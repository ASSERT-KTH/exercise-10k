## Exercise
Write a program that calculates an employee's salary using if/else statements

## Reference Solution
```python
def calculate_salary(job_title, years_of_experience, base_salary):
    """
    Calculates an employee's salary based on their job title and years of experience.

    Args:
        job_title (str): The employee's job title.
        years_of_experience (int): The employee's years of experience.
        base_salary (float): The base salary for the employee's job title.

    Returns:
        float: The employee's calculated salary.
    """
    if job_title == "Manager":
        if years_of_experience < 5:
            salary = base_salary * 1.1
        elif years_of_experience < 10:
            salary = base_salary * 1.2
        else:
            salary = base_salary * 1.3
    elif job_title == "Developer":
        if years_of_experience < 3:
            salary = base_salary * 1.05
        elif years_of_experience < 6:
            salary = base_salary * 1.1
        else:
            salary = base_salary * 1.2
    elif job_title == "Intern":
        if years_of_experience < 1:
            salary = base_salary * 0.8
        else:
            salary = base_salary * 0.9
    else:
        salary = base_salary

    return salary

def main():
    # Example usage:
    job_title = "Manager"
    years_of_experience = 8
    base_salary = 50000.0

    salary = calculate_salary(job_title, years_of_experience, base_salary)
    print(f"The salary for a {job_title} with {years_of_experience} years of experience is: ${salary:.2f}")

if __name__ == "__main__":
    main()
```
