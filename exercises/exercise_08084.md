## Exercise
Write a function that returns a closure with a context manager

## Reference Solution
```python
from contextlib import contextmanager
from typing import Callable

def create_closure_context_manager(resource_name: str) -> Callable:
    """
    Returns a closure that implements a context manager.
    
    The context manager is used to manage resources, such as files or database connections,
    and ensures that they are properly cleaned up after use.
    
    Args:
        resource_name (str): The name of the resource being managed.
    
    Returns:
        Callable: A closure that implements a context manager.
    """

    # Initialize a variable to store the resource
    resource = None

    @contextmanager
    def closure_context_manager():
        nonlocal resource  # Use the nonlocal keyword to modify the variable from the outer scope
        
        # Acquire the resource
        try:
            print(f"Acquiring resource: {resource_name}")
            resource = f"Resource {resource_name}"
            yield resource  # Yield the resource to the caller
        finally:
            # Release the resource
            if resource is not None:
                print(f"Releasing resource: {resource_name}")
                resource = None

    return closure_context_manager


# Example usage:
if __name__ == "__main__":
    # Create a closure with a context manager
    context_manager = create_closure_context_manager("example")

    # Use the context manager to manage a resource
    with context_manager() as resource:
        print(f"Using resource: {resource}")
```
