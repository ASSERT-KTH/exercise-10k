## Exercise
Write a program that calculates a student's grade based on their scores on several assignments.

## Reference Solution
```python
def calculate_grade(assignments):
    """
    Calculate the overall grade based on assignment scores.

    Args:
        assignments (dict): A dictionary where keys are assignment names and values are dictionaries with 'score' and 'weight' keys.

    Returns:
        float: The overall grade.
    """
    total_score = 0
    total_weight = 0

    for assignment, details in assignments.items():
        score = details['score']
        weight = details['weight']
        total_score += score * weight
        total_weight += weight

    overall_grade = total_score / total_weight
    return overall_grade


def main():
    # Example usage:
    assignments = {
        'Assignment 1': {'score': 90, 'weight': 0.2},
        'Assignment 2': {'score': 85, 'weight': 0.3},
        'Assignment 3': {'score': 95, 'weight': 0.5}
    }

    overall_grade = calculate_grade(assignments)
    print(f"Overall Grade: {overall_grade:.2f}")


if __name__ == "__main__":
    main()
```
