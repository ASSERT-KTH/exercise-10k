## Exercise
Write a function that extracts a substring from a given string based on a start and end index

## Reference Solution
```python
def extract_substring(input_string: str, start_index: int, end_index: int) -> str:
    """
    Extracts a substring from a given string based on a start and end index.

    Args:
        input_string (str): The input string.
        start_index (int): The start index of the substring.
        end_index (int): The end index of the substring.

    Returns:
        str: The extracted substring.

    Raises:
        ValueError: If the start index is greater than the end index.
        IndexError: If the start or end index is out of range.
    """

    # Check if the start index is greater than the end index
    if start_index > end_index:
        raise ValueError("Start index cannot be greater than end index")

    # Check if the start or end index is out of range
    if start_index < 0 or end_index > len(input_string):
        raise IndexError("Index out of range")

    # Extract the substring
    substring = input_string[start_index:end_index]

    return substring

# Example usage
def main():
    input_string = "Hello, World!"
    start_index = 7
    end_index = 12

    try:
        substring = extract_substring(input_string, start_index, end_index)
        print(f"Extracted substring: {substring}")
    except (ValueError, IndexError) as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
