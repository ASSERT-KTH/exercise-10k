## Exercise
Write a function that finds the difference between two dictionaries

## Reference Solution
```python
def dict_diff(dict1, dict2):
    """
    Calculate the difference between two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A dictionary containing the key-value pairs that are present in dict1 but not in dict2, or have different values.
    """
    diff = {}
    for key in dict1:
        if key not in dict2 or dict1[key] != dict2[key]:
            diff[key] = dict1[key]
    return diff

def dict_diff_both_ways(dict1, dict2):
    """
    Calculate the symmetric difference between two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A dictionary containing the key-value pairs that are present in either dict1 or dict2 but not both, or have different values.
    """
    diff1 = dict_diff(dict1, dict2)
    diff2 = dict_diff(dict2, dict1)
    return {**diff1, **diff2}

# Example usage
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2, "c": 3}
    dict2 = {"a": 1, "b": 3, "d": 4}

    print("Difference from dict1 to dict2:")
    print(dict_diff(dict1, dict2))

    print("Symmetric difference:")
    print(dict_diff_both_ways(dict1, dict2))
```
