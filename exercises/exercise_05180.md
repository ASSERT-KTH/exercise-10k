## Exercise
Write a function to check if two dictionaries are not equal.

def are_dictionaries_not_equal(dict1, dict2):
    """
    Checks if two dictionaries are not equal.

    Args:
        dict1 (dict): The first dictionary to compare.
        dict2 (dict): The second dictionary to compare.

    Returns:
        bool: True if the dictionaries are not equal, False otherwise.
    """
    return dict1 != dict2

# Example usage:
if __name__ == "__main__":

    print(are_dictionaries_not_equal(dict1, dict2))  # Output: False
    print(are_dictionaries_not_equal(dict1, dict3))  # Output: True
