## Exercise
Write a program that takes a string as input and prints out all the email addresses found in the string

## Reference Solution
```python
import re

def extract_emails(input_string):
    """
    Extracts email addresses from a given input string.

    Args:
        input_string (str): The input string to extract email addresses from.

    Returns:
        list: A list of extracted email addresses.
    """
    # Regular expression pattern to match email addresses
    email_pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'

    # Find all email addresses in the input string
    emails = re.findall(email_pattern, input_string)

    return emails

def main():
    # Example usage
    input_string = "Contact me at john.doe@example.com or jane.smith@example.org for more information."
    emails = extract_emails(input_string)

    print("Extracted Email Addresses:")
    for email in emails:
        print(email)

if __name__ == "__main__":
    main()
```
