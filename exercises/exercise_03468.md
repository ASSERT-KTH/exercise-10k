## Exercise
Write a program that asks the user a series of questions and keeps track of the score.

## Reference Solution
```python
# Program to ask the user a series of questions and keep track of the score

# Define the questions and answers
questions = [
    {
        "question": "What is the capital of France?",
        "answer": "Paris"
    },
    {
        "question": "What is the largest planet in our solar system?",
        "answer": "Jupiter"
    },
    {
        "question": "Who painted the Starry Night?",
        "answer": "Vincent van Gogh"
    },
    {
        "question": "What is the chemical symbol for gold?",
        "answer": "Au"
    },
    {
        "question": "What is the smallest country in the world?",
        "answer": "Vatican City"
    }
]

def quiz(questions):
    # Initialize the score
    score = 0

    # Ask the questions
    for i, q in enumerate(questions):
        print(f"Question {i+1}: {q['question']}")
        user_answer = input("Your answer: ")

        # Check if the user's answer is correct
        if user_answer.lower() == q['answer'].lower():
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer was {q['answer']}\n")

    # Print the final score
    print(f"Quiz complete! Your final score is {score} out of {len(questions)}")

# Run the quiz
quiz(questions)
```
