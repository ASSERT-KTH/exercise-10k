## Exercise
Write a function to find the minimum value in an array of integers.

## Reference Solution
```python
def find_minimum(array):
    """
    This function finds the minimum value in an array of integers.

    Args:
        array (list): A list of integers.

    Returns:
        int: The minimum value in the array.

    Raises:
        ValueError: If the input array is empty.
    """
    if not array:
        raise ValueError("Input array is empty")

    return min(array)

# Example usage
def main():
    array = [5, 2, 9, 1, 7]
    try:
        minimum_value = find_minimum(array)
        print(f"The minimum value in the array is: {minimum_value}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
