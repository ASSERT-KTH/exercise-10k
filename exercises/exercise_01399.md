## Exercise
Write a program that uses a for loop to apply a filter to an image.

## Reference Solution
```python
from PIL import Image

def apply_grayscale_filter(image):
    """
    Apply a grayscale filter to an image.
    
    Args:
        image (Image): The input image.
    
    Returns:
        Image: The filtered image.
    """
    # Get the dimensions of the image
    width, height = image.size
    
    # Create a new image with the same mode and size as the input image
    filtered_image = Image.new(image.mode, image.size)
    
    # Load the pixel data of the input image
    pixels = image.load()
    filtered_pixels = filtered_image.load()
    
    # Apply the grayscale filter to each pixel
    for x in range(width):
        for y in range(height):
            # Get the RGB values of the current pixel
            r, g, b = pixels[x, y]
            
            # Calculate the grayscale value using the formula: (r + g + b) / 3
            grayscale_value = int((r + g + b) / 3)
            
            # Set the grayscale value for the current pixel
            filtered_pixels[x, y] = (grayscale_value, grayscale_value, grayscale_value)
    
    return filtered_image

def main():
    # Open an image file
    image = Image.open('input.jpg')
    
    # Apply the grayscale filter to the image
    filtered_image = apply_grayscale_filter(image)
    
    # Save the filtered image to a new file
    filtered_image.save('output.jpg')

if __name__ == '__main__':
    main()
```
