## Exercise
Create a program that takes in a mathematical expression and evaluates it based on conditional statements. For example, if the user inputs '2 + 2', the program should output '4'. If the user inputs '2 * 2', the program should output '4'.

## Reference Solution
```python
def evaluate_expression(expression):
    """
    Evaluates a simple mathematical expression.

    Args:
    expression (str): A string representing a mathematical expression.

    Returns:
    float: The result of the evaluated expression.
    """

    # Split the expression into three parts: the first operand, the operator, and the second operand.
    try:
        operand1, operator, operand2 = expression.split()
    except ValueError:
        print("Invalid input. Please use the format 'operand operator operand'.")
        return None

    # Convert the operands to floats.
    try:
        operand1 = float(operand1)
        operand2 = float(operand2)
    except ValueError:
        print("Invalid input. Operands must be numbers.")
        return None

    # Evaluate the expression based on the operator.
    if operator == '+':
        result = operand1 + operand2
    elif operator == '-':
        result = operand1 - operand2
    elif operator == '*':
        result = operand1 * operand2
    elif operator == '/':
        # Check for division by zero.
        if operand2 == 0:
            print("Error: Division by zero.")
            return None
        result = operand1 / operand2
    else:
        print("Invalid operator. Supported operators are +, -, *, /.")
        return None

    return result


def main():
    print("Simple Mathematical Expression Evaluator")
    print("-----------------------------------------")

    while True:
        expression = input("Enter a mathematical expression (or 'quit' to exit): ")
        if expression.lower() == 'quit':
            break
        result = evaluate_expression(expression)
        if result is not None:
            print(f"Result: {result}")


if __name__ == "__main__":
    main()
```
