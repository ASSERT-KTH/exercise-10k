## Exercise
Create a dictionary and update the value for a given key

## Reference Solution
```python
def update_dict_value(dictionary, key, new_value):
    """
    Updates the value for a given key in a dictionary.

    Args:
        dictionary (dict): The dictionary to update.
        key: The key to update the value for.
        new_value: The new value for the given key.

    Returns:
        dict: The updated dictionary.
    """
    # Check if the key exists in the dictionary
    if key in dictionary:
        # Update the value for the given key
        dictionary[key] = new_value
        print(f"Updated the value for key '{key}' to '{new_value}'")
    else:
        print(f"Key '{key}' not found in the dictionary")

    return dictionary

# Example usage
if __name__ == "__main__":
    # Create a dictionary
    my_dict = {"name": "John", "age": 30, "city": "New York"}

    # Print the original dictionary
    print("Original Dictionary:")
    print(my_dict)

    # Update the value for a given key
    updated_dict = update_dict_value(my_dict, "age", 31)

    # Print the updated dictionary
    print("\nUpdated Dictionary:")
    print(updated_dict)
```
