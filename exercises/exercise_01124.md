## Exercise
Write a program that allows two players to play a game of Rock Paper Scissors

## Reference Solution
```python
import random

def get_player_choice(player_name, mode):
    """Get the choice of a player."""
    if mode == "manual":
        while True:
            choice = input(f"{player_name}, enter your choice (rock/paper/scissors): ")
            if choice.lower() in ["rock", "paper", "scissors"]:
                return choice.lower()
            else:
                print("Invalid choice. Please try again.")
    elif mode == "random":
        return random.choice(["rock", "paper", "scissors"])

def determine_winner(player1_choice, player2_choice):
    """Determine the winner of the game."""
    if player1_choice == player2_choice:
        return "It's a tie!"
    elif (player1_choice == "rock" and player2_choice == "scissors") or \
         (player1_choice == "scissors" and player2_choice == "paper") or \
         (player1_choice == "paper" and player2_choice == "rock"):
        return "Player 1 wins!"
    else:
        return "Player 2 wins!"

def play_game():
    """Play a game of Rock Paper Scissors."""
    print("Welcome to Rock Paper Scissors!")
    mode = input("Do you want to play in manual or random mode? ")
    while mode not in ["manual", "random"]:
        print("Invalid mode. Please enter 'manual' or 'random'.")
        mode = input("Do you want to play in manual or random mode? ")
    
    player1_name = input("Enter Player 1's name: ")
    player2_name = input("Enter Player 2's name: ")

    player1_choice = get_player_choice(player1_name, mode)
    player2_choice = get_player_choice(player2_name, mode)

    print(f"\n{player1_name} chose: {player1_choice}")
    print(f"{player2_name} chose: {player2_choice}")

    print(determine_winner(player1_choice, player2_choice))

if __name__ == "__main__":
    play_game()
```
