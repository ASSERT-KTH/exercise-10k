## Exercise
Write a program that calculates the frequency of each word in a given text.

## Reference Solution
```python
import re
from collections import Counter

def calculate_word_frequency(text):
    """
    Calculate the frequency of each word in a given text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary where the keys are the words and the values are their frequencies.
    """
    # Convert the text to lowercase and remove punctuation
    text = text.lower()
    text = re.sub(r'[^\w\s]', '', text)

    # Split the text into words
    words = text.split()

    # Calculate the frequency of each word
    frequency = Counter(words)

    return frequency

def main():
    # Example usage
    text = "This is an example sentence. This sentence is just an example."
    frequency = calculate_word_frequency(text)

    # Print the word frequencies
    print("Word Frequencies:")
    for word, count in frequency.items():
        print(f"{word}: {count}")

if __name__ == "__main__":
    main()
```
