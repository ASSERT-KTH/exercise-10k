## Exercise
Given an array of integers, write a program to delete a specific element.

## Reference Solution
```python
def delete_element(array, target):
    """
    Deletes the first occurrence of a target element in the array.

    Args:
    array (list): The input array.
    target (int): The element to be deleted.

    Returns:
    list: The array with the target element deleted.
    """
    try:
        # Find the index of the target element
        index = array.index(target)
        # Delete the target element
        del array[index]
        return array
    except ValueError:
        # If the target element is not found, return the original array
        print(f"{target} not found in the array.")
        return array


def main():
    # Example usage
    array = [1, 2, 3, 4, 5]
    target = 3
    print("Original array:", array)
    print("Target element:", target)
    updated_array = delete_element(array, target)
    print("Updated array:", updated_array)


if __name__ == "__main__":
    main()
```
