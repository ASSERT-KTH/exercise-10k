## Exercise
Write a program to calculate the total cost of an item including sales tax.

## Reference Solution
```python
def calculate_total_cost(price, tax_rate):
    """
    Calculate the total cost of an item including sales tax.

    Args:
        price (float): The price of the item.
        tax_rate (float): The sales tax rate as a decimal.

    Returns:
        float: The total cost of the item including sales tax.
    """
    sales_tax = price * tax_rate
    total_cost = price + sales_tax
    return total_cost

def main():
    # Get the item's price from the user
    price = float(input("Enter the item's price: $"))

    # Get the sales tax rate from the user
    tax_rate = float(input("Enter the sales tax rate (as a percentage, e.g., 8 for 8%): "))
    tax_rate /= 100  # Convert the tax rate to a decimal

    # Calculate the total cost
    total_cost = calculate_total_cost(price, tax_rate)

    # Print the result
    print(f"Sales tax: ${price * tax_rate:.2f}")
    print(f"Total cost: ${total_cost:.2f}")

if __name__ == "__main__":
    main()
```
