## Exercise
Write a program that updates the value associated with a given key in a dictionary.

## Reference Solution
```python
def update_dict_value(dictionary, key, new_value):
    """
    Updates the value associated with a given key in a dictionary.

    Args:
        dictionary (dict): The dictionary to be updated.
        key (any): The key whose value is to be updated.
        new_value (any): The new value to be assigned to the key.

    Returns:
        dict: The updated dictionary.

    Raises:
        KeyError: If the key is not found in the dictionary.
    """
    if key in dictionary:
        dictionary[key] = new_value
    else:
        raise KeyError(f"The key '{key}' is not found in the dictionary.")
    return dictionary


# Example usage:
if __name__ == "__main__":
    my_dict = {"name": "John", "age": 30, "city": "New York"}
    print("Original Dictionary:")
    print(my_dict)

    key_to_update = "age"
    new_value = 31
    updated_dict = update_dict_value(my_dict, key_to_update, new_value)
    print(f"\nUpdated Dictionary after updating '{key_to_update}' to {new_value}:")
    print(updated_dict)

    # Try updating a non-existent key
    try:
        update_dict_value(my_dict, "country", "USA")
    except KeyError as e:
        print(f"\nError: {e}")
```
