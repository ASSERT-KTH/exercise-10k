## Exercise
Write a program to shift the elements of an array by a specified number of positions.

## Reference Solution
```python
from collections import deque

def shift_array(arr, num_positions, direction='right'):
    """
    Shifts the elements of an array by a specified number of positions.

    Args:
    arr (list): The input array.
    num_positions (int): The number of positions to shift.
    direction (str): The direction of the shift. Can be 'right' or 'left'.

    Returns:
    list: The shifted array.
    """
    # Create a deque from the input array
    d = deque(arr)

    # Shift the elements
    if direction == 'right':
        d.rotate(num_positions)
    elif direction == 'left':
        d.rotate(-num_positions)
    else:
        raise ValueError("Invalid direction. Must be 'right' or 'left'.")

    # Convert the deque back to a list and return it
    return list(d)


# Example usage:
arr = [1, 2, 3, 4, 5]
num_positions = 2
direction = 'right'

shifted_arr = shift_array(arr, num_positions, direction)
print(f"Original array: {arr}")
print(f"Shifted array ({direction} by {num_positions} positions): {shifted_arr}")
```
