## Exercise
Encrypt and decrypt variables to and from a secure representation.

## Reference Solution
```python
from cryptography.fernet import Fernet

def generate_key():
    """Generate a key for encryption and decryption"""
    return Fernet.generate_key()

def encrypt_variable(variable, key):
    """Encrypt a variable using the provided key"""
    cipher_suite = Fernet(key)
    encrypted_variable = cipher_suite.encrypt(str(variable).encode('utf-8'))
    return encrypted_variable

def decrypt_variable(encrypted_variable, key):
    """Decrypt a variable using the provided key"""
    cipher_suite = Fernet(key)
    decrypted_variable = cipher_suite.decrypt(encrypted_variable)
    return decrypted_variable.decode('utf-8')

# Example usage
if __name__ == "__main__":
    key = generate_key()
    print("Generated Key:", key)

    variable = "Hello, World!"
    print("Original Variable:", variable)

    encrypted_variable = encrypt_variable(variable, key)
    print("Encrypted Variable:", encrypted_variable)

    decrypted_variable = decrypt_variable(encrypted_variable, key)
    print("Decrypted Variable:", decrypted_variable)
```
