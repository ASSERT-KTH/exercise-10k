## Exercise
Write a program that takes a string as input and returns the longest substring without repeating characters.

## Reference Solution
```python
def longest_substring(s: str) -> str:
    """
    Returns the longest substring without repeating characters.

    Args:
        s (str): The input string.

    Returns:
        str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    left = 0
    max_length = 0
    max_substring = ""
    char_set = set()

    for right in range(len(s)):
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1

        char_set.add(s[right])

        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring


# Example usage:
if __name__ == "__main__":
    input_string = "abcabcbb"
    result = longest_substring(input_string)
    print(f"Input String: {input_string}")
    print(f"Longest Substring Without Repeating Characters: {result}")
```
