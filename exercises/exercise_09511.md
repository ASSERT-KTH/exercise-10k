        s (str): The input string to be checked.

    """

    # We'll use the replace() method to remove spaces from the string before checking

    # Remove all spaces from the string
    s_without_spaces = s.replace(" ", "")

    # Check if the string without spaces is alphanumeric
    # If it is, and the original string only contains spaces and alphanumeric characters, return True
