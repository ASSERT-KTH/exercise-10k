## Exercise
Write a function that finds the index of the first occurrence of a character in a string

## Reference Solution
```python
def find_char_index(string, char):
    """
    Finds the index of the first occurrence of a character in a string.

    Args:
        string (str): The input string.
        char (str): The character to find.

    Returns:
        int: The index of the first occurrence of the character, or -1 if not found.
    """
    try:
        return string.index(char)
    except ValueError:
        return -1

# Example usage:
def main():
    string = "Hello, World!"
    char = "W"
    index = find_char_index(string, char)
    
    if index != -1:
        print(f"The character '{char}' is found at index {index}.")
    else:
        print(f"The character '{char}' is not found in the string.")

if __name__ == "__main__":
    main()
```
