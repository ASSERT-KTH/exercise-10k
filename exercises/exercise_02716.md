## Exercise
Write a program to format a given phone number in a specific format (e.g. (123) 456-7890).

## Reference Solution
```python
def format_phone_number(phone_number):
    """
    Formats a given phone number into (123) 456-7890 format.

    Args:
        phone_number (str): The phone number to be formatted.

    Returns:
        str: The formatted phone number.
    """
    # Remove any non-numeric characters from the phone number
    phone_number = ''.join(filter(str.isdigit, phone_number))

    # Check if the phone number is 10 digits long
    if len(phone_number) != 10:
        raise ValueError("Invalid phone number. Please enter a 10-digit number.")

    # Format the phone number
    formatted_phone_number = f"({phone_number[:3]}) {phone_number[3:6]}-{phone_number[6:]}"

    return formatted_phone_number


def main():
    phone_number = input("Enter a phone number: ")
    try:
        formatted_phone_number = format_phone_number(phone_number)
        print(f"Formatted phone number: {formatted_phone_number}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
