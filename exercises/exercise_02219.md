## Exercise
Write a program that takes a credit card number as input and checks if it is valid. Print 'Valid credit card' if the number is valid, otherwise print 'Invalid credit card'

def validate_credit_card(card_number):
    """
    Validate a credit card number using the Luhn algorithm.

    Args:
        card_number (str): The credit card number to validate.

    Returns:
        bool: True if the credit card number is valid, False otherwise.
    """
    # Remove any non-digit characters from the card number
    card_number = ''.join(filter(str.isdigit, card_number))

    # Check if the card number is empty or has less than 13 digits
    if len(card_number) < 13 or len(card_number) > 16:
        return False

    # Convert the card number to a list of integers
    card_number = [int(x) for x in card_number]

    # Reverse the card number
    card_number = card_number[::-1]

    # Double every second digit
    for i in range(1, len(card_number), 2):
        card_number[i] *= 2
        # Subtract 9 from numbers over 9
        if card_number[i] > 9:
            card_number[i] -= 9

    # If the sum of the digits is divisible by 10, the card number is valid
    return sum(card_number) % 10 == 0


def main():
    card_number = input("Enter a credit card number: ")
    if validate_credit_card(card_number):
        print("Valid credit card")
    else:
        print("Invalid credit card")


if __name__ == "__main__":
    main()
