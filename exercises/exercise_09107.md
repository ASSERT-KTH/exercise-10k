## Exercise
Create a program that stores student records in a dictionary. The dictionary should have the student's name as the key and a list of their grades as the value. The program should allow users to add a new student, add a grade to an existing student, and calculate the average grade of a student.

## Reference Solution
```python
class StudentRecords:
    def __init__(self):
        """
        Initialize an empty dictionary to store student records.
        """
        self.records = {}

    def add_student(self, name):
        """
        Add a new student to the records.

        Args:
            name (str): The name of the student.
        """
        if name not in self.records:
            self.records[name] = []
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def add_grade(self, name, grade):
        """
        Add a grade to an existing student.

        Args:
            name (str): The name of the student.
            grade (float): The grade to add.
        """
        if name in self.records:
            self.records[name].append(grade)
            print(f"Grade '{grade}' added to student '{name}' successfully.")
        else:
            print(f"Student '{name}' does not exist.")

    def calculate_average(self, name):
        """
        Calculate the average grade of a student.

        Args:
            name (str): The name of the student.

        Returns:
            float: The average grade of the student.
        """
        if name in self.records:
            grades = self.records[name]
            if len(grades) == 0:
                print(f"Student '{name}' has no grades.")
                return None
            average = sum(grades) / len(grades)
            print(f"The average grade of student '{name}' is {average:.2f}.")
            return average
        else:
            print(f"Student '{name}' does not exist.")
            return None

    def display_records(self):
        """
        Display all student records.
        """
        print("Student Records:")
        for name, grades in self.records.items():
            print(f"{name}: {grades}")


def main():
    records = StudentRecords()

    while True:
        print("\nStudent Records Menu:")
        print("1. Add a new student")
        print("2. Add a grade to an existing student")
        print("3. Calculate the average grade of a student")
        print("4. Display all student records")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter the student's name: ")
            records.add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            grade = float(input("Enter the grade: "))
            records.add_grade(name, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            records.calculate_average(name)
        elif choice == "4":
            records.display_records()
        elif choice == "5":
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
