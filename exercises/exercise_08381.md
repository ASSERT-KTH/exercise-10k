## Exercise
Write a program that calculates the sine, cosine, and tangent of a given angle using the Taylor series expansion.

## Reference Solution
```python
import math

def factorial(n):
    """
    Calculate the factorial of a number.
    
    Args:
    n (int): The number to calculate the factorial for.
    
    Returns:
    int: The factorial of n.
    """
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)

def power(x, n):
    """
    Calculate the power of a number.
    
    Args:
    x (float): The base number.
    n (int): The exponent.
    
    Returns:
    float: The result of x to the power of n.
    """
    if n == 0:
        return 1
    elif n < 0:
        return 1 / power(x, -n)
    else:
        return x * power(x, n-1)

def taylor_sine(x, n=10):
    """
    Calculate the sine of an angle using the Taylor series expansion.
    
    Args:
    x (float): The angle in radians.
    n (int, optional): The number of terms to use in the Taylor series. Defaults to 10.
    
    Returns:
    float: The sine of the angle.
    """
    result = 0
    for i in range(n):
        sign = (-1)**i
        result += ((power(x, 2.0*i+1)) / factorial(2*i+1)) * sign
    return result

def taylor_cosine(x, n=10):
    """
    Calculate the cosine of an angle using the Taylor series expansion.
    
    Args:
    x (float): The angle in radians.
    n (int, optional): The number of terms to use in the Taylor series. Defaults to 10.
    
    Returns:
    float: The cosine of the angle.
    """
    result = 0
    for i in range(n):
        sign = (-1)**i
        result += ((power(x, 2.0*i)) / factorial(2*i)) * sign
    return result

def taylor_tangent(x, n=10):
    """
    Calculate the tangent of an angle using the Taylor series expansion.
    
    Args:
    x (float): The angle in radians.
    n (int, optional): The number of terms to use in the Taylor series. Defaults to 10.
    
    Returns:
    float: The tangent of the angle.
    """
    return taylor_sine(x, n) / taylor_cosine(x, n)

# Test the functions
angle = math.pi / 4  # 45 degrees in radians
print("Sine of the angle: ", taylor_sine(angle))
print("Cosine of the angle: ", taylor_cosine(angle))
print("Tangent of the angle: ", taylor_tangent(angle))
```
