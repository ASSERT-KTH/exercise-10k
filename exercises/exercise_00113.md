## Exercise
Write a program to find the duplicate in an array of integers.

## Reference Solution
```python
def find_duplicate(nums):
    """
    Finds the duplicate in an array of integers.

    Args:
    - nums (list): A list of integers.

    Returns:
    - The duplicate integer.

    Raises:
    - ValueError: If the input list does not contain a duplicate.
    """
    num_set = set()
    for num in nums:
        if num in num_set:
            return num
        num_set.add(num)
    raise ValueError("No duplicate found")

# Example usage:
def main():
    nums = [1, 2, 3, 4, 5, 6, 7, 8, 9, 1]
    try:
        duplicate = find_duplicate(nums)
        print(f"Duplicate found: {duplicate}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
