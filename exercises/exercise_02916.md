## Exercise
Write a program to count the frequency of each word in a given string.

## Reference Solution
```python
from collections import Counter
import re

def count_word_frequency(input_string):
    """
    Count the frequency of each word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where keys are words and values are their frequencies.
    """
    # Convert the string to lowercase and remove punctuation
    cleaned_string = re.sub(r'[^\w\s]', '', input_string.lower())
    
    # Split the string into words
    words = cleaned_string.split()
    
    # Count the frequency of each word
    frequency = Counter(words)
    
    return frequency

def main():
    input_string = "This is a test string. The string is just a test."
    frequency = count_word_frequency(input_string)
    
    print("Word Frequency:")
    for word, count in frequency.items():
        print(f"{word}: {count}")

if __name__ == "__main__":
    main()
```
