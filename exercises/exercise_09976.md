## Exercise
Write a program that demonstrates the concept of variable scope.

## Reference Solution
```python
# Variable scope demonstration program

# Global variable
global_variable = "I am global"

def outer_function():
    # Enclosing variable
    enclosing_variable = "I am enclosing"

    def inner_function():
        # Local variable
        local_variable = "I am local"
        
        print("Inside inner_function:")
        print(f"global_variable: {global_variable}")
        print(f"enclosing_variable: {enclosing_variable}")
        print(f"local_variable: {local_variable}")

    inner_function()

    print("\nInside outer_function:")
    print(f"global_variable: {global_variable}")
    print(f"enclosing_variable: {enclosing_variable}")
    # Trying to access local_variable will result in an error
    # print(f"local_variable: {local_variable}")

def main():
    print("Inside main:")
    print(f"global_variable: {global_variable}")
    # Trying to access enclosing_variable or local_variable will result in an error
    # print(f"enclosing_variable: {enclosing_variable}")
    # print(f"local_variable: {local_variable}")

    outer_function()

if __name__ == "__main__":
    main()
```
