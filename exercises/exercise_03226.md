## Exercise
Write a program to find the maximum contiguous subarray of a given array.

## Reference Solution
```python
def max_contiguous_subarray(arr):
    """
    Find the maximum contiguous subarray of a given array.

    Args:
        arr (list): The input array.

    Returns:
        tuple: The maximum contiguous subarray and its sum.
    """
    if not arr:
        return [], 0

    max_sum = float('-inf')
    current_sum = 0
    max_subarray = []
    current_subarray = []

    for num in arr:
        if current_sum + num > num:
            current_sum += num
            current_subarray.append(num)
        else:
            current_sum = num
            current_subarray = [num]

        if current_sum > max_sum:
            max_sum = current_sum
            max_subarray = current_subarray[:]

    return max_subarray, max_sum


def main():
    # Example usage:
    arr = [-2, -3, 4, -1, -2, 1, 5, -3]
    max_subarray, max_sum = max_contiguous_subarray(arr)
    print("Maximum contiguous subarray:", max_subarray)
    print("Maximum sum:", max_sum)


if __name__ == "__main__":
    main()
```
