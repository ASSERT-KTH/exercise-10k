## Exercise
Develop a program that administers a simple quiz, where the user is asked a series of questions and the program keeps track of the score

## Reference Solution
```python
# Define a dictionary to store the questions, answers, and options
quiz_questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "answer": "B"
    }
}

def administer_quiz(questions):
    """
    Administer the quiz and keep track of the score.

    Args:
        questions (dict): A dictionary of questions, answers, and options.

    Returns:
        int: The final score.
    """
    score = 0
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        answer = input("Choose the correct answer (A, B, C, D): ")
        if answer.upper() == options["answer"]:
            print("Correct!\n")
            score += 1
        else:
            print(f"Incorrect. The correct answer is {options['answer']}.\n")
    return score

def main():
    print("Welcome to the quiz!")
    score = administer_quiz(quiz_questions)
    print(f"Quiz complete! Your final score is {score} out of {len(quiz_questions)}.")

if __name__ == "__main__":
    main()
```
