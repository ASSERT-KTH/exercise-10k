## Exercise
Create a program that acts as a timer, allowing the user to set a time in seconds, minutes, or hours, and then counts down to zero.

## Reference Solution
```python
import time

def countdown(t):
    """
    Countdown function that takes an integer t (time in seconds) and counts down to zero.
    
    Args:
        t (int): Time in seconds.
    """
    while t:
        mins, secs = divmod(t, 60)
        hours, mins = divmod(mins, 60)
        timer = '{:02d}:{:02d}:{:02d}'.format(hours, mins, secs)
        print(timer, end="\r")
        time.sleep(1)
        t -= 1
    print('Time Up!!')

def main():
    """
    Main function where the user can input the time and choose the unit of time.
    """
    print("Choose the unit of time:")
    print("1. Seconds")
    print("2. Minutes")
    print("3. Hours")
    
    choice = input("Enter your choice (1/2/3): ")
    
    if choice == "1":
        t = int(input("Enter the time in seconds: "))
    elif choice == "2":
        t = int(input("Enter the time in minutes: ")) * 60
    elif choice == "3":
        t = int(input("Enter the time in hours: ")) * 60 * 60
    else:
        print("Invalid choice. Exiting program.")
        return
    
    print("Starting countdown...")
    countdown(t)

if __name__ == "__main__":
    main()
```
