## Exercise
Design a program to perform arithmetic operations on complex numbers (e.g., addition, subtraction, multiplication, division).

## Reference Solution
```python
class ComplexNumber:
    def __init__(self, real, imaginary):
        """
        Initialize a ComplexNumber object.

        Args:
            real (float): The real part of the complex number.
            imaginary (float): The imaginary part of the complex number.
        """
        self.real = real
        self.imaginary = imaginary

    def __str__(self):
        """
        Return a string representation of the complex number.

        Returns:
            str: A string representation of the complex number.
        """
        if self.imaginary >= 0:
            return f"{self.real} + {self.imaginary}i"
        else:
            return f"{self.real} - {abs(self.imaginary)}i"

    def add(self, other):
        """
        Add two complex numbers.

        Args:
            other (ComplexNumber): The complex number to add.

        Returns:
            ComplexNumber: The sum of the two complex numbers.
        """
        real_sum = self.real + other.real
        imaginary_sum = self.imaginary + other.imaginary
        return ComplexNumber(real_sum, imaginary_sum)

    def subtract(self, other):
        """
        Subtract one complex number from another.

        Args:
            other (ComplexNumber): The complex number to subtract.

        Returns:
            ComplexNumber: The difference of the two complex numbers.
        """
        real_diff = self.real - other.real
        imaginary_diff = self.imaginary - other.imaginary
        return ComplexNumber(real_diff, imaginary_diff)

    def multiply(self, other):
        """
        Multiply two complex numbers.

        Args:
            other (ComplexNumber): The complex number to multiply.

        Returns:
            ComplexNumber: The product of the two complex numbers.
        """
        real_product = self.real * other.real - self.imaginary * other.imaginary
        imaginary_product = self.real * other.imaginary + self.imaginary * other.real
        return ComplexNumber(real_product, imaginary_product)

    def divide(self, other):
        """
        Divide one complex number by another.

        Args:
            other (ComplexNumber): The complex number to divide by.

        Returns:
            ComplexNumber: The quotient of the two complex numbers.

        Raises:
            ZeroDivisionError: If the divisor is zero.
        """
        if other.real == 0 and other.imaginary == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        denominator = other.real ** 2 + other.imaginary ** 2
        real_quotient = (self.real * other.real + self.imaginary * other.imaginary) / denominator
        imaginary_quotient = (self.imaginary * other.real - self.real * other.imaginary) / denominator
        return ComplexNumber(real_quotient, imaginary_quotient)


def main():
    num1 = ComplexNumber(1, 2)
    num2 = ComplexNumber(3, 4)

    print("Complex Number 1:", num1)
    print("Complex Number 2:", num2)

    print("Addition:", num1.add(num2))
    print("Subtraction:", num1.subtract(num2))
    print("Multiplication:", num1.multiply(num2))
    print("Division:", num1.divide(num2))


if __name__ == "__main__":
    main()
```
