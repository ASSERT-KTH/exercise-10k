## Exercise
Create a program that takes in two numbers and an operator (+, -, *, /) as input from the user and then performs the specified operation, printing out the result.

## Reference Solution
```python
def calculator():
    """
    A simple calculator program that takes in two numbers and an operator 
    as input from the user and then performs the specified operation, 
    printing out the result.
    """

    # Get the first number from the user
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Get the operator from the user
    while True:
        operator = input("Enter the operator (+, -, *, /): ")
        if operator in ['+', '-', '*', '/']:
            break
        else:
            print("Invalid operator. Please enter +, -, *, or /.")

    # Get the second number from the user
    while True:
        try:
            num2 = float(input("Enter the second number: "))
            if operator == '/' and num2 == 0:
                print("Cannot divide by zero. Please enter a non-zero number.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Perform the operation
    if operator == '+':
        result = num1 + num2
    elif operator == '-':
        result = num1 - num2
    elif operator == '*':
        result = num1 * num2
    elif operator == '/':
        result = num1 / num2

    # Print the result
    print(f"{num1} {operator} {num2} = {result}")

def main():
    calculator()
    while True:
        cont = input("Do you want to continue? (y/n): ")
        if cont.lower() == 'y':
            calculator()
        elif cont.lower() == 'n':
            break
        else:
            print("Invalid input. Please enter y or n.")

if __name__ == "__main__":
    main()
```
