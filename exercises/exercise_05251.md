## Exercise
Write a program to calculate the square root or cube root of a given number.

## Reference Solution
```python
import math

def calculate_root(number, root_type):
    """
    Calculate the root of a given number.

    Args:
        number (float): The number to calculate the root of.
        root_type (str): The type of root to calculate ('square' or 'cube').

    Returns:
        float: The calculated root.
    """
    if root_type == 'square':
        return math.sqrt(number)
    elif root_type == 'cube':
        return round(number ** (1/3), 3)
    else:
        raise ValueError("Invalid root type. Please choose 'square' or 'cube'.")

def main():
    print("Root Calculator Program")
    print("------------------------")

    number = float(input("Enter a number: "))
    root_type = input("Enter the root type ('square' or 'cube'): ")

    try:
        result = calculate_root(number, root_type)
        print(f"The {root_type} root of {number} is {result}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
