## Exercise
Create a program that displays student information using if/else statements

## Reference Solution
```python
# Define a dictionary to store student information
students = {
    "John": {"age": 20, "grade": "A", "major": "Computer Science"},
    "Alice": {"age": 22, "grade": "B", "major": "Mathematics"},
    "Bob": {"age": 21, "grade": "C", "major": "Physics"}
}

def display_student_info(student_name):
    """
    Display student information based on the student name.
    
    Args:
        student_name (str): The name of the student.
    """
    if student_name in students:
        print(f"Student Name: {student_name}")
        print(f"Age: {students[student_name]['age']}")
        print(f"Grade: {students[student_name]['grade']}")
        print(f"Major: {students[student_name]['major']}")
    else:
        print("Student not found.")

def display_student_info_by_age():
    """
    Display student information for students above a certain age.
    """
    age = int(input("Enter the minimum age: "))
    for student_name, info in students.items():
        if info["age"] >= age:
            print(f"Student Name: {student_name}")
            print(f"Age: {info['age']}")
            print(f"Grade: {info['grade']}")
            print(f"Major: {info['major']}\n")

def display_student_info_by_grade():
    """
    Display student information for students with a certain grade.
    """
    grade = input("Enter the grade (A, B, C): ")
    for student_name, info in students.items():
        if info["grade"] == grade.upper():
            print(f"Student Name: {student_name}")
            print(f"Age: {info['age']}")
            print(f"Grade: {info['grade']}")
            print(f"Major: {info['major']}\n")

def main():
    print("1. Display student information by name")
    print("2. Display student information by age")
    print("3. Display student information by grade")
    
    choice = input("Enter your choice (1, 2, 3): ")
    
    if choice == "1":
        student_name = input("Enter the student name: ")
        display_student_info(student_name)
    elif choice == "2":
        display_student_info_by_age()
    elif choice == "3":
        display_student_info_by_grade()
    else:
        print("Invalid choice.")

if __name__ == "__main__":
    main()
```
