## Exercise
Write a program to find the first duplicate in a given list of numbers.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Find the first duplicate in a given list of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        int: The first duplicate number. If no duplicates are found, returns None.
    """
    seen = {}
    for i, num in enumerate(nums):
        if num in seen:
            # If we've seen this number before, return it
            return num
        else:
            # Otherwise, add it to the seen dictionary
            seen[num] = i
    # If we've iterated through the entire list and found no duplicates, return None
    return None

# Example usage:
def main():
    numbers = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(numbers)
    if duplicate is not None:
        print(f"The first duplicate is: {duplicate}")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
