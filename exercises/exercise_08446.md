## Exercise
Write a program that takes in a shape and calculates its area using conditional statements.

## Reference Solution
```python
import math

def calculate_area(shape):
    """
    Calculate the area of a shape.

    Args:
    shape (dict): A dictionary containing the shape type and its dimensions.

    Returns:
    float: The area of the shape.
    """

    # Check if the shape type is valid
    if 'type' not in shape:
        raise ValueError("Shape type is required")

    # Calculate the area based on the shape type
    if shape['type'] == 'circle':
        # Check if the radius is provided
        if 'radius' not in shape:
            raise ValueError("Radius is required for a circle")
        # Calculate the area of the circle
        area = math.pi * (shape['radius'] ** 2)
    elif shape['type'] == 'rectangle':
        # Check if the length and width are provided
        if 'length' not in shape or 'width' not in shape:
            raise ValueError("Length and width are required for a rectangle")
        # Calculate the area of the rectangle
        area = shape['length'] * shape['width']
    elif shape['type'] == 'triangle':
        # Check if the base and height are provided
        if 'base' not in shape or 'height' not in shape:
            raise ValueError("Base and height are required for a triangle")
        # Calculate the area of the triangle
        area = 0.5 * shape['base'] * shape['height']
    elif shape['type'] == 'square':
        # Check if the side length is provided
        if 'side' not in shape:
            raise ValueError("Side length is required for a square")
        # Calculate the area of the square
        area = shape['side'] ** 2
    else:
        raise ValueError("Invalid shape type")

    return area

def main():
    # Example usage
    circle = {'type': 'circle', 'radius': 5}
    rectangle = {'type': 'rectangle', 'length': 4, 'width': 6}
    triangle = {'type': 'triangle', 'base': 3, 'height': 7}
    square = {'type': 'square', 'side': 4}

    print(f"Circle area: {calculate_area(circle)}")
    print(f"Rectangle area: {calculate_area(rectangle)}")
    print(f"Triangle area: {calculate_area(triangle)}")
    print(f"Square area: {calculate_area(square)}")

if __name__ == "__main__":
    main()
```
