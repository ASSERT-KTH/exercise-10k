## Exercise
Write a program that takes a user's name, age, and email as input and prints out a user profile.

## Reference Solution
```python
class UserProfile:
    def __init__(self, name, age, email):
        """
        Initialize the user profile with name, age, and email.
        
        Args:
            name (str): The user's name.
            age (int): The user's age.
            email (str): The user's email.
        """
        self.name = name
        self.age = age
        self.email = email

    def print_profile(self):
        """
        Print out the user's profile.
        """
        print("User Profile:")
        print(f"Name: {self.name}")
        print(f"Age: {self.age}")
        print(f"Email: {self.email}")


def main():
    # Get user's name, age, and email
    name = input("Please enter your name: ")
    while True:
        try:
            age = int(input("Please enter your age: "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid age.")
    email = input("Please enter your email: ")

    # Create a user profile and print it
    user_profile = UserProfile(name, age, email)
    user_profile.print_profile()


if __name__ == "__main__":
    main()
```
