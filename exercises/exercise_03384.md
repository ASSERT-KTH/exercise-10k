## Exercise
Write a function to find the index of a given element in an array.

## Reference Solution
```python
def find_index(array, target):
    """
    Find the index of a given element in an array.

    Args:
        array (list): The array to search.
        target: The element to find.

    Returns:
        int: The index of the target element if found, -1 otherwise.
    """
    try:
        return array.index(target)
    except ValueError:
        return -1

# Example usage:
array = [1, 2, 3, 4, 5]
target = 3
index = find_index(array, target)
print(f"Index of {target}: {index}")
```

def find_all_indices(array, target):
    return [i for i, x in enumerate(array) if x == target]

# Example usage:
array = [1, 2, 3, 2, 4, 2, 5]
target = 2
indices = find_all_indices(array, target)
print(f"Indices of {target}: {indices}")
