## Exercise
Write a program that performs a binary search on a sorted list of numbers to find a specific target value.

## Reference Solution
```python
def binary_search(sorted_list, target):
    low = 0
    high = len(sorted_list) - 1

    while low <= high:
        mid = (low + high) // 2
        guess = sorted_list[mid]

        if guess == target:
            return mid
        elif guess > target:
            high = mid - 1
        else:
            low = mid + 1

    return None

my_list = [1, 3, 5, 7, 9]
print(binary_search(my_list, 3))  # Output: 1

# Test cases
print(binary_search([1, 3, 5, 7, 9], 1))  # Output: 0
print(binary_search([1, 3, 5, 7, 9], 9))  # Output: 4
print(binary_search([1, 3, 5, 7, 9], 0))  # Output: None
print(binary_search([1, 3, 5, 7, 9], 10))  # Output: None
```
