## Exercise
Write a function that takes a list of dictionaries and returns a dictionary with the dictionaries grouped by a common key

## Reference Solution
```python
from collections import defaultdict

def group_dictionaries(dict_list, key):
    """
    Group a list of dictionaries by a common key.

    Args:
        dict_list (list): A list of dictionaries.
        key (str): The key to group by.

    Returns:
        dict: A dictionary with the dictionaries grouped by the common key.
    """
    grouped_dict = defaultdict(list)
    for dictionary in dict_list:
        if key in dictionary:
            grouped_dict[dictionary[key]].append(dictionary)
        else:
            raise KeyError(f"All dictionaries must have the key '{key}'")
    return dict(grouped_dict)

# Example usage
dict_list = [
    {"id": 1, "name": "John", "age": 30},
    {"id": 1, "name": "Jane", "age": 25},
    {"id": 2, "name": "Bob", "age": 40},
    {"id": 2, "name": "Alice", "age": 35},
    {"id": 3, "name": "Charlie", "age": 45}
]

grouped_dict = group_dictionaries(dict_list, "id")
print(grouped_dict)
```
