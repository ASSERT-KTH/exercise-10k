## Exercise
Create a program to calculate the Pearson correlation coefficient between two datasets. If the correlation coefficient is above 0.7, the datasets are strongly correlated.

## Reference Solution
```python
import numpy as np

def calculate_pearson_correlation(dataset1, dataset2):
    """
    Calculate the Pearson correlation coefficient between two datasets.

    Args:
    - dataset1 (list or numpy array): The first dataset.
    - dataset2 (list or numpy array): The second dataset.

    Returns:
    - correlation_coefficient (float): The Pearson correlation coefficient.
    """
    # Convert input data to numpy arrays
    dataset1 = np.array(dataset1)
    dataset2 = np.array(dataset2)

    # Calculate the means of the datasets
    mean1 = np.mean(dataset1)
    mean2 = np.mean(dataset2)

    # Calculate the deviations from the means
    deviations1 = dataset1 - mean1
    deviations2 = dataset2 - mean2

    # Calculate the covariance and variances
    covariance = np.sum(deviations1 * deviations2)
    variance1 = np.sum(deviations1 ** 2)
    variance2 = np.sum(deviations2 ** 2)

    # Calculate the Pearson correlation coefficient
    correlation_coefficient = covariance / np.sqrt(variance1 * variance2)

    return correlation_coefficient

def check_correlation(correlation_coefficient):
    """
    Check if the correlation coefficient indicates strong correlation.

    Args:
    - correlation_coefficient (float): The Pearson correlation coefficient.

    Returns:
    - correlation_status (str): A message indicating the correlation status.
    """
    if np.abs(correlation_coefficient) > 0.7:
        correlation_status = "The datasets are strongly correlated."
    else:
        correlation_status = "The datasets are not strongly correlated."

    return correlation_status

# Example usage
if __name__ == "__main__":
    dataset1 = [1, 2, 3, 4, 5]
    dataset2 = [2, 3, 5, 7, 11]

    correlation_coefficient = calculate_pearson_correlation(dataset1, dataset2)
    correlation_status = check_correlation(correlation_coefficient)

    print(f"Pearson correlation coefficient: {correlation_coefficient:.4f}")
    print(correlation_status)
```
