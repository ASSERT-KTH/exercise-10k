## Exercise
Create a program that accumulates snowfall amounts over a period of time, using conditional statements to determine the accumulation rules

## Reference Solution
```python
from datetime import datetime

class SnowfallAccumulator:
    def __init__(self):
        self.snowfall_amount = 0

    def add_snowfall(self, date, amount):
        """
        Add snowfall to the accumulator.

        Args:
            date (str): Date of the snowfall event in YYYY-MM-DD format.
            amount (float): Amount of snowfall in inches.
        """
        if amount < 1:
            # Do not accumulate if amount is less than 1 inch
            pass
        elif amount <= 3:
            # Accumulate as is if amount is between 1 inch and 3 inches
            self.snowfall_amount += amount
        else:
            # Accumulate with 10% reduction if amount is greater than 3 inches
            self.snowfall_amount += amount * 0.9

    def get_snowfall_amount(self):
        """
        Get the accumulated snowfall amount.

        Returns:
            float: Accumulated snowfall amount in inches.
        """
        return self.snowfall_amount


def main():
    # Create a SnowfallAccumulator instance
    accumulator = SnowfallAccumulator()

    # Add snowfall events
    snowfall_events = [
        ("2022-01-01", 0.5),
        ("2022-01-02", 2),
        ("2022-01-03", 4),
        ("2022-01-04", 1.5),
        ("2022-01-05", 5)
    ]

    for date, amount in snowfall_events:
        accumulator.add_snowfall(date, amount)

    # Get the accumulated snowfall amount
    snowfall_amount = accumulator.get_snowfall_amount()

    print(f"Accumulated snowfall amount: {snowfall_amount} inches")


if __name__ == "__main__":
    main()
```
