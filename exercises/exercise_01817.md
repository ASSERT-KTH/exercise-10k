## Exercise
Write a program that plans a university course schedule, using conditional statements to determine the course conflicts

## Reference Solution
```python
class Course:
    """Represents a university course."""
    def __init__(self, name, start_time, end_time, days):
        """
        Initializes a Course object.

        Args:
            name (str): Course name.
            start_time (str): Course start time (HH:MM).
            end_time (str): Course end time (HH:MM).
            days (list): List of days the course is held (e.g., ['M', 'T', 'W']).
        """
        self.name = name
        self.start_time = start_time
        self.end_time = end_time
        self.days = days

    def __str__(self):
        """Returns a string representation of the course."""
        return f"{self.name} - {self.start_time} to {self.end_time} on {', '.join(self.days)}"


class Schedule:
    """Represents a university course schedule."""
    def __init__(self):
        """Initializes an empty schedule."""
        self.courses = []

    def add_course(self, course):
        """
        Adds a course to the schedule.

        Args:
            course (Course): Course to add.
        """
        self.courses.append(course)

    def check_conflicts(self, course):
        """
        Checks if a course conflicts with any existing courses in the schedule.

        Args:
            course (Course): Course to check for conflicts.

        Returns:
            bool: True if the course has a conflict, False otherwise.
        """
        for existing_course in self.courses:
            if set(course.days) & set(existing_course.days):  # Check if courses are on the same day
                if (course.start_time < existing_course.end_time) and (existing_course.start_time < course.end_time):
                    return True
        return False

    def print_schedule(self):
        """Prints the current schedule."""
        for course in self.courses:
            print(course)


def main():
    schedule = Schedule()

    while True:
        print("\n1. Add course")
        print("2. Print schedule")
        print("3. Check course conflicts")
        print("4. Exit")
        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter course name: ")
            start_time = input("Enter course start time (HH:MM): ")
            end_time = input("Enter course end time (HH:MM): ")
            days = input("Enter course days (e.g., M, T, W): ").split(', ')
            course = Course(name, start_time, end_time, days)
            if not schedule.check_conflicts(course):
                schedule.add_course(course)
                print(f"Added {course.name} to schedule.")
            else:
                print(f"Cannot add {course.name} to schedule due to conflicts.")
        elif choice == "2":
            schedule.print_schedule()
        elif choice == "3":
            name = input("Enter course name: ")
            start_time = input("Enter course start time (HH:MM): ")
            end_time = input("Enter course end time (HH:MM): ")
            days = input("Enter course days (e.g., M, T, W): ").split(', ')
            course = Course(name, start_time, end_time, days)
            if schedule.check_conflicts(course):
                print(f"{course.name} has conflicts with existing courses.")
            else:
                print(f"{course.name} does not have conflicts with existing courses.")
        elif choice == "4":
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
