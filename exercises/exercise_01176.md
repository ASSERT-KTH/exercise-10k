## Exercise
Given two strings, check if the second string is a rotation of the first string

## Reference Solution

def is_rotation(s1: str, s2: str) -> bool:
    """
    Check if s2 is a rotation of s1.

    Args:
        s1 (str): The original string.
        s2 (str): The string to check.

    Returns:
        bool: True if s2 is a rotation of s1, False otherwise.
    """
    if len(s1) != len(s2):
        return False
    return s2 in s1 + s1

# Example usage
s1 = "abcde"
s2 = "cdeab"
print(is_rotation(s1, s2))  # Output: True

s1 = "abcde"
s2 = "abecd"
print(is_rotation(s1, s2))  # Output: False
