## Exercise
Create a program that calculates the average grade of a student based on their scores in different subjects.

## Reference Solution
```python
def calculate_average_grade(scores, weights=None):
    """
    Calculate the average grade of a student based on their scores in different subjects.

    Args:
        scores (dict): Dictionary of subject names and their respective scores.
        weights (dict, optional): Dictionary of subject names and their respective weights. Defaults to equal weights for all subjects.

    Returns:
        float: The average grade of the student.
    """
    if weights is None:
        # If no weights are provided, assign equal weights to all subjects
        weights = {subject: 1 / len(scores) for subject in scores}

    total_score = sum(scores[subject] * weights[subject] for subject in scores)
    total_weight = sum(weights.values())

    average_grade = total_score / total_weight

    return average_grade

def main():
    # Example usage
    scores = {
        "Math": 85,
        "Science": 90,
        "English": 78,
        "History": 92
    }

    # Optional weights
    weights = {
        "Math": 0.3,
        "Science": 0.3,
        "English": 0.2,
        "History": 0.2
    }

    average_grade = calculate_average_grade(scores, weights)
    print(f"Average Grade: {average_grade:.2f}")

if __name__ == "__main__":
    main()
```
