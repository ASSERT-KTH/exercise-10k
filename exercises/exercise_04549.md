## Exercise
Create a program that implements a Caesar cipher to encrypt and decrypt a given string.

## Reference Solution
```python
def caesar_encrypt(text: str, shift: int) -> str:
    """
    Encrypts a given string using a Caesar cipher.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The number of positions to shift the alphabet.

    Returns:
        str: The encrypted string.
    """
    result = ""

    for char in text:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            result += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            result += char

    return result


def caesar_decrypt(text: str, shift: int) -> str:
    """
    Decrypts a given string using a Caesar cipher.

    Args:
        text (str): The input string to be decrypted.
        shift (int): The number of positions to shift the alphabet.

    Returns:
        str: The decrypted string.
    """
    return caesar_encrypt(text, -shift)


def main():
    text = "Hello, World!"
    shift = 3

    encrypted_text = caesar_encrypt(text, shift)
    print(f"Encrypted text: {encrypted_text}")

    decrypted_text = caesar_decrypt(encrypted_text, shift)
    print(f"Decrypted text: {decrypted_text}")


if __name__ == "__main__":
    main()
```
