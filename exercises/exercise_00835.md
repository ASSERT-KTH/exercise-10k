Given three strings, check if the third string is an interleaving of the first two strings.

def is_interleave(s1: str, s2: str, s3: str) -> bool:
    """
    Check if s3 is an interleaving of s1 and s2.

    Args:
    s1 (str): The first string.
    s2 (str): The second string.
    s3 (str): The third string.

    Returns:
    bool: True if s3 is an interleaving of s1 and s2, False otherwise.
    """
    if len(s1) + len(s2) != len(s3):
        return False

    # Create a 2D table to store the results of subproblems
    dp = [[False] * (len(s2) + 1) for _ in range(len(s1) + 1)]

    # Initialize the base cases
    dp[0][0] = True
    for i in range(1, len(s1) + 1):
        dp[i][0] = dp[i - 1][0] and s1[i - 1] == s3[i - 1]
    for j in range(1, len(s2) + 1):
        dp[0][j] = dp[0][j - 1] and s2[j - 1] == s3[j - 1]

    # Fill in the rest of the table
    for i in range(1, len(s1) + 1):
        for j in range(1, len(s2) + 1):
            dp[i][j] = (dp[i - 1][j] and s1[i - 1] == s3[i + j - 1]) or (dp[i][j - 1] and s2[j - 1] == s3[i + j - 1])

    return dp[len(s1)][len(s2)]
