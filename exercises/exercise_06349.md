Declare and use boolean variables to represent true or false values. Explain their use cases and provide examples.

```python
# Declare a boolean variable
is_admin = True

# Use the boolean variable in a conditional statement
if is_admin:

else:
    print("You are not an admin")
```

```python
# Declare a boolean variable
is_running = True

# Use the boolean variable in a loop
while is_running:
    print("The program is running")
    # Add code to stop the program
    is_running = False
```

```python
# Declare boolean variables
is_admin = True
is_verified = False

# Use logical operations
if is_admin and is_verified:
    print("You are an admin and verified")
elif is_admin or is_verified:
    print("You are either an admin or verified")
else:
    print("You are neither an admin nor verified")
```

```python
# Declare a function with a boolean return type
def is_even(num: int) -> bool:
    return num % 2 == 0

# Use the function
print(is_even(10))  # Output: True
print(is_even(11))  # Output: False
```
