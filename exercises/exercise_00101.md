```python
def sum_array_elements(array):
    """
    This function calculates the sum of all elements in an array.

    Args:
        array (list): A list of numbers.

    Returns:
        int: The sum of all elements in the array.
    """
    total = 0
    for element in array:
        total += element
    return total


# Example usage:
array = [1, 2, 3, 4, 5]
result = sum_array_elements(array)
print("Sum of array elements:", result)
```
