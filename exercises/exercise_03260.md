## Exercise
Create a program that performs arithmetic operations on rational numbers.

```python
class Rational:
    def __init__(self, numerator, denominator):
        """
        Initialize a rational number with numerator and denominator.

        Args:
            numerator (int): The numerator of the rational number.
            denominator (int): The denominator of the rational number.

        Raises:
            ValueError: If the denominator is zero.
        """
        if denominator == 0:
            raise ValueError("Denominator cannot be zero")
        self.numerator = numerator
        self.denominator = denominator
        self.simplify()

    def simplify(self):
        """
        Simplify the rational number by dividing both numerator and denominator by their GCD.
        """
        def gcd(a, b):
            while b:
                a, b = b, a % b
            return a

        gcd_val = gcd(self.numerator, self.denominator)
        self.numerator //= gcd_val
        self.denominator //= gcd_val

    def __str__(self):
        """
        Return a string representation of the rational number.
        """
        return f"{self.numerator}/{self.denominator}"

    def __add__(self, other):
        """
        Add two rational numbers.

        Args:
            other (Rational): The other rational number to add.

        Returns:
            Rational: The sum of the two rational numbers.
        """
        numerator = self.numerator * other.denominator + other.numerator * self.denominator
        denominator = self.denominator * other.denominator
        return Rational(numerator, denominator)

    def __sub__(self, other):
        """
        Subtract one rational number from another.

        Args:
            other (Rational): The other rational number to subtract.

        Returns:
            Rational: The difference of the two rational numbers.
        """
        numerator = self.numerator * other.denominator - other.numerator * self.denominator
        denominator = self.denominator * other.denominator
        return Rational(numerator, denominator)

    def __mul__(self, other):
        """
        Multiply two rational numbers.

        Args:
            other (Rational): The other rational number to multiply.

        Returns:
            Rational: The product of the two rational numbers.
        """
        numerator = self.numerator * other.numerator
        denominator = self.denominator * other.denominator
        return Rational(numerator, denominator)

    def __truediv__(self, other):
        """
        Divide one rational number by another.

        Args:
            other (Rational): The other rational number to divide by.

        Returns:
            Rational: The quotient of the two rational numbers.

        Raises:
            ValueError: If the other rational number is zero.
        """
        if other.numerator == 0:
            raise ValueError("Cannot divide by zero")
        numerator = self.numerator * other.denominator
        denominator = self.denominator * other.numerator
        return Rational(numerator, denominator)


def main():
    # Example usage
    r1 = Rational(1, 2)
    r2 = Rational(1, 3)

    print(f"r1 = {r1}")
    print(f"r2 = {r2}")

    print(f"r1 + r2 = {r1 + r2}")
    print(f"r1 - r2 = {r1 - r2}")
    print(f"r1 * r2 = {r1 * r2}")
    print(f"r1 / r2 = {r1 / r2}")


if __name__ == "__main__":
    main()
```
