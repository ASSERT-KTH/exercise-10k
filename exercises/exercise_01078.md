## Exercise
Create a program that uses a for loop to print all prime numbers up to a given number.

## Reference Solution
```python
def is_prime(n):
    """Check if a number is prime."""
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

def print_primes(n):
    """Print all prime numbers up to n."""
    print("Prime numbers up to", n, ":")
    for i in range(2, n + 1):
        if is_prime(i):
            print(i, end=" ")

def main():
    num = int(input("Enter a number: "))
    print_primes(num)

if __name__ == "__main__":
    main()
```
