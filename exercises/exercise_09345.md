## Exercise
Write a program to demonstrate the concept of memory leaks.

## Reference Solution
```python
import os
import psutil
import time

# Function to get the current memory usage
def get_memory_usage():
    process = psutil.Process(os.getpid())
    return process.memory_info().rss / (1024 * 1024)  # in MB

# Main function
def main():
    # Initialize an empty list
    leaky_list = []

    print("Initial memory usage: {:.2f} MB".format(get_memory_usage()))

    # Continuously append new objects to the list
    for i in range(1000000):
        leaky_list.append(bytearray(1024))  # Append a 1KB bytearray
        if i % 10000 == 0:
            print("Memory usage after {} iterations: {:.2f} MB".format(i, get_memory_usage()))

    # Keep the program running to observe the memory usage
    while True:
        time.sleep(1)

if __name__ == "__main__":
    main()
```
