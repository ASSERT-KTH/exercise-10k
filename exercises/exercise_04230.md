## Exercise
Write a program to calculate the compound interest on an investment.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, frequency):
    """
    Calculate compound interest on an investment.

    Args:
        principal (float): Initial investment amount
        rate (float): Annual interest rate (in decimal form, e.g. 5% = 0.05)
        time (int): Time period in years
        frequency (int): Compounding frequency (e.g. 1 for annually, 12 for monthly)

    Returns:
        float: Total amount after compound interest
    """
    return principal * (1 + rate / frequency) ** (frequency * time)

def main():
    # Get user input
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period in years: "))
    frequency = int(input("Enter the compounding frequency (e.g. 1 for annually, 12 for monthly): "))

    # Calculate compound interest
    total_amount = calculate_compound_interest(principal, rate, time, frequency)

    # Print result
    print(f"The total amount after {time} years is: ${total_amount:.2f}")
    print(f"The compound interest earned is: ${total_amount - principal:.2f}")

if __name__ == "__main__":
    main()
```
