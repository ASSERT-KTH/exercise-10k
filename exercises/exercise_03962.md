# Importing Modules
import sys

# Defining Functions
def bubble_sort(arr):
    for i in range(len(arr) - 1):
        for j in range(len(arr) - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

# Main Program
if __name__ == '__main__':
    arr = [5, 2, 8, 1, 9]
    sorted_arr = bubble_sort(arr)
    print(sorted_arr)
