## Exercise
Create a program that finds the first duplicate element in a list using a while loop.

## Reference Solution
```python
def find_first_duplicate(input_list):
    """
    Finds the first duplicate element in a list.

    Args:
        input_list (list): The list to search for duplicates.

    Returns:
        The first duplicate element in the list, or None if no duplicates are found.
    """
    seen = set()
    i = 0
    while i < len(input_list):
        if input_list[i] in seen:
            return input_list[i]
        seen.add(input_list[i])
        i += 1
    return None


# Example usage
if __name__ == "__main__":
    input_list = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(input_list)
    if duplicate is not None:
        print(f"The first duplicate element in the list is {duplicate}.")
    else:
        print("No duplicates found in the list.")
```
