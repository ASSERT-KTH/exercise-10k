## Exercise
Write a program that takes two matrices as input, and returns their product.

## Reference Solution
```python
import numpy as np

def matrix_multiply(A, B):
    """
    Compute the product of two matrices A and B.

    Args:
        A (np.ndarray): The first matrix.
        B (np.ndarray): The second matrix.

    Returns:
        np.ndarray: The product of A and B.

    Raises:
        ValueError: If the number of columns in A is not equal to the number of rows in B.
    """
    # Check if the matrices can be multiplied
    if A.shape[1] != B.shape[0]:
        raise ValueError("The number of columns in A must be equal to the number of rows in B")

    # Use NumPy's built-in matrix multiplication function
    return np.matmul(A, B)

# Example usage
if __name__ == "__main__":
    # Define two example matrices
    A = np.array([[1, 2], [3, 4]])
    B = np.array([[5, 6], [7, 8]])

    # Compute the product of A and B
    product = matrix_multiply(A, B)

    # Print the result
    print("Matrix A:")
    print(A)
    print("Matrix B:")
    print(B)
    print("Product of A and B:")
    print(product)
```
