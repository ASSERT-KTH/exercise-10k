## Exercise
Perform numerical computations on two numbers and print out the result.

## Reference Solution
```python
def perform_computations(num1, num2):
    """
    Perform basic numerical computations on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.

    Returns:
        None
    """

    # Addition
    addition_result = num1 + num2
    print(f"Addition: {num1} + {num2} = {addition_result}")

    # Subtraction
    subtraction_result = num1 - num2
    print(f"Subtraction: {num1} - {num2} = {subtraction_result}")

    # Multiplication
    multiplication_result = num1 * num2
    print(f"Multiplication: {num1} * {num2} = {multiplication_result}")

    # Division
    if num2 != 0:
        division_result = num1 / num2
        print(f"Division: {num1} / {num2} = {division_result}")
    else:
        print("Error: Division by zero is not allowed.")

    # Modulus (remainder)
    if num2 != 0:
        modulus_result = num1 % num2
        print(f"Modulus: {num1} % {num2} = {modulus_result}")
    else:
        print("Error: Modulus operation by zero is not allowed.")

    # Exponentiation
    exponentiation_result = num1 ** num2
    print(f"Exponentiation: {num1} ^ {num2} = {exponentiation_result}")


def main():
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    perform_computations(num1, num2)


if __name__ == "__main__":
    main()
```
