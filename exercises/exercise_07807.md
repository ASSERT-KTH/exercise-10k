## Exercise
Write a program that converts a temperature from one unit to another (e.g. from Celsius to Fahrenheit).

## Reference Solution
```python
def convert_temperature(value, from_unit, to_unit):
    """
    Converts a temperature from one unit to another.

    Args:
        value (float): The temperature value to convert.
        from_unit (str): The unit of the input temperature. Can be 'celsius', 'fahrenheit', or 'kelvin'.
        to_unit (str): The unit to convert the temperature to. Can be 'celsius', 'fahrenheit', or 'kelvin'.

    Returns:
        float: The converted temperature value.
    """

    # Define conversion formulas
    conversions = {
        ('celsius', 'fahrenheit'): lambda x: x * 9/5 + 32,
        ('fahrenheit', 'celsius'): lambda x: (x - 32) * 5/9,
        ('celsius', 'kelvin'): lambda x: x + 273.15,
        ('kelvin', 'celsius'): lambda x: x - 273.15,
        ('fahrenheit', 'kelvin'): lambda x: (x - 32) * 5/9 + 273.15,
        ('kelvin', 'fahrenheit'): lambda x: (x - 273.15) * 9/5 + 32
    }

    # Check if units are valid
    valid_units = ['celsius', 'fahrenheit', 'kelvin']
    if from_unit not in valid_units or to_unit not in valid_units:
        raise ValueError("Invalid unit. Supported units are 'celsius', 'fahrenheit', and 'kelvin'.")

    # Check if from_unit and to_unit are the same
    if from_unit == to_unit:
        return value

    # Get conversion formula
    conversion = conversions.get((from_unit, to_unit))
    if conversion is None:
        raise ValueError("Invalid unit conversion. Please check the units.")

    # Apply conversion formula
    return conversion(value)


def main():
    print("Temperature Conversion Program")
    print("--------------------------------")

    # Get user input
    value = float(input("Enter the temperature value: "))
    from_unit = input("Enter the unit of the input temperature (celsius, fahrenheit, kelvin): ").lower()
    to_unit = input("Enter the unit to convert the temperature to (celsius, fahrenheit, kelvin): ").lower()

    # Convert temperature
    try:
        result = convert_temperature(value, from_unit, to_unit)
        print(f"{value} {from_unit} is equal to {result} {to_unit}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
