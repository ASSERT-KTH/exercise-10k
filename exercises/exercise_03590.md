## Exercise
Create a program that calculates the body mass index (BMI) given a person's weight and height.

## Reference Solution
```python
def calculate_bmi(weight, height):
    """
    Calculates the body mass index (BMI) given weight and height.

    Args:
        weight (float): Weight in kilograms (kg)
        height (float): Height in meters (m)

    Returns:
        float: Calculated BMI
    """
    bmi = weight / (height ** 2)
    return bmi

def get_bmi_category(bmi):
    """
    Determines the BMI category based on the calculated BMI.

    Args:
        bmi (float): Calculated BMI

    Returns:
        str: BMI category (Underweight, Normal weight, Overweight, Obese)
    """
    if bmi < 18.5:
        return "Underweight"
    elif bmi < 25:
        return "Normal weight"
    elif bmi < 30:
        return "Overweight"
    else:
        return "Obese"

def main():
    weight = float(input("Enter your weight in kilograms (kg): "))
    height = float(input("Enter your height in meters (m): "))

    bmi = calculate_bmi(weight, height)
    category = get_bmi_category(bmi)

    print(f"Your BMI is: {bmi:.2f}")
    print(f"Your BMI category is: {category}")

if __name__ == "__main__":
    main()
```
