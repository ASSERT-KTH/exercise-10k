## Exercise
Write a program to declare and use a dictionary to store a collection of key-value pairs.

## Reference Solution
```python
# Define a dictionary
def main():
    # Initialize an empty dictionary
    person = {}

    # Add key-value pairs to the dictionary
    person['name'] = 'John Doe'
    person['age'] = 30
    person['city'] = 'New York'

    # Print the dictionary
    print("Person Details:")
    print(person)

    # Access a value using its key
    print("\nName:", person['name'])
    print("Age:", person['age'])
    print("City:", person['city'])

    # Update a value
    person['age'] = 31
    print("\nUpdated Age:", person['age'])

    # Remove a key-value pair
    del person['city']
    print("\nUpdated Person Details:")
    print(person)

    # Check if a key exists in the dictionary
    if 'name' in person:
        print("\nName exists in the dictionary.")
    else:
        print("\nName does not exist in the dictionary.")

    # Iterate over the dictionary
    print("\nIterating over the dictionary:")
    for key, value in person.items():
        print(f"{key}: {value}")

if __name__ == "__main__":
    main()
```
