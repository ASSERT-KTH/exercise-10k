## Exercise
Write a function to filter a dictionary based on a condition (e.g., filter out keys with empty values).

## Reference Solution
```python
def filter_dict(input_dict, condition=None):
    """
    Filter a dictionary based on a condition.

    Args:
        input_dict (dict): The dictionary to be filtered.
        condition (function, optional): A function that takes a key-value pair and returns a boolean. Defaults to None.

    Returns:
        dict: The filtered dictionary.
    """
    if condition is None:
        # Default condition: filter out keys with empty values
        condition = lambda x: x[1] not in [None, "", [], {}, ()]

    return {key: value for key, value in input_dict.items() if condition((key, value))}

# Example usage:
if __name__ == "__main__":
    input_dict = {
        "key1": "value1",
        "key2": "",
        "key3": None,
        "key4": [],
        "key5": {},
        "key6": "value6"
    }

    # Filter out keys with empty values
    filtered_dict = filter_dict(input_dict)
    print("Filtered dictionary (default condition):")
    print(filtered_dict)

    # Filter out keys with values that are not strings
    filtered_dict = filter_dict(input_dict, condition=lambda x: isinstance(x[1], str))
    print("\nFiltered dictionary (custom condition):")
    print(filtered_dict)
```
