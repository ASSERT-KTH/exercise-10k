Create a function that generates a random piece of art, such as a fractal or a maze.

import numpy as np
import matplotlib.pyplot as plt

def generate_fractal(xmin,xmax,ymin,ymax,width,height,max_iter):
    """
    Generates a Mandelbrot fractal.

    Args:
        xmin (float): Minimum x-coordinate.
        xmax (float): Maximum x-coordinate.
        ymin (float): Minimum y-coordinate.
        ymax (float): Maximum y-coordinate.
        width (int): Width of the image.
        height (int): Height of the image.
        max_iter (int): Maximum number of iterations.

    Returns:
        A 2D array representing the fractal.
    """
    r1 = np.linspace(xmin, xmax, width)
    r2 = np.linspace(ymin, ymax, height)
    return (r1,r2,np.array([[mandelbrot(complex(r, i),max_iter) for r in r1] for i in r2]))

def mandelbrot(c, max_iter):
    """
    Computes the number of iterations until the point c escapes the Mandelbrot set.

    Args:
        c (complex): Point in the complex plane.
        max_iter (int): Maximum number of iterations.

    Returns:
        The number of iterations until the point c escapes the Mandelbrot set.
    """
    z = c
    for n in range(max_iter):
        if abs(z) > 2:
            return n
        z = z*z + c
    return max_iter

def draw_image(xmin,xmax,ymin,ymax,width,height,max_iter):
    """
    Draws the Mandelbrot fractal.

    Args:
        xmin (float): Minimum x-coordinate.
        xmax (float): Maximum x-coordinate.
        ymin (float): Minimum y-coordinate.
        ymax (float): Maximum y-coordinate.
        width (int): Width of the image.
        height (int): Height of the image.
        max_iter (int): Maximum number of iterations.
    """
    d = generate_fractal(xmin,xmax,ymin,ymax,width,height,max_iter)
    plt.imshow(d[2], extent=(xmin, xmax, ymin, ymax))
    plt.show()

# Example usage:
draw_image(-2.0,1.0,-1.5,1.5,1000,1000,256)
