## Exercise
Create a program that plays Rock, Paper, Scissors and determines the winner using if/else statements

## Reference Solution
```python
import random

def get_computer_choice():
    """Get the computer's random choice"""
    choices = ["rock", "paper", "scissors"]
    return random.choice(choices)

def determine_winner(player1, player2):
    """Determine the winner of the game"""
    if player1 == player2:
        return "It's a tie!"
    elif (player1 == "rock" and player2 == "scissors") or \
         (player1 == "scissors" and player2 == "paper") or \
         (player1 == "paper" and player2 == "rock"):
        return "Player 1 wins!"
    else:
        return "Player 2 wins!"

def play_game():
    """Play a game of Rock, Paper, Scissors"""
    print("Welcome to Rock, Paper, Scissors!")
    player1 = input("Enter your choice (rock, paper, or scissors): ").lower()
    while player1 not in ["rock", "paper", "scissors"]:
        player1 = input("Invalid choice. Please enter rock, paper, or scissors: ").lower()
    computer = get_computer_choice()
    print(f"\nYou chose: {player1}")
    print(f"Computer chose: {computer}\n")
    print(determine_winner(player1, computer))

def main():
    play_again = "yes"
    while play_again.lower() == "yes":
        play_game()
        play_again = input("\nPlay again? (yes/no): ")

if __name__ == "__main__":
    main()
```
