Create a function that initializes a dictionary with keys from 1 to 10 and values as their squares.

def initialize_dict():
    """
    This function initializes a dictionary with keys from 1 to 10 and values as their squares.

    Returns:
        dict: A dictionary containing keys from 1 to 10 and their corresponding squares.
    """

    # We start by defining the function initialize_dict that will return our dictionary.
    
    # We use dictionary comprehension to generate the dictionary. 
    # The expression {i: i**2 for i in range(1, 11)} generates a dictionary where each key is a number from 1 to 10 and its corresponding value is the square of the key.
    dict_with_squares = {i: i**2 for i in range(1, 11)}
    
    # Finally, we return the dictionary.
    return dict_with_squares
