## Exercise
Write a program that generates a random joke based on user input

## Reference Solution
```python
import random

def generate_joke(topic):
    # Dictionary of joke templates
    joke_templates = {
        "animal": [
            "Why did the {} go to the party? Because it was a {}!",
            "Why did the {} go to the doctor? It had a {}-ful cough!",
            "What did the {} say when it ran into its friend? 'Long time no {}!'"
        ],
        "food": [
            "Why was the {} sad? It was feeling a little {}-down.",
            "What did the {} say when it got old? 'I'm feeling a little {}-stale.'",
            "Why did the {} go to therapy? It was struggling with {}-related issues."
        ],
        "place": [
            "Why did the person bring a {} to the {}? In case they got {}-thirsty!",
            "What did the {} say to the {} when they met? 'It's great to finally {}-meet you!'",
            "Why did the person go to the {}? To get some {}-fresh air."
        ]
    }

    # Choose a random joke template
    template = random.choice(joke_templates[topic])

    # Generate a random punchline
    punchline = random.choice(["great", "amazing", "awesome", "funny", "silly"])

    # Return the joke
    return template.format(topic, punchline)

def main():
    print("Welcome to the joke generator!")
    print("Please choose a topic:")
    print("1. Animal")
    print("2. Food")
    print("3. Place")

    choice = input("Enter your choice (1, 2, or 3): ")

    if choice == "1":
        topic = "animal"
    elif choice == "2":
        topic = "food"
    elif choice == "3":
        topic = "place"
    else:
        print("Invalid choice. Defaulting to animal.")
        topic = "animal"

    joke = generate_joke(topic)
    print("Here's your joke:")
    print(joke)

if __name__ == "__main__":
    main()
```
