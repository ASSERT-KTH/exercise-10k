## Exercise
Write a function that finds the union of characters between two given strings.

## Reference Solution
```python
def union_of_characters(str1, str2):
    """
    Returns the union of characters between two given strings.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        set: A set of unique characters from both strings.
    """
    return set(str1).union(set(str2))

def main():
    # Example usage
    str1 = "hello"
    str2 = "world"
    union_chars = union_of_characters(str1, str2)
    print("Union of characters:", union_chars)

if __name__ == "__main__":
    main()
```
