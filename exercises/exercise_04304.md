## Exercise
Write a function that takes two player choices as input and returns the winner

def determine_winner(player1_choice, player2_choice):
    """
    Determine the winner of a Rock-Paper-Scissors game.

    Args:
        player1_choice (str): The choice of player 1 (rock, paper, or scissors).
        player2_choice (str): The choice of player 2 (rock, paper, or scissors).

    Returns:
        str: The winner of the game (player1, player2, or tie).
    """

    # Define the winning conditions
    winning_conditions = {
        "rock": "scissors",
        "scissors": "paper",
        "paper": "rock"
    }

    # Check if player 1 wins
    if winning_conditions[player1_choice] == player2_choice:
        return "player1"

    # Check if player 2 wins
    elif winning_conditions[player2_choice] == player1_choice:
        return "player2"

    # If neither player wins, it's a tie
    else:
        return "tie"

# Example usage
if __name__ == "__main__":
    print(determine_winner("rock", "scissors"))  # Output: player1
    print(determine_winner("scissors", "paper"))  # Output: player1
    print(determine_winner("paper", "rock"))  # Output: player1
    print(determine_winner("rock", "rock"))  # Output: tie
