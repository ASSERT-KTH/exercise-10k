## Exercise
Write a program that asks the user 3 trivia questions and keeps track of the score.

## Reference Solution
```python
# Define a dictionary of trivia questions
trivia_questions = {
    "What is the largest planet in our solar system?": "Jupiter",
    "Which country is the largest in Scandinavia?": "Sweden",
    "What is the chemical symbol for gold?": "Au"
}

def ask_question(question, answer):
    """
    Ask the user a trivia question and check their answer.

    Args:
        question (str): The trivia question.
        answer (str): The correct answer.

    Returns:
        bool: True if the user's answer is correct, False otherwise.
    """
    user_answer = input(question + " ")
    return user_answer.lower() == answer.lower()

def main():
    score = 0
    for question, answer in trivia_questions.items():
        if ask_question(question, answer):
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer was {answer}.\n")

    print(f"Game over! Your final score is {score} out of {len(trivia_questions)}.")

if __name__ == "__main__":
    main()
```
