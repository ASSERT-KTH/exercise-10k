## Exercise
Write a program that prints the factorials of numbers from 1 to 10 using a for loop.

## Reference Solution
```python
def calculate_factorial(n):
    """Calculates the factorial of a given number."""
    factorial = 1
    for i in range(1, n + 1):
        factorial *= i
    return factorial

def main():
    """Prints the factorials of numbers from 1 to 10."""
    for i in range(1, 11):
        factorial = calculate_factorial(i)
        print(f"The factorial of {i} is: {factorial}")

if __name__ == "__main__":
    main()
```

def calculate_factorial(n, memo = {}):
    """Calculates the factorial of a given number using memoization."""
    if n in memo:
        return memo[n]
    if n == 1 or n == 0:
        return 1
    else:
        factorial = n * calculate_factorial(n - 1, memo)
        memo[n] = factorial
        return factorial

def main():
    """Prints the factorials of numbers from 1 to 10."""
