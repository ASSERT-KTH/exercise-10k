## Exercise
Write a program to interact with a database using a database library.

## Reference Solution
```python
import sqlite3

class DatabaseInteraction:
    def __init__(self, db_name):
        """
        Initializes the database connection.

        :param db_name: Name of the database file
        """
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()

    def create_table(self, table_name, columns):
        """
        Creates a new table in the database.

        :param table_name: Name of the table
        :param columns: List of column names and data types (e.g., ["id INTEGER PRIMARY KEY", "name TEXT"])
        """
        column_defs = ", ".join(columns)
        query = f"CREATE TABLE IF NOT EXISTS {table_name} ({column_defs})"
        self.cursor.execute(query)
        self.conn.commit()

    def insert_data(self, table_name, data):
        """
        Inserts data into the specified table.

        :param table_name: Name of the table
        :param data: List of tuples containing the data to insert (e.g., [(1, "John"), (2, "Jane")])
        """
        query = f"INSERT INTO {table_name} VALUES ({', '.join(['?'] * len(data[0]))})"
        self.cursor.executemany(query, data)
        self.conn.commit()

    def retrieve_data(self, table_name):
        """
        Retrieves all data from the specified table.

        :param table_name: Name of the table
        :return: List of tuples containing the retrieved data
        """
        query = f"SELECT * FROM {table_name}"
        self.cursor.execute(query)
        return self.cursor.fetchall()

    def close_connection(self):
        """
        Closes the database connection.
        """
        self.conn.close()

# Example usage
if __name__ == "__main__":
    db_name = "example.db"
    table_name = "users"

    db = DatabaseInteraction(db_name)

    # Create table
    columns = ["id INTEGER PRIMARY KEY", "name TEXT"]
    db.create_table(table_name, columns)

    # Insert data
    data = [(1, "John"), (2, "Jane")]
    db.insert_data(table_name, data)

    # Retrieve data
    retrieved_data = db.retrieve_data(table_name)
    print("Retrieved Data:")
    for row in retrieved_data:
        print(row)

    # Close connection
    db.close_connection()
```
