## Exercise
Write a program that takes a story with missing words as input and asks the user to fill them in

## Reference Solution
```python
def get_user_input(prompt):
    """Get user input for a specific prompt."""
    while True:
        user_input = input(prompt)
        if user_input:
            return user_input
        print("Please enter a value.")

def main():
    # Example story with missing words
    story = """
Once upon a time, in a land far, far away, there was a {adjective1} {noun1}. 
This {noun1} loved to {verb1} in the {adjective2} {noun2}. 
One day, the {noun1} decided to {verb2} to the {adjective3} {noun3}.
"""

    # List of missing word types
    missing_words = [
        {"type": "adjective1", "prompt": "Enter an adjective (e.g. happy, tall, green): "},
        {"type": "noun1", "prompt": "Enter a noun (e.g. dog, castle, car): "},
        {"type": "verb1", "prompt": "Enter a verb (e.g. run, jump, read): "},
        {"type": "adjective2", "prompt": "Enter an adjective (e.g. big, old, sunny): "},
        {"type": "noun2", "prompt": "Enter a noun (e.g. park, house, tree): "},
        {"type": "verb2", "prompt": "Enter a verb (e.g. walk, drive, fly): "},
        {"type": "adjective3", "prompt": "Enter an adjective (e.g. beautiful, large, mysterious): "},
        {"type": "noun3", "prompt": "Enter a noun (e.g. city, forest, mountain): "},
    ]

    # Get user input for each missing word
    user_inputs = {}
    for word in missing_words:
        user_input = get_user_input(word["prompt"])
        user_inputs[word["type"]] = user_input

    # Fill in the missing words in the story
    filled_story = story
    for word_type, user_input in user_inputs.items():
        filled_story = filled_story.replace("{" + word_type + "}", user_input)

    # Print the filled story
    print("\nHere is the completed story:")
    print(filled_story)

if __name__ == "__main__":
    main()
```
