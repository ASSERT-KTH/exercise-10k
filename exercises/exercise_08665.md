## Exercise
Write a program that plays Rock, Paper, Scissors with the user.

## Reference Solution
```python
import random

def get_computer_choice():
    """Returns the computer's random choice"""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def get_user_choice():
    """Returns the user's choice"""
    while True:
        user_choice = input("Enter your choice (rock, paper, scissors): ").lower()
        if user_choice in ['rock', 'paper', 'scissors']:
            return user_choice
        else:
            print("Invalid choice. Please try again.")

def determine_winner(user_choice, computer_choice):
    """Determines the winner of the round"""
    if user_choice == computer_choice:
        return "It's a tie!"
    elif (user_choice == 'rock' and computer_choice == 'scissors') or \
         (user_choice == 'paper' and computer_choice == 'rock') or \
         (user_choice == 'scissors' and computer_choice == 'paper'):
        return "You win!"
    else:
        return "Computer wins!"

def play_single_round():
    """Plays a single round of Rock, Paper, Scissors"""
    user_choice = get_user_choice()
    computer_choice = get_computer_choice()
    print(f"\nYou chose: {user_choice}")
    print(f"Computer chose: {computer_choice}\n")
    print(determine_winner(user_choice, computer_choice))

def play_multiple_rounds():
    """Plays multiple rounds of Rock, Paper, Scissors"""
    user_score = 0
    computer_score = 0
    while True:
        user_choice = get_user_choice()
        computer_choice = get_computer_choice()
        print(f"\nYou chose: {user_choice}")
        print(f"Computer chose: {computer_choice}\n")
        result = determine_winner(user_choice, computer_choice)
        print(result)
        if result == "You win!":
            user_score += 1
        elif result == "Computer wins!":
            computer_score += 1
        print(f"Score - You: {user_score}, Computer: {computer_score}\n")
        play_again = input("Do you want to play again? (yes/no): ").lower()
        if play_again != "yes":
            break

def main():
    print("Welcome to Rock, Paper, Scissors!")
    while True:
        print("1. Play a single round")
        print("2. Play multiple rounds")
        print("3. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            play_single_round()
        elif choice == "2":
            play_multiple_rounds()
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
