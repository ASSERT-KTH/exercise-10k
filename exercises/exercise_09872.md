## Exercise
Write a program that asks the user a series of questions and keeps track of their score.

## Reference Solution
```python
# Define a dictionary with questions, answers, and scores
questions = {
    "What is the capital of France?": {"A": "Paris", "B": "London", "C": "Berlin", "D": "Rome", "answer": "A", "score": 10},
    "Which planet is the largest in our solar system?": {"A": "Earth", "B": "Saturn", "C": "Jupiter", "D": "Uranus", "answer": "C", "score": 20},
    "Who painted the Mona Lisa?": {"A": "Leonardo da Vinci", "B": "Michelangelo", "C": "Raphael", "D": "Caravaggio", "answer": "A", "score": 30},
    "What is the largest mammal?": {"A": "Blue whale", "B": "African elephant", "C": "Hippopotamus", "D": "Giraffe", "answer": "A", "score": 40},
    "What is the chemical symbol for gold?": {"A": "Ag", "B": "Au", "C": "Hg", "D": "Pb", "answer": "B", "score": 50},
}

def ask_question(question, options):
    """Ask a question and return the user's answer."""
    print(question)
    for option, value in options.items():
        if option != "answer" and option != "score":
            print(f"{option}: {value}")
    answer = input("Enter your answer (A, B, C, D): ")
    return answer.upper()

def check_answer(user_answer, correct_answer):
    """Check if the user's answer is correct and return a message."""
    if user_answer == correct_answer:
        return "Correct!"
    else:
        return "Incorrect."

def main():
    """Main function that asks questions and keeps track of the score."""
    score = 0
    for question, options in questions.items():
        user_answer = ask_question(question, options)
        correct_answer = options["answer"]
        print(check_answer(user_answer, correct_answer))
        if user_answer == correct_answer:
            score += options["score"]
        print(f"Your current score is: {score}\n")
    print(f"Game over! Your final score is: {score}")

if __name__ == "__main__":
    main()
```
