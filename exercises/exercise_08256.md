## Exercise
Write a program that simulates a menu system using if/else statements.

## Reference Solution
```python
def display_menu():
    print("\n--- Menu ---")
    print("1. View Account Balance")
    print("2. Deposit Money")
    print("3. Withdraw Money")
    print("4. Exit")

def view_balance(balance):
    print(f"\nYour current balance is: ${balance}")

def deposit_money(balance):
    amount = float(input("\nEnter the amount to deposit: $"))
    balance += amount
    print(f"\nDeposited ${amount}. Your new balance is: ${balance}")
    return balance

def withdraw_money(balance):
    amount = float(input("\nEnter the amount to withdraw: $"))
    if amount > balance:
        print("\nInsufficient funds.")
    else:
        balance -= amount
        print(f"\nWithdrew ${amount}. Your new balance is: ${balance}")
    return balance

def main():
    balance = 1000.0
    while True:
        display_menu()
        choice = input("\nEnter your choice (1-4): ")
        if choice == "1":
            view_balance(balance)
        elif choice == "2":
            balance = deposit_money(balance)
        elif choice == "3":
            balance = withdraw_money(balance)
        elif choice == "4":
            print("\nExiting the program. Goodbye!")
            break
        else:
            print("\nInvalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
