## Exercise
Develop a program that allows two players to play Rock, Paper, Scissors. The program should determine the winner based on the game's rules.

## Reference Solution
```python
import random

def get_computer_choice():
    """Return the computer's random choice."""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def determine_winner(player1, player2):
    """Determine the winner based on the game's rules."""
    if player1 == player2:
        return "It's a tie!"
    if (player1 == 'rock' and player2 == 'scissors') or \
       (player1 == 'scissors' and player2 == 'paper') or \
       (player1 == 'paper' and player2 == 'rock'):
        return "Player 1 wins!"
    return "Player 2 wins!"

def main():
    print("Welcome to Rock, Paper, Scissors!")
    while True:
        player1 = input("Player 1, enter your choice (rock, paper, scissors): ").lower()
        while player1 not in ['rock', 'paper', 'scissors']:
            player1 = input("Invalid choice. Please enter rock, paper, or scissors: ").lower()
        
        player2 = input("Player 2, enter your choice (rock, paper, scissors): ").lower()
        while player2 not in ['rock', 'paper', 'scissors']:
            player2 = input("Invalid choice. Please enter rock, paper, or scissors: ").lower()
        
        print(f"\nPlayer 1 chose: {player1}")
        print(f"Player 2 chose: {player2}\n")
        
        print(determine_winner(player1, player2))
        
        play_again = input("\nDo you want to play again? (yes/no): ").lower()
        while play_again not in ['yes', 'no']:
            play_again = input("Invalid input. Please enter yes or no: ").lower()
        if play_again != 'yes':
            break

if __name__ == "__main__":
    main()
```
