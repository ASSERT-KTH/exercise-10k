Create a function that takes two functions as arguments and returns a new function that is the composition of the two.

def compose_functions(func1, func2):
    """
    This function takes two functions as arguments and returns a new function 
    that is the composition of the two. The new function will first apply func2 
    to its input, then apply func1 to the result.

    Args:
        func1 (function): The first function to be applied in the composition.
        func2 (function): The second function to be applied in the composition.

    Returns:
        function: A new function that is the composition of func1 and func2.
    """

    # We want to return a new function, so we define a new function inside our compose_functions function.
    def composed_function(*args, **kwargs):
        
        # First, we apply func2 to the input.
        # We use *args and **kwargs to allow for any number of arguments and keyword arguments.
        result_from_func2 = func2(*args, **kwargs)
        
        # Then, we apply func1 to the result of func2.
        # We use *args and **kwargs to allow for any number of arguments and keyword arguments.
        result_from_func1 = func1(result_from_func2)
        
        # Finally, we return the result of the composition.
        return result_from_func1
    
    # We return the new composed function.
    return composed_function

# Example usage:
def add_one(x):
    return x + 1

def multiply_by_two(x):
    return x * 2

composed = compose_functions(add_one, multiply_by_two)
print(composed(5))  # Output: 11
