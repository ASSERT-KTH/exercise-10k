## Exercise
Write a function that takes a dictionary and a list of key-value pairs as input and updates the dictionary with the new key-value pairs

## Reference Solution
```python
def update_dictionary(existing_dict, new_pairs):
    """
    Updates an existing dictionary with new key-value pairs.

    Args:
        existing_dict (dict): The dictionary to be updated.
        new_pairs (list): A list of tuples containing the new key-value pairs.

    Returns:
        dict: The updated dictionary.
    """
    # Check if the input is a dictionary and a list
    if not isinstance(existing_dict, dict) or not isinstance(new_pairs, list):
        raise TypeError("Invalid input type. Expected a dictionary and a list.")

    # Check if all elements in the list are tuples
    if not all(isinstance(pair, tuple) for pair in new_pairs):
        raise TypeError("Invalid input type. Expected a list of tuples.")

    # Check if all tuples have exactly two elements
    if not all(len(pair) == 2 for pair in new_pairs):
        raise TypeError("Invalid input type. Expected tuples with two elements.")

    # Update the dictionary with the new key-value pairs
    for key, value in new_pairs:
        existing_dict[key] = value

    return existing_dict

# Example usage
if __name__ == "__main__":
    existing_dict = {"name": "John", "age": 30}
    new_pairs = [("city", "New York"), ("country", "USA")]

    updated_dict = update_dictionary(existing_dict, new_pairs)
    print(updated_dict)
```
