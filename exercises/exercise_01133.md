## Exercise
Create a program that asks the user a series of questions and keeps track of their score

## Reference Solution
```python
# quiz_program.py

# Dictionary to store questions, answers, and scores
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B",
        "score": 10
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C",
        "score": 10
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "answer": "B",
        "score": 10
    }
}

def quiz(questions):
    """
    Asks the user a series of questions and keeps track of their score.

    Args:
        questions (dict): Dictionary containing questions, answers, and scores.

    Returns:
        int: The user's final score.
    """
    score = 0
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "answer" and option != "score":
                print(f"{option}: {value}")
        answer = input("Enter your answer (A, B, C, D): ")
        if answer.upper() == options["answer"]:
            score += options["score"]
            print("Correct!\n")
        else:
            print(f"Incorrect. The correct answer is {options['answer']}.\n")
    return score

def main():
    print("Welcome to the quiz program!")
    score = quiz(questions)
    print(f"Your final score is {score} out of {sum(question['score'] for question in questions.values())}.")

if __name__ == "__main__":
    main()
```
