## Exercise
Write a function to find the nearest neighbor to a query point in a set of points.

## Reference Solution
```python
import numpy as np

def nearest_neighbor(query_point, points):
    """
    Find the nearest neighbor to a query point in a set of points.

    Args:
    - query_point (numpy array): The point for which to find the nearest neighbor.
    - points (numpy array): The set of points.

    Returns:
    - nearest_point (numpy array): The nearest point to the query point.
    - distance (float): The Euclidean distance between the query point and its nearest neighbor.
    """
    # Calculate the Euclidean distance between the query point and all points
    distances = np.linalg.norm(points - query_point, axis=1)

    # Find the index of the point with the minimum distance
    nearest_index = np.argmin(distances)

    # Get the nearest point and its distance
    nearest_point = points[nearest_index]
    distance = distances[nearest_index]

    return nearest_point, distance

# Example usage
if __name__ == "__main__":
    # Define a set of points
    points = np.array([[1, 2], [3, 4], [5, 6]])

    # Define a query point
    query_point = np.array([2, 3])

    # Find the nearest neighbor
    nearest_point, distance = nearest_neighbor(query_point, points)

    print("Nearest point:", nearest_point)
    print("Distance:", distance)
```
