## Exercise
Given a list of integers and a target sum, find the first subarray with the given sum

## Reference Solution
```python
def find_subarray(nums, target_sum):
    """
    Find the first subarray with the given sum.

    Args:
        nums (list): A list of integers.
        target_sum (int): The target sum.

    Returns:
        list: The first subarray with the given sum. If no such subarray exists, returns None.
    """
    prefix_sum = 0
    sum_map = {0: -1}  # Initialize the sum_map with 0 sum at index -1

    for i, num in enumerate(nums):
        prefix_sum += num

        if prefix_sum - target_sum in sum_map:
            return nums[sum_map[prefix_sum - target_sum] + 1:i + 1]

        if prefix_sum not in sum_map:
            sum_map[prefix_sum] = i

    return None

# Example usage
nums = [1, 4, 20, 3, 10, 5]
target_sum = 33
result = find_subarray(nums, target_sum)

if result:
    print("First subarray with the given sum:", result)
else:
    print("No subarray with the given sum exists.")
```
