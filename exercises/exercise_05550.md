## Exercise
Write a program that counts the number of consonants in a given string using a for loop.

## Reference Solution
```python
def count_consonants(input_string):
    # Convert the string to lowercase
    input_string = input_string.lower()
    
    # Initialize a counter for consonants
    consonant_count = 0
    
    # Iterate over each character in the string
    for char in input_string:
        # Check if the character is a letter
        if char.isalpha():
            # Check if the character is not a vowel
            if char not in 'aeiou':
                # If it's not a vowel, increment the consonant count
                consonant_count += 1
                
    # Return the consonant count
    return consonant_count


def main():
    # Test the function with a string
    input_string = "Hello, World!"
    print(f"The string '{input_string}' contains {count_consonants(input_string)} consonants.")


if __name__ == "__main__":
    main()
```
