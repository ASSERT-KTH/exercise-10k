## Exercise
Write a program that demonstrates the use of complex number operations in programming.

## Reference Solution
```python
import cmath

def add_complex_numbers(num1, num2):
    """Add two complex numbers."""
    return num1 + num2

def subtract_complex_numbers(num1, num2):
    """Subtract two complex numbers."""
    return num1 - num2

def multiply_complex_numbers(num1, num2):
    """Multiply two complex numbers."""
    return num1 * num2

def divide_complex_numbers(num1, num2):
    """Divide two complex numbers."""
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return num1 / num2

def calculate_magnitude(num):
    """Calculate the magnitude of a complex number."""
    return abs(num)

def calculate_conjugate(num):
    """Calculate the conjugate of a complex number."""
    return num.conjugate()

def main():
    # Define complex numbers
    num1 = complex(3, 4)
    num2 = complex(2, 1)

    print("Complex Number 1:", num1)
    print("Complex Number 2:", num2)

    # Perform operations
    print("\nAddition:", add_complex_numbers(num1, num2))
    print("Subtraction:", subtract_complex_numbers(num1, num2))
    print("Multiplication:", multiply_complex_numbers(num1, num2))
    print("Division:", divide_complex_numbers(num1, num2))

    # Calculate magnitude and conjugate
    print("\nMagnitude of Complex Number 1:", calculate_magnitude(num1))
    print("Conjugate of Complex Number 1:", calculate_conjugate(num1))

if __name__ == "__main__":
    main()
```
