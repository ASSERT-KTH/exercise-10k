## Exercise
Write a program that stores and retrieves music using conditional statements, allowing users to search for songs based on artist, album, and genre.

## Reference Solution
```python
class MusicLibrary:
    def __init__(self):
        self.music_library = {
            "song1": {"artist": "Artist1", "album": "Album1", "genre": "Rock"},
            "song2": {"artist": "Artist2", "album": "Album2", "genre": "Pop"},
            "song3": {"artist": "Artist3", "album": "Album3", "genre": "Jazz"},
        }

    def add_song(self, song, artist, album, genre):
        self.music_library[song] = {"artist": artist, "album": album, "genre": genre}

    def search_by_artist(self, artist):
        return [song for song, info in self.music_library.items() if info["artist"] == artist]

    def search_by_album(self, album):
        return [song for song, info in self.music_library.items() if info["album"] == album]

    def search_by_genre(self, genre):
        return [song for song, info in self.music_library.items() if info["genre"] == genre]

    def retrieve_song(self, song):
        return self.music_library.get(song, "Song not found in library")

    def display_library(self):
        for song, info in self.music_library.items():
            print(f"Song: {song}, Artist: {info['artist']}, Album: {info['album']}, Genre: {info['genre']}")


def main():
    library = MusicLibrary()

    while True:
        print("\nMusic Library Menu:")
        print("1. Add song to library")
        print("2. Search for songs by artist")
        print("3. Search for songs by album")
        print("4. Search for songs by genre")
        print("5. Retrieve a song")
        print("6. Display library")
        print("7. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            song = input("Enter song name: ")
            artist = input("Enter artist name: ")
            album = input("Enter album name: ")
            genre = input("Enter genre: ")
            library.add_song(song, artist, album, genre)
        elif choice == "2":
            artist = input("Enter artist name: ")
            songs = library.search_by_artist(artist)
            if songs:
                print(f"Songs by {artist}: {songs}")
            else:
                print("No songs found by that artist")
        elif choice == "3":
            album = input("Enter album name: ")
            songs = library.search_by_album(album)
            if songs:
                print(f"Songs from {album}: {songs}")
            else:
                print("No songs found from that album")
        elif choice == "4":
            genre = input("Enter genre: ")
            songs = library.search_by_genre(genre)
            if songs:
                print(f"Songs of {genre} genre: {songs}")
            else:
                print("No songs found of that genre")
        elif choice == "5":
            song = input("Enter song name: ")
            info = library.retrieve_song(song)
            if info != "Song not found in library":
                print(f"Song: {song}, Artist: {info['artist']}, Album: {info['album']}, Genre: {info['genre']}")
            else:
                print(info)
        elif choice == "6":
            library.display_library()
        elif choice == "7":
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
