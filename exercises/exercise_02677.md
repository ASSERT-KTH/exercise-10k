## Exercise
Write a program that takes a string as input and returns the string in sentence case.

## Reference Solution
```python
def convert_to_sentence_case(input_string: str) -> str:
    """
    Converts a string to sentence case.

    Args:
        input_string (str): The input string to be converted.

    Returns:
        str: The input string in sentence case.
    """
    # Strip leading and trailing whitespace
    input_string = input_string.strip()
    
    # Check if the string is empty
    if not input_string:
        return input_string
    
    # Capitalize the first character and make the rest lowercase
    sentence_case_string = input_string[0].upper() + input_string[1:].lower()
    
    return sentence_case_string

def main():
    # Example usage
    input_string = "tHis IS a TeSt StRiNg"
    print("Original String:", input_string)
    print("Sentence Case String:", convert_to_sentence_case(input_string))

if __name__ == "__main__":
    main()
```
