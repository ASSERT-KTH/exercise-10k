## Exercise
Write a program that calculates the fuel efficiency of a vehicle given its distance traveled and fuel consumption.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumption, unit='kmpl'):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
        distance (float): Distance traveled by the vehicle in kilometers.
        fuel_consumption (float): Fuel consumed by the vehicle in liters.
        unit (str, optional): Unit of fuel efficiency. Defaults to 'kmpl'.

    Returns:
        float: Fuel efficiency of the vehicle in the specified unit.
    """
    if unit == 'kmpl':
        return distance / fuel_consumption
    elif unit == 'mpg':
        # 1 km = 0.621371 miles, 1 liter = 0.264172 gallons
        return (distance * 0.621371) / (fuel_consumption * 0.264172)
    else:
        raise ValueError("Invalid unit. Supported units are 'kmpl' and 'mpg'.")

def main():
    distance = float(input("Enter the distance traveled in kilometers: "))
    fuel_consumption = float(input("Enter the fuel consumption in liters: "))
    unit = input("Enter the unit of fuel efficiency (kmpl/mpg): ")

    fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumption, unit)

    print(f"The fuel efficiency of the vehicle is {fuel_efficiency:.2f} {unit}.")

if __name__ == "__main__":
    main()
```
