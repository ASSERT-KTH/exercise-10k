## Exercise
Write a program that takes an income as input and prints out the calculated tax

## Reference Solution
```python
# Define tax brackets and rates
tax_brackets = [
    (0, 9875, 0.10),  # 10% on the first $9,875
    (9876, 40125, 0.12),  # 12% on income between $9,876 and $40,125
    (40126, 85525, 0.22),  # 22% on income between $40,126 and $85,525
    (85526, 163300, 0.24),  # 24% on income between $85,526 and $163,300
    (163301, 207350, 0.32),  # 32% on income between $163,301 and $207,350
    (207351, 518400, 0.35),  # 35% on income between $207,351 and $518,400
    (518401, float('inf'), 0.37)  # 37% on income above $518,400
]

def calculate_tax(income):
    """
    Calculate tax based on the given income and tax brackets.
    
    Args:
    income (float): The income to calculate tax for.
    
    Returns:
    float: The calculated tax.
    """
    tax = 0.0
    remaining_income = income
    
    for bracket in tax_brackets:
        lower_bound, upper_bound, rate = bracket
        
        # Calculate the income within the current bracket
        income_in_bracket = min(upper_bound, remaining_income) - lower_bound
        
        # If the income is within the current bracket, calculate the tax and break
        if income_in_bracket > 0:
            tax += income_in_bracket * rate
            remaining_income -= income_in_bracket
            
            # If all income has been processed, break the loop
            if remaining_income <= 0:
                break
    
    return tax

def main():
    income = float(input("Enter your income: $"))
    tax = calculate_tax(income)
    print(f"Your tax is: ${tax:.2f}")

if __name__ == "__main__":
    main()
```
