## Exercise
Create a function that allows the user to track their expenses. The function takes in a category and an amount, and stores the expense in a list.

## Reference Solution
```python
class ExpenseTracker:
    def __init__(self):
        self.expenses = []

    def add_expense(self, category, amount):
        """
        Add an expense to the tracker.

        Args:
            category (str): Category of the expense (e.g., 'food', 'transportation', etc.)
            amount (float): Amount of the expense

        Returns:
            None
        """
        self.expenses.append({"category": category, "amount": amount})

    def get_expenses(self):
        """
        Get all expenses in the tracker.

        Returns:
            list: List of dictionaries, each containing the category and amount of an expense
        """
        return self.expenses

    def get_expenses_by_category(self, category):
        """
        Get all expenses in a specific category.

        Args:
            category (str): Category of the expenses to retrieve

        Returns:
            list: List of dictionaries, each containing the category and amount of an expense in the specified category
        """
        return [expense for expense in self.expenses if expense["category"] == category]

    def calculate_total_expenses(self):
        """
        Calculate the total amount of all expenses.

        Returns:
            float: Total amount of all expenses
        """
        return sum(expense["amount"] for expense in self.expenses)

    def calculate_total_expenses_by_category(self, category):
        """
        Calculate the total amount of expenses in a specific category.

        Args:
            category (str): Category of the expenses to calculate

        Returns:
            float: Total amount of expenses in the specified category
        """
        return sum(expense["amount"] for expense in self.expenses if expense["category"] == category)


def main():
    tracker = ExpenseTracker()

    while True:
        print("\nExpense Tracker Menu:")
        print("1. Add Expense")
        print("2. Get All Expenses")
        print("3. Get Expenses by Category")
        print("4. Calculate Total Expenses")
        print("5. Calculate Total Expenses by Category")
        print("6. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            category = input("Enter category: ")
            amount = float(input("Enter amount: "))
            tracker.add_expense(category, amount)
        elif choice == "2":
            print(tracker.get_expenses())
        elif choice == "3":
            category = input("Enter category: ")
            print(tracker.get_expenses_by_category(category))
        elif choice == "4":
            print(tracker.calculate_total_expenses())
        elif choice == "5":
            category = input("Enter category: ")
            print(tracker.calculate_total_expenses_by_category(category))
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
