def concatenate_lists(list1, list2):
    """
    Concatenates two lists of strings.

    Args:
        list1 (list): The first list of strings.
        list2 (list): The second list of strings.

    Returns:
        list: A new list containing all the strings from both input lists.
    """
    return list1 + list2
