## Exercise
Create a function that generates another function at runtime

## Reference Solution
```python
def generate_function(operator):
    """
    Generates a function that performs a specific mathematical operation.
    
    Args:
    operator (str): The mathematical operation to be performed. 
                    Supported operators are '+', '-', '*', '/'.
                    
    Returns:
    function: A function that takes two numbers as input and returns the result of the operation.
    """
    
    # Define a dictionary to map operators to their corresponding functions
    operators = {
        '+': lambda x, y: x + y,
        '-': lambda x, y: x - y,
        '*': lambda x, y: x * y,
        '/': lambda x, y: x / y if y != 0 else float('inf')
    }
    
    # Check if the operator is supported
    if operator not in operators:
        raise ValueError("Unsupported operator. Supported operators are '+', '-', '*', '/'")
    
    # Generate the function
    def dynamic_function(x, y):
        return operators[operator](x, y)
    
    return dynamic_function

# Example usage:
if __name__ == "__main__":
    # Generate an addition function
    add_function = generate_function('+')
    print(add_function(5, 3))  # Output: 8
    
    # Generate a subtraction function
    subtract_function = generate_function('-')
    print(subtract_function(5, 3))  # Output: 2
    
    # Generate a multiplication function
    multiply_function = generate_function('*')
    print(multiply_function(5, 3))  # Output: 15
    
    # Generate a division function
    divide_function = generate_function('/')
    print(divide_function(5, 3))  # Output: 1.6666666666666667
```
