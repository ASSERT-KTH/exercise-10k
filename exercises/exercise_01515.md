## Exercise
Write a program that swaps two specified characters in a given string.

## Reference Solution
```python
def swap_chars(string, char1, char2):
    """
    Swaps two specified characters in a given string.

    Args:
        string (str): The input string.
        char1 (str): The first character to swap.
        char2 (str): The second character to swap.

    Returns:
        str: The string with the specified characters swapped.

    Raises:
        ValueError: If either char1 or char2 is not found in the string.
    """

    # Check if the input string is empty
    if not string:
        raise ValueError("Input string is empty")

    # Check if char1 and char2 are single characters
    if len(char1) != 1 or len(char2) != 1:
        raise ValueError("Both char1 and char2 must be single characters")

    # Check if char1 and char2 are in the string
    if char1 not in string or char2 not in string:
        raise ValueError("Both char1 and char2 must be in the string")

    # Convert the string to a list of characters
    char_list = list(string)

    # Find the indices of char1 and char2 in the string
    index1 = [i for i, char in enumerate(char_list) if char == char1]
    index2 = [i for i, char in enumerate(char_list) if char == char2]

    # Swap the characters at the found indices
    for i, j in zip(index1, index2):
        char_list[i], char_list[j] = char_list[j], char_list[i]

    # Convert the list of characters back to a string
    result = "".join(char_list)

    return result

# Example usage
string = "hello world"
char1 = "h"
char2 = "w"
print(swap_chars(string, char1, char2))  # Output: "wello horld"
```
