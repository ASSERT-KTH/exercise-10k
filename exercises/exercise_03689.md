## Exercise
Write a program to calculate the time difference between two given times in hours, minutes, and seconds.

## Reference Solution
```python
from datetime import datetime

def calculate_time_difference(time1, time2):
    # Convert the time strings to datetime objects
    time1 = datetime.strptime(time1, "%H:%M:%S")
    time2 = datetime.strptime(time2, "%H:%M:%S")

    # Calculate the time difference
    time_difference = time2 - time1

    # Calculate the hours, minutes, and seconds
    hours = time_difference.seconds // 3600
    minutes = (time_difference.seconds // 60) % 60
    seconds = time_difference.seconds % 60

    return hours, minutes, seconds


def main():
    time1 = input("Enter the first time (HH:MM:SS): ")
    time2 = input("Enter the second time (HH:MM:SS): ")

    hours, minutes, seconds = calculate_time_difference(time1, time2)

    print(f"The time difference is {hours} hours, {minutes} minutes, and {seconds} seconds.")


if __name__ == "__main__":
    main()
```
