## Exercise
Write a program that formats a given number as a currency value, with two decimal places and a dollar sign.

## Reference Solution
```python
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

string currencyFormat(int n) {
    string dollars = to_string(n / 100);
    string cents = to_string(n % 100);

    if (cents.length() == 1) {
        cents.insert(0, "0");
    }

    string formatted = "$" + dollars + "." + cents;
    return formatted;
}

int main() {
    int n;
    cin >> n;
    cout << currencyFormat(n) << endl;
    return 0;
}
```
