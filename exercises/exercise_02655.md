## Exercise
Write a function to create a copy of a dictionary using the copy() method

## Reference Solution
```python
def create_dict_copy(original_dict):
    """
    Creates a copy of the original dictionary.

    Args:
        original_dict (dict): The dictionary to be copied.

    Returns:
        dict: A copy of the original dictionary.
    """
    # Create a copy of the original dictionary
    dict_copy = original_dict.copy()

    return dict_copy

# Example usage:
original_dict = {"name": "John", "age": 30, "city": "New York"}
dict_copy = create_dict_copy(original_dict)

print("Original Dictionary:")
print(original_dict)

print("\nCopied Dictionary:")
print(dict_copy)

# Modifying the copied dictionary to demonstrate it's a separate copy
dict_copy["age"] = 31

print("\nAfter Modifying the Copied Dictionary:")
print("Original Dictionary:")
print(original_dict)

print("\nCopied Dictionary:")
print(dict_copy)
```
