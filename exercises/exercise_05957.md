## Exercise
Create a program that generates the Mandelbrot Set. The program should ask for the size, the maximum iterations, and then print out the generated set.

## Reference Solution
```python
import numpy as np

def mandelbrot(c, max_iter):
    """
    Compute the number of iterations until the point c escapes the Mandelbrot Set.
    
    Args:
        c (complex): The point to check.
        max_iter (int): The maximum number of iterations.
    
    Returns:
        int: The number of iterations until the point escapes.
    """
    z = c
    for n in range(max_iter):
        if abs(z) > 2:
            return n
        z = z*z + c
    return max_iter

def generate_mandelbrot_set(size, max_iter):
    """
    Generate the Mandelbrot Set and print it out as a 2D representation.
    
    Args:
        size (int): The size of the output.
        max_iter (int): The maximum number of iterations.
    """
    # Define the boundaries of the Mandelbrot Set
    x_min, x_max, y_min, y_max = -2.0, 1.0, -1.5, 1.5
    
    # Create a 2D array to store the number of iterations
    mandelbrot_set = np.zeros((size, size), dtype=int)
    
    # Iterate over each point in the 2D array
    for i in range(size):
        for j in range(size):
            # Map the point to the Mandelbrot Set boundaries
            x = x_min + (x_max - x_min) * i / (size - 1)
            y = y_min + (y_max - y_min) * j / (size - 1)
            c = complex(x, y)
            
            # Compute the number of iterations until the point escapes
            mandelbrot_set[i, j] = mandelbrot(c, max_iter)
    
    # Print out the Mandelbrot Set
    for i in range(size):
        for j in range(size):
            if mandelbrot_set[i, j] == max_iter:
                print('*', end=' ')
            else:
                print(' ', end=' ')
        print()

def main():
    # Ask for user input
    size = int(input("Enter the size of the output: "))
    max_iter = int(input("Enter the maximum number of iterations: "))
    
    # Generate and print the Mandelbrot Set
    generate_mandelbrot_set(size, max_iter)

if __name__ == "__main__":
    main()
```
