## Exercise
Create a program that takes in three numbers and performs addition, subtraction, multiplication, and division based on the user's choice. The program should also handle invalid inputs and division by zero errors.

## Reference Solution
```python
def add(num1, num2):
    """Return the sum of two numbers."""
    return num1 + num2

def subtract(num1, num2):
    """Return the difference of two numbers."""
    return num1 - num2

def multiply(num1, num2, num3):
    """Return the product of three numbers."""
    return num1 * num2 * num3

def divide(num1, num2, num3):
    """Return the result of dividing the product of two numbers by the third number."""
    if num3 == 0:
        raise ZeroDivisionError("Cannot divide by zero.")
    return (num1 * num2) / num3

def main():
    print("Simple Calculator Program")
    print("1. Addition")
    print("2. Subtraction")
    print("3. Multiplication")
    print("4. Division")

    while True:
        choice = input("Enter your choice (1, 2, 3, 4): ")
        
        if choice in ['1', '2', '3', '4']:
            try:
                num1 = float(input("Enter first number: "))
                num2 = float(input("Enter second number: "))
                num3 = float(input("Enter third number: "))

                if choice == '1':
                    print(f"{num1} + {num2} + {num3} = {add(num1, num2) + num3}")
                elif choice == '2':
                    print(f"{num1} - {num2} - {num3} = {subtract(num1, num2) - num3}")
                elif choice == '3':
                    print(f"{num1} * {num2} * {num3} = {multiply(num1, num2, num3)}")
                elif choice == '4':
                    print(f"({num1} * {num2}) / {num3} = {divide(num1, num2, num3)}")
            except ValueError:
                print("Invalid input. Please enter a number.")
            except ZeroDivisionError as e:
                print(e)
        else:
            print("Invalid choice. Please enter a number between 1 and 4.")
        
        cont = input("Do you want to continue? (yes/no): ")
        if cont.lower() != 'yes':
            break

if __name__ == "__main__":
    main()
```
