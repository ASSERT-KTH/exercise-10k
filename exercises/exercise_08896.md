## Exercise
Write a program that takes the lengths of the sides of a triangle as input and returns the length of its inradius.

## Reference Solution
```python
import math

def calculate_inradius(a, b, c):
    """
    Calculate the inradius of a triangle given the lengths of its sides.

    Parameters:
    a (float): The length of the first side of the triangle.
    b (float): The length of the second side of the triangle.
    c (float): The length of the third side of the triangle.

    Returns:
    float: The length of the inradius of the triangle.
    """
    # Calculate the semi-perimeter of the triangle
    s = (a + b + c) / 2

    # Calculate the area of the triangle using Heron's formula
    area = math.sqrt(s * (s - a) * (s - b) * (s - c))

    # Calculate the inradius of the triangle
    inradius = area / s

    return inradius

# Example usage:
a = float(input("Enter the length of the first side: "))
b = float(input("Enter the length of the second side: "))
c = float(input("Enter the length of the third side: "))

# Check if the input values can form a triangle
if a + b > c and a + c > b and b + c > a:
    inradius = calculate_inradius(a, b, c)
    print("The length of the inradius is: ", inradius)
else:
    print("The input values cannot form a triangle.")
```
