## Exercise
Write a program that declares and operates on dictionaries.

## Reference Solution
```python
# dictionary_operations.py

def main():
    # Declare a dictionary
    person = {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }

    # Print the dictionary
    print("Person Dictionary:")
    print(person)

    # Access a value by key
    print("\nAccessing values by key:")
    print("Name:", person["name"])
    print("Age:", person["age"])
    print("City:", person["city"])

    # Update a value
    person["age"] = 31
    print("\nUpdated age:")
    print("Age:", person["age"])

    # Add a new key-value pair
    person["country"] = "USA"
    print("\nAdded country:")
    print("Country:", person["country"])

    # Remove a key-value pair
    del person["city"]
    print("\nRemoved city:")
    print("Person Dictionary:")
    print(person)

    # Check if a key exists
    print("\nChecking if key exists:")
    print("Has 'name' key:", "name" in person)
    print("Has 'state' key:", "state" in person)

    # Get all keys
    print("\nGetting all keys:")
    print("Keys:", list(person.keys()))

    # Get all values
    print("\nGetting all values:")
    print("Values:", list(person.values()))

    # Get all key-value pairs
    print("\nGetting all key-value pairs:")
    print("Items:", list(person.items()))

if __name__ == "__main__":
    main()
```
