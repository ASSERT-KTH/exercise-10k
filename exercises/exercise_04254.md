## Exercise
Create a program that manages a customer loyalty program. The program should take the customer ID, purchase amount, and loyalty type as input, and print the updated loyalty points.

## Reference Solution
```python
class Customer:
    def __init__(self, customer_id, loyalty_type):
        self.customer_id = customer_id
        self.loyalty_type = loyalty_type
        self.loyalty_points = 0

    def update_loyalty_points(self, purchase_amount):
        if self.loyalty_type == "bronze":
            self.loyalty_points += purchase_amount * 0.01
        elif self.loyalty_type == "silver":
            self.loyalty_points += purchase_amount * 0.02
        elif self.loyalty_type == "gold":
            self.loyalty_points += purchase_amount * 0.03
        else:
            print("Invalid loyalty type")

    def get_loyalty_points(self):
        return self.loyalty_points


class LoyaltyProgram:
    def __init__(self):
        self.customers = {}

    def add_customer(self, customer_id, loyalty_type):
        self.customers[customer_id] = Customer(customer_id, loyalty_type)

    def update_customer_loyalty_points(self, customer_id, purchase_amount):
        if customer_id in self.customers:
            self.customers[customer_id].update_loyalty_points(purchase_amount)
        else:
            print("Customer not found")

    def print_customer_loyalty_points(self, customer_id):
        if customer_id in self.customers:
            print(f"Customer ID: {customer_id}, Loyalty Points: {self.customers[customer_id].get_loyalty_points()}")
        else:
            print("Customer not found")


def main():
    loyalty_program = LoyaltyProgram()

    while True:
        print("\n1. Add Customer")
        print("2. Update Customer Loyalty Points")
        print("3. Print Customer Loyalty Points")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            customer_id = input("Enter customer ID: ")
            loyalty_type = input("Enter loyalty type (bronze, silver, gold): ")
            loyalty_program.add_customer(customer_id, loyalty_type)
        elif choice == "2":
            customer_id = input("Enter customer ID: ")
            purchase_amount = float(input("Enter purchase amount: "))
            loyalty_program.update_customer_loyalty_points(customer_id, purchase_amount)
        elif choice == "3":
            customer_id = input("Enter customer ID: ")
            loyalty_program.print_customer_loyalty_points(customer_id)
        elif choice == "4":
            break
        else:
            print("Invalid choice")


if __name__ == "__main__":
    main()
```
