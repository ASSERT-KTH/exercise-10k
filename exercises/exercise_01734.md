## Exercise
Write a program that takes a string as input and returns the longest substring without repeating characters.

## Reference Solution

def longest_substring_without_repeating_chars(s: str) -> str:
    """
    Returns the longest substring without repeating characters.

    Args:
    s (str): The input string.

    Returns:
    str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    # Initialize variables to keep track of the longest substring
    max_length = 0
    max_substring = ""

    # Initialize variables for the sliding window
    left = 0
    char_set = set()

    # Iterate over the string
    for right in range(len(s)):
        # While the character at the right pointer is in the set, remove the character at the left pointer from the set and move the left pointer to the right
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1

        # Add the character at the right pointer to the set
        char_set.add(s[right])

        # Update the maximum length and substring if the current substring is longer
        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring


# Example usage
if __name__ == "__main__":
    input_string = "abcabcbb"
    result = longest_substring_without_repeating_chars(input_string)
    print(f"Longest substring without repeating characters: {result}")
