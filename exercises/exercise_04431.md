## Exercise
Write a program that prints out a calendar for a given month and year using a for loop.

## Reference Solution
```python
import calendar

def print_calendar(year, month):
    """
    Prints a calendar for the given month and year.

    Args:
        year (int): The year of the calendar.
        month (int): The month of the calendar (1-12).
    """
    # Get the month name
    month_name = calendar.month_name[month]

    # Print the month and year
    print(f"{month_name} {year}")
    print("Mo Tu We Th Fr Sa Su")

    # Get the month calendar
    month_calendar = calendar.monthcalendar(year, month)

    # Print the calendar
    for week in month_calendar:
        for day in week:
            if day == 0:
                print("  ", end=" ")
            else:
                print(f"{day:2}", end=" ")
        print()

def main():
    year = int(input("Enter the year: "))
    month = int(input("Enter the month (1-12): "))

    print_calendar(year, month)

if __name__ == "__main__":
    main()
```
