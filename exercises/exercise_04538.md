## Exercise
Write a program that takes an array of pairs as input and returns two separate arrays: one containing the first elements of the pairs and one containing the second elements of the pairs.

## Reference Solution
```python
#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    vector<vector<int>> separateElements(vector<pair<int, int>>& nums) {
        vector<int> a;
        vector<int> b;

        for (auto& x : nums) {
            a.push_back(x.first);
            b.push_back(x.second);
        }

        vector<vector<int>> result;
        result.push_back(a);
        result.push_back(b);

        return result;
    }
};

int main() {
    Solution solution;
    vector<pair<int, int>> nums = {{1, 2}, {3, 4}, {5, 6}};
    vector<vector<int>> result = solution.separateElements(nums);

    for (auto& row : result) {
        for (auto& x : row) {
            cout << x << " ";
        }
        cout << endl;
    }

    return 0;
}
```
