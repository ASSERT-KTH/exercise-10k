Create a function that takes two dictionaries and returns a new dictionary containing all key-value pairs from both dictionaries.

def merge_dictionaries(dict1, dict2):
    """
    This function takes two dictionaries as input and returns a new dictionary containing all key-value pairs from both dictionaries.

    Args:
        dict1 (dict): The first dictionary to be merged.
        dict2 (dict): The second dictionary to be merged.

    Returns:
        dict: A new dictionary containing all key-value pairs from both input dictionaries.
    """

    # First, we'll create a new empty dictionary that will store the merged key-value pairs.
    merged_dict = {}

    # Now, we'll iterate over the key-value pairs in the first dictionary.
    for key, value in dict1.items():
        # For each key-value pair, we'll add it to our new merged dictionary.
        merged_dict[key] = value

    # Next, we'll do the same for the second dictionary.
    for key, value in dict2.items():
        # If a key already exists in the merged dictionary, its value will be updated with the value from the second dictionary.
        merged_dict[key] = value

    # Finally, we'll return our merged dictionary.
    return merged_dict
