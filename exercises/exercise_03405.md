## Exercise
Create a program to delete a specific key-value pair from a given dictionary.

## Reference Solution
```python
def delete_key_value_pair(dictionary, key):
    """
    Deletes a specific key-value pair from a given dictionary.

    Args:
        dictionary (dict): The dictionary from which to delete the key-value pair.
        key (any): The key of the key-value pair to delete.

    Returns:
        dict: The dictionary with the key-value pair deleted.
    """
    if key in dictionary:
        del dictionary[key]
    else:
        print(f"Key '{key}' not found in dictionary.")
    return dictionary

def main():
    # Example usage
    example_dict = {"name": "John", "age": 30, "city": "New York"}
    print("Original Dictionary:")
    print(example_dict)

    key_to_delete = "age"
    updated_dict = delete_key_value_pair(example_dict, key_to_delete)
    print(f"\nDictionary after deleting key '{key_to_delete}':")
    print(updated_dict)

if __name__ == "__main__":
    main()
```
