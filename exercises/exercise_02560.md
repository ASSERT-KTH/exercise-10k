## Exercise
Create a function to partially apply another function and use it to calculate the sum.

## Reference Solution
```python
from functools import partial

def add(a, b):
    """Add two numbers."""
    return a + b

def partial_apply(func, *args):
    """
    Partially apply a function with given arguments.

    Args:
        func (function): The function to partially apply.
        *args: The arguments to apply to the function.

    Returns:
        function: A new function with the applied arguments.
    """
    return partial(func, *args)

def calculate_sum(numbers, func):
    """
    Calculate the sum of a list of numbers using a given function.

    Args:
        numbers (list): A list of numbers.
        func (function): A function that takes two arguments and returns their sum.

    Returns:
        int: The sum of the list of numbers.
    """
    result = numbers[0]
    for num in numbers[1:]:
        result = func(result, num)
    return result

# Example usage
if __name__ == "__main__":
    numbers = [1, 2, 3, 4, 5]
    partial_add = partial_apply(add, 0)  # Partially apply the add function with 0 as the first argument
    sum_result = calculate_sum(numbers, add)  # Calculate the sum using the add function
    partial_sum_result = calculate_sum(numbers, partial_apply(add, 0))  # Calculate the sum using the partially applied add function

    print("Sum using add function:", sum_result)
    print("Sum using partially applied add function:", partial_sum_result)
```
