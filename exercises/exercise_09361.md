# Define a function to calculate the area of a rectangle
def calculate_rectangle_area(length_in_meters: float, width_in_meters: float) -> float:
    """
    Calculate the area of a rectangle.

    Args:
        length_in_meters (float): The length of the rectangle in meters.
        width_in_meters (float): The width of the rectangle in meters.

    Returns:
        float: The area of the rectangle in square meters.
    """
    area_in_square_meters = length_in_meters * width_in_meters
    return area_in_square_meters

# Define a function to calculate the volume of a cube
def calculate_cube_volume(side_length_in_meters: float) -> float:
    """
    Calculate the volume of a cube.

    Args:
        side_length_in_meters (float): The side length of the cube in meters.

    Returns:
        float: The volume of the cube in cubic meters.
    """
    volume_in_cubic_meters = side_length_in_meters ** 3
    return volume_in_cubic_meters

# Main function
def main():
    # Example usage
    rectangle_length = 5.0  # in meters
    rectangle_width = 3.0  # in meters
    cube_side_length = 4.0  # in meters

    rectangle_area = calculate_rectangle_area(rectangle_length, rectangle_width)
    cube_volume = calculate_cube_volume(cube_side_length)

    print(f"Rectangle area: {rectangle_area} square meters")
    print(f"Cube volume: {cube_volume} cubic meters")

if __name__ == "__main__":
    main()
