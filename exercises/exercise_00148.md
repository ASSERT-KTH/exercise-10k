## Exercise
Create a function that takes in a dictionary and a reducing function, and returns a single value resulting from applying the reducing function to all values in the dictionary

## Reference Solution
```python
from functools import reduce

def reduce_dict_values(dictionary, reducing_function, initial_value=None):
    """
    Applies a reducing function to all values in a dictionary.

    Args:
        dictionary (dict): The dictionary to reduce.
        reducing_function (function): The function to apply to the dictionary values.
        initial_value (any, optional): The initial value for the reducing function. Defaults to None.

    Returns:
        any: The result of applying the reducing function to all dictionary values.
    """

    # Get the dictionary values
    values = dictionary.values()

    # If an initial value is provided, use it as the starting point for the reduction
    if initial_value is not None:
        # Use the reduce function with the initial value
        return reduce(reducing_function, values, initial_value)
    else:
        # If no initial value is provided, use the first value in the dictionary as the starting point
        return reduce(reducing_function, values)

# Example usage
if __name__ == "__main__":
    # Define a dictionary
    my_dict = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

    # Define a reducing function (e.g., sum)
    def sum_function(x, y):
        return x + y

    # Apply the reducing function to the dictionary values
    result = reduce_dict_values(my_dict, sum_function)

    print("Result:", result)  # Output: 10

    # Apply the reducing function with an initial value
    result_with_initial = reduce_dict_values(my_dict, sum_function, 10)

    print("Result with initial value:", result_with_initial)  # Output: 20
```
