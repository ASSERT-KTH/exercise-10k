## Exercise
Create a program that asks the user a series of questions and keeps track of their score.

## Reference Solution
```python
# Import the required libraries
import random

# Define the questions and answers
questions = {
    "What is the capital of France?": ["Paris", "London", "Berlin", "Rome"],
    "What is the largest planet in our solar system?": ["Earth", "Saturn", "Jupiter", "Uranus"],
    "What is the chemical symbol for gold?": ["Ag", "Au", "Hg", "Pb"],
    "What is the largest mammal?": ["Blue whale", "African elephant", "Hippopotamus", "Giraffe"],
    "What is the smallest country in the world?": ["Vatican City", "Monaco", "Nauru", "Tuvalu"]
}

answers = {
    "What is the capital of France?": "Paris",
    "What is the largest planet in our solar system?": "Jupiter",
    "What is the chemical symbol for gold?": "Au",
    "What is the largest mammal?": "Blue whale",
    "What is the smallest country in the world?": "Vatican City"
}

def quiz():
    # Initialize the score
    score = 0

    # Ask each question
    for question, options in questions.items():
        print(f"\nQuestion: {question}")
        for i, option in enumerate(options):
            print(f"{i+1}. {option}")

        # Get the user's answer
        user_answer = input("Enter the number of your answer: ")

        # Check if the user's answer is correct
        if options[int(user_answer) - 1] == answers[question]:
            print("Correct!")
            score += 1
        else:
            print(f"Incorrect. The correct answer is {answers[question]}.")

    # Print the final score
    print(f"\nQuiz complete! Your final score is {score}/{len(questions)}.")

# Run the quiz
if __name__ == "__main__":
    quiz()
```
