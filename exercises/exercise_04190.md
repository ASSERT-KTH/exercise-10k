## Exercise
Write a function that simulates a simple chatbot, responding to basic user queries.

## Reference Solution
```python
import random

def chatbot(query):
    """
    Simulates a simple chatbot that responds to basic user queries.

    Args:
        query (str): The user's query.

    Returns:
        str: The chatbot's response.
    """

    # Define a dictionary of possible responses to different queries
    responses = {
        'greeting': [
            'Hello! How can I assist you today?',
            'Hi! What brings you here?',
            'Hey! How are you doing?'
        ],
        'farewell': [
            'Goodbye! It was nice chatting with you.',
            'See you later! Have a great day.',
            'Bye! Feel free to come back anytime.'
        ],
        'help': [
            'I can help you with basic queries. What do you need help with?',
            'How can I assist you? Please provide more details.',
            'What do you need help with? I\'m here to assist you.'
        ],
        'default': [
            'I didn\'t quite understand that. Could you please rephrase?',
            'Sorry, I\'m not sure what you mean. Could you provide more context?',
            'I\'m not sure I understand your query. Please try again.'
        ]
    }

    # Convert the query to lowercase for case-insensitive comparison
    query = query.lower()

    # Check if the query is a greeting
    if any(word in query for word in ['hello', 'hi', 'hey']):
        return random.choice(responses['greeting'])

    # Check if the query is a farewell
    elif any(word in query for word in ['bye', 'goodbye', 'see you later']):
        return random.choice(responses['farewell'])

    # Check if the query is a help request
    elif any(word in query for word in ['help', 'assist']):
        return random.choice(responses['help'])

    # If none of the above conditions are met, return a default response
    else:
        return random.choice(responses['default'])

# Example usage:
def main():
    print("Welcome to the simple chatbot!")
    while True:
        query = input("You: ")
        print("Chatbot: ", chatbot(query))
        if chatbot(query).startswith('Goodbye') or chatbot(query).startswith('See you later'):
            break

if __name__ == "__main__":
    main()
```
