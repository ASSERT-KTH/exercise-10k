## Exercise
Filter the students in a dictionary based on their age

## Reference Solution
```python
# Define a dictionary with student information
students = {
    "John": {"age": 20, "grade": 3.5},
    "Alice": {"age": 22, "grade": 3.8},
    "Bob": {"age": 19, "grade": 3.2},
    "Emma": {"age": 21, "grade": 3.6},
    "Michael": {"age": 23, "grade": 3.9},
}

# Function to filter students by age
def filter_students_by_age(students, min_age=None, max_age=None):
    """
    Filter students in a dictionary based on their age.

    Args:
    - students (dict): Dictionary with student information.
    - min_age (int, optional): Minimum age to filter by. Defaults to None.
    - max_age (int, optional): Maximum age to filter by. Defaults to None.

    Returns:
    - dict: Dictionary with filtered student information.
    """
    filtered_students = {}
    for name, info in students.items():
        if (min_age is None or info["age"] >= min_age) and (max_age is None or info["age"] <= max_age):
            filtered_students[name] = info
    return filtered_students

# Example usage
print("All Students:")
for name, info in students.items():
    print(f"{name}: {info}")

# Filter students by age
min_age = 20
max_age = 22
filtered_students = filter_students_by_age(students, min_age, max_age)

print(f"\nStudents between {min_age} and {max_age} years old:")
for name, info in filtered_students.items():
    print(f"{name}: {info}")
```
