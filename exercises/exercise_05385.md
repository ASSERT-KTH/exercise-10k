## Exercise
Create a function that calculates the sum of all numbers in a list.

## Reference Solution
```python
def calculate_sum(numbers):
    """
    Calculate the sum of all numbers in a list.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The sum of all numbers in the list.
    """
    return sum(numbers)

# Example usage:
numbers = [1, 2, 3, 4, 5]
result = calculate_sum(numbers)
print("The sum of the numbers is:", result)
```

```python
def calculate_sum_manual(numbers):
    """
    Calculate the sum of all numbers in a list manually.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The sum of all numbers in the list.
    """
    total = 0
    for num in numbers:
        total += num
    return total

# Example usage:
numbers = [1, 2, 3, 4, 5]
result = calculate_sum_manual(numbers)
print("The sum of the numbers is:", result)
```

```python
def calculate_sum_with_error_handling(numbers):
    """
    Calculate the sum of all numbers in a list with error handling.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The sum of all numbers in the list.

    Raises:
        TypeError: If the input is not a list or if the list contains non-numeric values.
    """
    if not isinstance(numbers, list):
        raise TypeError("Input must be a list")
    try:
        return sum(numbers)
    except TypeError:
        raise TypeError("List must only contain numeric values")

# Example usage:
numbers = [1, 2, 3, 4, 5]
try:
    result = calculate_sum_with_error_handling(numbers)
    print("The sum of the numbers is:", result)
except TypeError as e:
    print("Error:", e)
```
