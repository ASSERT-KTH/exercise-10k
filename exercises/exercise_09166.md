## Exercise
Create a program with poor code maintainability (e.g. duplicated code, magic numbers) and refactor it to improve maintainability, then print out the refactored code.

```python
def calculate_area(length, width):
    if length == 0 or width == 0:
        return 0
    return length * width

def calculate_perimeter(length, width):
    if length == 0 or width == 0:
        return 0
    return 2 * (length + width)

def calculate_area_and_perimeter(length, width):
    area = calculate_area(length, width)
    perimeter = calculate_perimeter(length, width)
    return area, perimeter

def main():
    length = 5
    width = 3
    area1, perimeter1 = calculate_area_and_perimeter(length, width)
    print(f"Area: {area1}, Perimeter: {perimeter1}")

    length = 10
    width = 2
    area2, perimeter2 = calculate_area_and_perimeter(length, width)
    print(f"Area: {area2}, Perimeter: {perimeter2}")

    length = 7
    width = 4
    area3, perimeter3 = calculate_area_and_perimeter(length, width)
    print(f"Area: {area3}, Perimeter: {perimeter3}")

if __name__ == "__main__":
    main()
```

```python
class Rectangle:
    def __init__(self, length, width):
        self.length = length
        self.width = width

    def calculate_area(self):
        """Calculate the area of the rectangle."""
        if self.length <= 0 or self.width <= 0:
            return 0
        return self.length * self.width

    def calculate_perimeter(self):
        """Calculate the perimeter of the rectangle."""
        if self.length <= 0 or self.width <= 0:
            return 0
        return 2 * (self.length + self.width)

    def print_details(self):
        """Print the area and perimeter of the rectangle."""
        area = self.calculate_area()
        perimeter = self.calculate_perimeter()
        print(f"Length: {self.length}, Width: {self.width}, Area: {area}, Perimeter: {perimeter}")


def main():
    rectangles = [
        Rectangle(5, 3),
        Rectangle(10, 2),
        Rectangle(7, 4),
    ]

    for rectangle in rectangles:
        rectangle.print_details()


if __name__ == "__main__":
    main()
```
