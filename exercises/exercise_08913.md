## Exercise
Write a program to find the first non-repeated character in a given string.

## Reference Solution
```python
def first_non_repeated_char(s):
    """
    Find the first non-repeated character in a given string.

    Args:
    s (str): The input string.

    Returns:
    str: The first non-repeated character if found, otherwise None.
    """
    # Create a dictionary to store the frequency of each character
    char_frequency = {}
    for char in s:
        if char in char_frequency:
            char_frequency[char] += 1
        else:
            char_frequency[char] = 1

    # Iterate over the string to find the first non-repeated character
    for char in s:
        if char_frequency[char] == 1:
            return char

    # If no non-repeated character is found, return None
    return None


# Example usage:
if __name__ == "__main__":
    s = "hello"
    result = first_non_repeated_char(s)
    if result:
        print(f"The first non-repeated character in '{s}' is '{result}'")
    else:
        print(f"No non-repeated character found in '{s}'")
```
