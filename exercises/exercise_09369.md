## Exercise
Write a program that demonstrates the use of object methods.

## Reference Solution
```python
# Define a Vehicle class
class Vehicle:
    def __init__(self, brand, model, year):
        """
        Initialize a Vehicle object.

        Args:
            brand (str): The brand of the vehicle.
            model (str): The model of the vehicle.
            year (int): The year of the vehicle.
        """
        self.brand = brand
        self.model = model
        self.year = year
        self.mileage = 0

    def start_engine(self):
        """
        Start the vehicle's engine.
        """
        print("The engine is started.")

    def drive(self, miles):
        """
        Drive the vehicle for a specified number of miles.

        Args:
            miles (int): The number of miles to drive.
        """
        self.mileage += miles
        print(f"You have driven {miles} miles.")

    def get_mileage(self):
        """
        Get the current mileage of the vehicle.

        Returns:
            int: The current mileage.
        """
        return self.mileage

    def describe_vehicle(self):
        """
        Print a description of the vehicle.
        """
        print(f"This vehicle is a {self.year} {self.brand} {self.model} with {self.mileage} miles.")


# Create a Vehicle object
my_car = Vehicle("Toyota", "Corolla", 2015)

# Demonstrate object methods
print("Demonstrating object methods:")
my_car.describe_vehicle()
my_car.start_engine()
my_car.drive(100)
print(f"Current mileage: {my_car.get_mileage()} miles")
my_car.drive(50)
print(f"Current mileage: {my_car.get_mileage()} miles")
my_car.describe_vehicle()
```
