## Exercise
Set a value for a key in a dictionary if the key is not already present

## Reference Solution
```python
def set_value_if_not_present(dictionary, key, value):
    """
    Set a value for a key in a dictionary if the key is not already present.

    Args:
        dictionary (dict): The dictionary to update.
        key: The key to set the value for.
        value: The value to set for the key.

    Returns:
        The value that was set for the key, either the new value or the existing value.
    """
    return dictionary.setdefault(key, value)

# Example usage
my_dict = {'a': 1, 'b': 2}
print(set_value_if_not_present(my_dict, 'c', 3))  # Output: 3
print(my_dict)  # Output: {'a': 1, 'b': 2, 'c': 3}

print(set_value_if_not_present(my_dict, 'a', 4))  # Output: 1 (the existing value)
print(my_dict)  # Output: {'a': 1, 'b': 2, 'c': 3}
```

```python
    return dictionary.get(key)
```
