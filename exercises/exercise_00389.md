## Exercise
Write a program that calculates the monthly payment for a loan given the principal amount, interest rate, and number of payments.

## Reference Solution
```python
def calculate_monthly_payment(principal, interest_rate, num_payments):
    """
    Calculate the monthly payment for a loan.

    Args:
        principal (float): The principal amount of the loan.
        interest_rate (float): The annual interest rate as a decimal.
        num_payments (int): The number of payments.

    Returns:
        float: The monthly payment amount.
    """
    monthly_interest_rate = interest_rate / 12
    monthly_payment = principal * monthly_interest_rate * (1 + monthly_interest_rate) ** num_payments / ((1 + monthly_interest_rate) ** num_payments - 1)
    return round(monthly_payment, 2)

def main():
    # Example usage
    principal = 200000  # Principal amount
    interest_rate = 0.05  # Annual interest rate (5%)
    num_payments = 360  # Number of payments (30 years \* 12 months/year)

    monthly_payment = calculate_monthly_payment(principal, interest_rate, num_payments)
    print(f"The monthly payment is: ${monthly_payment}")

if __name__ == "__main__":
    main()
```
