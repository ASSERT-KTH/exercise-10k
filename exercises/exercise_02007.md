## Exercise
Write a program that takes a password as input and checks whether it meets certain conditions using conditional statements.

## Reference Solution
```python
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s;
    cin >> s;
    int upper = 0, lower = 0, num = 0, special = 0;
    int size = s.length();

    for (int i = 0; i < size; i++) {
        char c = s[i];
        if (isupper(c)) {
            upper++;
        } else if (islower(c)) {
            lower++;
        } else if (isdigit(c)) {
            num++;
        } else {
            special++;
        }
    }

    if (upper >= 2 && lower >= 3 && num >= 1 && special >= 2 && size >= 7) {
        cout << "Strong" << endl;
    } else {
        cout << "Weak" << endl;
    }

    return 0;
}
```
