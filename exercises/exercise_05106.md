## Exercise
Perform common operations on characters and strings, such as concatenation, substring extraction, and case conversion.

## Reference Solution
```python
def string_operations():
    # Define two strings for concatenation
    str1 = "Hello, "
    str2 = "World!"

    # Concatenation of two strings
    print("Concatenation of '{}' and '{}': {}".format(str1, str2, str1 + str2))

    # Define a string for substring extraction
    original_str = "Hello, World!"

    # Extract a substring from the original string
    start_idx = 7
    end_idx = 12
    print("Substring of '{}' from index {} to {}: {}".format(original_str, start_idx, end_idx, original_str[start_idx:end_idx]))

    # Define a string for case conversion
    case_str = "hElLo, WoRlD!"

    # Case conversion of a string (lowercase to uppercase and vice versa)
    print("Original string: {}".format(case_str))
    print("Lowercase: {}".format(case_str.lower()))
    print("Uppercase: {}".format(case_str.upper()))
    print("Title case: {}".format(case_str.title()))

    # Define a string for swapping case
    swap_case_str = "HeLlO, wOrLd!"

    # Swap case of a string
    print("Original string: {}".format(swap_case_str))
    print("Swapped case: {}".format(swap_case_str.swapcase()))

# Test the function
string_operations()
```
