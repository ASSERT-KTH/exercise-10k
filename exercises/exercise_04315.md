## Exercise
Create a function that takes multiple arguments and returns a new function with the same arguments

## Reference Solution
```python
def create_function(*args):
    """
    Creates a new function with the same arguments.

    Args:
        *args: Variable number of arguments.

    Returns:
        A new function with the same arguments.
    """

    # Define the new function
    def new_function(*new_args):
        # Check if the number of arguments is the same
        if len(args) != len(new_args):
            raise ValueError("The number of arguments must be the same")

        # Return the new arguments
        return new_args

    return new_function


# Example usage:
if __name__ == "__main__":
    # Create a new function with 3 arguments
    new_func = create_function("arg1", "arg2", "arg3")

    # Call the new function with 3 arguments
    print(new_func("value1", "value2", "value3"))  # Output: ('value1', 'value2', 'value3')

    # Try calling the new function with a different number of arguments
    try:
        print(new_func("value1", "value2"))
    except ValueError as e:
        print(e)  # Output: The number of arguments must be the same
```
