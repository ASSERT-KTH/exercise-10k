## Exercise
Write a program that takes a simple programming language as input and outputs the compiled machine code.

## Reference Solution
```python
import re

# Define the grammar for our programming language
class Compiler:
    def __init__(self, code):
        self.code = code
        self.variables = {}
        self.machine_code = []

    def compile(self):
        # Split the code into individual statements
        statements = self.code.splitlines()

        for statement in statements:
            # Remove leading and trailing whitespace
            statement = statement.strip()

            # Check if the statement is a variable declaration
            if statement.startswith("var"):
                self.compile_var_declaration(statement)
            # Check if the statement is an assignment
            elif "=" in statement:
                self.compile_assignment(statement)
            # Check if the statement is a print statement
            elif statement.startswith("print"):
                self.compile_print_statement(statement)
            # Check if the statement is an arithmetic operation
            elif any(operator in statement for operator in ["+", "-", "*", "/"]):
                self.compile_arithmetic_operation(statement)

    def compile_var_declaration(self, statement):
        # Extract the variable name
        var_name = statement.split()[1]

        # Add a machine code instruction to declare the variable
        self.machine_code.append(f"DECLARE {var_name}")

    def compile_assignment(self, statement):
        # Split the statement into variable name and value
        var_name, value = statement.split("=")

        # Remove leading and trailing whitespace
        var_name = var_name.strip()
        value = value.strip()

        # Check if the value is a number
        if value.isdigit():
            # Add a machine code instruction to assign the value to the variable
            self.machine_code.append(f"ASSIGN {var_name} {value}")
        else:
            # Check if the value is an arithmetic expression
            if any(operator in value for operator in ["+", "-", "*", "/"]):
                self.compile_arithmetic_operation(value)
                # Add a machine code instruction to assign the result to the variable
                self.machine_code.append(f"ASSIGN {var_name} RESULT")
            else:
                # Check if the value is a variable
                if value in self.variables:
                    # Add a machine code instruction to assign the value of the variable to the new variable
                    self.machine_code.append(f"ASSIGN {var_name} {value}")

    def compile_print_statement(self, statement):
        # Extract the value to be printed
        value = statement.split()[1]

        # Check if the value is a number
        if value.isdigit():
            # Add a machine code instruction to print the value
            self.machine_code.append(f"PRINT {value}")
        else:
            # Check if the value is a variable
            if value in self.variables:
                # Add a machine code instruction to print the value of the variable
                self.machine_code.append(f"PRINT {value}")
            else:
                # Check if the value is an arithmetic expression
                if any(operator in value for operator in ["+", "-", "*", "/"]):
                    self.compile_arithmetic_operation(value)
                    # Add a machine code instruction to print the result
                    self.machine_code.append("PRINT RESULT")

    def compile_arithmetic_operation(self, expression):
        # Split the expression into operands and operator
        operands = re.findall(r"[0-9]+|[a-zA-Z]+", expression)
        operator = re.findall(r"[+\-*/]", expression)[0]

        # Add machine code instructions to load the operands
        for operand in operands:
            if operand.isdigit():
                self.machine_code.append(f"LOAD {operand}")
            else:
                self.machine_code.append(f"LOAD {operand}")

        # Add a machine code instruction to perform the arithmetic operation
        if operator == "+":
            self.machine_code.append("ADD")
        elif operator == "-":
            self.machine_code.append("SUB")
        elif operator == "*":
            self.machine_code.append("MUL")
        elif operator == "/":
            self.machine_code.append("DIV")

    def get_machine_code(self):
        return self.machine_code


# Example usage:
code = """
var x
x = 5
var y
y = x + 3
print y
"""

compiler = Compiler(code)
compiler.compile()
print(compiler.get_machine_code())
```
