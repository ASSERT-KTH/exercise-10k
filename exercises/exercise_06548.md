## Exercise
Create a function that searches for a substring within a given string.

## Reference Solution
```python
def search_substring(main_string, sub_string):
    """
    Searches for a substring within a given string.

    Args:
        main_string (str): The string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the first occurrence of the substring, or -1 if not found.
    """
    try:
        # Use the str.find() method to search for the substring
        index = main_string.find(sub_string)
        return index
    except TypeError:
        # Handle the case where the input is not a string
        print("Error: Input must be a string.")
        return None

# Example usage
if __name__ == "__main__":
    main_string = "Hello, World!"
    sub_string = "World"
    index = search_substring(main_string, sub_string)
    if index != -1:
        print(f"The substring '{sub_string}' was found at index {index}.")
    else:
        print(f"The substring '{sub_string}' was not found.")
```
