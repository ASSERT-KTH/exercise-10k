## Exercise
Write a program that takes a user input and determines its data type. The program should then convert the input to other data types (integer, float, string) if possible and print the converted values. If the conversion is not possible, print an error message.

## Reference Solution
```python
def convert_to_int(input_value):
    """Attempt to convert input to integer."""
    try:
        return int(input_value)
    except ValueError:
        return "Error: Cannot convert to integer."

def convert_to_float(input_value):
    """Attempt to convert input to float."""
    try:
        return float(input_value)
    except ValueError:
        if isinstance(input_value, int):
            return float(input_value)
        else:
            return "Error: Cannot convert to float."

def convert_to_str(input_value):
    """Attempt to convert input to string."""
    try:
        return str(input_value)
    except Exception:
        return "Error: Cannot convert to string."

def determine_and_convert_input():
    """Determine the data type of the input and attempt to convert it to other types."""
    user_input = input("Please enter a value: ")
    
    # Determine the input type
    if user_input.isdigit():
        input_type = "integer"
        input_value = int(user_input)
    elif user_input.replace('.', '', 1).isdigit():
        input_type = "float"
        input_value = float(user_input)
    else:
        input_type = "string"
        input_value = user_input

    print(f"Input type: {input_type}")

    # Convert the input to other types
    conversions = {
        "integer": {
            "float": convert_to_float,
            "string": convert_to_str
        },
        "float": {
            "integer": convert_to_int,
            "string": convert_to_str
        },
        "string": {
            "integer": convert_to_int,
            "float": convert_to_float
        }
    }

    for target_type, conversion_func in conversions[input_type].items():
        converted_value = conversion_func(input_value)
        print(f"Converted to {target_type}: {converted_value}")

if __name__ == "__main__":
    determine_and_convert_input()
```
