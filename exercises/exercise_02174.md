## Exercise
Ask the user for the principal amount, interest rate, and number of years, and print the simple interest.

## Reference Solution
```python
def calculate_simple_interest(principal, rate, years):
    """
    Calculate the simple interest.

    Args:
        principal (float): The principal amount.
        rate (float): The interest rate (in %).
        years (int): The number of years.

    Returns:
        float: The simple interest.
    """
    return (principal * rate * years) / 100

def main():
    # Ask the user for the principal amount
    while True:
        try:
            principal = float(input("Enter the principal amount: $"))
            if principal <= 0:
                print("Please enter a positive number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Ask the user for the interest rate
    while True:
        try:
            rate = float(input("Enter the interest rate (%): "))
            if rate < 0:
                print("Please enter a non-negative number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Ask the user for the number of years
    while True:
        try:
            years = int(input("Enter the number of years: "))
            if years <= 0:
                print("Please enter a positive integer.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid integer.")

    # Calculate and print the simple interest
    interest = calculate_simple_interest(principal, rate, years)
    total_amount = principal + interest
    print(f"\nPrincipal Amount: ${principal:.2f}")
    print(f"Interest Rate: {rate}%")
    print(f"Number of Years: {years} years")
    print(f"Simple Interest: ${interest:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
