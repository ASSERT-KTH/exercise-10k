## Exercise
Write a program to compare two fractions and determine which one is larger.

## Reference Solution
```python
class Fraction:
    def __init__(self, numerator, denominator):
        """
        Initialize a Fraction object.

        Args:
            numerator (int): The numerator of the fraction.
            denominator (int): The denominator of the fraction.

        Raises:
            ZeroDivisionError: If the denominator is zero.
        """
        if denominator == 0:
            raise ZeroDivisionError("Denominator cannot be zero")
        self.numerator = numerator
        self.denominator = denominator

    def __lt__(self, other):
        """
        Compare this fraction with another fraction.

        Args:
            other (Fraction): The other fraction to compare with.

        Returns:
            bool: True if this fraction is less than the other, False otherwise.
        """
        return self.numerator * other.denominator < other.numerator * self.denominator

    def __gt__(self, other):
        """
        Compare this fraction with another fraction.

        Args:
            other (Fraction): The other fraction to compare with.

        Returns:
            bool: True if this fraction is greater than the other, False otherwise.
        """
        return self.numerator * other.denominator > other.numerator * self.denominator

    def __eq__(self, other):
        """
        Compare this fraction with another fraction for equality.

        Args:
            other (Fraction): The other fraction to compare with.

        Returns:
            bool: True if this fraction is equal to the other, False otherwise.
        """
        return self.numerator * other.denominator == other.numerator * self.denominator

    def __str__(self):
        """
        Return a string representation of the fraction.

        Returns:
            str: A string representation of the fraction.
        """
        return f"{self.numerator}/{self.denominator}"


def compare_fractions(fraction1, fraction2):
    """
    Compare two fractions and determine which one is larger.

    Args:
        fraction1 (Fraction): The first fraction to compare.
        fraction2 (Fraction): The second fraction to compare.

    Returns:
        str: A string indicating which fraction is larger.
    """
    if fraction1 > fraction2:
        return f"{fraction1} is larger than {fraction2}"
    elif fraction1 < fraction2:
        return f"{fraction1} is smaller than {fraction2}"
    else:
        return f"{fraction1} is equal to {fraction2}"


# Example usage
fraction1 = Fraction(1, 2)
fraction2 = Fraction(1, 3)

print(compare_fractions(fraction1, fraction2))
```
