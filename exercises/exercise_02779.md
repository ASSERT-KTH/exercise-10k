## Exercise
Write a program that takes a user's name and age as input and prints out a personalized message.

## Reference Solution
```python
def get_user_info():
    """Gets user's name and age as input."""
    name = input("Please enter your name: ")
    while True:
        try:
            age = int(input("Please enter your age: "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid age.")
    return name, age


def print_personalized_message(name, age):
    """Prints out a personalized message."""
    print(f"Hello, {name}! You are {age} years old.")


def main():
    """Main function."""
    name, age = get_user_info()
    print_personalized_message(name, age)


if __name__ == "__main__":
    main()
```
