## Exercise
Write a function to convert a list of tuples to a dictionary

## Reference Solution
```python
def convert_to_dict(tuple_list):
    """
    Convert a list of tuples into a dictionary.

    Args:
        tuple_list (list): A list of tuples, where each tuple contains a key-value pair.

    Returns:
        dict: A dictionary created from the list of tuples.

    Raises:
        ValueError: If the list of tuples contains duplicate keys.
    """
    try:
        # Use the dict() function to convert the list of tuples into a dictionary
        return dict(tuple_list)
    except Exception as e:
        # Raise an error if there are duplicate keys
        raise ValueError("The list of tuples contains duplicate keys.") from e


# Example usage:
def main():
    # Create a list of tuples
    tuple_list = [("apple", 5), ("banana", 10), ("orange", 7)]

    # Convert the list of tuples into a dictionary
    fruit_dict = convert_to_dict(tuple_list)

    # Print the resulting dictionary
    print(fruit_dict)


if __name__ == "__main__":
    main()
```

```python
def convert_to_dict_with_lists(tuple_list):
    """
    Convert a list of tuples into a dictionary, preserving all values for duplicate keys.

    Args:
        tuple_list (list): A list of tuples, where each tuple contains a key-value pair.

    Returns:
        dict: A dictionary created from the list of tuples, where each value is a list.
    """
    result = {}
    for key, value in tuple_list:
        if key in result:
            result[key].append(value)
        else:
            result[key] = [value]
    return result


# Example usage:
def main():
    # Create a list of tuples
    tuple_list = [("apple", 5), ("banana", 10), ("apple", 7)]

    # Convert the list of tuples into a dictionary with list values
    fruit_dict = convert_to_dict_with_lists(tuple_list)

    # Print the resulting dictionary
    print(fruit_dict)


if __name__ == "__main__":
    main()
```
