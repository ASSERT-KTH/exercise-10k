## Exercise
Declare and use object variables to store key-value pairs.

## Reference Solution
```python
// Declare an object variable
let person = {
  // Key-value pairs
  name: "John Doe",
  age: 30,
  occupation: "Software Developer"
};

// Accessing values using dot notation
console.log(person.name);  // Output: John Doe
console.log(person.age);   // Output: 30

// Accessing values using bracket notation
console.log(person["occupation"]);  // Output: Software Developer

// Updating values
person.name = "Jane Doe";
console.log(person.name);  // Output: Jane Doe

// Adding new key-value pairs
person.location = "New York";
console.log(person);  // Output: { name: 'Jane Doe', age: 30, occupation: 'Software Developer', location: 'New York' }

// Removing key-value pairs
delete person.age;
console.log(person);  // Output: { name: 'Jane Doe', occupation: 'Software Developer', location: 'New York' }
```

```python
// Using Object.keys()
let keys = Object.keys(person);
console.log(keys);  // Output: ['name', 'occupation', 'location']

// Using Object.values()
let values = Object.values(person);
console.log(values);  // Output: ['Jane Doe', 'Software Developer', 'New York']

// Using Object.entries()
let entries = Object.entries(person);
console.log(entries);  
// Output:

//   [ 'location', 'New York' ]
```
