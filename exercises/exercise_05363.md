## Exercise
Write a program to append, insert, and remove elements from an array.

## Reference Solution
```python
class ArrayOperations:
    def __init__(self):
        self.array = []

    def append(self, element):
        """
        Adds a new element to the end of the array.

        Args:
            element (any): The element to be appended.
        """
        self.array.append(element)
        print(f"Element {element} appended to the array.")

    def insert(self, index, element):
        """
        Inserts a new element at the specified index in the array.

        Args:
            index (int): The index where the element will be inserted.
            element (any): The element to be inserted.
        """
        if index < 0 or index > len(self.array):
            print("Invalid index. Please enter an index between 0 and", len(self.array))
        else:
            self.array.insert(index, element)
            print(f"Element {element} inserted at index {index}.")

    def remove(self, element=None, index=None):
        """
        Removes the specified element or the element at the specified index from the array.

        Args:
            element (any, optional): The element to be removed. Defaults to None.
            index (int, optional): The index of the element to be removed. Defaults to None.
        """
        if element is not None:
            if element in self.array:
                self.array.remove(element)
                print(f"Element {element} removed from the array.")
            else:
                print(f"Element {element} not found in the array.")
        elif index is not None:
            if index < 0 or index >= len(self.array):
                print("Invalid index. Please enter an index between 0 and", len(self.array) - 1)
            else:
                removed_element = self.array.pop(index)
                print(f"Element {removed_element} removed from index {index}.")
        else:
            print("Please specify either an element or an index to remove.")

    def display(self):
        """
        Prints the current state of the array.
        """
        print("Current array:", self.array)


def main():
    array_operations = ArrayOperations()
    
    while True:
        print("\nArray Operations Menu:")
        print("1. Append an element")
        print("2. Insert an element")
        print("3. Remove an element")
        print("4. Display the array")
        print("5. Quit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            element = input("Enter the element to append: ")
            array_operations.append(element)
        elif choice == "2":
            index = int(input("Enter the index to insert: "))
            element = input("Enter the element to insert: ")
            array_operations.insert(index, element)
        elif choice == "3":
            print("\nRemove Options:")
            print("1. Remove by element")
            print("2. Remove by index")
            remove_choice = input("Enter your choice: ")
            if remove_choice == "1":
                element = input("Enter the element to remove: ")
                array_operations.remove(element=element)
            elif remove_choice == "2":
                index = int(input("Enter the index to remove: "))
                array_operations.remove(index=index)
        elif choice == "4":
            array_operations.display()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
