## Exercise
Create a program that takes in an income and calculates the corresponding tax using conditional statements.

## Reference Solution
```python
def calculate_tax(income):
    """
    Calculates the tax amount based on the given income.

    Args:
        income (float): The income amount.

    Returns:
        float: The calculated tax amount.
    """
    if income <= 0:
        return 0
    elif income <= 10000:
        return income * 0.10
    elif income <= 20000:
        return (10000 * 0.10) + ((income - 10000) * 0.15)
    elif income <= 50000:
        return (10000 * 0.10) + (10000 * 0.15) + ((income - 20000) * 0.20)
    else:
        return (10000 * 0.10) + (10000 * 0.15) + (30000 * 0.20) + ((income - 50000) * 0.25)

def main():
    income = float(input("Enter your income: $"))
    tax = calculate_tax(income)
    print(f"Your tax amount is: ${tax:.2f}")

if __name__ == "__main__":
    main()
```
