## Exercise
Create a program that uses a for loop to generate all permutations of a given string.

## Reference Solution
```python
def generate_permutations(string):
    """
    Generate all permutations of a given string.

    Args:
    string (str): The input string.

    Returns:
    list: A list of all permutations.
    """
    # Base case: if the string is empty or only contains one character
    if len(string) <= 1:
        return [string]

    permutations = []
    for i, char in enumerate(string):
        # Remove the current character from the string
        remaining_string = string[:i] + string[i+1:]
        
        # Generate all permutations of the remaining string
        for p in generate_permutations(remaining_string):
            # Add the current character to the beginning of each permutation
            permutations.append(char + p)

    return permutations


def main():
    string = "abc"
    print("Input string: ", string)
    print("Permutations: ", generate_permutations(string))


if __name__ == "__main__":
    main()
```

import itertools

def generate_permutations(string):
    """
    Generate all permutations of a given string.

    Args:
    string (str): The input string.

    Returns:
    list: A list of all permutations.
    """
    permutations = [''.join(p) for p in itertools.permutations(string)]
    return permutations


def main():
    string = "abc"
    print("Input string: ", string)
    print("Permutations: ", generate_permutations(string))


if __name__ == "__main__":
