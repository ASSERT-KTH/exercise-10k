## Exercise
Write a program that takes a list of integers and a target value as input and returns the index of the target value in the list using linear search.

## Reference Solution
```python
#include <stdio.h>

void linear_search(int a[], int n, int t, int *ind) {
    for (int i = 0; i < n; i++) {
        if (a[i] == t) {
            *ind = i;
            return;
        }
    }
    *ind = -1;
}

int main() {
    int a[] = {1, 2, 3, 4, 5};
    int n = sizeof(a) / sizeof(a[0]);
    int t = 3;
    int ind;

    linear_search(a, n, t, &ind);

    if (ind == -1) {
        printf("Target value not found in the array.\n");
    } else {
        printf("Target value found at index %d\n", ind);
    }

    return 0;
}
```
