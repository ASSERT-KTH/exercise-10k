## Exercise
Write a program that demonstrates the concept of variable scope by declaring variables inside and outside a function.

## Reference Solution
```python
# Declare a variable outside a function
x = 10  # Global variable

def my_function():
    # Declare a variable inside a function
    y = 20  # Local variable
    
    # Access the global variable inside the function
    print("Global variable x:", x)
    
    # Access the local variable inside the function
    print("Local variable y:", y)

# Call the function
my_function()

# Try to access the local variable outside the function
try:
    print("Local variable y:", y)
except NameError:
    print("Error: Local variable y is not accessible outside the function.")

# Modify the global variable inside the function
def modify_global_variable():
    global x  # Declare x as a global variable
    x = 30  # Modify the global variable
    
# Call the function
modify_global_variable()

# Print the modified global variable
print("Modified global variable x:", x)

# Declare a variable with the same name inside and outside a function
z = 40  # Global variable

def modify_variable_with_same_name():
    z = 50  # Local variable with the same name
    print("Local variable z:", z)

# Call the function
modify_variable_with_same_name()

# Print the global variable
print("Global variable z:", z)
```
