## Exercise
Write a program that takes a string and a subsequence as input and returns the indices of the subsequence in the string.

## Reference Solution

def find_subsequence_indices(string, subsequence):
    """
    Find the indices of a subsequence in a string.

    Args:
        string (str): The input string.
        subsequence (str): The subsequence to find.

    Returns:
        list: A list of indices where the subsequence appears in the string.
    """
    # Initialize a 2D table to store the lengths of common prefixes
    dp = [[0] * (len(subsequence) + 1) for _ in range(len(string) + 1)]

    # Fill the table in a bottom-up manner
    for i in range(1, len(string) + 1):
        for j in range(1, len(subsequence) + 1):
            if string[i - 1] == subsequence[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = 0

    # Find the indices of the subsequence
    indices = []
    for i in range(len(string)):
        if dp[i + 1][len(subsequence)] == len(subsequence):
            indices.append(i - len(subsequence) + 1)

    return indices

# Example usage:
string = "banana"
subsequence = "ana"
indices = find_subsequence_indices(string, subsequence)
print(indices)  # Output: [1, 3]
