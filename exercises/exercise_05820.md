## Exercise
Create a program that uses a while loop to manage a to-do list, allowing users to add, remove, and mark tasks as completed.

## Reference Solution
```python
# Initialize the to-do list
to_do_list = []

def display_to_do_list():
    """Displays the to-do list."""
    print("\nTo-Do List:")
    for i, task in enumerate(to_do_list, start=1):
        status = "Completed" if task["completed"] else "Not Completed"
        print(f"{i}. {task['task']} - {status}")

def add_task():
    """Adds a task to the to-do list."""
    task_name = input("Enter a task name: ")
    to_do_list.append({"task": task_name, "completed": False})

def remove_task():
    """Removes a task from the to-do list."""
    if not to_do_list:
        print("No tasks to remove.")
        return

    display_to_do_list()
    task_number = int(input("Enter the task number to remove: ")) - 1
    if task_number < 0 or task_number >= len(to_do_list):
        print("Invalid task number.")
        return

    del to_do_list[task_number]

def mark_task_as_completed():
    """Marks a task as completed."""
    if not to_do_list:
        print("No tasks to mark as completed.")
        return

    display_to_do_list()
    task_number = int(input("Enter the task number to mark as completed: ")) - 1
    if task_number < 0 or task_number >= len(to_do_list):
        print("Invalid task number.")
        return

    to_do_list[task_number]["completed"] = True

def main():
    while True:
        print("\nTo-Do List Manager")
        print("1. Display To-Do List")
        print("2. Add Task")
        print("3. Remove Task")
        print("4. Mark Task as Completed")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            display_to_do_list()
        elif choice == "2":
            add_task()
        elif choice == "3":
            remove_task()
        elif choice == "4":
            mark_task_as_completed()
        elif choice == "5":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
