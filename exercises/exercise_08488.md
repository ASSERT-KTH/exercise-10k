## Exercise
Create a program that takes a list of integers as input and prints out the sum of all the elements in the list, along with the average value. The program should also print out the largest and smallest numbers in the list.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Masukkan jumlah data: ";
    cin >> n;

    int* list = new int[n];

    for (int i = 0; i < n; i++) {
        cout << "Masukkan data ke-" << i + 1 << ": ";
        cin >> list[i];
    }

    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += list[i];
    }

    double rata = (double)sum / n;

    int max = list[0];
    int min = list[0];

    for (int i = 1; i < n; i++) {
        if (list[i] > max) {
            max = list[i];
        }
        if (list[i] < min) {
            min = list[i];
        }
    }

    cout << "Jumlah data: " << n << endl;
    cout << "Jumlah: " << sum << endl;
    cout << "Rata-rata: " << rata << endl;
    cout << "Nilai tertinggi: " << max << endl;
    cout << "Nilai terendah: " << min << endl;

    delete[] list;

    return 0;
}
```
