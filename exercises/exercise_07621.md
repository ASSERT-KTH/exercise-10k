## Exercise
Write a program that uses a for loop to check if two given strings are anagrams of each other.

## Reference Solution
```python
def are_anagrams(str1, str2):
    # Remove any white spaces and convert to lower case
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # Check if both strings have the same length
    if len(str1) != len(str2):
        return False

    # Create a dictionary to store the frequency of characters
    char_freq = {}

    # Count the frequency of characters in the first string
    for char in str1:
        if char in char_freq:
            char_freq[char] += 1
        else:
            char_freq[char] = 1

    # Subtract the frequency of characters in the second string
    for char in str2:
        if char in char_freq:
            char_freq[char] -= 1
        else:
            return False

    # Check if all frequencies are zero
    for freq in char_freq.values():
        if freq != 0:
            return False

    return True


# Test the function
str1 = "Listen"
str2 = "Silent"
print(are_anagrams(str1, str2))  # Output: True

str1 = "Hello"
str2 = "World"
print(are_anagrams(str1, str2))  # Output: False
```
