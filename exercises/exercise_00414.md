## Exercise
Given an array of integers, find the longest increasing subsequence.

## Reference Solution
```python
def longest_increasing_subsequence(nums):
    """
    Find the longest increasing subsequence in a given array of integers.

    Args:
    - nums (list): The input array of integers.

    Returns:
    - list: The longest increasing subsequence.
    """
    if not nums:
        return []

    n = len(nums)
    dp = [1] * n
    prev = [-1] * n

    max_length = 1
    max_index = 0

    for i in range(1, n):
        for j in range(i):
            if nums[i] > nums[j] and dp[i] < dp[j] + 1:
                dp[i] = dp[j] + 1
                prev[i] = j
        if dp[i] > max_length:
            max_length = dp[i]
            max_index = i

    # Reconstruct the LIS
    lis = []
    while max_index != -1:
        lis.append(nums[max_index])
        max_index = prev[max_index]

    return lis[::-1]

# Example usage
nums = [10, 22, 9, 33, 21, 50, 41, 60, 80]
print(longest_increasing_subsequence(nums))  # [10, 22, 33, 50, 60, 80]
```
