## Exercise
Create a function that scrambles a word and asks the user to unscramble it.

## Reference Solution
```python
import random

def word_scramble_game():
    # List of words to scramble
    words = ["apple", "banana", "cherry", "date", "elderberry"]

    # Choose a random word from the list
    word_to_scramble = random.choice(words)

    # Scramble the word
    scrambled_word = "".join(random.sample(word_to_scramble, len(word_to_scramble)))

    # Ask the user to unscramble the word
    print(f"Unscramble the word: {scrambled_word}")
    user_answer = input("Your answer: ")

    # Check if the user's answer is correct
    if user_answer.lower() == word_to_scramble:
        print("Congratulations! You unscrambled the word correctly.")
    else:
        print(f"Sorry, that's incorrect. The correct answer was {word_to_scramble}.")

# Run the game
word_scramble_game()
```
