## Exercise
Write a program that takes a string and a delimiter as input, and returns a list of substrings split by the delimiter.

## Reference Solution
```python
function splitString(string, delimiter) {
  var array = [];
  var index = string.indexOf(delimiter);
  var string1 = string;

  while (index !== -1) {
    array.push(string1.substring(0, index));
    string1 = string1.substring(index + 1);
    index = string1.indexOf(delimiter);
  }

  array.push(string1);
  return array;
}

// Test the function
console.log(splitString("01/02/03", "/"));
```
