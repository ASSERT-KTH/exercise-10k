## Exercise
Write a function to check if a given graph has a cycle.

from typing import Dict, List

def has_cycle(graph: Dict[int, List[int]]) -> bool:
    """
    Checks if a given graph has a cycle.

    Args:
    graph: A dictionary representing the graph as an adjacency list.

    Returns:
    True if the graph has a cycle, False otherwise.
    """
    visited = set()
    recursion_stack = set()

    def dfs(node: int) -> bool:
        """
        Recursive DFS function to detect cycles.

        Args:
        node: The current node being visited.

        Returns:
        True if a cycle is detected, False otherwise.
        """
        visited.add(node)
        recursion_stack.add(node)

        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                if dfs(neighbor):
                    return True
            elif neighbor in recursion_stack:
                return True

        recursion_stack.remove(node)
        return False

    for node in graph:
        if node not in visited:
            if dfs(node):
                return True

    return False


# Example usage:
if __name__ == "__main__":
    # Define a graph with a cycle
    graph_with_cycle = {
        0: [1, 2],
        1: [2],
        2: [0, 3],
        3: [3]
    }

    # Define a graph without a cycle
    graph_without_cycle = {
        0: [1, 2],
        1: [2],
        2: [3],
        3: []
    }

    print(has_cycle(graph_with_cycle))  # Output: True
    print(has_cycle(graph_without_cycle))  # Output: False
