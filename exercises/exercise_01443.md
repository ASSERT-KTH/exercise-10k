## Exercise
Write a program that takes two numbers as input and outputs their division and remainder.

## Reference Solution
```python
def divide_and_remainder(dividend, divisor):
    """
    This function takes two numbers as input and returns their division and remainder.

    Args:
        dividend (float): The number being divided.
        divisor (float): The number by which we are dividing.

    Returns:
        tuple: A tuple containing the division result and the remainder.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")

    division_result = dividend / divisor
    remainder = dividend % divisor

    return division_result, remainder


def main():
    # Example usage
    dividend = float(input("Enter the dividend: "))
    divisor = float(input("Enter the divisor: "))

    try:
        division_result, remainder = divide_and_remainder(dividend, divisor)
        print(f"The division result is: {division_result}")
        print(f"The remainder is: {remainder}")
    except ZeroDivisionError as e:
        print(e)


if __name__ == "__main__":
    main()
```
