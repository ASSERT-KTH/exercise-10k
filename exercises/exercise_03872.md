## Exercise
Create a dictionary with names of countries as keys and their capitals as values. Filter the dictionary to include only countries in a specific continent and print the filtered dictionary.

## Reference Solution
```python
# Define a dictionary with countries as keys and their capitals as values
countries = {
    "France": "Paris",
    "Germany": "Berlin",
    "Italy": "Rome",
    "Spain": "Madrid",
    "China": "Beijing",
    "Japan": "Tokyo",
    "India": "New Delhi",
    "United States": "Washington D.C.",
    "Canada": "Ottawa",
    "Mexico": "Mexico City",
    "Brazil": "Bras√≠lia",
    "Argentina": "Buenos Aires",
}

# Define a dictionary with countries as keys and their continents as values
country_continents = {
    "France": "Europe",
    "Germany": "Europe",
    "Italy": "Europe",
    "Spain": "Europe",
    "China": "Asia",
    "Japan": "Asia",
    "India": "Asia",
    "United States": "North America",
    "Canada": "North America",
    "Mexico": "North America",
    "Brazil": "South America",
    "Argentina": "South America",
}

def filter_countries_by_continent(countries, country_continents, continent):
    """
    Filter a dictionary of countries by continent.

    Args:
        countries (dict): A dictionary with countries as keys and their capitals as values.
        country_continents (dict): A dictionary with countries as keys and their continents as values.
        continent (str): The continent to filter by.

    Returns:
        dict: A dictionary with countries in the specified continent as keys and their capitals as values.
    """
    filtered_countries = {
        country: capital
        for country, capital in countries.items()
        if country_continents.get(country) == continent
    }
    return filtered_countries

# Filter the dictionary to include only countries in Europe
continent = "Europe"
filtered_countries = filter_countries_by_continent(countries, country_continents, continent)

# Print the filtered dictionary
print(f"Countries in {continent}:")
print(filtered_countries)
```
