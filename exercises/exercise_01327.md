## Exercise
Write a function that takes a string and a shift value as input and returns the encrypted string using the Caesar cipher.

## Reference Solution
```python
def caesar_cipher(text: str, shift: int) -> str:
    """
    Encrypts a string using the Caesar cipher.

    Args:
    text (str): The input string to be encrypted.
    shift (int): The number of positions to shift each letter.

    Returns:
    str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
        # If the character is not a letter, add it as is
        else:
            result += char

    return result

# Example usage:
if __name__ == "__main__":
    text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_cipher(text, shift)
    print(f"Text: {text}")
    print(f"Shift: {shift}")
    print(f"Encrypted Text: {encrypted_text}")
```
