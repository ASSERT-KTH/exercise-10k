## Exercise
Write a program to calculate the average grade of a student given their grades in different subjects.

## Reference Solution
```python
def calculate_average_grade(grades):
    """
    Calculate the average grade of a student given their grades in different subjects.

    Args:
        grades (dict): A dictionary where the keys are the subject names and the values are the grades.

    Returns:
        float: The average grade of the student.
    """
    total_grade = sum(grades.values())
    average_grade = total_grade / len(grades)
    return average_grade


def main():
    # Get the number of subjects
    num_subjects = int(input("Enter the number of subjects: "))

    # Initialize an empty dictionary to store the grades
    grades = {}

    # Get the grades for each subject
    for i in range(num_subjects):
        subject = input(f"Enter the name of subject {i+1}: ")
        grade = float(input(f"Enter the grade for {subject}: "))
        grades[subject] = grade

    # Calculate the average grade
    average_grade = calculate_average_grade(grades)

    # Print the average grade
    print(f"The average grade is: {average_grade:.2f}")


if __name__ == "__main__":
    main()
```
