## Exercise
Implement the depth-first search algorithm to traverse a graph. The program should take a graph represented as an adjacency list and a starting node as input and output the order of visited nodes.

## Reference Solution
```python
def dfs(graph, start):
    """
    Performs a depth-first search on a graph.

    Args:
    graph: A dictionary representing the adjacency list of the graph.
    start: The node to start the search from.

    Returns:
    A list of nodes in the order they were visited.
    """
    # Create a set to store visited nodes
    visited = set()
    
    # Create a list to store the order of visited nodes
    traversal_order = []
    
    # Define a helper function to perform the recursive DFS
    def dfs_helper(node):
        # Mark the current node as visited
        visited.add(node)
        
        # Add the current node to the traversal order
        traversal_order.append(node)
        
        # Iterate over the neighbors of the current node
        for neighbor in graph[node]:
            # If the neighbor has not been visited, recursively visit it
            if neighbor not in visited:
                dfs_helper(neighbor)
    
    # Start the DFS from the given starting node
    dfs_helper(start)
    
    # Return the traversal order
    return traversal_order

# Example usage
graph = {
    'A': ['B', 'C'],
    'B': ['A', 'D', 'E'],
    'C': ['A', 'F'],
    'D': ['B'],
    'E': ['B', 'F'],
    'F': ['C', 'E']
}

start_node = 'A'

traversal_order = dfs(graph, start_node)

print("Depth-first traversal order:", traversal_order)
```
