## Exercise
Create a sparse array where most elements are zero, and only non-zero elements are stored.

## Reference Solution
```python
class SparseArray:
    def __init__(self, size):
        """
        Initialize a sparse array of a given size.

        Args:
            size (int): The size of the sparse array.
        """
        self.size = size
        self.data = {}

    def set(self, index, value):
        """
        Set the value at a given index.

        Args:
            index (int): The index to set the value at.
            value (int): The value to set.
        """
        if index < 0 or index >= self.size:
            raise IndexError("Index out of range")

        if value != 0:
            self.data[index] = value
        elif index in self.data:
            del self.data[index]

    def get(self, index):
        """
        Get the value at a given index.

        Args:
            index (int): The index to get the value at.

        Returns:
            int: The value at the given index, or 0 if the index is not in the dictionary.
        """
        if index < 0 or index >= self.size:
            raise IndexError("Index out of range")

        return self.data.get(index, 0)

    def __str__(self):
        """
        Return a string representation of the sparse array.
        """
        return str(self.data)


# Example usage
if __name__ == "__main__":
    sparse_array = SparseArray(10)
    sparse_array.set(1, 5)
    sparse_array.set(3, 10)
    sparse_array.set(7, 15)

    print(sparse_array.get(1))  # Output: 5
    print(sparse_array.get(3))  # Output: 10
    print(sparse_array.get(7))  # Output: 15
    print(sparse_array.get(5))  # Output: 0

    print(sparse_array)  # Output: {1: 5, 3: 10, 7: 15}
```
