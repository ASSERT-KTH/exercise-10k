## Exercise
Write a program that prints out a table with user-provided data

## Reference Solution
```python
from tabulate import tabulate

def print_table(data, headers):
    """
    Prints a table with the provided data and headers.

    Args:
        data (list): A list of lists, where each sublist represents a row in the table.
        headers (list): A list of strings, where each string represents a column header.
    """
    print(tabulate(data, headers, tablefmt="grid"))

def main():
    # Get the number of rows and columns from the user
    num_rows = int(input("Enter the number of rows: "))
    num_cols = int(input("Enter the number of columns: "))

    # Get the column headers from the user
    headers = []
    for i in range(num_cols):
        headers.append(input(f"Enter column {i+1} header: "))

    # Get the data from the user
    data = []
    for i in range(num_rows):
        row = []
        for j in range(num_cols):
            row.append(input(f"Enter row {i+1}, column {j+1} data: "))
        data.append(row)

    # Print the table
    print_table(data, headers)

if __name__ == "__main__":
    main()
```
