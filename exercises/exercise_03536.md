## Exercise
Create a program that checks if a given string is a palindrome using a while loop.

## Reference Solution
```python
def is_palindrome(s):
    """
    Checks if a given string is a palindrome using a while loop.

    Args:
        s (str): The input string to be checked.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    # Remove any non-alphanumeric characters and convert to lowercase
    s = ''.join(c for c in s if c.isalnum()).lower()
    
    # Initialize two pointers, one at the start and one at the end of the string
    left = 0
    right = len(s) - 1
    
    # Use a while loop to compare characters from the start and end of the string
    while left < right:
        # If the characters at the current positions do not match, return False
        if s[left] != s[right]:
            return False
        # Move the pointers towards the center of the string
        left += 1
        right -= 1
    
    # If the loop completes without finding any mismatches, return True
    return True


# Example usage
def main():
    # Test the function with some examples
    print(is_palindrome("A man, a plan, a canal: Panama"))  # True
    print(is_palindrome("hello"))  # False
    print(is_palindrome("Was it a car or a cat I saw?"))  # True
    print(is_palindrome("Not a palindrome"))  # False


if __name__ == "__main__":
    main()
```
