## Exercise
Create a program that stores student grades in a dictionary. The dictionary should have student names as keys and lists of grades as values. The program should allow users to add students, add grades to existing students, and calculate the average grade for each student.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.student_grades = {}

    def add_student(self, name):
        """
        Add a new student to the dictionary.

        Args:
            name (str): The name of the student.
        """
        if name not in self.student_grades:
            self.student_grades[name] = []
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def add_grade(self, name, grade):
        """
        Add a grade to an existing student.

        Args:
            name (str): The name of the student.
            grade (float): The grade to be added.
        """
        if name in self.student_grades:
            self.student_grades[name].append(grade)
            print(f"Grade {grade} added to student '{name}'.")
        else:
            print(f"Student '{name}' does not exist. Please add the student first.")

    def calculate_average(self, name):
        """
        Calculate the average grade for a student.

        Args:
            name (str): The name of the student.

        Returns:
            float: The average grade of the student.
        """
        if name in self.student_grades:
            grades = self.student_grades[name]
            if grades:
                return sum(grades) / len(grades)
            else:
                print(f"Student '{name}' has no grades.")
                return None
        else:
            print(f"Student '{name}' does not exist.")
            return None

    def display_grades(self):
        """
        Display all student grades.
        """
        for name, grades in self.student_grades.items():
            print(f"Student: {name}")
            print(f"Grades: {grades}")
            average = self.calculate_average(name)
            if average is not None:
                print(f"Average: {average:.2f}")
            print()


def main():
    grades = StudentGrades()

    while True:
        print("1. Add student")
        print("2. Add grade")
        print("3. Calculate average")
        print("4. Display grades")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter student name: ")
            grades.add_student(name)
        elif choice == "2":
            name = input("Enter student name: ")
            grade = float(input("Enter grade: "))
            grades.add_grade(name, grade)
        elif choice == "3":
            name = input("Enter student name: ")
            average = grades.calculate_average(name)
            if average is not None:
                print(f"Average grade for {name}: {average:.2f}")
        elif choice == "4":
            grades.display_grades()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
