## Exercise
Create a program that allows the user to add, remove, and search for phone numbers.

## Reference Solution
```python
# phone_book.py

class PhoneBook:
    def __init__(self):
        self.phone_book = {}

    def add_number(self, name, phone_number):
        """Add a phone number to the phone book."""
        self.phone_book[name] = phone_number
        print(f"Phone number for {name} added successfully.")

    def remove_number(self, name):
        """Remove a phone number from the phone book."""
        if name in self.phone_book:
            del self.phone_book[name]
            print(f"Phone number for {name} removed successfully.")
        else:
            print(f"No phone number found for {name}.")

    def search_number(self, name):
        """Search for a phone number in the phone book."""
        if name in self.phone_book:
            print(f"Phone number for {name}: {self.phone_book[name]}")
        else:
            print(f"No phone number found for {name}.")

    def display_phone_book(self):
        """Display all phone numbers in the phone book."""
        if self.phone_book:
            print("Phone Book:")
            for name, phone_number in self.phone_book.items():
                print(f"{name}: {phone_number}")
        else:
            print("Phone book is empty.")


def main():
    phone_book = PhoneBook()

    while True:
        print("\nPhone Book Menu:")
        print("1. Add phone number")
        print("2. Remove phone number")
        print("3. Search phone number")
        print("4. Display phone book")
        print("5. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter name: ")
            phone_number = input("Enter phone number: ")
            phone_book.add_number(name, phone_number)
        elif choice == "2":
            name = input("Enter name: ")
            phone_book.remove_number(name)
        elif choice == "3":
            name = input("Enter name: ")
            phone_book.search_number(name)
        elif choice == "4":
            phone_book.display_phone_book()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
