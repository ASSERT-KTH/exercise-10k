## Exercise
Write a program that calculates the monthly payment for a mortgage based on the principal, interest rate, and term.

## Reference Solution
```python
def calculate_mortgage_payment(principal, annual_interest_rate, term_in_years):
    """
    Calculate the monthly payment for a mortgage.

    Args:
        principal (float): The initial amount borrowed.
        annual_interest_rate (float): The annual interest rate as a decimal.
        term_in_years (int): The term of the mortgage in years.

    Returns:
        float: The monthly payment amount.
    """
    # Convert annual interest rate to monthly interest rate
    monthly_interest_rate = annual_interest_rate / 12

    # Calculate the number of monthly payments
    num_payments = term_in_years * 12

    # Calculate the monthly payment
    monthly_payment = (principal * monthly_interest_rate * (1 + monthly_interest_rate) ** num_payments) / (
                (1 + monthly_interest_rate) ** num_payments - 1)

    return monthly_payment


def main():
    # Example usage
    principal = 200000  # Initial amount borrowed
    annual_interest_rate = 0.04  # Annual interest rate as a decimal
    term_in_years = 30  # Term of the mortgage in years

    monthly_payment = calculate_mortgage_payment(principal, annual_interest_rate, term_in_years)

    print(f"Monthly payment: ${monthly_payment:.2f}")


if __name__ == "__main__":
    main()
```
