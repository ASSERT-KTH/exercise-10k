## Exercise
Write a program that takes a time as input and converts it to a different time zone. Print the converted time to the console.

## Reference Solution
```python
from datetime import datetime
import pytz

def convert_time_zone(input_time, input_time_zone, output_time_zone):
    # Parse input time
    input_time = datetime.strptime(input_time, '%Y-%m-%d %H:%M:%S')
    
    # Set input time zone
    input_time = pytz.timezone(input_time_zone).localize(input_time)
    
    # Convert to output time zone
    output_time = input_time.astimezone(pytz.timezone(output_time_zone))
    
    # Return output time as string
    return output_time.strftime('%Y-%m-%d %H:%M:%S')

def main():
    input_time = input("Enter the time in the format 'YYYY-MM-DD HH:MM:SS': ")
    input_time_zone = input("Enter the input time zone (e.g. US/Pacific, US/Eastern, etc.): ")
    output_time_zone = input("Enter the output time zone (e.g. US/Pacific, US/Eastern, etc.): ")

    output_time = convert_time_zone(input_time, input_time_zone, output_time_zone)
    print("The converted time is: ", output_time)

if __name__ == "__main__":
    main()
```
