## Exercise
Write a program to validate a password based on certain rules (e.g. length, characters, digits).

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validates a password based on certain rules.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the rules
    rules = {
        "length": lambda p: len(p) >= 8,
        "uppercase": lambda p: any(c.isupper() for c in p),
        "lowercase": lambda p: any(c.islower() for c in p),
        "digit": lambda p: any(c.isdigit() for c in p),
        "special_char": lambda p: bool(re.search(r"[^A-Za-z0-9]", p)),
    }

    # Check each rule
    for rule, func in rules.items():
        if not func(password):
            print(f"Password does not meet the '{rule}' rule.")
            return False

    print("Password is valid.")
    return True

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
