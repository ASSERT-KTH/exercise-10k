## Exercise
Create a function to check if a given string is a substring of another string or not

## Reference Solution
```python
def check_substring(main_string, sub_string):
    """
    Checks if a given string is a substring of another string.

    Args:
        main_string (str): The main string to check against.
        sub_string (str): The substring to check for.

    Returns:
        bool: True if the substring is found, False otherwise.
    """
    return sub_string in main_string

# Example usage:
def main():
    main_str = "Hello, World!"
    sub_str = "World"
    result = check_substring(main_str, sub_str)
    print(f"'{sub_str}' is a substring of '{main_str}': {result}")

    main_str = "Hello, World!"
    sub_str = "Universe"
    result = check_substring(main_str, sub_str)
    print(f"'{sub_str}' is a substring of '{main_str}': {result}")

if __name__ == "__main__":
    main()
```
