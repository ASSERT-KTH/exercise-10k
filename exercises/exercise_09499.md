## Exercise
Create a program that uses a dictionary to store names and phone numbers. The program should allow users to add, remove, and search for phone numbers.

## Reference Solution
```python
class PhoneBook:
    def __init__(self):
        self.phone_book = {}

    def add_number(self):
        """Adds a new name and phone number to the phone book."""
        name = input("Enter the name: ")
        number = input("Enter the phone number: ")
        self.phone_book[name] = number
        print(f"Added {name} with phone number {number}.")

    def remove_number(self):
        """Removes a name and phone number from the phone book."""
        name = input("Enter the name to remove: ")
        if name in self.phone_book:
            del self.phone_book[name]
            print(f"Removed {name}.")
        else:
            print(f"{name} not found in phone book.")

    def search_number(self):
        """Searches for a phone number by name."""
        name = input("Enter the name to search for: ")
        if name in self.phone_book:
            print(f"{name}'s phone number is {self.phone_book[name]}.")
        else:
            print(f"{name} not found in phone book.")

    def display_phone_book(self):
        """Displays all names and phone numbers in the phone book."""
        if self.phone_book:
            print("Phone Book:")
            for name, number in self.phone_book.items():
                print(f"{name}: {number}")
        else:
            print("Phone book is empty.")

def main():
    phone_book = PhoneBook()
    while True:
        print("\nPhone Book Menu:")
        print("1. Add a number")
        print("2. Remove a number")
        print("3. Search for a number")
        print("4. Display phone book")
        print("5. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            phone_book.add_number()
        elif choice == "2":
            phone_book.remove_number()
        elif choice == "3":
            phone_book.search_number()
        elif choice == "4":
            phone_book.display_phone_book()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
