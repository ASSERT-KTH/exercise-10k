## Exercise
Write a program that takes a student's name and grade as input and uses if/else statements to determine their eligibility for a scholarship (e.g., based on grade).

## Reference Solution
```python
# Define a function to determine scholarship eligibility
def check_scholarship_eligibility(name, grade):
    """
    Determine if a student is eligible for a scholarship based on their grade.

    Args:
        name (str): The student's name.
        grade (float): The student's grade (out of 100).

    Returns:
        bool: True if the student is eligible for a scholarship, False otherwise.
    """

    # Check if the grade is within the valid range (0-100)
    if grade < 0 or grade > 100:
        return False

    # Define the scholarship eligibility criteria
    # For this example, we'll assume a student needs a grade of 85 or higher to be eligible
    if grade >= 85:
        return True
    else:
        return False


# Define a main function to get user input and display the result
def main():
    # Get the student's name and grade from the user
    name = input("Please enter the student's name: ")
    while True:
        try:
            grade = float(input("Please enter the student's grade (out of 100): "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Check the student's eligibility for a scholarship
    eligible = check_scholarship_eligibility(name, grade)

    # Display the result
    if eligible:
        print(f"Congratulations, {name}! You are eligible for a scholarship with a grade of {grade}.")
    else:
        print(f"Sorry, {name}. You are not eligible for a scholarship with a grade of {grade}.")


# Run the main function
if __name__ == "__main__":
    main()
```
