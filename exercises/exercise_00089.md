## Exercise
Given a list of integers, find the most frequent element

## Reference Solution
```python
from collections import Counter

def most_frequent_element(lst):
    """
    Find the most frequent element in a list of integers.

    Args:
        lst (list): A list of integers.

    Returns:
        int: The most frequent element.
    """
    # Count the frequency of each element
    freq = Counter(lst)

    # Find the element with the maximum frequency
    most_frequent = max(freq, key=freq.get)

    return most_frequent

# Example usage
numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
print(most_frequent_element(numbers))  # Output: 4
```

```python
from collections import Counter

def most_frequent_elements(lst):
    """
    Find all elements with the maximum frequency in a list of integers.

    Args:
        lst (list): A list of integers.

    Returns:
        list: A list of elements with the maximum frequency.
    """
    # Count the frequency of each element
    freq = Counter(lst)

    # Find the maximum frequency
    max_freq = max(freq.values())

    # Find all elements with the maximum frequency
    most_frequent = [elem for elem, count in freq.items() if count == max_freq]

    return most_frequent

# Example usage
numbers = [1, 2, 2, 3, 3]
print(most_frequent_elements(numbers))  # Output: [2, 3]
```
