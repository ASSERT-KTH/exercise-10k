## Exercise
Create a program to perform advanced arithmetic operations (e.g., logarithms, trigonometry) using a library or built-in functions.

## Reference Solution
```python
import math

def logarithm(base, number):
    """
    Calculate the logarithm of a number with a given base.

    Args:
        base (float): The base of the logarithm.
        number (float): The number to calculate the logarithm for.

    Returns:
        float: The logarithm of the number with the given base.
    """
    return math.log(number, base)

def natural_logarithm(number):
    """
    Calculate the natural logarithm of a number.

    Args:
        number (float): The number to calculate the natural logarithm for.

    Returns:
        float: The natural logarithm of the number.
    """
    return math.log(number)

def trigonometric_operation(operation, angle):
    """
    Perform a trigonometric operation (sine, cosine, or tangent) on an angle.

    Args:
        operation (str): The trigonometric operation to perform (sin, cos, or tan).
        angle (float): The angle in radians.

    Returns:
        float: The result of the trigonometric operation.
    """
    if operation == "sin":
        return math.sin(angle)
    elif operation == "cos":
        return math.cos(angle)
    elif operation == "tan":
        return math.tan(angle)
    else:
        raise ValueError("Invalid operation")

def exponential(base, exponent):
    """
    Calculate the exponential of a base raised to a power.

    Args:
        base (float): The base of the exponential.
        exponent (float): The exponent to which the base is raised.

    Returns:
        float: The exponential of the base raised to the power.
    """
    return math.pow(base, exponent)

def main():
    print("Advanced Arithmetic Operations Program")
    print("-------------------------------------")

    while True:
        print("1. Logarithm")
        print("2. Natural Logarithm")
        print("3. Trigonometric Operation")
        print("4. Exponential")
        print("5. Quit")

        choice = input("Choose an operation: ")

        if choice == "1":
            base = float(input("Enter the base: "))
            number = float(input("Enter the number: "))
            result = logarithm(base, number)
            print(f"The logarithm of {number} with base {base} is {result}")
        elif choice == "2":
            number = float(input("Enter the number: "))
            result = natural_logarithm(number)
            print(f"The natural logarithm of {number} is {result}")
        elif choice == "3":
            operation = input("Enter the operation (sin, cos, or tan): ")
            angle = float(input("Enter the angle in radians: "))
            result = trigonometric_operation(operation, angle)
            print(f"The {operation} of {angle} radians is {result}")
        elif choice == "4":
            base = float(input("Enter the base: "))
            exponent = float(input("Enter the exponent: "))
            result = exponential(base, exponent)
            print(f"{base} raised to the power of {exponent} is {result}")
        elif choice == "5":
            break
        else:
            print("Invalid choice")

if __name__ == "__main__":
    main()
```
