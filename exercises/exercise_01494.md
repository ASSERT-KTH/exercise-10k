## Exercise
Write a program that demonstrates the use of bytes operations in programming.

## Reference Solution
```python
def main():
    # Creating bytes objects
    bytes_obj1 = b'Hello'
    bytes_obj2 = b'World'

    print("Bytes Object 1:", bytes_obj1)
    print("Bytes Object 2:", bytes_obj2)

    # Concatenating bytes objects
    concatenated_bytes = bytes_obj1 + bytes_obj2
    print("Concatenated Bytes:", concatenated_bytes)

    # Slicing bytes objects
    sliced_bytes = bytes_obj1[1:3]
    print("Sliced Bytes:", sliced_bytes)

    # Comparing bytes objects
    print("Bytes Object 1 == Bytes Object 2:", bytes_obj1 == bytes_obj2)
    print("Bytes Object 1 != Bytes Object 2:", bytes_obj1 != bytes_obj2)

    # Encoding and decoding strings to/from bytes
    string = "Hello, World!"
    encoded_bytes = string.encode('utf-8')
    print("Encoded Bytes:", encoded_bytes)
    decoded_string = encoded_bytes.decode('utf-8')
    print("Decoded String:", decoded_string)

    # Creating bytes from a list of integers
    integers = [72, 101, 108, 108, 111]
    bytes_from_integers = bytes(integers)
    print("Bytes from Integers:", bytes_from_integers)

if __name__ == "__main__":
    main()
```
