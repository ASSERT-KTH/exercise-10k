## Exercise
Write a program to calculate the total cost of a meal including a tip.

## Reference Solution
```python
def calculate_total_cost(meal_cost, tip_percentage):
    """
    Calculate the total cost of a meal, including a tip.

    Args:
        meal_cost (float): The cost of the meal.
        tip_percentage (float): The desired tip percentage.

    Returns:
        float: The total cost of the meal, including the tip.
    """
    tip_amount = meal_cost * (tip_percentage / 100)
    total_cost = meal_cost + tip_amount
    return total_cost

def main():
    print("Meal Cost Calculator")
    print("--------------------")

    meal_cost = float(input("Enter the cost of the meal: $"))
    tip_percentage = float(input("Enter the desired tip percentage (e.g. 15, 20): "))

    total_cost = calculate_total_cost(meal_cost, tip_percentage)

    print(f"Meal cost: ${meal_cost:.2f}")
    print(f"Tip ({tip_percentage}%): ${total_cost - meal_cost:.2f}")
    print(f"Total cost: ${total_cost:.2f}")

if __name__ == "__main__":
    main()
```
