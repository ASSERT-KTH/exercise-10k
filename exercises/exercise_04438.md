## Exercise
Create a program that manages an address book using a for loop to display contacts and a while loop to handle user input.

## Reference Solution
```python
class AddressBook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self):
        name = input("Enter contact name: ")
        number = input("Enter contact number: ")
        self.contacts[name] = number
        print(f"Contact {name} added successfully.")

    def display_contacts(self):
        if not self.contacts:
            print("No contacts in address book.")
        else:
            print("Address Book:")
            for name, number in self.contacts.items():
                print(f"Name: {name}, Number: {number}")

    def delete_contact(self):
        name = input("Enter contact name to delete: ")
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact {name} deleted successfully.")
        else:
            print(f"No contact found with name {name}.")

    def search_contact(self):
        name = input("Enter contact name to search: ")
        if name in self.contacts:
            print(f"Name: {name}, Number: {self.contacts[name]}")
        else:
            print(f"No contact found with name {name}.")

    def update_contact(self):
        name = input("Enter contact name to update: ")
        if name in self.contacts:
            number = input("Enter new contact number: ")
            self.contacts[name] = number
            print(f"Contact {name} updated successfully.")
        else:
            print(f"No contact found with name {name}.")

def main():
    address_book = AddressBook()
    while True:
        print("\nAddress Book Menu:")
        print("1. Add Contact")
        print("2. Display Contacts")
        print("3. Delete Contact")
        print("4. Search Contact")
        print("5. Update Contact")
        print("6. Exit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            address_book.add_contact()
        elif choice == "2":
            address_book.display_contacts()
        elif choice == "3":
            address_book.delete_contact()
        elif choice == "4":
            address_book.search_contact()
        elif choice == "5":
            address_book.update_contact()
        elif choice == "6":
            print("Exiting address book.")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
