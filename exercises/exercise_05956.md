## Exercise
Develop a program that draws different shapes (e.g., square, circle, triangle) based on user input. The program should ask for the shape type, the size, and then print out the drawn shape.

## Reference Solution
```python
import math

def draw_square(size):
    """Draws a square of a given size."""
    for i in range(size):
        for j in range(size):
            print("*", end=" ")
        print()

def draw_circle(size):
    """Draws a circle of a given size."""
    radius = size // 2
    for y in range(size):
        for x in range(size):
            distance = math.sqrt((x - radius) ** 2 + (y - radius) ** 2)
            if distance <= radius:
                print("*", end=" ")
            else:
                print(" ", end=" ")
        print()

def draw_triangle(size):
    """Draws a triangle of a given size."""
    for i in range(size):
        for j in range(i + 1):
            print("*", end=" ")
        print()

def main():
    print("Shape Drawing Program")
    print("1. Square")
    print("2. Circle")
    print("3. Triangle")

    shape_type = input("Enter the shape type (1/2/3): ")
    size = int(input("Enter the size: "))

    if shape_type == "1":
        draw_square(size)
    elif shape_type == "2":
        draw_circle(size)
    elif shape_type == "3":
        draw_triangle(size)
    else:
        print("Invalid shape type.")

if __name__ == "__main__":
    main()
```
