## Exercise
Write a program that takes a password as input from the user and checks if it meets certain criteria.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password against the following criteria:
    - At least 8 characters long
    - At least one uppercase letter
    - At least one lowercase letter
    - At least one digit
    - At least one special character

    Args:
        password (str): The password to validate

    Returns:
        bool: True if the password is valid, False otherwise
    """
    # Define the regular expression patterns
    patterns = {
        "uppercase": r"[A-Z]",
        "lowercase": r"[a-z]",
        "digit": r"\d",
        "special_char": r"[^A-Za-z0-9]"
    }

    # Check the length of the password
    if len(password) < 8:
        return False

    # Check each pattern
    for pattern in patterns.values():
        if not re.search(pattern, password):
            return False

    # If all checks pass, the password is valid
    return True

def main():
    # Get the password from the user
    password = input("Enter a password: ")

    # Validate the password
    if validate_password(password):
        print("Password is valid")
    else:
        print("Password is not valid. Please ensure it meets the following criteria:")
        print("* At least 8 characters long")
        print("* At least one uppercase letter")
        print("* At least one lowercase letter")
        print("* At least one digit")
        print("* At least one special character")

if __name__ == "__main__":
    main()
```
