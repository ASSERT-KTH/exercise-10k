## Exercise
Unflatten a single-level dictionary into a nested dictionary

## Reference Solution
```python
def unflatten(dictionary, delimiter='.'):
    """
    Unflattens a single-level dictionary into a nested dictionary.
    
    Args:
        dictionary (dict): The flat dictionary to unflatten.
        delimiter (str): The delimiter used in the keys to separate levels. Defaults to '.'.
    
    Returns:
        dict: The unflattened, nested dictionary.
    """
    unflattened_dict = {}
    for key, value in dictionary.items():
        parts = key.split(delimiter)
        current_dict = unflattened_dict
        for part in parts[:-1]:
            if part not in current_dict:
                current_dict[part] = {}
            current_dict = current_dict[part]
        current_dict[parts[-1]] = value
    return unflattened_dict

# Example usage:
flat_dict = {
    'a.b.c': 'value1',
    'x.y.z': 'value2',
    'foo.bar': 'value3'
}

nested_dict = unflatten(flat_dict)
print(nested_dict)
```
