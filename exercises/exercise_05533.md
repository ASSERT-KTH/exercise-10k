## Exercise
Write a program to perform bitwise AND, OR, and XOR operations on two integers.

## Reference Solution
```python
# Define a function to perform bitwise operations
def bitwise_operations(num1, num2):
    """
    This function performs bitwise AND, OR, and XOR operations on two integers.

    Args:
        num1 (int): The first integer.
        num2 (int): The second integer.

    Returns:
        A dictionary containing the results of the bitwise operations.
    """
    # Perform bitwise AND operation
    and_result = num1 & num2

    # Perform bitwise OR operation
    or_result = num1 | num2

    # Perform bitwise XOR operation
    xor_result = num1 ^ num2

    # Return the results as a dictionary
    return {
        "AND": and_result,
        "OR": or_result,
        "XOR": xor_result
    }

# Example usage
if __name__ == "__main__":
    num1 = 5  # Binary: 101
    num2 = 3  # Binary: 011

    results = bitwise_operations(num1, num2)

    print(f"Bitwise AND of {num1} and {num2}: {results['AND']}")
    print(f"Bitwise OR of {num1} and {num2}: {results['OR']}")
    print(f"Bitwise XOR of {num1} and {num2}: {results['XOR']}")
```
