## Exercise
Write a program to demonstrate implicit type conversion, including conversions between numbers and strings.

## Reference Solution
```python
# Implicit type conversion between numbers
def implicit_conversion_numbers():
    """
    This function demonstrates implicit type conversion between integers and floats.
    """
    # Initialize variables
    int_num = 5
    float_num = 3.5

    # Implicit conversion from int to float
    result = int_num + float_num
    print("Implicit conversion from int to float: ", result, type(result))

    # Implicit conversion from float to complex
    complex_num = 2 + 3j
    result = float_num + complex_num
    print("Implicit conversion from float to complex: ", result, type(result))

# Implicit type conversion between strings and numbers
def implicit_conversion_strings():
    """
    This function demonstrates implicit type conversion between strings and numbers.
    """
    # Initialize variables
    str_num = "10"
    int_num = 5

    # Implicit conversion from string to int
    # Note: This will throw an error if the string cannot be converted to an integer
    try:
        result = int(str_num) + int_num
        print("Implicit conversion from string to int: ", result, type(result))
    except ValueError:
        print("Error: String cannot be converted to an integer.")

    # Implicit conversion from string to float
    # Note: This will throw an error if the string cannot be converted to a float
    str_num = "10.5"
    try:
        result = float(str_num) + int_num
        print("Implicit conversion from string to float: ", result, type(result))
    except ValueError:
        print("Error: String cannot be converted to a float.")

# Main function
def main():
    print("Implicit Type Conversion Between Numbers:")
    implicit_conversion_numbers()
    print("\nImplicit Type Conversion Between Strings and Numbers:")
    implicit_conversion_strings()

# Run the program
if __name__ == "__main__":
    main()
```
