## Exercise
Check if a variable is null or undefined and handle it correctly

## Reference Solution
```python
// Function to handle null or undefined variables
function handleNullUndefined(variable) {
    if (variable === null || variable === undefined) {
        // Provide a default value
        variable = "Default Value";
    }
    return variable;
}

// Example usage
let myVariable = null;
let result = handleNullUndefined(myVariable);
console.log(result); // Output: "Default Value"

// Example with a non-null variable
myVariable = "Hello, World!";
result = handleNullUndefined(myVariable);
console.log(result); // Output: "Hello, World!"
```
