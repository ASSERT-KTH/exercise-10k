## Exercise
Write a function that takes a dictionary as input and returns a copy of the dictionary

## Reference Solution
```python
import copy

def copy_dictionary(input_dict):
    """
    Creates a deep copy of a given dictionary.
    
    Args:
        input_dict (dict): The dictionary to be copied.
    
    Returns:
        dict: A deep copy of the input dictionary.
    """
    return copy.deepcopy(input_dict)

# Example usage
if __name__ == "__main__":
    original_dict = {
        "key1": "value1",
        "key2": [1, 2, 3],
        "key3": {"nested_key1": "nested_value1"}
    }
    
    copied_dict = copy_dictionary(original_dict)
    
    # Modify the original dictionary to verify the copy
    original_dict["key1"] = "new_value1"
    original_dict["key2"].append(4)
    original_dict["key3"]["nested_key1"] = "new_nested_value1"
    
    print("Original Dictionary:")
    print(original_dict)
    print("\nCopied Dictionary:")
    print(copied_dict)
```
