## Exercise
Create a program that checks the data type of a given variable and prints out whether it is an integer, float, or string. The program should also handle cases where the variable is a list, dictionary, or boolean.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a given variable and prints out the type.
    
    Args:
        var (any): The variable to check the data type of.
    """
    if isinstance(var, int):
        print("The variable is an integer.")
    elif isinstance(var, float):
        print("The variable is a float.")
    elif isinstance(var, str):
        print("The variable is a string.")
    elif isinstance(var, list):
        print("The variable is a list.")
    elif isinstance(var, dict):
        print("The variable is a dictionary.")
    elif isinstance(var, bool):
        print("The variable is a boolean.")
    else:
        print("The variable is of an unknown data type.")

def main():
    # Example usage
    var1 = 10
    var2 = 3.14
    var3 = "Hello, World!"
    var4 = [1, 2, 3]
    var5 = {"name": "John", "age": 30}
    var6 = True

    print("Variable 1:")
    check_data_type(var1)
    print("Variable 2:")
    check_data_type(var2)
    print("Variable 3:")
    check_data_type(var3)
    print("Variable 4:")
    check_data_type(var4)
    print("Variable 5:")
    check_data_type(var5)
    print("Variable 6:")
    check_data_type(var6)

if __name__ == "__main__":
    main()
```
