## Exercise
Write a program to map a given dictionary to a new dictionary with different keys or values.

## Reference Solution
```python
def map_dictionary(input_dict, key_map=None, value_map=None):
    """
    Maps a given dictionary to a new dictionary with different keys or values.

    Args:
        input_dict (dict): The input dictionary to be mapped.
        key_map (dict, optional): A dictionary mapping old keys to new keys. Defaults to None.
        value_map (dict, optional): A dictionary mapping old values to new values. Defaults to None.

    Returns:
        dict: The mapped dictionary.
    """
    output_dict = {}

    # Iterate over the input dictionary
    for key, value in input_dict.items():
        # Check if key mapping is provided
        if key_map and key in key_map:
            new_key = key_map[key]
        else:
            new_key = key

        # Check if value mapping is provided
        if value_map and value in value_map:
            new_value = value_map[value]
        else:
            new_value = value

        # Add the mapped key-value pair to the output dictionary
        output_dict[new_key] = new_value

    return output_dict


# Example usage
if __name__ == "__main__":
    input_dict = {"a": 1, "b": 2, "c": 3}

    # Mapping keys
    key_map = {"a": "x", "b": "y", "c": "z"}
    output_dict = map_dictionary(input_dict, key_map=key_map)
    print("Mapped dictionary with new keys:", output_dict)

    # Mapping values
    value_map = {1: "one", 2: "two", 3: "three"}
    output_dict = map_dictionary(input_dict, value_map=value_map)
    print("Mapped dictionary with new values:", output_dict)

    # Mapping both keys and values
    output_dict = map_dictionary(input_dict, key_map=key_map, value_map=value_map)
    print("Mapped dictionary with new keys and values:", output_dict)
```
