## Exercise
Create and print examples of composite data types such as arrays and objects.

## Reference Solution

#!/bin/bash

# Example of a tuple (array in Bash)
tuple_example=(1 2 3 4 5)
echo "${tuple_example[@]}"

# Example of a dictionary ( associative array in Bash )
declare -A dict_example
dict_example["name"]="John"
dict_example["age"]=30
dict_example["city"]="New York"

for key in "${!dict_example[@]}"; do
  echo "$key: ${dict_example[$key]}"
done

# Example of a set (array in Bash with unique values)
set_example=(1 2 3 4 5)
echo "${set_example[@]}"

# Example of a list (array in Bash)
list_example=(1 2 3 4 5)
echo "${list_example[@]}"

# Example of a dictionary with list as value ( associative array with array as value in Bash )
declare -A dict_with_list_example
dict_with_list_example["name"]="John"
dict_with_list_example["age"]="30"
dict_with_list_example["hobbies"]="reading swimming cycling"

for key in "${!dict_with_list_example[@]}"; do
  echo "$key: ${dict_with_list_example[$key]}"
done

# Example of a set of tuples (array of arrays in Bash)
set_of_tuples_example=("1 2" "3 4" "5 6")
for tuple in "${set_of_tuples_example[@]}"; do
  echo "($tuple)"
done

# Example of a dictionary with tuple as value ( associative array with array as value in Bash )

for key in "${!dict_with_tuple_example[@]}"; do

done
