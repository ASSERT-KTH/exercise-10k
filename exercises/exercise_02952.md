## Exercise
Write a program to find the shortest path between two vertices in a graph using Dijkstra's algorithm.

## Reference Solution
```python
import sys
import heapq

def dijkstra(graph, start):
    """
    Find the shortest path between two vertices in a graph using Dijkstra's algorithm.

    Args:
        graph (dict): The graph represented as a dictionary where each key is a vertex and each value is a dictionary of its neighbors and their distances.
        start (str): The starting vertex.

    Returns:
        dict: A dictionary where each key is a vertex and each value is its shortest distance from the start vertex.
    """

    # Create a dictionary to store the shortest distances
    distances = {vertex: sys.maxsize for vertex in graph}
    distances[start] = 0

    # Create a priority queue to store the vertices to be processed
    pq = [(0, start)]

    while pq:
        # Extract the vertex with the shortest distance from the priority queue
        current_distance, current_vertex = heapq.heappop(pq)

        # If the current distance is greater than the already found distance, skip this vertex
        if current_distance > distances[current_vertex]:
            continue

        # Iterate over the neighbors of the current vertex
        for neighbor, weight in graph[current_vertex].items():
            distance = current_distance + weight

            # If a shorter path to the neighbor is found, update its distance
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))

    return distances

def find_shortest_path(graph, start, end):
    """
    Find the shortest path between two vertices in a graph using Dijkstra's algorithm.

    Args:
        graph (dict): The graph represented as a dictionary where each key is a vertex and each value is a dictionary of its neighbors and their distances.
        start (str): The starting vertex.
        end (str): The ending vertex.

    Returns:
        list: A list of vertices representing the shortest path from the start vertex to the end vertex.
    """

    # Find the shortest distances from the start vertex to all other vertices
    distances = dijkstra(graph, start)

    # If there is no path from the start vertex to the end vertex, return None
    if distances[end] == sys.maxsize:
        return None

    # Build the shortest path by backtracking from the end vertex to the start vertex
    path = []
    current_vertex = end
    while current_vertex != start:
        path.append(current_vertex)
        for neighbor, weight in graph[current_vertex].items():
            if distances[current_vertex] - weight == distances[neighbor]:
                current_vertex = neighbor
                break
    path.append(start)
    path.reverse()

    return path

# Example usage:
graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

start_vertex = 'A'
end_vertex = 'D'

distances = dijkstra(graph, start_vertex)
print("Shortest distances from {}:".format(start_vertex))
for vertex, distance in distances.items():
    print("{}: {}".format(vertex, distance))

shortest_path = find_shortest_path(graph, start_vertex, end_vertex)
print("Shortest path from {} to {}: {}".format(start_vertex, end_vertex, shortest_path))
```
