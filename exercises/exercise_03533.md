## Exercise
Create a program that counts the number of words in a given string using a for loop.

## Reference Solution
```python
def count_words(input_string):
    """
    Counts the number of words in a given string.

    Args:
        input_string (str): The input string to count words from.

    Returns:
        int: The number of words in the input string.
    """
    word_count = 0
    in_word = False

    for char in input_string:
        # Check if the character is a space
        if char.isspace():
            in_word = False
        # If the character is not a space and we're not in a word, increment the word count
        elif not in_word:
            word_count += 1
            in_word = True

    return word_count


def main():
    # Example usage
    input_string = "Hello World this is a test string"
    print(f"Number of words: {count_words(input_string)}")


if __name__ == "__main__":
    main()
```
