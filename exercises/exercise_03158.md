## Exercise
Write a function to check if a string is a subsequence of another string.

## Reference Solution

def is_subsequence(main_string: str, subsequence: str) -> bool:
    """
    Checks if a given string is a subsequence of another string.

    Args:
    main_string (str): The main string to check against.
    subsequence (str): The string to check if it's a subsequence.

    Returns:
    bool: True if the string is a subsequence, False otherwise.
    """
    # Initialize two pointers, one for each string
    main_pointer = 0
    sub_pointer = 0

    # Iterate through the main string
    while main_pointer < len(main_string) and sub_pointer < len(subsequence):
        # If the current characters match, move to the next character in both strings
        if main_string[main_pointer] == subsequence[sub_pointer]:
            sub_pointer += 1
        # Move to the next character in the main string
        main_pointer += 1

    # If we've reached the end of the subsequence, it's a subsequence of the main string
    return sub_pointer == len(subsequence)


# Example usage

    main_string = "abcdef"
    subsequence = "ace"
    print(is_subsequence(main_string, subsequence))  # Output: True

    main_string = "abcdef"
    subsequence = "aec"
    print(is_subsequence(main_string, subsequence))  # Output: False
