## Exercise
Write a program that simulates a coin flip and keeps track of the number of heads and tails using a while loop.

## Reference Solution
```python
import random

def coin_flip():
    """Simulate a coin flip and return the result."""
    return "Heads" if random.randint(0, 1) == 0 else "Tails"

def main():
    """Simulate coin flips and keep track of the number of heads and tails."""
    heads = 0
    tails = 0
    flip_count = 0

    while True:
        print("\n1. Flip Coin")
        print("2. View Results")
        print("3. Quit")
        
        choice = input("Choose an option: ")

        if choice == "1":
            result = coin_flip()
            print(f"\nYou flipped: {result}")
            if result == "Heads":
                heads += 1
            else:
                tails += 1
            flip_count += 1
        elif choice == "2":
            print(f"\nHeads: {heads}")
            print(f"Tails: {tails}")
            print(f"Total Flips: {flip_count}")
            if flip_count > 0:
                print(f"Heads Percentage: {(heads / flip_count) * 100:.2f}%")
                print(f"Tails Percentage: {(tails / flip_count) * 100:.2f}%")
            else:
                print("No flips have been made yet.")
        elif choice == "3":
            print("\nGoodbye!")
            break
        else:
            print("\nInvalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
