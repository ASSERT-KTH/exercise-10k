Apply a function to each value in a dictionary

def square(x):
    """Returns the square of x."""
    return x ** 2

# Example dictionary
data = {'a': 1, 'b': 2, 'c': 3}

# Apply the square function to each value in the dictionary
squared_data = {key: square(value) for key, value in data.items()}

print(squared_data)  # Output: {'a': 1, 'b': 4, 'c': 9}
```

def square(x):
    """Returns the square of x."""
    return x ** 2

# Example dictionary
data = {'a': 1, 'b': 2, 'c': 3}

# Apply the square function to each value in the dictionary
squared_data = dict(map(lambda item: (item[0], square(item[1])), data.items()))

print(squared_data)  # Output: {'a': 1, 'b': 4, 'c': 9}
```
