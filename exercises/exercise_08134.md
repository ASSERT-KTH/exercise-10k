## Exercise
Create a program that checks if a given string is a palindrome using if/else statements

## Reference Solution
```python
def is_palindrome(s):
    """
    Checks if a given string is a palindrome.

    Args:
        s (str): The input string to check.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    s = ''.join(c for c in s if c.isalnum()).lower()  # Remove non-alphanumeric characters and convert to lowercase
    reversed_s = s[::-1]  # Reverse the string
    if s == reversed_s:
        return True
    else:
        return False

def main():
    # Example usage
    strings_to_check = [
        "radar",
        "hello",
        "A man, a plan, a canal: Panama",
        "Not a palindrome"
    ]

    for s in strings_to_check:
        if is_palindrome(s):
            print(f'"{s}" is a palindrome.')
        else:
            print(f'"{s}" is not a palindrome.')

if __name__ == "__main__":
    main()
```
