## Exercise
Write a program that plays a game of Rock, Paper, Scissors between two players and determines the winner based on the game's rules

## Reference Solution
```python
import random

def play_game():
    while True:
        user1 = input("Enter your choice (rock/paper/scissors): ").lower()
        user2 = random.choice(['rock', 'paper', 'scissors'])

        print(f"\nYou chose: {user1}")
        print(f"Computer chose: {user2}\n")

        if user1 == user2:
            print(f"Both players selected {user1}. It's a tie!")
        elif user1 == 'rock':
            if user2 == 'scissors':
                print("Rock smashes scissors! You win!")
            else:
                print("Paper covers rock! You lose.")
        elif user1 == 'paper':
            if user2 == 'rock':
                print("Paper covers rock! You win!")
            else:
                print("Scissors cuts paper! You lose.")
        elif user1 == 'scissors':
            if user2 == 'paper':
                print("Scissors cuts paper! You win!")
            else:
                print("Rock smashes scissors! You lose.")

        play_again = input("Play again? (yes/no): ").lower()
        if play_again != "yes":
            break

play_game()
```
