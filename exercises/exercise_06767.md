## Exercise
Write a program to create a calendar program that displays the current date and allows users to navigate through different dates.

## Reference Solution
```python
import calendar
import datetime

class CalendarProgram:
    def __init__(self):
        self.current_date = datetime.date.today()

    def display_calendar(self):
        print(calendar.month(self.current_date.year, self.current_date.month))

    def navigate_date(self):
        while True:
            print("\n1. Go to previous month")
            print("2. Go to next month")
            print("3. Go to previous year")
            print("4. Go to next year")
            print("5. Go to specific date")
            print("6. Exit")
            choice = input("Enter your choice: ")

            if choice == "1":
                if self.current_date.month == 1:
                    self.current_date = self.current_date.replace(year=self.current_date.year - 1, month=12)
                else:
                    self.current_date = self.current_date.replace(month=self.current_date.month - 1)
            elif choice == "2":
                if self.current_date.month == 12:
                    self.current_date = self.current_date.replace(year=self.current_date.year + 1, month=1)
                else:
                    self.current_date = self.current_date.replace(month=self.current_date.month + 1)
            elif choice == "3":
                self.current_date = self.current_date.replace(year=self.current_date.year - 1)
            elif choice == "4":
                self.current_date = self.current_date.replace(year=self.current_date.year + 1)
            elif choice == "5":
                year = int(input("Enter year: "))
                month = int(input("Enter month: "))
                day = int(input("Enter day: "))
                self.current_date = datetime.date(year, month, day)
            elif choice == "6":
                break
            else:
                print("Invalid choice. Please try again.")

            self.display_calendar()

    def run(self):
        print("Current Date: ", self.current_date)
        self.display_calendar()
        self.navigate_date()


if __name__ == "__main__":
    calendar_program = CalendarProgram()
    calendar_program.run()
```
