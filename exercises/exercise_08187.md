## Exercise
Write a program that plays Rock Paper Scissors using a while loop and asks the user for their choice.

## Reference Solution
```python
import random

def get_computer_choice():
    """Get the computer's choice by selecting a random option."""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def get_user_choice():
    """Get the user's choice by asking for input."""
    while True:
        user_choice = input("Enter your choice (rock, paper, or scissors): ").lower()
        if user_choice in ['rock', 'paper', 'scissors']:
            return user_choice
        print("Invalid choice. Please try again.")

def determine_winner(user_choice, computer_choice):
    """Determine the winner based on the user's and computer's choices."""
    if user_choice == computer_choice:
        return "It's a tie!"
    if (user_choice == 'rock' and computer_choice == 'scissors') or \
       (user_choice == 'paper' and computer_choice == 'rock') or \
       (user_choice == 'scissors' and computer_choice == 'paper'):
        return "You win!"
    return "Computer wins!"

def main():
    """Play Rock Paper Scissors using a while loop."""
    play_again = 'y'
    while play_again.lower() == 'y':
        user_choice = get_user_choice()
        computer_choice = get_computer_choice()
        print(f"\nYou chose: {user_choice}")
        print(f"Computer chose: {computer_choice}\n")
        print(determine_winner(user_choice, computer_choice))
        play_again = input("Do you want to play again? (y/n): ")
        while play_again.lower() not in ['y', 'n']:
            play_again = input("Invalid input. Please enter 'y' or 'n': ")

if __name__ == "__main__":
    main()
```
