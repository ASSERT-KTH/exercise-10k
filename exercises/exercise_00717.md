## Exercise
Write a function to apply a function to each value in a dictionary

## Reference Solution
```python
def apply_function_to_dict_values(dictionary, func):
    """
    Applies a given function to each value in a dictionary.

    Args:
        dictionary (dict): The dictionary whose values will be transformed.
        func (function): The function to be applied to each value.

    Returns:
        dict: A new dictionary with the transformed values.
    """
    return {key: func(value) for key, value in dictionary.items()}

# Example usage:
if __name__ == "__main__":
    # Define a sample dictionary
    sample_dict = {"a": 1, "b": 2, "c": 3}
    
    # Define a function to square a number
    def square(x):
        return x ** 2
    
    # Apply the function to the dictionary values
    transformed_dict = apply_function_to_dict_values(sample_dict, square)
    
    # Print the original and transformed dictionaries
    print("Original Dictionary:", sample_dict)
    print("Transformed Dictionary:", transformed_dict)
```
