## Exercise
Create a program that takes two numbers and an operator as input and outputs the result of the operation.

## Reference Solution
```python
# Import the operator module for mapping operator symbols to their corresponding functions
import operator

# Dictionary mapping operator symbols to their corresponding functions
operators = {
    '+': operator.add,
    '-': operator.sub,
    '*': operator.mul,
    '/': operator.truediv,
    '%': operator.mod,
    '**': operator.pow
}

def calculate(num1, num2, op):
    """
    Perform an arithmetic operation on two numbers.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.
    op (str): The operator symbol.

    Returns:
    float: The result of the operation.

    Raises:
    ValueError: If the operator is not supported.
    ZeroDivisionError: If the operator is '/' or '%' and the second number is zero.
    """

    # Check if the operator is supported
    if op not in operators:
        raise ValueError("Unsupported operator")

    # Perform the operation
    try:
        result = operators[op](num1, num2)
    except ZeroDivisionError:
        raise ZeroDivisionError("Cannot divide by zero")

    return result

def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    op = input("Enter the operator (+, -, *, /, %, **): ")

    # Perform the operation and display the result
    try:
        result = calculate(num1, num2, op)
        print(f"{num1} {op} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
