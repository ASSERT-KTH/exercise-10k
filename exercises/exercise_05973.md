## Exercise
Write a program that takes an array and a modification function as input, and applies the modification function to each element of the array.

## Reference Solution
```python
#include <iostream>
#include <vector>

template <typename T>
void modifyArray(std::vector<T>& array, T (*modify)(T)) {
    for (int i = 0; i < array.size(); i++) {
        array[i] = modify(array[i]);
    }
}

int modifyFunction(int value) {
    return value * 2;
}

int modifyFunction2(int value) {
    return value + 2;
}

int main() {
    std::vector<int> array = {1, 2, 3};
    modifyArray(array, modifyFunction);
    for (int i = 0; i < array.size(); i++) {
        std::cout << array[i] << " ";
    }
    std::cout << std::endl;

    array = {1, 2, 3};
    modifyArray(array, modifyFunction2);
    for (int i = 0; i < array.size(); i++) {
        std::cout << array[i] << " ";
    }
    std::cout << std::endl;

    array = {1, 2, 3};
    modifyArray(array, modifyFunction);
    for (int i = 0; i < array.size(); i++) {
        std::cout << array[i] << " ";
    }
    std::cout << std::endl;

    return 0;
}
```
