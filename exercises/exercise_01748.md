## Exercise
Write a program to calculate the mean, median, and mode of an array of integers.

## Reference Solution
```python
import statistics
from collections import Counter

def calculate_statistics(arr):
    """
    Calculate the mean, median, and mode of an array of integers.

    Args:
        arr (list): A list of integers.

    Returns:
        tuple: A tuple containing the mean, median, and mode.
    """
    # Calculate the mean
    mean = statistics.mean(arr)

    # Calculate the median
    median = statistics.median(arr)

    # Calculate the mode
    counter = Counter(arr)
    mode = counter.most_common(1)[0][0]

    return mean, median, mode

def main():
    # Example usage
    arr = [1, 2, 3, 4, 4, 5, 5, 5]
    mean, median, mode = calculate_statistics(arr)
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Mode: {mode}")

if __name__ == "__main__":
    main()
```
