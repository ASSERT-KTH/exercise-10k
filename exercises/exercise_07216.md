## Exercise
Create a program that finds the minimum number in a list using a while loop.

## Reference Solution
```python
def find_min_number(num_list):
    """
    Finds the minimum number in a list using a while loop.

    Args:
        num_list (list): A list of numbers.

    Returns:
        int: The minimum number in the list.
    """
    if not num_list:
        return None  # Return None if the list is empty

    min_num = num_list[0]
    i = 1

    while i < len(num_list):
        if num_list[i] < min_num:
            min_num = num_list[i]
        i += 1

    return min_num


def main():
    # Example usage
    numbers = [10, 5, 8, 20, 3]
    min_number = find_min_number(numbers)

    if min_number is not None:
        print(f"The minimum number in the list is: {min_number}")
    else:
        print("The list is empty.")


if __name__ == "__main__":
    main()
```
