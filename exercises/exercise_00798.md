## Exercise
Create a program to calculate the covariance given two lists of numbers.

## Reference Solution
```python
import numpy as np

def calculate_covariance(list1, list2):
    """
    Calculate the covariance between two lists of numbers.

    Args:
        list1 (list): The first list of numbers.
        list2 (list): The second list of numbers.

    Returns:
        float: The covariance between the two lists.
    """
    # Convert the lists to NumPy arrays
    arr1 = np.array(list1)
    arr2 = np.array(list2)

    # Calculate the means of the arrays
    mean1 = np.mean(arr1)
    mean2 = np.mean(arr2)

    # Calculate the deviations from the means
    deviations1 = arr1 - mean1
    deviations2 = arr2 - mean2

    # Calculate the covariance
    covariance = np.mean(deviations1 * deviations2)

    return covariance

# Example usage
list1 = [1, 2, 3, 4, 5]
list2 = [2, 3, 5, 7, 11]

covariance = calculate_covariance(list1, list2)
print("Covariance:", covariance)
```
