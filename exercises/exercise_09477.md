## Exercise
Create a program that takes in two numbers and an operator (+, -, *, /) from the user, performs the specified operation, and outputs the result.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    Perform the specified operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not one of '+', '-', '*', or '/'.
        ZeroDivisionError: If the operator is '/' and num2 is 0.
    """

    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return num1 / num2
    else:
        raise ValueError("Invalid operator. Must be '+', '-', '*', or '/'")


def main():
    # Get the numbers and operator from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operator = input("Enter the operator (+, -, *, /): ")

    try:
        # Calculate and print the result
        result = calculate(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
