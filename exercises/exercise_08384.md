## Exercise
Write a program that calculates the nth Fibonacci number using dynamic programming.

## Reference Solution
```python
def fibonacci(n):
    # Create a dictionary to store the Fibonacci numbers
    fib_dict = {0: 0, 1: 1}

    # Calculate the Fibonacci numbers from 2 to n
    for i in range(2, n + 1):
        # Calculate the ith Fibonacci number as the sum of the two preceding numbers
        fib_dict[i] = fib_dict[i - 1] + fib_dict[i - 2]

    # Return the nth Fibonacci number
    return fib_dict[n]

# Test the function
for i in range(10):
    print(f"The {i}th Fibonacci number is: {fibonacci(i)}")
```

def fibonacci(n):
    if n <= 1:
        return n

    # Initialize the last two Fibonacci numbers
    a, b = 0, 1

    # Calculate the Fibonacci numbers from 2 to n
    for _ in range(2, n + 1):
        # Calculate the next Fibonacci number as the sum of the last two
        a, b = b, a + b

    # Return the nth Fibonacci number
    return b

# Test the function
for i in range(10):
    print(f"The {i}th Fibonacci number is: {fibonacci(i)}")
