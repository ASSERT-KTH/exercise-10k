## Exercise
Create a program that reads a string from the user and prints it out with all special characters escaped.

## Reference Solution
```python
import sys

def escape_special_characters(input_string):
    """
    Escapes all special characters in the input string.
    
    Args:
        input_string (str): The input string to be processed.
    
    Returns:
        str: The input string with all special characters escaped.
    """
    escaped_string = ""
    for char in input_string:
        if not char.isalnum() and not char.isspace():
            # Use the built-in repr function to escape special characters
            escaped_string += repr(char)[1:-1]
        else:
            escaped_string += char
    return escaped_string

def main():
    # Read the input string from the user
    input_string = input("Enter a string: ")
    
    # Escape special characters in the input string
    escaped_string = escape_special_characters(input_string)
    
    # Print the escaped string
    print("Escaped string: ", escaped_string)

if __name__ == "__main__":
    main()
```
