## Exercise
Write a program that uses a while loop to generate a strong password with a given length and characteristics (uppercase, lowercase, digits, special characters).

## Reference Solution
```python
import string
import secrets

def generate_password(length, use_uppercase, use_numbers, use_special_chars):
    """
    Generate a strong password with the given characteristics.

    Args:
        length (int): The length of the password.
        use_uppercase (bool): Whether to include uppercase letters.
        use_numbers (bool): Whether to include digits.
        use_special_chars (bool): Whether to include special characters.

    Returns:
        str: The generated password.
    """
    # Define the characters that can be used in the password
    chars = string.ascii_lowercase
    if use_uppercase:
        chars += string.ascii_uppercase
    if use_numbers:
        chars += string.digits
    if use_special_chars:
        chars += string.punctuation

    # Generate the password
    password = ''
    while len(password) < length:
        password += secrets.choice(chars)

    return password

def main():
    # Get the password length and characteristics from the user
    length = int(input("Enter the password length: "))
    use_uppercase = input("Include uppercase letters? (y/n): ").lower() == 'y'
    use_numbers = input("Include digits? (y/n): ").lower() == 'y'
    use_special_chars = input("Include special characters? (y/n): ").lower() == 'y'

    # Generate and print the password
    password = generate_password(length, use_uppercase, use_numbers, use_special_chars)
    print("Generated password:", password)

if __name__ == "__main__":
    main()
```
