## Exercise
Create a function that uses the reduce() function to apply a given function to a list of numbers

## Reference Solution
```python
import functools
import operator

def apply_function_to_list(numbers, func):
    """
    Applies a given function to a list of numbers using reduce.

    Args:
        numbers (list): A list of numbers.
        func (function): A function that takes two arguments.

    Returns:
        The result of applying the function to the list of numbers.
    """
    return functools.reduce(func, numbers)

# Example usage
numbers = [1, 2, 3, 4, 5]

# Define a function to add two numbers
def add(x, y):
    return x + y

# Define a function to multiply two numbers
def multiply(x, y):
    return x * y

# Apply the add function to the list of numbers
result_add = apply_function_to_list(numbers, add)
print("Sum of numbers:", result_add)

# Apply the multiply function to the list of numbers
result_multiply = apply_function_to_list(numbers, multiply)
print("Product of numbers:", result_multiply)

# Alternatively, you can use the operator module for common functions
result_add_operator = apply_function_to_list(numbers, operator.add)
print("Sum of numbers using operator.add:", result_add_operator)

result_multiply_operator = apply_function_to_list(numbers, operator.mul)
print("Product of numbers using operator.mul:", result_multiply_operator)
```
