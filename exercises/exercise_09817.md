## Exercise
Create a program that takes two integers and an arithmetic operator as input and performs the corresponding operation. The operators to consider are: addition (+), subtraction (-), multiplication (*), and division (/). Handle cases where the operator is not one of the above or where division by zero occurs.

## Reference Solution
```python
def arithmetic_operation(num1, num2, operator):
    """
    This function takes two integers and an arithmetic operator as input and 
    performs the corresponding operation.

    Args:
        num1 (int): The first integer.
        num2 (int): The second integer.
        operator (str): The arithmetic operator.

    Returns:
        float: The result of the operation.
    """

    # First, we define the allowed operators
    allowed_operators = ['+', '-', '*', '/']

    # Check if the operator is one of the allowed operators
    if operator not in allowed_operators:
        raise ValueError("Invalid operator. Allowed operators are: +, -, *, /")

    # Perform the operation based on the operator
    if operator == '+':
        # If the operator is '+', perform addition
        result = num1 + num2
    elif operator == '-':
        # If the operator is '-', perform subtraction
        result = num1 - num2
    elif operator == '*':
        # If the operator is '*', perform multiplication
        result = num1 * num2
    elif operator == '/':
        # If the operator is '/', perform division
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        result = num1 / num2

    return result


def main():
    try:
        num1 = int(input("Enter the first number: "))
        num2 = int(input("Enter the second number: "))
        operator = input("Enter the operator (+, -, *, /): ")
        
        result = arithmetic_operation(num1, num2, operator)
        print(f"The result of {num1} {operator} {num2} is {result}")
    
    except ValueError as e:
        print(f"Error: {e}")
    
    except ZeroDivisionError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
