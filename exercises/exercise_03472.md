## Exercise
Write a program that scrambles a word and asks the user to unscramble it.

## Reference Solution
```python
import random

def scramble_word(word):
    """Scramble the letters in a word"""
    word_list = list(word)
    random.shuffle(word_list)
    return ''.join(word_list)

def unscramble_word_game():
    """Play the unscramble word game"""
    words = ["apple", "banana", "cherry", "date", "elderberry"]
    word = random.choice(words)
    scrambled_word = scramble_word(word)
    attempts = 3

    print(f"Unscramble the word: {scrambled_word}")
    print(f"You have {attempts} attempts.")

    while attempts > 0:
        guess = input("Enter your guess: ")
        if guess.lower() == word:
            print("Congratulations! You unscrambled the word correctly.")
            return
        else:
            attempts -= 1
            print(f"Incorrect. You have {attempts} attempts remaining.")

    print(f"Sorry, you didn't unscramble the word. The correct answer was {word}.")

if __name__ == "__main__":
    unscramble_word_game()
```
