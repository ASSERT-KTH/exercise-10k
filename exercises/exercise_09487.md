## Exercise
Create a program that uses a dictionary to store student grades. The program should allow users to add students, add grades for a specific student, and calculate the average grade for a student. The dictionary keys should be the student names and the values should be lists of grades.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.student_grades = {}

    def add_student(self, student_name):
        """
        Add a student to the dictionary.

        Args:
            student_name (str): The name of the student.
        """
        if student_name not in self.student_grades:
            self.student_grades[student_name] = []
            print(f"Student '{student_name}' added successfully.")
        else:
            print(f"Student '{student_name}' already exists.")

    def add_grade(self, student_name, grade):
        """
        Add a grade for a specific student.

        Args:
            student_name (str): The name of the student.
            grade (float): The grade to be added.
        """
        if student_name in self.student_grades:
            self.student_grades[student_name].append(grade)
            print(f"Grade '{grade}' added for student '{student_name}'.")
        else:
            print(f"Student '{student_name}' does not exist. Please add the student first.")

    def calculate_average_grade(self, student_name):
        """
        Calculate the average grade for a student.

        Args:
            student_name (str): The name of the student.

        Returns:
            float: The average grade of the student. If the student has no grades, returns 0.
        """
        if student_name in self.student_grades:
            if self.student_grades[student_name]:
                average_grade = sum(self.student_grades[student_name]) / len(self.student_grades[student_name])
                print(f"The average grade for student '{student_name}' is {average_grade:.2f}.")
                return average_grade
            else:
                print(f"Student '{student_name}' has no grades.")
                return 0
        else:
            print(f"Student '{student_name}' does not exist. Please add the student first.")
            return None

    def display_student_grades(self):
        """
        Display all student grades.
        """
        for student, grades in self.student_grades.items():
            print(f"Student: {student}, Grades: {grades}")


def main():
    student_grades = StudentGrades()

    while True:
        print("\n1. Add Student")
        print("2. Add Grade")
        print("3. Calculate Average Grade")
        print("4. Display Student Grades")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            student_name = input("Enter student name: ")
            student_grades.add_student(student_name)
        elif choice == "2":
            student_name = input("Enter student name: ")
            grade = float(input("Enter grade: "))
            student_grades.add_grade(student_name, grade)
        elif choice == "3":
            student_name = input("Enter student name: ")
            student_grades.calculate_average_grade(student_name)
        elif choice == "4":
            student_grades.display_student_grades()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
