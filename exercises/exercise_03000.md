## Exercise
Write a program that finds the longest palindromic substring of a given string.

## Reference Solution
```python
def is_palindrome(s: str) -> bool:
    """Check if a string is a palindrome."""
    return s == s[::-1]


def longest_palindromic_substring(s: str) -> str:
    """Find the longest palindromic substring of a given string."""
    longest = ""
    for i in range(len(s)):
        for j in range(i + 1, len(s) + 1):
            substring = s[i:j]
            if is_palindrome(substring) and len(substring) > len(longest):
                longest = substring
    return longest


# Example usage
if __name__ == "__main__":
    input_string = "babad"
    print("Longest palindromic substring of '{}' is '{}'".format(
        input_string, longest_palindromic_substring(input_string)))
```
