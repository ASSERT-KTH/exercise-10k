## Exercise
Write a function that generates all permutations of a given string.

## Reference Solution
```python
def generate_permutations(s):
    """
    Generates all permutations of a given string.

    Args:
        s (str): The input string.

    Returns:
        list: A list of all permutations of the input string.
    """
    # Base case: if the string is empty or has only one character, return a list containing the string
    if len(s) <= 1:
        return [s]

    # Initialize an empty list to store the permutations
    permutations = []

    # Iterate over each character in the string
    for i, char in enumerate(s):
        # Get the remaining characters
        remaining_chars = s[:i] + s[i+1:]

        # Recursively generate permutations of the remaining characters
        for perm in generate_permutations(remaining_chars):
            # Append the current character to each permutation
            permutations.append(char + perm)

    # Return the list of permutations
    return permutations

# Example usage
def main():
    input_str = "abc"
    permutations = generate_permutations(input_str)
    print("Permutations of '{}'".format(input_str))
    for perm in permutations:
        print(perm)

if __name__ == "__main__":
    main()
```
