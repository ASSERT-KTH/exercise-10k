## Exercise
Write a program that counts the frequency of each word in a given text using for loops.

## Reference Solution
```python
def count_word_frequency(text):
    """
    Counts the frequency of each word in a given text.
    
    Args:
    text (str): The input text.
    
    Returns:
    dict: A dictionary where the keys are the words and the values are their frequencies.
    """
    
    # Convert the text to lower case and split it into words
    words = text.lower().split()
    
    # Initialize an empty dictionary to store the word frequencies
    word_frequency = {}
    
    # Iterate over each word in the text
    for word in words:
        # Remove punctuation from the word
        word = ''.join(e for e in word if e.isalnum())
        
        # If the word is already in the dictionary, increment its frequency
        if word in word_frequency:
            word_frequency[word] += 1
        # If the word is not in the dictionary, add it with a frequency of 1
        else:
            word_frequency[word] = 1
    
    return word_frequency

# Example usage
text = "This is a test. This test is only a test."
print(count_word_frequency(text))
```
