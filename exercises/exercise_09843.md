## Exercise
Create a program that stores student grades in a dictionary. The dictionary should have student names as keys and lists of grades as values. The program should be able to add a student, add a grade to a student's list of grades, and calculate the average grade for a student.

## Reference Solution
```python
class Gradebook:
    def __init__(self):
        """
        Initialize an empty gradebook.
        """
        self.gradebook = {}

    def add_student(self, name):
        """
        Add a student to the gradebook.

        Args:
        name (str): The name of the student to add.
        """
        if name not in self.gradebook:
            self.gradebook[name] = []
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists in the gradebook.")

    def add_grade(self, name, grade):
        """
        Add a grade to a student's list of grades.

        Args:
        name (str): The name of the student to add a grade for.
        grade (float): The grade to add.
        """
        if name in self.gradebook:
            self.gradebook[name].append(grade)
            print(f"Grade '{grade}' added successfully for student '{name}'.")
        else:
            print(f"Student '{name}' does not exist in the gradebook.")

    def calculate_average(self, name):
        """
        Calculate the average grade for a student.

        Args:
        name (str): The name of the student to calculate the average grade for.

        Returns:
        float: The average grade for the student.
        """
        if name in self.gradebook:
            if self.gradebook[name]:
                average = sum(self.gradebook[name]) / len(self.gradebook[name])
                print(f"The average grade for student '{name}' is {average:.2f}.")
                return average
            else:
                print(f"Student '{name}' has no grades to calculate an average.")
                return None
        else:
            print(f"Student '{name}' does not exist in the gradebook.")
            return None

    def display_gradebook(self):
        """
        Display the current state of the gradebook.
        """
        print("\nCurrent Gradebook:")
        for student, grades in self.gradebook.items():
            print(f"{student}: {grades}")


def main():
    gradebook = Gradebook()

    while True:
        print("\nGradebook Menu:")
        print("1. Add a student")
        print("2. Add a grade")
        print("3. Calculate average grade")
        print("4. Display gradebook")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter the student's name: ")
            gradebook.add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            grade = float(input("Enter the grade: "))
            gradebook.add_grade(name, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            gradebook.calculate_average(name)
        elif choice == "4":
            gradebook.display_gradebook()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
