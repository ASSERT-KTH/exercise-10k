## Exercise
Write a program that allows the user to create and manage a to-do list. The program should print out the current list and allow the user to add or remove items.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def display_tasks(self):
        if not self.tasks:
            print("Your to-do list is empty.")
        else:
            print("Your current to-do list:")
            for index, task in enumerate(self.tasks, start=1):
                print(f"{index}. {task}")

    def add_task(self):
        task = input("Enter a new task: ")
        self.tasks.append(task)
        print(f"Task '{task}' added successfully.")

    def remove_task(self):
        if not self.tasks:
            print("Your to-do list is empty.")
            return

        self.display_tasks()
        try:
            task_number = int(input("Enter the task number to remove: "))
            if task_number < 1 or task_number > len(self.tasks):
                print("Invalid task number.")
            else:
                task = self.tasks.pop(task_number - 1)
                print(f"Task '{task}' removed successfully.")
        except ValueError:
            print("Invalid input. Please enter a number.")


def main():
    todo_list = ToDoList()

    while True:
        print("\nTo-Do List Menu:")
        print("1. Display tasks")
        print("2. Add task")
        print("3. Remove task")
        print("4. Quit")

        try:
            choice = int(input("Choose an option: "))
        except ValueError:
            print("Invalid input. Please enter a number.")
            continue

        if choice == 1:
            todo_list.display_tasks()
        elif choice == 2:
            todo_list.add_task()
        elif choice == 3:
            todo_list.remove_task()
        elif choice == 4:
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
