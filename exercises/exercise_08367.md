## Exercise
Write a program that generates a geometric sequence of a given length with a given common ratio.

## Reference Solution
```python
def geometric_sequence(length, ratio, start=1):
    """
    Generate a geometric sequence of a given length with a given common ratio.

    Parameters:
    length (int): The length of the sequence.
    ratio (float): The common ratio of the sequence.
    start (float): The starting value of the sequence. Defaults to 1.

    Returns:
    list: The generated geometric sequence.
    """
    sequence = [start]
    for _ in range(length - 1):
        sequence.append(sequence[-1] * ratio)
    return sequence


def main():
    # Example usage:
    length = 10  # length of the sequence
    ratio = 2    # common ratio
    sequence = geometric_sequence(length, ratio)
    print(f"Geometric sequence with length {length} and common ratio {ratio}:")
    print(sequence)


if __name__ == "__main__":
    main()
```
