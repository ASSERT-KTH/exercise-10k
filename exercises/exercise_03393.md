## Exercise
Write a function to find the median of a list of integers.

## Reference Solution
```python
def find_median(numbers):
    """
    Calculate the median of a list of integers.

    Args:
        numbers (list): A list of integers.

    Returns:
        float: The median of the input list.
    """
    # Check if the input list is empty
    if not numbers:
        raise ValueError("Input list is empty")

    # Sort the list in ascending order
    numbers.sort()

    # Calculate the length of the list
    n = len(numbers)

    # If the length is even, the median is the average of the two middle numbers
    if n % 2 == 0:
        median = (numbers[n // 2 - 1] + numbers[n // 2]) / 2
    # If the length is odd, the median is the middle number
    else:
        median = numbers[n // 2]

    return median

# Example usage
def main():
    numbers = [12, 4, 56, 7, 23, 5, 9]
    print(find_median(numbers))

if __name__ == "__main__":
    main()
```
