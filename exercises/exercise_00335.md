## Exercise
Write a program to demonstrate the promotion of variables from one data type to another during arithmetic operations.

## Reference Solution
```python
def demonstrate_type_promotion():
    # Define variables of different data types
    a_int = 10  # integer
    b_float = 20.5  # float
    c_complex = 30 + 40j  # complex

    # Arithmetic operations with integer and float
    print("Arithmetic operations with integer and float:")
    print(f"Integer: {a_int}, Float: {b_float}")
    print(f"Addition: {a_int + b_float}, Type: {type(a_int + b_float)}")
    print(f"Subtraction: {a_int - b_float}, Type: {type(a_int - b_float)}")
    print(f"Multiplication: {a_int * b_float}, Type: {type(a_int * b_float)}")
    print(f"Division: {a_int / b_float}, Type: {type(a_int / b_float)}")

    # Arithmetic operations with integer and complex
    print("\nArithmetic operations with integer and complex:")
    print(f"Integer: {a_int}, Complex: {c_complex}")
    print(f"Addition: {a_int + c_complex}, Type: {type(a_int + c_complex)}")
    print(f"Subtraction: {a_int - c_complex}, Type: {type(a_int - c_complex)}")
    print(f"Multiplication: {a_int * c_complex}, Type: {type(a_int * c_complex)}")
    print(f"Division: {a_int / c_complex}, Type: {type(a_int / c_complex)}")

    # Arithmetic operations with float and complex
    print("\nArithmetic operations with float and complex:")
    print(f"Float: {b_float}, Complex: {c_complex}")
    print(f"Addition: {b_float + c_complex}, Type: {type(b_float + c_complex)}")
    print(f"Subtraction: {b_float - c_complex}, Type: {type(b_float - c_complex)}")
    print(f"Multiplication: {b_float * c_complex}, Type: {type(b_float * c_complex)}")
    print(f"Division: {b_float / c_complex}, Type: {type(b_float / c_complex)}")

if __name__ == "__main__":
    demonstrate_type_promotion()
```
