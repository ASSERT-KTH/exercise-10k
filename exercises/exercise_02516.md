## Exercise
Create a program that takes a list of numbers as input, and returns the mean, median, and standard deviation.

## Reference Solution
```python
import numpy as np

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and standard deviation of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        dict: A dictionary containing the mean, median, and standard deviation.
    """
    # Calculate the mean
    mean = np.mean(numbers)

    # Calculate the median
    median = np.median(numbers)

    # Calculate the standard deviation
    std_dev = np.std(numbers)

    return {
        "mean": mean,
        "median": median,
        "standard_deviation": std_dev
    }

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    stats = calculate_statistics(numbers)

    print("Statistics:")
    print(f"Mean: {stats['mean']}")
    print(f"Median: {stats['median']}")
    print(f"Standard Deviation: {stats['standard_deviation']}")

if __name__ == "__main__":
    main()
```
