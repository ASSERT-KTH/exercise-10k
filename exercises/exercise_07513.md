## Exercise
Create a Simple Chatbot program using variables and conditional statements.

## Reference Solution
```python
# Import the required libraries
import time

# Define the chatbot's name and greeting
chatbot_name = "Meta Llama"
greeting = f"Hello! I'm {chatbot_name}. How can I assist you today?"

# Define the chatbot's responses
responses = {
    "hello": "Hi! How are you today?",
    "how are you": "I'm doing great, thanks for asking!",
    "what is your name": f"My name is {chatbot_name}.",
    "goodbye": "It was nice chatting with you. Goodbye!",
    "default": "I didn't understand that. Can you please rephrase?"
}

# Function to get the user's input
def get_user_input():
    user_input = input("You: ")
    return user_input.lower()

# Function to get the chatbot's response
def get_chatbot_response(user_input):
    for keyword, response in responses.items():
        if keyword in user_input:
            return response
    return responses["default"]

# Function to display the chatbot's greeting and start the conversation
def start_conversation():
    print(greeting)
    while True:
        user_input = get_user_input()
        if "goodbye" in user_input:
            print("Chatbot: ", responses["goodbye"])
            break
        else:
            print("Chatbot: ", get_chatbot_response(user_input))
            time.sleep(1)  # Pause for 1 second before responding again

# Start the conversation
start_conversation()
```
