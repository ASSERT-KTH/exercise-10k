## Exercise
Create a string and search for a substring using a specific search algorithm, then print out the indices of the found substrings.

## Reference Solution

def compute_prefix_function(pattern):
    """
    Compute the prefix function for the KMP algorithm.

    Args:
    pattern (str): The substring to search for.

    Returns:
    list: A list of prefix values.
    """
    prefix = [0] * len(pattern)
    j = 0
    for i in range(1, len(pattern)):
        while j > 0 and pattern[j] != pattern[i]:
            j = prefix[j - 1]
        if pattern[j] == pattern[i]:
            j += 1
        prefix[i] = j
    return prefix


def kmp_search(text, pattern):
    """
    Search for a substring in a text using the KMP algorithm.

    Args:
    text (str): The text to search in.
    pattern (str): The substring to search for.

    Returns:
    list: A list of indices where the substring was found.
    """
    prefix = compute_prefix_function(pattern)
    indices = []
    j = 0
    for i in range(len(text)):
        while j > 0 and text[i] != pattern[j]:
            j = prefix[j - 1]
        if text[i] == pattern[j]:
            j += 1
        if j == len(pattern):
            indices.append(i - j + 1)
            j = prefix[j - 1]
    return indices


# Example usage
if __name__ == "__main__":
    text = "abxabcabcaby"
    pattern = "abc"
    indices = kmp_search(text, pattern)
    print("Indices of found substrings:", indices)
