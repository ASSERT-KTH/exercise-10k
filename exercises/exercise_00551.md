## Exercise
Write a program that takes a list of strings as input and returns the longest common prefix among all the strings.

## Reference Solution
```python
function longestCommonPrefix(strs) {
  const prefix = [];
  for (let i = 0; i < strs[0].length; i++) {
    const letter = strs[0][i];
    for (let j = 1; j < strs.length; j++) {
      if (strs[j][i] !== letter) {
        return prefix.join('');
      }
    }
    prefix.push(letter);
  }
  return prefix.join('');
}

// Test cases

console.log(longestCommonPrefix(["dog","racecar","car"])); // Output: ""
console.log(longestCommonPrefix(["interview","interrupt","interior"])); // Output: "inter"
```
