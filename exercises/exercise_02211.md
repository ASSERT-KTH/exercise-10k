## Exercise
Write a program that takes a date as input and prints it in a standard format (MM/DD/YYYY).

## Reference Solution
```python
from datetime import datetime

def print_standard_date(date_str, input_format):
    """
    Prints the input date in the standard format (MM/DD/YYYY).
    
    Args:
    date_str (str): The input date string.
    input_format (str): The format of the input date string.
    """
    try:
        date = datetime.strptime(date_str, input_format)
        standard_date = date.strftime("%m/%d/%Y")
        print(standard_date)
    except ValueError:
        print("Invalid date format. Please use the correct input format.")

def main():
    date_str = input("Enter a date: ")
    input_formats = ["%Y-%m-%d", "%d-%m-%Y", "%m-%d-%Y", "%B %d, %Y"]
    for input_format in input_formats:
        try:
            print_standard_date(date_str, input_format)
            break
        except ValueError:
            continue
    else:
        print("Unsupported date format. Supported formats: YYYY-MM-DD, DD-MM-YYYY, MM-DD-YYYY, Month DD, YYYY")

if __name__ == "__main__":
    main()
```
