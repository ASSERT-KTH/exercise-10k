## Exercise
Create a function to calculate the percentage of consonants in a given string.

## Reference Solution
```python
def calculate_consonant_percentage(input_string):
    """
    Calculate the percentage of consonants in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        float: The percentage of consonants in the input string.
    """
    # Define vowels
    vowels = 'aeiou'

    # Convert the input string to lowercase
    input_string = input_string.lower()

    # Initialize counters for total alphabetic characters and consonants
    total_alphabetic_chars = 0
    consonants = 0

    # Iterate over each character in the input string
    for char in input_string:
        # Check if the character is an alphabetic character
        if char.isalpha():
            total_alphabetic_chars += 1
            # Check if the character is not a vowel
            if char not in vowels:
                consonants += 1

    # Check if there are any alphabetic characters
    if total_alphabetic_chars == 0:
        return 0
    else:
        # Calculate the percentage of consonants
        consonant_percentage = (consonants / total_alphabetic_chars) * 100
        return consonant_percentage

# Example usage
input_string = "Hello, World!"
consonant_percentage = calculate_consonant_percentage(input_string)
print(f"The percentage of consonants in '{input_string}' is {consonant_percentage:.2f}%")
```
