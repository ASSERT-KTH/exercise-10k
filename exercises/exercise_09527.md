## Exercise
Write a function to find the maximum contiguous subarray of a given array of numbers. The function should return the maximum sum of a subarray and the subarray itself.

## Reference Solution
```python
def max_sub_array_sum_and_array(a):
    size = len(a)
    max_so_far = a[0]
    max_ending_here = a[0]
    start = 0
    end = 0
    tempstart = 0

    for i in range(1, size):
        if a[i] > max_ending_here + a[i]:
            max_ending_here = a[i]
            tempstart = i
        else:
            max_ending_here = max_ending_here + a[i]
        
        if max_so_far < max_ending_here:
            max_so_far = max_ending_here
            start = tempstart
            end = i
    
    return max_so_far, a[start:end+1]

# Example usage:
a = [-2, -3, 4, -1, -2, 1, 5, -3]
max_sum, max_subarray = max_sub_array_sum_and_array(a)
print("Maximum contiguous sum is", max_sum)
print("Maximum contiguous subarray is", max_subarray)
```
