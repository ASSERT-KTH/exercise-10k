## Exercise
Manipulate a string by concatenating, uppercasing, and lowercasing it.

## Reference Solution
```python
def manipulate_string(input_string, append_string):
    """
    Manipulates the input string by concatenating, uppercasing, and lowercasing it.

    Args:
        input_string (str): The original string.
        append_string (str): The string to be appended to the input string.

    Returns:
        tuple: A tuple containing the concatenated string, the uppercased string, and the lowercased string.
    """
    # Concatenate the input string with the append string
    concatenated_string = input_string + append_string
    
    # Uppercase the concatenated string
    uppercased_string = concatenated_string.upper()
    
    # Lowercase the concatenated string
    lowercased_string = concatenated_string.lower()
    
    return concatenated_string, uppercased_string, lowercased_string

def main():
    input_string = "Hello, "
    append_string = "World!"
    
    concatenated, uppercased, lowercased = manipulate_string(input_string, append_string)
    
    print("Input String:", input_string)
    print("Append String:", append_string)
    print("Concatenated String:", concatenated)
    print("Uppercased String:", uppercased)
    print("Lowercased String:", lowercased)

if __name__ == "__main__":
    main()
```
