## Exercise
Create a program that calculates the solutions to a quadratic equation given the coefficients.

import math

def calculate_quadratic_solutions(a, b, c):
    """
    Calculates the solutions to a quadratic equation given the coefficients.

    Args:
    a (float): The coefficient of the quadratic term.
    b (float): The coefficient of the linear term.
    c (float): The constant term.

    Returns:
    tuple: A tuple containing the solutions to the quadratic equation.
    """
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c

    # Check if the discriminant is positive, zero, or negative
    if discriminant > 0:
        # Two distinct real solutions
        solution1 = (-b + math.sqrt(discriminant)) / (2*a)
        solution2 = (-b - math.sqrt(discriminant)) / (2*a)
        return solution1, solution2
    elif discriminant == 0:
        # One real solution
        solution = -b / (2*a)
        return solution, solution
    else:
        # Two complex solutions
        real_part = -b / (2*a)
        imaginary_part = math.sqrt(-discriminant) / (2*a)
        solution1 = complex(real_part, imaginary_part)
        solution2 = complex(real_part, -imaginary_part)
        return solution1, solution2

def main():
    # Get the coefficients from the user
    a = float(input("Enter the coefficient of the quadratic term (a): "))
    b = float(input("Enter the coefficient of the linear term (b): "))
    c = float(input("Enter the constant term (c): "))

    # Calculate the solutions
    solution1, solution2 = calculate_quadratic_solutions(a, b, c)

    # Print the solutions
    print("The solutions to the quadratic equation are:")
    print(f"x = {solution1}")
    print(f"x = {solution2}")

if __name__ == "__main__":
    main()
