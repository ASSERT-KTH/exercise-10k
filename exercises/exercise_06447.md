## Exercise
Write a program that converts a given amount of money from one currency to another using a fixed exchange rate.

## Reference Solution
```python
def convert_currency(amount, exchange_rate):
    """
    Converts a given amount of money from one currency to another using a fixed exchange rate.

    Args:
        amount (float): The amount of money to be converted.
        exchange_rate (float): The exchange rate between the two currencies.

    Returns:
        float: The converted amount of money.
    """
    # Check if the amount and exchange rate are valid
    if amount < 0 or exchange_rate <= 0:
        raise ValueError("Invalid input. Amount must be non-negative and exchange rate must be positive.")

    # Convert the currency
    converted_amount = amount * exchange_rate

    return converted_amount


def main():
    # Example usage
    amount = 100  # Amount of money to be converted
    exchange_rate = 1.31  # Exchange rate between USD and CAD (1 USD = 1.31 CAD)

    converted_amount = convert_currency(amount, exchange_rate)

    print(f"{amount} USD is equal to {converted_amount} CAD")


if __name__ == "__main__":
    main()
```
