## Exercise
Create a dictionary and sort its values in ascending order

## Reference Solution
```python
def sort_dict_values(input_dict):
    """
    Sorts dictionary values in ascending order.

    Args:
        input_dict (dict): Input dictionary.

    Returns:
        list: Sorted dictionary values.
    """
    # Check if input is a dictionary
    if not isinstance(input_dict, dict):
        raise ValueError("Input must be a dictionary")

    # Sort dictionary values in ascending order
    sorted_values = sorted(input_dict.values())

    return sorted_values


# Example usage
if __name__ == "__main__":
    # Create a dictionary
    student_grades = {
        "Alice": 85,
        "Bob": 92,
        "Charlie": 78,
        "David": 95,
        "Eve": 88
    }

    # Sort dictionary values
    sorted_grades = sort_dict_values(student_grades)

    print("Sorted Grades:")
    for grade in sorted_grades:
        print(grade)
```
