## Exercise
Create a program that takes the time in one unit as input, and returns the time in another unit.

## Reference Solution
```python
def time_unit_converter(time_value, input_unit, output_unit):
    """
    Converts time from one unit to another.

    Args:
        time_value (float): The time value to be converted.
        input_unit (str): The unit of the input time value.
        output_unit (str): The unit to which the time value should be converted.

    Returns:
        float: The time value in the output unit.
    """

    # Define conversion factors
    conversion_factors = {
        's': 1,  # seconds
        'min': 60,  # minutes
        'h': 3600,  # hours
        'd': 86400,  # days
        'w': 604800,  # weeks
        'm': 2629800,  # months (approximate)
        'y': 31557600  # years (approximate)
    }

    # Convert input time to seconds
    time_in_seconds = time_value * conversion_factors[input_unit]

    # Convert seconds to output unit
    time_in_output_unit = time_in_seconds / conversion_factors[output_unit]

    return time_in_output_unit


def main():
    print("Time Unit Converter Program")
    print("---------------------------")

    # Get user input
    time_value = float(input("Enter the time value: "))
    input_unit = input("Enter the input unit (s, min, h, d, w, m, y): ")
    output_unit = input("Enter the output unit (s, min, h, d, w, m, y): ")

    # Convert time
    output_time = time_unit_converter(time_value, input_unit, output_unit)

    # Print result
    print(f"{time_value} {input_unit} is equal to {output_time} {output_unit}")


if __name__ == "__main__":
    main()
```
