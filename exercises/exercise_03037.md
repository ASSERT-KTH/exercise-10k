## Exercise
Write a program that takes a string as input and returns a sentiment score based on the text.

## Reference Solution
```python
import nltk
from nltk.sentiment.vader import SentimentIntensityAnalyzer

def calculate_sentiment(text):
    """
    Calculate the sentiment score of a given text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary containing the sentiment scores.
    """
    # Initialize the VADER sentiment analyzer
    sia = SentimentIntensityAnalyzer()

    # Calculate the sentiment scores
    scores = sia.polarity_scores(text)

    return scores

def main():
    # Example usage
    text = "I love this product! It's amazing."
    scores = calculate_sentiment(text)

    print("Sentiment Scores:")
    print(f"  - Positive: {scores['pos']:.2f}")
    print(f"  - Negative: {scores['neg']:.2f}")
    print(f"  - Neutral: {scores['neu']:.2f}")
    print(f"  - Compound: {scores['compound']:.2f}")

if __name__ == "__main__":
    # Download the VADER sentiment lexicon if not already downloaded
    nltk.download('vader_lexicon', quiet=True)

    main()
```
