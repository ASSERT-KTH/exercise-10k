## Exercise
Write a program that allows users to create and manage a dictionary of words and their definitions.

## Reference Solution
```python
class Dictionary:
    def __init__(self):
        self.words = {}

    def add_word(self, word, definition):
        """Add a word to the dictionary."""
        if word not in self.words:
            self.words[word] = definition
            print(f"Word '{word}' added successfully.")
        else:
            print(f"Word '{word}' already exists in the dictionary.")

    def remove_word(self, word):
        """Remove a word from the dictionary."""
        if word in self.words:
            del self.words[word]
            print(f"Word '{word}' removed successfully.")
        else:
            print(f"Word '{word}' does not exist in the dictionary.")

    def update_word(self, word, new_definition):
        """Update the definition of a word in the dictionary."""
        if word in self.words:
            self.words[word] = new_definition
            print(f"Definition of word '{word}' updated successfully.")
        else:
            print(f"Word '{word}' does not exist in the dictionary.")

    def search_word(self, word):
        """Search for a word in the dictionary."""
        if word in self.words:
            print(f"Definition of '{word}': {self.words[word]}")
        else:
            print(f"Word '{word}' not found in the dictionary.")

    def display_dictionary(self):
        """Display all words and their definitions in the dictionary."""
        if not self.words:
            print("Dictionary is empty.")
        else:
            for word, definition in self.words.items():
                print(f"{word}: {definition}")


def main():
    dictionary = Dictionary()

    while True:
        print("\nDictionary Manager")
        print("1. Add Word")
        print("2. Remove Word")
        print("3. Update Word")
        print("4. Search Word")
        print("5. Display Dictionary")
        print("6. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            word = input("Enter the word: ")
            definition = input("Enter the definition: ")
            dictionary.add_word(word, definition)
        elif choice == "2":
            word = input("Enter the word to remove: ")
            dictionary.remove_word(word)
        elif choice == "3":
            word = input("Enter the word to update: ")
            new_definition = input("Enter the new definition: ")
            dictionary.update_word(word, new_definition)
        elif choice == "4":
            word = input("Enter the word to search: ")
            dictionary.search_word(word)
        elif choice == "5":
            dictionary.display_dictionary()
        elif choice == "6":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
