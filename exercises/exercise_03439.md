## Exercise
Write a function that takes a string as input and returns the Base64 encoded string

## Reference Solution
```python
import base64

def encode_base64(input_string: str) -> str:
    """
    This function takes a string as input and returns the Base64 encoded string.

    Args:
        input_string (str): The input string to be encoded.

    Returns:
        str: The Base64 encoded string.
    """
    # Convert the input string into bytes
    input_bytes = input_string.encode('utf-8')
    
    # Encode the bytes into Base64
    encoded_bytes = base64.b64encode(input_bytes)
    
    # Convert the encoded bytes back into a string
    encoded_string = encoded_bytes.decode('utf-8')
    
    return encoded_string

# Example usage:
if __name__ == "__main__":
    input_str = "Hello, World!"
    encoded_str = encode_base64(input_str)
    print(f"Input: {input_str}")
    print(f"Base64 Encoded: {encoded_str}")
```
