## Exercise
Write a program to check if a given matrix is symmetric.

## Reference Solution
```python
import numpy as np

def is_symmetric(matrix):
    """
    Checks if a given matrix is symmetric.

    Args:
        matrix (numpy.ndarray): The input matrix.

    Returns:
        bool: True if the matrix is symmetric, False otherwise.
    """
    # Check if the matrix is square
    if matrix.shape[0] != matrix.shape[1]:
        return False

    # Check if the matrix is equal to its transpose
    return np.allclose(matrix, matrix.T)

def main():
    # Example usage
    matrix1 = np.array([[1, 2, 3], [2, 4, 5], [3, 5, 6]])
    matrix2 = np.array([[1, 2], [3, 4]])

    print(f"Is matrix1 symmetric? {is_symmetric(matrix1)}")
    print(f"Is matrix2 symmetric? {is_symmetric(matrix2)}")

if __name__ == "__main__":
    main()
```
