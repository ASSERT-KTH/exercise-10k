## Exercise
Write a program to group a list of objects by multiple attributes and print the resulting groups.

## Reference Solution
```python
from itertools import groupby
from operator import attrgetter

class Person:
    def __init__(self, name, age, city):
        self.name = name
        self.age = age
        self.city = city

    def __repr__(self):
        return f"{self.name}, {self.age}, {self.city}"

def group_objects(object_list, *attrs):
    """Group a list of objects by multiple attributes."""
    # Sort the list of objects based on the attributes
    sorted_list = sorted(object_list, key=attrgetter(*attrs))
    
    # Group the sorted list by the attributes
    groups = groupby(sorted_list, key=attrgetter(*attrs))
    
    return groups

def print_groups(groups):
    """Print the resulting groups."""
    for key, group in groups:
        print(f"Group: {key}")
        for obj in group:
            print(obj)
        print()

def main():
    # Create a list of Person objects
    persons = [
        Person("John", 25, "New York"),
        Person("Alice", 25, "New York"),
        Person("Bob", 30, "Los Angeles"),
        Person("Charlie", 30, "Los Angeles"),
        Person("David", 25, "Chicago"),
    ]

    # Group the persons by age and city
    groups = group_objects(persons, "age", "city")

    # Print the resulting groups
    print_groups(groups)

if __name__ == "__main__":
    main()
```
