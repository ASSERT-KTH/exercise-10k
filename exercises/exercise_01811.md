## Exercise
Write a program that plans a travel itinerary based on user input, using conditional statements to determine the travel routes

## Reference Solution
```python
def travel_itinerary():
    print("Welcome to the Travel Itinerary Planner!")
    
    # Get user input
    start_location = input("Enter your starting location: ")
    end_location = input("Enter your destination: ")
    mode_of_transport = input("Enter your preferred mode of transport (car, train, plane, bus): ")
    budget = float(input("Enter your budget: "))

    # Determine travel route based on mode of transport and budget
    if mode_of_transport.lower() == "car":
        if budget < 500:
            print(f"Route: Drive from {start_location} to {end_location} via the highway.")
        else:
            print(f"Route: Drive from {start_location} to {end_location} via the scenic route.")
    elif mode_of_transport.lower() == "train":
        if budget < 200:
            print(f"Route: Take the local train from {start_location} to {end_location}.")
        else:
            print(f"Route: Take the express train from {start_location} to {end_location}.")
    elif mode_of_transport.lower() == "plane":
        if budget < 1000:
            print(f"Route: Fly from {start_location} to {end_location} with a budget airline.")
        else:
            print(f"Route: Fly from {start_location} to {end_location} with a premium airline.")
    elif mode_of_transport.lower() == "bus":
        if budget < 100:
            print(f"Route: Take the local bus from {start_location} to {end_location}.")
        else:
            print(f"Route: Take the express bus from {start_location} to {end_location}.")
    else:
        print("Invalid mode of transport. Please try again.")

    # Additional suggestions based on mode of transport
    if mode_of_transport.lower() == "car":
        print("Consider stopping at local attractions along the way.")
    elif mode_of_transport.lower() == "train":
        print("Consider booking a seat with extra legroom.")
    elif mode_of_transport.lower() == "plane":
        print("Consider checking in online and printing your boarding pass.")
    elif mode_of_transport.lower() == "bus":
        print("Consider bringing a book or portable charger.")

# Run the program
travel_itinerary()
```
