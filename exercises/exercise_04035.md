## Exercise
Create a function to find the minimum number of coins needed to make change for a given amount

## Reference Solution
```python
def min_coins(coins, amount):
    """
    Calculate the minimum number of coins needed to make change for a given amount.

    Args:
    - coins (list): A list of available coin denominations.
    - amount (int): The amount for which we need to make change.

    Returns:
    - int: The minimum number of coins needed to make change for the given amount.
    """
    # Create a list to store the minimum number of coins needed for each amount from 0 to the given amount
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # We need 0 coins to make change for 0 amount

    # Iterate over each coin denomination
    for coin in coins:
        # Iterate over each amount from the coin denomination to the given amount
        for i in range(coin, amount + 1):
            # Update the minimum number of coins needed for the current amount
            dp[i] = min(dp[i], dp[i - coin] + 1)

    # If we cannot make change for the given amount, return -1
    if dp[amount] == float('inf'):
        return -1
    else:
        return dp[amount]

# Example usage
coins = [1, 2, 5]
amount = 11
print(min_coins(coins, amount))  # Output: 3 (11 = 5 + 5 + 1)
```
