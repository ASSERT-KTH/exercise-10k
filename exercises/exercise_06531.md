## Exercise
Create a function that calculates the factorial of a given number.

## Reference Solution
```python
def factorial_recursive(n):
    """
    Calculate the factorial of a number using recursion.

    Args:
    n (int): The input number.

    Returns:
    int: The factorial of the input number.

    Raises:
    ValueError: If the input number is negative.
    """
    if n < 0:
        raise ValueError("Input number must be non-negative")
    elif n == 0 or n == 1:
        return 1
    else:
        return n * factorial_recursive(n-1)


def factorial_iterative(n):
    """
    Calculate the factorial of a number using iteration.

    Args:
    n (int): The input number.

    Returns:
    int: The factorial of the input number.

    Raises:
    ValueError: If the input number is negative.
    """
    if n < 0:
        raise ValueError("Input number must be non-negative")
    result = 1
    for i in range(1, n+1):
        result *= i
    return result


def main():
    # Example usage
    num = 5
    print(f"Factorial of {num} (recursive): {factorial_recursive(num)}")
    print(f"Factorial of {num} (iterative): {factorial_iterative(num)}")


if __name__ == "__main__":
    main()
```
