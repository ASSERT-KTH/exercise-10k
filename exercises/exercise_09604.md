## Exercise
Write a program that allows the user to navigate a text-based adventure game

## Reference Solution
```python
# Import required libraries
import time

# Define game classes
class Room:
    def __init__(self, name, description):
        self.name = name
        self.description = description
        self.exits = {}
        self.items = []

class Player:
    def __init__(self, name):
        self.name = name
        self.inventory = []
        self.current_room = None

class Game:
    def __init__(self):
        self.rooms = []
        self.player = None

    def create_rooms(self):
        # Create rooms
        kitchen = Room("Kitchen", "A room with a table and chairs.")
        garden = Room("Garden", "A beautiful garden with flowers and trees.")
        hall = Room("Hall", "A grand hall with a staircase.")

        # Add exits to rooms
        kitchen.exits = {"north": hall, "east": garden}
        garden.exits = {"west": kitchen}
        hall.exits = {"south": kitchen}

        # Add rooms to game
        self.rooms.append(kitchen)
        self.rooms.append(garden)
        self.rooms.append(hall)

        # Set player's starting room
        self.player.current_room = kitchen

    def create_player(self, name):
        self.player = Player(name)

    def play(self):
        print("Welcome to the adventure game!")
        print("You are", self.player.name)
        print("You are in the", self.player.current_room.name)
        print(self.player.current_room.description)

        while True:
            # Get user input
            action = input("> ").lower().split()

            # Handle movement
            if action[0] in ["go", "move"]:
                if len(action) < 2:
                    print("Which direction do you want to go?")
                else:
                    direction = action[1]
                    if direction in self.player.current_room.exits:
                        self.player.current_room = self.player.current_room.exits[direction]
                        print("You are now in the", self.player.current_room.name)
                        print(self.player.current_room.description)
                    else:
                        print("You cannot go that way.")

            # Handle item collection
            elif action[0] == "take":
                if len(action) < 2:
                    print("What do you want to take?")
                else:
                    item_name = " ".join(action[1:])
                    for item in self.player.current_room.items:
                        if item == item_name:
                            self.player.inventory.append(item)
                            self.player.current_room.items.remove(item)
                            print("You took", item)
                            break
                    else:
                        print("There is no such item in the room.")

            # Handle inventory check
            elif action[0] == "inventory":
                if self.player.inventory:
                    print("You have:")
                    for item in self.player.inventory:
                        print(item)
                else:
                    print("You have no items.")

            # Handle quit
            elif action[0] == "quit":
                print("Goodbye!")
                break

            # Handle unknown commands
            else:
                print("Unknown command. Type 'go' to move, 'take' to take an item, 'inventory' to check your inventory, or 'quit' to quit the game.")

# Create game instance
game = Game()

# Create player
player_name = input("Enter your name: ")
game.create_player(player_name)

# Create rooms
game.create_rooms()

# Start game
game.play()
```
