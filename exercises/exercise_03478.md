## Exercise
Write a program that asks for the user's location and prints out the current weather.

## Reference Solution
```python
import requests

def get_current_weather(api_key, location):
    """
    Get the current weather for a given location.

    Args:
        api_key (str): OpenWeatherMap API key
        location (str): City name or zip code

    Returns:
        dict: Current weather data
    """
    base_url = "http://api.openweathermap.org/data/2.5/weather"
    params = {
        "q": location,
        "appid": api_key,
        "units": "imperial"  # Use imperial units (Fahrenheit)
    }

    response = requests.get(base_url, params=params)
    response.raise_for_status()  # Raise an exception for bad status codes

    return response.json()

def print_weather(weather_data):
    """
    Print the current weather data.

    Args:
        weather_data (dict): Current weather data
    """
    print(f"Weather in {weather_data['name']}:")
    print(f"Temperature: {weather_data['main']['temp']}Â°F")
    print(f"Conditions: {weather_data['weather'][0]['description']}")
    print(f"Humidity: {weather_data['main']['humidity']}%")
    print(f"Wind: {weather_data['wind']['speed']} mph")

def main():
    api_key = input("Enter your OpenWeatherMap API key: ")
    location = input("Enter your location (city name or zip code): ")

    weather_data = get_current_weather(api_key, location)
    print_weather(weather_data)

if __name__ == "__main__":
    main()
```
