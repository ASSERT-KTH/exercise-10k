## Exercise
Develop a program that takes in the image width and height, then prints out the resized image dimensions. If the width is greater than the height, print 'Landscape', otherwise print 'Portrait'

## Reference Solution
```python
def resize_image(width, height, max_size=1024):
    """
    Resizes an image while maintaining its aspect ratio.

    Args:
        width (int): The original width of the image.
        height (int): The original height of the image.
        max_size (int, optional): The maximum size of the resized image. Defaults to 1024.

    Returns:
        tuple: The resized width and height of the image.
    """
    # Calculate the aspect ratio
    aspect_ratio = width / height

    # If the width is greater than the height, the image is in landscape mode
    orientation = "Landscape" if width > height else "Portrait"

    # Resize the image while maintaining the aspect ratio
    if width > height:
        new_width = max_size
        new_height = int(max_size / aspect_ratio)
    else:
        new_width = int(max_size * aspect_ratio)
        new_height = max_size

    return new_width, new_height, orientation


def main():
    # Get the image width and height from the user
    width = int(input("Enter the image width: "))
    height = int(input("Enter the image height: "))

    # Resize the image and print the results
    new_width, new_height, orientation = resize_image(width, height)
    print(f"Resized Image Dimensions: {new_width}x{new_height}")
    print(f"Image Orientation: {orientation}")


if __name__ == "__main__":
    main()
```
