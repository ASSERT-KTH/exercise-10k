## Exercise
Write a program that takes an array and an index as input, and returns the array with the element at the specified index deleted.

## Reference Solution

#include <stdlib.h>
#include <stdio.h>
#include <string.h>

int* Delete(int* array, int arraySize, int index, int* returnSize) {
    if (index < 0 || index >= arraySize) {
        return NULL;
    }

    int* new_array = (int*)malloc((arraySize - 1) * sizeof(int));

    if (index == 0) {
        memcpy(new_array, array + 1, (arraySize - 1) * sizeof(int));
    } else if (index == arraySize - 1) {
        memcpy(new_array, array, (arraySize - 1) * sizeof(int));
    } else {
        memcpy(new_array, array, index * sizeof(int));
        memcpy(new_array + index, array + index + 1, (arraySize - index - 1) * sizeof(int));
    }

    *returnSize = arraySize - 1;
    return new_array;
}

int main() {
    int array[] = {1, 2, 3, 4, 5};
    int arraySize = sizeof(array) / sizeof(array[0]);
    int index = 2;

    int returnSize;
    int* result = Delete(array, arraySize, index, &returnSize);

    if (result != NULL) {
        printf("Array after deletion: [");
        for (int i = 0; i < returnSize; i++) {
            printf("%d", result[i]);
            if (i < returnSize - 1) {
                printf(", ");
            }
        }
        printf("]\n");
    } else {
        printf("Error: Invalid index\n");
    }

    return 0;
}
