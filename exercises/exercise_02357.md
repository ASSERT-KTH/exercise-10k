## Exercise
Write a program to take a string as input and count the frequency of each word in the string

## Reference Solution
```python
from collections import Counter
import re

def count_word_frequencies(input_string):
    """
    Count the frequency of each word in the input string.

    Args:
        input_string (str): The input string to process.

    Returns:
        dict: A dictionary where the keys are the unique words and the values are their frequencies.
    """
    # Convert the input string to lowercase and remove punctuation
    cleaned_string = re.sub(r'[^\w\s]', '', input_string.lower())

    # Split the string into individual words
    words = cleaned_string.split()

    # Count the frequency of each word
    word_frequencies = Counter(words)

    return word_frequencies

def main():
    # Example usage
    input_string = "This is an example sentence. This sentence is just an example."
    word_frequencies = count_word_frequencies(input_string)

    # Print the word frequencies
    print("Word Frequencies:")
    for word, frequency in word_frequencies.items():
        print(f"{word}: {frequency}")

if __name__ == "__main__":
    main()
```
