## Exercise
Write a program that asks for two strings and checks if the first string contains the second string.

## Reference Solution
```python
def check_containment(main_string, sub_string):
    """
    Checks if the main string contains the sub string.

    Args:
        main_string (str): The string to search in.
        sub_string (str): The string to search for.

    Returns:
        bool: True if the main string contains the sub string, False otherwise.
    """
    return sub_string.lower() in main_string.lower()

def main():
    # Get the two strings from the user
    main_string = input("Enter the main string: ")
    sub_string = input("Enter the string to search for: ")

    # Check if the main string contains the sub string
    if check_containment(main_string, sub_string):
        print(f"'{main_string}' contains '{sub_string}'")
    else:
        print(f"'{main_string}' does not contain '{sub_string}'")

if __name__ == "__main__":
    main()
```
