## Exercise
Create a function that uses interfaces to define a contract. Then, call this function with an input value and print the result.

## Reference Solution
```python
from abc import ABC, abstractmethod

# Define the Calculator interface
class Calculator(ABC):
    @abstractmethod
    def add(self, x, y):
        pass

    @abstractmethod
    def subtract(self, x, y):
        pass

    @abstractmethod
    def multiply(self, x, y):
        pass

    @abstractmethod
    def divide(self, x, y):
        pass

# Implement the Calculator interface
class BasicCalculator(Calculator):
    def add(self, x, y):
        return x + y

    def subtract(self, x, y):
        return x - y

    def multiply(self, x, y):
        return x * y

    def divide(self, x, y):
        if y == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return x / y

# Define the calculate function
def calculate(calculator, x, y, operation):
    """
    Perform a calculation using the provided calculator and operation.

    Args:
    - calculator (Calculator): An object implementing the Calculator interface.
    - x (float): The first operand.
    - y (float): The second operand.
    - operation (str): The operation to perform (add, subtract, multiply, divide).

    Returns:
    - float: The result of the calculation.
    """
    operations = {
        "add": calculator.add,
        "subtract": calculator.subtract,
        "multiply": calculator.multiply,
        "divide": calculator.divide
    }

    if operation not in operations:
        raise ValueError("Invalid operation")

    return operations[operation](x, y)

# Example usage
if __name__ == "__main__":
    calculator = BasicCalculator()
    result = calculate(calculator, 10, 2, "add")
    print(f"Result: {result}")  # Output: Result: 12
```
