## Exercise
Write a program that helps users plan trips using conditional statements, considering factors such as destination, budget, and time of year.

## Reference Solution
```python
# Import necessary libraries
import datetime

# Define a dictionary to store destination information
destinations = {
    "beach": {
        "description": "Relaxing beach vacation",
        "best_time": "summer",
        "budget": "high",
        "accommodations": ["hotel", "resort"],
        "activities": ["swimming", "surfing", "sunbathing"],
        "transportation": ["car", "taxi"]
    },
    "city": {
        "description": "Urban city break",
        "best_time": "spring",
        "budget": "medium",
        "accommodations": ["hotel", "hostel"],
        "activities": ["museum", "shopping", "dining"],
        "transportation": ["public transportation", "walk"]
    },
    "mountains": {
        "description": "Scenic mountain retreat",
        "best_time": "winter",
        "budget": "low",
        "accommodations": ["cabin", "lodge"],
        "activities": ["hiking", "skiing", "camping"],
        "transportation": ["car", "shuttle"]
    }
}

# Define a function to get user input
def get_user_input():
    destination = input("Enter your destination (beach, city, mountains): ")
    budget = input("Enter your budget (low, medium, high): ")
    time_of_year = input("Enter the time of year (spring, summer, winter, fall): ")
    return destination, budget, time_of_year

# Define a function to provide trip recommendations
def provide_recommendations(destination, budget, time_of_year):
    # Check if the destination is valid
    if destination not in destinations:
        print("Invalid destination. Please try again.")
        return

    # Check if the budget is valid
    if budget not in ["low", "medium", "high"]:
        print("Invalid budget. Please try again.")
        return

    # Check if the time of year is valid
    if time_of_year not in ["spring", "summer", "winter", "fall"]:
        print("Invalid time of year. Please try again.")
        return

    # Get the destination information
    destination_info = destinations[destination]

    # Check if the time of year is the best time for the destination
    if time_of_year == destination_info["best_time"]:
        print(f"The {time_of_year} season is the best time to visit {destination}.")
    else:
        print(f"The {time_of_year} season is not the best time to visit {destination}. Consider visiting during the {destination_info['best_time']} season instead.")

    # Check if the budget is suitable for the destination
    if budget == destination_info["budget"]:
        print(f"Your budget of {budget} is suitable for {destination}.")
    else:
        print(f"Your budget of {budget} may not be suitable for {destination}. Consider adjusting your budget to {destination_info['budget']}.")

    # Provide recommendations for accommodations, activities, and transportation
    print(f"Recommended accommodations for {destination}: {', '.join(destination_info['accommodations'])}")
    print(f"Recommended activities for {destination}: {', '.join(destination_info['activities'])}")
    print(f"Recommended transportation for {destination}: {', '.join(destination_info['transportation'])}")

# Main function
def main():
    destination, budget, time_of_year = get_user_input()
    provide_recommendations(destination, budget, time_of_year)

# Run the main function
if __name__ == "__main__":
    main()
```
