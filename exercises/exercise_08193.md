## Exercise
Write a program that implements a Caesar cipher using a for loop.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts or decrypts the input text using a Caesar cipher.

    Args:
        text (str): The input text to be encrypted or decrypted.
        shift (int): The number of positions to shift the letters.

    Returns:
        str: The encrypted or decrypted text.
    """
    result = ""

    for char in text:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            result += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            result += char

    return result

def main():
    text = input("Enter the text to be encrypted or decrypted: ")
    shift = int(input("Enter the shift value: "))

    encrypted_text = caesar_cipher(text, shift)
    decrypted_text = caesar_cipher(encrypted_text, -shift)

    print(f"Encrypted text: {encrypted_text}")
    print(f"Decrypted text: {decrypted_text}")

if __name__ == "__main__":
    main()
```
