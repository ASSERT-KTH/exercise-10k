## Exercise
Create a function that catches and handles exceptions raised by a given function.

## Reference Solution
```python
def catch_exceptions(func):
    """
    A decorator function to catch and handle exceptions raised by a given function.

    Args:
        func (function): The function to be decorated.

    Returns:
        function: A wrapper function that catches and handles exceptions.
    """

    # Define a wrapper function that will wrap the original function
    def wrapper(*args, **kwargs):
        
        # Try to execute the original function
        try:
            # Call the original function with the provided arguments
            return func(*args, **kwargs)
        
        # Catch all exceptions
        except Exception as e:
            # Print the exception message
            print(f"An error occurred: {str(e)}")
            
            # You can also log the exception or perform any other action
            # For example, you can raise a custom exception or return a default value
            
            # Return None to indicate that an exception occurred
            return None
    
    # Return the wrapper function
    return wrapper

# Example usage:

# Define a function that may raise an exception
@catch_exceptions
def divide(a, b):
    """
    Divide two numbers.

    Args:
        a (float): The dividend.
        b (float): The divisor.

    Returns:
        float: The quotient.
    """
    return a / b

# Test the function
print(divide(10, 2))  # Should print: 5.0
print(divide(10, 0))  # Should print: An error occurred: division by zero
```
