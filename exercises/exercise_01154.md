## Exercise
Create a program that lets the user manage a to-do list. The program should let the user add, remove, and view tasks.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self):
        """Add a task to the to-do list"""
        task = input("Enter a task: ")
        self.tasks.append(task)
        print(f"Task '{task}' added successfully.")

    def remove_task(self):
        """Remove a task from the to-do list"""
        if not self.tasks:
            print("No tasks to remove.")
            return

        print("Available tasks:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")

        task_number = input("Enter the task number to remove: ")
        try:
            task_number = int(task_number)
            if task_number < 1 or task_number > len(self.tasks):
                print("Invalid task number.")
                return
            task = self.tasks.pop(task_number - 1)
            print(f"Task '{task}' removed successfully.")
        except ValueError:
            print("Invalid task number.")

    def view_tasks(self):
        """View all tasks in the to-do list"""
        if not self.tasks:
            print("No tasks available.")
            return

        print("Available tasks:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")


def main():
    todo_list = ToDoList()

    while True:
        print("\nTo-Do List Manager Menu:")
        print("1. Add task")
        print("2. Remove task")
        print("3. View tasks")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            todo_list.add_task()
        elif choice == "2":
            todo_list.remove_task()
        elif choice == "3":
            todo_list.view_tasks()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
