## Exercise
Write a function that takes an array as input and returns the sum of its elements.

## Reference Solution
```python
def sum_array_elements(array):
    """
    This function calculates the sum of elements in an array.

    Args:
        array (list): A list of numbers.

    Returns:
        int: The sum of all elements in the array.

    Raises:
        TypeError: If the input is not a list or if the list contains non-numeric values.
    """
    # Check if the input is a list
    if not isinstance(array, list):
        raise TypeError("Input must be a list.")

    # Initialize the sum variable to 0
    total_sum = 0

    # Iterate over each element in the array
    for element in array:
        # Check if the element is a number (int or float)
        if isinstance(element, (int, float)):
            # Add the element to the total sum
            total_sum += element
        else:
            raise TypeError("The list must only contain numeric values.")

    # Return the total sum
    return total_sum


# Example usage:
array = [1, 2, 3, 4, 5]
result = sum_array_elements(array)
print("The sum of the array elements is:", result)
```

```python
def sum_array_elements(array):
    """
    This function calculates the sum of elements in an array.

    Args:
        array (list): A list of numbers.

    Returns:
        int: The sum of all elements in the array.

    Raises:
        TypeError: If the input is not a list or if the list contains non-numeric values.
    """
    # Check if the input is a list
    if not isinstance(array, list):
        raise TypeError("Input must be a list.")

    # Check if all elements in the list are numbers
    if not all(isinstance(element, (int, float)) for element in array):
        raise TypeError("The list must only contain numeric values.")

    # Return the sum of the array elements
    return sum(array)


# Example usage:
array = [1, 2, 3, 4, 5]
result = sum_array_elements(array)
print("The sum of the array elements is:", result)
```
