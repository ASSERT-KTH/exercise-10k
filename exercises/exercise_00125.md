## Exercise
Write a program that takes an integer as input and outputs its modulus (absolute value).

def absolute_value(n):
    """
    Returns the absolute value of an integer.

    Args:
        n (int): The input integer.

    Returns:
        int: The absolute value of the input integer.
    """
    return abs(n)

# Example usage:
num = int(input("Enter an integer: "))
print("The absolute value of {} is {}".format(num, absolute_value(num)))

def absolute_value(n):
    """
    Returns the absolute value of an integer.

    Args:
        n (int): The input integer.

    Returns:
        int: The absolute value of the input integer.
    """
    if n < 0:
        return -n
    else:
        return n

# Example usage:
num = int(input("Enter an integer: "))
print("The absolute value of {} is {}".format(num, absolute_value(num)))
