## Exercise
Create a program to convert weight from pounds to kilograms using conditional statements

## Reference Solution
```python
def convert_weight(pounds):
    """
    Converts weight from pounds to kilograms.

    Args:
        pounds (float): Weight in pounds.

    Returns:
        float: Weight in kilograms.
    """
    kilograms = pounds * 0.45359237

    # Conditional statements to determine the unit of measurement
    if pounds < 0:
        print("Error: Weight cannot be negative.")
        return None
    elif pounds == 0:
        print("Weight is 0 pounds, which is equivalent to 0 kilograms.")
    elif pounds < 1:
        print(f"{pounds} pounds is equivalent to {kilograms} kilograms (approximately).")
    else:
        print(f"{pounds} pounds is equivalent to {kilograms:.2f} kilograms (approximately).")

    return kilograms

def main():
    # Get user input
    pounds = float(input("Enter weight in pounds: "))

    # Convert weight
    convert_weight(pounds)

if __name__ == "__main__":
    main()
```
