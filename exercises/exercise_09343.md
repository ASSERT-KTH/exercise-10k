## Exercise
Write a program to demonstrate the difference between static and dynamic memory allocation.

## Reference Solution
```python
#include <stdio.h>
#include <stdlib.h>

// Function to demonstrate static memory allocation
void staticMemoryAllocation() {
    int arr[5];  // Static memory allocation

    // Assign values to the array
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 10;
    }

    // Print the array
    printf("Static Memory Allocation:\n");
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

// Function to demonstrate dynamic memory allocation
void dynamicMemoryAllocation() {
    int* arr = (int*) malloc(5 * sizeof(int));  // Dynamic memory allocation

    // Check if memory allocation was successful
    if (arr == NULL) {
        printf("Memory allocation failed\n");
        return;
    }

    // Assign values to the array
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 10;
    }

    // Print the array
    printf("Dynamic Memory Allocation:\n");
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    // Free the allocated memory
    free(arr);
}

int main() {
    staticMemoryAllocation();
    dynamicMemoryAllocation();

    return 0;
}
```
