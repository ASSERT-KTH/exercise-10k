## Exercise
Write a program that uses a while loop to let the user guess a secret number between 1 and 10.

## Reference Solution
```python
# Secret Number Guessing Game

def main():
    # Define the secret number
    secret_number = 7  # You can change this number

    # Initialize the number of attempts
    attempts = 0

    print("Welcome to the Secret Number Guessing Game!")
    print("I'm thinking of a number between 1 and 10.")

    # Start the game loop
    while True:
        # Ask the user for their guess
        user_guess = input("What's your guess? (or 'q' to quit): ")

        # Check if the user wants to quit
        if user_guess.lower() == 'q':
            print("Thanks for playing!")
            break

        # Try to convert the user's guess to an integer
        try:
            user_guess = int(user_guess)
        except ValueError:
            print("Invalid input! Please enter a number.")
            continue

        # Check if the guess is in the valid range
        if user_guess < 1 or user_guess > 10:
            print("Please enter a number between 1 and 10.")
            continue

        # Increment the number of attempts
        attempts += 1

        # Check if the guess is correct
        if user_guess == secret_number:
            print(f"Congratulations! You found the secret number in {attempts} attempts.")
            break
        elif user_guess < secret_number:
            print("Too low! Try again.")
        else:
            print("Too high! Try again.")

if __name__ == "__main__":
    main()
```
