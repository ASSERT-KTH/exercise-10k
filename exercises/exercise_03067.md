## Exercise
Create a program that simulates a toaster with different settings. The user can choose from light, medium, or dark toast, and the program should print out the corresponding toast settings.

## Reference Solution
```python
class Toaster:
    def __init__(self):
        self.settings = {
            "light": {"heat": 3, "time": 2},
            "medium": {"heat": 5, "time": 3},
            "dark": {"heat": 7, "time": 4}
        }

    def toast(self, setting):
        if setting in self.settings:
            print(f"Toasting to {setting} setting:")
            print(f"Heat: {self.settings[setting]['heat']}")
            print(f"Time: {self.settings[setting]['time']} minutes")
        else:
            print("Invalid setting. Please choose from light, medium, or dark.")

def main():
    toaster = Toaster()
    while True:
        print("Toaster Settings:")
        print("1. Light")
        print("2. Medium")
        print("3. Dark")
        print("4. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            toaster.toast("light")
        elif choice == "2":
            toaster.toast("medium")
        elif choice == "3":
            toaster.toast("dark")
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
