def solve_linear_inequality(a, b, c, inequality_type):
    """
    Solves a linear inequality of the form ax + b <= c or ax + b >= c.

    Args:
        a (float): Coefficient of x.
        b (float): Constant term.
        c (float): Right-hand side of the inequality.
        inequality_type (str): Type of inequality, either '<=' or '>='.

    Returns:
        str: Solution to the inequality.
    """
    if inequality_type not in ['<=', '>=']:
        raise ValueError("Invalid inequality type. Must be '<=' or '>='.")

    # Subtract b from both sides
    new_c = c - b

    # Divide both sides by a
    if a == 0:
        if new_c == 0:
            return "All values of x satisfy the inequality."
        else:
            return "No values of x satisfy the inequality."
    else:
        x_value = new_c / a

        if inequality_type == '<=':
            if a > 0:
                return f"x <= {x_value}"
            else:
                return f"x >= {x_value}"
        else:
            if a > 0:
                return f"x >= {x_value}"
            else:
                return f"x <= {x_value}"
