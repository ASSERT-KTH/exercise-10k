## Exercise
Create a program that takes an array of integers as input and returns a new array with the same elements, but in a random order. The program should also handle cases where the input array is empty or contains only one element.

## Reference Solution
```python
import random

def shuffle_list(input_list):
    if not input_list:
        return []
    if len(input_list) <= 1:
        return input_list
    random.shuffle(input_list)
    return input_list

def shuffle_list_slice(input_list):
    if not input_list:
        return []
    if len(input_list) <= 1:
        return input_list
    random.shuffle(input_list)
    return input_list[:]

def shuffle_list_new(input_list):
    if not input_list:
        return []
    if len(input_list) <= 1:
        return input_list
    new_list = input_list[:]
    random.shuffle(new_list)
    return new_list

# Example usage
input_list = [1, 2, 3, 4, 5]
print(shuffle_list(input_list))
print(shuffle_list_slice(input_list))
print(shuffle_list_new(input_list))
```
