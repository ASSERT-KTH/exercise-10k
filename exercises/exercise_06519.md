## Exercise
Write a program to initialize variables with default values.

## Reference Solution
```python
# Initialize variables with default values
def initialize_variables():
    # Initialize variables with default values
    name = "John Doe"
    age = 30
    city = "New York"
    country = "USA"

    # Print the default values
    print("Default Values:")
    print(f"Name: {name}")
    print(f"Age: {age}")
    print(f"City: {city}")
    print(f"Country: {country}")

    # Return the default values as a dictionary
    return {
        "name": name,
        "age": age,
        "city": city,
        "country": country
    }

# Main function
def main():
    default_values = initialize_variables()
    print("\nReturned Dictionary:")
    print(default_values)

if __name__ == "__main__":
    main()
```
