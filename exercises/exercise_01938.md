## Exercise
Write a program that demonstrates the immutability of variables (e.g., tuples, strings).

## Reference Solution
```python
def main():
    # Create a tuple
    my_tuple = (1, 2, 3)
    print("Original Tuple:", my_tuple)

    # Attempt to modify the tuple
    try:
        my_tuple[0] = 10
    except TypeError as e:
        print("Error:", e)

    # Create a string
    my_string = "Hello, World!"
    print("\nOriginal String:", my_string)

    # Attempt to modify the string
    try:
        my_string[0] = 'J'
    except TypeError as e:
        print("Error:", e)

    # Create a new tuple and string to demonstrate immutability
    new_tuple = my_tuple + (4, 5)
    new_string = my_string.replace('H', 'J')
    print("\nNew Tuple:", new_tuple)
    print("New String:", new_string)

if __name__ == "__main__":
    main()
```
