## Exercise
Write a program to validate user input data against a set of rules or constraints, using data types and type checking.

## Reference Solution
```python
from datetime import datetime

class Validator:
    def __init__(self):
        self.rules = {
            "name": self.validate_string,
            "age": self.validate_integer,
            "email": self.validate_email,
            "phone": self.validate_phone,
            "date_of_birth": self.validate_date
        }

    def validate_string(self, value):
        """Validate a string input."""
        if not isinstance(value, str):
            return False
        return len(value) > 0

    def validate_integer(self, value):
        """Validate an integer input."""
        if not isinstance(value, int):
            try:
                int(value)
            except ValueError:
                return False
        return True

    def validate_email(self, value):
        """Validate an email input."""
        if not isinstance(value, str):
            return False
        if "@" not in value:
            return False
        return True

    def validate_phone(self, value):
        """Validate a phone number input."""
        if not isinstance(value, str):
            return False
        if not value.isdigit() or len(value) != 10:
            return False
        return True

    def validate_date(self, value):
        """Validate a date input."""
        if not isinstance(value, str):
            return False
        try:
            datetime.strptime(value, "%Y-%m-%d")
        except ValueError:
            return False
        return True

    def validate_input(self, field, value):
        """Validate user input against a set of rules."""
        if field in self.rules:
            return self.rules[field](value)
        return False


def main():
    validator = Validator()

    # Get user input
    name = input("Enter your name: ")
    age = input("Enter your age: ")
    email = input("Enter your email: ")
    phone = input("Enter your phone number: ")
    date_of_birth = input("Enter your date of birth (YYYY-MM-DD): ")

    # Validate user input
    print("Validation Results:")
    print(f"Name: {validator.validate_input('name', name)}")
    print(f"Age: {validator.validate_input('age', age)}")
    print(f"Email: {validator.validate_input('email', email)}")
    print(f"Phone: {validator.validate_input('phone', phone)}")
    print(f"Date of Birth: {validator.validate_input('date_of_birth', date_of_birth)}")


if __name__ == "__main__":
    main()
```
