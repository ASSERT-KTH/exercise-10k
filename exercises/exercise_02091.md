## Exercise
Write a program to calculate the square root and cube root of a given number.

## Reference Solution
```python
import math

def calculate_roots(number):
    """
    Calculate the square root and cube root of a given number.

    Args:
        number (float): The input number.

    Returns:
        tuple: A tuple containing the square root and cube root of the input number.
    """
    if number < 0:
        raise ValueError("Input number cannot be negative")

    square_root = math.sqrt(number)
    cube_root = round(number ** (1/3), 3)  # Rounded to 3 decimal places

    return square_root, cube_root

def main():
    number = float(input("Enter a number: "))
    try:
        square_root, cube_root = calculate_roots(number)
        print(f"Square root: {square_root}")
        print(f"Cube root: {cube_root}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
