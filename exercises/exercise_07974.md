## Exercise
Write a program that calculates the number of coins needed to make change for a given amount.

## Reference Solution
```python
def calculate_change(amount, denominations):
    """
    Calculate the minimum number of coins needed to make change for a given amount.

    Args:
        amount (int): The amount to make change for.
        denominations (list): A list of available coin denominations.

    Returns:
        int: The minimum number of coins needed to make change.
        dict: A dictionary mapping coin denominations to their respective counts.
    """
    # Sort the denominations in descending order
    denominations.sort(reverse=True)

    # Initialize the coin count dictionary
    coin_count = {denomination: 0 for denomination in denominations}

    # Calculate the change
    remaining_amount = amount
    for denomination in denominations:
        count = remaining_amount // denomination
        if count > 0:
            coin_count[denomination] = count
            remaining_amount -= denomination * count

    # Calculate the total number of coins
    total_coins = sum(coin_count.values())

    return total_coins, coin_count


def main():
    # Define the available coin denominations
    denominations = [25, 10, 5, 1]  # quarters, dimes, nickels, pennies

    # Get the amount from the user
    amount = int(input("Enter the amount to make change for: "))

    # Calculate the change
    total_coins, coin_count = calculate_change(amount, denominations)

    # Print the results
    print(f"Minimum number of coins needed: {total_coins}")
    print("Coin count:")
    for denomination, count in coin_count.items():
        if count > 0:
            print(f"{denomination} cents: {count}")


if __name__ == "__main__":
    main()
```
