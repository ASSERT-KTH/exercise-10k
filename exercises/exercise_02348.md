## Exercise
Write a program to take a list of tasks as input and allow the user to add, remove, and mark tasks as completed

## Reference Solution
```python
class Task:
    def __init__(self, name):
        self.name = name
        self.completed = False

    def mark_as_completed(self):
        self.completed = True

    def __str__(self):
        status = "Completed" if self.completed else "Not Completed"
        return f"{self.name}: {status}"


class TaskManager:
    def __init__(self):
        self.tasks = []

    def add_task(self, name):
        self.tasks.append(Task(name))
        print(f"Task '{name}' added successfully.")

    def remove_task(self, task_index):
        try:
            del self.tasks[task_index]
            print("Task removed successfully.")
        except IndexError:
            print("Invalid task index.")

    def mark_task_as_completed(self, task_index):
        try:
            self.tasks[task_index].mark_as_completed()
            print("Task marked as completed.")
        except IndexError:
            print("Invalid task index.")

    def display_tasks(self):
        if not self.tasks:
            print("No tasks available.")
        else:
            for index, task in enumerate(self.tasks):
                print(f"{index + 1}. {task}")


def main():
    task_manager = TaskManager()

    while True:
        print("\nTask Manager Menu:")
        print("1. Add Task")
        print("2. Remove Task")
        print("3. Mark Task as Completed")
        print("4. Display Tasks")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter task name: ")
            task_manager.add_task(name)
        elif choice == "2":
            task_manager.display_tasks()
            task_index = int(input("Enter task number to remove: ")) - 1
            task_manager.remove_task(task_index)
        elif choice == "3":
            task_manager.display_tasks()
            task_index = int(input("Enter task number to mark as completed: ")) - 1
            task_manager.mark_task_as_completed(task_index)
        elif choice == "4":
            task_manager.display_tasks()
        elif choice == "5":
            print("Exiting Task Manager.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
