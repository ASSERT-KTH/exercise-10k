## Exercise
Write a function to check if two lists are equal.

## Reference Solution
```python
def are_lists_equal(list1, list2):
    """
    Check if two lists are equal.

    Args:
        list1 (list): The first list to compare.
        list2 (list): The second list to compare.

    Returns:
        bool: True if the lists are equal, False otherwise.
    """
    return list1 == list2

# Example usage:
def main():
    list1 = [1, 2, 3]
    list2 = [1, 2, 3]
    list3 = [1, 2, 4]

    print(are_lists_equal(list1, list2))  # Expected output: True
    print(are_lists_equal(list1, list3))  # Expected output: False

if __name__ == "__main__":
    main()
```
