## Exercise
Create a program that takes a list of student scores as input and rearranges them in ascending order. The program should also calculate the average score.

## Reference Solution
```python
def sort_and_average_scores(scores):
    """
    Sorts a list of scores in ascending order and calculates the average score.

    Args:
        scores (list): A list of student scores.

    Returns:
        tuple: A tuple containing the sorted list of scores and the average score.
    """
    # Sort the scores in ascending order
    sorted_scores = sorted(scores)

    # Calculate the average score
    average_score = sum(sorted_scores) / len(sorted_scores) if sorted_scores else 0

    return sorted_scores, average_score

def main():
    # Example usage:
    scores = [85, 90, 78, 92, 88, 76, 95, 89]
    sorted_scores, average_score = sort_and_average_scores(scores)

    # Print the sorted scores
    print("Sorted Scores:")
    for i, score in enumerate(sorted_scores, start=1):
        print(f"Student {i}: {score}")

    # Print the average score
    print(f"\nAverage Score: {average_score:.2f}")

if __name__ == "__main__":
    main()
```
