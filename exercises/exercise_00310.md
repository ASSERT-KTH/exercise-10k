## Exercise
Write a function that inserts a new element at a specified position in a given list.

## Reference Solution
```python
def insert_element(input_list, index, element):
    """
    Inserts a new element at a specified position in a given list.

    Args:
        input_list (list): The input list.
        index (int): The index at which the element should be inserted.
        element: The element to be inserted.

    Returns:
        list: The updated list with the new element inserted.

    Raises:
        IndexError: If the index is out of range.
    """
    if index < 0 or index > len(input_list):
        raise IndexError("Index out of range")

    # Create a copy of the input list to avoid modifying the original list
    updated_list = input_list.copy()

    # Insert the element at the specified index
    updated_list.insert(index, element)

    return updated_list

# Example usage:
if __name__ == "__main__":
    input_list = [1, 2, 3, 5]
    index = 3
    element = 4

    updated_list = insert_element(input_list, index, element)
    print("Updated List:", updated_list)
```
