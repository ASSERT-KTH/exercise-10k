## Exercise
Write a program that allows the user to add, remove, and search for phone numbers.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.phonebook = {}

    def add_phone_number(self, name, phone_number):
        """Add a phone number to the phonebook."""
        if name in self.phonebook:
            print(f"A phone number for {name} already exists. Do you want to update it? (yes/no)")
            response = input().lower()
            if response == "yes":
                self.phonebook[name] = phone_number
                print(f"Phone number for {name} updated successfully.")
            else:
                print("Operation cancelled.")
        else:
            self.phonebook[name] = phone_number
            print(f"Phone number for {name} added successfully.")

    def remove_phone_number(self, name):
        """Remove a phone number from the phonebook."""
        if name in self.phonebook:
            del self.phonebook[name]
            print(f"Phone number for {name} removed successfully.")
        else:
            print(f"No phone number found for {name}.")

    def search_phone_number(self, name):
        """Search for a phone number in the phonebook."""
        if name in self.phonebook:
            print(f"Phone number for {name}: {self.phonebook[name]}")
        else:
            print(f"No phone number found for {name}.")

    def display_phonebook(self):
        """Display all phone numbers in the phonebook."""
        if self.phonebook:
            print("Phonebook:")
            for name, phone_number in self.phonebook.items():
                print(f"{name}: {phone_number}")
        else:
            print("Phonebook is empty.")


def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook Menu:")
        print("1. Add phone number")
        print("2. Remove phone number")
        print("3. Search phone number")
        print("4. Display phonebook")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter name: ")
            phone_number = input("Enter phone number: ")
            phonebook.add_phone_number(name, phone_number)
        elif choice == "2":
            name = input("Enter name: ")
            phonebook.remove_phone_number(name)
        elif choice == "3":
            name = input("Enter name: ")
            phonebook.search_phone_number(name)
        elif choice == "4":
            phonebook.display_phonebook()
        elif choice == "5":
            print("Exiting program.")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
