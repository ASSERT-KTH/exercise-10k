def find_min_value(array):
    if len(array) == 0:
        return None  # or raise an exception, depending on your requirements
    min_value = array[0]
    for value in array[1:]:
        if value < min_value:
            min_value = value
    return min_value

# Example usage:
array = [10, 20, 3, 40, 5]
min_value = find_min_value(array)
print("Minimum value in the array:", min_value)
