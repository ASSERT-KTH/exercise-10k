## Exercise
Develop a program that determines whether a student is eligible for university admission based on their grades and test scores. The program should use if/else statements to determine the student's eligibility.

## Reference Solution
```python
def determine_eligibility(gpa, sat_score, act_score):
    """
    Determine whether a student is eligible for university admission.
    
    Args:
        gpa (float): Student's high school GPA
        sat_score (int): Student's SAT score
        act_score (int): Student's ACT score
    
    Returns:
        bool: Whether the student is eligible for university admission
    """
    if gpa >= 3.0:
        # Check SAT score if available
        if sat_score >= 1000:
            return True
        # Check ACT score if available
        elif act_score >= 20:
            return True
        else:
            return False
    else:
        return False

def main():
    # Example usage
    gpa = 3.5
    sat_score = 1200
    act_score = 25
    
    eligible = determine_eligibility(gpa, sat_score, act_score)
    
    if eligible:
        print("The student is eligible for university admission.")
    else:
        print("The student is not eligible for university admission.")

if __name__ == "__main__":
    main()
```
