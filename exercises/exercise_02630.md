## Exercise
Create a program that takes a task as input and generates a to-do list entry.

## Reference Solution
```python
class ToDoListEntry:
    def __init__(self, task, due_date=None, priority=None):
        """
        Initialize a to-do list entry.

        Args:
        - task (str): The task to be completed.
        - due_date (str): The due date for the task (optional).
        - priority (str): The priority of the task (optional).
        """
        self.task = task
        self.due_date = due_date
        self.priority = priority

    def __str__(self):
        """
        Return a string representation of the to-do list entry.
        """
        output = f"Task: {self.task}"
        if self.due_date:
            output += f"\nDue Date: {self.due_date}"
        if self.priority:
            output += f"\nPriority: {self.priority}"
        return output


def main():
    # Get user input
    task = input("Enter the task: ")
    due_date = input("Enter the due date (optional): ")
    if due_date.strip() == "":
        due_date = None
    priority = input("Enter the priority (optional): ")
    if priority.strip() == "":
        priority = None

    # Create a to-do list entry
    entry = ToDoListEntry(task, due_date, priority)

    # Print the to-do list entry
    print("\nTo-Do List Entry:")
    print(entry)


if __name__ == "__main__":
    main()
```
