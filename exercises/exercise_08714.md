## Exercise
Generate and print out a random number between two user-defined numbers.

## Reference Solution
```python
import random

def get_user_input():
    """Get user input for the range of numbers"""
    while True:
        try:
            num1 = int(input("Enter the first number: "))
            num2 = int(input("Enter the second number: "))
            if num1 < num2:
                return num1, num2
            else:
                print("The first number should be less than the second number.")
        except ValueError:
            print("Invalid input. Please enter a valid integer.")

def generate_random_number(num1, num2):
    """Generate a random number between num1 and num2"""
    return random.randint(num1, num2)

def main():
    num1, num2 = get_user_input()
    random_number = generate_random_number(num1, num2)
    print(f"A random number between {num1} and {num2} is: {random_number}")

if __name__ == "__main__":
    main()
```
