## Exercise
Create a program that takes two numbers as input, adds them, subtracts the second number from the first, multiplies them, and divides the first number by the second. The program should then output the results of these operations in a specific format.

#include <iostream>

int main() {
    int num1, num2;
    int sumResult, subtractResult, multiplyResult;
    float divideResult;

    std::cout << "Enter the first number: ";
    std::cin >> num1;
    std::cout << "Enter the second number: ";
    std::cin >> num2;

    sumResult = num1 + num2;
    subtractResult = num1 - num2;
    multiplyResult = num1 * num2;
    divideResult = static_cast<float>(num1) / num2;

    std::cout << "Sum: " << sumResult << std::endl;
    std::cout << "Subtract: " << subtractResult << std::endl;
    std::cout << "Multiply: " << multiplyResult << std::endl;
    std::cout << "Divide: " << divideResult << std::endl;

    return 0;
}
