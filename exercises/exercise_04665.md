## Exercise
Write a program that takes two strings as input and checks if the first string is a subsequence of the second string.

def is_subsequence(s1: str, s2: str) -> bool:
    """
    Checks if s1 is a subsequence of s2.

    Args:
        s1 (str): The potential subsequence string.
        s2 (str): The original string.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """
    i, j = 0, 0
    while i < len(s1) and j < len(s2):
        if s1[i] == s2[j]:
            i += 1
        j += 1
    return i == len(s1)


def main():
    # Example usage:
    s1 = "abc"
    s2 = "ahbgdc"
    print(f"Is '{s1}' a subsequence of '{s2}'? {is_subsequence(s1, s2)}")

    s1 = "axc"
    s2 = "ahbgdc"
    print(f"Is '{s1}' a subsequence of '{s2}'? {is_subsequence(s1, s2)}")


if __name__ == "__main__":
    main()
