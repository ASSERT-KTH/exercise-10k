from collections import namedtuple

# Define a function to create and print student information
def print_student_info():
    # Create a named tuple to store student information
    Student = namedtuple("Student", ["name", "age", "major", "gpa"])
    student_info = Student("John Doe", 20, "Computer Science", 3.8)

    # Print out the student information
    print("Student Name:", student_info.name)
    print("Age:", student_info.age)
    print("Major:", student_info.major)
    print("GPA:", student_info.gpa)

# Call the function to print student information
print_student_info()
