## Exercise
Write a program that takes input from the user and writes it to a URL.

## Reference Solution
```python
import requests

def write_to_url(url, data):
    """
    Writes user input to a specified URL.

    Args:
        url (str): The URL to write to.
        data (str): The user input to write.

    Returns:
        bool: True if the write was successful, False otherwise.
    """
    try:
        response = requests.post(url, data=data)
        response.raise_for_status()  # Raise an exception for HTTP errors
        return True
    except requests.exceptions.RequestException as e:
        print(f"Error writing to URL: {e}")
        return False

def main():
    url = input("Enter the URL to write to: ")
    data = input("Enter the data to write: ")

    if write_to_url(url, data):
        print("Data written to URL successfully.")
    else:
        print("Failed to write data to URL.")

if __name__ == "__main__":
    main()
```
