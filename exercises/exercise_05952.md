## Exercise
Develop a program that asks the user a series of questions and keeps track of the score. The program should ask for the question, the possible answers, and the correct answer, and then print out the final score.

## Reference Solution
```python
class Question:
    def __init__(self, question, possible_answers, correct_answer):
        self.question = question
        self.possible_answers = possible_answers
        self.correct_answer = correct_answer

    def ask_question(self):
        print(f"Question: {self.question}")
        for i, answer in enumerate(self.possible_answers):
            print(f"{i + 1}. {answer}")
        user_answer = input("Enter the number of your answer: ")
        try:
            user_answer = int(user_answer)
            if user_answer < 1 or user_answer > len(self.possible_answers):
                print("Invalid answer. Please enter a number corresponding to one of the answers.")
                return self.ask_question()
            if self.possible_answers[user_answer - 1] == self.correct_answer:
                print("Correct!\n")
                return True
            else:
                print(f"Sorry, that's incorrect. The correct answer was {self.correct_answer}.\n")
                return False
        except ValueError:
            print("Invalid input. Please enter a number.")
            return self.ask_question()


def main():
    questions = []
    num_questions = int(input("How many questions do you want to ask? "))

    for i in range(num_questions):
        question = input(f"Enter question {i + 1}: ")
        num_answers = int(input("How many possible answers does this question have? "))
        possible_answers = []
        for j in range(num_answers):
            possible_answers.append(input(f"Enter possible answer {j + 1}: "))
        correct_answer = input("Enter the correct answer: ")
        questions.append(Question(question, possible_answers, correct_answer))

    score = 0
    for i, question in enumerate(questions):
        print(f"Question {i + 1}:")
        if question.ask_question():
            score += 1

    print(f"\nFinal score: {score} out of {num_questions}")


if __name__ == "__main__":
    main()
```
