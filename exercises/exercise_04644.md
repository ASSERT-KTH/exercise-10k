## Exercise
Create a program that manages a bank account. The user can deposit or withdraw money, and the program displays the current balance. If the user tries to withdraw more money than the balance, the program displays an error message.

## Reference Solution
```python
class BankAccount:
    def __init__(self, initial_balance=0):
        """
        Initializes a new bank account with an optional initial balance.

        Args:
            initial_balance (float, optional): The initial balance of the account. Defaults to 0.
        """
        self.balance = initial_balance

    def deposit(self, amount):
        """
        Deposits money into the account.

        Args:
            amount (float): The amount to deposit.
        """
        if amount > 0:
            self.balance += amount
            print(f"Deposited ${amount:.2f}. Current balance: ${self.balance:.2f}")
        else:
            print("Invalid deposit amount. Please enter a positive value.")

    def withdraw(self, amount):
        """
        Withdraws money from the account.

        Args:
            amount (float): The amount to withdraw.
        """
        if 0 < amount <= self.balance:
            self.balance -= amount
            print(f"Withdrew ${amount:.2f}. Current balance: ${self.balance:.2f}")
        elif amount <= 0:
            print("Invalid withdrawal amount. Please enter a positive value.")
        else:
            print("Insufficient funds. Withdrawal not allowed.")

    def display_balance(self):
        """
        Displays the current balance of the account.
        """
        print(f"Current balance: ${self.balance:.2f}")


def main():
    account = BankAccount()

    while True:
        print("\nBank Account Management")
        print("1. Deposit money")
        print("2. Withdraw money")
        print("3. Display balance")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            amount = float(input("Enter the amount to deposit: $"))
            account.deposit(amount)
        elif choice == "2":
            amount = float(input("Enter the amount to withdraw: $"))
            account.withdraw(amount)
        elif choice == "3":
            account.display_balance()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
