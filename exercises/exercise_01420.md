## Exercise
Create a program that converts a color from RGB to HEX or vice versa. If the user inputs 'rgb', convert to HEX, if 'hex', convert to RGB.

## Reference Solution
```python
def rgb_to_hex(rgb):
    """
    Convert RGB to HEX.

    Args:
        rgb (list): A list containing RGB values.

    Returns:
        str: The HEX representation of the color.
    """
    return '#{:02x}{:02x}{:02x}'.format(rgb[0], rgb[1], rgb[2])


def hex_to_rgb(hex_color):
    """
    Convert HEX to RGB.

    Args:
        hex_color (str): A string containing the HEX representation of the color.

    Returns:
        list: A list containing RGB values.
    """
    hex_color = hex_color.lstrip('#')
    return tuple(int(hex_color[i:i+2], 16) for i in (0, 2, 4))


def main():
    while True:
        conversion_type = input("Enter 'rgb' to convert to HEX or 'hex' to convert to RGB, 'quit' to exit: ")
        if conversion_type.lower() == 'rgb':
            rgb = input("Enter RGB values separated by comma (e.g., 255,255,255): ")
            rgb_values = [int(x) for x in rgb.split(',')]
            print(f"HEX representation: {rgb_to_hex(rgb_values)}")
        elif conversion_type.lower() == 'hex':
            hex_color = input("Enter HEX color (e.g., #ffffff): ")
            print(f"RGB representation: {hex_to_rgb(hex_color)}")
        elif conversion_type.lower() == 'quit':
            break
        else:
            print("Invalid input. Please try again.")

if __name__ == "__main__":
    main()
```
