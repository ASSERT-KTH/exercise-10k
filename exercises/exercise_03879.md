## Exercise
Given two dictionaries with names of cities as keys and their populations as values, write a program to check if one dictionary is a subset of the other and print the result.

## Reference Solution
```python
def is_subset(dict1, dict2):
    """
    Check if dict1 is a subset of dict2.
    
    Args:
        dict1 (dict): The dictionary to check.
        dict2 (dict): The dictionary to check against.
    
    Returns:
        bool: True if dict1 is a subset of dict2, False otherwise.
    """
    return all(key in dict2 and dict2[key] == value for key, value in dict1.items())

def main():
    # Example dictionaries
    city_populations1 = {
        "New York": 8537673,
        "Los Angeles": 3990456,
        "Chicago": 2720546
    }
    
    city_populations2 = {
        "New York": 8537673,
        "Los Angeles": 3990456,
        "Chicago": 2720546,
        "Houston": 2319092
    }
    
    # Check if city_populations1 is a subset of city_populations2
    result = is_subset(city_populations1, city_populations2)
    print(f"Is city_populations1 a subset of city_populations2? {result}")
    
    # Check if city_populations2 is a subset of city_populations1
    result = is_subset(city_populations2, city_populations1)
    print(f"Is city_populations2 a subset of city_populations1? {result}")

if __name__ == "__main__":
    main()
```
