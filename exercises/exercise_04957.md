## Exercise
Write a program that identifies the type of shape (circle, square, triangle, etc.) based on user input using conditional statements.

## Reference Solution
```python
def get_shape_type():
    print("Select a shape type:")
    print("1. Circle")
    print("2. Square")
    print("3. Triangle")
    print("4. Rectangle")

    choice = input("Enter your choice (1/2/3/4): ")

    if choice == "1":
        return "Circle"
    elif choice == "2":
        return "Square"
    elif choice == "3":
        return "Triangle"
    elif choice == "4":
        return "Rectangle"
    else:
        return "Invalid choice"

def get_circle_properties():
    radius = float(input("Enter the radius of the circle: "))
    return f"Circle with radius {radius}"

def get_square_properties():
    side = float(input("Enter the side length of the square: "))
    return f"Square with side length {side}"

def get_triangle_properties():
    print("Select a triangle type:")
    print("1. Equilateral")
    print("2. Isosceles")
    print("3. Scalene")
    triangle_choice = input("Enter your choice (1/2/3): ")

    if triangle_choice == "1":
        side = float(input("Enter the side length of the equilateral triangle: "))
        return f"Equilateral triangle with side length {side}"
    elif triangle_choice == "2":
        side1 = float(input("Enter the length of the equal sides of the isosceles triangle: "))
        side2 = float(input("Enter the length of the base of the isosceles triangle: "))
        return f"Isosceles triangle with equal sides {side1} and base {side2}"
    elif triangle_choice == "3":
        side1 = float(input("Enter the length of the first side of the scalene triangle: "))
        side2 = float(input("Enter the length of the second side of the scalene triangle: "))
        side3 = float(input("Enter the length of the third side of the scalene triangle: "))
        return f"Scalene triangle with sides {side1}, {side2}, and {side3}"
    else:
        return "Invalid triangle choice"

def get_rectangle_properties():
    length = float(input("Enter the length of the rectangle: "))
    width = float(input("Enter the width of the rectangle: "))
    return f"Rectangle with length {length} and width {width}"

def main():
    shape_type = get_shape_type()

    if shape_type == "Invalid choice":
        print("Invalid choice. Please try again.")
    elif shape_type == "Circle":
        print(get_circle_properties())
    elif shape_type == "Square":
        print(get_square_properties())
    elif shape_type == "Triangle":
        print(get_triangle_properties())
    elif shape_type == "Rectangle":
        print(get_rectangle_properties())

if __name__ == "__main__":
    main()
```
