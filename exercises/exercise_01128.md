## Exercise
Write a program that takes a password as input and checks if it meets certain criteria (e.g. length, characters)

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password against certain criteria.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the criteria
    criteria = {
        "min_length": lambda p: len(p) >= 8,
        "has_uppercase": lambda p: any(c.isupper() for c in p),
        "has_lowercase": lambda p: any(c.islower() for c in p),
        "has_digit": lambda p: any(c.isdigit() for c in p),
        "has_special_char": lambda p: bool(re.search(r"[^a-zA-Z0-9]", p))
    }

    # Check each criterion
    for name, func in criteria.items():
        if not func(password):
            print(f"Password does not meet criterion: {name}")
            return False

    print("Password is valid")
    return True

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
