## Exercise
Write a program that takes the lengths of the sides of a triangle as input and returns the length of its circumradius.

## Reference Solution
```python
import math

def calculate_circumradius(a, b, c):
    """
    Calculate the circumradius of a triangle.

    Args:
        a (float): The length of the first side of the triangle.
        b (float): The length of the second side of the triangle.
        c (float): The length of the third side of the triangle.

    Returns:
        float: The length of the circumradius of the triangle.

    Raises:
        ValueError: If the input sides cannot form a triangle.
    """

    # Check if the input sides can form a triangle
    if a + b <= c or a + c <= b or b + c <= a:
        raise ValueError("The input sides cannot form a triangle.")

    # Calculate the semi-perimeter of the triangle
    s = (a + b + c) / 2

    # Calculate the area of the triangle using Heron's formula
    area = math.sqrt(s * (s - a) * (s - b) * (s - c))

    # Calculate the circumradius of the triangle
    circumradius = (a * b * c) / (4 * area)

    return circumradius

# Test the function
a = float(input("Enter the length of the first side: "))
b = float(input("Enter the length of the second side: "))
c = float(input("Enter the length of the third side: "))

try:
    circumradius = calculate_circumradius(a, b, c)
    print(f"The length of the circumradius is: {circumradius}")
except ValueError as e:
    print(e)
```
