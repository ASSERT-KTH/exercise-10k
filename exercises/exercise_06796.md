## Exercise
Write a program to find the sum of all prime factors of a given number. The program should use a while loop to continuously divide the input number by the smallest divisor until it is no longer divisible. Use a for loop to check for primality.

## Reference Solution
```python
def is_prime(n):
    """Check if a number is prime."""
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

def smallest_divisor(n):
    """Find the smallest divisor of a number."""
    for i in range(2, n + 1):
        if n % i == 0:
            return i

def sum_of_prime_factors(n):
    """Find the sum of all prime factors of a number."""
    sum_of_factors = 0
    while n > 1:
        divisor = smallest_divisor(n)
        if is_prime(divisor):
            sum_of_factors += divisor
        n = n // divisor
    return sum_of_factors

# Test the function
n = 315
print("Sum of prime factors of", n, "is", sum_of_prime_factors(n))
```
