## Exercise
Write a program that takes an array of integers as input and returns a new array containing every other element from the original array, starting from the first element. The program should also handle cases where the input array is empty or contains only one element.

## Reference Solution
```python
#include <stdio.h>

// Function to print an array
void print_array(int arr[], int size) {
    printf("{");
    for (int i = 0; i < size; i++) {
        printf("%d", arr[i]);
        if (i < size - 1) {
            printf(", ");
        }
    }
    printf("}\n");
}

// Function to create a new array with every other element
int* every_other(int arr[], int size, int* new_size) {
    *new_size = size / 2 + size % 2; // Calculate new array size
    int* new_arr = (int*)malloc(*new_size * sizeof(int)); // Allocate memory
    for (int i = 0; i < *new_size; i++) {
        new_arr[i] = arr[i * 2]; // Copy every other element
    }
    return new_arr;
}

int main() {
    int arr[] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
    int size = sizeof(arr) / sizeof(arr[0]);
    int new_size;
    int* new_arr = every_other(arr, size, &new_size);

    print_array(new_arr, new_size);

    free(new_arr); // Don't forget to free the allocated memory!

    return 0;
}
```
