## Exercise
Write a function that implements a Caesar cipher to shift characters in a string by a given number of positions

## Reference Solution
```python
def caesar_cipher(text: str, shift: int) -> str:
    """
    Shifts characters in a string by a given number of positions using a Caesar cipher.

    Args:
        text (str): The input string to be shifted.
        shift (int): The number of positions to shift the characters.

    Returns:
        str: The shifted string.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        # If the character is not a letter, add it to the result as is
        else:
            result += char

    return result


# Example usage
def main():
    text = "Hello, World!"
    shift = 3
    print(f"Original text: {text}")
    print(f"Shifted text: {caesar_cipher(text, shift)}")


if __name__ == "__main__":
    main()
```
