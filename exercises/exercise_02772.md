## Exercise
Write a program that takes a word as input and prints out its definition from a dictionary.

## Reference Solution
```python
import requests

def get_definition(word):
    """
    Fetches the definition of a word from a dictionary API.
    
    Args:
        word (str): The word to define.
    
    Returns:
        str: The definition of the word.
    """
    api_url = f"https://api.dictionaryapi.dev/api/v2/entries/en/{word}"
    response = requests.get(api_url)
    
    if response.status_code == 200:
        data = response.json()
        definition = data[0]['meanings'][0]['definitions'][0]['definition']
        return definition
    else:
        return None

def main():
    word = input("Enter a word: ")
    definition = get_definition(word)
    
    if definition:
        print(f"The definition of '{word}' is: {definition}")
    else:
        print(f"Sorry, no definition found for '{word}'")

if __name__ == "__main__":
    main()
```
