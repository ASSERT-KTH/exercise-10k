## Exercise
Develop a simple chatbot that responds to basic user queries. The chatbot should be able to greet the user, tell the user their name, and provide a basic help message.

## Reference Solution
```python
import random

class Chatbot:
    def __init__(self, name):
        self.name = name
        self.username = None

    def greet(self):
        """Greets the user with a random greeting message."""
        greetings = ["Hello!", "Hi!", "Hey!", "Welcome!"]
        return random.choice(greetings)

    def get_username(self):
        """Gets the username from the user."""
        if self.username is None:
            self.username = input("Please enter your name: ")
        return self.username

    def tell_name(self):
        """Tells the user their name."""
        return f"Your name is {self.get_username()}."

    def help(self):
        """Provides a basic help message."""
        return "I can greet you, tell you your name, and provide this help message. Type 'greet' to greet, 'name' to get your name, 'help' for this message, or 'quit' to exit."

    def respond(self, message):
        """Responds to the user's query."""
        if message.lower() == "greet":
            return self.greet()
        elif message.lower() == "name":
            return self.tell_name()
        elif message.lower() == "help":
            return self.help()
        elif message.lower() == "quit":
            return "Goodbye!"
        else:
            return "I didn't understand that. Type 'help' for available commands."

def main():
    chatbot = Chatbot("Simple Chatbot")
    print(f"Welcome to {chatbot.name}!")
    while True:
        message = input("You: ")
        response = chatbot.respond(message)
        print(f"{chatbot.name}: {response}")
        if message.lower() == "quit":
            break

if __name__ == "__main__":
    main()
```
