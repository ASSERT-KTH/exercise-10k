## Exercise
Write a program that checks whether an ISBN (International Standard Book Number) is valid.

## Reference Solution

def is_valid_isbn(isbn):
    """
    Checks whether an ISBN is valid.

    Args:
    isbn (str): The ISBN to validate.

    Returns:
    bool: True if the ISBN is valid, False otherwise.
    """

    # Remove hyphens and convert to uppercase
    isbn = isbn.replace("-", "").upper()

    # Check if the ISBN is in the ISBN-10 format
    if len(isbn) == 10:
        return is_valid_isbn10(isbn)
    # Check if the ISBN is in the ISBN-13 format
    elif len(isbn) == 13 and isbn.isdigit():
        return is_valid_isbn13(isbn)
    # If the ISBN is neither in the ISBN-10 nor ISBN-13 format, it's invalid
    else:
        return False


def is_valid_isbn10(isbn):
    """
    Checks whether an ISBN-10 is valid.

    Args:
    isbn (str): The ISBN-10 to validate.

    Returns:
    bool: True if the ISBN-10 is valid, False otherwise.
    """

    # Check if the last character is 'X' (which represents 10)
    if isbn[-1] == 'X':
        check_digit = 10
        isbn = isbn[:-1]
    # If the last character is not 'X', it should be a digit
    elif not isbn[-1].isdigit():
        return False
    else:
        check_digit = int(isbn[-1])

    # Calculate the check digit
    calculated_check_digit = 0
    for i in range(10):
        calculated_check_digit += (10 - i) * int(isbn[i])

    # Check if the calculated check digit matches the given check digit
    return calculated_check_digit % 11 == check_digit


def is_valid_isbn13(isbn):
    """
    Checks whether an ISBN-13 is valid.

    Args:
    isbn (str): The ISBN-13 to validate.

    Returns:
    bool: True if the ISBN-13 is valid, False otherwise.
    """

    # Calculate the check digit
    calculated_check_digit = 0
    for i in range(12):
        digit = int(isbn[i])
        if i % 2 == 0:
            calculated_check_digit += digit
        else:
            calculated_check_digit += digit * 3

    # Check if the calculated check digit matches the given check digit
    return (10 - (calculated_check_digit % 10)) % 10 == int(isbn[-1])
