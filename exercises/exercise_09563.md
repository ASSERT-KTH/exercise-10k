## Exercise
Create a program that takes a dictionary of student names and their corresponding grades as input. The program should calculate and display the average grade, the highest grade, and the names of students who scored above the average grade.

## Reference Solution
```python
def calculate_student_grades(student_grades):
    """
    Calculate the average grade, the highest grade, and the names of students who scored above the average grade.

    Args:
        student_grades (dict): A dictionary of student names and their corresponding grades.

    Returns:
        tuple: A tuple containing the average grade, the highest grade, and the names of students who scored above the average grade.
    """
    # Check if the input dictionary is empty
    if not student_grades:
        return None, None, []

    # Calculate the average grade
    average_grade = sum(student_grades.values()) / len(student_grades)

    # Find the highest grade
    highest_grade = max(student_grades.values())

    # Find the names of students who scored above the average grade
    above_average_students = [name for name, grade in student_grades.items() if grade > average_grade]

    return average_grade, highest_grade, above_average_students


def main():
    # Example usage
    student_grades = {
        "John": 85,
        "Alice": 90,
        "Bob": 78,
        "Charlie": 92,
        "David": 88
    }

    average_grade, highest_grade, above_average_students = calculate_student_grades(student_grades)

    print("Average Grade: {:.2f}".format(average_grade))
    print("Highest Grade: {:.2f}".format(highest_grade))
    print("Students with grades above average: {}".format(above_average_students))


if __name__ == "__main__":
    main()
```
