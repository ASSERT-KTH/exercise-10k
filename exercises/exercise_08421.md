## Exercise
Create a program that takes in a math operation and two numbers, and performs the operation using conditional statements.

## Reference Solution
```python
def math_operation(num1, num2, operation):
    """
    Perform a math operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The math operation to perform. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the math operation.

    Raises:
        ValueError: If the operation is not one of '+', '-', '*', or '/'.
        ZeroDivisionError: If the operation is '/' and num2 is 0.
    """

    # Check if the operation is valid
    if operation not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operation. Must be '+', '-', '*', or '/'.")

    # Perform the operation
    if operation == '+':
        return num1 + num2
    elif operation == '-':
        return num1 - num2
    elif operation == '*':
        return num1 * num2
    elif operation == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


def main():
    # Get the two numbers from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    # Get the operation from the user
    operation = input("Enter the operation (+, -, *, /): ")

    try:
        # Perform the operation and print the result
        result = math_operation(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except ValueError as e:
        print(f"Error: {e}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
