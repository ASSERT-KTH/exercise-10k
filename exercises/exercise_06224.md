## Exercise
Write a function that checks if a given element is present in a list.

## Reference Solution
```python
def check_element_in_list(input_list, target_element):
    """
    Checks if a given element is present in a list.

    Args:
        input_list (list): The list to search in.
        target_element: The element to search for.

    Returns:
        bool: True if the element is found, False otherwise.
    """
    return target_element in input_list

# Example usage:
def main():
    my_list = [1, 2, 3, 4, 5]
    target = 3
    result = check_element_in_list(my_list, target)
    print(f"Is {target} in the list? {result}")

    target = 6
    result = check_element_in_list(my_list, target)
    print(f"Is {target} in the list? {result}")

if __name__ == "__main__":
    main()
```
