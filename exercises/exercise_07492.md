## Exercise
Convert a string to an integer and vice versa, then print out the converted values.

## Reference Solution
```python
def convert_string_to_int(string):
    """Converts a string to an integer."""
    try:
        integer = int(string)
        return integer
    except ValueError:
        print("Error: The string cannot be converted to an integer.")
        return None

def convert_int_to_string(integer):
    """Converts an integer to a string."""
    return str(integer)

def main():
    # Test string to integer conversion
    string = "123"
    integer = convert_string_to_int(string)
    if integer is not None:
        print(f"The string '{string}' has been converted to an integer: {integer}")

    # Test integer to string conversion
    integer = 456
    string = convert_int_to_string(integer)
    print(f"The integer {integer} has been converted to a string: '{string}'")

if __name__ == "__main__":
    main()
```
