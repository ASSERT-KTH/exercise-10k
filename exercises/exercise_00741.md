def get_user_name():
    while True:
        user_name = input("Please enter your name: ")
        if user_name.strip():  # check if the input is not empty
            return user_name
        print("Invalid input. Please try again.")

def main():
    user_name = get_user_name()
    print(f"Hello, {user_name}! Welcome!")

if __name__ == "__main__":
    main()
