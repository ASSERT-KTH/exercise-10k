## Exercise
Create a program that calculates a person's daily nutrition needs. The user can input their age, weight, and height, and the program calculates the daily calorie needs.

## Reference Solution
```python
def calculate_bmr(age, weight, height, sex):
    """
    Calculate the basal metabolic rate (BMR) using the Harris-Benedict equation.

    Args:
        age (int): Age in years
        weight (float): Weight in kg
        height (float): Height in cm
        sex (str): Sex (male or female)

    Returns:
        float: BMR in calories/day
    """
    if sex.lower() == 'male':
        bmr = 66 + (6.2 * weight) + (12.7 * height) - (6.8 * age)
    elif sex.lower() == 'female':
        bmr = 655 + (4.35 * weight) + (4.7 * height) - (4.7 * age)
    else:
        raise ValueError("Invalid sex. Please enter 'male' or 'female'.")

    return bmr

def calculate_daily_calories(bmr, activity_level):
    """
    Calculate the daily calorie needs based on the BMR and activity level.

    Args:
        bmr (float): BMR in calories/day
        activity_level (str): Activity level (sedentary, lightly active, moderately active, very active, extra active)

    Returns:
        float: Daily calorie needs in calories/day
    """
    activity_factors = {
        'sedentary': 1.2,
        'lightly active': 1.375,
        'moderately active': 1.55,
        'very active': 1.725,
        'extra active': 1.9
    }

    daily_calories = bmr * activity_factors[activity_level]

    return daily_calories

def main():
    print("Daily Nutrition Needs Calculator")
    print("-----------------------------------")

    age = int(input("Enter your age: "))
    weight = float(input("Enter your weight in kg: "))
    height = float(input("Enter your height in cm: "))
    sex = input("Enter your sex (male or female): ")
    activity_level = input("Enter your activity level (sedentary, lightly active, moderately active, very active, extra active): ")

    bmr = calculate_bmr(age, weight, height, sex)
    daily_calories = calculate_daily_calories(bmr, activity_level)

    print(f"\nYour BMR is: {bmr:.2f} calories/day")
    print(f"Your daily calorie needs are: {daily_calories:.2f} calories/day")

if __name__ == "__main__":
    main()
```
