## Exercise
Write a program that demonstrates the scope of variables (local, global, nonlocal).

## Reference Solution
```python
# Global variable
global_variable = "I am global"

def outer_function():
    # Nonlocal variable (local to outer_function)
    nonlocal_variable = "I am nonlocal"

    def inner_function():
        # Local variable (local to inner_function)
        local_variable = "I am local"

        print("Inside inner_function:")
        print(f"Local variable: {local_variable}")
        print(f"Nonlocal variable: {nonlocal_variable}")
        print(f"Global variable: {global_variable}")

        # Try to modify global variable
        # This will create a new local variable with the same name
        global_variable = "I am local global"
        print(f"Global variable (modified locally): {global_variable}")

    inner_function()

    print("\nInside outer_function:")
    print(f"Nonlocal variable: {nonlocal_variable}")
    print(f"Global variable: {global_variable}")

    # Try to access local variable
    try:
        print(f"Local variable: {local_variable}")
    except NameError:
        print("Local variable is not accessible here")

outer_function()

print("\nOutside all functions:")
print(f"Global variable: {global_variable}")
```

def outer_function():

    nonlocal_variable = "I am nonlocal"

    def inner_function():
        global global_variable  # Declare global variable
        nonlocal nonlocal_variable  # Declare nonlocal variable

        global_variable = "I am modified global"
        nonlocal_variable = "I am modified nonlocal"

    inner_function()

    print(f"Global variable: {global_variable}")
    print(f"Nonlocal variable: {nonlocal_variable}")

outer_function()
