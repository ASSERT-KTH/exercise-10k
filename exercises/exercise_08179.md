## Exercise
Create an array of 5 integers and find the last index of a specific element.

## Reference Solution
```python
import numpy as np

def find_last_index(arr, target):
    """
    Find the last index of a specific element in an array.

    Args:
    arr (numpy.ndarray): The input array.
    target (int): The target element.

    Returns:
    int: The last index of the target element. Returns -1 if the target element is not found.
    """
    for i in range(len(arr) - 1, -1, -1):
        if arr[i] == target:
            return i
    return -1

# Create an array of 5 integers
arr = np.array([1, 2, 3, 2, 4])

# Specify the target element
target = 2

# Find the last index of the target element
last_index = find_last_index(arr, target)

if last_index != -1:
    print(f"The last index of {target} is {last_index}")
else:
    print(f"{target} is not found in the array")
```
