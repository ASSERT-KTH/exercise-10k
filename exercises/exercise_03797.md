## Exercise
Create a function that finds the number of unique paths from the top-left corner to the bottom-right corner of a grid.

## Reference Solution
```python
def unique_paths(rows: int, cols: int) -> int:
    """
    Calculate the number of unique paths from the top-left corner to the bottom-right corner of a grid.

    Args:
    rows (int): The number of rows in the grid.
    cols (int): The number of columns in the grid.

    Returns:
    int: The number of unique paths.
    """

    # Create a 2D array to store the number of unique paths to each cell.
    dp = [[1] * cols for _ in range(rows)]

    # Fill in the rest of the 2D array.
    for i in range(1, rows):
        for j in range(1, cols):
            # The number of unique paths to a cell is the sum of the number of unique paths to the cell above it and the cell to its left.
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1]

    # The number of unique paths to the bottom-right corner is stored in the bottom-right cell of the 2D array.
    return dp[rows - 1][cols - 1]


# Example usage:
if __name__ == "__main__":
    rows = 3
    cols = 7
    print(f"Number of unique paths in a {rows}x{cols} grid: {unique_paths(rows, cols)}")
```
