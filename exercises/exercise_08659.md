## Exercise
Create a function that finds the common elements in a list of lists.

## Reference Solution
```python
def find_common_elements(lists):
    """
    Finds the common elements in a list of lists.

    Args:
        lists (list): A list of lists.

    Returns:
        list: A list of common elements.
    """
    # Check if the input list is empty
    if not lists:
        return []

    # Initialize the set of common elements with the first list
    common_elements = set(lists[0])

    # Iterate over the remaining lists
    for lst in lists[1:]:
        # Update the set of common elements by intersecting it with the current list
        common_elements &= set(lst)

    # Convert the set of common elements back to a list and return it
    return list(common_elements)


# Example usage:
if __name__ == "__main__":
    lists = [[1, 2, 3], [2, 3, 4], [2, 3, 5]]
    common = find_common_elements(lists)
    print("Common elements:", common)
```
