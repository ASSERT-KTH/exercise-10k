## Exercise
Write a program to perform data type conversion between different data types using built-in functions.

## Reference Solution
```python
def convert_data_type(value, target_type):
    """
    Converts the given value to the target data type.

    Args:
        value (any): The value to be converted.
        target_type (type): The target data type.

    Returns:
        any: The converted value.

    Raises:
        ValueError: If the conversion is not possible.
        TypeError: If the target type is not supported.
    """

    # Define the supported data types
    supported_types = [int, float, str]

    # Check if the target type is supported
    if target_type not in supported_types:
        raise TypeError("Unsupported target type")

    # Perform the conversion
    if target_type == int:
        try:
            return int(value)
        except ValueError:
            raise ValueError("Cannot convert to integer")
    elif target_type == float:
        try:
            return float(value)
        except ValueError:
            raise ValueError("Cannot convert to float")
    elif target_type == str:
        return str(value)


def main():
    # Test the convert_data_type function
    print("Data Type Conversion Program")

    # Integer to Float
    print("Integer to Float:")
    value = 10
    target_type = float
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")

    # Float to Integer
    print("\nFloat to Integer:")
    value = 10.5
    target_type = int
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")

    # String to Integer
    print("\nString to Integer:")
    value = "10"
    target_type = int
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")

    # String to Float
    print("\nString to Float:")
    value = "10.5"
    target_type = float
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")

    # Integer to String
    print("\nInteger to String:")
    value = 10
    target_type = str
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")

    # Float to String
    print("\nFloat to String:")
    value = 10.5
    target_type = str
    print(f"{value} ({type(value).__name__}) -> {convert_data_type(value, target_type)} ({target_type.__name__})")


if __name__ == "__main__":
    main()
```
