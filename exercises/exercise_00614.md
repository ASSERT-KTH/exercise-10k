## Exercise
Write a program to calculate the sales tax on a given purchase amount and tax rate.

## Reference Solution
```python
def calculate_sales_tax(purchase_amount, tax_rate):
    """
    Calculates the sales tax on a given purchase amount and tax rate.

    Args:
        purchase_amount (float): The amount of the purchase.
        tax_rate (float): The tax rate as a decimal (e.g., 8% = 0.08).

    Returns:
        float: The sales tax amount.
    """
    sales_tax = purchase_amount * tax_rate
    return sales_tax

def main():
    purchase_amount = float(input("Enter the purchase amount: $"))
    tax_rate = float(input("Enter the tax rate (as a percentage, e.g., 8 for 8%): ")) / 100

    sales_tax = calculate_sales_tax(purchase_amount, tax_rate)
    total_amount = purchase_amount + sales_tax

    print(f"Purchase Amount: ${purchase_amount:.2f}")
    print(f"Tax Rate: {tax_rate * 100}%")
    print(f"Sales Tax: ${sales_tax:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
