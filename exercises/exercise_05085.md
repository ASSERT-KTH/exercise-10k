## Exercise
Write a program that takes a story with blank spaces as input and fills in the blanks with words provided by the user.

## Reference Solution
```python
def fill_blanks(story, words):
    """
    Fill in the blanks in the story with the provided words.

    Args:
    story (str): The story with blank spaces.
    words (list): A list of words to fill in the blanks.

    Returns:
    str: The story with the blanks filled.
    """
    filled_story = story
    for i, word in enumerate(words):
        filled_story = filled_story.replace(f"{{{i}}}", word, 1)
    return filled_story

def main():
    # Get the story from the user
    story = input("Enter a story with blank spaces (use {} to represent a blank): ")

    # Get the words from the user
    num_blanks = story.count("{}")
    words = []
    for i in range(num_blanks):
        word = input(f"Enter word {i+1}: ")
        words.append(word)

    # Fill in the blanks
    filled_story = fill_blanks(story, words)

    # Print the filled story
    print("Filled Story:")
    print(filled_story)

if __name__ == "__main__":
    main()
```
