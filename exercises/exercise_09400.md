## Exercise
Create a program that calculates the permutation and combination of two given numbers.

## Reference Solution
```python
import math

def calculate_permutation(n, r):
    """
    Calculate the permutation of n items taken r at a time.

    Args:
        n (int): The total number of items.
        r (int): The number of items to choose.

    Returns:
        int: The permutation of n items taken r at a time.
    """
    return math.factorial(n) // math.factorial(n - r)

def calculate_combination(n, r):
    """
    Calculate the combination of n items taken r at a time.

    Args:
        n (int): The total number of items.
        r (int): The number of items to choose.

    Returns:
        int: The combination of n items taken r at a time.
    """
    return math.comb(n, r)

def main():
    n = int(input("Enter the total number of items (n): "))
    r = int(input("Enter the number of items to choose (r): "))

    permutation = calculate_permutation(n, r)
    combination = calculate_combination(n, r)

    print(f"The permutation of {n} items taken {r} at a time is: {permutation}")
    print(f"The combination of {n} items taken {r} at a time is: {combination}")

if __name__ == "__main__":
    main()
```
