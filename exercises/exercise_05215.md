## Exercise
Develop a program that calculates the fuel efficiency based on the distance traveled and fuel consumed provided by the user.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumed, unit='miles'):
    """
    Calculates the fuel efficiency based on the distance traveled and fuel consumed.

    Args:
        distance (float): The distance traveled.
        fuel_consumed (float): The amount of fuel consumed.
        unit (str, optional): The unit of measurement for distance. Defaults to 'miles'.

    Returns:
        float: The fuel efficiency in miles per gallon (mpg) or kilometers per liter (km/L).
    """
    if unit == 'miles':
        fuel_efficiency = distance / fuel_consumed
        return fuel_efficiency
    elif unit == 'kilometers':
        fuel_efficiency = distance / fuel_consumed
        return fuel_efficiency
    else:
        raise ValueError("Invalid unit. Please use 'miles' or 'kilometers'.")

def main():
    print("Fuel Efficiency Calculator")
    print("---------------------------")

    distance_unit = input("Enter the unit of measurement for distance (miles/kilometers): ")
    distance = float(input("Enter the distance traveled: "))
    fuel_consumed = float(input("Enter the amount of fuel consumed (gallons/liters): "))

    fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumed, unit=distance_unit)

    if distance_unit == 'miles':
        print(f"Fuel efficiency: {fuel_efficiency:.2f} miles per gallon (mpg)")
    elif distance_unit == 'kilometers':
        print(f"Fuel efficiency: {fuel_efficiency:.2f} kilometers per liter (km/L)")

if __name__ == "__main__":
    main()
```
