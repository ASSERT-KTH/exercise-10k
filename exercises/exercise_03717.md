## Exercise
Create a function that generates a random tree of a specified depth.

## Reference Solution
```python
import random

class Node:
    """Represents a node in the tree."""
    def __init__(self, value):
        self.value = value
        self.children = []

class Tree:
    """Represents a tree."""
    def __init__(self, root):
        self.root = Node(root)

    def add_child(self, parent, child):
        """Adds a child node to a parent node."""
        parent.children.append(Node(child))

    def generate_random_tree(self, depth, max_children=3):
        """Generates a random tree of a specified depth."""
        def recursive_generate(node, current_depth):
            if current_depth < depth:
                num_children = random.randint(1, max_children)
                for _ in range(num_children):
                    child_value = random.randint(1, 100)
                    self.add_child(node, child_value)
                    recursive_generate(node.children[-1], current_depth + 1)

        recursive_generate(self.root, 1)

    def print_tree(self, node=None, level=0):
        """Prints the tree."""
        if node is None:
            node = self.root
        print(' ' * 4 * level + '->', node.value)
        for child in node.children:
            self.print_tree(child, level + 1)


# Example usage:
tree = Tree("Root")
tree.generate_random_tree(3)
tree.print_tree()
```
