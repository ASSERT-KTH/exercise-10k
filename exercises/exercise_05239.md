## Exercise
Write a program that takes a time and time zone as input, and prints the equivalent time in a different time zone.

## Reference Solution
```python
from datetime import datetime
import pytz

def convert_time(time_string, input_tz, output_tz):
    """
    Converts a time from one time zone to another.

    Args:
        time_string (str): The time to convert, in the format 'YYYY-MM-DD HH:MM:SS'.
        input_tz (str): The time zone of the input time.
        output_tz (str): The time zone to convert the time to.

    Returns:
        str: The converted time, in the format 'YYYY-MM-DD HH:MM:SS'.
    """

    # Parse the input time
    input_time = datetime.strptime(time_string, '%Y-%m-%d %H:%M:%S')

    # Localize the input time to the input time zone
    input_tz = pytz.timezone(input_tz)
    localized_input_time = input_tz.localize(input_time)

    # Convert the input time to the output time zone
    output_tz = pytz.timezone(output_tz)
    output_time = localized_input_time.astimezone(output_tz)

    # Format the output time as a string
    output_time_string = output_time.strftime('%Y-%m-%d %H:%M:%S')

    return output_time_string

def main():
    time_string = '2022-01-01 12:00:00'
    input_tz = 'US/Pacific'
    output_tz = 'US/Eastern'

    converted_time = convert_time(time_string, input_tz, output_tz)

    print(f'The time {time_string} in {input_tz} is equivalent to {converted_time} in {output_tz}.')

if __name__ == "__main__":
    main()
```
