## Exercise
Create a program that takes two numbers and an arithmetic operation as input and returns the result. The program should support addition, subtraction, multiplication, and division.

## Reference Solution
```python
def calculate(num1, num2, operation):
    """
    Performs arithmetic operations on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operation is not one of '+', '-', '*', or '/'.
        ZeroDivisionError: If the operation is '/' and num2 is 0.
    """

    # Check if the operation is valid
    if operation not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operation. Must be one of '+', '-', '*', or '/'.")

    # Perform the operation
    if operation == '+':
        # Add the numbers
        result = num1 + num2
    elif operation == '-':
        # Subtract num2 from num1
        result = num1 - num2
    elif operation == '*':
        # Multiply the numbers
        result = num1 * num2
    elif operation == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        # Divide num1 by num2
        result = num1 / num2

    return result


def main():
    # Get user input for numbers and operation
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, /): ")

    try:
        # Calculate and print the result
        result = calculate(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(e)


if __name__ == "__main__":
    main()
```
