## Exercise
Write a program that calculates a student's grade based on their scores on several assignments.

## Reference Solution
```python
def calculate_grade(assignments, scores):
    """
    Calculate the student's grade based on their scores on several assignments.

    Args:
        assignments (list): A list of assignments with their respective weights.
        scores (list): A list of scores for each assignment.

    Returns:
        float: The student's grade as a percentage.
    """

    # Check if the number of assignments and scores match
    if len(assignments) != len(scores):
        raise ValueError("Number of assignments and scores do not match")

    # Initialize total score and total weight
    total_score = 0
    total_weight = 0

    # Calculate total score and total weight
    for i in range(len(assignments)):
        assignment = assignments[i]
        score = scores[i]

        # Check if the score is valid (between 0 and 100)
        if score < 0 or score > 100:
            raise ValueError("Invalid score: {}".format(score))

        # Add the weighted score to the total score
        total_score += assignment['weight'] * score

        # Add the weight to the total weight
        total_weight += assignment['weight']

    # Check if the total weight is zero to avoid division by zero
    if total_weight == 0:
        raise ValueError("Total weight is zero")

    # Calculate the student's grade
    grade = total_score / total_weight

    return grade


def main():
    # Define assignments with their respective weights
    assignments = [
        {'name': 'Assignment 1', 'weight': 20},
        {'name': 'Assignment 2', 'weight': 30},
        {'name': 'Assignment 3', 'weight': 50}
    ]

    # Define scores for each assignment
    scores = [80, 90, 85]

    # Calculate the student's grade
    try:
        grade = calculate_grade(assignments, scores)
        print("The student's grade is: {:.2f}%".format(grade))
    except ValueError as e:
        print("Error: {}".format(e))


if __name__ == "__main__":
    main()
```
