## Exercise
Write a function to encode a given string as Base64

## Reference Solution
```python
import base64

def encode_string_to_base64(input_string: str) -> str:
    """
    Encodes a given string as Base64.

    Args:
        input_string (str): The string to be encoded.

    Returns:
        str: The Base64 encoded string.
    """
    # Convert the string to bytes
    input_bytes = input_string.encode("utf-8")
    
    # Encode the bytes as Base64
    encoded_bytes = base64.b64encode(input_bytes)
    
    # Convert the encoded bytes back to a string
    encoded_string = encoded_bytes.decode("utf-8")
    
    return encoded_string

# Example usage:
if __name__ == "__main__":
    input_str = "Hello, World!"
    encoded_str = encode_string_to_base64(input_str)
    print(f"Input: {input_str}")
    print(f"Base64 Encoded: {encoded_str}")
```
