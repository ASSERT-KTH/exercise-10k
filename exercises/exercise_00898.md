## Exercise
Create a program that responds to basic user queries, such as 'hello', 'how are you', and 'what is your name'.

## Reference Solution
```python
import random

class QueryResponder:
    def __init__(self, name):
        self.name = name

    def respond(self, query):
        query = query.lower()

        if "hello" in query or "hi" in query:
            return self.greet()
        elif "how are you" in query:
            return self.check_wellbeing()
        elif "what is your name" in query:
            return self.introduce()
        else:
            return "I'm not sure I understand your query."

    def greet(self):
        greetings = ["Hello!", "Hi!", "Hey!", "Hello there!"]
        return random.choice(greetings)

    def check_wellbeing(self):
        wellbeing_responses = ["I'm doing great, thanks!", "I'm fine, thanks!", "I'm just a program, I don't have feelings, but thanks for asking!"]
        return random.choice(wellbeing_responses)

    def introduce(self):
        return f"My name is {self.name}."

def main():
    responder = QueryResponder("Llama")

    print("Welcome to the query responder program!")
    while True:
        query = input("You: ")
        response = responder.respond(query)
        print(f"{responder.name}: {response}")

if __name__ == "__main__":
    main()
```
