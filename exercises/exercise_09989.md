## Exercise
Use a string variable to store a person's name and then perform string operations on it

## Reference Solution
```python
def main():
    # Store a person's name in a string variable
    person_name = "John Doe"

    # Print the original name
    print("Original Name:", person_name)

    # String operations
    print("\nString Operations:")
    print("--------------------")

    # 1. Upper case
    print("Upper Case:", person_name.upper())

    # 2. Lower case
    print("Lower Case:", person_name.lower())

    # 3. Title case
    print("Title Case:", person_name.title())

    # 4. Split the name into first and last names
    names = person_name.split()
    print("First Name:", names[0])
    print("Last Name:", names[1])

    # 5. Replace a part of the name
    new_name = person_name.replace("John", "Jane")
    print("New Name:", new_name)

    # 6. Check if a substring exists in the name
    print("Does the name contain 'Doe'?", "Doe" in person_name)

if __name__ == "__main__":
    main()
```
