## Exercise
Create a program that displays the weather forecast for a given city. The user can choose to view the current weather, tomorrow's weather, or the weather for the next 5 days.

## Reference Solution
```python
import requests
import json

def get_weather(api_key, city, units='metric'):
    """
    Fetches the current weather for a given city.
    
    Args:
        api_key (str): The OpenWeatherMap API key.
        city (str): The name of the city.
        units (str): The unit of measurement. Defaults to 'metric'.
    
    Returns:
        dict: The current weather data.
    """
    base_url = f"http://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units={units}"
    response = requests.get(base_url)
    return response.json()

def get_tomorrow_weather(api_key, city, units='metric'):
    """
    Fetches the weather forecast for tomorrow for a given city.
    
    Args:
        api_key (str): The OpenWeatherMap API key.
        city (str): The name of the city.
        units (str): The unit of measurement. Defaults to 'metric'.
    
    Returns:
        dict: The weather forecast for tomorrow.
    """
    base_url = f"http://api.openweathermap.org/data/2.5/forecast?q={city}&appid={api_key}&units={units}"
    response = requests.get(base_url)
    data = response.json()
    tomorrow = data['list'][0]['dt_txt']
    for forecast in data['list']:
        if forecast['dt_txt'] > tomorrow:
            return forecast

def get_5day_weather(api_key, city, units='metric'):
    """
    Fetches the weather forecast for the next 5 days for a given city.
    
    Args:
        api_key (str): The OpenWeatherMap API key.
        city (str): The name of the city.
        units (str): The unit of measurement. Defaults to 'metric'.
    
    Returns:
        dict: The weather forecast for the next 5 days.
    """
    base_url = f"http://api.openweathermap.org/data/2.5/forecast?q={city}&appid={api_key}&units={units}"
    response = requests.get(base_url)
    return response.json()

def main():
    api_key = "YOUR_API_KEY"  # Replace with your own API key
    city = input("Enter the city name: ")
    
    while True:
        print("\n1. Current Weather")
        print("2. Tomorrow's Weather")
        print("3. 5-Day Weather Forecast")
        print("4. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == '1':
            weather = get_weather(api_key, city)
            print(f"\nCurrent Weather in {city}:")
            print(f"Temperature: {weather['main']['temp']}°C")
            print(f"Weather Condition: {weather['weather'][0]['description']}")
            print(f"Humidity: {weather['main']['humidity']}%")
            print(f"Wind Speed: {weather['wind']['speed']} m/s")
        elif choice == '2':
            tomorrow_weather = get_tomorrow_weather(api_key, city)
            print(f"\nTomorrow's Weather in {city}:")
            print(f"Temperature: {tomorrow_weather['main']['temp']}°C")
            print(f"Weather Condition: {tomorrow_weather['weather'][0]['description']}")
            print(f"Humidity: {tomorrow_weather['main']['humidity']}%")
            print(f"Wind Speed: {tomorrow_weather['wind']['speed']} m/s")
        elif choice == '3':
            five_day_weather = get_5day_weather(api_key, city)
            print(f"\n5-Day Weather Forecast in {city}:")
            for forecast in five_day_weather['list']:
                print(f"Date: {forecast['dt_txt']}")
                print(f"Temperature: {forecast['main']['temp']}°C")
                print(f"Weather Condition: {forecast['weather'][0]['description']}")
                print(f"Humidity: {forecast['main']['humidity']}%")
                print(f"Wind Speed: {forecast['wind']['speed']} m/s")
                print("--------------------")
        elif choice == '4':
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
