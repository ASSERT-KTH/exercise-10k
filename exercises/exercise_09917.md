## Exercise
Write a program that checks the typography of a given text (e.g. font, size, style).

## Reference Solution
```python
import fontTools.ttLib

def check_typography(text, font_path):
    """
    Checks the typography of a given text.

    Args:
        text (str): The input text to check.
        font_path (str): The path to the font file (e.g. .ttf, .otf).

    Returns:
        dict: A dictionary containing the typography information.
    """
    # Load the font file
    font = fontTools.ttLib.TTFont(font_path)

    # Get the font name and style
    font_name = font['name'].getDebugName(4)
    font_style = font['name'].getDebugName(2)

    # Get the font size
    font_size = 12  # default font size

    # Analyze the text
    typography_info = {
        'font': font_name,
        'style': font_style,
        'size': font_size,
        'text': text
    }

    return typography_info

def main():
    # Example usage
    text = "Hello, World!"
    font_path = "path/to/font.ttf"

    typography_info = check_typography(text, font_path)

    print("Typography Information:")
    print(f"Font: {typography_info['font']}")
    print(f"Style: {typography_info['style']}")
    print(f"Size: {typography_info['size']}")
    print(f"Text: {typography_info['text']}")

if __name__ == "__main__":
    main()
```
