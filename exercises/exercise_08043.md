## Exercise
Write a program that uses an if/else statement to administer a simple quiz based on user input.

## Reference Solution
```python
# Define a dictionary with quiz questions and answers
quiz_questions = {
    "What is the capital of France?": "Paris",
    "What is the largest planet in our solar system?": "Jupiter",
    "What is the chemical symbol for gold?": "Au",
    "Who painted the Mona Lisa?": "Leonardo da Vinci",
    "What is the largest mammal on Earth?": "Blue whale"
}

def run_quiz(questions):
    """
    Run the quiz and check user answers.
    """
    score = 0
    for question, answer in questions.items():
        user_answer = input(question + " ")
        if user_answer.lower() == answer.lower():
            print("Correct!\n")
            score += 1
        else:
            print(f"Incorrect. The correct answer is {answer}.\n")
    print(f"Quiz complete! Your final score is {score} out of {len(questions)}.")

def main():
    print("Welcome to the quiz!")
    run_quiz(quiz_questions)

if __name__ == "__main__":
    main()
```
