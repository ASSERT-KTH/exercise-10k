## Exercise
Create a program that simulates a conversation with the user.

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary to store user inputs and their corresponding responses
responses = {
    "hello": ["Hi, how are you?", "Hello! What's up?", "Hey, what can I do for you?"],
    "how are you": ["I'm good, thanks!", "I'm doing well, thanks for asking!", "I'm just a program, I don't have feelings, but thanks for asking!"],
    "what's your name": ["My name is Conversation Simulator", "I don't have a personal name, but I'm here to chat with you!", "You can call me Sim, for short!"],
    "default": ["Sorry, I didn't understand that.", "Can you please rephrase?", "I'm not sure what you mean."]
}

def get_response(user_input):
    """
    Returns a response based on the user's input.

    Args:
        user_input (str): The user's input.

    Returns:
        str: A response to the user's input.
    """
    user_input = user_input.lower()
    for key in responses:
        if key in user_input:
            return random.choice(responses[key])
    return random.choice(responses["default"])

def main():
    print("Welcome to the Conversation Simulator!")
    print("Type 'quit' to exit the program.")

    while True:
        user_input = input("You: ")
        if user_input.lower() == "quit":
            break
        print("Sim: ", get_response(user_input))

if __name__ == "__main__":
    main()
```
