## Exercise
Create a simple student information system using variables and data types.

## Reference Solution
```python
class Student:
    def __init__(self, student_id, name, age, grade, gpa):
        """
        Initialize a Student object.

        Args:
        - student_id (int): Unique identifier for the student.
        - name (str): Student's name.
        - age (int): Student's age.
        - grade (int): Student's current grade level.
        - gpa (float): Student's current GPA.
        """
        self.student_id = student_id
        self.name = name
        self.age = age
        self.grade = grade
        self.gpa = gpa

    def display_info(self):
        """
        Display the student's information.
        """
        print(f"Student ID: {self.student_id}")
        print(f"Name: {self.name}")
        print(f"Age: {self.age}")
        print(f"Grade: {self.grade}")
        print(f"GPA: {self.gpa:.2f}")


def main():
    # Create a new Student object
    student1 = Student(12345, "John Doe", 17, 12, 3.8)

    # Display the student's information
    print("Student Information:")
    student1.display_info()


if __name__ == "__main__":
    main()
```
