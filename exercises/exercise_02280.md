## Exercise
Write a program that calculates the least common multiple of two numbers.

## Reference Solution
```python
import math

def calculate_lcm(num1, num2):
    """
    Calculate the least common multiple of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The least common multiple of num1 and num2.
    """
    # Calculate the greatest common divisor using the math.gcd function
    gcd = math.gcd(num1, num2)
    
    # Calculate the least common multiple using the formula: lcm(a, b) = |a*b| / gcd(a, b)
    lcm = abs(num1 * num2) // gcd
    
    return lcm

def main():
    # Example usage:
    num1 = 12
    num2 = 15
    
    lcm = calculate_lcm(num1, num2)
    
    print(f"The least common multiple of {num1} and {num2} is: {lcm}")

if __name__ == "__main__":
    main()
```
