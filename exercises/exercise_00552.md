## Exercise
Write a program that takes a string and a rotation value as input and returns the rotated string.

## Reference Solution
```python
def rotate_string(input_str, rotation, direction='clockwise'):
    """
    Rotate a string by a given rotation value.

    Args:
    input_str (str): The input string to be rotated.
    rotation (int): The number of positions to rotate the string.
    direction (str, optional): The direction of rotation. Defaults to 'clockwise'.

    Returns:
    str: The rotated string.
    """
    if direction == 'clockwise':
        rotation = rotation % len(input_str)
        return input_str[-rotation:] + input_str[:-rotation]
    elif direction == 'counter-clockwise':
        rotation = rotation % len(input_str)
        return input_str[rotation:] + input_str[:rotation]
    else:
        raise ValueError("Invalid direction. Please specify 'clockwise' or 'counter-clockwise'.")

# Example usage
if __name__ == "__main__":
    input_str = "HelloWorld"
    rotation = 3
    print("Original string:", input_str)
    print("Rotated string (clockwise):", rotate_string(input_str, rotation))
    print("Rotated string (counter-clockwise):", rotate_string(input_str, rotation, 'counter-clockwise'))
```
