## Exercise
Find the minimum number of deletions needed to make an array sorted

## Reference Solution

def min_deletions_to_sort(arr):
    """
    Find the minimum number of deletions needed to make an array sorted.

    Args:
    arr (list): The input array.

    Returns:
    int: The minimum number of deletions needed.
    """
    n = len(arr)
    if n == 0:
        return 0

    # Initialize a list to store the length of LIS ending at each position
    lis = [1] * n

    # Compute the length of LIS ending at each position
    for i in range(1, n):
        for j in range(i):
            if arr[i] > arr[j] and lis[i] < lis[j] + 1:
                lis[i] = lis[j] + 1

    # The minimum number of deletions is the difference between the length of the array and the length of the LIS
    return n - max(lis)


# Example usage
arr = [4, 5, 2, 10, 3]
print("Minimum deletions:", min_deletions_to_sort(arr))
