## Exercise
Write a program to simulate a quiz show where the user has to answer questions to win prizes.

## Reference Solution
```python
# Quiz Show Simulator

# Dictionary of questions, answers, and prizes
quiz_questions = {
    "What is the capital of France?": {"A": "Paris", "B": "London", "C": "Berlin", "D": "Rome", "Answer": "A", "Prize": 100},
    "Who painted the Mona Lisa?": {"A": "Leonardo da Vinci", "B": "Michelangelo", "C": "Raphael", "D": "Caravaggio", "Answer": "A", "Prize": 200},
    "What is the largest planet in our solar system?": {"A": "Earth", "B": "Saturn", "C": "Jupiter", "D": "Uranus", "Answer": "C", "Prize": 300},
    "Who wrote Romeo and Juliet?": {"A": "William Shakespeare", "B": "Jane Austen", "C": "Charles Dickens", "D": "J.K. Rowling", "Answer": "A", "Prize": 400},
    "What is the chemical symbol for gold?": {"A": "Ag", "B": "Au", "C": "Hg", "D": "Pb", "Answer": "B", "Prize": 500},
}

def quiz_show():
    print("Welcome to the Quiz Show!")
    print("You will be asked a series of questions, and for each question you answer correctly, you will win a prize.")
    print("If you answer a question incorrectly, the game will end.")
    
    total_prize = 0
    for question, options in quiz_questions.items():
        print(f"\nQuestion: {question}")
        for option, value in options.items():
            if option != "Answer" and option != "Prize":
                print(f"{option}: {value}")
        
        answer = input("Enter your answer (A, B, C, D): ")
        if answer.upper() == options["Answer"]:
            print(f"Correct! You won ${options['Prize']}.")
            total_prize += options["Prize"]
        else:
            print(f"Incorrect. The correct answer was {options['Answer']}. Game over.")
            break
    
    print(f"\nYou won a total of ${total_prize}.")

if __name__ == "__main__":
    quiz_show()
```
