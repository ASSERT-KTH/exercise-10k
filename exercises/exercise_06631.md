## Exercise
Write a function that uses async/await to perform an asynchronous operation

## Reference Solution
```python
import asyncio
import time

# Define an asynchronous function
async def perform_async_operation(operation_name: str):
    """
    Simulates an asynchronous operation with a delay.
    
    Args:
    operation_name (str): The name of the operation.
    
    Returns:
    str: A success message.
    """
    print(f"Starting {operation_name}...")
    await asyncio.sleep(2)  # Simulate a delay of 2 seconds
    print(f"{operation_name} completed.")
    return f"{operation_name} successful."

# Define the main function
async def main():
    """
    Demonstrates the usage of the perform_async_operation function.
    """
    start_time = time.time()
    
    # Perform multiple asynchronous operations concurrently
    tasks = [
        perform_async_operation("Operation 1"),
        perform_async_operation("Operation 2"),
        perform_async_operation("Operation 3")
    ]
    results = await asyncio.gather(*tasks)
    
    print("Results:")
    for result in results:
        print(result)
    
    print(f"Total execution time: {time.time() - start_time} seconds")

# Run the main function
if __name__ == "__main__":
    asyncio.run(main())
```
