## Exercise
Write a program that takes a string and an integer as input, and returns the justified version of the string, where the words are spaced evenly to fill the given width.

## Reference Solution
```python
class Solution:
    def fullJustify(self, words, maxWidth):
        result, current_line, current_width = [], [], 0
        for word in words:
            if current_width + len(word) + len(current_line) > maxWidth:
                for i in range(len(current_line) - 1, 0, -1):
                    current_line[i] += ' '
                if len(current_line) == 1:
                    current_line[0] += ' ' * (maxWidth - len(current_line[0]))
                result.append(''.join(current_line))
                current_line, current_width = [], 0
            current_line.append(word)
            current_width += len(word)
        last_line = ' '.join(current_line)
        last_line += ' ' * (maxWidth - len(last_line))
        result.append(last_line)
        return result
```
