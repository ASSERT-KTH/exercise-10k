## Exercise
Create a program that uses a while loop to simulate a Twitter feed, where users can post tweets and view the timeline until they choose to quit.

## Reference Solution
```python
class TwitterFeed:
    def __init__(self):
        self.tweets = []

    def post_tweet(self):
        """Posts a new tweet to the timeline."""
        tweet = input("Enter your tweet: ")
        self.tweets.append(tweet)
        print("Tweet posted successfully!")

    def view_timeline(self):
        """Displays all tweets in the timeline."""
        if not self.tweets:
            print("No tweets in the timeline.")
        else:
            print("\nTwitter Timeline:")
            for i, tweet in enumerate(self.tweets, start=1):
                print(f"{i}. {tweet}")

    def run(self):
        """Runs the Twitter feed simulation."""
        while True:
            print("\nTwitter Feed Menu:")
            print("1. Post a tweet")
            print("2. View timeline")
            print("3. Quit")
            choice = input("Enter your choice: ")
            if choice == "1":
                self.post_tweet()
            elif choice == "2":
                self.view_timeline()
            elif choice == "3":
                print("Goodbye!")
                break
            else:
                print("Invalid choice. Please try again.")


if __name__ == "__main__":
    twitter_feed = TwitterFeed()
    twitter_feed.run()
```
