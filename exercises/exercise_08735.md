Declare and use map variables to store key-value pairs.

```python

# Declare a map with initial key-value pairs
person = {
    "name": "John Doe",
    "age": 30,
    "city": "New York"
}
```

```python
# Insert a new key-value pair
person["country"] = "USA"

# Retrieve a value by its key
print(person["name"])  # Output: John Doe

# Update an existing key-value pair
person["age"] = 31

# Delete a key-value pair
del person["city"]

# Check if a key exists in the map
if "age" in person:
    print("Age:", person["age"])  # Output: Age: 31

# Iterate over the key-value pairs
for key, value in person.items():
    print(f"{key}: {value}")

# Output:
# name: John Doe
# age: 31
# country: USA
```

```python
# Declare a map to store student information
students = {}

# Function to add a student
def add_student(student_id, name, grade):
    students[student_id] = {
        "name": name,
        "grade": grade
    }

# Function to retrieve a student's information
def get_student(student_id):
    return students.get(student_id)

# Add students
add_student("S1", "John Doe", 90)
add_student("S2", "Jane Doe", 85)

# Retrieve a student's information
student_info = get_student("S1")
if student_info:
    print(f"Name: {student_info['name']}, Grade: {student_info['grade']}")
else:
    print("Student not found")
```
