## Exercise
Write a function to find the most common value in a dictionary

```python
from collections import Counter

def most_common_value(dictionary):
    """
    Find the most common value in a dictionary.

    Args:
        dictionary (dict): The dictionary to search.

    Returns:
        The most common value in the dictionary, or None if the dictionary is empty.

    Raises:
        ValueError: If there are multiple most common values.
    """
    # Get the values from the dictionary
    values = dictionary.values()

    # Count the occurrences of each value
    value_counts = Counter(values)

    # Find the maximum count
    max_count = max(value_counts.values(), default=0)

    # Find all values with the maximum count
    most_common_values = [value for value, count in value_counts.items() if count == max_count]

    # Check if there are multiple most common values
    if len(most_common_values) > 1:
        raise ValueError("There are multiple most common values: {}".format(most_common_values))

    # Return the most common value, or None if the dictionary is empty
    return most_common_values[0] if most_common_values else None

# Example usage:
example_dict = {'a': 1, 'b': 2, 'c': 1, 'd': 3, 'e': 1}
print(most_common_value(example_dict))  # Output: 1
```
