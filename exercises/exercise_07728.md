## Exercise
Find the longest substring with k distinct characters in a given string.

## Reference Solution

from collections import defaultdict

def longest_substring_with_k_distinct(s: str, k: int) -> str:
    """
    Find the longest substring with k distinct characters in a given string.

    Args:
    s (str): The input string.
    k (int): The number of distinct characters.

    Returns:
    str: The longest substring with k distinct characters.
    """
    if not s or k == 0:
        return ""

    # Initialize variables
    max_length = 0
    max_substring = ""
    char_count = defaultdict(int)
    window_start = 0

    # Iterate over the string
    for window_end in range(len(s)):
        # Add the current character to the window
        char_count[s[window_end]] += 1

        # Shrink the window if the number of distinct characters exceeds k
        while len(char_count) > k:
            char_count[s[window_start]] -= 1
            if char_count[s[window_start]] == 0:
                del char_count[s[window_start]]
            window_start += 1

        # Update the maximum length and substring
        if len(char_count) == k and window_end - window_start + 1 > max_length:
            max_length = window_end - window_start + 1
            max_substring = s[window_start:window_end + 1]

    return max_substring

# Example usage
s = "araaci"
k = 2
print(longest_substring_with_k_distinct(s, k))  # Output: "araa"
