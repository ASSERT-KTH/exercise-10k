## Exercise
Write a program to dynamically check the data type of a variable and perform actions based on its type.

## Reference Solution
```python
def check_data_type(var):
    """
    Dynamically checks the data type of a variable and performs actions based on its type.

    Args:
        var: The variable to check.

    Returns:
        None
    """

    # Check if the variable is an integer
    if isinstance(var, int):
        print(f"The variable {var} is an integer.")
        print("Performing integer-specific action...")
        print(f"The square of {var} is {var ** 2}")

    # Check if the variable is a float
    elif isinstance(var, float):
        print(f"The variable {var} is a float.")
        print("Performing float-specific action...")
        print(f"The square root of {var} is {var ** 0.5}")

    # Check if the variable is a string
    elif isinstance(var, str):
        print(f"The variable {var} is a string.")
        print("Performing string-specific action...")
        print(f"The length of the string {var} is {len(var)}")

    # Check if the variable is a list
    elif isinstance(var, list):
        print(f"The variable {var} is a list.")
        print("Performing list-specific action...")
        print(f"The length of the list {var} is {len(var)}")

    # Check if the variable is a dictionary
    elif isinstance(var, dict):
        print(f"The variable {var} is a dictionary.")
        print("Performing dictionary-specific action...")
        print(f"The keys of the dictionary {var} are {list(var.keys())}")

    # If the variable is of any other type, print a message
    else:
        print(f"The variable {var} is of type {type(var)}")
        print("No specific action defined for this type.")

# Test the function with different types of variables
check_data_type(10)  # Integer
check_data_type(10.5)  # Float
check_data_type("Hello")  # String
check_data_type([1, 2, 3])  # List
check_data_type({"a": 1, "b": 2})  # Dictionary
check_data_type((1, 2, 3))  # Tuple
```
