## Exercise
Write a program to declare and use a set to store a collection of unique values.

## Reference Solution
```python
# Define a function to demonstrate set usage
def use_set():
    # Declare an empty set
    unique_values = set()

    # Add values to the set
    unique_values.add(10)
    unique_values.add(20)
    unique_values.add(10)  # Duplicate value, will not be added
    unique_values.add(30)

    # Print the set
    print("Unique Values:", unique_values)

    # Add multiple values at once
    unique_values.update([40, 50, 20])  # 20 is a duplicate, will not be added

    # Print the updated set
    print("Updated Unique Values:", unique_values)

    # Remove a value from the set
    unique_values.remove(20)

    # Print the set after removal
    print("Unique Values after removal:", unique_values)

    # Check if a value exists in the set
    print("Is 30 in the set?", 30 in unique_values)

# Example usage
if __name__ == "__main__":
    use_set()
```
