## Exercise
Create a program that reads input from a file specified by the user, processes it by converting all text to uppercase, and writes the result to another file also specified by the user. Handle potential errors such as file not found or permission denied.

## Reference Solution
```python
def process_file(input_file, output_file):
    """
    Reads input from a file, converts the text to uppercase, and writes the result to another file.

    Args:
        input_file (str): The path to the input file.
        output_file (str): The path to the output file.

    Returns:
        None
    """

    try:
        # Attempt to open the input file in read mode
        with open(input_file, 'r') as file:
            # Read the contents of the file
            text = file.read()
        
        # Convert the text to uppercase
        uppercase_text = text.upper()

        # Attempt to open the output file in write mode
        with open(output_file, 'w') as file:
            # Write the uppercase text to the output file
            file.write(uppercase_text)

        print(f"Text successfully converted to uppercase and written to {output_file}")

    except FileNotFoundError:
        print(f"Error: Input file '{input_file}' not found.")
    except PermissionError:
        print(f"Error: Permission denied to read '{input_file}' or write to '{output_file}'.")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")


def main():
    # Get the input file path from the user
    input_file = input("Enter the path to the input file: ")

    # Get the output file path from the user
    output_file = input("Enter the path to the output file: ")

    # Process the file
    process_file(input_file, output_file)


if __name__ == "__main__":
    main()
```
