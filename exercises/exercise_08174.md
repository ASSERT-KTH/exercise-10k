## Exercise
Create a list of 5 integers and check if all elements meet a condition.

## Reference Solution
```python
def check_condition(numbers, condition):
    """
    Checks if all elements in the list meet the specified condition.

    Args:
    - numbers (list): A list of integers.
    - condition (function): A function that takes an integer and returns a boolean.

    Returns:
    - bool: True if all elements meet the condition, False otherwise.
    """
    return all(condition(num) for num in numbers)


# Example usage
if __name__ == "__main__":
    # Create a list of 5 integers
    numbers = [10, 20, 30, 40, 50]

    # Define a condition function (e.g., checking if a number is even)
    def is_even(num):
        return num % 2 == 0

    # Check if all elements meet the condition
    result = check_condition(numbers, is_even)

    print(f"List of numbers: {numbers}")
    print(f"Condition: is_even")
    print(f"All elements meet the condition: {result}")
```
