## Exercise
Create a program that plays Rock, Paper, Scissors with the user. The program should generate a random choice and determine the winner based on the game's rules.

## Reference Solution
```python
import random

def play_game():
    # Define the possible choices
    choices = ['rock', 'paper', 'scissors']

    # Generate a random choice for the computer
    computer_choice = random.choice(choices)

    # Ask the user for their choice
    user_choice = input("Enter your choice (rock, paper, or scissors): ")

    # Validate the user's choice
    while user_choice.lower() not in choices:
        user_choice = input("Invalid choice. Please enter rock, paper, or scissors: ")

    # Determine the winner
    print(f"\nComputer chose: {computer_choice}")
    print(f"User chose: {user_choice}\n")

    if computer_choice == user_choice:
        print(f"It's a tie! Both chose {user_choice}.")
    elif (computer_choice == 'rock' and user_choice == 'scissors') or \
         (computer_choice == 'paper' and user_choice == 'rock') or \
         (computer_choice == 'scissors' and user_choice == 'paper'):
        print(f"Computer wins! {computer_choice.capitalize()} beats {user_choice}.")
    else:
        print(f"User wins! {user_choice.capitalize()} beats {computer_choice}.")

def main():
    play_again = 'y'
    while play_again.lower() == 'y':
        play_game()
        play_again = input("\nDo you want to play again? (y/n): ")
        while play_again.lower() not in ['y', 'n']:
            play_again = input("Invalid input. Please enter y or n: ")

if __name__ == "__main__":
    main()
```
