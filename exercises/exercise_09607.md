## Exercise
Write a program that asks the user a series of multiple-choice questions and keeps track of the score

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary with questions, options, and answers
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "Answer": "B"
    },
    "Which planet is the largest in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "Answer": "C"
    },
    "Who painted the Mona Lisa?": {
        "A": "Leonardo da Vinci",
        "B": "Michelangelo",
        "C": "Raphael",
        "D": "Caravaggio",
        "Answer": "A"
    },
    "What is the largest mammal on Earth?": {
        "A": "Blue whale",
        "B": "Fin whale",
        "C": "Humpback whale",
        "D": "Sperm whale",
        "Answer": "A"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "Answer": "B"
    }
}

def quiz():
    # Initialize the score
    score = 0

    # Iterate over the questions
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "Answer":
                print(f"{option}: {value}")
        answer = input("Enter the correct option (A, B, C, D): ")
        if answer.upper() == options["Answer"]:
            print("Correct answer!\n")
            score += 1
        else:
            print(f"Incorrect answer. The correct answer was {options['Answer']}\n")

    # Print the final score
    print(f"Quiz finished! Your final score is {score}/{len(questions)}")

# Run the quiz
quiz()
```
