## Exercise
Create a function that uses async/await to perform asynchronous operations.

## Reference Solution
```python
import asyncio
import time

async def fetch_data(api_name):
    """Simulate fetching data from an API."""
    print(f"Fetching data from {api_name}...")
    await asyncio.sleep(2)  # Simulate API call delay
    print(f"Data fetched from {api_name}.")
    return f"Data from {api_name}"

async def main():
    """Perform asynchronous operations."""
    start_time = time.time()

    # Perform asynchronous operations concurrently
    task1 = asyncio.create_task(fetch_data("API 1"))
    task2 = asyncio.create_task(fetch_data("API 2"))

    # Wait for both tasks to complete
    result1 = await task1
    result2 = await task2

    print(f"Results: {result1}, {result2}")
    print(f"Total execution time: {time.time() - start_time} seconds")

# Run the asynchronous main function
asyncio.run(main())
```
