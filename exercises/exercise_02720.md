## Exercise
Write a program to validate a given credit card number.

def validate_credit_card_number(card_number):
    """
    Validate a credit card number using the Luhn algorithm.

    Args:
        card_number (str): The credit card number to validate.

    Returns:
        bool: True if the credit card number is valid, False otherwise.
    """
    # Remove any non-digit characters from the card number
    card_number = ''.join(filter(str.isdigit, card_number))

    # Check if the card number is empty or has less than 13 digits
    if len(card_number) < 13 or len(card_number) > 19:
        return False

    # Reverse the order of the digits in the card number
    reversed_card_number = card_number[::-1]

    # Initialize the sum of the digits
    digit_sum = 0

    # Iterate over the digits in the reversed card number
    for i, digit in enumerate(reversed_card_number):
        # Convert the digit to an integer
        digit = int(digit)

        # If the digit is at an odd position, double its value
        if i % 2 != 0:
            digit *= 2
            # If the doubled value is greater than 9, subtract 9
            if digit > 9:
                digit -= 9

        # Add the digit to the sum
        digit_sum += digit

    # Check if the sum is divisible by 10
    return digit_sum % 10 == 0

if __name__ == "__main__":
    card_number = input("Enter a credit card number: ")
    if validate_credit_card_number(card_number):
        print("The credit card number is valid.")
    else:
        print("The credit card number is not valid.")
