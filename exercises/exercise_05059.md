## Exercise
Create a program that calculates the roots of a quadratic equation of the form ax^2 + bx + c = 0.

import math

def calculate_roots(a, b, c):
    """
    Calculates the roots of a quadratic equation.

    Args:
        a (float): Coefficient of x^2
        b (float): Coefficient of x
        c (float): Constant term

    Returns:
        tuple: Roots of the quadratic equation
    """
    determinant = b**2 - 4*a*c

    if determinant > 0:
        root1 = (-b + math.sqrt(determinant)) / (2*a)
        root2 = (-b - math.sqrt(determinant)) / (2*a)
        return root1, root2
    elif determinant == 0:
        root = -b / (2*a)
        return root, root
    else:
        real_part = -b / (2*a)
        imaginary_part = math.sqrt(-determinant) / (2*a)
        root1 = complex(real_part, imaginary_part)
        root2 = complex(real_part, -imaginary_part)
        return root1, root2

def main():
    print("Quadratic Equation Roots Calculator")

    b = float(input("Enter coefficient of x (b): "))
    c = float(input("Enter constant term (c): "))

    root1, root2 = calculate_roots(a, b, c)

    print(f"Roots of the quadratic equation {a}x^2 + {b}x + {c} = 0:")
