## Exercise
Create a program that uses a while loop to repeatedly ask the user to guess a random number between 1 and 100 until they guess correctly.

## Reference Solution
```python
import random

def guess_the_number():
    # Generate a random number between 1 and 100
    number_to_guess = random.randint(1, 100)

    # Initialize the number of tries the player has made
    tries = 0

    print("Welcome to the number guessing game!")
    print("I'm thinking of a number between 1 and 100.")

    # Continue to ask the user for input until they guess the number
    while True:
        # Ask the user for their guess
        user_guess = input("Take a guess: ")

        # Try to convert the user's guess to an integer
        try:
            user_guess = int(user_guess)
        except ValueError:
            print("That's not a valid number!")
            continue

        # Increment the number of tries the player has made
        tries += 1

        # Check if the user's guess is higher or lower than the number to guess
        if user_guess < number_to_guess:
            print("Too low!")
        elif user_guess > number_to_guess:
            print("Too high!")
        else:
            # If the user guessed the number, print a success message and exit the loop
            print(f"Congratulations! You found the number in {tries} tries.")
            break

if __name__ == "__main__":
    guess_the_number()
```
