## Exercise
Write a program to perform basic dictionary operations (key-value pair, get, update, delete).

## Reference Solution
```python
class DictionaryOperations:
    def __init__(self):
        self.dictionary = {}

    def add_key_value(self, key, value):
        """Add a key-value pair to the dictionary."""
        self.dictionary[key] = value
        print(f"Added key-value pair: {key} - {value}")

    def get_value(self, key):
        """Retrieve the value associated with a given key."""
        if key in self.dictionary:
            print(f"Value for key '{key}': {self.dictionary[key]}")
        else:
            print(f"Key '{key}' not found in dictionary.")

    def update_value(self, key, new_value):
        """Update the value associated with a given key."""
        if key in self.dictionary:
            self.dictionary[key] = new_value
            print(f"Updated value for key '{key}': {new_value}")
        else:
            print(f"Key '{key}' not found in dictionary.")

    def delete_key_value(self, key):
        """Delete a key-value pair from the dictionary."""
        if key in self.dictionary:
            del self.dictionary[key]
            print(f"Deleted key-value pair: {key}")
        else:
            print(f"Key '{key}' not found in dictionary.")

    def print_dictionary(self):
        """Print the current state of the dictionary."""
        print("Current dictionary:")
        for key, value in self.dictionary.items():
            print(f"{key}: {value}")


def main():
    operations = DictionaryOperations()

    while True:
        print("\nDictionary Operations Menu:")
        print("1. Add key-value pair")
        print("2. Get value")
        print("3. Update value")
        print("4. Delete key-value pair")
        print("5. Print dictionary")
        print("6. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            key = input("Enter key: ")
            value = input("Enter value: ")
            operations.add_key_value(key, value)
        elif choice == "2":
            key = input("Enter key: ")
            operations.get_value(key)
        elif choice == "3":
            key = input("Enter key: ")
            new_value = input("Enter new value: ")
            operations.update_value(key, new_value)
        elif choice == "4":
            key = input("Enter key: ")
            operations.delete_key_value(key)
        elif choice == "5":
            operations.print_dictionary()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
