## Exercise
Write a program to check if a given string has balanced parentheses.

## Reference Solution
```python
def has_balanced_parentheses(s):
    """
    Checks if a given string has balanced parentheses.

    Args:
        s (str): The input string.

    Returns:
        bool: True if the string has balanced parentheses, False otherwise.
    """
    stack = []
    parentheses_map = {')': '(', '}': '{', ']': '['}

    for char in s:
        if char in parentheses_map.values():
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in parentheses_map.keys():
            # Check if the stack is empty or the top of the stack does not match the current closing parenthesis
            if not stack or stack.pop() != parentheses_map[char]:
                return False

    # If the stack is empty after iterating over the string, the parentheses are balanced
    return not stack


def main():
    # Example usage
    test_cases = [
        ("(hello world)", True),
        ("(hello world", False),
        ("({[]})", True),
        ("({[}])", False),
        ("(a(b)c)", True),
    ]

    for s, expected in test_cases:
        result = has_balanced_parentheses(s)
        print(f"Input: {s}, Expected: {expected}, Result: {result}")


if __name__ == "__main__":
    main()
```
