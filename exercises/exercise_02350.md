## Exercise
Write a program to take a word as input and scramble its letters, then ask the user to unscramble the word

## Reference Solution
```python
import random

def scramble_word(word):
    """Scramble the letters of a word"""
    word_list = list(word)
    random.shuffle(word_list)
    return ''.join(word_list)

def main():
    # Get the word from the user
    word = input("Enter a word: ")

    # Scramble the word
    scrambled_word = scramble_word(word)

    # Ask the user to unscramble the word
    print(f"Unscramble the word: {scrambled_word}")
    answer = input("Enter your answer: ")

    # Check if the user's answer is correct
    if answer.lower() == word.lower():
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {word}.")

if __name__ == "__main__":
    main()
```
