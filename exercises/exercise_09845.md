## Exercise
Create a program that uses functions to perform basic arithmetic operations. The program should take in two numbers and an operator as input, and then use functions to perform the specified operation. The functions should be able to handle addition, subtraction, multiplication, and division.

## Reference Solution
```python
def add(num1, num2):
    return num1 + num2

def subtract(num1, num2):
    return num1 - num2

def multiply(num1, num2):
    return num1 * num2

def divide(num1, num2):
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return num1 / num2

def calculator(num1, num2, operator):
    operations = {
        '+': add,
        '-': subtract,
        '*': multiply,
        '/': divide
    }

    if operator in operations:
        return operations[operator](num1, num2)
    else:
        raise ValueError("Invalid operator. Please use one of the following: +, -, *, /")

def main():
    num1 = float(input("Enter the first number: "))
    operator = input("Enter the operator (+, -, *, /): ")
    num2 = float(input("Enter the second number: "))

    try:
        result = calculator(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except ZeroDivisionError as e:
        print(e)
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
