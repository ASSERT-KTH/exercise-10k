## Exercise
Write a program that takes a person's weight and height as input and calculates their BMI

## Reference Solution
```python
def calculate_bmi(weight, height):
    """
    Calculate the BMI of a person.

    Args:
        weight (float): The weight of the person in kilograms.
        height (float): The height of the person in meters.

    Returns:
        float: The calculated BMI.
    """
    bmi = weight / (height ** 2)
    return bmi

def get_bmi_category(bmi):
    """
    Get the BMI category based on the calculated BMI.

    Args:
        bmi (float): The calculated BMI.

    Returns:
        str: The BMI category.
    """
    if bmi < 18.5:
        return "Underweight"
    elif bmi < 25:
        return "Normal weight"
    elif bmi < 30:
        return "Overweight"
    else:
        return "Obese"

def main():
    # Get user input
    weight = float(input("Enter your weight in kilograms: "))
    height = float(input("Enter your height in meters: "))

    # Calculate BMI
    bmi = calculate_bmi(weight, height)

    # Get BMI category
    category = get_bmi_category(bmi)

    # Print results
    print(f"Your BMI is: {bmi:.2f}")
    print(f"Your BMI category is: {category}")

if __name__ == "__main__":
    main()
```
