## Exercise
Write a program that takes an array of numbers as input and returns the most frequent element in the array.

## Reference Solution
```python
def most_frequent(nums):
    """
    This function finds the most frequent element in an array of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        The most frequent number in the list.
    """
    # Create a dictionary to store the frequency of each element
    frequency = {}
    
    # Iterate over the list to count the frequency of each element
    for num in nums:
        if num in frequency:
            frequency[num] += 1
        else:
            frequency[num] = 1
    
    # Find the key with the maximum value in the dictionary
    most_frequent_num = max(frequency, key=frequency.get)
    
    return most_frequent_num

# Example usage:
numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
print(most_frequent(numbers))  # Output: 4
```

from collections import Counter

def most_frequent(nums):
    """
    This function finds the most frequent element in an array of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        The most frequent number in the list.
    """
    # Create a Counter object to count the frequency of each element
    frequency = Counter(nums)
    
    # Find the key with the maximum value in the Counter object
    most_frequent_num = frequency.most_common(1)[0][0]
    
    return most_frequent_num

# Example usage:
numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
print(most_frequent(numbers))  # Output: 4
