## Exercise
Write a program to remove a given element from an array.

## Reference Solution
```python
def remove_element(array, target):
    """
    Removes the first occurrence of the target element in the array.

    Args:
    array (list): The input array.
    target: The element to be removed.

    Returns:
    list: The array with the target element removed.
    """
    if target in array:
        array.remove(target)
    return array

# Example usage:
array = [1, 2, 3, 4, 5, 2, 6]
target = 2
print("Original array:", array)
print("Array after removing", target, ":", remove_element(array, target))
```

```python
def remove_all_occurrences(array, target):
    """
    Removes all occurrences of the target element in the array.

    Args:
    array (list): The input array.
    target: The element to be removed.

    Returns:
    list: The array with all occurrences of the target element removed.
    """
    return [element for element in array if element != target]

# Example usage:
array = [1, 2, 3, 4, 5, 2, 6]
target = 2
print("Original array:", array)
print("Array after removing all occurrences of", target, ":", remove_all_occurrences(array, target))
```
