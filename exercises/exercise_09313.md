## Exercise
Write a function that takes a string as input and returns the string with alternating character case swapped.

## Reference Solution
```python
def swap_alternating_case(input_string: str) -> str:
    """
    Swaps the case of alternating characters in a string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The string with alternating character case swapped.
    """
    result = ""
    swap = True  # Flag to track whether to swap case or not

    for char in input_string:
        if char.isalpha():  # Check if character is a letter
            if swap:
                result += char.swapcase()  # Swap case
            else:
                result += char  # Don't swap case
            swap = not swap  # Toggle swap flag
        else:
            result += char  # Non-letter characters are left unchanged

    return result

# Example usage
def main():
    input_str = "Hello, World!"
    print("Input:", input_str)
    print("Output:", swap_alternating_case(input_str))

if __name__ == "__main__":
    main()
```
