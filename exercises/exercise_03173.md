## Exercise
Develop a function that retrieves the value associated with a given key from a dictionary.

## Reference Solution
```python
def get_dict_value(dictionary, key):
    """
    Retrieves the value associated with a given key from a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.

    Returns:
        The value associated with the given key.

    Raises:
        KeyError: If the key is not found in the dictionary.
    """
    return dictionary[key]

# Example usage:
if __name__ == "__main__":
    example_dict = {"name": "John", "age": 30}
    key_to_find = "name"
    
    try:
        value = get_dict_value(example_dict, key_to_find)
        print(f"The value for key '{key_to_find}' is: {value}")
    except KeyError as e:
        print(f"Key '{key_to_find}' not found in the dictionary: {e}")
```

def get_dict_value_with_default(dictionary, key, default=None):
    """
    Retrieves the value associated with a given key from a dictionary.
    If the key is not found, returns the default value.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.
        default: The default value to return if the key is not found.

    Returns:
        The value associated with the given key, or the default value if not found.
    """
    return dictionary.get(key, default)

# Example usage:
if __name__ == "__main__":
    example_dict = {"name": "John", "age": 30}
    key_to_find = "occupation"
    default_value = "Unknown"
    
    value = get_dict_value_with_default(example_dict, key_to_find, default_value)
    print(f"The value for key '{key_to_find}' is: {value}")
