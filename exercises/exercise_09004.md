## Exercise
Write a function that takes a function as an argument and returns a new function with an additional feature.

## Reference Solution
```python
import logging
import functools

# Set up logging
logging.basicConfig(level=logging.INFO)

def log_function_call(func):
    """
    A function wrapper that logs the input arguments and the output of the wrapped function.

    Args:
        func: The function to be wrapped.

    Returns:
        A new function with the additional logging feature.
    """
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # Log the input arguments
        logging.info(f"Calling {func.__name__} with arguments: args={args}, kwargs={kwargs}")

        # Call the wrapped function and store the result
        result = func(*args, **kwargs)

        # Log the output
        logging.info(f"{func.__name__} returned: {result}")

        return result

    return wrapper

# Example usage
@log_function_call
def add(a, b):
    """
    A simple function that adds two numbers.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The sum of a and b.
    """
    return a + b

# Test the wrapped function
result = add(2, 3)
print(result)
```
