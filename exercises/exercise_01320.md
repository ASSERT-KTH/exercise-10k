## Exercise
Write a function to reverse a dictionary where the keys become the values and the values become the keys

## Reference Solution
```python
def reverse_dictionary(input_dict):
    """
    Reverses a dictionary where the keys become the values and the values become the keys.

    Args:
    input_dict (dict): The dictionary to be reversed.

    Returns:
    dict: The reversed dictionary.

    Raises:
    ValueError: If the input dictionary has non-unique values.
    """
    # Check if the input dictionary values are unique
    if len(input_dict.values()) != len(set(input_dict.values())):
        raise ValueError("Input dictionary values must be unique")

    # Reverse the dictionary using dictionary comprehension
    reversed_dict = {v: k for k, v in input_dict.items()}
    return reversed_dict

# Example usage:
input_dict = {"a": 1, "b": 2, "c": 3}
try:
    reversed_dict = reverse_dictionary(input_dict)
    print(reversed_dict)  # Output: {1: 'a', 2: 'b', 3: 'c'}
except ValueError as e:
    print(e)
```
