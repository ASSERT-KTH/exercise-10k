## Exercise
Write a program that takes a story template and word types as input and generates a completed story.

## Reference Solution
```python
import random

# Define a dictionary of word types and their corresponding words
word_types = {
    "noun": ["cat", "dog", "house", "car", "tree"],
    "verb": ["ran", "jumped", "danced", "sang", "played"],
    "adjective": ["happy", "sad", "big", "small", "red"],
    "adverb": ["quickly", "slowly", "loudly", "quietly", "happily"],
    "place": ["park", "beach", "city", "forest", "mountain"]
}

def generate_story(template):
    """
    Generate a completed story based on a given story template.

    Args:
        template (str): The story template with placeholders for word types.

    Returns:
        str: The completed story with placeholders replaced with random words.
    """
    # Split the template into words
    words = template.split()

    # Initialize an empty list to store the completed story
    completed_story = []

    # Iterate over each word in the template
    for word in words:
        # Check if the word is a placeholder for a word type
        if word.startswith("{") and word.endswith("}"):
            # Extract the word type from the placeholder
            word_type = word[1:-1]

            # Check if the word type is valid
            if word_type in word_types:
                # Replace the placeholder with a random word of the corresponding type
                completed_story.append(random.choice(word_types[word_type]))
            else:
                # If the word type is invalid, raise an error
                raise ValueError(f"Invalid word type: {word_type}")
        else:
            # If the word is not a placeholder, add it to the completed story as is
            completed_story.append(word)

    # Join the completed story into a single string
    completed_story = " ".join(completed_story)

    return completed_story

# Example usage
template = "Once upon a time, a {adjective} {noun} {verb} {adverb} through the {place}."
print(generate_story(template))
```
