## Exercise
Write a program to take a set of words as input and generate a random sentence using the words

## Reference Solution
```python
import random

# Predefined sentence templates
sentence_templates = [
    "{subject} {verb} {object}.",
    "{subject} {verb} {object} {adverb}.",
    "{subject} {verb} {object} because {reason}."
]

# Function to generate a random sentence
def generate_sentence(words):
    # Randomly select a sentence template
    template = random.choice(sentence_templates)

    # Define word roles
    word_roles = {
        "subject": ["I", "You", "He", "She", "It"],
        "verb": ["run", "jump", "read", "write"],
        "object": words,
        "adverb": ["quickly", "slowly", "loudly", "wisely"],
        "reason": ["I am happy", "I am sad", "I am tired", "I am hungry"]
    }

    # Randomly select words for each role
    sentence = template.format(
        subject=random.choice(word_roles["subject"]),
        verb=random.choice(word_roles["verb"]),
        object=random.choice(word_roles["object"]),
        adverb=random.choice(word_roles["adverb"]),
        reason=random.choice(word_roles["reason"])
    )

    return sentence

# Example usage
def main():
    words = input("Enter a set of words separated by spaces: ").split()
    print("Random Sentence:", generate_sentence(words))

if __name__ == "__main__":
    main()
```
