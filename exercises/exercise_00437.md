## Exercise
Ask the user for a string and print out the number of consonants in it.

## Reference Solution
```python
def count_consonants(input_string):
    """
    Counts the number of consonants in a given string.

    Args:
        input_string (str): The input string to count consonants from.

    Returns:
        int: The number of consonants in the input string.
    """
    vowels = 'aeiou'
    consonant_count = 0

    for char in input_string.lower():
        if char.isalpha() and char not in vowels:
            consonant_count += 1

    return consonant_count


def main():
    user_string = input("Please enter a string: ")
    consonant_count = count_consonants(user_string)
    print(f"The string '{user_string}' contains {consonant_count} consonants.")


if __name__ == "__main__":
    main()
```
