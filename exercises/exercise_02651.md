## Exercise
Write a function to get a value from a dictionary using the get() method

## Reference Solution
```python
def get_dictionary_value(dictionary, key, default=None):
    """
    Retrieves a value from a dictionary using the get() method.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.
        default: The default value to return if the key is not found. Defaults to None.

    Returns:
        The value associated with the key if found, otherwise the default value.
    """
    return dictionary.get(key, default)

# Example usage:
if __name__ == "__main__":
    example_dict = {"name": "John", "age": 30}
    key_to_find = "name"
    default_value = "Key not found"

    value = get_dictionary_value(example_dict, key_to_find, default_value)
    print(f"The value for the key '{key_to_find}' is: {value}")

    key_not_in_dict = "city"
    value = get_dictionary_value(example_dict, key_not_in_dict, default_value)
    print(f"The value for the key '{key_not_in_dict}' is: {value}")
```
