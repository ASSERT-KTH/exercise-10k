## Exercise
Create a function that takes the length and width of a rectangle as arguments and returns the perimeter

## Reference Solution
```python
def calculate_perimeter(length: float, width: float) -> float:
    """
    Calculate the perimeter of a rectangle.

    Args:
    length (float): The length of the rectangle.
    width (float): The width of the rectangle.

    Returns:
    float: The perimeter of the rectangle.
    """
    if length <= 0 or width <= 0:
        raise ValueError("Length and width must be greater than zero.")

    perimeter = 2 * (length + width)
    return perimeter

# Example usage:
def main():
    length = 5.0
    width = 3.0
    try:
        perimeter = calculate_perimeter(length, width)
        print(f"The perimeter of the rectangle with length {length} and width {width} is {perimeter}.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
