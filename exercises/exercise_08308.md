## Exercise
Create a dictionary to store student grades. The dictionary should have student names as keys and a list of their grades as values. Write a function to add a student, add a grade to a student, and calculate the average grade of a student.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.student_grades = {}

    def add_student(self, name):
        """
        Add a student to the dictionary.

        Args:
            name (str): The name of the student.
        """
        if name not in self.student_grades:
            self.student_grades[name] = []
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def add_grade(self, name, grade):
        """
        Add a grade to a student's list of grades.

        Args:
            name (str): The name of the student.
            grade (float): The grade to be added.
        """
        if name in self.student_grades:
            self.student_grades[name].append(grade)
            print(f"Grade '{grade}' added to student '{name}'.")
        else:
            print(f"Student '{name}' does not exist. Please add the student first.")

    def calculate_average_grade(self, name):
        """
        Calculate the average grade of a student.

        Args:
            name (str): The name of the student.

        Returns:
            float: The average grade of the student.
        """
        if name in self.student_grades:
            if self.student_grades[name]:
                average_grade = sum(self.student_grades[name]) / len(self.student_grades[name])
                return average_grade
            else:
                print(f"Student '{name}' has no grades.")
                return None
        else:
            print(f"Student '{name}' does not exist.")
            return None

    def print_student_grades(self):
        """
        Print the grades of all students.
        """
        for name, grades in self.student_grades.items():
            print(f"Student: {name}, Grades: {grades}")


# Example usage
if __name__ == "__main__":
    student_grades = StudentGrades()

    # Add students
    student_grades.add_student("John Doe")
    student_grades.add_student("Jane Doe")

    # Add grades to students
    student_grades.add_grade("John Doe", 85)
    student_grades.add_grade("John Doe", 90)
    student_grades.add_grade("Jane Doe", 78)
    student_grades.add_grade("Jane Doe", 92)

    # Calculate average grades
    print(f"Average grade of John Doe: {student_grades.calculate_average_grade('John Doe')}")
    print(f"Average grade of Jane Doe: {student_grades.calculate_average_grade('Jane Doe')}")

    # Print student grades
    student_grades.print_student_grades()
```
