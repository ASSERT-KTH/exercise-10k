## Exercise
Create a program that calculates the tip amount based on the total bill and tip percentage.

## Reference Solution
```python
def calculate_tip(total_bill, tip_percentage):
    """
    Calculate the tip amount based on the total bill and tip percentage.

    Args:
        total_bill (float): The total bill amount.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The tip amount.
    """
    tip_amount = (total_bill / 100) * tip_percentage
    return tip_amount

def main():
    print("Tip Calculator Program")
    print("------------------------")

    # Get the total bill from the user
    while True:
        try:
            total_bill = float(input("Enter the total bill amount: $"))
            if total_bill < 0:
                print("Total bill amount cannot be negative.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip percentage from the user
    while True:
        try:
            tip_percentage = float(input("Enter the tip percentage (e.g., 15 for 15%): "))
            if tip_percentage < 0:
                print("Tip percentage cannot be negative.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the tip amount
    tip_amount = calculate_tip(total_bill, tip_percentage)

    # Calculate the total amount (bill + tip)
    total_amount = total_bill + tip_amount

    # Print the results
    print(f"Total Bill: ${total_bill:.2f}")
    print(f"Tip Percentage: {tip_percentage}%")
    print(f"Tip Amount: ${tip_amount:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
