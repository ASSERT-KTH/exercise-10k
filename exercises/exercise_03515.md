## Exercise
Write a program that demonstrates how to handle data type errors using try-except blocks.

## Reference Solution
```python
def divide_numbers(dividend, divisor):
    """
    Divide two numbers and handle potential data type errors.

    Args:
        dividend (int or float): The number being divided.
        divisor (int or float): The number by which we are dividing.

    Returns:
        float: The result of the division.
    """
    try:
        # Attempt to divide the numbers
        result = dividend / divisor
        return result
    except TypeError:
        # Handle the case where either number is not a number
        print("Error: Both inputs must be numbers.")
    except ZeroDivisionError:
        # Handle the case where the divisor is zero
        print("Error: Cannot divide by zero.")
    except Exception as e:
        # Catch any other unexpected errors
        print(f"An unexpected error occurred: {e}")

def main():
    # Test the function with valid inputs
    print("Valid division:")
    print(divide_numbers(10, 2))

    # Test the function with invalid inputs (non-numeric)
    print("\nInvalid division (non-numeric):")
    divide_numbers("ten", 2)

    # Test the function with invalid inputs (division by zero)
    print("\nInvalid division (division by zero):")
    divide_numbers(10, 0)

if __name__ == "__main__":
    main()
```
