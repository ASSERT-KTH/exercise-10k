## Exercise
Write a program that takes the loan amount, interest rate, and number of years as input and returns the monthly payment.

## Reference Solution
```python
def calculate_monthly_payment(loan_amount, interest_rate, years):
    """
    Calculate the monthly payment for a loan.

    Args:
        loan_amount (float): The total amount borrowed.
        interest_rate (float): The annual interest rate as a decimal.
        years (int): The number of years to repay the loan.

    Returns:
        float: The monthly payment amount.
    """
    monthly_interest_rate = interest_rate / 12
    number_of_payments = years * 12
    monthly_payment = loan_amount * monthly_interest_rate * (1 + monthly_interest_rate) ** number_of_payments / ((1 + monthly_interest_rate) ** number_of_payments - 1)
    return round(monthly_payment, 2)

def main():
    loan_amount = float(input("Enter the loan amount: $"))
    interest_rate = float(input("Enter the annual interest rate (in %): ")) / 100
    years = int(input("Enter the number of years: "))

    monthly_payment = calculate_monthly_payment(loan_amount, interest_rate, years)
    print(f"The monthly payment is: ${monthly_payment}")

if __name__ == "__main__":
    main()
```
