```python
def update_dict_key(dict, key, new_value):
    """
    Update the value of a specific key in a dictionary.

    Args:
        dict (dict): The dictionary to update.
        key (any): The key to update.
        new_value (any): The new value for the key.

    Returns:
        dict: The updated dictionary.
    """
    if key in dict:
        dict[key] = new_value
    else:
        raise KeyError(f"The key '{key}' does not exist in the dictionary.")
    return dict
```

```python
def update_dict_key(dict, key, new_value):
    """
    Update the value of a specific key in a dictionary.

    Args:
        dict (dict): The dictionary to update.
        key (any): The key to update.
        new_value (any): The new value for the key.

    Returns:
        dict: The updated dictionary.
    """
    dict.setdefault(key, None)
    dict[key] = new_value
    return dict

# Example usage:
my_dict = {"key1": "value1", "key2": "value2"}
```
