## Exercise
Create a program that manages a sports team roster, using conditional statements to determine the player positions

## Reference Solution
```python
class Player:
    def __init__(self, name, skills, preference):
        self.name = name
        self.skills = skills
        self.preference = preference
        self.position = None

    def assign_position(self):
        if self.skills['speed'] > 7 and self.preference == 'offense':
            self.position = 'Running Back'
        elif self.skills['strength'] > 7 and self.preference == 'offense':
            self.position = 'Tight End'
        elif self.skills['agility'] > 7 and self.preference == 'defense':
            self.position = 'Cornerback'
        elif self.skills['intelligence'] > 7 and self.preference == 'defense':
            self.position = 'Safety'
        else:
            self.position = 'Lineman'

class Team:
    def __init__(self, name):
        self.name = name
        self.players = []

    def add_player(self, player):
        self.players.append(player)

    def display_roster(self):
        print(f"{self.name} Roster:")
        for player in self.players:
            print(f"{player.name} - {player.position}")

def main():
    team = Team("Lions")

    while True:
        print("\n1. Add Player")
        print("2. Display Roster")
        print("3. Exit")
        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter player name: ")
            speed = int(input("Enter player speed (1-10): "))
            strength = int(input("Enter player strength (1-10): "))
            agility = int(input("Enter player agility (1-10): "))
            intelligence = int(input("Enter player intelligence (1-10): "))
            preference = input("Enter player preference (offense/defense): ")
            player = Player(name, {'speed': speed, 'strength': strength, 'agility': agility, 'intelligence': intelligence}, preference)
            player.assign_position()
            team.add_player(player)
        elif choice == "2":
            team.display_roster()
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
