## Exercise
Write a program to check the data type of a variable.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a variable.

    Args:
        var: The variable to check.

    Returns:
        The data type of the variable.
    """
    return type(var)

def main():
    # Test variables
    test_vars = [
        10,  # Integer
        10.5,  # Float
        "Hello",  # String
        True,  # Boolean
        [1, 2, 3],  # List
        (1, 2, 3),  # Tuple
        {"a": 1, "b": 2},  # Dictionary
        {1, 2, 3},  # Set
        None  # NoneType
    ]

    # Check data type for each variable
    for var in test_vars:
        print(f"Variable: {var}, Data Type: {check_data_type(var)}")

if __name__ == "__main__":
    main()
```
