## Exercise
Write a program to calculate the equation of a line

## Reference Solution
```python
def calculate_line_equation(point1, point2):
    """
    Calculate the equation of a line given two points.

    Args:
        point1 (tuple): The first point on the line (x, y)
        point2 (tuple): The second point on the line (x, y)

    Returns:
        tuple: The slope (m) and y-intercept (b) of the line
    """
    x1, y1 = point1
    x2, y2 = point2

    # Calculate the slope
    if x2 - x1 == 0:
        raise ValueError("The line is vertical and has no slope")
    m = (y2 - y1) / (x2 - x1)

    # Calculate the y-intercept
    b = y1 - m * x1

    return m, b

def print_line_equation(m, b):
    """
    Print the equation of a line in slope-intercept form.

    Args:
        m (float): The slope of the line
        b (float): The y-intercept of the line
    """
    if b < 0:
        print(f"y = {m}x - {abs(b)}")
    elif b > 0:
        print(f"y = {m}x + {b}")
    else:
        print(f"y = {m}x")

def main():
    # Example usage
    point1 = (1, 2)
    point2 = (3, 4)
    m, b = calculate_line_equation(point1, point2)
    print(f"Slope (m): {m}")
    print(f"Y-intercept (b): {b}")
    print("Equation of the line:")
    print_line_equation(m, b)

if __name__ == "__main__":
    main()
```
