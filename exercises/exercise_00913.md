## Exercise
Create a program that inspects a given variable and prints out its data type, value, and other relevant information.

## Reference Solution
```python
def inspect_variable(var):
    """
    Inspect a given variable and print out its data type, value, and other relevant information.

    Args:
        var: The variable to be inspected.

    Returns:
        None
    """

    # Get the data type of the variable
    data_type = type(var)

    # Print the data type of the variable
    print(f"Data Type: {data_type.__name__}")

    # Print the value of the variable
    print(f"Value: {var}")

    # Check if the variable is a collection (like list, tuple, dict, set)
    if hasattr(var, '__len__'):
        print(f"Length: {len(var)}")

    # Check if the variable is a dictionary
    if isinstance(var, dict):
        print("Dictionary Keys:")
        for key in var.keys():
            print(key)

    # Check if the variable is a list or a tuple
    if isinstance(var, (list, tuple)):
        print("List/Tuple Elements:")
        for i, element in enumerate(var):
            print(f"Element {i+1}: {element}")

    # Check if the variable is a set
    if isinstance(var, set):
        print("Set Elements:")
        for i, element in enumerate(var):
            print(f"Element {i+1}: {element}")
```
