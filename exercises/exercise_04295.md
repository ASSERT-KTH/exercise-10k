## Exercise
Write a function to find the cartesian product of two dictionaries.

## Reference Solution
```python
import itertools

def cartesian_product(dict1, dict2):
    """
    Calculate the cartesian product of two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        list: A list of tuples, where each tuple is a pair of items from the two dictionaries.
    """
    # Get the items from the dictionaries
    items1 = list(dict1.items())
    items2 = list(dict2.items())

    # Calculate the cartesian product
    product = list(itertools.product(items1, items2))

    return product

# Example usage
dict1 = {'a': 1, 'b': 2}
dict2 = {'c': 3, 'd': 4}

result = cartesian_product(dict1, dict2)
for item in result:
    print(item)
```
