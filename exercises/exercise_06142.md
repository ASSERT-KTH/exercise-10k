## Exercise
Write a function to concatenate two or more strings together.

## Reference Solution
```python
def concatenate_strings(*strings):
    """
    Concatenates two or more strings together.

    Args:
        *strings (str): Variable number of strings to concatenate.

    Returns:
        str: The concatenated string.

    Raises:
        TypeError: If no strings are provided or if a non-string argument is passed.
    """
    if not strings:
        raise TypeError("At least one string is required")
    if not all(isinstance(s, str) for s in strings):
        raise TypeError("All arguments must be strings")

    return "".join(strings)

# Example usage:
if __name__ == "__main__":
    print(concatenate_strings("Hello", " ", "World"))  # Output: "Hello World"
    print(concatenate_strings("This", " is", " a", " test"))  # Output: "This is a test"
```
