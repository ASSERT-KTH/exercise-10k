# Initialize score to 0
score = 0

# Define the questions, answers, and messages
questions = [
    {"question": "What is the capital of France?", "answer": "paris", "message": "Sorry, the correct answer is Paris."},
    {"question": "What is the largest planet in our solar system?", "answer": "jupiter", "message": "Sorry, the correct answer is Jupiter."},
    {"question": "What is the chemical symbol for gold?", "answer": "au", "message": "Sorry, the correct answer is Au."},
    {"question": "Who painted the Starry Night?", "answer": "vincent van gogh", "message": "Sorry, the correct answer is Vincent van Gogh."},
    {"question": "What is the largest mammal?", "answer": "blue whale", "message": "Sorry, the correct answer is Blue Whale."},
]

# Ask the user the questions and update the score
for i, q in enumerate(questions):
    print(f"\nQuestion {i+1}: {q['question']}")
    answer = input("Type your answer: ")
    
    if answer.lower() == q["answer"]:
        print("Correct!")
        score += 1
    else:
        print(q["message"])

# Print the final score
print(f"\nYou scored {score} out of {len(questions)}")
