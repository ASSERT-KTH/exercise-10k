## Exercise
Write a program that takes an input and processes it using a function.

## Reference Solution
```python
def process_input(input_str):
    """
    Processes the input string by converting it to uppercase and counting the number of vowels.
    
    Args:
        input_str (str): The input string to process.
    
    Returns:
        tuple: A tuple containing the processed string and the vowel count.
    """
    # Convert the input string to uppercase
    processed_str = input_str.upper()
    
    # Initialize a counter for vowels
    vowel_count = 0
    
    # Count the number of vowels in the input string
    for char in input_str.lower():
        if char in 'aeiou':
            vowel_count += 1
    
    return processed_str, vowel_count

def main():
    # Get the input from the user
    input_str = input("Please enter a string: ")
    
    # Process the input using the function
    processed_str, vowel_count = process_input(input_str)
    
    # Print the result
    print(f"Processed string: {processed_str}")
    print(f"Vowel count: {vowel_count}")

if __name__ == "__main__":
    main()
```
