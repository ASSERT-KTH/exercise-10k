## Exercise
Write a function to reverse the order of elements in a given array. The function should take an array as input and return the reversed array.

## Reference Solution
```python
function reverseArr(arr, start, end) {
  while (start < end) {
    let temp = arr[start];
    arr[start] = arr[end];
    arr[end] = temp;
    start++;
    end--;
  }
  return arr;
}

// Example usage:
let arr = [1, 2, 3, 4, 5, 6, 7];
let start = 0;
let end = arr.length - 1;
let reversedArr = reverseArr(arr, start, end);
```
