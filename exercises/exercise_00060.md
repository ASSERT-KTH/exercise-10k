## Exercise
Implement the KMP algorithm to search for a substring in a string

## Reference Solution
```python
def compute_prefix_function(pattern):
    """
    Compute the prefix function for the given pattern.

    The prefix function is a list where the i-th element is the length of the
    longest proper prefix of the pattern that is also a proper suffix of the
    pattern up to the i-th position.

    Args:
        pattern (str): The pattern to search for.

    Returns:
        list: The prefix function for the pattern.
    """
    prefix = [0] * len(pattern)
    j = 0
    for i in range(1, len(pattern)):
        while j > 0 and pattern[j] != pattern[i]:
            j = prefix[j - 1]
        if pattern[j] == pattern[i]:
            j += 1
        prefix[i] = j
    return prefix


def kmp_search(text, pattern):
    """
    Search for the given pattern in the given text using the KMP algorithm.

    Args:
        text (str): The text to search in.
        pattern (str): The pattern to search for.

    Returns:
        list: A list of indices where the pattern was found in the text.
    """
    prefix = compute_prefix_function(pattern)
    indices = []
    j = 0
    for i in range(len(text)):
        while j > 0 and text[i] != pattern[j]:
            j = prefix[j - 1]
        if text[i] == pattern[j]:
            j += 1
        if j == len(pattern):
            indices.append(i - j + 1)
            j = prefix[j - 1]
    return indices


# Example usage:
text = "abxabcabcaby"
pattern = "abcaby"
indices = kmp_search(text, pattern)
print("Pattern found at indices:", indices)
```
