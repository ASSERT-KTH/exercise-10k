## Exercise
Create a function that uses alerting to notify others of issues and errors

```python
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
import requests

def send_alert(notification_type, message, email_recipients=None, webhook_url=None):
    """
    Sends an alert notification via email or webhook.

    Args:
    - notification_type (str): The type of notification (e.g., "error", "warning", "info").
    - message (str): The message to be sent with the notification.
    - email_recipients (list): A list of email recipients. Defaults to None.
    - webhook_url (str): The URL of the webhook to send the notification to. Defaults to None.
    """

    # Email configuration
    email_config = {
        "smtp_server": "smtp.gmail.com",
        "smtp_port": 587,
        "from_email": "your-email@gmail.com",
        "password": "your-email-password"
    }

    # Send email notification
    if email_recipients:
        msg = MIMEMultipart()
        msg["From"] = email_config["from_email"]
        msg["To"] = ", ".join(email_recipients)
        msg["Subject"] = f"{notification_type.capitalize()} Alert"
        msg.attach(MIMEText(message, "plain"))

        server = smtplib.SMTP(email_config["smtp_server"], email_config["smtp_port"])
        server.starttls()
        server.login(email_config["from_email"], email_config["password"])
        server.sendmail(email_config["from_email"], email_recipients, msg.as_string())
        server.quit()

    # Send webhook notification
    if webhook_url:
        payload = {
            "notification_type": notification_type,
            "message": message
        }
        response = requests.post(webhook_url, json=payload)
        if response.status_code != 200:
            print(f"Failed to send webhook notification: {response.text}")

# Example usage:
if __name__ == "__main__":
    send_alert(
        notification_type="error",
        message="An error occurred in the application.",
        email_recipients=["recipient1@example.com", "recipient2@example.com"],
        webhook_url="https://example.com/webhook"
    )
```
