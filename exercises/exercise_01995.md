## Exercise
Develop a program to generate a student grade report using conditional statements to determine the grade

## Reference Solution
```python
def calculate_grade(score):
    """
    Calculate the grade based on the score.

    Args:
        score (float): The score of the student.

    Returns:
        str: The grade of the student.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

def generate_report(student_name, scores):
    """
    Generate the student grade report.

    Args:
        student_name (str): The name of the student.
        scores (dict): A dictionary containing the scores of the student in different subjects.

    Returns:
        str: The student grade report.
    """
    report = f"Student Name: {student_name}\n"
    report += "------------------------------------\n"
    for subject, score in scores.items():
        grade = calculate_grade(score)
        report += f"{subject}: {score} - {grade}\n"
    report += "------------------------------------\n"
    return report

def main():
    student_name = input("Enter the student name: ")
    num_subjects = int(input("Enter the number of subjects: "))

    scores = {}
    for i in range(num_subjects):
        subject = input(f"Enter the subject {i+1}: ")
        score = float(input(f"Enter the score for {subject}: "))
        scores[subject] = score

    report = generate_report(student_name, scores)
    print(report)

if __name__ == "__main__":
    main()
```
