## Exercise
Given a list of integers, write a program to determine if it is a subset of another list.

## Reference Solution
```python
def is_subset(subset, main_set):
    """
    Checks if a list is a subset of another list.

    Args:
        subset (list): The list to check.
        main_set (list): The main list.

    Returns:
        bool: True if subset is a subset of main_set, False otherwise.
    """
    return set(subset).issubset(set(main_set))


def main():
    # Example usage
    main_list = [1, 2, 3, 4, 5, 6]
    subset_list = [2, 4, 6]

    if is_subset(subset_list, main_list):
        print(f"{subset_list} is a subset of {main_list}")
    else:
        print(f"{subset_list} is not a subset of {main_list}")


if __name__ == "__main__":
    main()
```
