## Exercise
Write a program to scramble a word and ask the user to unscramble it. After each guess, print out a message indicating if the guess is correct or not.

## Reference Solution
```python
import random

def scramble_word(word):
    """Scramble the letters of a word"""
    letters = list(word)
    random.shuffle(letters)
    return ''.join(letters)

def main():
    # List of words to choose from
    words = ['apple', 'banana', 'cherry', 'date', 'elderberry']

    # Choose a random word
    word = random.choice(words)

    # Scramble the word
    scrambled_word = scramble_word(word)

    print("Unscramble the word: ", scrambled_word)

    while True:
        # Ask the user for their guess
        guess = input("Enter your guess: ")

        # Check if the guess is correct
        if guess.lower() == word:
            print("Congratulations! You unscrambled the word correctly.")
            break
        else:
            print("Sorry, that's not correct. Try again.")

if __name__ == "__main__":
    main()
```
