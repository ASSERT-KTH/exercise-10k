## Exercise
Write a program that sorts a given list in ascending or descending order

## Reference Solution
```python
def sort_list(input_list, order):
    """
    Sorts a given list in ascending or descending order.

    Args:
        input_list (list): The list to be sorted.
        order (str): The order of sorting. Can be 'asc' or 'desc'.

    Returns:
        list: The sorted list.
    """
    if order.lower() == 'asc':
        return sorted(input_list)
    elif order.lower() == 'desc':
        return sorted(input_list, reverse=True)
    else:
        raise ValueError("Invalid order. Please use 'asc' or 'desc'.")

def main():
    # Example usage
    numbers = [64, 34, 25, 12, 22, 11, 90]
    print("Original List:", numbers)

    sorted_numbers_asc = sort_list(numbers, 'asc')
    print("Sorted List (Ascending):", sorted_numbers_asc)

    sorted_numbers_desc = sort_list(numbers, 'desc')
    print("Sorted List (Descending):", sorted_numbers_desc)

if __name__ == "__main__":
    main()
```
