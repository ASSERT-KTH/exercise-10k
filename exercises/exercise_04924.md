## Exercise
Write a program that calculates a student's GPA given their grades and credits for each course. The program should ask the user to enter the grades and credits for each course and then calculate the GPA.

## Reference Solution
```python
# Define a function to calculate the GPA
def calculate_gpa(grades, credits):
    """
    Calculate the GPA given a list of grades and credits.

    Args:
        grades (list): A list of grades (e.g., 'A', 'B', 'C', etc.)
        credits (list): A list of credits corresponding to each grade

    Returns:
        float: The calculated GPA
    """
    # Define a dictionary to map grades to their corresponding GPA values
    grade_map = {
        'A': 4.0,
        'A-': 3.7,
        'B+': 3.3,
        'B': 3.0,
        'B-': 2.7,
        'C+': 2.3,
        'C': 2.0,
        'C-': 1.7,
        'D+': 1.3,
        'D': 1.0,
        'D-': 0.7,
        'F': 0.0
    }

    # Initialize variables to calculate the GPA
    total_credits = 0
    total_gpa = 0.0

    # Iterate over the grades and credits
    for grade, credit in zip(grades, credits):
        # Check if the grade is valid
        if grade.upper() in grade_map:
            # Calculate the GPA for this course
            course_gpa = grade_map[grade.upper()] * credit
            # Add the course GPA to the total GPA
            total_gpa += course_gpa
            # Add the course credits to the total credits
            total_credits += credit
        else:
            print(f"Invalid grade: {grade}. Skipping this course.")

    # Check if there are any valid courses
    if total_credits == 0:
        return 0.0

    # Calculate the overall GPA
    gpa = total_gpa / total_credits

    return gpa


# Define the main function
def main():
    """
    Ask the user for grades and credits for each course and calculate the GPA.
    """
    # Initialize lists to store grades and credits
    grades = []
    credits = []

    # Ask the user for the number of courses
    num_courses = int(input("Enter the number of courses: "))

    # Ask the user for grades and credits for each course
    for i in range(num_courses):
        grade = input(f"Enter the grade for course {i+1} (e.g., A, B, C, etc.): ")
        credit = float(input(f"Enter the credits for course {i+1}: "))
        grades.append(grade)
        credits.append(credit)

    # Calculate the GPA
    gpa = calculate_gpa(grades, credits)

    # Print the GPA
    print(f"Your GPA is: {gpa:.2f}")


# Call the main function
if __name__ == "__main__":
    main()
```
