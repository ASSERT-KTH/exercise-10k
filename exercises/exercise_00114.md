```python
def count_occurrences(array, num):
    """
    Counts the occurrences of a given number in an array.

    Args:
        array (list): The input array.
        num (int): The number to be searched.

    Returns:
        int: The number of occurrences of the given number.
    """
    count_dict = {}
    for i in array:
        if i in count_dict:
            count_dict[i] += 1
        else:
            count_dict[i] = 1
    return count_dict.get(num, 0)


# Example usage:
array = [1, 2, 3, 4, 2, 3, 4, 5, 2, 3, 4]
num = 2
print(f"The number {num} occurs {count_occurrences(array, num)} times in the array.")
```
