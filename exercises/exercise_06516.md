## Exercise
Write a program to compare different data types (int, float, str, etc.).

## Reference Solution
```python
def compare_data_types():
    # Define variables of different data types
    int_var = 10
    float_var = 10.5
    str_var = "Hello, World!"
    bool_var = True
    list_var = [1, 2, 3]
    tuple_var = (1, 2, 3)
    dict_var = {"name": "John", "age": 30}

    # Print data type and value of each variable
    print("Data Type Comparison:")
    print("------------------------")
    print(f"Integer: {type(int_var)} - {int_var}")
    print(f"Float: {type(float_var)} - {float_var}")
    print(f"String: {type(str_var)} - {str_var}")
    print(f"Boolean: {type(bool_var)} - {bool_var}")
    print(f"List: {type(list_var)} - {list_var}")
    print(f"Tuple: {type(tuple_var)} - {tuple_var}")
    print(f"Dictionary: {type(dict_var)} - {dict_var}")

    # Compare data types
    print("\nData Type Comparison Results:")
    print("---------------------------------")
    print(f"Is int_var an integer? {isinstance(int_var, int)}")
    print(f"Is float_var a float? {isinstance(float_var, float)}")
    print(f"Is str_var a string? {isinstance(str_var, str)}")
    print(f"Is bool_var a boolean? {isinstance(bool_var, bool)}")
    print(f"Is list_var a list? {isinstance(list_var, list)}")
    print(f"Is tuple_var a tuple? {isinstance(tuple_var, tuple)}")
    print(f"Is dict_var a dictionary? {isinstance(dict_var, dict)}")

if __name__ == "__main__":
    compare_data_types()
```
