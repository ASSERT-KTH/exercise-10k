## Exercise
Find the longest increasing subsequence in an array of integers

## Reference Solution

def longest_increasing_subsequence(arr):
    """
    Find the longest increasing subsequence in an array of integers.

    Args:
        arr (list): The input array of integers.

    Returns:
        list: The longest increasing subsequence.
    """

    # Create a dynamic programming table
    dp = [1] * len(arr)

    # Initialize a table to store the previous element of each element
    prev = [-1] * len(arr)

    # Fill the table
    for i in range(1, len(arr)):
        for j in range(i):
            if arr[i] > arr[j] and dp[i] < dp[j] + 1:
                dp[i] = dp[j] + 1
                prev[i] = j

    # Find the maximum length
    max_length = max(dp)
    max_index = dp.index(max_length)

    # Reconstruct the subsequence
    subsequence = []
    while max_index != -1:
        subsequence.append(arr[max_index])
        max_index = prev[max_index]

    # Return the subsequence in the correct order
    return subsequence[::-1]

# Example usage
arr = [10, 22, 9, 33, 21, 50, 41, 60, 80]
print(longest_increasing_subsequence(arr))
