## Exercise
Write a program that takes a string input from the user, reverses the string, and checks if the original string is a palindrome. A palindrome is a string that reads the same backward as forward.

## Reference Solution
```python
def reverse_string(input_str):
    """
    Reverses the input string.

    Args:
    input_str (str): The string to be reversed.

    Returns:
    str: The reversed string.
    """
    return input_str[::-1]


def is_palindrome(input_str):
    """
    Checks if the input string is a palindrome.

    Args:
    input_str (str): The string to be checked.

    Returns:
    bool: True if the string is a palindrome, False otherwise.
    """
    reversed_str = reverse_string(input_str)
    return input_str == reversed_str


def main():
    # Take user input
    input_str = input("Please enter a string: ")

    # Remove leading and trailing whitespace
    input_str = input_str.strip()

    # Reverse the string
    reversed_str = reverse_string(input_str)

    # Check if the string is a palindrome
    is_palindrome_result = is_palindrome(input_str)

    # Print the results
    print(f"Original string: {input_str}")
    print(f"Reversed string: {reversed_str}")
    print(f"Is '{input_str}' a palindrome? {is_palindrome_result}")


if __name__ == "__main__":
    main()
```
