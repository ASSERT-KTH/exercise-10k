## Exercise
Convert a string to an integer and perform arithmetic operations on it.

## Reference Solution

def convert_and_operate(num_str, operation, value):
    """
    Converts a string to an integer and performs an arithmetic operation.

    Args:
        num_str (str): The string to be converted to an integer.
        operation (str): The arithmetic operation to be performed. Can be 'add', 'subtract', 'multiply', or 'divide'.
        value (int): The value to be used in the arithmetic operation.

    Returns:
        int: The result of the arithmetic operation.
    """
    try:
        # Attempt to convert the string to an integer
        num = int(num_str)
        
        # Perform the specified arithmetic operation
        if operation == 'add':
            result = num + value
        elif operation == 'subtract':
            result = num - value
        elif operation == 'multiply':
            result = num * value
        elif operation == 'divide':
            # Check for division by zero
            if value == 0:
                raise ZeroDivisionError("Cannot divide by zero")
            result = num // value  # Use integer division
        else:
            raise ValueError("Invalid operation")
        
        return result
    
    except ValueError as e:
        print(f"Error: {e}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")

# Example usage
if __name__ == "__main__":
    num_str = "10"
    operation = "add"
    value = 5
    
    result = convert_and_operate(num_str, operation, value)
    print(f"The result of {operation}ing {value} to {num_str} is: {result}")
