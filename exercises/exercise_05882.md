## Exercise
Write a function that takes an array and an index as input and returns the element at that index.

## Reference Solution
```python
def get_element_at_index(array, index):
    """
    Retrieves the element at a given index in an array.

    Args:
    array (list): The input array.
    index (int): The index of the element to retrieve.

    Returns:
    any: The element at the specified index if it exists.

    Raises:
    TypeError: If the input array is not a list or the index is not an integer.
    IndexError: If the index is out of range.
    """
    # Check if the input array is a list
    if not isinstance(array, list):
        raise TypeError("Input array must be a list.")

    # Check if the index is an integer
    if not isinstance(index, int):
        raise TypeError("Index must be an integer.")

    # Check if the index is within the bounds of the array
    if index < 0 or index >= len(array):
        raise IndexError("Index out of range.")

    # Return the element at the specified index
    return array[index]

# Example usage
if __name__ == "__main__":
    array = [1, 2, 3, 4, 5]
    index = 2
    try:
        element = get_element_at_index(array, index)
        print(f"Element at index {index}: {element}")
    except (TypeError, IndexError) as e:
        print(f"Error: {e}")
```
