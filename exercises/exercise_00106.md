## Exercise
Write a program to find all prime numbers up to a given number.

## Reference Solution
```python
def find_primes(n):
    """
    Finds all prime numbers up to n.

    Args:
        n (int): The upper limit for finding prime numbers.

    Returns:
        list: A list of prime numbers up to n.
    """
    sieve = [True] * (n + 1)
    sieve[0:2] = [False, False]  # 0 and 1 are not prime numbers

    for num in range(2, int(n**0.5) + 1):
        if sieve[num]:
            for multiple in range(num**2, n + 1, num):
                sieve[multiple] = False

    return [num for num, is_prime in enumerate(sieve) if is_prime]


def main():
    n = int(input("Enter a number: "))
    primes = find_primes(n)
    print("Prime numbers up to", n, "are:")
    print(primes)


if __name__ == "__main__":
    main()
```
