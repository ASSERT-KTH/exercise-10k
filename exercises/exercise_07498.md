## Exercise
Perform basic set operations (add, remove, union, intersection) and print out the resulting set.

## Reference Solution
```python
def main():
    # Initialize two sets
    set1 = {1, 2, 3, 4, 5}
    set2 = {4, 5, 6, 7, 8}

    print("Initial Sets:")
    print("Set 1:", set1)
    print("Set 2:", set2)

    # Add an element to set1
    print("\nAdding 9 to Set 1:")
    set1.add(9)
    print("Set 1:", set1)

    # Remove an element from set1
    print("\nRemoving 5 from Set 1:")
    set1.discard(5)
    print("Set 1:", set1)

    # Union of set1 and set2
    print("\nUnion of Set 1 and Set 2:")
    union_set = set1.union(set2)
    print("Union Set:", union_set)

    # Intersection of set1 and set2
    print("\nIntersection of Set 1 and Set 2:")
    intersection_set = set1.intersection(set2)
    print("Intersection Set:", intersection_set)

if __name__ == "__main__":
    main()
```
