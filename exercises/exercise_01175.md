## Exercise
Given two strings, find the first occurrence of the second string in the first string

## Reference Solution
```python
def find_first_occurrence(main_string, sub_string):
    """
    Find the first occurrence of a substring in a string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the first occurrence of the substring. Returns -1 if not found.
    """
    return main_string.find(sub_string)

# Example usage:
main_string = "Hello, World! Hello again!"
sub_string = "Hello"
index = find_first_occurrence(main_string, sub_string)
if index != -1:
    print(f"The first occurrence of '{sub_string}' is at index {index}.")
else:
    print(f"'{sub_string}' not found in the main string.")
```

def find_first_occurrence(main_string, sub_string):
    """
    Find the first occurrence of a substring in a string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the first occurrence of the substring. Returns -1 if not found.
    """
    for i in range(len(main_string) - len(sub_string) + 1):
        if main_string[i:i + len(sub_string)] == sub_string:
            return i
    return -1

import re

def find_first_occurrence(main_string, sub_string):
    """
    Find the first occurrence of a substring in a string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the first occurrence of the substring. Returns -1 if not found.
    """
    match = re.search(sub_string, main_string)
    if match:
        return match.start()
    return -1

# Example usage:
main_string = "Hello, World! Hello again!"
