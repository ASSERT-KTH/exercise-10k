## Exercise
Write a program that takes a word as input and generates a list of spelling suggestions.

## Reference Solution
```python
import nltk
from nltk.metrics import edit_distance
from collections import defaultdict
from difflib import get_close_matches

# Download the required NLTK data
nltk.download('words')

# Load the English words dataset
english_words = set(w.lower() for w in nltk.corpus.words.words())

def generate_suggestions(word, num_suggestions=5):
    """
    Generate a list of spelling suggestions for a given word.

    Args:
        word (str): The word for which to generate suggestions.
        num_suggestions (int): The number of suggestions to generate.

    Returns:
        list: A list of spelling suggestions.
    """
    # Convert the word to lowercase
    word = word.lower()

    # Get the close matches from the English words dataset
    close_matches = get_close_matches(word, english_words, n=num_suggestions, cutoff=0.6)

    # If no close matches are found, use edit distance to find similar words
    if not close_matches:
        similar_words = []
        for english_word in english_words:
            distance = edit_distance(word, english_word)
            if distance <= 2:
                similar_words.append((english_word, distance))
        similar_words.sort(key=lambda x: x[1])
        close_matches = [word for word, distance in similar_words[:num_suggestions]]

    return close_matches

def main():
    word = input("Enter a word: ")
    suggestions = generate_suggestions(word)
    print("Spelling suggestions:")
    for suggestion in suggestions:
        print(suggestion)

if __name__ == "__main__":
    main()
```
