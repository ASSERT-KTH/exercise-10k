## Exercise
Create a program that plays a game of Rock, Paper, Scissors with the user. The program should use a while loop to keep playing until the user chooses to quit.

## Reference Solution
```python
import random

def get_computer_choice():
    """Returns the computer's choice."""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def determine_winner(user_choice, computer_choice):
    """Determines the winner based on the user's and computer's choices."""
    if user_choice == computer_choice:
        return 'tie'
    elif (user_choice == 'rock' and computer_choice == 'scissors') or \
         (user_choice == 'paper' and computer_choice == 'rock') or \
         (user_choice == 'scissors' and computer_choice == 'paper'):
        return 'user'
    else:
        return 'computer'

def main():
    """Plays a game of Rock, Paper, Scissors with the user."""
    while True:
        user_choice = input("Enter your choice (rock, paper, scissors, or quit): ").lower()
        if user_choice == 'quit':
            print("Thanks for playing!")
            break
        elif user_choice not in ['rock', 'paper', 'scissors']:
            print("Invalid choice. Please enter rock, paper, or scissors.")
            continue

        computer_choice = get_computer_choice()
        print(f"\nYou chose {user_choice}, computer chose {computer_choice}.\n")

        winner = determine_winner(user_choice, computer_choice)
        if winner == 'tie':
            print(f"Both players selected {user_choice}. It's a tie!")
        elif winner == 'user':
            print("You win!")
        else:
            print("Computer wins!\n")

if __name__ == "__main__":
    main()
```
