## Exercise
Create a program that calculates the shipping cost based on the package weight and destination. The program should use if/else statements to determine the shipping cost.

## Reference Solution
```python
def calculate_shipping_cost(weight, destination):
    """
    Calculate the shipping cost based on the package weight and destination.

    Args:
        weight (float): The weight of the package in pounds.
        destination (str): The destination of the package (e.g., "local", "regional", "national", "international").

    Returns:
        float: The shipping cost.
    """
    if destination == "local":
        if weight <= 2:
            return 3.00
        elif weight <= 6:
            return 5.00
        else:
            return 7.00
    elif destination == "regional":
        if weight <= 2:
            return 5.00
        elif weight <= 6:
            return 8.00
        else:
            return 12.00
    elif destination == "national":
        if weight <= 2:
            return 8.00
        elif weight <= 6:
            return 12.00
        else:
            return 18.00
    elif destination == "international":
        if weight <= 2:
            return 15.00
        elif weight <= 6:
            return 25.00
        else:
            return 40.00
    else:
        raise ValueError("Invalid destination")

def main():
    # Example usage:
    weight = float(input("Enter the package weight in pounds: "))
    destination = input("Enter the destination (local, regional, national, international): ")
    shipping_cost = calculate_shipping_cost(weight, destination)
    print(f"The shipping cost is: ${shipping_cost:.2f}")

if __name__ == "__main__":
    main()
```
