## Exercise
Create a function that generates the Fibonacci sequence up to a specified number of terms.

## Reference Solution
```python
def generate_fibonacci(n):
    """
    Generate the Fibonacci sequence up to a specified number of terms.

    Args:
        n (int): The number of terms in the Fibonacci sequence.

    Returns:
        list: A list of integers representing the Fibonacci sequence.

    Raises:
        ValueError: If n is a non-positive integer.
    """
    if not isinstance(n, int) or n <= 0:
        raise ValueError("n must be a positive integer")

    # Base case: Fibonacci sequence with 1 term
    if n == 1:
        return [0]

    # Base case: Fibonacci sequence with 2 terms
    if n == 2:
        return [0, 1]

    # Initialize the Fibonacci sequence with the first two terms
    fib_sequence = [0, 1]

    # Generate the Fibonacci sequence up to n terms
    while len(fib_sequence) < n:
        # Calculate the next term as the sum of the last two terms
        next_term = fib_sequence[-1] + fib_sequence[-2]
        fib_sequence.append(next_term)

    return fib_sequence


# Example usage:
def main():
    num_terms = 10
    fib_sequence = generate_fibonacci(num_terms)
    print(f"Fibonacci sequence with {num_terms} terms: {fib_sequence}")


if __name__ == "__main__":
    main()
```
