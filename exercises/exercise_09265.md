## Exercise
Create a program that checks the data type of a given input and prints it out. The program should be able to handle integers, floats, and strings.

## Reference Solution
```python
def check_data_type(input_value):
    """
    Checks the data type of the input value and prints it out.
    
    Args:
        input_value: The input value to check.
    
    Returns:
        None
    """
    # Check the type of the input value
    if isinstance(input_value, int):
        print(f"The input '{input_value}' is of type integer.")
    elif isinstance(input_value, float):
        print(f"The input '{input_value}' is of type float.")
    elif isinstance(input_value, str):
        print(f"The input '{input_value}' is of type string.")
    else:
        print(f"The input '{input_value}' is of type {type(input_value).__name__}.")

def main():
    # Get the input from the user
    input_value = input("Enter a value: ")
    
    # Try to convert the input to an integer or float
    try:
        input_value = int(input_value)
    except ValueError:
        try:
            input_value = float(input_value)
        except ValueError:
            # If it's not a number, treat it as a string
            pass
    
    # Check the data type of the input
    check_data_type(input_value)

if __name__ == "__main__":
    main()
```
