## Exercise
Create a program to solve a complex problem using nested conditional statements

## Reference Solution
```python
# Define a dictionary of courses with their prerequisites
courses = {
    "Math": {
        "Course1": {"name": "Calculus", "prerequisites": []},
        "Course2": {"name": "Linear Algebra", "prerequisites": ["Calculus"]},
        "Course3": {"name": "Differential Equations", "prerequisites": ["Linear Algebra"]}
    },
    "Computer Science": {
        "Course1": {"name": "Introduction to Programming", "prerequisites": []},
        "Course2": {"name": "Data Structures", "prerequisites": ["Introduction to Programming"]},
        "Course3": {"name": "Algorithms", "prerequisites": ["Data Structures"]}
    },
    "Engineering": {
        "Course1": {"name": "Introduction to Engineering", "prerequisites": []},
        "Course2": {"name": "Thermodynamics", "prerequisites": ["Introduction to Engineering"]},
        "Course3": {"name": "Mechanics of Materials", "prerequisites": ["Thermodynamics"]}
    }
}

def get_major():
    """Ask for the student's major"""
    print("Available majors:")
    for i, major in enumerate(courses.keys()):
        print(f"{i+1}. {major}")
    major_choice = int(input("Enter the number of your major: "))
    return list(courses.keys())[major_choice - 1]

def get_semester():
    """Ask for the student's semester"""
    print("Available semesters:")
    for i in range(1, 9):
        print(f"{i}. Semester {i}")
    semester_choice = int(input("Enter the number of your semester: "))
    return semester_choice

def display_courses(major):
    """Display the available courses for the student's major"""
    print(f"Available courses for {major}:")
    for i, course in enumerate(courses[major].values()):
        print(f"{i+1}. {course['name']}")

def enroll_in_course(major, course_name, student_schedule, student_prerequisites):
    """Enroll the student in a course if they have the required prerequisites"""
    course = next((course for course in courses[major].values() if course["name"] == course_name), None)
    if course:
        prerequisites = course["prerequisites"]
        if all(prerequisite in student_prerequisites for prerequisite in prerequisites):
            student_schedule.append(course_name)
            print(f"You have been enrolled in {course_name}.")
        else:
            print(f"You do not have the required prerequisites for {course_name}.")
    else:
        print(f"{course_name} is not available for {major}.")

def main():
    major = get_major()
    semester = get_semester()
    student_schedule = []
    student_prerequisites = []

    display_courses(major)
    for i, course in enumerate(courses[major].values()):
        course_name = course["name"]
        enroll_choice = input(f"Do you want to enroll in {course_name}? (yes/no): ")
        if enroll_choice.lower() == "yes":
            enroll_in_course(major, course_name, student_schedule, student_prerequisites)

    print(f"Your schedule for semester {semester} is:")
    for course in student_schedule:
        print(course)

if __name__ == "__main__":
    main()
```
