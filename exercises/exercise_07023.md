## Exercise
Create a program that simulates a phonebook using a dictionary. The program should have the following functionalities: add a new contact, delete a contact, search for a contact by name, and display all contacts.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone_number):
        """Add a new contact to the phonebook."""
        self.contacts[name] = phone_number
        print(f"Contact {name} added successfully.")

    def delete_contact(self, name):
        """Delete a contact from the phonebook."""
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact {name} deleted successfully.")
        else:
            print(f"Contact {name} not found.")

    def search_contact(self, name):
        """Search for a contact by name."""
        if name in self.contacts:
            print(f"Contact {name} found with phone number: {self.contacts[name]}")
        else:
            print(f"Contact {name} not found.")

    def display_contacts(self):
        """Display all contacts in the phonebook."""
        if self.contacts:
            print("Contacts:")
            for name, phone_number in self.contacts.items():
                print(f"{name}: {phone_number}")
        else:
            print("No contacts found.")


def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook Menu:")
        print("1. Add Contact")
        print("2. Delete Contact")
        print("3. Search Contact")
        print("4. Display Contacts")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter contact name to delete: ")
            phonebook.delete_contact(name)
        elif choice == "3":
            name = input("Enter contact name to search: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Exiting phonebook.")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
