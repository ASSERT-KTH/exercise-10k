## Exercise
Write a program that creates a dictionary of words and their meanings. The program should allow the user to look up words and provide their meanings.

## Reference Solution
```python
class Dictionary:
    def __init__(self):
        self.dictionary = {}

    def add_word(self, word, meaning):
        """Add a new word to the dictionary."""
        self.dictionary[word.lower()] = meaning

    def look_up_word(self, word):
        """Look up the meaning of a word in the dictionary."""
        return self.dictionary.get(word.lower(), "Word not found in dictionary.")

    def display_dictionary(self):
        """Display all words and their meanings in the dictionary."""
        for word, meaning in self.dictionary.items():
            print(f"{word.capitalize()}: {meaning}")


def main():
    dictionary = Dictionary()

    # Pre-populate the dictionary with some words
    dictionary.add_word("hello", "a greeting")
    dictionary.add_word("world", "the earth and its inhabitants")
    dictionary.add_word("python", "a programming language")

    while True:
        print("\nDictionary Program")
        print("1. Look up a word")
        print("2. Add a new word")
        print("3. Display dictionary")
        print("4. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            word = input("Enter a word to look up: ")
            print(f"Meaning: {dictionary.look_up_word(word)}")
        elif choice == "2":
            word = input("Enter a new word: ")
            meaning = input("Enter the meaning of the word: ")
            dictionary.add_word(word, meaning)
            print(f"Added '{word}' to the dictionary.")
        elif choice == "3":
            print("\nDictionary:")
            dictionary.display_dictionary()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
