## Exercise
Write a function that unflattens a dictionary into a nested dictionary

## Reference Solution
```python
def unflatten_dict(flattened_dict):
    """
    Unflattens a dictionary into a nested dictionary.

    Args:
        flattened_dict (dict): The dictionary to unflatten.

    Returns:
        dict: The unflattened dictionary.
    """
    unflattened_dict = {}

    for key, value in flattened_dict.items():
        current_dict = unflattened_dict
        key_parts = key.split('.')

        # Iterate over all key parts except the last one
        for key_part in key_parts[:-1]:
            if key_part not in current_dict:
                current_dict[key_part] = {}
            current_dict = current_dict[key_part]

        # Set the value for the last key part
        current_dict[key_parts[-1]] = value

    return unflattened_dict


# Example usage
if __name__ == "__main__":
    flattened_dict = {
        "a.b.c": 1,
        "a.b.d": 2,
        "e.f.g": 3
    }

    unflattened_dict = unflatten_dict(flattened_dict)
    print(unflattened_dict)
```
