## Exercise
Develop a program to simulate a coin flip using conditional statements

## Reference Solution
```python
import random

def coin_flip():
    """
    Simulates a coin flip and returns the outcome.
    """
    # Generate a random number between 0 and 1
    outcome = random.randint(0, 1)
    
    # Use conditional statements to determine the outcome
    if outcome == 0:
        return "Heads"
    else:
        return "Tails"

def main():
    """
    Main function to run the coin flip simulator.
    """
    print("Coin Flip Simulator")
    print("--------------------")
    
    # Ask the user if they want to flip the coin
    response = input("Do you want to flip the coin? (yes/no): ")
    
    # Use conditional statements to handle user input
    if response.lower() == "yes":
        outcome = coin_flip()
        print(f"The result is: {outcome}")
    elif response.lower() == "no":
        print("Goodbye!")
    else:
        print("Invalid input. Please try again.")

if __name__ == "__main__":
    main()
```
