## Exercise
Write a function that takes a name and an age, with the age defaulting to 25 if not provided.

## Reference Solution
```python
class Person:
    """A class to represent a person."""
    def __init__(self, name: str, age: int = 25):
        """
        Initialize a Person object.

        Args:
        - name (str): The person's name.
        - age (int): The person's age (default is 25).
        """
        self.name = name
        self.age = age

def create_person(name: str, age: int = 25) -> Person:
    """
    Create a person object with a name and an age.

    Args:
    - name (str): The person's name.
    - age (int): The person's age (default is 25).

    Returns:
    - Person: A person object with the given name and age.
    """
    return Person(name, age)

# Example usage:
def main():
    person1 = create_person("John Doe", 30)
    person2 = create_person("Jane Doe")  # Age defaults to 25

    print(f"Name: {person1.name}, Age: {person1.age}")
    print(f"Name: {person2.name}, Age: {person2.age}")

if __name__ == "__main__":
    main()
```
