## Exercise
Write a program that simulates a simple calendar, allowing the user to add, remove, and view events

## Reference Solution
```python
class Calendar:
    def __init__(self):
        self.events = {}

    def add_event(self, date, event):
        """
        Adds an event to the calendar.

        Args:
            date (str): The date of the event (in YYYY-MM-DD format).
            event (str): The description of the event.
        """
        if date in self.events:
            self.events[date].append(event)
        else:
            self.events[date] = [event]
        print(f"Event '{event}' added to {date}.")

    def remove_event(self, date, event):
        """
        Removes an event from the calendar.

        Args:
            date (str): The date of the event (in YYYY-MM-DD format).
            event (str): The description of the event.
        """
        if date in self.events:
            if event in self.events[date]:
                self.events[date].remove(event)
                print(f"Event '{event}' removed from {date}.")
                if not self.events[date]:
                    del self.events[date]
            else:
                print(f"No event '{event}' found on {date}.")
        else:
            print(f"No events found on {date}.")

    def view_events(self, date=None):
        """
        Displays all events on the calendar, or all events on a specific date.

        Args:
            date (str, optional): The date to view events for (in YYYY-MM-DD format). Defaults to None.
        """
        if date:
            if date in self.events:
                print(f"Events on {date}:")
                for event in self.events[date]:
                    print(event)
            else:
                print(f"No events found on {date}.")
        else:
            print("All events:")
            for date, events in self.events.items():
                print(f"{date}:")
                for event in events:
                    print(f"- {event}")


def main():
    calendar = Calendar()

    while True:
        print("\nCalendar Menu:")
        print("1. Add event")
        print("2. Remove event")
        print("3. View events")
        print("4. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            date = input("Enter the date (YYYY-MM-DD): ")
            event = input("Enter the event description: ")
            calendar.add_event(date, event)
        elif choice == "2":
            date = input("Enter the date (YYYY-MM-DD): ")
            event = input("Enter the event description: ")
            calendar.remove_event(date, event)
        elif choice == "3":
            date = input("Enter a date to view events (YYYY-MM-DD), or leave blank to view all events: ")
            if date:
                calendar.view_events(date)
            else:
                calendar.view_events()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
