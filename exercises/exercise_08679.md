## Exercise
Write a program that allows the user to add and remove items from a to-do list, and assigns a priority level to each item.

## Reference Solution
```python
class ToDoItem:
    def __init__(self, description, priority):
        self.description = description
        self.priority = priority

class ToDoList:
    def __init__(self):
        self.items = []

    def add_item(self):
        description = input("Enter the description of the item: ")
        priority = input("Enter the priority level (High, Medium, Low): ")
        item = ToDoItem(description, priority)
        self.items.append(item)
        print(f"Item '{description}' added to the list with priority '{priority}'.")

    def remove_item(self):
        if not self.items:
            print("The list is empty.")
            return
        print("Current items in the list:")
        for i, item in enumerate(self.items, start=1):
            print(f"{i}. {item.description} - {item.priority}")
        choice = input("Enter the number of the item to remove: ")
        try:
            choice = int(choice)
            if 1 <= choice <= len(self.items):
                item = self.items.pop(choice - 1)
                print(f"Item '{item.description}' removed from the list.")
            else:
                print("Invalid choice.")
        except ValueError:
            print("Invalid input.")

    def view_list(self):
        if not self.items:
            print("The list is empty.")
            return
        print("Current items in the list:")
        for i, item in enumerate(self.items, start=1):
            print(f"{i}. {item.description} - {item.priority}")

def main():
    todo_list = ToDoList()
    while True:
        print("\nOptions:")
        print("1. Add item to the list")
        print("2. Remove item from the list")
        print("3. View the list")
        print("4. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            todo_list.add_item()
        elif choice == "2":
            todo_list.remove_item()
        elif choice == "3":
            todo_list.view_list()
        elif choice == "4":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice.")

if __name__ == "__main__":
    main()
```
