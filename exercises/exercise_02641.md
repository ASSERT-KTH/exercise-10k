## Exercise
Create a dictionary with 5 key-value pairs and print out the keys and values

## Reference Solution
```python
def create_and_print_dict():
    # Create a dictionary with 5 key-value pairs
    my_dict = {
        "name": "John",
        "age": 30,
        "city": "New York",
        "country": "USA",
        "job": "Engineer"
    }

    # Print out the keys
    print("Keys:")
    for key in my_dict.keys():
        print(key)

    # Print out the values
    print("\nValues:")
    for value in my_dict.values():
        print(value)

    # Print out the key-value pairs
    print("\nKey-Value Pairs:")
    for key, value in my_dict.items():
        print(f"{key}: {value}")

# Example usage
if __name__ == "__main__":
    create_and_print_dict()
```
