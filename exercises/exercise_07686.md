```python
def is_sorted(numbers):
    """
    Checks if a list of numbers is sorted in ascending order.

    Args:
        numbers (list): A list of numbers.

    Returns:
        bool: True if the list is sorted, False otherwise.
    """
    return numbers == sorted(numbers)
```

```python
def is_sorted(numbers):
    """
    Checks if a list of numbers is sorted in ascending order.

    Args:
        numbers (list): A list of numbers.

    Returns:
        bool: True if the list is sorted, False otherwise.
    """
    return all(numbers[i] <= numbers[i + 1] for i in range(len(numbers) - 1))

# Example usage:
if __name__ == "__main__":
    # Test with a sorted list
    sorted_list = [1, 2, 3, 4, 5]
    print(is_sorted(sorted_list))  # Expected output: True

    # Test with an unsorted list
    unsorted_list = [5, 2, 3, 1, 4]
    print(is_sorted(unsorted_list))  # Expected output: False
```
