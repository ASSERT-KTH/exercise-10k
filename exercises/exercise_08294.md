Create a program that calculates the angle between two line segments.

import numpy as np

def calculate_angle(line1, line2):
    """
    Calculate the angle between two line segments using the Law of Cosines.

    Args:
        line1 (list): [x1, y1, x2, y2] coordinates of the first line segment
        line2 (list): [x1, y1, x2, y2] coordinates of the second line segment

    Returns:
        float: The angle between the two line segments in degrees
    """
    # Extract coordinates
    x1, y1, x2, y2 = line1
    x3, y3, x4, y4 = line2

    # Calculate vector lengths
    a = np.sqrt((x2 - x1)**2 + (y2 - y1)**2)
    b = np.sqrt((x4 - x3)**2 + (y4 - y3)**2)
    c = np.sqrt((x3 - x2)**2 + (y3 - y2)**2)

    # Calculate dot product
    dot_product = (x2 - x1) * (x4 - x3) + (y2 - y1) * (y4 - y3)

    # Calculate cosine of the angle
    cos_angle = dot_product / (a * b)

    # Calculate the angle in radians
    angle_rad = np.arccos(cos_angle)

    # Convert the angle to degrees
    angle_deg = np.degrees(angle_rad)

    return angle_deg

def main():
    # Example usage:
    line1 = [0, 0, 3, 4]  # Line segment from (0, 0) to (3, 4)
    line2 = [3, 4, 6, 8]  # Line segment from (3, 4) to (6, 8)

    angle = calculate_angle(line1, line2)
    print(f"The angle between the two line segments is: {angle:.2f} degrees")

if __name__ == "__main__":
    main()
