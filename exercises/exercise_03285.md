## Exercise
Write a program that finds the maximum number in a list of numbers using a for loop.

## Reference Solution
```python
def find_max_number(numbers):
    """
    This function finds the maximum number in a list of numbers.

    Parameters:
    numbers (list): A list of numbers.

    Returns:
    max_number (int): The maximum number in the list.
    """
    if len(numbers) == 0:
        return None

    max_number = numbers[0]
    for num in numbers:
        if num > max_number:
            max_number = num

    return max_number


def main():
    # Test the function
    numbers = [4, 2, 9, 6, 5, 1, 8, 3, 7]
    max_number = find_max_number(numbers)

    if max_number is not None:
        print("The maximum number in the list is:", max_number)
    else:
        print("The list is empty.")


if __name__ == "__main__":
    main()
```
