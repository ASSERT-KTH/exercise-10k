## Exercise
Write a program that uses a nested for loop to multiply two matrices.

## Reference Solution
```python
def multiply_matrices(matrix1, matrix2):
    """
    Multiply two matrices using a nested for loop.

    Args:
        matrix1 (list): The first matrix.
        matrix2 (list): The second matrix.

    Returns:
        list: The product of the two matrices.
    """
    # Get the dimensions of the matrices
    rows1 = len(matrix1)
    cols1 = len(matrix1[0])
    rows2 = len(matrix2)
    cols2 = len(matrix2[0])

    # Check if the matrices can be multiplied
    if cols1 != rows2:
        raise ValueError("Matrices cannot be multiplied")

    # Create a result matrix filled with zeros
    result = [[0 for _ in range(cols2)] for _ in range(rows1)]

    # Multiply the matrices using a nested for loop
    for i in range(rows1):
        for j in range(cols2):
            for k in range(cols1):
                result[i][j] += matrix1[i][k] * matrix2[k][j]

    return result


def print_matrix(matrix):
    """
    Print a matrix in a readable format.

    Args:
        matrix (list): The matrix to print.
    """
    for row in matrix:
        print(" ".join(str(x) for x in row))


def main():
    # Define two example matrices
    matrix1 = [[1, 2, 3], [4, 5, 6]]
    matrix2 = [[7, 8], [9, 10], [11, 12]]

    # Multiply the matrices
    result = multiply_matrices(matrix1, matrix2)

    # Print the result
    print("Matrix 1:")
    print_matrix(matrix1)
    print("\nMatrix 2:")
    print_matrix(matrix2)
    print("\nResult:")
    print_matrix(result)


if __name__ == "__main__":
    main()
```
