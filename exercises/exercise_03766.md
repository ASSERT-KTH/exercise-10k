## Exercise
Write a program to iterate over a dictionary's key-value pairs.

## Reference Solution
```python
def iterate_dictionary(dictionary):
    """
    Iterate over a dictionary's key-value pairs.

    Args:
        dictionary (dict): The dictionary to iterate over.

    Returns:
        None
    """
    # Method 1: Using items() method
    print("Method 1: Using items() method")
    for key, value in dictionary.items():
        print(f"Key: {key}, Value: {value}")

    # Method 2: Using enumerate() method
    print("\nMethod 2: Using enumerate() method")
    for index, (key, value) in enumerate(dictionary.items()):
        print(f"Index: {index}, Key: {key}, Value: {value}")

    # Method 3: Using zip() method with keys() and values() methods
    print("\nMethod 3: Using zip() method with keys() and values() methods")
    for key, value in zip(dictionary.keys(), dictionary.values()):
        print(f"Key: {key}, Value: {value}")


# Example usage
if __name__ == "__main__":
    example_dict = {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }

    iterate_dictionary(example_dict)
```
