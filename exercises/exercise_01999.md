Write a program to determine the day of the week based on the date using conditional statements

def find_day_of_week(day, month, year):
    """
    This function calculates the day of the week based on the date using Zeller's Congruence algorithm.

    Args:
        day (int): The day of the month (1-31).
        month (int): The month of the year (1-12).
        year (int): The year.

    Returns:
        str: The day of the week.
    """

    # Adjust month and year for January and February
    if month < 3:
        month += 12
        year -= 1

    # Calculate the day of the week using Zeller's Congruence algorithm
    q = day
    m = month
    k = year % 100
    j = year // 100

    f = q + ((13 * (m + 1)) // 5) + k + (k // 4) + (j // 4) - (2 * j)

    # Find the remainder of f divided by 7
    remainder = f % 7

    # Determine the day of the week based on the remainder
    if remainder == 0:
        return "Saturday"
    elif remainder == 1:
        return "Sunday"
    elif remainder == 2:
        return "Monday"
    elif remainder == 3:
        return "Tuesday"
    elif remainder == 4:
        return "Wednesday"
    elif remainder == 5:
        return "Thursday"
    else:
        return "Friday"


def main():
    day = int(input("Enter the day of the month (1-31): "))
    month = int(input("Enter the month of the year (1-12): "))
    year = int(input("Enter the year: "))

    day_of_week = find_day_of_week(day, month, year)

    print(f"The day of the week is: {day_of_week}")


if __name__ == "__main__":
    main()
