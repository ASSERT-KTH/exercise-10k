## Exercise
Create a program that uses a dictionary to store student grades. The dictionary keys should be student names and the values should be another dictionary with the subject as the key and the grade as the value. The program should allow the user to add a student, add a subject and grade for a student, and print out the grades for a specific student or all students.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        self.grades = {}

    def add_student(self, name):
        """Add a student to the grades dictionary"""
        if name not in self.grades:
            self.grades[name] = {}
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def add_grade(self, name, subject, grade):
        """Add a subject and grade for a student"""
        if name in self.grades:
            self.grades[name][subject] = grade
            print(f"Grade for '{subject}' added successfully for student '{name}'.")
        else:
            print(f"Student '{name}' does not exist.")

    def print_grades(self, name=None):
        """Print out the grades for a specific student or all students"""
        if name:
            if name in self.grades:
                print(f"Grades for student '{name}':")
                for subject, grade in self.grades[name].items():
                    print(f"{subject}: {grade}")
            else:
                print(f"Student '{name}' does not exist.")
        else:
            if self.grades:
                for name, grades in self.grades.items():
                    print(f"Grades for student '{name}':")
                    for subject, grade in grades.items():
                        print(f"{subject}: {grade}")
            else:
                print("No students or grades available.")

def main():
    grades = StudentGrades()
    while True:
        print("\nStudent Grades Program")
        print("1. Add student")
        print("2. Add grade for student")
        print("3. Print grades for student")
        print("4. Print all grades")
        print("5. Quit")
        choice = input("Choose an option: ")
        if choice == "1":
            name = input("Enter student name: ")
            grades.add_student(name)
        elif choice == "2":
            name = input("Enter student name: ")
            subject = input("Enter subject: ")
            grade = input("Enter grade: ")
            grades.add_grade(name, subject, grade)
        elif choice == "3":
            name = input("Enter student name: ")
            grades.print_grades(name)
        elif choice == "4":
            grades.print_grades()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
