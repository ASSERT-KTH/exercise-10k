def are_strings_equal(str1: str, str2: str) -> bool:
    """
    Checks if two strings are equal.

    Args:
        str1 (str): The first string to compare.
        str2 (str): The second string to compare.

    Returns:
        bool: True if the strings are equal, False otherwise.
    """
    return str1 == str2
