## Exercise
Write a program that declares and operates on sets.

## Reference Solution
```python
def create_set(elements):
    """Create a set from a list of elements."""
    return set(elements)

def union(set1, set2):
    """Return the union of two sets."""
    return set1.union(set2)

def intersection(set1, set2):
    """Return the intersection of two sets."""
    return set1.intersection(set2)

def difference(set1, set2):
    """Return the difference of two sets."""
    return set1.difference(set2)

def is_subset(set1, set2):
    """Check if set1 is a subset of set2."""
    return set1.issubset(set2)

def is_superset(set1, set2):
    """Check if set1 is a superset of set2."""
    return set1.issuperset(set2)

def main():
    # Create sets
    set1 = create_set([1, 2, 3, 4])
    set2 = create_set([3, 4, 5, 6])

    # Print sets
    print("Set 1:", set1)
    print("Set 2:", set2)

    # Perform set operations
    print("Union:", union(set1, set2))
    print("Intersection:", intersection(set1, set2))
    print("Difference:", difference(set1, set2))

    # Check subset and superset relationships
    print("Is Set 1 a subset of Set 2?", is_subset(set1, set2))
    print("Is Set 1 a superset of Set 2?", is_superset(set1, set2))

if __name__ == "__main__":
    main()
```
