## Exercise
Write a program to check if a specific key exists in a given dictionary.

## Reference Solution
```python
def check_key(dictionary, key):
    """
    Checks if a specific key exists in a given dictionary.

    Args:
        dictionary (dict): The dictionary to check.
        key (any): The key to look for.

    Returns:
        bool: True if the key exists, False otherwise.
    """
    return key in dictionary

def main():
    # Example usage:
    example_dict = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }

    key_to_check = "age"
    result = check_key(example_dict, key_to_check)

    print(f"Key '{key_to_check}' exists: {result}")

if __name__ == "__main__":
    main()
```
