## Exercise
Write a program to access the value of a specific key in a given dictionary.

## Reference Solution
```python
def access_dict_key(dictionary, key):
    """
    Access the value of a specific key in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key (any): The key to access.

    Returns:
        any: The value associated with the key if it exists, otherwise None.
    """
    try:
        return dictionary[key]
    except KeyError:
        print(f"Key '{key}' not found in the dictionary.")
        return None

def main():
    # Example dictionary
    example_dict = {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }

    # Access a key
    key_to_access = "age"
    value = access_dict_key(example_dict, key_to_access)

    if value is not None:
        print(f"The value of '{key_to_access}' is: {value}")

if __name__ == "__main__":
    main()
```
