## Exercise
Write a program that takes an array and a shift amount as input, and returns the array shifted cyclically by the specified amount.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int n = sizeof(arr) / sizeof(arr[0]);
    int k = 1;

    cout << "Original array: ";
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    k = k % n;
    int b[n];

    for (int i = 0; i < n; i++) {
        b[i] = arr[(n - k + i) % n];
    }

    for (int i = 0; i < n; i++) {
        arr[i] = b[i];
    }

    cout << "Shifted array: ";
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    return 0;
}
```
