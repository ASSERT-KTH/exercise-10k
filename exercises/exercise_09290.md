## Exercise
Create a program that takes two numbers and an operator as input and performs the specified arithmetic operation. The operators can be +, -, *, /, or %. The program should handle invalid operators and division by zero errors.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int num1, num2;
    char op;

    cout << "Enter the first number: ";
    cin >> num1;

    cout << "Enter the operator (+, -, *, /, %): ";
    cin >> op;

    cout << "Enter the second number: ";
    cin >> num2;

    switch (op) {
        case '+':
            cout << "Result: " << num1 + num2 << endl;
            break;
        case '-':
            cout << "Result: " << num1 - num2 << endl;
            break;
        case '*':
            cout << "Result: " << num1 * num2 << endl;
            break;
        case '/':
            if (num2 != 0) {
                cout << "Result: " << num1 / num2 << endl;
            } else {
                cout << "Error: Division by zero is not allowed." << endl;
            }
            break;
        case '%':
            if (num2 != 0) {
                cout << "Result: " << num1 % num2 << endl;
            } else {
                cout << "Error: Modulus operation by zero is not allowed." << endl;
            }
            break;
        default:
            cout << "Error: Invalid operator." << endl;
            break;
    }

    return 0;
}
```
