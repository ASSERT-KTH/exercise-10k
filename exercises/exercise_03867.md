## Exercise
Given a dictionary with names of planets as keys and their diameters as values, write a program to retrieve and print the diameter of a specific planet.

## Reference Solution
```python
def get_planet_diameter(planets, planet_name):
    """
    Retrieves the diameter of a specific planet from a dictionary.

    Args:
        planets (dict): A dictionary with planet names as keys and their diameters as values.
        planet_name (str): The name of the planet to retrieve the diameter for.

    Returns:
        float: The diameter of the specified planet, or None if the planet is not found.

    Raises:
        TypeError: If the input dictionary or planet name is not of the correct type.
    """
    if not isinstance(planets, dict):
        raise TypeError("Input must be a dictionary")
    if not isinstance(planet_name, str):
        raise TypeError("Planet name must be a string")

    # Check if the planet exists in the dictionary
    if planet_name in planets:
        # Return the diameter of the planet
        return planets[planet_name]
    else:
        # Return None if the planet is not found
        return None


def main():
    # Create a dictionary with planet names as keys and their diameters as values
    planets = {
        "Mercury": 4879.4,
        "Venus": 12103.6,
        "Earth": 12756.2,
        "Mars": 6794,
        "Jupiter": 142984,
        "Saturn": 116460,
        "Uranus": 50724,
        "Neptune": 49528
    }

    # Specify the planet to retrieve the diameter for
    planet_name = "Earth"

    # Retrieve and print the diameter of the planet
    diameter = get_planet_diameter(planets, planet_name)

    if diameter is not None:
        print(f"The diameter of {planet_name} is {diameter} kilometers.")
    else:
        print(f"{planet_name} not found in the dictionary.")


if __name__ == "__main__":
    main()
```
