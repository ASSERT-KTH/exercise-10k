## Exercise
Demonstrate the concept of implicit type conversion in programming

## Reference Solution
```python
# Implicit conversion from integer to float
a = 5
b = 2.0
result = a + b
print(type(result))  # Output: <class 'float'>
print(result)        # Output: 7.0

# Implicit conversion from string to integer (not possible, raises an error)
try:
    a = "5"
    b = 2
    result = a + b
    print(result)
except TypeError:
    print("Error: Cannot convert string to integer implicitly")

# Implicit conversion from boolean to integer
a = True
b = 2
result = a + b
print(type(result))  # Output: <class 'int'>
print(result)        # Output: 3
```

```python
public class Main {
    public static void main(String[] args) {
        // Implicit conversion from int to double
        int a = 5;
        double b = 2.0;
        double result = a + b;
        System.out.println(result);  // Output: 7.0

        // Implicit conversion from byte to int
        byte c = 10;
        int d = c;
        System.out.println(d);  // Output: 10

        // No implicit conversion from String to int (compile-time error)
        // String e = "5";
        // int f = e;  // Error: incompatible types: String cannot be converted to int

}
```
