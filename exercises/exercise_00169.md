## Exercise
Write a program to check the data type of a variable using built-in functions.

## Reference Solution
```python
# Function to check data type of a variable
def check_data_type(var):
    """Prints the data type of the input variable."""
    print(f"Data type of {var} is {type(var).__name__}")

# Main function
def main():
    # Example variables
    var_int = 10
    var_float = 10.5
    var_string = "Hello"
    var_list = [1, 2, 3]
    var_tuple = (1, 2, 3)
    var_dict = {"name": "John", "age": 30}

    # Check data type of each variable
    check_data_type(var_int)
    check_data_type(var_float)
    check_data_type(var_string)
    check_data_type(var_list)
    check_data_type(var_tuple)
    check_data_type(var_dict)

if __name__ == "__main__":
    main()
```
