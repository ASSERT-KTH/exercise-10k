## Exercise
Write a program that demonstrates the concept of variable scope, including local and global variables.

## Reference Solution
```python
# Global variable
global_variable = "I am a global variable"

def local_variable_example():
    """
    Demonstrates the use of local variables.
    """
    # Local variable
    local_variable = "I am a local variable"
    print("Inside local_variable_example function:")
    print("Local variable:", local_variable)
    print("Global variable:", global_variable)

def modify_global_variable():
    """
    Demonstrates how to modify a global variable inside a function.
    """
    global global_variable  # Declare global variable
    global_variable = "I am a modified global variable"
    print("Inside modify_global_variable function:")
    print("Global variable:", global_variable)

def main():
    print("Outside any function:")
    print("Global variable:", global_variable)
    
    local_variable_example()
    
    print("After calling local_variable_example function:")
    try:
        print("Local variable:", local_variable)
    except NameError:
        print("Local variable is not accessible here.")
    
    modify_global_variable()
    
    print("After calling modify_global_variable function:")
    print("Global variable:", global_variable)

if __name__ == "__main__":
    main()
```
