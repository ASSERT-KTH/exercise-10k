## Exercise
Write a program that calculates the square root of a given number using the Babylonian method.

## Reference Solution

def babylonian_sqrt(n, precision=0.00001):
    """
    Calculate the square root of a number using the Babylonian method.

    Args:
        n (float): The number to calculate the square root of.
        precision (float, optional): The desired precision of the result. Defaults to 0.00001.

    Returns:
        float: The approximate square root of the given number.
    """

    # Check if the input number is negative
    if n < 0:
        raise ValueError("Input number cannot be negative.")

    # Check if the input number is zero or one
    if n == 0 or n == 1:
        return n

    # Initial guess for the square root
    x = n / 2.0

    # Iterate until the desired precision is reached
    while True:
        # Calculate the next estimate using the Babylonian method formula
        y = (x + n / x) / 2.0

        # Check if the desired precision is reached
        if abs(y - x) < precision:
            return y

        # Update the current estimate
        x = y


# Example usage
if __name__ == "__main__":
    num = 9
    sqrt = babylonian_sqrt(num)
    print(f"The square root of {num} is approximately {sqrt}")
