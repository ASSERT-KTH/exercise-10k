## Exercise
Write a program to calculate the coin change for a given amount using conditional statements

## Reference Solution
```python
def calculate_coin_change(amount):
    # Define the available coin denominations
    coin_denominations = [25, 10, 5, 1]

    # Initialize the coin change dictionary
    coin_change = {
        "quarters": 0,
        "dimes": 0,
        "nickels": 0,
        "pennies": 0
    }

    # Calculate the coin change
    for i, denomination in enumerate(coin_denominations):
        if denomination == 25:
            coin_change["quarters"] = amount // denomination
            amount %= denomination
        elif denomination == 10:
            coin_change["dimes"] = amount // denomination
            amount %= denomination
        elif denomination == 5:
            coin_change["nickels"] = amount // denomination
            amount %= denomination
        elif denomination == 1:
            coin_change["pennies"] = amount // denomination
            amount %= denomination

    return coin_change


def main():
    # Get the amount from the user
    amount = int(input("Enter the amount in cents: "))

    # Calculate the coin change
    coin_change = calculate_coin_change(amount)

    # Print the coin change
    print("Coin Change:")
    print(f"Quarters: {coin_change['quarters']}")
    print(f"Dimes: {coin_change['dimes']}")
    print(f"Nickels: {coin_change['nickels']}")
    print(f"Pennies: {coin_change['pennies']}")


if __name__ == "__main__":
    main()
```
