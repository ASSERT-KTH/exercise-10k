## Exercise
Write a program that allows two players to play Rock, Paper, Scissors, Lizard, Spock. The program should determine the winner based on the game's rules and output the result.

## Reference Solution
```python
import random

# Define the game rules
game_rules = {
    "rock": {"scissors": "crushes", "lizard": "crushes"},
    "paper": {"rock": "covers", "spock": "disproves"},
    "scissors": {"paper": "cuts", "lizard": "decapitates"},
    "lizard": {"spock": "poisons", "paper": "eats"},
    "spock": {"scissors": "smashes", "rock": "vaporizes"}
}

def get_computer_choice():
    """Get the computer's random choice."""
    choices = list(game_rules.keys())
    return random.choice(choices)

def determine_winner(player1, player2):
    """Determine the winner based on the game rules."""
    if player1 == player2:
        return "It's a tie!"
    elif player2 in game_rules[player1]:
        return f"{player1.capitalize()} {game_rules[player1][player2]} {player2}. Player 1 wins!"
    else:
        return f"{player2.capitalize()} {game_rules[player2][player1]} {player1}. Player 2 wins!"

def play_game():
    """Play the Rock, Paper, Scissors, Lizard, Spock game."""
    print("Welcome to Rock, Paper, Scissors, Lizard, Spock!")
    player1 = input("Enter your choice (rock, paper, scissors, lizard, spock): ").lower()
    while player1 not in game_rules:
        player1 = input("Invalid choice. Please enter rock, paper, scissors, lizard, or spock: ").lower()
    
    player2 = get_computer_choice()
    print(f"\nYou chose: {player1}")
    print(f"Computer chose: {player2}\n")
    
    print(determine_winner(player1, player2))

if __name__ == "__main__":
    play_game()
```
