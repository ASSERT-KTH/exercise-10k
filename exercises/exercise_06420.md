## Exercise
Write a program that simulates a chat room and allows users to send and receive messages. Use if/else statements to determine the message.

## Reference Solution
```python
class ChatRoom:
    def __init__(self):
        self.users = {}
        self.messages = {}

    def add_user(self, name):
        """Add a user to the chat room."""
        if name not in self.users:
            self.users[name] = []
            self.messages[name] = []
            print(f"User {name} added to the chat room.")
        else:
            print(f"User {name} already exists in the chat room.")

    def send_message(self, sender, recipient, message):
        """Send a message from one user to another."""
        if sender in self.users and recipient in self.users:
            self.messages[recipient].append((sender, message))
            print(f"Message sent from {sender} to {recipient}.")
        else:
            print("One or both users do not exist in the chat room.")

    def receive_message(self, recipient):
        """Receive messages for a user."""
        if recipient in self.users:
            if self.messages[recipient]:
                print(f"Messages for {recipient}:")
                for sender, message in self.messages[recipient]:
                    print(f"{sender}: {message}")
                self.messages[recipient] = []  # clear messages
            else:
                print(f"No messages for {recipient}.")
        else:
            print(f"User {recipient} does not exist in the chat room.")

    def display_users(self):
        """Display all users in the chat room."""
        if self.users:
            print("Users in the chat room:")
            for user in self.users:
                print(user)
        else:
            print("No users in the chat room.")


def main():
    chat_room = ChatRoom()

    while True:
        print("\n1. Add user")
        print("2. Send message")
        print("3. Receive message")
        print("4. Display users")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter user name: ")
            chat_room.add_user(name)
        elif choice == "2":
            sender = input("Enter sender's name: ")
            recipient = input("Enter recipient's name: ")
            message = input("Enter message: ")
            chat_room.send_message(sender, recipient, message)
        elif choice == "3":
            recipient = input("Enter recipient's name: ")
            chat_room.receive_message(recipient)
        elif choice == "4":
            chat_room.display_users()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
