## Exercise
Create a game that allows two players to play Rock, Paper, Scissors.

## Reference Solution
```python
import random

def get_computer_choice():
    """Get a random choice from the computer."""
    choices = ["rock", "paper", "scissors"]
    return random.choice(choices)

def get_player_choice(player_name):
    """Get the choice from a player."""
    while True:
        choice = input(f"{player_name}, enter your choice (rock, paper, or scissors): ").lower()
        if choice in ["rock", "paper", "scissors"]:
            return choice
        else:
            print("Invalid choice. Please try again.")

def determine_winner(player1_choice, player2_choice):
    """Determine the winner based on the game rules."""
    if player1_choice == player2_choice:
        return "It's a tie!"
    elif (player1_choice == "rock" and player2_choice == "scissors") or \
         (player1_choice == "scissors" and player2_choice == "paper") or \
         (player1_choice == "paper" and player2_choice == "rock"):
        return "Player 1 wins!"
    else:
        return "Player 2 wins!"

def play_game():
    """Play a game of Rock, Paper, Scissors."""
    player1_name = input("Enter Player 1's name: ")
    player2_name = input("Enter Player 2's name: ")

    player1_choice = get_player_choice(player1_name)
    player2_choice = get_player_choice(player2_name)

    print(f"\n{player1_name} chose {player1_choice}.")
    print(f"{player2_name} chose {player2_choice}.\n")

    result = determine_winner(player1_choice, player2_choice)
    print(result)

play_game()
```
