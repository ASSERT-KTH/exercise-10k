## Exercise
Write a program that demonstrates the use of list operations in programming.

## Reference Solution
```python
# List Operations in Python

def main():
    # Create a list
    numbers = [1, 2, 3, 4, 5]
    print("Original List:", numbers)

    # Indexing a list
    print("\nIndexing:")
    print("First Element:", numbers[0])
    print("Last Element:", numbers[-1])

    # Slicing a list
    print("\nSlicing:")
    print("First Two Elements:", numbers[:2])
    print("Last Two Elements:", numbers[-2:])

    # Appending an element
    numbers.append(6)
    print("\nAfter Appending 6:", numbers)

    # Inserting an element
    numbers.insert(2, 2.5)
    print("\nAfter Inserting 2.5 at Index 2:", numbers)

    # Removing an element
    numbers.remove(2)
    print("\nAfter Removing 2:", numbers)

    # Sorting a list
    numbers.sort()
    print("\nAfter Sorting:", numbers)

    # Reversing a list
    numbers.reverse()
    print("\nAfter Reversing:", numbers)

    # List methods
    print("\nList Methods:")
    print("Length:", len(numbers))
    print("Minimum:", min(numbers))
    print("Maximum:", max(numbers))

if __name__ == "__main__":
    main()
```
