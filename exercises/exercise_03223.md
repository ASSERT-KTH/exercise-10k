## Exercise
Write a program that calculates the minimum number of coins required to make change for a given amount using coins of denominations 1, 5, and 10.

## Reference Solution
```python
function main() {
  let change = 20;
  let coins = [1, 5, 10];
  let minCoins = getMinCoins(coins, change);
  console.log(`Minimum number of coins required to make change for ${change} is ${minCoins}`);
}

function getMinCoins(coins, change) {
  let minCoins = [];
  let i = coins.length - 1;

  while (change > 0) {
    if (coins[i] <= change) {
      minCoins.push(coins[i]);
      change = change - coins[i];
    } else if (i <= 0) {
      break;
    } else {
      i--;
    }
  }

  return minCoins.length;
}

main();
```
