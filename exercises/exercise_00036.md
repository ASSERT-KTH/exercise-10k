## Exercise
Write a function that takes a 2D list as input and returns True if it's a symmetric matrix, False otherwise.

## Reference Solution

def is_symmetric(matrix):
    """
    Checks if a given 2D list represents a symmetric matrix.

    Args:
        matrix (list): A 2D list representing a matrix.

    Returns:
        bool: True if the matrix is symmetric, False otherwise.
    """
    # Check if the matrix is square
    if len(matrix) != len(matrix[0]):
        return False

    # Check if the matrix is equal to its transpose
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            if matrix[i][j] != matrix[j][i]:
                return False

    return True

# Example usage
if __name__ == "__main__":
    # Define a symmetric matrix
    symmetric_matrix = [[1, 2, 3], [2, 4, 5], [3, 5, 6]]
    print(is_symmetric(symmetric_matrix))  # Output: True

    # Define a non-symmetric matrix
    non_symmetric_matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    print(is_symmetric(non_symmetric_matrix))  # Output: False

    # Define a non-square matrix
    non_square_matrix = [[1, 2, 3], [4, 5, 6]]
    print(is_symmetric(non_square_matrix))  # Output: False
