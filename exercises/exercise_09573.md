## Exercise
Create a program that takes in a string input from the user and converts it into different data types (integer, float, boolean) based on the user's choice. The program should handle invalid conversions and provide appropriate error messages.

## Reference Solution
```python
# Importing Libraries
import math
from ast import literal_eval

# Defining Functions
def isInt(s):
    try:
        int(s)
        return True
    except ValueError:
        return False

def isFloat(s):
    try:
        float(s)
        return True
    except ValueError:
        return False

def isBool(s):
    return s.lower() == "true" or s.lower() == "false"

def convertToInt(s):
    try:
        return int(s)
    except ValueError:
        raise Exception("String cannot be converted to integer")

def convertToFloat(s):
    try:
        return float(s)
    except ValueError:
        raise Exception("String cannot be converted to float")

def convertToBool(s):
    if s.lower() == "true":
        return True
    elif s.lower() == "false":
        return False
    else:
        raise Exception("String cannot be converted to boolean")

def convertData(s, ch):
    if ch == "int":
        if isInt(s):
            return convertToInt(s)
        else:
            raise Exception("String cannot be converted to integer")
    elif ch == "float":
        if isFloat(s):
            return convertToFloat(s)
        else:
            raise Exception("String cannot be converted to float")
    elif ch == "bool":
        if isBool(s):
            return convertToBool(s)
        else:
            raise Exception("String cannot be converted to boolean")
    else:
        raise Exception("Invalid choice")

# Main Program
print("Data Type Converter")
print("---------------------")
string = input("Enter a string: ")
choice = input("Enter the data type you want to convert to (int/float/bool): ")
choice = choice.lower()

try:
    result = convertData(string, choice)
    print(f"Converted value: {result}")
except Exception as e:
    print(f"Error: {e}")
```
