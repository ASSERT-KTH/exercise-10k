## Exercise
Create a dictionary with 5 key-value pairs and print the values for specific keys

## Reference Solution
```python
def main():
    # Create a dictionary with 5 key-value pairs
    example_dict = {
        "name": "John Doe",
        "age": 30,
        "city": "New York",
        "country": "USA",
        " occupation": "Software Engineer"
    }

    # Define the keys for which we want to print the values
    keys_to_print = ["name", "age", "city"]

    # Print the values for the specified keys
    for key in keys_to_print:
        if key in example_dict:
            print(f"The value for key '{key}' is: {example_dict[key]}")
        else:
            print(f"Key '{key}' not found in the dictionary.")

if __name__ == "__main__":
    main()
```
