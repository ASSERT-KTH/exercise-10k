## Exercise
Write a program that calculates the mean, median, and mode of a list of numbers.

## Reference Solution
```python
import statistics
from collections import Counter

def calculate_mean(num_list):
    """Calculate the mean of a list of numbers."""
    return sum(num_list) / len(num_list)

def calculate_median(num_list):
    """Calculate the median of a list of numbers."""
    return statistics.median(num_list)

def calculate_mode(num_list):
    """Calculate the mode of a list of numbers."""
    counts = Counter(num_list)
    max_count = max(counts.values())
    modes = [num for num, count in counts.items() if count == max_count]
    return modes

def main():
    # Example usage
    num_list = [1, 2, 3, 4, 4, 5, 5, 5]
    mean = calculate_mean(num_list)
    median = calculate_median(num_list)
    mode = calculate_mode(num_list)
    
    print("Mean:", mean)
    print("Median:", median)
    print("Mode:", mode)

if __name__ == "__main__":
    main()
```
