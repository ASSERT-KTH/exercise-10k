## Exercise
Create a function that takes a list of nodes as input and returns a binary tree

## Reference Solution
```python
class TreeNode:
    """Represents a node in a binary tree."""
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None

def create_binary_tree(nodes):
    """
    Creates a binary tree from a list of nodes.

    Args:
    nodes (list): A list of node values in level order traversal.

    Returns:
    TreeNode: The root node of the created binary tree.
    """
    if not nodes:
        return None

    root = TreeNode(nodes[0])
    queue = [root]
    i = 1

    while i < len(nodes):
        current_node = queue.pop(0)
        
        # Create left child
        if i < len(nodes):
            current_node.left = TreeNode(nodes[i])
            queue.append(current_node.left)
            i += 1
        
        # Create right child
        if i < len(nodes):
            current_node.right = TreeNode(nodes[i])
            queue.append(current_node.right)
            i += 1

    return root

# Example usage:
def print_tree(node, level=0):
    """Prints the binary tree in a pretty format."""
    if node is not None:
        print_tree(node.right, level + 1)
        print(' ' * 4 * level + '->', node.value)
        print_tree(node.left, level + 1)

# Create a binary tree from a list of nodes
nodes = [1, 2, 3, 4, 5, 6, 7]
root = create_binary_tree(nodes)

# Print the binary tree
print_tree(root)
```
