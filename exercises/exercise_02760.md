## Exercise
Write a program that calculates an employee's gross pay given their hourly wage and number of hours worked.

## Reference Solution
```python
def calculate_gross_pay(hourly_wage, hours_worked):
    """
    Calculates an employee's gross pay.

    Args:
        hourly_wage (float): The employee's hourly wage.
        hours_worked (float): The number of hours worked.

    Returns:
        float: The employee's gross pay.
    """
    if hourly_wage < 0 or hours_worked < 0:
        raise ValueError("Hourly wage and hours worked must be non-negative")

    gross_pay = hourly_wage * hours_worked
    return gross_pay

def main():
    # Get user input
    hourly_wage = float(input("Enter hourly wage: $"))
    hours_worked = float(input("Enter number of hours worked: "))

    # Calculate gross pay
    gross_pay = calculate_gross_pay(hourly_wage, hours_worked)

    # Print result
    print(f"Gross pay: ${gross_pay:.2f}")

if __name__ == "__main__":
    main()
```
