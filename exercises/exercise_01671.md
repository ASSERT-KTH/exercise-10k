## Exercise
Develop a program that takes the day of the month and the month as input and determines the day of the week (Monday to Sunday) using if/else statements

def find_day_of_week(day, month, year):
    """
    This function calculates the day of the week using Zeller's Congruence algorithm.

    Args:
        day (int): Day of the month (1-31)
        month (int): Month of the year (1-12)
        year (int): Year

    Returns:
        str: The day of the week (Monday to Sunday)
    """

    # Adjust month and year for January and February
    if month < 3:
        month += 12
        year -= 1

    # Calculate the day of the week
    q = day
    m = month
    k = year % 100
    j = year // 100

    f = q + ((13 * (m + 1)) // 5) + k + (k // 4) + (j // 4) - (2 * j)
    result = f % 7

    # Map the result to the corresponding day of the week
    if result == 0:
        return "Saturday"
    elif result == 1:
        return "Sunday"
    elif result == 2:
        return "Monday"
    elif result == 3:
        return "Tuesday"
    elif result == 4:
        return "Wednesday"
    elif result == 5:
        return "Thursday"
    else:
        return "Friday"


def main():
    day = int(input("Enter the day of the month (1-31): "))
    month = int(input("Enter the month (1-12): "))
    year = int(input("Enter the year: "))

    day_of_week = find_day_of_week(day, month, year)

    print(f"The day of the week is: {day_of_week}")


if __name__ == "__main__":
    main()
