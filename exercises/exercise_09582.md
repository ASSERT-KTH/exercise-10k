## Exercise
Write a program to find the maximum contiguous subarray of an array of integers. The subarray must have at least one element. If the array is empty, return 0.

## Reference Solution
```python
def max_contiguous_subarray(arr):
    """
    Find the maximum contiguous subarray of an array of integers.

    Args:
        arr (list): A list of integers.

    Returns:
        int: The maximum sum of a contiguous subarray. If the array is empty, returns 0.
    """
    if not arr:
        return 0

    max_sum = float('-inf')
    current_sum = 0

    for num in arr:
        current_sum = max(num, current_sum + num)
        max_sum = max(max_sum, current_sum)

    return max_sum

# Example usage
def main():
    arr1 = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
    print(max_contiguous_subarray(arr1))  # Output: 6

    arr2 = []
    print(max_contiguous_subarray(arr2))  # Output: 0

    arr3 = [-1, -2, -3]
    print(max_contiguous_subarray(arr3))  # Output: -1

if __name__ == "__main__":
    main()
```
