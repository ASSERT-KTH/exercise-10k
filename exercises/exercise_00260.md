## Exercise
Write a function that takes multiple lists as input, and returns a new list that contains only the elements that are in the first list but not in any of the other lists.

def filter_elements(*lists):
    """
    This function takes multiple lists as input, and returns a new list that contains only the elements 
    that are in the first list but not in any of the other lists.

    Args:
        *lists (list): A variable number of lists

    Returns:
        list: A new list with elements unique to the first input list
    """

    # Check if there are at least two lists provided
    if len(lists) < 2:
        raise ValueError("At least two lists are required")

    # Initialize an empty set to store the elements from the first list
    # We use a set here because it allows for efficient lookups
    first_list_elements = set()

    # Iterate over each element in the first list
    for element in lists[0]:
        # Add each element to the set
        first_list_elements.add(element)

    # Initialize an empty set to store the elements from the other lists
    other_list_elements = set()

    # Iterate over each of the other lists
    for lst in lists[1:]:
        # Iterate over each element in the current list
        for element in lst:
            # Add each element to the set
            other_list_elements.add(element)

    # Use a list comprehension to create a new list that contains only the elements 
    # that are in the first list but not in any of the other lists
    # This is done by checking if each element in the first list is not in the set of elements from the other lists
    result = [element for element in lists[0] if element not in other_list_elements]

    # Return the resulting list
    return result
