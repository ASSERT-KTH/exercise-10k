## Exercise
Write a program that takes two arrays as input, and returns a new array containing all the unique elements of both arrays.

## Reference Solution
```python
#include <iostream>
#include <vector>

std::vector<int> mergeUniqueElements(std::vector<int> array1, std::vector<int> array2) {
    std::vector<int> mergedArray;

    for (int element : array1) {
        bool found = false;
        for (int existingElement : mergedArray) {
            if (element == existingElement) {
                found = true;
                break;
            }
        }
        if (!found) {
            mergedArray.push_back(element);
        }
    }

    for (int element : array2) {
        bool found = false;
        for (int existingElement : mergedArray) {
            if (element == existingElement) {
                found = true;
                break;
            }
        }
        if (!found) {
            mergedArray.push_back(element);
        }
    }

    return mergedArray;
}

int main() {
    std::vector<int> array1 = {1, 2, 3, 4, 5};
    std::vector<int> array2 = {4, 5, 6, 7, 8};

    std::vector<int> result = mergeUniqueElements(array1, array2);

    std::cout << "Merged array: ";
    for (int element : result) {
        std::cout << element << " ";
    }
    std::cout << std::endl;

    return 0;
}
```
