## Exercise
Write a program that generates the Fibonacci sequence up to a given number.

## Reference Solution
```python
def generate_fibonacci(n):
    """
    Generate Fibonacci sequence up to the nth number.

    Args:
        n (int): The number of terms in the sequence.

    Returns:
        list: A list of Fibonacci numbers up to the nth term.
    """
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]

    fib_sequence = [0, 1]
    while len(fib_sequence) < n:
        fib_sequence.append(fib_sequence[-1] + fib_sequence[-2])

    return fib_sequence


def main():
    # Example usage
    n = int(input("Enter the number of terms: "))
    fib_sequence = generate_fibonacci(n)
    print("Fibonacci sequence up to {} terms:".format(n))
    print(fib_sequence)


if __name__ == "__main__":
    main()
```
