## Exercise
Write a program that validates a credit card number using if/else statements.

def validate_credit_card_number(card_number):
    """
    Validates a credit card number using the Luhn algorithm.

    Args:
        card_number (str): The credit card number to validate.

    Returns:
        bool: True if the credit card number is valid, False otherwise.
    """
    # Convert the card number to a list of integers
    card_number = [int(digit) for digit in card_number.replace(" ", "")]

    # Check if the card number has a valid length
    if len(card_number) < 13 or len(card_number) > 16:
        return False

    # Reverse the card number
    card_number = card_number[::-1]

    # Double every second digit
    for i in range(1, len(card_number), 2):
        card_number[i] *= 2
        # Subtract 9 from numbers over 9
        if card_number[i] > 9:
            card_number[i] -= 9

    # Check if the sum of the digits is divisible by 10
    if sum(card_number) % 10 == 0:
        return True
    else:
        return False

def main():
    # Test the function
    card_number = input("Enter a credit card number: ")
    if validate_credit_card_number(card_number):
        print("The credit card number is valid.")
    else:
        print("The credit card number is not valid.")

if __name__ == "__main__":
    main()
