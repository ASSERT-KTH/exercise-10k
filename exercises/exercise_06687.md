## Exercise
Create a program that takes two numbers as input and returns the remainder of their division.

## Reference Solution
```python
def get_remainder(dividend, divisor):
    """
    Returns the remainder of the division of dividend by divisor.
    
    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.
    
    Returns:
        int: The remainder of the division.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero.")
    return dividend % divisor


def main():
    # Get user input
    try:
        dividend = int(input("Enter the dividend: "))
        divisor = int(input("Enter the divisor: "))
        
        # Calculate and display the remainder
        remainder = get_remainder(dividend, divisor)
        print(f"The remainder of {dividend} divided by {divisor} is {remainder}.")
        
    except ZeroDivisionError as e:
        print(e)
    except ValueError:
        print("Invalid input. Please enter a valid number.")


if __name__ == "__main__":
    main()
```
