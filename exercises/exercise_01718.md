## Exercise
Create a program that uses a while loop to simulate a game of Baseball, where the user can pitch and swing until they choose to quit.

## Reference Solution
```python
import random

def pitch():
    """Simulate a pitch"""
    pitch_type = random.choice(["Strike", "Ball", "Foul Ball"])
    return pitch_type

def swing():
    """Simulate a swing"""
    swing_result = random.choice(["Hit", "Miss", "Foul Ball"])
    return swing_result

def game():
    """Simulate a game of Baseball"""
    strikes = 0
    balls = 0
    outs = 0
    while True:
        print("\nOptions:")
        print("1. Pitch")
        print("2. Swing")
        print("3. Quit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            pitch_result = pitch()
            print(f"\nYou pitched a {pitch_result}.")
            if pitch_result == "Strike":
                strikes += 1
                print(f"You have {strikes} strike(s).")
                if strikes == 3:
                    outs += 1
                    strikes = 0
                    print(f"You got a strikeout! You have {outs} out(s).")
            elif pitch_result == "Ball":
                balls += 1
                print(f"You have {balls} ball(s).")
                if balls == 4:
                    balls = 0
                    print("The batter walked!")
            elif pitch_result == "Foul Ball":
                print("The batter hit a foul ball. The pitch counts as a strike.")
                strikes += 1
                print(f"You have {strikes} strike(s).")
                if strikes == 3:
                    outs += 1
                    strikes = 0
                    print(f"You got a strikeout! You have {outs} out(s).")
        elif choice == "2":
            swing_result = swing()
            print(f"\nYou swung and got a {swing_result}.")
            if swing_result == "Hit":
                print("You hit the ball! You got a hit!")
                strikes = 0
                balls = 0
            elif swing_result == "Miss":
                strikes += 1
                print(f"You missed the ball. You have {strikes} strike(s).")
                if strikes == 3:
                    outs += 1
                    strikes = 0
                    print(f"You got a strikeout! You have {outs} out(s).")
            elif swing_result == "Foul Ball":
                print("You hit a foul ball. The swing counts as a strike.")
                strikes += 1
                print(f"You have {strikes} strike(s).")
                if strikes == 3:
                    outs += 1
                    strikes = 0
                    print(f"You got a strikeout! You have {outs} out(s).")
        elif choice == "3":
            print("\nGame over!")
            break
        else:
            print("Invalid choice. Please try again.")

        if outs == 3:
            print("\nGame over! Three outs have been reached.")
            break

if __name__ == "__main__":
    game()
```
