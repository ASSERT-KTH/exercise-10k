## Exercise
Write a program that thinks of a number between 1 and 100 and asks the user to guess it

## Reference Solution
```python
import random

def get_computer_number():
    """Generate a random number between 1 and 100"""
    return random.randint(1, 100)

def get_user_guess():
    """Get the user's guess"""
    while True:
        try:
            guess = int(input("Guess a number between 1 and 100: "))
            if 1 <= guess <= 100:
                return guess
            else:
                print("Please enter a number between 1 and 100.")
        except ValueError:
            print("Invalid input. Please enter a whole number.")

def check_guess(computer_number, user_guess):
    """Check if the user's guess is correct"""
    if user_guess < computer_number:
        print("Too low! Try again.")
    elif user_guess > computer_number:
        print("Too high! Try again.")
    else:
        print("Congratulations! You guessed the number correctly.")

def play_game():
    """Play the number guessing game"""
    computer_number = get_computer_number()
    attempts = 0
    while True:
        user_guess = get_user_guess()
        attempts += 1
        check_guess(computer_number, user_guess)
        if user_guess == computer_number:
            print(f"It took you {attempts} attempts to guess the number.")
            break

if __name__ == "__main__":
    play_game()
```
