## Exercise
Create a program that simulates a voting system. The program should ask for the candidates, the votes, and then print out the winner.

## Reference Solution
```python
def voting_system():
    # Get the candidates
    candidates = input("Enter the candidates separated by commas: ")
    candidates = [candidate.strip() for candidate in candidates.split(",")]

    # Initialize the votes dictionary
    votes = {candidate: 0 for candidate in candidates}

    # Get the votes
    num_voters = int(input("Enter the number of voters: "))
    for i in range(num_voters):
        print(f"Voter {i+1}'s turn:")
        for candidate in candidates:
            print(f"{candidates.index(candidate)+1}. {candidate}")
        choice = int(input("Enter the number of your chosen candidate: ")) - 1
        if 0 <= choice < len(candidates):
            votes[candidates[choice]] += 1
        else:
            print("Invalid choice. Your vote will not be counted.")

    # Print out the results
    print("\nResults:")
    for candidate, vote in votes.items():
        print(f"{candidate}: {vote} votes")

    # Determine the winner
    max_votes = max(votes.values())
    winners = [candidate for candidate, vote in votes.items() if vote == max_votes]

    if len(winners) == 1:
        print(f"\nThe winner is {winners[0]} with {max_votes} votes!")
    else:
        print(f"\nIt's a tie between {', '.join(winners)} with {max_votes} votes each.")

if __name__ == "__main__":
    voting_system()
```
