## Exercise
Create a program that uses a while loop to play a text adventure game with the user. The program should present the user with a series of choices and respond accordingly.

## Reference Solution
```python
def start_game():
    print("Welcome to the text adventure game!")
    print("You are standing at the entrance of a dark cave.")
    print("There is a path to the left and a path to the right.")
    
    while True:
        choice = input("Which path do you want to take? Type 'left' or 'right': ")
        
        if choice.lower() == "left":
            print("You have taken the path to the left.")
            print("You come across a river. Do you want to swim across or follow the river?")
            
            river_choice = input("Type 'swim' or 'follow': ")
            
            if river_choice.lower() == "swim":
                print("You have swum across the river.")
                print("On the other side, you find a treasure chest.")
                print("Congratulations, you have won the game!")
                break
            elif river_choice.lower() == "follow":
                print("You have followed the river.")
                print("The river leads you to a waterfall.")
                print("Behind the waterfall, you find a hidden cave.")
                print("Unfortunately, the cave is empty.")
                print("Game over.")
                break
            else:
                print("Invalid choice. Please try again.")
                
        elif choice.lower() == "right":
            print("You have taken the path to the right.")
            print("You come across a dense forest. Do you want to enter the forest or go back to the entrance?")
            
            forest_choice = input("Type 'enter' or 'back': ")
            
            if forest_choice.lower() == "enter":
                print("You have entered the forest.")
                print("In the forest, you find a group of friendly animals.")
                print("They offer to guide you through the forest.")
                print("After a while, they lead you to a beautiful meadow.")
                print("Congratulations, you have won the game!")
                break
            elif forest_choice.lower() == "back":
                print("You have gone back to the entrance.")
                print("You are now back at the start of the game.")
            else:
                print("Invalid choice. Please try again.")
                
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    start_game()
```
