## Exercise
Create a program that allows users to reserve flights. The program should take the flight number, departure date, and return date as input, and print the reservation details.

## Reference Solution
```python
from datetime import datetime

class Flight:
    def __init__(self, flight_number, departure_date, return_date):
        self.flight_number = flight_number
        self.departure_date = datetime.strptime(departure_date, '%Y-%m-%d')
        self.return_date = datetime.strptime(return_date, '%Y-%m-%d')

    def __str__(self):
        return f"Flight Number: {self.flight_number}\nDeparture Date: {self.departure_date.strftime('%Y-%m-%d')}\nReturn Date: {self.return_date.strftime('%Y-%m-%d')}"

def reserve_flight(flight_number, departure_date, return_date):
    flight = Flight(flight_number, departure_date, return_date)
    print("Flight Reservation Details:")
    print(flight)

def main():
    print("Flight Reservation Program")
    print("---------------------------")
    flight_number = input("Enter flight number: ")
    departure_date = input("Enter departure date (YYYY-MM-DD): ")
    return_date = input("Enter return date (YYYY-MM-DD): ")
    reserve_flight(flight_number, departure_date, return_date)

if __name__ == "__main__":
    main()
```
