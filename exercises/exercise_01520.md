## Exercise
Write a program that finds all palindromic substrings in a given string.

## Reference Solution
```python
def is_palindrome(s):
    """Check if a string is a palindrome."""
    return s == s[::-1]


def find_palindromic_substrings(s):
    """Find all palindromic substrings in a given string."""
    palindromic_substrings = set()
    for i in range(len(s)):
        for j in range(i + 1, len(s) + 1):
            substring = s[i:j]
            if is_palindrome(substring):
                palindromic_substrings.add(substring)
    return palindromic_substrings


def main():
    # Example usage
    s = "abba"
    palindromic_substrings = find_palindromic_substrings(s)
    print(f"Palindromic substrings in '{s}': {palindromic_substrings}")


if __name__ == "__main__":
    main()
```
