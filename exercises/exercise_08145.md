## Exercise
Write a program that implements a simple encryption algorithm using if/else statements

## Reference Solution
```python
def caesar_encrypt(plaintext, shift):
    """
    Encrypts the given plaintext using a Caesar cipher with the specified shift.

    Args:
        plaintext (str): The text to be encrypted.
        shift (int): The number of positions to shift each letter.

    Returns:
        str: The encrypted ciphertext.
    """
    ciphertext = ""
    for char in plaintext:
        if char.isalpha():  # Check if the character is a letter
            ascii_offset = 97 if char.islower() else 65  # Get the ASCII offset for lowercase or uppercase letters
            encrypted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            ciphertext += encrypted_char
        else:
            ciphertext += char  # If the character is not a letter, leave it unchanged
    return ciphertext

def caesar_decrypt(ciphertext, shift):
    """
    Decrypts the given ciphertext using a Caesar cipher with the specified shift.

    Args:
        ciphertext (str): The text to be decrypted.
        shift (int): The number of positions to shift each letter.

    Returns:
        str: The decrypted plaintext.
    """
    return caesar_encrypt(ciphertext, -shift)  # Decrypting is equivalent to encrypting with a negative shift

def main():
    plaintext = "Hello, World!"
    shift = 3
    print(f"Plaintext: {plaintext}")
    print(f"Shift: {shift}")

    ciphertext = caesar_encrypt(plaintext, shift)
    print(f"Ciphertext: {ciphertext}")

    decrypted_text = caesar_decrypt(ciphertext, shift)
    print(f"Decrypted Text: {decrypted_text}")

if __name__ == "__main__":
    main()
```
