## Exercise
Ask the user for a string and print out all possible anagrams.

## Reference Solution
```python
import itertools

def generate_anagrams(input_string):
    """
    Generate all possible anagrams of a given string.

    Args:
        input_string (str): The input string.

    Returns:
        list: A list of all possible anagrams.
    """
    # Remove any whitespace and convert to lowercase
    input_string = input_string.replace(" ", "").lower()

    # Generate all permutations of the input string
    anagrams = set("".join(p) for p in itertools.permutations(input_string))

    return list(anagrams)

def main():
    # Ask the user for a string
    input_string = input("Please enter a string: ")

    # Generate and print all possible anagrams
    anagrams = generate_anagrams(input_string)
    print("Anagrams:")
    for i, anagram in enumerate(anagrams):
        print(f"{i+1}. {anagram}")

if __name__ == "__main__":
    main()
```
