def check_element_in_list(element, lst):
    """
    Checks if a given element is in a list.

    Args:
        element (any): The element to be searched for in the list.
        lst (list): The list to search in.

    Returns:
        bool: True if the element is in the list, False otherwise.
    """
    return element in lst
