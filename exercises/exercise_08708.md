## Exercise
Create a simple game where the user has to collect points by answering questions correctly.

## Reference Solution
```python
# Import the random module to shuffle the questions
import random

# Define a dictionary with questions, options, and correct answers
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "correct": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "correct": "C"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "correct": "B"
    },
    "What is the smallest country in the world?": {
        "A": "Vatican City",
        "B": "Monaco",
        "C": "Nauru",
        "D": "Tuvalu",
        "correct": "A"
    },
    "What is the largest mammal?": {
        "A": "Blue whale",
        "B": "Fin whale",
        "C": "Humpback whale",
        "D": "Sperm whale",
        "correct": "A"
    }
}

def play_game():
    # Initialize the score
    score = 0

    # Shuffle the questions
    question_list = list(questions.items())
    random.shuffle(question_list)

    # Iterate over the questions
    for question, options in question_list:
        print(f"\nQuestion: {question}")
        for option, value in options.items():
            if option != "correct":
                print(f"{option}: {value}")

        # Get the user's answer
        answer = input("Choose your answer (A, B, C, D): ")

        # Check if the answer is correct
        if answer.upper() == options["correct"]:
            print("Correct answer!")
            score += 10
        else:
            print(f"Incorrect answer. The correct answer is {options['correct']}.")
            score -= 5

        print(f"Your current score is: {score}")

    print(f"\nGame over! Your final score is: {score}")

def main():
    print("Welcome to Quiz Master!")
    play_game()

if __name__ == "__main__":
    main()
```
