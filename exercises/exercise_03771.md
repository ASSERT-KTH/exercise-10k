## Exercise
Validate a dictionary based on a set of rules.

## Reference Solution
```python
from typing import Any, Dict

class Validator:
    def __init__(self, rules: Dict[str, Dict[str, Any]]):
        """
        Initialize the Validator with a set of rules.

        Args:
        rules (Dict[str, Dict[str, Any]]): A dictionary containing validation rules.
        """
        self.rules = rules

    def validate(self, data: Dict[str, Any]) -> bool:
        """
        Validate the provided data against the defined rules.

        Args:
        data (Dict[str, Any]): The dictionary to be validated.

        Returns:
        bool: True if the data is valid, False otherwise.
        """
        for key, rule in self.rules.items():
            # Check if the key is required
            if "required" in rule and key not in data:
                return False

            # Check if the key exists in the data
            if key not in data:
                continue

            # Check the type of the value
            if "type" in rule and not isinstance(data[key], rule["type"]):
                return False

            # Check the length of the value (if applicable)
            if "type" in rule and rule["type"] == str:
                if "min_length" in rule and len(data[key]) < rule["min_length"]:
                    return False
                if "max_length" in rule and len(data[key]) > rule["max_length"]:
                    return False

            # Check the value range (if applicable)
            if "type" in rule and rule["type"] in [int, float]:
                if "min_value" in rule and data[key] < rule["min_value"]:
                    return False
                if "max_value" in rule and data[key] > rule["max_value"]:
                    return False

        return True


# Example usage:
if __name__ == "__main__":
    rules = {
        "name": {"required": True, "type": str, "min_length": 3, "max_length": 50},
        "age": {"required": True, "type": int, "min_value": 18, "max_value": 100},
        "email": {"required": True, "type": str, "min_length": 5, "max_length": 100},
    }

    validator = Validator(rules)

    # Valid data
    valid_data = {
        "name": "John Doe",
        "age": 30,
        "email": "john@example.com",
    }
    print(validator.validate(valid_data))  # Output: True

    # Invalid data (missing required key)
    invalid_data_1 = {
        "name": "John Doe",
        "email": "john@example.com",
    }
    print(validator.validate(invalid_data_1))  # Output: False

    # Invalid data (invalid type)
    invalid_data_2 = {
        "name": "John Doe",
        "age": "thirty",
        "email": "john@example.com",
    }
    print(validator.validate(invalid_data_2))  # Output: False

    # Invalid data (value out of range)
    invalid_data_3 = {
        "name": "John Doe",
        "age": 150,
        "email": "john@example.com",
    }
    print(validator.validate(invalid_data_3))  # Output: False
```
