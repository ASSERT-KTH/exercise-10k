## Exercise
Write a program to check if a given array is sorted in ascending or descending order.

## Reference Solution
```python
def is_sorted(arr):
    """
    Checks if the given array is sorted in ascending or descending order.

    Args:
        arr (list): The input array to check.

    Returns:
        str: A message indicating the sorting order of the array.
    """
    ascending = all(arr[i] <= arr[i+1] for i in range(len(arr)-1))
    descending = all(arr[i] >= arr[i+1] for i in range(len(arr)-1))

    if ascending:
        return "The array is sorted in ascending order."
    elif descending:
        return "The array is sorted in descending order."
    else:
        return "The array is not sorted."

def main():
    # Example usage
    arr1 = [1, 2, 3, 4, 5]
    arr2 = [5, 4, 3, 2, 1]
    arr3 = [1, 3, 2, 4, 5]

    print(is_sorted(arr1))  # Output: The array is sorted in ascending order.
    print(is_sorted(arr2))  # Output: The array is sorted in descending order.
    print(is_sorted(arr3))  # Output: The array is not sorted.

if __name__ == "__main__":
    main()
```
