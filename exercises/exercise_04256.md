## Exercise
Create a program that helps users track their personal finances. The program should take the income, expenses, and budget type as input, and print the updated budget.

## Reference Solution
```python
class BudgetTracker:
    def __init__(self, income, expenses, budget_type):
        """
        Initialize the BudgetTracker class.

        Args:
        income (float): The user's income.
        expenses (list): A list of expenses.
        budget_type (str): The type of budget (e.g., 'monthly', 'yearly', etc.).
        """
        self.income = income
        self.expenses = expenses
        self.budget_type = budget_type

    def calculate_total_expenses(self):
        """
        Calculate the total expenses.

        Returns:
        float: The total expenses.
        """
        return sum(self.expenses)

    def calculate_remaining_balance(self):
        """
        Calculate the remaining balance after deducting expenses from income.

        Returns:
        float: The remaining balance.
        """
        return self.income - self.calculate_total_expenses()

    def print_budget(self):
        """
        Print the updated budget.
        """
        print(f"Budget Type: {self.budget_type}")
        print(f"Income: ${self.income:.2f}")
        print("Expenses:")
        for i, expense in enumerate(self.expenses, start=1):
            print(f"  {i}. ${expense:.2f}")
        print(f"Total Expenses: ${self.calculate_total_expenses():.2f}")
        print(f"Remaining Balance: ${self.calculate_remaining_balance():.2f}")


def main():
    # Get user input
    income = float(input("Enter your income: $"))
    num_expenses = int(input("Enter the number of expenses: "))
    expenses = []
    for i in range(num_expenses):
        expense = float(input(f"Enter expense {i+1}: $"))
        expenses.append(expense)
    budget_type = input("Enter the budget type (e.g., 'monthly', 'yearly', etc.): ")

    # Create a BudgetTracker object
    budget_tracker = BudgetTracker(income, expenses, budget_type)

    # Print the updated budget
    budget_tracker.print_budget()


if __name__ == "__main__":
    main()
```
