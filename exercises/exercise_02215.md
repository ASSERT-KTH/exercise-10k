## Exercise
Write a program that takes a story template and user inputs as input and generates a completed story. Print the completed story to the console.

## Reference Solution
```python
def get_user_input(prompt, input_type):
    """Gets user input and converts it to the specified type."""
    while True:
        try:
            return input_type(input(prompt))
        except ValueError:
            print("Invalid input. Please try again.")

def generate_story(template, inputs):
    """Replaces placeholders in the template with user inputs."""
    for key, value in inputs.items():
        template = template.replace(f"[{key}]", str(value))
    return template

def main():
    # Define the story template
    story_template = """
Once upon a time, in a land far, far away, there was a {adjective1} {noun1} named {name}. 
This {adjective1} {noun1} had a {adjective2} {body_part} and loved to {verb}. 
One day, {name} decided to go on a {adjective3} adventure to find the legendary {noun2}. 
Along the way, {name} met a {adjective4} {animal} who became their trusty sidekick. 
Together, they encountered many {plural_noun} and overcame incredible {adjective5} challenges.
In the end, {name} found the {noun2} and lived {adverb} ever after.
"""

    # Get user inputs
    inputs = {
        "name": get_user_input("Enter the main character's name: ", str),
        "adjective1": get_user_input("Enter an adjective (e.g., brave, kind): ", str),
        "noun1": get_user_input("Enter a noun (e.g., knight, wizard): ", str),
        "adjective2": get_user_input("Enter an adjective (e.g., big, small): ", str),
        "body_part": get_user_input("Enter a body part (e.g., nose, eyes): ", str),
        "verb": get_user_input("Enter a verb (e.g., run, jump): ", str),
        "adjective3": get_user_input("Enter an adjective (e.g., exciting, boring): ", str),
        "noun2": get_user_input("Enter a noun (e.g., treasure, castle): ", str),
        "adjective4": get_user_input("Enter an adjective (e.g., friendly, fierce): ", str),
        "animal": get_user_input("Enter an animal (e.g., dog, dragon): ", str),
        "plural_noun": get_user_input("Enter a plural noun (e.g., obstacles, enemies): ", str),
        "adjective5": get_user_input("Enter an adjective (e.g., difficult, easy): ", str),
        "adverb": get_user_input("Enter an adverb (e.g., happily, sadly): ", str),
    }

    # Generate and print the completed story
    completed_story = generate_story(story_template, inputs)
    print(completed_story)

if __name__ == "__main__":
    main()
```
