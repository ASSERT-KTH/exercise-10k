## Exercise
Write a program that prints out the day of the year based on a given date using if/else statements

## Reference Solution

def calculate_day_of_year(month, day):
    """
    Calculate the day of the year based on the given month and day.

    Args:
        month (int): The month of the year (1-12).
        day (int): The day of the month (1-31).

    Returns:
        int: The day of the year (1-365).
    """

    # Define the number of days in each month
    days_in_month = {
        1: 31,  # January
        2: 28,  # February (non-leap year)
        3: 31,  # March
        4: 30,  # April
        5: 31,  # May
        6: 30,  # June
        7: 31,  # July
        8: 31,  # August
        9: 30,  # September
        10: 31,  # October
        11: 30,  # November
        12: 31  # December
    }

    # Initialize the day of the year
    day_of_year = 0

    # Calculate the day of the year
    for i in range(1, month):
        if i == 2:  # February
            day_of_year += 28
        elif i in [1, 3, 5, 7, 8, 10, 12]:  # Months with 31 days
            day_of_year += 31
        else:  # Months with 30 days
            day_of_year += 30

    # Add the day of the month to the day of the year
    day_of_year += day

    return day_of_year


def main():
    # Get the month and day from the user
    month = int(input("Enter the month (1-12): "))
    day = int(input("Enter the day (1-31): "))

    # Validate the input
    if month < 1 or month > 12:
        print("Invalid month. Please enter a number between 1 and 12.")
    elif day < 1 or day > 31:
        print("Invalid day. Please enter a number between 1 and 31.")
    else:
        # Calculate and print the day of the year
        day_of_year = calculate_day_of_year(month, day)
        print(f"The day of the year is: {day_of_year}")


if __name__ == "__main__":
    main()
