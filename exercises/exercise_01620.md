## Exercise
Create a program that takes the coordinates of two points as input and outputs the distance between them.

## Reference Solution
```python
import math

def distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.

    Args:
        point1 (list): Coordinates of the first point.
        point2 (list): Coordinates of the second point.

    Returns:
        float: The distance between the two points.
    """
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of dimensions")

    squared_distances = [(a - b) ** 2 for a, b in zip(point1, point2)]
    return math.sqrt(sum(squared_distances))

def main():
    # Get the coordinates of the two points from the user
    num_dimensions = int(input("Enter the number of dimensions (2 or 3): "))
    point1 = [float(input(f"Enter x{'' if num_dimensions == 2 else f'{i+1}'} coordinate of point 1: ")) for i in range(num_dimensions)]
    point2 = [float(input(f"Enter x{'' if num_dimensions == 2 else f'{i+1}'} coordinate of point 2: ")) for i in range(num_dimensions)]

    # Calculate and print the distance
    distance_between_points = distance(point1, point2)
    print(f"The distance between the two points is: {distance_between_points}")

if __name__ == "__main__":
    main()
```
