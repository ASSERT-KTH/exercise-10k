## Exercise
Write a program to demonstrate the scope of a variable, including local and global variables.

## Reference Solution
```python
# Global variable
global_variable = 10

def local_variable_scope():
    """Demonstrate local variable scope"""
    # Local variable
    local_variable = 20
    print("Inside local_variable_scope function:")
    print(f"Local variable: {local_variable}")
    print(f"Global variable: {global_variable}")

def modify_global_variable():
    """Demonstrate modifying global variable"""
    global global_variable  # Declare global variable
    global_variable = 30
    print("Inside modify_global_variable function:")
    print(f"Global variable: {global_variable}")

def main():
    print("Outside any function:")
    print(f"Global variable: {global_variable}")
    
    local_variable_scope()
    
    print("After local_variable_scope function:")
    try:
        print(f"Local variable: {local_variable}")
    except NameError:
        print("Local variable is not accessible outside its scope.")
    
    modify_global_variable()
    
    print("After modify_global_variable function:")
    print(f"Global variable: {global_variable}")

if __name__ == "__main__":
    main()
```
