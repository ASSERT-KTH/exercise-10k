## Exercise
Create a program that calculates a tip based on the total bill using if/else statements.

## Reference Solution
```python
def calculate_tip(bill, service):
    """
    Calculate a tip based on the total bill and service quality.

    Args:
        bill (float): The total bill amount.
        service (str): The quality of service (poor, good, excellent).

    Returns:
        float: The calculated tip amount.
    """
    if service == "poor":
        tip_percentage = 0.05
    elif service == "good":
        tip_percentage = 0.10
    elif service == "excellent":
        tip_percentage = 0.15
    else:
        print("Invalid service quality. Defaulting to 10% tip.")
        tip_percentage = 0.10

    tip_amount = bill * tip_percentage
    return tip_amount

def main():
    # Get user input
    bill = float(input("Enter the total bill amount: $"))
    service = input("Enter the quality of service (poor, good, excellent): ")

    # Calculate the tip
    tip_amount = calculate_tip(bill, service)

    # Print the result
    print(f"Bill: ${bill:.2f}")
    print(f"Tip ({service} service): ${tip_amount:.2f}")
    print(f"Total: ${bill + tip_amount:.2f}")

if __name__ == "__main__":
    main()
```
