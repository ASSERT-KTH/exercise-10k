## Exercise
Write a program that takes a list of integers as input and returns the element at the index specified by the user.

## Reference Solution
```python
def get_element_at_index(lst, index):
    """
    Retrieves the element at the specified index from the given list.

    Args:
        lst (list): The list of integers.
        index (int): The index of the desired element.

    Returns:
        int: The element at the specified index if it exists, otherwise None.
    """
    try:
        return lst[index]
    except IndexError:
        return None


def main():
    # Example list of integers
    numbers = [1, 2, 3, 4, 5]

    # Get the index from the user
    while True:
        try:
            index = int(input("Enter the index of the element you want to retrieve: "))
            break
        except ValueError:
            print("Invalid input. Please enter an integer.")

    # Get the element at the specified index
    element = get_element_at_index(numbers, index)

    if element is not None:
        print(f"The element at index {index} is {element}.")
    else:
        print(f"Index {index} is out of range.")


if __name__ == "__main__":
    main()
```
