## Exercise
Write a program that calculates the sum of all elements in a given array. The array will contain only integers.

## Reference Solution
```python
def calculate_array_sum(array):
    """
    Calculates the sum of all elements in a given array.

    Args:
        array (list): A list of integers.

    Returns:
        int: The sum of all elements in the array.

    Raises:
        TypeError: If the input is not a list.
        ValueError: If the list contains non-integer values.
    """
    if not isinstance(array, list):
        raise TypeError("Input must be a list.")
    if not all(isinstance(x, int) for x in array):
        raise ValueError("List must contain only integers.")
    return sum(array)


def main():
    # Example usage
    array = [1, 2, 3, 4, 5]
    try:
        result = calculate_array_sum(array)
        print(f"The sum of the array is: {result}")
    except (TypeError, ValueError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
