## Exercise
Write a function to compress a given string by replacing repeated characters with their count

## Reference Solution
```python
def compress_string(input_str):
    """
    Compress a given string by replacing repeated characters with their count.

    Args:
        input_str (str): The input string to be compressed.

    Returns:
        str: The compressed string.
    """
    if not input_str:
        return ""

    compressed_str = []
    count = 1

    for i in range(1, len(input_str)):
        if input_str[i] == input_str[i - 1]:
            count += 1
        else:
            compressed_str.append(input_str[i - 1] + str(count))
            count = 1

    compressed_str.append(input_str[-1] + str(count))
    compressed_str = "".join(compressed_str)

    return compressed_str


# Example usage:
if __name__ == "__main__":
    input_str = "AAABBBCCCDDDD"
    compressed = compress_string(input_str)
    print(f"Input String: {input_str}")
    print(f"Compressed String: {compressed}")
```
