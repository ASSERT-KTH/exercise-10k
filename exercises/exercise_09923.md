## Exercise
Write a program that takes an array of integers as input and reverses the order of its elements. The program should then print the reversed array.

## Reference Solution
```python
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> v(n);

    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }

    reverse(v.begin(), v.end());

    for (int i = 0; i < n; i++) {
        cout << v[i] << " ";
    }

    return 0;
}
```
