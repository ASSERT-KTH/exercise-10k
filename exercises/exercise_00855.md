## Exercise
Create a function that curries another function, transforming it into a sequence of functions each taking a single argument.

def curry(func):
    """
    This function curries another function, transforming it into a sequence of functions each taking a single argument.

    Args:
        func (function): The function to be curried.

    Returns:
        function: A curried version of the input function.
    """

    # Define a helper function that will be used to curry the input function
    def curried_func(*args, **kwargs):
        
        # If the number of arguments passed is equal to the number of arguments required by the function,
        # then call the function with the passed arguments and return the result
        if len(args) + len(kwargs) == func.__code__.co_argcount:
            return func(*args, **kwargs)
        
        # If the number of arguments passed is less than the number of arguments required by the function,
        # then return a new function that takes the remaining arguments
        else:
            # Define a new function that takes the remaining arguments
            def new_func(*new_args, **new_kwargs):
                return curried_func(*(args + new_args), **dict(kwargs, **new_kwargs))
            
            # Return the new function
            return new_func
    
    # Return the curried function
    return curried_func

# Example usage:
def add(a, b, c):
    return a + b + c

curried_add = curry(add)
print(curried_add(1)(2)(3))  # Outputs: 6
