## Exercise
Create a dictionary and count the occurrences of each value

## Reference Solution
```python
from collections import Counter

def count_value_occurrences(dictionary):
    """
    Counts the occurrences of each value in a dictionary.
    
    Args:
    dictionary (dict): The input dictionary.
    
    Returns:
    dict: A dictionary where keys are the unique values from the input dictionary
          and values are their respective occurrence counts.
    """
    # Get all values from the dictionary
    values = dictionary.values()
    
    # Use Counter to count the occurrences of each value
    value_counts = Counter(values)
    
    return dict(value_counts)

# Example usage:
if __name__ == "__main__":
    example_dict = {
        "key1": "value1",
        "key2": "value2",
        "key3": "value1",
        "key4": "value3",
        "key5": "value2",
        "key6": "value1"
    }
    
    print("Original Dictionary:")
    print(example_dict)
    
    value_occurrences = count_value_occurrences(example_dict)
    print("\nValue Occurrences:")
    print(value_occurrences)
```
