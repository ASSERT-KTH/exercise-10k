## Exercise
Write a program that finds the first duplicate in a list of numbers.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Finds the first duplicate in a list of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        The first duplicate number, or None if no duplicates are found.
    """
    seen = {}
    for i, num in enumerate(nums):
        if num in seen:
            return num
        seen[num] = i
    return None

# Example usage:
def main():
    nums = [2, 1, 3, 4, 5, 3, 6]
    duplicate = find_first_duplicate(nums)
    if duplicate is not None:
        print(f"The first duplicate is {duplicate}.")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
