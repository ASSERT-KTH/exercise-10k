## Exercise
Write a program that finds the longest word in a given sentence using a for loop.

## Reference Solution
```python
def find_longest_word(sentence):
    """
    Finds the longest word in a given sentence.

    Args:
        sentence (str): The input sentence.

    Returns:
        str: The longest word in the sentence.
    """
    words = sentence.split()  # Split the sentence into words
    longest_word = ""  # Initialize the longest word
    max_length = 0  # Initialize the maximum length

    for word in words:
        word = word.strip(".,!?:;")  # Remove punctuation
        if len(word) > max_length:
            max_length = len(word)
            longest_word = word

    return longest_word

def main():
    sentence = input("Enter a sentence: ")
    longest_word = find_longest_word(sentence)
    print(f"The longest word is: {longest_word}")

if __name__ == "__main__":
    main()
```
