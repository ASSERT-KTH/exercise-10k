## Exercise
Write a program that calculates the distance between two points given their coordinates.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the distance between two points.

    Args:
        point1 (list): Coordinates of the first point (x, y, z)
        point2 (list): Coordinates of the second point (x, y, z)

    Returns:
        float: Distance between the two points
    """
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of coordinates")

    squared_differences = [(a - b) ** 2 for a, b in zip(point1, point2)]
    distance = math.sqrt(sum(squared_differences))
    return distance

def main():
    # Example usage
    point1 = [1, 2, 3]  # x, y, z coordinates of the first point
    point2 = [4, 5, 6]  # x, y, z coordinates of the second point

    distance = calculate_distance(point1, point2)
    print(f"The distance between the two points is: {distance:.2f}")

if __name__ == "__main__":
    main()
```
