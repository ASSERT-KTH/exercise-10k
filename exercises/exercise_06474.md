## Exercise
Write a function that takes a dictionary as input and removes and returns an arbitrary key-value pair

## Reference Solution
```python
def remove_arbitrary_pair(input_dict):
    """
    Removes and returns an arbitrary key-value pair from the input dictionary.

    Args:
    input_dict (dict): The dictionary to remove the key-value pair from.

    Returns:
    tuple: A tuple containing the removed key and value, or None if the dictionary is empty.
    """
    if not input_dict:
        return None

    # Get an arbitrary key from the dictionary
    key = next(iter(input_dict))

    # Get the corresponding value
    value = input_dict[key]

    # Remove the key-value pair from the dictionary
    del input_dict[key]

    # Return the removed key-value pair
    return key, value


# Example usage:
def main():
    example_dict = {"apple": 1, "banana": 2, "cherry": 3}
    print("Original dictionary:", example_dict)

    removed_pair = remove_arbitrary_pair(example_dict)
    print("Removed pair:", removed_pair)

    print("Dictionary after removal:", example_dict)


if __name__ == "__main__":
    main()
```
