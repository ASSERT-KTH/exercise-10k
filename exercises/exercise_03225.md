Write a program to solve the Towers of Hanoi problem, where we need to move a stack of disks from one peg to another, subject to certain constraints.

def towers_of_hanoi(n, from_peg, to_peg, aux_peg):
    """
    Solve the Towers of Hanoi problem.

    Args:
        n (int): The number of disks.
        from_peg (str): The peg where the disks are initially located.
        to_peg (str): The peg where the disks need to be moved.
        aux_peg (str): The auxiliary peg.

    Returns:
        None
    """
    if n == 1:
        # Base case: move one disk from the from_peg to the to_peg.
        print(f"Move disk 1 from {from_peg} to {to_peg}")
        return

    # Move n-1 disks from the from_peg to the aux_peg.
    towers_of_hanoi(n - 1, from_peg, aux_peg, to_peg)

    # Move the nth disk from the from_peg to the to_peg.
    print(f"Move disk {n} from {from_peg} to {to_peg}")

    # Move the n-1 disks from the aux_peg to the to_peg.
    towers_of_hanoi(n - 1, aux_peg, to_peg, from_peg)


def main():
    # Example usage:
    n = 3  # Number of disks.
    from_peg = "A"
    to_peg = "C"
    aux_peg = "B"

    print("Towers of Hanoi solution:")
    towers_of_hanoi(n, from_peg, to_peg, aux_peg)


if __name__ == "__main__":
    main()
