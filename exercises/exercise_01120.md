## Exercise
Write a program to calculate the compound interest on an investment given the principal amount, interest rate, and time period

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, n=1):
    """
    Calculate compound interest.

    Args:
        principal (float): The initial investment amount.
        rate (float): The annual interest rate (in decimal form, e.g. 4% = 0.04).
        time (int): The time period in years.
        n (int, optional): The number of times interest is compounded per year. Defaults to 1.

    Returns:
        float: The compound interest amount.
    """
    return principal * ((1 + rate / n) ** (n * time)) - principal

def main():
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period in years: "))
    n = int(input("Enter the number of times interest is compounded per year (default=1): ") or 1)

    interest = calculate_compound_interest(principal, rate, time, n)
    print(f"The compound interest is: ${interest:.2f}")
    print(f"The total amount after {time} years is: ${principal + interest:.2f}")

if __name__ == "__main__":
    main()
```
