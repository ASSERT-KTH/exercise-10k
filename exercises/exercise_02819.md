## Exercise
Write a program to create a simple weather forecast that prints out the current weather and a 5-day forecast.

## Reference Solution
```python
import requests
import json

def get_weather(api_key, location):
    """
    Fetches the current weather for a specified location.

    Args:
        api_key (str): The OpenWeatherMap API key.
        location (str): The location for which to fetch the weather.

    Returns:
        dict: The current weather data.
    """
    base_url = f"http://api.openweathermap.org/data/2.5/weather?q={location}&appid={api_key}&units=metric"
    response = requests.get(base_url)
    weather_data = response.json()
    return weather_data

def get_forecast(api_key, location):
    """
    Fetches the 5-day forecast for a specified location.

    Args:
        api_key (str): The OpenWeatherMap API key.
        location (str): The location for which to fetch the forecast.

    Returns:
        dict: The 5-day forecast data.
    """
    base_url = f"http://api.openweathermap.org/data/2.5/forecast?q={location}&appid={api_key}&units=metric"
    response = requests.get(base_url)
    forecast_data = response.json()
    return forecast_data

def print_weather(weather_data):
    """
    Prints the current weather.

    Args:
        weather_data (dict): The current weather data.
    """
    print("Current Weather:")
    print(f"Location: {weather_data['name']}, {weather_data['sys']['country']}")
    print(f"Temperature: {weather_data['main']['temp']}°C")
    print(f"Weather: {weather_data['weather'][0]['description']}")
    print(f"Humidity: {weather_data['main']['humidity']}%")
    print(f"Wind Speed: {weather_data['wind']['speed']} m/s")

def print_forecast(forecast_data):
    """
    Prints the 5-day forecast.

    Args:
        forecast_data (dict): The 5-day forecast data.
    """
    print("\n5-Day Forecast:")
    for i, forecast in enumerate(forecast_data['list'][:5]):
        print(f"Day {i+1}:")
        print(f"Date: {forecast['dt_txt']}")
        print(f"Temperature: {forecast['main']['temp']}°C")
        print(f"Weather: {forecast['weather'][0]['description']}")
        print(f"Humidity: {forecast['main']['humidity']}%")
        print(f"Wind Speed: {forecast['wind']['speed']} m/s")
        print()

def main():
    api_key = "YOUR_OPENWEATHERMAP_API_KEY"
    location = "London"
    weather_data = get_weather(api_key, location)
    forecast_data = get_forecast(api_key, location)
    print_weather(weather_data)
    print_forecast(forecast_data)

if __name__ == "__main__":
    main()
```
