## Exercise
Create a program to find the length of the longest increasing subsequence in a given list of integers.

## Reference Solution
```python
def longest_increasing_subsequence(sequence):
    """
    Find the length of the longest increasing subsequence in a given list of integers.

    Args:
        sequence (list): A list of integers.

    Returns:
        int: The length of the longest increasing subsequence.
    """
    if not sequence:
        return 0

    # Initialize a list to store the lengths of LIS ending at each position
    lengths = [1] * len(sequence)

    # Iterate through the sequence
    for i in range(1, len(sequence)):
        # Compare the current element with all previous elements
        for j in range(i):
            # If the current element is greater than the previous element, update the length
            if sequence[i] > sequence[j]:
                lengths[i] = max(lengths[i], lengths[j] + 1)

    # Return the maximum length found
    return max(lengths)


# Example usage
def main():
    sequence = [10, 22, 9, 33, 21, 50, 41, 60]
    print("Length of the longest increasing subsequence:", longest_increasing_subsequence(sequence))


if __name__ == "__main__":
    main()
```
