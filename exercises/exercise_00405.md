## Exercise
Given an array of integers representing a sequence of numbers, find the missing number.

## Reference Solution

def find_missing_number(sequence):
    """
    Find the missing number in a sequence of numbers.

    Args:
        sequence (list): A list of integers representing a sequence of numbers.

    Returns:
        int: The missing number in the sequence.

    Raises:
        ValueError: If the input sequence is not a list of integers.
    """
    if not isinstance(sequence, list) or not all(isinstance(x, int) for x in sequence):
        raise ValueError("Input sequence must be a list of integers")

    # Calculate the sum of the sequence using the formula for the sum of an arithmetic series
    n = len(sequence) + 1
    expected_sum = (n * (min(sequence) + max(sequence))) // 2

    # Calculate the actual sum of the sequence
    actual_sum = sum(sequence)

    # The missing number is the difference between the expected sum and the actual sum
    missing_number = expected_sum - actual_sum

    return missing_number

# Example usage:
sequence = [1, 2, 3, 5]
missing_number = find_missing_number(sequence)
print(f"The missing number in the sequence is: {missing_number}")
