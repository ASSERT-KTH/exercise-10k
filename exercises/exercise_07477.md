## Exercise
Write a program that unpacks a given tuple into separate variables

## Reference Solution
```python
def unpack_tuple(input_tuple):
    """
    Unpacks a given tuple into separate variables.

    Args:
        input_tuple (tuple): The tuple to be unpacked.

    Returns:
        tuple: A tuple containing the unpacked variables.
    """
    # Check if the tuple is not empty
    if len(input_tuple) == 0:
        return None

    # Unpack the tuple into separate variables
    variables = input_tuple

    return variables

def main():
    # Example usage
    input_tuple = (1, 2, 3, 4, 5)
    print("Input Tuple:", input_tuple)

    unpacked_variables = unpack_tuple(input_tuple)

    if unpacked_variables is not None:
        # Unpack the tuple into separate variables
        var1, var2, var3, var4, var5 = unpacked_variables

        print("Unpacked Variables:")
        print("Var 1:", var1)
        print("Var 2:", var2)
        print("Var 3:", var3)
        print("Var 4:", var4)
        print("Var 5:", var5)
    else:
        print("The input tuple is empty.")

if __name__ == "__main__":
    main()
```
