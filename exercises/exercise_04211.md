## Exercise
Find the maximum contiguous subarray of an array.

## Reference Solution
```python
def max_contiguous_subarray(arr):
    """
    Find the maximum contiguous subarray of an array.

    Args:
        arr (list): The input array.

    Returns:
        tuple: The maximum sum and the corresponding subarray.
    """
    max_sum = float('-inf')
    current_sum = 0
    start = 0
    end = 0
    temp_start = 0

    for i in range(len(arr)):
        current_sum += arr[i]
        if current_sum > max_sum:
            max_sum = current_sum
            start = temp_start
            end = i
        if current_sum < 0:
            current_sum = 0
            temp_start = i + 1

    return max_sum, arr[start:end+1]

# Example usage
arr = [-2, -3, 4, -1, -2, 1, 5, -3]
max_sum, subarray = max_contiguous_subarray(arr)
print(f"Maximum sum: {max_sum}")
print(f"Maximum contiguous subarray: {subarray}")
```
