## Exercise
Write a program that takes a user input string and converts it into different data types (integer, float, string). The program should handle potential errors during the conversion process.

## Reference Solution
```python
def convert_to_int(user_input):
    try:
        return int(user_input)
    except ValueError:
        return None

def convert_to_float(user_input):
    try:
        return float(user_input)
    except ValueError:
        return None

def convert_to_string(user_input):
    return str(user_input)

def main():
    user_input = input("Please enter a string: ")

    # Convert to integer
    int_value = convert_to_int(user_input)
    if int_value is not None:
        print(f"Converted to integer: {int_value}")
    else:
        print("Failed to convert to integer.")

    # Convert to float
    float_value = convert_to_float(user_input)
    if float_value is not None:
        print(f"Converted to float: {float_value}")
    else:
        print("Failed to convert to float.")

    # Convert to string
    string_value = convert_to_string(user_input)
    print(f"Converted to string: {string_value}")

if __name__ == "__main__":
    main()
```
