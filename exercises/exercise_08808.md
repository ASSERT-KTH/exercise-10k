## Exercise
Write a program that takes a polynomial and a value as input and returns the result of the polynomial evaluation.

## Reference Solution
```python
def evaluate_polynomial(coefficients, x):
    """
    Evaluate a polynomial at a given point.

    Args:
    coefficients (list): A list of coefficients of the polynomial, 
                         where coefficients[i] is the coefficient of x^i.
    x (float): The point at which to evaluate the polynomial.

    Returns:
    float: The value of the polynomial at the given point.
    """
    result = 0
    for i, coefficient in enumerate(coefficients):
        result += coefficient * (x ** i)
    return result


def main():
    # Get the coefficients of the polynomial from the user
    degree = int(input("Enter the degree of the polynomial: "))
    coefficients = []
    for i in range(degree + 1):
        coefficient = float(input(f"Enter the coefficient of x^{i}: "))
        coefficients.append(coefficient)

    # Get the value at which to evaluate the polynomial from the user
    x = float(input("Enter the value at which to evaluate the polynomial: "))

    # Evaluate the polynomial
    result = evaluate_polynomial(coefficients, x)

    # Print the result
    print(f"The value of the polynomial at x = {x} is {result}")


if __name__ == "__main__":
    main()
```
