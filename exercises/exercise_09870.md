## Exercise
Write a program that generates a sequence of numbers based on user input.

## Reference Solution
```python
def arithmetic_sequence(a, d, n):
    """
    Generates an arithmetic sequence.

    Args:
        a (int): The first term of the sequence.
        d (int): The common difference.
        n (int): The number of terms.

    Returns:
        list: The generated arithmetic sequence.
    """
    sequence = [a + i * d for i in range(n)]
    return sequence


def geometric_sequence(a, r, n):
    """
    Generates a geometric sequence.

    Args:
        a (int): The first term of the sequence.
        r (int): The common ratio.
        n (int): The number of terms.

    Returns:
        list: The generated geometric sequence.
    """
    sequence = [a * (r ** i) for i in range(n)]
    return sequence


def fibonacci_sequence(n):
    """
    Generates a Fibonacci sequence.

    Args:
        n (int): The number of terms.

    Returns:
        list: The generated Fibonacci sequence.
    """
    sequence = [0, 1]
    while len(sequence) < n:
        sequence.append(sequence[-1] + sequence[-2])
    return sequence[:n]


def main():
    print("Sequence Generator Program")
    print("-------------------------------")
    print("1. Arithmetic Sequence")
    print("2. Geometric Sequence")
    print("3. Fibonacci Sequence")

    choice = int(input("Enter your choice (1/2/3): "))

    if choice == 1:
        a = int(input("Enter the first term: "))
        d = int(input("Enter the common difference: "))
        n = int(input("Enter the number of terms: "))
        print("Arithmetic Sequence:", arithmetic_sequence(a, d, n))
    elif choice == 2:
        a = int(input("Enter the first term: "))
        r = int(input("Enter the common ratio: "))
        n = int(input("Enter the number of terms: "))
        print("Geometric Sequence:", geometric_sequence(a, r, n))
    elif choice == 3:
        n = int(input("Enter the number of terms: "))
        print("Fibonacci Sequence:", fibonacci_sequence(n))
    else:
        print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
