Convert a temperature from Celsius to Fahrenheit or vice versa using variables and operators.

def convert_temperature(temp, unit):
    """
    Convert temperature from Celsius to Fahrenheit or vice versa.

    Args:
        temp (float): Temperature value.
        unit (str): Unit of temperature ('C' for Celsius, 'F' for Fahrenheit).

    Returns:
        float: Converted temperature value.
    """
    if unit.upper() == 'C':
        # Convert Celsius to Fahrenheit
        return (temp * 9/5) + 32
    elif unit.upper() == 'F':
        # Convert Fahrenheit to Celsius
        return (temp - 32) * 5/9
    else:
        raise ValueError("Invalid unit. Please use 'C' for Celsius or 'F' for Fahrenheit.")
