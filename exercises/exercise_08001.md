## Exercise
Create a program that uses a dictionary to store student names as keys and their grades as values. The program should allow users to add students, update grades, and display all student grades. The dictionary should be initialized with a set of students and their grades.

## Reference Solution
```python
# Initialize a dictionary with a set of students and their grades
student_grades = {
    "John": 85,
    "Alice": 90,
    "Bob": 78,
    "Charlie": 92,
    "David": 88
}

def add_student():
    """Add a new student to the dictionary"""
    name = input("Enter the student's name: ")
    grade = float(input("Enter the student's grade: "))
    
    # Check if the student already exists
    if name in student_grades:
        print("Student already exists. Please update their grade instead.")
    else:
        student_grades[name] = grade
        print(f"Student '{name}' added successfully.")

def update_grade():
    """Update the grade of an existing student"""
    name = input("Enter the student's name: ")
    
    # Check if the student exists
    if name in student_grades:
        grade = float(input("Enter the new grade: "))
        student_grades[name] = grade
        print(f"Grade of '{name}' updated successfully.")
    else:
        print("Student does not exist. Please add them first.")

def display_grades():
    """Display all student grades"""
    if not student_grades:
        print("No students in the database.")
    else:
        print("\nStudent Grades:")
        for name, grade in student_grades.items():
            print(f"{name}: {grade}")

def main():
    while True:
        print("\nStudent Gradebook Menu:")
        print("1. Add Student")
        print("2. Update Grade")
        print("3. Display Grades")
        print("4. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            add_student()
        elif choice == "2":
            update_grade()
        elif choice == "3":
            display_grades()
        elif choice == "4":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
