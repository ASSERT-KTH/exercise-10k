## Exercise
Write a program to demonstrate implicit type conversion.

## Reference Solution
```python
# Implicit Type Conversion Demonstration

def implicit_type_conversion():
    # Integer and Float
    print("Integer and Float:")
    int_val = 10
    float_val = 10.5
    result = int_val + float_val  # Implicit conversion of int to float
    print(f"{int_val} (int) + {float_val} (float) = {result} (float)")

    # Integer and Complex
    print("\nInteger and Complex:")
    int_val = 10
    complex_val = 10 + 5j
    result = int_val + complex_val  # Implicit conversion of int to complex
    print(f"{int_val} (int) + {complex_val} (complex) = {result} (complex)")

    # Boolean and Integer
    print("\nBoolean and Integer:")
    bool_val = True
    int_val = 10
    result = bool_val + int_val  # Implicit conversion of bool to int
    print(f"{bool_val} (bool) + {int_val} (int) = {result} (int)")


if __name__ == "__main__":
    implicit_type_conversion()
```
