## Exercise
Write a program to perform basic list operations (indexing, slicing, append, insert).

## Reference Solution
```python
def main():
    # Initialize a list
    my_list = [1, 2, 3, 4, 5]
    print("Original List:", my_list)

    # Indexing
    print("\nIndexing:")
    print("Element at index 0:", my_list[0])
    print("Element at index -1 (last element):", my_list[-1])

    # Slicing
    print("\nSlicing:")
    print("Elements from index 1 to 3:", my_list[1:4])
    print("Elements from index 2 to the end:", my_list[2:])
    print("Elements from the start to index 3:", my_list[:4])

    # Append
    print("\nAppend:")
    my_list.append(6)
    print("List after appending 6:", my_list)

    # Insert
    print("\nInsert:")
    my_list.insert(2, 7)
    print("List after inserting 7 at index 2:", my_list)

if __name__ == "__main__":
    main()
```
