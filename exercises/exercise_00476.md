## Exercise
Write a program that takes a string as input and returns a snake pattern of the string

## Reference Solution
```python
def generate_snake_pattern(input_string):
    """
    Generates a snake pattern of the input string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The snake pattern of the input string.
    """
    rows = []
    direction = 1  # 1 for right, -1 for left
    row = []

    for char in input_string:
        row.append(char)
        if direction == 1:
            if len(row) == len(input_string) // len(rows) + 1:
                rows.append(row)
                row = []
                direction = -1
        else:
            if len(row) == len(input_string) // len(rows) + 1:
                rows.append(row)
                row = []
                direction = 1

    if row:
        rows.append(row)

    snake_pattern = []
    for i in range(len(rows)):
        if i % 2 == 0:
            snake_pattern.extend(rows[i])
        else:
            snake_pattern.extend(reversed(rows[i]))

    return ''.join(snake_pattern)


def main():
    input_string = input("Enter a string: ")
    print(generate_snake_pattern(input_string))


if __name__ == "__main__":
    main()
```
