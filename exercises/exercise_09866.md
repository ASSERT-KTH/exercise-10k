## Exercise
Write a program that takes a set of words as input and generates a silly story.

## Reference Solution
```python
import random

def generate_silly_story(words):
    """
    Generates a silly story using the provided words.

    Args:
        words (list): A list of words to be used in the story.

    Returns:
        str: The generated silly story.
    """
    story = ""

    # Randomly select a main character from the input words
    main_character = random.choice(words)
    story += f"Once upon a time, in a land far, far away, there lived a {main_character} named Bob.\n"

    # Create a random adventure for the main character
    adventure = random.choice(["went on a quest", "discovered a hidden treasure", "encountered a magical creature"])
    story += f"One day, Bob {adventure} and found himself in a strange and wonderful place.\n"

    # Introduce a random obstacle or challenge
    obstacle = random.choice(["a fierce dragon", "a treacherous mountain", "a mysterious puzzle"])
    story += f"But, to his surprise, he was confronted by {obstacle}.\n"

    # Have the main character overcome the obstacle using the input words
    solution = random.choice(words)
    story += f"Using his quick thinking and a trusty {solution}, Bob was able to outsmart the {obstacle} and continue his journey.\n"

    # Conclude the story with a happy ending
    story += f"In the end, Bob returned home a hero, with a tale to tell and a heart full of joy.\n"

    return story

def main():
    # Get the input words from the user
    num_words = int(input("Enter the number of words: "))
    words = []
    for i in range(num_words):
        word = input(f"Enter word {i+1}: ")
        words.append(word)

    # Generate and print the silly story
    story = generate_silly_story(words)
    print("\nHere's your silly story:")
    print(story)

if __name__ == "__main__":
    main()
```
