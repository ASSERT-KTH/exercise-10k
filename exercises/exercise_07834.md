## Exercise
Create a program that stores student grades in a dictionary. The dictionary should have student names as keys and a list of their grades as values. The program should prompt the user to input a student's name and their grades for a certain number of assignments. It should then calculate the average grade for each student and print out the student's name, grades, and average grade.

## Reference Solution
```python
def get_student_grades():
    """Get student grades from user input."""
    student_grades = {}
    
    # Get the number of students
    num_students = int(input("Enter the number of students: "))
    
    for _ in range(num_students):
        # Get the student's name
        student_name = input("Enter the student's name: ")
        
        # Get the number of assignments
        num_assignments = int(input(f"Enter the number of assignments for {student_name}: "))
        
        # Get the grades for each assignment
        grades = []
        for i in range(num_assignments):
            while True:
                try:
                    grade = float(input(f"Enter grade for assignment {i+1}: "))
                    if 0 <= grade <= 100:
                        grades.append(grade)
                        break
                    else:
                        print("Grade must be between 0 and 100.")
                except ValueError:
                    print("Invalid input. Please enter a number.")
        
        # Store the grades in the dictionary
        student_grades[student_name] = grades
    
    return student_grades


def calculate_average_grades(student_grades):
    """Calculate the average grade for each student."""
    average_grades = {}
    
    for student, grades in student_grades.items():
        average_grade = sum(grades) / len(grades)
        average_grades[student] = average_grade
    
    return average_grades


def print_student_grades(student_grades, average_grades):
    """Print out the student's name, grades, and average grade."""
    for student, grades in student_grades.items():
        print(f"Student: {student}")
        print(f"Grades: {grades}")
        print(f"Average Grade: {average_grades[student]:.2f}")
        print()


def main():
    student_grades = get_student_grades()
    average_grades = calculate_average_grades(student_grades)
    print_student_grades(student_grades, average_grades)


if __name__ == "__main__":
    main()
```
