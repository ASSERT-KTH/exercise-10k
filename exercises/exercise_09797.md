## Exercise
Write a program that takes an input and validates it using set operations.

## Reference Solution
```python
def validate_set(input_set, reference_set):
    """
    Validate the input set using set operations.

    Args:
        input_set (set): The input set to be validated.
        reference_set (set): The reference set used for validation.

    Returns:
        dict: A dictionary containing the validation results.
    """
    validation_results = {
        "is_subset": input_set.issubset(reference_set),
        "is_superset": input_set.issuperset(reference_set),
        "is_equal": input_set == reference_set,
        "intersection": input_set & reference_set,
        "union": input_set | reference_set,
        "difference": input_set - reference_set,
        "symmetric_difference": input_set ^ reference_set
    }
    return validation_results


def main():
    # Example usage
    input_set = {1, 2, 3, 4}
    reference_set = {1, 2, 3, 4, 5}

    validation_results = validate_set(input_set, reference_set)

    print("Validation Results:")
    print(f"Is Subset: {validation_results['is_subset']}")
    print(f"Is Superset: {validation_results['is_superset']}")
    print(f"Is Equal: {validation_results['is_equal']}")
    print(f"Intersection: {validation_results['intersection']}")
    print(f"Union: {validation_results['union']}")
    print(f"Difference: {validation_results['difference']}")
    print(f"Symmetric Difference: {validation_results['symmetric_difference']}")


if __name__ == "__main__":
    main()
```
