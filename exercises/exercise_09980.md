## Exercise
Write a program that demonstrates the concept of immutable variables.

## Reference Solution
```python
# Immutable Variables Program

# Immutable Variables
# ------------------

# In Python, the following data types are immutable:
# - Integers
# - Floats
# - Complex numbers
# - Strings
# - Tuples

# Example Usage
# -------------

def main():
    # Immutable integer
    immutable_int = 10
    print("Immutable Integer:", immutable_int)

    # Trying to change the immutable integer
    # This will not change the original integer, but create a new one
    immutable_int += 1
    print("Immutable Integer after change:", immutable_int)

    # Immutable string
    immutable_str = "Hello"
    print("Immutable String:", immutable_str)

    # Trying to change the immutable string
    # This will not change the original string, but create a new one
    immutable_str += " World"
    print("Immutable String after change:", immutable_str)

    # Immutable tuple
    immutable_tuple = (1, 2, 3)
    print("Immutable Tuple:", immutable_tuple)

    # Trying to change the immutable tuple
    # This will raise an error
    try:
        immutable_tuple[0] = 10
    except TypeError as e:
        print("Error:", e)

if __name__ == "__main__":
    main()
```
