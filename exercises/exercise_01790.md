## Exercise
Write a function to sort a dictionary by its keys or values and return a new dictionary containing the sorted key-value pairs.

## Reference Solution
```python
def sort_dict(input_dict, sort_by='keys', reverse=False):
    """
    Sorts a dictionary by its keys or values and returns a new dictionary containing the sorted key-value pairs.

    Args:
        input_dict (dict): The dictionary to be sorted.
        sort_by (str, optional): The criteria to sort by. Defaults to 'keys'. Can be 'keys' or 'values'.
        reverse (bool, optional): Whether to sort in descending order. Defaults to False.

    Returns:
        dict: A new dictionary containing the sorted key-value pairs.
    """
    if sort_by == 'keys':
        # Sort by keys
        sorted_dict = dict(sorted(input_dict.items(), key=lambda item: item[0], reverse=reverse))
    elif sort_by == 'values':
        # Sort by values
        sorted_dict = dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))
    else:
        raise ValueError("Invalid sort_by parameter. Must be 'keys' or 'values'.")

    return sorted_dict

# Example usage:
if __name__ == "__main__":
    # Create a sample dictionary
    sample_dict = {'c': 3, 'b': 2, 'a': 1}

    # Sort by keys
    sorted_by_keys = sort_dict(sample_dict, sort_by='keys')
    print("Sorted by keys:", sorted_by_keys)

    # Sort by values
    sorted_by_values = sort_dict(sample_dict, sort_by='values')
    print("Sorted by values:", sorted_by_values)

    # Sort by keys in descending order
    sorted_by_keys_desc = sort_dict(sample_dict, sort_by='keys', reverse=True)
    print("Sorted by keys in descending order:", sorted_by_keys_desc)

    # Sort by values in descending order
    sorted_by_values_desc = sort_dict(sample_dict, sort_by='values', reverse=True)
    print("Sorted by values in descending order:", sorted_by_values_desc)
```
