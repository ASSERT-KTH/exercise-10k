## Exercise
Write a program that calculates an employee's salary based on their hourly wage and number of hours worked.

## Reference Solution
```python
def calculate_salary(hourly_wage, hours_worked):
    """
    Calculate an employee's salary based on their hourly wage and number of hours worked.

    Args:
        hourly_wage (float): The employee's hourly wage.
        hours_worked (float): The number of hours the employee worked.

    Returns:
        float: The employee's salary.
    """
    return hourly_wage * hours_worked

def main():
    print("Employee Salary Calculator")
    print("-------------------------------")

    # Get the hourly wage from the user
    while True:
        try:
            hourly_wage = float(input("Enter the employee's hourly wage: $"))
            if hourly_wage < 0:
                print("Hourly wage cannot be negative.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Get the number of hours worked from the user
    while True:
        try:
            hours_worked = float(input("Enter the number of hours the employee worked: "))
            if hours_worked < 0:
                print("Hours worked cannot be negative.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Calculate and display the employee's salary
    salary = calculate_salary(hourly_wage, hours_worked)
    print(f"The employee's salary is: ${salary:.2f}")

if __name__ == "__main__":
    main()
```
