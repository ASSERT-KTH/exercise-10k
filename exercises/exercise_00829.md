## Exercise
Given two strings, check if the first string is a subsequence of the second string and return True or False.

## Reference Solution
```python
def is_subsequence(s1: str, s2: str) -> bool:
    """
    Check if the first string is a subsequence of the second string.

    Args:
        s1 (str): The first string.
        s2 (str): The second string.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """

    # Initialize two pointers, one for each string
    i, j = 0, 0

    # Iterate over the second string
    while j < len(s2):
        # If the current characters in both strings match, move the pointer for the first string
        if i < len(s1) and s1[i] == s2[j]:
            i += 1
        # Move the pointer for the second string
        j += 1

    # If we have iterated over the entire first string, it is a subsequence of the second string
    return i == len(s1)


# Example usage:
if __name__ == "__main__":
    print(is_subsequence("abc", "ahbgdc"))  # True
    print(is_subsequence("axc", "ahbgdc"))  # False
```
