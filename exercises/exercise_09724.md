## Exercise
Write a program that takes two hexadecimal numbers as input and returns their sum.

## Reference Solution
```python
def hex_sum(hex1, hex2):
    """
    Returns the sum of two hexadecimal numbers.

    Args:
        hex1 (str): The first hexadecimal number.
        hex2 (str): The second hexadecimal number.

    Returns:
        str: The sum of the two hexadecimal numbers as a string.
    """
    # Convert the hexadecimal numbers to integers
    num1 = int(hex1, 16)
    num2 = int(hex2, 16)

    # Calculate the sum
    total = num1 + num2

    # Convert the sum back to a hexadecimal string
    hex_total = hex(total)[2:].upper()

    return hex_total

def main():
    # Get the two hexadecimal numbers from the user
    hex1 = input("Enter the first hexadecimal number: ")
    hex2 = input("Enter the second hexadecimal number: ")

    # Validate the input
    try:
        # Check if the input is a valid hexadecimal number
        int(hex1, 16)
        int(hex2, 16)
    except ValueError:
        print("Invalid hexadecimal number. Please try again.")
        return

    # Calculate and print the sum
    print(f"The sum of {hex1} and {hex2} is: {hex_sum(hex1, hex2)}")

if __name__ == "__main__":
    main()
```
