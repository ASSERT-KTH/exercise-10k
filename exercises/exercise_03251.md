## Exercise
Write a program that generates an arithmetic sequence given the first term, common difference, and number of terms.

## Reference Solution
```python
def generate_arithmetic_sequence(first_term, common_difference, num_terms):
    """
    Generates an arithmetic sequence given the first term, common difference, and number of terms.

    Args:
        first_term (float): The first term of the sequence.
        common_difference (float): The difference between each term in the sequence.
        num_terms (int): The number of terms in the sequence.

    Returns:
        list: A list of floats representing the arithmetic sequence.
    """
    sequence = []
    for i in range(num_terms):
        term = first_term + (i * common_difference)
        sequence.append(term)
    return sequence

def main():
    # Example usage:
    first_term = float(input("Enter the first term: "))
    common_difference = float(input("Enter the common difference: "))
    num_terms = int(input("Enter the number of terms: "))

    sequence = generate_arithmetic_sequence(first_term, common_difference, num_terms)
    print("Arithmetic Sequence:")
    print(sequence)

if __name__ == "__main__":
    main()
```
