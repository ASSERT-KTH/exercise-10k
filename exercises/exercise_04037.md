## Exercise
Write a function to convert a Roman numeral to an integer

def roman_to_int(roman_numeral: str) -> int:
    """
    Convert a Roman numeral to an integer.

    Args:
    roman_numeral (str): The Roman numeral to convert.

    Returns:
    int: The integer equivalent of the Roman numeral.

    Raises:
    ValueError: If the input is not a valid Roman numeral.
    """

    # Define a dictionary to map Roman numerals to their integer values
    roman_numerals = {
        'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000
    }

    # Initialize the result variable
    result = 0

    # Iterate over the input string
    for i in range(len(roman_numeral)):
        # Check if the current numeral is valid
        if roman_numeral[i] not in roman_numerals:
            raise ValueError("Invalid Roman numeral")

        # If the current numeral is not the last one and its value is less than the next one,
        # subtract its value from the result
        if i < len(roman_numeral) - 1 and roman_numerals[roman_numeral[i]] < roman_numerals[roman_numeral[i + 1]]:
            result -= roman_numerals[roman_numeral[i]]
        else:
            # Otherwise, add its value to the result
            result += roman_numerals[roman_numeral[i]]

    return result


# Example usage:
if __name__ == "__main__":
    print(roman_to_int("III"))  # Output: 3
    print(roman_to_int("IV"))  # Output: 4
    print(roman_to_int("IX"))  # Output: 9
    print(roman_to_int("LVIII"))  # Output: 58
    print(roman_to_int("MCMXCIV"))  # Output: 1994
