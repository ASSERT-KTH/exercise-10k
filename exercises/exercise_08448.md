## Exercise
Write a program that takes in a 3D shape and calculates its volume using conditional statements.

## Reference Solution
```python
import math

def calculate_volume(shape):
    if shape == "cube":
        side_length = float(input("Enter the side length of the cube: "))
        return side_length ** 3
    elif shape == "sphere":
        radius = float(input("Enter the radius of the sphere: "))
        return (4/3) * math.pi * (radius ** 3)
    elif shape == "rectangular prism":
        length = float(input("Enter the length of the rectangular prism: "))
        width = float(input("Enter the width of the rectangular prism: "))
        height = float(input("Enter the height of the rectangular prism: "))
        return length * width * height
    elif shape == "cylinder":
        radius = float(input("Enter the radius of the cylinder: "))
        height = float(input("Enter the height of the cylinder: "))
        return math.pi * (radius ** 2) * height
    elif shape == "cone":
        radius = float(input("Enter the radius of the cone: "))
        height = float(input("Enter the height of the cone: "))
        return (1/3) * math.pi * (radius ** 2) * height
    else:
        return "Invalid shape"

def main():
    print("Available shapes:")
    print("1. Cube")
    print("2. Sphere")
    print("3. Rectangular Prism")
    print("4. Cylinder")
    print("5. Cone")
    
    choice = int(input("Enter the number of your chosen shape: "))
    
    if choice == 1:
        shape = "cube"
    elif choice == 2:
        shape = "sphere"
    elif choice == 3:
        shape = "rectangular prism"
    elif choice == 4:
        shape = "cylinder"
    elif choice == 5:
        shape = "cone"
    else:
        print("Invalid choice")
        return
    
    volume = calculate_volume(shape)
    print(f"The volume of the {shape} is: {volume}")

if __name__ == "__main__":
    main()
```
