## Exercise
Write a function to aggregate a dictionary by applying a function to each value (e.g., sum, average).

## Reference Solution
```python
from functools import reduce
from operator import add, mul
from statistics import mean

def aggregate_dict(dictionary, func='sum'):
    """
    Aggregate a dictionary by applying a specified function to each value.

    Args:
        dictionary (dict): The dictionary to aggregate.
        func (str, optional): The function to apply. Defaults to 'sum'.
            Options: 'sum', 'average', 'product', 'min', 'max'.

    Returns:
        dict: The aggregated dictionary.
    """

    aggregation_functions = {
        'sum': sum,
        'average': mean,
        'product': lambda x: reduce(mul, x, 1),
        'min': min,
        'max': max
    }

    if func not in aggregation_functions:
        raise ValueError("Invalid function. Options: 'sum', 'average', 'product', 'min', 'max'.")

    aggregated_dict = {}

    for key, value in dictionary.items():
        if isinstance(value, list):
            aggregated_dict[key] = aggregation_functions[func](value)
        else:
            aggregated_dict[key] = value

    return aggregated_dict

# Example usage:
if __name__ == "__main__":
    dictionary = {
        'a': [1, 2, 3],
        'b': [4, 5, 6],
        'c': [7, 8, 9]
    }

    print(aggregate_dict(dictionary, 'sum'))
    print(aggregate_dict(dictionary, 'average'))
    print(aggregate_dict(dictionary, 'product'))
    print(aggregate_dict(dictionary, 'min'))
    print(aggregate_dict(dictionary, 'max'))
```
