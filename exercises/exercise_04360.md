## Exercise
Write a program that checks if a given number is part of a simple sequence (e.g., even numbers, multiples of 3, etc.)

## Reference Solution
```python
def is_even(number):
    """Check if a number is even."""
    return number % 2 == 0

def is_multiple_of_3(number):
    """Check if a number is a multiple of 3."""
    return number % 3 == 0

def is_multiple_of_n(number, n):
    """Check if a number is a multiple of n."""
    return number % n == 0

def main():
    print("Sequence Checker Program")
    print("-------------------------")

    num = int(input("Enter a number: "))

    print(f"Is {num} even? {is_even(num)}")
    print(f"Is {num} a multiple of 3? {is_multiple_of_3(num)}")

    n = int(input("Enter a number to check for multiples (or 0 to skip): "))
    if n != 0:
        print(f"Is {num} a multiple of {n}? {is_multiple_of_n(num, n)}")

if __name__ == "__main__":
    main()
```
