## Exercise
Ask the user for the total bill and the sales tax percentage, and print the total amount to pay.

## Reference Solution
```python
def calculate_total_bill():
    # Ask the user for the total bill
    while True:
        try:
            total_bill = float(input("Enter the total bill: $"))
            if total_bill < 0:
                print("Please enter a non-negative number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Ask the user for the sales tax percentage
    while True:
        try:
            sales_tax_percentage = float(input("Enter the sales tax percentage (%): "))
            if sales_tax_percentage < 0:
                print("Please enter a non-negative number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Calculate the sales tax amount
    sales_tax_amount = (sales_tax_percentage / 100) * total_bill

    # Calculate the total amount to pay
    total_amount_to_pay = total_bill + sales_tax_amount

    # Print the result
    print(f"Total bill: ${total_bill:.2f}")
    print(f"Sales tax ({sales_tax_percentage}%): ${sales_tax_amount:.2f}")
    print(f"Total amount to pay: ${total_amount_to_pay:.2f}")

# Run the program
if __name__ == "__main__":
    calculate_total_bill()
```
