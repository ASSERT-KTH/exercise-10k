## Exercise
Write a program that demonstrates the use of boolean expressions, including logical operators and conditional statements, using variables.

## Reference Solution
```python
# Define variables
x = 5
y = 10
z = 15

# Boolean expressions
print("Boolean expressions:")
print(x > y)  # False
print(x < y)  # True
print(x == y)  # False
print(x != y)  # True

# Logical operators
print("\nLogical operators:")
print(x > y and x < z)  # False
print(x > y or x < z)  # True
print(not x > y)  # True

# Conditional statements
print("\nConditional statements:")
if x > y:
    print("x is greater than y")
else:
    print("x is less than or equal to y")

if x < y and y < z:
    print("x is less than y and y is less than z")
elif x < y or y < z:
    print("x is less than y or y is less than z")
else:
    print("Neither condition is true")

# Example usage
age = 25
has_driver_license = True
has_car_insurance = False

if age >= 18 and has_driver_license and has_car_insurance:
    print("\nYou are eligible to drive.")
else:
    print("\nYou are not eligible to drive.")
```
