## Exercise
Write a program that generates Mad Libs stories using a for loop.

## Reference Solution
```python
# Define the story template
story_template = """
Once upon a time, in a land far, far away, there was a {adjective1} {noun1}.
This {noun1} had a {adjective2} {body_part} and loved to {verb1}.
One day, while {verb2} through the forest, the {noun1} stumbled upon a {adjective3} {noun2}.
The {noun1} was so {adjective4} that it decided to {verb3} the {noun2} home.
From that day on, the {noun1} and {noun2} became the best of {plural_noun}.
"""

def get_word(prompt):
    """Get a word from the user."""
    while True:
        word = input(prompt)
        if word:
            return word
        print("Please enter a word.")

def generate_story():
    """Generate a Mad Libs story."""
    words = {
        "adjective1": get_word("Enter an adjective: "),
        "noun1": get_word("Enter a noun: "),
        "adjective2": get_word("Enter an adjective: "),
        "body_part": get_word("Enter a body part: "),
        "verb1": get_word("Enter a verb: "),
        "verb2": get_word("Enter a verb: "),
        "adjective3": get_word("Enter an adjective: "),
        "noun2": get_word("Enter a noun: "),
        "adjective4": get_word("Enter an adjective: "),
        "verb3": get_word("Enter a verb: "),
        "plural_noun": get_word("Enter a plural noun: ")
    }

    # Use a for loop to fill in the story template
    story = story_template
    for key, value in words.items():
        story = story.replace("{" + key + "}", value)

    print("Here is your Mad Libs story:")
    print(story)

def main():
    """Run the program."""
    generate_story()

if __name__ == "__main__":
    main()
```
