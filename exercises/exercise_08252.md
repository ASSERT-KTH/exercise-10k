## Exercise
Create a program that manages a student's record using if/else statements.

## Reference Solution
```python
class StudentRecord:
    def __init__(self):
        self.records = {}

    def add_record(self):
        name = input("Enter student's name: ")
        grade = input("Enter student's grade: ")
        self.records[name] = grade
        print(f"Record added for {name} with grade {grade}")

    def update_record(self):
        name = input("Enter student's name: ")
        if name in self.records:
            new_grade = input("Enter new grade: ")
            self.records[name] = new_grade
            print(f"Record updated for {name} with grade {new_grade}")
        else:
            print(f"No record found for {name}")

    def delete_record(self):
        name = input("Enter student's name: ")
        if name in self.records:
            del self.records[name]
            print(f"Record deleted for {name}")
        else:
            print(f"No record found for {name}")

    def display_records(self):
        if self.records:
            print("Student Records:")
            for name, grade in self.records.items():
                print(f"{name}: {grade}")
        else:
            print("No records found")

def main():
    record = StudentRecord()
    while True:
        print("\nStudent Record Management Menu:")
        print("1. Add Record")
        print("2. Update Record")
        print("3. Delete Record")
        print("4. Display Records")
        print("5. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            record.add_record()
        elif choice == "2":
            record.update_record()
        elif choice == "3":
            record.delete_record()
        elif choice == "4":
            record.display_records()
        elif choice == "5":
            print("Exiting program")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
