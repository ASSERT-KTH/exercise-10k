## Exercise
Create a program that solves a cubic equation.

import numpy as np

def cubic_equation(a, b, c, d):
    """
    Define the cubic equation.

    Parameters:
    a (float): Coefficient of x^3
    b (float): Coefficient of x^2
    c (float): Coefficient of x
    d (float): Constant term

    Returns:
    function: The cubic equation function
    """
    def equation(x):
        return a * x**3 + b * x**2 + c * x + d
    return equation

def derivative(equation):
    """
    Calculate the derivative of the equation.

    Parameters:
    equation (function): The equation function

    Returns:
    function: The derivative function
    """
    def derivative_func(x):
        h = 0.0001
        return (equation(x + h) - equation(x - h)) / (2.0 * h)
    return derivative_func

def newton_raphson(equation, derivative, initial_guess, tolerance=1e-5, max_iter=1000):
    """
    Solve the equation using the Newton-Raphson method.

    Parameters:
    equation (function): The equation function
    derivative (function): The derivative function
    initial_guess (float): The initial guess for the root
    tolerance (float): The tolerance for convergence
    max_iter (int): The maximum number of iterations

    Returns:
    float: The root of the equation
    """
    x = initial_guess
    for _ in range(max_iter):
        x_next = x - equation(x) / derivative(x)
        if abs(x_next - x) < tolerance:
            return x_next
        x = x_next
    return x

def solve_cubic_equation(a, b, c, d, initial_guess):
    """
    Solve the cubic equation.

    Parameters:
    a (float): Coefficient of x^3
    b (float): Coefficient of x^2
    c (float): Coefficient of x
    d (float): Constant term
    initial_guess (float): The initial guess for the root

    Returns:
    float: The root of the equation
    """
    equation = cubic_equation(a, b, c, d)
    derivative_func = derivative(equation)
    return newton_raphson(equation, derivative_func, initial_guess)

# Example usage

c = 11
d = -6

root = solve_cubic_equation(a, b, c, d, initial_guess)
print(f"The root of the equation is: {root}")
