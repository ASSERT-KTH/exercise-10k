## Exercise
Create a program that performs basic matrix operations (addition, subtraction, multiplication) on 2x2 matrices.

## Reference Solution
```python
import numpy as np

class Matrix:
    def __init__(self, matrix):
        self.matrix = np.array(matrix)

    def add(self, other):
        """Add two matrices"""
        return Matrix(self.matrix + other.matrix)

    def subtract(self, other):
        """Subtract one matrix from another"""
        return Matrix(self.matrix - other.matrix)

    def multiply(self, other):
        """Multiply two matrices"""
        return Matrix(np.dot(self.matrix, other.matrix))

    def __str__(self):
        return str(self.matrix)


def main():
    # Create two 2x2 matrices
    matrix1 = Matrix([[1, 2], [3, 4]])
    matrix2 = Matrix([[5, 6], [7, 8]])

    print("Matrix 1:")
    print(matrix1)
    print("Matrix 2:")
    print(matrix2)

    # Perform addition
    result_add = matrix1.add(matrix2)
    print("\nAddition:")
    print(result_add)

    # Perform subtraction
    result_subtract = matrix1.subtract(matrix2)
    print("\nSubtraction:")
    print(result_subtract)

    # Perform multiplication
    result_multiply = matrix1.multiply(matrix2)
    print("\nMultiplication:")
    print(result_multiply)


if __name__ == "__main__":
    main()
```
