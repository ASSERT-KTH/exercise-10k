## Exercise
Pivot a dictionary from rows to columns.

## Reference Solution
```python
def pivot_dict(row_dict):
    """
    Pivot a dictionary from rows to columns.

    Args:
        row_dict (dict): A dictionary where each key represents a row and its corresponding value is another dictionary with column names as keys and values as cell values.

    Returns:
        dict: A dictionary where each key represents a column and its corresponding value is a list of cell values.
    """
    # Initialize an empty dictionary to store the pivoted data
    col_dict = {}

    # Iterate over each row in the input dictionary
    for row, row_values in row_dict.items():
        # Iterate over each column in the current row
        for col, value in row_values.items():
            # If the column is not already in the pivoted dictionary, add it with an empty list as its value
            if col not in col_dict:
                col_dict[col] = []
            # Append the current cell value to the list of values for the current column
            col_dict[col].append(value)

    return col_dict

# Example usage:
row_dict = {
    'row1': {'col1': 1, 'col2': 2, 'col3': 3},
    'row2': {'col1': 4, 'col2': 5, 'col3': 6},
    'row3': {'col1': 7, 'col2': 8, 'col3': 9}
}

pivoted_dict = pivot_dict(row_dict)
print(pivoted_dict)
```
