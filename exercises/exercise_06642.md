def greet(name: str) -> str:
    """
    This function generates a greeting message for a given name.

    Args:
        name (str): The name of the person to be greeted.

    Returns:
        str: A greeting message including the provided name.

    Raises:
        TypeError: If the input is not a string.

    Example:
        >>> greet("John")
        'Hello, John!'
    """
    if not isinstance(name, str):
        raise TypeError("Input must be a string.")
    return f'Hello, {name}!'

# Test the function
print(greet("John"))  # Outputs: Hello, John!
