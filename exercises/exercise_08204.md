## Exercise
Write a program that prints a hex dump of a binary file using a for loop.

## Reference Solution
```python
def hex_dump(filename):
    """
    Prints a hex dump of a binary file.

    Args:
        filename (str): The name of the binary file.
    """
    try:
        with open(filename, 'rb') as file:
            address = 0
            for chunk in iter(lambda: file.read(16), b''):
                # Print the address
                print(f'{address:08x}: ', end='')

                # Print the hex representation of the chunk
                for byte in chunk:
                    print(f'{byte:02x} ', end='')
                print(' ' * (16 - len(chunk) * 3), end='')

                # Print the ASCII representation of the chunk
                for byte in chunk:
                    if 32 <= byte <= 126:
                        print(chr(byte), end='')
                    else:
                        print('.', end='')

                print()
                address += 16

    except FileNotFoundError:
        print(f'File {filename} not found.')

def main():
    filename = input('Enter the filename: ')
    hex_dump(filename)

if __name__ == '__main__':
    main()
```
