## Exercise
Write a program that takes a string as input and returns the most frequent character in the string.

## Reference Solution
```python
from collections import Counter

def most_frequent_char(input_str):
    """
    Returns the most frequent character in the input string.

    Args:
        input_str (str): The input string.

    Returns:
        str: The most frequent character.
    """
    # Remove whitespace and convert to lowercase
    input_str = input_str.replace(" ", "").lower()
    
    # Count character frequencies
    char_counts = Counter(input_str)
    
    # Get the most common character
    most_common_char = char_counts.most_common(1)
    
    # Return the most frequent character
    return most_common_char[0][0]

def main():
    # Example usage
    input_str = "Hello, World!"
    print("Most frequent character:", most_frequent_char(input_str))

if __name__ == "__main__":
    main()
```
