## Exercise
Create a function that uses a decorator to log its input and output

## Reference Solution
```python
import logging
import functools

# Create a logger
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def log_input_output(func):
    """
    A decorator that logs the input and output of a function.

    Args:
        func: The function to be decorated.

    Returns:
        A wrapper function that logs the input and output of the original function.
    """
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # Log the input
        logger.info(f"Calling {func.__name__} with arguments: {args}, {kwargs}")

        # Call the original function
        result = func(*args, **kwargs)

        # Log the output
        logger.info(f"{func.__name__} returned: {result}")

        return result

    return wrapper

# Example usage:
@log_input_output
def add(a, b):
    """
    A simple function that adds two numbers.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The sum of the two numbers.
    """
    return a + b

# Test the function
if __name__ == "__main__":
    result = add(2, 3)
    print(f"Result: {result}")
```
