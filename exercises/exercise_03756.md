## Exercise
Create a program to find the most frequent element in a given list using a basic algorithm

## Reference Solution
```python
def most_frequent_element(lst):
    """
    Finds the most frequent element in a given list.

    Args:
        lst (list): The input list.

    Returns:
        The most frequent element in the list.
    """
    # Create a dictionary to store the frequency of each element
    frequency_dict = {}

    # Iterate over the list to count the frequency of each element
    for element in lst:
        if element in frequency_dict:
            frequency_dict[element] += 1
        else:
            frequency_dict[element] = 1

    # Find the element with the maximum frequency
    max_frequency = max(frequency_dict.values())
    most_frequent_elements = [element for element, frequency in frequency_dict.items() if frequency == max_frequency]

    # Return the most frequent element(s)
    return most_frequent_elements


def main():
    # Example usage
    lst = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
    most_frequent = most_frequent_element(lst)
    print("Most frequent element(s):", most_frequent)


if __name__ == "__main__":
    main()
```
