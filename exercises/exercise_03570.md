## Exercise
Write a function that reverses a dictionary (swaps keys and values)

## Reference Solution
```python
def reverse_dict(input_dict):
    """
    Reverses a dictionary by swapping keys and values.

    Args:
        input_dict (dict): The dictionary to be reversed.

    Returns:
        dict: A new dictionary with the keys and values swapped.

    Raises:
        ValueError: If the input dictionary has non-unique values.
    """
    # Check for non-unique values
    if len(input_dict.values()) != len(set(input_dict.values())):
        raise ValueError("Input dictionary has non-unique values.")

    # Use dictionary comprehension to swap keys and values
    reversed_dict = {v: k for k, v in input_dict.items()}

    return reversed_dict

# Example usage
if __name__ == "__main__":
    input_dict = {"apple": 1, "banana": 2, "cherry": 3}
    print("Original Dictionary:", input_dict)

    try:
        reversed_dict = reverse_dict(input_dict)
        print("Reversed Dictionary:", reversed_dict)
    except ValueError as e:
        print("Error:", e)
```
