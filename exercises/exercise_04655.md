## Exercise
Create a program that manages a game scoreboard. The user can add or remove points from the scoreboard, and the program displays the current score.

## Reference Solution
```python
class Scoreboard:
    def __init__(self):
        self.score = 0

    def add_points(self, points):
        """Add points to the scoreboard."""
        self.score += points
        print(f"Added {points} points. Current score: {self.score}")

    def remove_points(self, points):
        """Remove points from the scoreboard."""
        if points > self.score:
            print("Not enough points to remove.")
        else:
            self.score -= points
            print(f"Removed {points} points. Current score: {self.score}")

    def display_score(self):
        """Display the current score."""
        print(f"Current score: {self.score}")


def main():
    scoreboard = Scoreboard()
    while True:
        print("\nScoreboard Menu:")
        print("1. Add points")
        print("2. Remove points")
        print("3. Display score")
        print("4. Quit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            points = int(input("Enter points to add: "))
            scoreboard.add_points(points)
        elif choice == "2":
            points = int(input("Enter points to remove: "))
            scoreboard.remove_points(points)
        elif choice == "3":
            scoreboard.display_score()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
