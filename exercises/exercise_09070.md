## Exercise
Create a program that acts as a simple timer and counts down from a given time using if/else statements

## Reference Solution
```python
import time

def countdown(hours, minutes, seconds):
    """
    Counts down from the given time.

    Args:
        hours (int): The number of hours to count down from.
        minutes (int): The number of minutes to count down from.
        seconds (int): The number of seconds to count down from.
    """
    total_seconds = hours * 3600 + minutes * 60 + seconds

    while total_seconds > 0:
        # Calculate the remaining hours, minutes, and seconds
        remaining_hours = total_seconds // 3600
        remaining_minutes = (total_seconds % 3600) // 60
        remaining_seconds = total_seconds % 60

        # Print the remaining time
        print(f"Time remaining: {remaining_hours:02d}:{remaining_minutes:02d}:{remaining_seconds:02d}", end='\r')

        # Wait for 1 second
        time.sleep(1)

        # Decrement the total seconds
        total_seconds -= 1

    print("\nTime's up!")

def main():
    hours = int(input("Enter the number of hours: "))
    minutes = int(input("Enter the number of minutes: "))
    seconds = int(input("Enter the number of seconds: "))

    # Validate the input time
    if hours < 0 or minutes < 0 or seconds < 0:
        print("Error: Time cannot be negative.")
    elif minutes >= 60 or seconds >= 60:
        print("Error: Minutes and seconds must be less than 60.")
    else:
        countdown(hours, minutes, seconds)

if __name__ == "__main__":
    main()
```
