## Exercise
Create a program that takes in two numbers and an arithmetic operation (addition, subtraction, multiplication, division) as input and outputs the result of the operation. The program should handle invalid operations and division by zero.

## Reference Solution
```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
This script performs arithmetic operations on two numbers.
"""

def main():
    numb1 = input("Enter the first number: ")
    numb2 = input("Enter the second number: ")
    oper = input("Enter the operation (+, -, *, /): ")

    # Check if inputs are valid numbers
    if not numb1.replace('.', '', 1).replace('-', '', 1).isdigit():
        print("Error: Invalid input for first number")
        return
    if not numb2.replace('.', '', 1).replace('-', '', 1).isdigit():
        print("Error: Invalid input for second number")
        return

    # Convert inputs to floats
    numb1 = float(numb1)
    numb2 = float(numb2)

    # Perform operation
    if oper == "+":
        print("Result: ", numb1 + numb2)
    elif oper == "-":
        print("Result: ", numb1 - numb2)
    elif oper == "*":
        print("Result: ", numb1 * numb2)
    elif oper == "/":
        if numb2 == 0:
            print("Error: Division by zero is not allowed")
        else:
            print("Result: ", numb1 / numb2)
    else:
        print("Error: Invalid operation")

if __name__ == "__main__":
    main()
```
