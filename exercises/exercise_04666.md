## Exercise
Write a program that takes a list of strings as input and returns the longest common prefix.

## Reference Solution
```python
def longest_common_prefix(strings):
    """
    Returns the longest common prefix of a list of strings.

    Args:
        strings (list): A list of strings.

    Returns:
        str: The longest common prefix.
    """
    if not strings:
        return ""

    shortest_str = min(strings, key=len)

    for i, char in enumerate(shortest_str):
        for other in strings:
            if other[i] != char:
                return shortest_str[:i]

    return shortest_str

# Example usage
def main():
    strings = ["hello", "hello world", "hello python"]
    print(longest_common_prefix(strings))  # Output: "hello"

    strings = ["apple", "application", "apply"]
    print(longest_common_prefix(strings))  # Output: "app"

    strings = ["cat", "dog", "elephant"]
    print(longest_common_prefix(strings))  # Output: ""

if __name__ == "__main__":
    main()
```
