## Exercise
Write a program that takes in a temperature in one unit and converts it to another unit using conditional statements.

## Reference Solution
```python
def convert_temperature(temp, unit_from, unit_to):
    """
    Converts temperature from one unit to another.

    Args:
    - temp (float): The temperature value to be converted.
    - unit_from (str): The unit of the input temperature. Can be 'celsius', 'fahrenheit', or 'kelvin'.
    - unit_to (str): The unit to which the temperature should be converted. Can be 'celsius', 'fahrenheit', or 'kelvin'.

    Returns:
    - float: The converted temperature value.
    """

    # Check if unit_from and unit_to are valid
    valid_units = ['celsius', 'fahrenheit', 'kelvin']
    if unit_from not in valid_units or unit_to not in valid_units:
        raise ValueError("Invalid unit. Please use 'celsius', 'fahrenheit', or 'kelvin'.")

    # Convert temperature to the desired unit
    if unit_from == unit_to:
        return temp  # No conversion needed if units are the same

    # Convert from Celsius
    if unit_from == 'celsius':
        if unit_to == 'fahrenheit':
            return temp * 9/5 + 32
        elif unit_to == 'kelvin':
            return temp + 273.15

    # Convert from Fahrenheit
    elif unit_from == 'fahrenheit':
        if unit_to == 'celsius':
            return (temp - 32) * 5/9
        elif unit_to == 'kelvin':
            return (temp - 32) * 5/9 + 273.15

    # Convert from Kelvin
    elif unit_from == 'kelvin':
        if unit_to == 'celsius':
            return temp - 273.15
        elif unit_to == 'fahrenheit':
            return (temp - 273.15) * 9/5 + 32


def main():
    temp = float(input("Enter the temperature value: "))
    unit_from = input("Enter the unit of the temperature (celsius, fahrenheit, kelvin): ").lower()
    unit_to = input("Enter the unit to which you want to convert (celsius, fahrenheit, kelvin): ").lower()

    try:
        converted_temp = convert_temperature(temp, unit_from, unit_to)
        print(f"{temp} {unit_from} is equal to {converted_temp} {unit_to}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
