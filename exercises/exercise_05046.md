## Exercise
Write a program that evaluates an expression with mixed arithmetic operations.

def evaluate_expression(expression):
    """
    Evaluates an expression with mixed arithmetic operations.

    Args:
    expression (str): The expression to evaluate.

    Returns:
    float: The result of the evaluation.
    """
    # Define operator precedence
    precedence = {
        '+': 1,
        '-': 1,
        '*': 2,
        '/': 2
    }

    # Split the expression into tokens
    tokens = expression.replace('(', ' ( ').replace(')', ' ) ').split()

    # Initialize operator and operand stacks
    operator_stack = []
    operand_stack = []

    # Iterate through tokens
    for token in tokens:
        # Check if token is an operand
        if token.replace('.', '', 1).isdigit():
            operand_stack.append(float(token))
        # Check if token is an operator
        elif token in precedence:
            # Pop operators with higher or equal precedence
            while (operator_stack and
                   operator_stack[-1] in precedence and
                   precedence[operator_stack[-1]] >= precedence[token]):
                apply_operator(operator_stack, operand_stack)
            operator_stack.append(token)
        # Check if token is a left parenthesis
        elif token == '(':
            operator_stack.append(token)
        # Check if token is a right parenthesis
        elif token == ')':
            # Pop operators until left parenthesis is found
            while operator_stack[-1] != '(':
                apply_operator(operator_stack, operand_stack)
            operator_stack.pop()  # Remove left parenthesis

    # Apply remaining operators
    while operator_stack:
        apply_operator(operator_stack, operand_stack)

    # Return the final result
    return operand_stack[0]


def apply_operator(operator_stack, operand_stack):
    """
    Applies the top operator to the top two operands.

    Args:
    operator_stack (list): The stack of operators.
    operand_stack (list): The stack of operands.
    """
    # Pop the top operator and operands
    operator = operator_stack.pop()
    operand2 = operand_stack.pop()
    operand1 = operand_stack.pop()

    # Apply the operator
    if operator == '+':
        result = operand1 + operand2
    elif operator == '-':
        result = operand1 - operand2
    elif operator == '*':
        result = operand1 * operand2
    elif operator == '/':
        result = operand1 / operand2

    # Push the result back onto the operand stack
    operand_stack.append(result)


# Example usage
if __name__ == "__main__":
    expression = "2 + 3 * (4 - 1)"
    result = evaluate_expression(expression)
    print(f"The result of '{expression}' is {result}")
