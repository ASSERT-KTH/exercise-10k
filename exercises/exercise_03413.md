## Exercise
Create a program to merge two given dictionaries into one.

## Reference Solution
```python
def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: The merged dictionary.
    """
    # Use the built-in dictionary update method to merge the dictionaries
    merged_dict = dict1.copy()  # Create a copy of the first dictionary
    merged_dict.update(dict2)  # Update the copy with the key-value pairs from the second dictionary

    return merged_dict

# Example usage
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2}
    dict2 = {"c": 3, "d": 4}

    print("Dictionary 1:", dict1)
    print("Dictionary 2:", dict2)

    merged_dict = merge_dictionaries(dict1, dict2)
    print("Merged Dictionary:", merged_dict)
```
