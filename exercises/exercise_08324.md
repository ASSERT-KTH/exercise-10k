## Exercise
Write a program that takes an array of integers as input, reverses the order of its elements, and then prints the modified array. The program should also calculate and print the average of the elements in the original array.

## Reference Solution
```python
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        int[] v = new int[n];

        for (int i = 0; i < n; i++) {
            v[i] = sc.nextInt();
        }

        int[] v2 = new int[n];

        for (int i = 0; i < n; i++) {
            v2[i] = v[n - i - 1];
        }

        System.out.println("Reversed array: ");
        for (int i = 0; i < n; i++) {
            System.out.print(v2[i] + " ");
        }

        double media = 0;
        for (int i = 0; i < n; i++) {
            media += v[i];
        }
        media /= n;

        System.out.println("\nAverage: " + media);

        sc.close();
    }
}
```
