## Exercise
Write a program to find the shortest path between two nodes in a weighted graph.

## Reference Solution
```python
import sys
import heapq

class Graph:
    def __init__(self):
        self.nodes = {}

    def add_node(self, node):
        if node not in self.nodes:
            self.nodes[node] = {}

    def add_edge(self, node1, node2, weight):
        if node1 in self.nodes and node2 in self.nodes:
            self.nodes[node1][node2] = weight
            self.nodes[node2][node1] = weight  # for undirected graph

    def dijkstra(self, start_node):
        distances = {node: sys.maxsize for node in self.nodes}
        distances[start_node] = 0
        priority_queue = [(0, start_node)]

        while priority_queue:
            current_distance, current_node = heapq.heappop(priority_queue)

            if current_distance > distances[current_node]:
                continue

            for neighbor, weight in self.nodes[current_node].items():
                distance = current_distance + weight

                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    heapq.heappush(priority_queue, (distance, neighbor))

        return distances

    def shortest_path(self, start_node, end_node):
        distances = self.dijkstra(start_node)
        return distances[end_node]

# Example usage
if __name__ == "__main__":
    graph = Graph()

    # Add nodes
    graph.add_node('A')
    graph.add_node('B')
    graph.add_node('C')
    graph.add_node('D')

    # Add edges
    graph.add_edge('A', 'B', 2)
    graph.add_edge('A', 'C', 4)
    graph.add_edge('B', 'C', 1)
    graph.add_edge('B', 'D', 7)
    graph.add_edge('C', 'D', 3)

    # Find shortest path
    start_node = 'A'
    end_node = 'D'
    shortest_distance = graph.shortest_path(start_node, end_node)

    print(f"Shortest distance from {start_node} to {end_node}: {shortest_distance}")
```
