## Exercise
Create a program that uses binary search to find an element in a sorted array. The program should return the index of the element if found, or -1 if not found.

## Reference Solution

#include <stdio.h>
#include <stdlib.h>

int binary_search(int arr[], int n, int x) {
    int l = 0;
    int u = n - 1;

    while (l <= u) {
        int mid = l + (u - l) / 2;

        if (arr[mid] == x) {
            return mid;
        } else if (arr[mid] < x) {
            l = mid + 1;
        } else {
            u = mid - 1;
        }
    }

    return -1;
}

int main() {
    int arr[] = {10, 20, 30, 40, 50, 60, 70, 80, 90};
    int n = sizeof(arr) / sizeof(arr[0]);
    int x = 50;

    int result = binary_search(arr, n, x);

    if (result == -1) {
        printf("Element not found in array\n");
    } else {
        printf("Element found at index %d\n", result);
    }

    return 0;
}
