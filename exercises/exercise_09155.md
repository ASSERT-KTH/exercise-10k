## Exercise
Create a regular expression pattern and match it against a string, then print out the matches.

## Reference Solution
```python
import re

def main():
    # Define the regular expression pattern
    pattern = r"\d{4}-\d{2}-\d{2}"  # Matches dates in YYYY-MM-DD format
    
    # Define the input string
    input_string = "My birthday is 1995-12-25 and my anniversary is 2010-06-15."
    
    # Use the re.findall function to find all matches of the pattern in the string
    matches = re.findall(pattern, input_string)
    
    # Print out the matches
    if matches:
        print("Matches found:")
        for i, match in enumerate(matches):
            print(f"Match {i+1}: {match}")
    else:
        print("No matches found.")

if __name__ == "__main__":
    main()
```
