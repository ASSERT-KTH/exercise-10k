## Exercise
Write a function to check the strength of a given password based on its length, characters, and digits

## Reference Solution
```python
import re

def check_password_strength(password: str) -> dict:
    """
    Checks the strength of a given password based on its length, characters, and digits.

    Args:
    password (str): The password to be checked.

    Returns:
    dict: A dictionary containing the password strength and suggestions for improvement.
    """

    # Initialize the password strength and suggestions
    password_strength = {
        "strength": "weak",
        "suggestions": []
    }

    # Check password length
    if len(password) < 8:
        password_strength["suggestions"].append("Password should be at least 8 characters long.")

    # Check for digits
    if not re.search(r"\d", password):
        password_strength["suggestions"].append("Password should contain at least one digit.")

    # Check for uppercase letters
    if not re.search(r"[A-Z]", password):
        password_strength["suggestions"].append("Password should contain at least one uppercase letter.")

    # Check for lowercase letters
    if not re.search(r"[a-z]", password):
        password_strength["suggestions"].append("Password should contain at least one lowercase letter.")

    # Check for special characters
    if not re.search(r"[!@#$%^&*()_+=-{};:'<>,./?]", password):
        password_strength["suggestions"].append("Password should contain at least one special character.")

    # Evaluate password strength
    if len(password) >= 8 and re.search(r"\d", password) and re.search(r"[A-Z]", password) and re.search(r"[a-z]", password) and re.search(r"[!@#$%^&*()_+=-{};:'<>,./?]", password):
        password_strength["strength"] = "strong"
    elif len(password) >= 8 and (re.search(r"\d", password) or re.search(r"[A-Z]", password) or re.search(r"[a-z]", password) or re.search(r"[!@#$%^&*()_+=-{};:'<>,./?]", password)):
        password_strength["strength"] = "medium"

    return password_strength

def main():
    password = input("Enter a password: ")
    password_strength = check_password_strength(password)
    print(f"Password strength: {password_strength['strength']}")
    if password_strength["suggestions"]:
        print("Suggestions for improvement:")
        for suggestion in password_strength["suggestions"]:
            print(f"- {suggestion}")

if __name__ == "__main__":
    main()
```
