## Exercise
Write a program that demonstrates the use of infinity and NaN values in programming.

## Reference Solution
```python
# Positive and negative infinity
pos_inf = float('inf')
neg_inf = float('-inf')

print("Positive Infinity: ", pos_inf)
print("Negative Infinity: ", neg_inf)

# Comparison with infinity
print("\nComparison with infinity:")
print(pos_inf > 1000000)  # True
print(pos_inf < 1000000)  # False
print(neg_inf > -1000000)  # False
print(neg_inf < -1000000)  # True

# NaN (Not a Number)
nan_val = float('nan')

print("\nNaN (Not a Number): ", nan_val)

# Comparison with NaN
print("\nComparison with NaN:")
print(nan_val > 0)  # False
print(nan_val < 0)  # False
print(nan_val == 0)  # False
print(nan_val != 0)  # True

# Arithmetic operations with NaN
print("\nArithmetic operations with NaN:")
print(nan_val + 1)  # nan
print(nan_val - 1)  # nan
print(nan_val * 1)  # nan
print(nan_val / 1)  # nan

# Arithmetic operations with infinity
print("\nArithmetic operations with infinity:")
print(pos_inf + 1)  # inf
print(neg_inf - 1)  # -inf
print(pos_inf * 1)  # inf
print(neg_inf / 1)  # -inf
```
