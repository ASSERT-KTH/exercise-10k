```python
def main():
    # Create a list
    my_list = [1, 2, 3, 4, 5]

    # Indexing
    print("Indexing:")
    print(my_list[0])  # Output: 1
    print(my_list[4])  # Output: 5

    # Slicing
    print("\nSlicing:")
    print(my_list[0:3])  # Output: [1, 2, 3]
    print(my_list[3:])  # Output: [4, 5]
    print(my_list[::2])  # Output: [1, 3, 5]
    print(my_list[::-1])  # Output: [5, 4, 3, 2, 1]

    # Concatenation
    print("\nConcatenation:")
    list1 = [1, 2, 3]
    list2 = [4, 5, 6]
    print(list1 + list2)  # Output: [1, 2, 3, 4, 5, 6]

    # Extending a list
    print("\nExtending a list:")
    list1 = [1, 2, 3]
    list2 = [4, 5, 6]
    list1.extend(list2)
    print(list1)  # Output: [1, 2, 3, 4, 5, 6]

    # Replication
    print("\nReplication:")
    my_list = [1, 2, 3]
    print(my_list * 2)  # Output: [1, 2, 3, 1, 2, 3]

if __name__ == "__main__":
    main()
```
