## Exercise
Write a program that debounces input data to prevent multiple triggers

## Reference Solution
```python
import time
from typing import Callable

class Debouncer:
    def __init__(self, timeout: float, callback: Callable):
        """
        Initialize the debouncer.

        Args:
            timeout (float): The time in seconds to wait before triggering the callback.
            callback (Callable): The function to call when the debouncer is triggered.
        """
        self.timeout = timeout
        self.callback = callback
        self.last_trigger = 0

    def trigger(self):
        """
        Trigger the debouncer.
        """
        current_time = time.time()
        if current_time - self.last_trigger > self.timeout:
            self.callback()
            self.last_trigger = current_time

# Example usage
def example_callback():
    print("Callback triggered")

debouncer = Debouncer(timeout=0.5, callback=example_callback)

# Simulate input data
for i in range(10):
    debouncer.trigger()
    time.sleep(0.2)
```
