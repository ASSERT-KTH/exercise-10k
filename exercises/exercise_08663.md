## Exercise
Write a program that asks the user for their age and prints out their age in 10 years.

## Reference Solution
```python
def get_user_age():
    """Gets the user's current age"""
    while True:
        try:
            age = int(input("Please enter your current age: "))
            if age < 0:
                print("Age cannot be negative. Please try again.")
            else:
                return age
        except ValueError:
            print("Invalid input. Please enter a whole number.")

def calculate_future_age(current_age):
    """Calculates the user's age in 10 years"""
    return current_age + 10

def main():
    current_age = get_user_age()
    future_age = calculate_future_age(current_age)
    print(f"In 10 years, you will be {future_age} years old.")

if __name__ == "__main__":
    main()
```
