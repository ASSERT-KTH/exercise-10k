## Exercise
Create a program that reserves a hotel room using if/else statements

## Reference Solution
```python
class Hotel:
    def __init__(self):
        self.rooms = {
            "Single": {"price": 100, "available": True},
            "Double": {"price": 150, "available": True},
            "Suite": {"price": 200, "available": True},
        }

    def display_rooms(self):
        print("Available Rooms:")
        for room, details in self.rooms.items():
            print(f"{room}: ${details['price']}, Available: {details['available']}")

    def reserve_room(self, room_type):
        if self.rooms[room_type]["available"]:
            self.rooms[room_type]["available"] = False
            print(f"Room {room_type} reserved successfully.")
        else:
            print(f"Sorry, Room {room_type} is not available.")

    def cancel_reservation(self, room_type):
        if not self.rooms[room_type]["available"]:
            self.rooms[room_type]["available"] = True
            print(f"Reservation for Room {room_type} cancelled successfully.")
        else:
            print(f"Room {room_type} is already available.")


def main():
    hotel = Hotel()
    while True:
        print("\nHotel Room Reservation Menu:")
        print("1. Display Available Rooms")
        print("2. Reserve a Room")
        print("3. Cancel a Reservation")
        print("4. Exit")
        choice = input("Enter your choice: ")

        if choice == "1":
            hotel.display_rooms()
        elif choice == "2":
            hotel.display_rooms()
            room_type = input("Enter the room type to reserve (Single, Double, Suite): ")
            if room_type in hotel.rooms:
                hotel.reserve_room(room_type)
            else:
                print("Invalid room type. Please try again.")
        elif choice == "3":
            hotel.display_rooms()
            room_type = input("Enter the room type to cancel (Single, Double, Suite): ")
            if room_type in hotel.rooms:
                hotel.cancel_reservation(room_type)
            else:
                print("Invalid room type. Please try again.")
        elif choice == "4":
            print("Exiting the program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
