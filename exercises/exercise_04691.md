## Exercise
Write a function that scrambles the letters in each word of a given string.

## Reference Solution
```python
import random

def scramble_string(input_string):
    """
    Scrambles the letters in each word of a given string.

    Args:
        input_string (str): The input string to be scrambled.

    Returns:
        str: The scrambled string.
    """
    words = input_string.split()
    scrambled_words = []

    for word in words:
        # Separate punctuation from the word
        punctuation = ''
        while word and not word[-1].isalpha():
            punctuation = word[-1] + punctuation
            word = word[:-1]

        # Scramble the word
        scrambled_word = list(word)
        random.shuffle(scrambled_word)
        scrambled_word = ''.join(scrambled_word) + punctuation

        scrambled_words.append(scrambled_word)

    return ' '.join(scrambled_words)

# Example usage
if __name__ == "__main__":
    input_string = "Hello, World!"
    print("Input String:", input_string)
    print("Scrambled String:", scramble_string(input_string))
```
