Convert a string representation of an integer to an actual integer and perform arithmetic operations.

def convert_and_perform_arithmetic(string_representation, operation, value):
    """
    Converts a string representation of an integer to an actual integer and performs arithmetic operations.

    Args:
        string_representation (str): A string representation of an integer.
        operation (str): The arithmetic operation to perform. Can be 'add', 'subtract', 'multiply', or 'divide'.
        value (int): The value to use for the arithmetic operation.

    Returns:
        int: The result of the arithmetic operation.
    """

    # Convert the string representation to an integer
    integer_value = int(string_representation)

    # Perform the specified arithmetic operation
    if operation == 'add':
        result = integer_value + value
    elif operation == 'subtract':
        result = integer_value - value
    elif operation == 'multiply':
        result = integer_value * value
    elif operation == 'divide':
        # Check for division by zero
        if value == 0:
            raise ValueError("Cannot divide by zero")
        result = integer_value // value  # Use integer division to get an integer result
    else:
        raise ValueError("Invalid operation")

    return result

# Example usage:
print(convert_and_perform_arithmetic('10', 'add', 5))  # Output: 15
print(convert_and_perform_arithmetic('10', 'subtract', 5))  # Output: 5
print(convert_and_perform_arithmetic('10', 'multiply', 5))  # Output: 50
print(convert_and_perform_arithmetic('10', 'divide', 2))  # Output: 5
