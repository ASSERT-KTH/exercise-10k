Remove and return an arbitrary key-value pair from a dictionary

def remove_and_return_arbitrary_pair(dictionary):
    """
    Remove and return an arbitrary key-value pair from a dictionary.

    Args:
        dictionary (dict): The dictionary from which to remove the pair.

    Returns:
        tuple: A tuple containing the key and value of the removed pair.
    """

        raise ValueError("Dictionary is empty")
