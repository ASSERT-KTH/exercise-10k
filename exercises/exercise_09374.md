## Exercise
Write a program that demonstrates object manipulation techniques.

## Reference Solution
```python
# Define a class for demonstration purposes
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __str__(self):
        return f"{self.name}, {self.age} years old"

# Create a dictionary to store objects
people = {}

def create_object(name, age):
    """
    Create a new Person object and add it to the dictionary.
    """
    person = Person(name, age)
    people[name] = person
    print(f"Created: {person}")

def modify_object(name, attribute, value):
    """
    Modify an existing Person object in the dictionary.
    """
    if name in people:
        if attribute == "name":
            people[name].name = value
        elif attribute == "age":
            people[name].age = value
        print(f"Modified: {people[name]}")
    else:
        print(f"Object '{name}' not found.")

def delete_object(name):
    """
    Delete a Person object from the dictionary.
    """
    if name in people:
        del people[name]
        print(f"Deleted: {name}")
    else:
        print(f"Object '{name}' not found.")

def display_objects():
    """
    Display all Person objects in the dictionary.
    """
    print("Objects:")
    for person in people.values():
        print(person)

def main():
    # Create objects
    create_object("John Doe", 30)
    create_object("Jane Doe", 25)

    # Display objects
    display_objects()

    # Modify an object
    modify_object("John Doe", "age", 31)

    # Display objects again
    display_objects()

    # Delete an object
    delete_object("Jane Doe")

    # Display objects one last time
    display_objects()

if __name__ == "__main__":
    main()
```
