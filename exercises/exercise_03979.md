## Exercise
Write a program that finds the longest increasing subsequence in a given array of integers. The program should take an array of integers as input and output the longest increasing subsequence.

## Reference Solution
```python
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;

    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }

    std::vector<int> d(n, 1);
    std::vector<int> p(n, -1);

    int ans = 0;
    int index = -1;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (a[j] < a[i] && d[j] + 1 > d[i]) {
                d[i] = d[j] + 1;
                p[i] = j;
            }
        }

        if (d[i] > ans) {
            ans = d[i];
            index = i;
        }
    }

    std::vector<int> l;
    while (index != -1) {
        l.push_back(a[index]);
        index = p[index];
    }

    std::reverse(l.begin(), l.end());

    for (int i = 0; i < l.size(); i++) {
        std::cout << l[i] << " ";
    }

    return 0;
}
```
