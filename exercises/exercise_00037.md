## Exercise
Create a function that takes a 2D list as input and returns the determinant of the matrix.

## Reference Solution
```python
import numpy as np

def calculate_determinant(matrix):
    """
    Calculate the determinant of a given 2D list (matrix).
    
    Args:
        matrix (list): A 2D list representing the matrix.
    
    Returns:
        float: The determinant of the matrix.
    
    Raises:
        ValueError: If the input matrix is not square (i.e., it does not have the same number of rows and columns).
    """
    # Check if the input matrix is square
    if len(matrix) != len(matrix[0]):
        raise ValueError("The input matrix must be square.")
    
    # Convert the input list to a NumPy array
    array = np.array(matrix)
    
    # Calculate the determinant using NumPy's linalg.det function
    determinant = np.linalg.det(array)
    
    return determinant

# Example usage
if __name__ == "__main__":
    matrix = [[1, 2], [3, 4]]
    print(calculate_determinant(matrix))
```
