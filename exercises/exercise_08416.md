## Exercise
Write a program that takes a string input from the user and converts it into different data types such as integer, float, and boolean. The program should handle potential errors that may occur during the conversion process.

## Reference Solution
```python
def convert_to_int(input_str):
    """Converts the input string to an integer."""
    try:
        return int(input_str)
    except ValueError:
        print("Error: Unable to convert the string to an integer.")
        return None


def convert_to_float(input_str):
    """Converts the input string to a floating-point number."""
    try:
        return float(input_str)
    except ValueError:
        print("Error: Unable to convert the string to a float.")
        return None


def convert_to_bool(input_str):
    """Converts the input string to a boolean value."""
    input_str = input_str.lower()
    if input_str == "true":
        return True
    elif input_str == "false":
        return False
    else:
        print("Error: Unable to convert the string to a boolean.")
        return None


def main():
    input_str = input("Enter a string: ")

    int_value = convert_to_int(input_str)
    if int_value is not None:
        print(f"Integer value: {int_value}")

    float_value = convert_to_float(input_str)
    if float_value is not None:
        print(f"Float value: {float_value}")

    bool_value = convert_to_bool(input_str)
    if bool_value is not None:
        print(f"Boolean value: {bool_value}")


if __name__ == "__main__":
    main()
```
