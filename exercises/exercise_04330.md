## Exercise
Write a program to check if a given element is in an array.

## Reference Solution
```python
def check_element(array, target):
    """
    Checks if a given element is in an array.

    Args:
        array (list): The input array.
        target (any): The element to be searched.

    Returns:
        bool: True if the element is found, False otherwise.
    """
    return target in array

def main():
    # Example usage
    array = [1, 2, 3, 4, 5]
    target = 3
    result = check_element(array, target)

    if result:
        print(f"The element {target} is found in the array.")
    else:
        print(f"The element {target} is not found in the array.")

if __name__ == "__main__":
    main()
```
