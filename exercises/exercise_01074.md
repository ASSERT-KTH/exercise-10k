## Exercise
Create a program that uses a for loop to analyze the grades of students and calculate the average grade.

## Reference Solution
```python
# Define a dictionary with student names and their grades
student_grades = {
    "John": [85, 90, 78, 92],
    "Alice": [88, 95, 85, 90],
    "Bob": [76, 82, 90, 88],
    "Eve": [92, 98, 89, 85],
}

def calculate_average_grade(student_grades):
    """
    Calculate the average grade for each student.

    Args:
        student_grades (dict): A dictionary with student names and their grades.

    Returns:
        dict: A dictionary with student names and their average grades.
    """
    average_grades = {}

    # Iterate over each student in the dictionary
    for student, grades in student_grades.items():
        # Calculate the sum of grades
        total_grades = sum(grades)
        
        # Calculate the average grade
        average_grade = total_grades / len(grades)

        # Store the average grade in the average_grades dictionary
        average_grades[student] = average_grade

    return average_grades

def main():
    average_grades = calculate_average_grade(student_grades)

    # Print the average grades for each student
    for student, average_grade in average_grades.items():
        print(f"Student: {student}, Average Grade: {average_grade:.2f}")

if __name__ == "__main__":
    main()
```
