## Exercise
Write a function that takes a nested dictionary and returns a flattened dictionary

## Reference Solution
```python
def flatten_dict(nested_dict, parent_key='', separator='_'):
    """
    Flattens a nested dictionary.

    Args:
        nested_dict (dict): The dictionary to be flattened.
        parent_key (str): The key of the parent dictionary (used for recursion).
        separator (str): The separator to use between keys in the flattened dictionary.

    Returns:
        dict: The flattened dictionary.
    """
    flattened_dict = {}

    for key, value in nested_dict.items():
        new_key = f"{parent_key}{separator}{key}" if parent_key else key

        if isinstance(value, dict):
            # If the value is a dictionary, recursively call this function
            flattened_dict.update(flatten_dict(value, new_key, separator))
        else:
            # If the value is not a dictionary, add it to the flattened dictionary
            flattened_dict[new_key] = value

    return flattened_dict

# Example usage
if __name__ == "__main__":
    nested_dict = {
        "a": 1,
        "b": 2,
        "c": {
            "d": 3,
            "e": {
                "f": 4,
                "g": 5
            }
        }
    }

    flattened_dict = flatten_dict(nested_dict)
    print(flattened_dict)
```
