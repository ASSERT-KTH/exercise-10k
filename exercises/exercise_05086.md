## Exercise
Create a program that generates a random number and asks the user to guess it.

## Reference Solution
```python
import random

def random_number_guessing_game():
    # Generate a random number between 1 and 100
    number_to_guess = random.randint(1, 100)
    guess = None
    attempts = 0

    print("Welcome to the random number guessing game!")
    print("I'm thinking of a number between 1 and 100.")

    while guess != number_to_guess:
        # Ask the user for their guess
        guess = input("Take a guess: ")

        # Check if the user wants to quit
        if guess.lower() == "quit":
            print("Okay, the number was {}.".format(number_to_guess))
            break

        # Try to convert the guess to an integer
        try:
            guess = int(guess)
        except ValueError:
            print("That's not a valid number!")
            continue

        # Check if the guess is in the correct range
        if guess < 1 or guess > 100:
            print("Please guess a number between 1 and 100.")
            continue

        # Increment the number of attempts
        attempts += 1

        # Check if the guess is correct
        if guess < number_to_guess:
            print("Too low!")
        elif guess > number_to_guess:
            print("Too high!")

    if guess == number_to_guess:
        print("Congratulations! You found the number in {} attempts.".format(attempts))

if __name__ == "__main__":
    random_number_guessing_game()
```
