Create a function that uses documentation strings to provide a description of its purpose and usage

## Reference Solution

def greet(name: str) -> str:
    """
    A function that generates a personalized greeting message.

    Args:
        name (str): The name of the person to be greeted.

    Returns:
        str: A greeting message with the provided name.

    Examples:
        >>> greet('John')
        'Hello, John!'
        >>> greet('Alice')
        'Hello, Alice!'
    """
    return f'Hello, {name}!'

# Test the function
print(greet('John'))  # Output: Hello, John!
print(greet('Alice'))  # Output: Hello, Alice!
