## Exercise
Write a program that takes a string as input and encrypts it by shifting each character by a certain number of positions.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts the input text by shifting each character by the specified number of positions.

    Args:
        text (str): The input text to be encrypted.
        shift (int): The number of positions to shift each character.

    Returns:
        str: The encrypted text.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character, and add the shift
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character, and add the shift
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        # If the character is not a letter, leave it unchanged
        else:
            result += char

    return result


def main():
    text = input("Enter the text to be encrypted: ")
    shift = int(input("Enter the shift value: "))

    encrypted_text = caesar_cipher(text, shift)

    print("Encrypted text:", encrypted_text)


if __name__ == "__main__":
    main()
```
