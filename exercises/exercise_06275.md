## Exercise
Reverse the key-value pairs in a dictionary

## Reference Solution
```python
def reverse_dict(dictionary):
    """
    Reverses the key-value pairs in a dictionary.

    Args:
    dictionary (dict): The dictionary to be reversed.

    Returns:
    dict: A new dictionary with the key-value pairs reversed.

    Raises:
    ValueError: If the dictionary has duplicate values.
    """
    # Check for duplicate values in the dictionary
    if len(dictionary.values()) != len(set(dictionary.values())):
        raise ValueError("Dictionary has duplicate values")

    # Reverse the key-value pairs using dictionary comprehension
    reversed_dict = {v: k for k, v in dictionary.items()}

    return reversed_dict

# Example usage
original_dict = {"a": 1, "b": 2, "c": 3}
print("Original Dictionary:", original_dict)

try:
    reversed_dict = reverse_dict(original_dict)
    print("Reversed Dictionary:", reversed_dict)
except ValueError as e:
    print("Error:", e)
```
