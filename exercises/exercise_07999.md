## Exercise
Write a program that finds the shortest path between two nodes in a graph.

## Reference Solution
```python
import sys
import heapq

def dijkstra(graph, start, end):
    """
    Dijkstra's algorithm to find the shortest path between two nodes in a graph.

    Args:
    graph (dict): A dictionary representing the graph, where each key is a node and each value is a dictionary of neighboring nodes and their edge weights.
    start (str): The starting node.
    end (str): The ending node.

    Returns:
    tuple: A tuple containing the shortest distance and the shortest path.
    """
    # Initialize the distance to the start node as 0, and all other nodes as infinity
    distances = {node: float('infinity') for node in graph}
    distances[start] = 0

    # Initialize the priority queue with the start node
    priority_queue = [(0, start, [])]

    while priority_queue:
        # Extract the node with the minimum distance from the priority queue
        (dist, current_node, path) = heapq.heappop(priority_queue)

        # If the current node is the end node, return the distance and path
        if current_node == end:
            return dist, path + [end]

        # If the distance to the current node is greater than the known distance, skip it
        if dist > distances[current_node]:
            continue

        # For each neighbor of the current node
        for neighbor, neighbor_dist in graph[current_node].items():
            # Calculate the distance to the neighbor
            old_dist = distances[neighbor]
            new_dist = dist + neighbor_dist

            # If the new distance is less than the old distance, update the distance and priority queue
            if new_dist < old_dist:
                distances[neighbor] = new_dist
                heapq.heappush(priority_queue, (new_dist, neighbor, path + [current_node]))

    # If there is no path to the end node, return None
    return None

# Example usage
graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

start_node = 'A'
end_node = 'D'

result = dijkstra(graph, start_node, end_node)

if result:
    distance, path = result
    print(f"The shortest distance from {start_node} to {end_node} is {distance}")
    print(f"The shortest path from {start_node} to {end_node} is {path}")
else:
    print(f"There is no path from {start_node} to {end_node}")
```
