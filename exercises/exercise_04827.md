## Exercise
Write a program to declare and use a list to store a collection of values.

## Reference Solution
```python
# List Declaration and Usage Program

def main():
    # Declare an empty list
    values = []

    # Add values to the list
    values.append(10)
    values.append(20)
    values.append(30)

    # Print the list
    print("List of Values:", values)

    # Access and print individual values
    print("First Value:", values[0])
    print("Last Value:", values[-1])

    # Update a value in the list
    values[1] = 25
    print("Updated List:", values)

    # Remove a value from the list
    values.remove(30)
    print("List after removal:", values)

if __name__ == "__main__":
    main()
```
