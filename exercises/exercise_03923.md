## Exercise
Write a program that takes two numbers as input and returns the result of the division and the remainder.

## Reference Solution
```python
def divide_numbers(dividend, divisor):
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    quotient = dividend // divisor
    remainder = dividend % divisor
    return quotient, remainder

def main():
    dividend = float(input("Enter the dividend: "))
    divisor = float(input("Enter the divisor: "))
    try:
        quotient, remainder = divide_numbers(dividend, divisor)
        print(f"{dividend} divided by {divisor} is {quotient} with a remainder of {remainder}")
    except ZeroDivisionError as e:
        print(e)

if __name__ == "__main__":
    main()
```
