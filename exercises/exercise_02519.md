## Exercise
Write a program that takes two fractions as input, and returns the result of adding, subtracting, multiplying, or dividing them.

## Reference Solution
```python
from fractions import Fraction

def add_fractions(frac1, frac2):
    """Add two fractions."""
    return frac1 + frac2

def subtract_fractions(frac1, frac2):
    """Subtract two fractions."""
    return frac1 - frac2

def multiply_fractions(frac1, frac2):
    """Multiply two fractions."""
    return frac1 * frac2

def divide_fractions(frac1, frac2):
    """Divide two fractions."""
    if frac2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return frac1 / frac2

def main():
    # Get the two fractions from the user
    frac1 = input("Enter the first fraction (e.g., 1/2): ")
    frac2 = input("Enter the second fraction (e.g., 1/2): ")

    # Convert the input strings to Fraction objects
    frac1 = Fraction(frac1)
    frac2 = Fraction(frac2)

    # Ask the user for the operation to perform
    print("Operations:")
    print("1. Addition")
    print("2. Subtraction")
    print("3. Multiplication")
    print("4. Division")
    operation = int(input("Enter the operation number: "))

    # Perform the operation and print the result
    if operation == 1:
        result = add_fractions(frac1, frac2)
    elif operation == 2:
        result = subtract_fractions(frac1, frac2)
    elif operation == 3:
        result = multiply_fractions(frac1, frac2)
    elif operation == 4:
        try:
            result = divide_fractions(frac1, frac2)
        except ZeroDivisionError as e:
            print(str(e))
            return
    else:
        print("Invalid operation number")
        return

    print(f"The result of the operation is {result}")

if __name__ == "__main__":
    main()
```
