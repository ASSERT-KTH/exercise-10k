## Exercise
Write a program to find the first duplicate in a list of integers.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Finds the first duplicate in a list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The first duplicate found, or None if no duplicates exist.
    """
    seen = set()
    for num in nums:
        if num in seen:
            return num
        seen.add(num)
    return None

# Example usage
def main():
    nums = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(nums)
    if duplicate:
        print(f"The first duplicate is: {duplicate}")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
