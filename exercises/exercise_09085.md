## Exercise
Create a program that uses a dictionary to store student records. The dictionary should have student names as keys and their corresponding grades as values. The program should allow users to add, remove, and update student records, as well as display all records.

## Reference Solution
```python
class StudentRecordSystem:
    def __init__(self):
        """
        Initialize an empty dictionary to store student records.
        """
        self.student_records = {}

    def add_student(self, name, grade):
        """
        Add a new student record to the dictionary.

        Args:
            name (str): The name of the student.
            grade (float): The grade of the student.
        """
        if name not in self.student_records:
            self.student_records[name] = grade
            print(f"Student {name} added successfully.")
        else:
            print(f"Student {name} already exists.")

    def remove_student(self, name):
        """
        Remove a student record from the dictionary.

        Args:
            name (str): The name of the student to remove.
        """
        if name in self.student_records:
            del self.student_records[name]
            print(f"Student {name} removed successfully.")
        else:
            print(f"Student {name} does not exist.")

    def update_student(self, name, new_grade):
        """
        Update the grade of a student in the dictionary.

        Args:
            name (str): The name of the student to update.
            new_grade (float): The new grade of the student.
        """
        if name in self.student_records:
            self.student_records[name] = new_grade
            print(f"Student {name}'s grade updated successfully.")
        else:
            print(f"Student {name} does not exist.")

    def display_records(self):
        """
        Display all student records in the dictionary.
        """
        if not self.student_records:
            print("No student records found.")
        else:
            print("Student Records:")
            for name, grade in self.student_records.items():
                print(f"{name}: {grade}")


def main():
    record_system = StudentRecordSystem()

    while True:
        print("\nStudent Record System Menu:")
        print("1. Add Student")
        print("2. Remove Student")
        print("3. Update Student Grade")
        print("4. Display Student Records")
        print("5. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter student name: ")
            grade = float(input("Enter student grade: "))
            record_system.add_student(name, grade)
        elif choice == "2":
            name = input("Enter student name: ")
            record_system.remove_student(name)
        elif choice == "3":
            name = input("Enter student name: ")
            new_grade = float(input("Enter new grade: "))
            record_system.update_student(name, new_grade)
        elif choice == "4":
            record_system.display_records()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
