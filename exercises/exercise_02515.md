## Exercise
Write a program that takes the scores of several assignments as input, and returns the average score and grade.

## Reference Solution
```python
def calculate_average_score(scores):
    """
    Calculate the average score from a list of scores.

    Args:
        scores (list): A list of assignment scores.

    Returns:
        float: The average score.
    """
    return sum(scores) / len(scores)


def determine_grade(score):
    """
    Determine the grade based on the score.

    Args:
        score (float): The average score.

    Returns:
        str: The corresponding grade.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"


def main():
    # Get the number of assignments
    num_assignments = int(input("Enter the number of assignments: "))

    # Get the scores for each assignment
    scores = []
    for i in range(num_assignments):
        score = float(input(f"Enter the score for assignment {i+1}: "))
        scores.append(score)

    # Calculate the average score
    average_score = calculate_average_score(scores)

    # Determine the grade
    grade = determine_grade(average_score)

    # Print the results
    print(f"Average Score: {average_score:.2f}")
    print(f"Grade: {grade}")


if __name__ == "__main__":
    main()
```
