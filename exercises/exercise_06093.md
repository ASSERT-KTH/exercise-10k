## Exercise
Create a program that uses a for loop to translate a message from English to Pig Latin.

## Reference Solution
```python
def translate_to_pig_latin(message):
    """
    Translates a message from English to Pig Latin.

    Args:
        message (str): The message to be translated.

    Returns:
        str: The translated message.
    """
    vowels = 'aeiou'
    translated_message = []

    # Split the message into words
    words = message.split()

    # Iterate over each word
    for word in words:
        # Check if the word begins with a vowel
        if word[0].lower() in vowels:
            # Add "way" to the end of the word
            translated_word = word + 'way'
        else:
            # Find the index of the first vowel in the word
            for i, char in enumerate(word):
                if char.lower() in vowels:
                    # Move all characters before the first vowel to the end and add "ay"
                    translated_word = word[i:] + word[:i] + 'ay'
                    break
            else:
                # If the word contains no vowels, just add "ay" to the end
                translated_word = word + 'ay'

        translated_message.append(translated_word)

    # Join the translated words back into a message
    translated_message = ' '.join(translated_message)

    return translated_message

# Example usage
def main():
    message = "Hello world this is a test"
    print("Original message:", message)
    print("Translated message:", translate_to_pig_latin(message))

if __name__ == "__main__":
    main()
```
