## Exercise
Given an array of integers, write a program to modify the elements by applying a specific operation.

## Reference Solution
```python
def modify_array(arr, operation, value):
    """
    Modify the elements of an array by applying a specific operation.

    Args:
        arr (list): The input array of integers.
        operation (str): The operation to apply (add, subtract, multiply, divide).
        value (int): The value used in the operation.

    Returns:
        list: The modified array.
    """
    modified_arr = arr.copy()  # Create a copy of the original array

    if operation == "add":
        modified_arr = [x + value for x in modified_arr]
    elif operation == "subtract":
        modified_arr = [x - value for x in modified_arr]
    elif operation == "multiply":
        modified_arr = [x * value for x in modified_arr]
    elif operation == "divide":
        if value != 0:
            modified_arr = [x / value for x in modified_arr]
        else:
            raise ZeroDivisionError("Cannot divide by zero")
    else:
        raise ValueError("Invalid operation")

    return modified_arr


def main():
    # Example usage
    arr = [1, 2, 3, 4, 5]
    print("Original array:", arr)

    # Apply addition operation
    modified_arr = modify_array(arr, "add", 2)
    print("Array after adding 2:", modified_arr)

    # Apply subtraction operation
    modified_arr = modify_array(arr, "subtract", 1)
    print("Array after subtracting 1:", modified_arr)

    # Apply multiplication operation
    modified_arr = modify_array(arr, "multiply", 3)
    print("Array after multiplying by 3:", modified_arr)

    # Apply division operation
    modified_arr = modify_array(arr, "divide", 2)
    print("Array after dividing by 2:", modified_arr)


if __name__ == "__main__":
    main()
```
