## Exercise
Write a program that asks the user a series of questions and prints out their score at the end.

## Reference Solution
```python
# Import the required libraries
import sys

# Define a function to ask questions and calculate the score
def ask_questions(questions):
    """
    Asks the user a series of questions and returns their score.

    Args:
        questions (list): A list of dictionaries containing the question, options, and correct answer.

    Returns:
        int: The user's score.
    """
    score = 0
    for i, question in enumerate(questions):
        print(f"\nQuestion {i+1}: {question['question']}")
        for j, option in enumerate(question['options']):
            print(f"{j+1}. {option}")
        answer = input("Enter the number of your answer: ")
        if question['options'][int(answer) - 1] == question['correct_answer']:
            print("Correct!")
            score += 1
        else:
            print(f"Incorrect. The correct answer was {question['correct_answer']}.")
    return score

# Define the questions
questions = [
    {
        'question': 'What is the capital of France?',
        'options': ['Paris', 'London', 'Berlin', 'Rome'],
        'correct_answer': 'Paris'
    },
    {
        'question': 'What is the largest planet in our solar system?',
        'options': ['Earth', 'Saturn', 'Jupiter', 'Uranus'],
        'correct_answer': 'Jupiter'
    },
    {
        'question': 'Who painted the Mona Lisa?',
        'options': ['Leonardo da Vinci', 'Michelangelo', 'Raphael', 'Caravaggio'],
        'correct_answer': 'Leonardo da Vinci'
    }
]

# Ask the questions and print the score
def main():
    print("Welcome to the quiz!")
    score = ask_questions(questions)
    print(f"\nYour final score is {score} out of {len(questions)}.")

if __name__ == "__main__":
    main()
```
