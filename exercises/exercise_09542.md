## Exercise
Write a program that checks the data type of a given variable and prints out whether it is an integer, float, string, or boolean.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a given variable and prints out whether it is an integer, float, string, or boolean.
    
    Args:
        var: The variable to check the data type of.
    """
    
    # Use the type() function to get the data type of the variable
    var_type = type(var)
    
    # Check the data type and print out the corresponding message
    if var_type == int:
        print(f"The variable {var} is an integer.")
    elif var_type == float:
        print(f"The variable {var} is a float.")
    elif var_type == str:
        print(f"The variable {var} is a string.")
    elif var_type == bool:
        print(f"The variable {var} is a boolean.")
    else:
        print(f"The variable {var} is of type {var_type.__name__}.")

# Example usage
if __name__ == "__main__":
    # Test the function with different data types
    check_data_type(10)       # integer
    check_data_type(3.14)     # float
    check_data_type("hello")  # string
    check_data_type(True)     # boolean
    check_data_type([1, 2])   # list
```
