## Exercise
Write a program that performs bitwise operations on two variables.

## Reference Solution
```python
def bitwise_operations(num1, num2):
    """
    Performs bitwise operations on two variables.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        dict: A dictionary containing the results of the bitwise operations.
    """
    # Perform bitwise AND operation
    and_result = num1 & num2

    # Perform bitwise OR operation
    or_result = num1 | num2

    # Perform bitwise XOR operation
    xor_result = num1 ^ num2

    # Perform bitwise NOT operation
    not_result_num1 = ~num1
    not_result_num2 = ~num2

    # Perform left shift operation
    left_shift_result_num1 = num1 << 1
    left_shift_result_num2 = num2 << 1

    # Perform right shift operation
    right_shift_result_num1 = num1 >> 1
    right_shift_result_num2 = num2 >> 1

    # Return the results as a dictionary
    return {
        "Bitwise AND": and_result,
        "Bitwise OR": or_result,
        "Bitwise XOR": xor_result,
        "Bitwise NOT (num1)": not_result_num1,
        "Bitwise NOT (num2)": not_result_num2,
        "Left Shift (num1)": left_shift_result_num1,
        "Left Shift (num2)": left_shift_result_num2,
        "Right Shift (num1)": right_shift_result_num1,
        "Right Shift (num2)": right_shift_result_num2,
    }

def main():
    # Example usage
    num1 = 5  # Binary: 101
    num2 = 3  # Binary: 011

    results = bitwise_operations(num1, num2)

    print("Bitwise Operations Results:")
    for operation, result in results.items():
        print(f"{operation}: {result} (Binary: {bin(result)[2:]})")

if __name__ == "__main__":
    main()
```
