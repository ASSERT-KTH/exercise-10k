## Exercise
Find the minimum number of coins needed to make change for a given amount

## Reference Solution

def min_coins(coins, amount):
    """
    Find the minimum number of coins needed to make change for a given amount.

    Args:
        coins (list): A list of coin denominations.
        amount (int): The target amount.

    Returns:
        int: The minimum number of coins needed.
    """
    # Initialize the coins table
    coins_table = [float('inf')] * (amount + 1)
    coins_table[0] = 0

    # Iterate over each coin denomination
    for denom in coins:
        # Iterate over each amount from denom to amount
        for i in range(denom, amount + 1):
            # Update the coins table with the minimum of the current value and coins[i - denom] + 1
            coins_table[i] = min(coins_table[i], coins_table[i - denom] + 1)

    # Return the value in the coins table at index amount
    return coins_table[amount] if coins_table[amount] != float('inf') else -1

# Example usage
coins = [1, 2, 5]
amount = 11
print(min_coins(coins, amount))  # Output: 3
