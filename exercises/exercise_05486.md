## Exercise
Write a program to find all prime numbers within a given range.

## Reference Solution
```python
def sieve_of_eratosthenes(start, end):
    """
    Finds all prime numbers within a given range using the Sieve of Eratosthenes algorithm.

    Args:
    start (int): The start of the range (inclusive).
    end (int): The end of the range (inclusive).

    Returns:
    list: A list of prime numbers within the given range.
    """
    # Create a boolean array, prime, of size end+1
    prime = [True] * (end + 1)
    prime[0] = prime[1] = False

    # Iterate from 2 to sqrt(end)
    for p in range(2, int(end ** 0.5) + 1):
        # If p is a prime, mark as composite all the multiples of p
        if prime[p]:
            for i in range(p * p, end + 1, p):
                prime[i] = False

    # Return a list of all prime numbers in the range
    return [p for p in range(start, end + 1) if prime[p]]

def main():
    start = int(input("Enter the start of the range: "))
    end = int(input("Enter the end of the range: "))
    primes = sieve_of_eratosthenes(start, end)
    print("Prime numbers in the range:", primes)

if __name__ == "__main__":
    main()
```
