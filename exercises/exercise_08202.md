## Exercise
Write a program that encodes a string to UTF-8 using a for loop.

## Reference Solution
```python
def encode_utf8(input_string):
    """
    Encode a string to UTF-8.

    Args:
        input_string (str): The input string to be encoded.

    Returns:
        bytes: The UTF-8 encoded bytes.
    """
    encoded_bytes = bytearray()
    for char in input_string:
        # Get the Unicode code point of the character
        code_point = ord(char)

        # Determine the number of bytes required to encode the character
        if code_point <= 0x7F:
            # 1 byte (ASCII)
            encoded_bytes.append(code_point)
        elif code_point <= 0x7FF:
            # 2 bytes
            encoded_bytes.append(0xC0 | (code_point >> 6))
            encoded_bytes.append(0x80 | (code_point & 0x3F))
        elif code_point <= 0xFFFF:
            # 3 bytes
            encoded_bytes.append(0xE0 | (code_point >> 12))
            encoded_bytes.append(0x80 | ((code_point >> 6) & 0x3F))
            encoded_bytes.append(0x80 | (code_point & 0x3F))
        else:
            # 4 bytes
            encoded_bytes.append(0xF0 | (code_point >> 18))
            encoded_bytes.append(0x80 | ((code_point >> 12) & 0x3F))
            encoded_bytes.append(0x80 | ((code_point >> 6) & 0x3F))
            encoded_bytes.append(0x80 | (code_point & 0x3F))

    return bytes(encoded_bytes)


# Example usage
if __name__ == "__main__":
    input_string = "Hello, World!"
    encoded_bytes = encode_utf8(input_string)
    print(f"Encoded bytes: {encoded_bytes}")
```
