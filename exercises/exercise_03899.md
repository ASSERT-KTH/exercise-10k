## Exercise
Develop a program that takes user input for game scores and updates a leaderboard with the top scores.

## Reference Solution
```python
class Leaderboard:
    def __init__(self):
        self.scores = []

    def add_score(self, name, score):
        """Add a score to the leaderboard."""
        self.scores.append((name, score))
        self.scores.sort(key=lambda x: x[1], reverse=True)

    def display_leaderboard(self):
        """Display the current leaderboard."""
        print("\nCurrent Leaderboard:")
        for i, (name, score) in enumerate(self.scores[:10], start=1):
            print(f"{i}. {name} - {score}")

    def get_user_input(self):
        """Get user input for name and score."""
        name = input("Enter player name: ")
        while True:
            try:
                score = int(input("Enter player score: "))
                break
            except ValueError:
                print("Invalid input. Please enter a valid score.")
        return name, score


def main():
    leaderboard = Leaderboard()
    while True:
        print("\n1. Add score to leaderboard")
        print("2. Display leaderboard")
        print("3. Quit")
        choice = input("Choose an option: ")
        if choice == "1":
            name, score = leaderboard.get_user_input()
            leaderboard.add_score(name, score)
        elif choice == "2":
            leaderboard.display_leaderboard()
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
