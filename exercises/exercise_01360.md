## Exercise
Write a program to demonstrate the use of type checking

## Reference Solution
```python
# type_checking_demo.py

def add_numbers(a: int, b: int) -> int:
    """
    Adds two integers.

    Args:
    a (int): The first integer.
    b (int): The second integer.

    Returns:
    int: The sum of a and b.

    Raises:
    TypeError: If either a or b is not an integer.
    """
    # Check if both a and b are integers
    if not isinstance(a, int) or not isinstance(b, int):
        raise TypeError("Both inputs must be integers.")
    return a + b

def main():
    # Test cases
    test_cases = [
        (1, 2),  # Valid integers
        (1.5, 2),  # Invalid: float and integer
        ("1", 2),  # Invalid: string and integer
        (1, 2.5),  # Invalid: integer and float
        ("1", "2")  # Invalid: string and string
    ]

    for a, b in test_cases:
        try:
            result = add_numbers(a, b)
            print(f"Adding {a} and {b} results in {result}")
        except TypeError as e:
            print(f"Error: Adding {a} and {b}: {e}")

if __name__ == "__main__":
    main()
```
