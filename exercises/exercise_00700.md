## Exercise
Write a program to create a scientific calculator that can perform trigonometric functions, exponentiation, and logarithms.

## Reference Solution
```python
import math

def scientific_calculator():
    print("Scientific Calculator")
    print("1. Trigonometric Functions")
    print("2. Exponentiation")
    print("3. Logarithms")
    print("4. Exit")

    while True:
        choice = input("Enter your choice (1/2/3/4): ")

        if choice == "1":
            print("Trigonometric Functions")
            print("1. Sine")
            print("2. Cosine")
            print("3. Tangent")
            print("4. Back")

            trig_choice = input("Enter your choice (1/2/3/4): ")

            if trig_choice == "1":
                angle = float(input("Enter angle in degrees: "))
                print(f"Sine of {angle} degrees is {math.sin(math.radians(angle))}")
            elif trig_choice == "2":
                angle = float(input("Enter angle in degrees: "))
                print(f"Cosine of {angle} degrees is {math.cos(math.radians(angle))}")
            elif trig_choice == "3":
                angle = float(input("Enter angle in degrees: "))
                print(f"Tangent of {angle} degrees is {math.tan(math.radians(angle))}")
            elif trig_choice == "4":
                continue
            else:
                print("Invalid choice. Please try again.")

        elif choice == "2":
            print("Exponentiation")
            base = float(input("Enter base: "))
            exponent = float(input("Enter exponent: "))
            print(f"{base} raised to the power of {exponent} is {math.pow(base, exponent)}")

        elif choice == "3":
            print("Logarithms")
            print("1. Natural Logarithm")
            print("2. Logarithm to the base 10")
            print("3. Logarithm to the base 2")
            print("4. Back")

            log_choice = input("Enter your choice (1/2/3/4): ")

            if log_choice == "1":
                number = float(input("Enter a number: "))
                print(f"Natural logarithm of {number} is {math.log(number)}")
            elif log_choice == "2":
                number = float(input("Enter a number: "))
                print(f"Logarithm to the base 10 of {number} is {math.log10(number)}")
            elif log_choice == "3":
                number = float(input("Enter a number: "))
                print(f"Logarithm to the base 2 of {number} is {math.log2(number)}")
            elif log_choice == "4":
                continue
            else:
                print("Invalid choice. Please try again.")

        elif choice == "4":
            print("Exiting the calculator. Goodbye!")
            break

        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    scientific_calculator()
```
