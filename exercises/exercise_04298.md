## Exercise
Write a function to reduce a dictionary's values to a single value using a given function.

## Reference Solution
```python
def reduce_dict_values(dictionary, func):
    """
    Reduces a dictionary's values to a single value using a given function.

    Args:
        dictionary (dict): The dictionary whose values are to be reduced.
        func (function): The function used to reduce the values.

    Returns:
        The result of applying the given function to all values in the dictionary.
    """

    # First, we check if the dictionary is empty
    if not dictionary:
        return None  # or any other default value

    # We get the dictionary's values
    # This will return a view object that displays a list of all values in the dictionary
    values = dictionary.values()

    # We use the built-in reduce function from the functools module
    # This function applies a rolling computation to sequential pairs of values in a list
    # We import the reduce function from the functools module
    from functools import reduce

    # We apply the given function to all values in the dictionary
    # The reduce function takes two arguments: the function to apply and the iterable to apply it to
    # We pass the given function and the dictionary's values to the reduce function
    result = reduce(func, values)

    # Finally, we return the result
    return result

# Example usage:
# We define a dictionary
my_dict = {'a': 1, 'b': 2, 'c': 3}

# We define a function to add two numbers
def add(a, b):
    return a + b

# We reduce the dictionary's values using the add function
result = reduce_dict_values(my_dict, add)

print(result)  # Outputs: 6
```
