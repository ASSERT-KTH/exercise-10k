## Exercise
Write a function to find the intersection of two dictionaries

## Reference Solution
```python
def dict_intersection(dict1, dict2):
    """
    Finds the intersection of two dictionaries based on their keys.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A new dictionary containing the key-value pairs that are common to both input dictionaries.
    """
    return {key: dict1[key] for key in dict1 if key in dict2 and dict1[key] == dict2[key]}

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 2, 'c': 3, 'd': 4}

intersection = dict_intersection(dict1, dict2)
print(intersection)  # Output: {'b': 2, 'c': 3}
```

```python
def dict_intersection(dict1, dict2):
    """
    Finds the intersection of two dictionaries based on their keys.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A new dictionary containing the key-value pairs that are common to both input dictionaries.
    """
    common_keys = set(dict1.keys()) & set(dict2.keys())
    return {key: dict1[key] for key in common_keys if dict1[key] == dict2[key]}

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 2, 'c': 3, 'd': 4}

intersection = dict_intersection(dict1, dict2)
```
