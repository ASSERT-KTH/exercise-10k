## Exercise
Write a program that checks if a given variable is a float.

## Reference Solution
```python
def is_float(var):
    """
    Checks if a given variable is a float.

    Args:
        var: The variable to check.

    Returns:
        bool: True if the variable is a float, False otherwise.
    """
    return isinstance(var, float)

# Example usage:
if __name__ == "__main__":
    var1 = 5.5
    var2 = "hello"
    var3 = 10

    print(f"Is {var1} a float? {is_float(var1)}")  # Expected output: True
    print(f"Is {var2} a float? {is_float(var2)}")  # Expected output: False
    print(f"Is {var3} a float? {is_float(var3)}")  # Expected output: False
```
