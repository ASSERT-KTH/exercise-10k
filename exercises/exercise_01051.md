Write a program that takes an array and a function as input, and returns a new array containing the results of applying the function to each element of the input array.

#include <stdlib.h>

int main() {
    int n = 0, x, i, y;
    int* result = NULL;

    scanf("%d", &n);
    result = (int*)malloc(n * sizeof(int));

    for (i = 0; i < n; i++) {

        if (x < 0) {
            result[i] = x * -1;
        } else {
            result[i] = x;

    }

    free(result);

}
