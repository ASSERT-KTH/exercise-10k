## Exercise
Write a program that evaluates a given boolean expression

# Define a class to represent a boolean expression
class BooleanExpression:
    def __init__(self, expression):
        self.expression = expression

    def evaluate(self):
        # Remove any whitespace from the expression
        expression = self.expression.replace(" ", "")

        # Evaluate the expression
        return self._evaluate_expression(expression)

    def _evaluate_expression(self, expression):
        # Base case: if the expression is a single character, return its boolean value
        if len(expression) == 1:
            return expression == "T" or expression == "True"

        # Find the operator with the lowest precedence
        operator_index = self._find_operator(expression)

        # If no operator is found, return the boolean value of the expression
        if operator_index == -1:
            return self._evaluate_term(expression)

        # Evaluate the left and right operands
        left_operand = expression[:operator_index]
        right_operand = expression[operator_index + 1:]

        # Evaluate the operator
        operator = expression[operator_index]

        # Recursively evaluate the left and right operands
        left_value = self._evaluate_expression(left_operand) if len(left_operand) > 1 else self._evaluate_term(left_operand)
        right_value = self._evaluate_expression(right_operand) if len(right_operand) > 1 else self._evaluate_term(right_operand)

        # Apply the operator
        if operator == "&":
            return left_value and right_value
        elif operator == "|":
            return left_value or right_value

    def _find_operator(self, expression):
        # Find the operator with the lowest precedence
        for i, char in enumerate(expression):
            if char in ["&", "|"]:
                return i
        return -1

    def _evaluate_term(self, term):
        # Evaluate a term (a single boolean value or a parenthesized expression)
        if term[0] == "(" and term[-1] == ")":
            return self._evaluate_expression(term[1:-1])
        else:
            return term == "T" or term == "True"


# Example usage:
if __name__ == "__main__":
    expression = "T&T|F"
    boolean_expression = BooleanExpression(expression)
    result = boolean_expression.evaluate()
    print(f"The result of '{expression}' is {result}")
