## Exercise
Create a function that searches for an element in a list.

## Reference Solution
```python
def search_list(element, lst):
    """
    Searches for an element in a list.

    Args:
        element (any): The element to search for.
        lst (list): The list to search in.

    Returns:
        int: The index of the element if found, -1 otherwise.
    """
    try:
        # Use the index method to find the element
        return lst.index(element)
    except ValueError:
        # Return -1 if the element is not found
        return -1

# Example usage
def main():
    my_list = [1, 2, 3, 4, 5]
    target = 3
    result = search_list(target, my_list)

    if result != -1:
        print(f"Element {target} found at index {result}.")
    else:
        print(f"Element {target} not found in the list.")

if __name__ == "__main__":
    main()
```
