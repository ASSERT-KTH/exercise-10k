## Exercise
Write a program that takes two strings as input and returns the common characters between the two strings.

## Reference Solution
```python
def common_characters(str1, str2):
    """
    Returns the common characters between two strings.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        list: A list of common characters between the two strings.
    """
    # Convert the input strings to sets of characters
    set1 = set(str1)
    set2 = set(str2)

    # Find the intersection of the two sets (common characters)
    common_chars = set1.intersection(set2)

    # Convert the set of common characters back to a list
    common_chars_list = list(common_chars)

    return common_chars_list

def main():
    # Example usage
    str1 = "hello"
    str2 = "world"
    common_chars = common_characters(str1, str2)
    print(f"Common characters between '{str1}' and '{str2}': {common_chars}")

if __name__ == "__main__":
    main()
```
