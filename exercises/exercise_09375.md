## Exercise
Write a program that demonstrates variable initialization techniques.

## Reference Solution
```python
# Basic Initialization
def basic_initialization():
    """Initialize variables with basic data types"""
    name = "John Doe"  # string
    age = 30  # integer
    height = 1.75  # float
    is_employed = True  # boolean
    print(f"Name: {name}, Age: {age}, Height: {height}, Employed: {is_employed}")

# Type Hints
def type_hints_initialization():
    """Initialize variables with type hints"""
    name: str = "Jane Doe"
    age: int = 25
    height: float = 1.65
    is_employed: bool = False
    print(f"Name: {name}, Age: {age}, Height: {height}, Employed: {is_employed}")

# Default Values
def default_values_initialization():
    """Initialize variables with default values"""
    name = "Unknown"
    age = 0
    height = 0.0
    is_employed = False
    print(f"Name: {name}, Age: {age}, Height: {height}, Employed: {is_employed}")

# Main Function
def main():
    basic_initialization()
    type_hints_initialization()
    default_values_initialization()

if __name__ == "__main__":
    main()
```
