## Exercise
Write a program that takes two arrays of integers as input and returns their concatenation.

## Reference Solution
```python
#include <iostream>

int* concat(int a[], int b[], int sizea, int sizeb) {
  int* c = new int[sizea + sizeb];
  int idx = 0;

  for (int i = 0; i < sizea; i++) {
    c[idx++] = a[i];
  }

  for (int i = 0; i < sizeb; i++) {
    c[idx++] = b[i];
  }

  return c;
}

int main() {
  int a[] = {1, 2, 3, 4, 5};
  int b[] = {6, 7, 8, 9, 10};

  int* c = concat(a, b, 5, 5);

  for (int i = 0; i < 10; i++) {
    std::cout << c[i] << " ";
  }

  delete[] c; // don't forget to free the memory!

  return 0;
}
```
