## Exercise
Create a program that counts the frequency of each word in a given text. Use a for loop to iterate over the words.

## Reference Solution
```python
# Import the required libraries
import re
from collections import Counter

def count_word_frequency(text):
    """
    Counts the frequency of each word in a given text.
    
    Parameters:
    text (str): The input text.
    
    Returns:
    dict: A dictionary with words as keys and their frequencies as values.
    """

    # Convert the text to lower case to ensure the word count is case-insensitive
    text = text.lower()

    # Replace non-alphanumeric characters with spaces
    text = re.sub(r'[^a-z0-9\s]', ' ', text)

    # Split the text into words
    words = text.split()

    # Initialize an empty Counter object to store the word frequencies
    word_frequency = Counter()

    # Use a for loop to iterate over the words and update the word frequency
    for word in words:
        word_frequency[word] += 1

    return word_frequency


def main():
    # Get the input text from the user
    text = input("Enter a text: ")

    # Count the word frequency
    word_frequency = count_word_frequency(text)

    # Print the word frequency
    print("Word Frequency:")
    for word, frequency in word_frequency.items():
        print(f"{word}: {frequency}")


if __name__ == "__main__":
    main()
```
