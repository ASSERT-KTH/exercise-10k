## Exercise
Ask the user for two numbers and print their least common multiple.

## Reference Solution
```python
import math

def calculate_lcm(num1, num2):
    """
    Calculate the least common multiple of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The least common multiple of num1 and num2.
    """
    # Calculate the greatest common divisor
    gcd = math.gcd(num1, num2)
    
    # Calculate the least common multiple
    lcm = abs(num1 * num2) // gcd
    
    return lcm

def main():
    # Ask the user for two numbers
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))

    # Calculate and print the LCM
    lcm = calculate_lcm(num1, num2)
    print(f"The least common multiple of {num1} and {num2} is: {lcm}")

if __name__ == "__main__":
    main()
```
