## Exercise
Write a program that encodes a string to Base64 using a for loop.

## Reference Solution
```python
def base64_encode(input_string):
    # Define the Base64 character set
    base64_chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'

    # Convert input string to bytes
    input_bytes = input_string.encode('utf-8')

    # Initialize the output string
    output_string = ''

    # Loop over the input bytes in chunks of 3
    for i in range(0, len(input_bytes), 3):
        # Get the current chunk of 3 bytes
        chunk = input_bytes[i:i+3]

        # Convert the chunk to a single integer
        integer = 0
        for j, byte in enumerate(chunk):
            integer |= byte << (16 - j * 8)

        # Convert the integer to 4 Base64 characters
        for j in range(4):
            if i + j < len(chunk):
                output_string += base64_chars[(integer >> (18 - j * 6)) & 0x3F]
            else:
                output_string += '='

    return output_string

# Example usage:
input_string = "Hello, World!"
encoded_string = base64_encode(input_string)
print(f"Encoded string: {encoded_string}")
```
