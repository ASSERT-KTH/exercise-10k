## Exercise
Write a function that creates a playlist from a list of songs.

## Reference Solution
```python
class Song:
    def __init__(self, title, artist):
        """
        Initialize a Song object.

        Args:
        title (str): The title of the song.
        artist (str): The artist of the song.
        """
        self.title = title
        self.artist = artist

    def __repr__(self):
        """
        Return a string representation of the Song object.
        """
        return f"{self.title} by {self.artist}"


class Playlist:
    def __init__(self, name, songs=None):
        """
        Initialize a Playlist object.

        Args:
        name (str): The name of the playlist.
        songs (list, optional): A list of Song objects. Defaults to None.
        """
        self.name = name
        self.songs = songs if songs is not None else []

    def add_song(self, song):
        """
        Add a song to the playlist.

        Args:
        song (Song): A Song object to add to the playlist.
        """
        self.songs.append(song)

    def __repr__(self):
        """
        Return a string representation of the Playlist object.
        """
        return f"{self.name}: {self.songs}"


def create_playlist(playlist_name, songs):
    """
    Create a playlist from a list of songs.

    Args:
    playlist_name (str): The name of the playlist.
    songs (list): A list of dictionaries, where each dictionary contains the title and artist of a song.

    Returns:
    Playlist: A Playlist object containing the list of songs.
    """
    playlist = Playlist(playlist_name)
    for song in songs:
        title = song["title"]
        artist = song["artist"]
        playlist.add_song(Song(title, artist))
    return playlist


# Example usage:
songs = [
    {"title": "Song 1", "artist": "Artist 1"},
    {"title": "Song 2", "artist": "Artist 2"},
    {"title": "Song 3", "artist": "Artist 3"}
]

playlist = create_playlist("My Playlist", songs)
print(playlist)
```
