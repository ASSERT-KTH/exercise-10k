## Exercise
Write a program to calculate the compound interest earned on a bank deposit.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, frequency):
    """
    Calculate compound interest.

    Args:
        principal (float): Initial deposit amount.
        rate (float): Annual interest rate (in decimal).
        time (int): Time period in years.
        frequency (int): Compounding frequency per year.

    Returns:
        float: Total amount after compound interest.
    """
    amount = principal * (1 + rate / frequency) ** (frequency * time)
    return amount

def main():
    print("Compound Interest Calculator")
    print("--------------------------------")

    # Get user input
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period in years: "))
    frequency = int(input("Enter the compounding frequency per year: "))

    # Calculate compound interest
    amount = calculate_compound_interest(principal, rate, time, frequency)

    # Print result
    print(f"Total amount after {time} years: ${amount:.2f}")
    print(f"Compound interest earned: ${amount - principal:.2f}")

if __name__ == "__main__":
    main()
```
