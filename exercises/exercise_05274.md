## Exercise
Develop a program to determine the winner of a Tic-Tac-Toe game using conditional statements

def determine_winner(board):
    """
    Determine the winner of the Tic-Tac-Toe game.

    Args:
        board (list): 3x3 matrix representing the game board.

    Returns:
        str: The winner of the game ('X' or 'O'), or 'Tie' if the game is a tie.
    """
    # Check rows for a win
    for row in board:
        if row[0] == row[1] == row[2] != ' ':
            return row[0]

    # Check columns for a win
    for col in range(3):
        if board[0][col] == board[1][col] == board[2][col] != ' ':
            return board[0][col]

    # Check diagonals for a win
    if board[0][0] == board[1][1] == board[2][2] != ' ':
        return board[0][0]
    if board[0][2] == board[1][1] == board[2][0] != ' ':
        return board[0][2]

    # Check for a tie
    for row in board:
        if ' ' in row:
            return None  # Game is not over yet

    return 'Tie'

def print_board(board):
    """
    Print the Tic-Tac-Toe game board.

    Args:
        board (list): 3x3 matrix representing the game board.
    """
    for row in board:
        print(' | '.join(row))
        print('---------')

def main():
    # Example game board
    board = [
        ['X', 'O', ' '],
        [' ', 'X', 'O'],
        ['O', ' ', 'X']
    ]

    print("Game Board:")
    print_board(board)

    winner = determine_winner(board)
    if winner == 'X' or winner == 'O':
        print(f"Player {winner} wins!")
    elif winner == 'Tie':
        print("It's a tie!")
    else:
        print("Game is not over yet.")

if __name__ == "__main__":
    main()
