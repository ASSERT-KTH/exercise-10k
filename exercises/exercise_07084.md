## Exercise
Read an array of numbers from the user and print the maximum of the array

## Reference Solution
```python
def read_array_from_user():
    """Reads an array of numbers from the user."""
    user_input = input("Enter a list of numbers separated by space: ")
    try:
        array = [float(num) for num in user_input.split()]
        return array
    except ValueError:
        print("Invalid input. Please enter numbers only.")
        return read_array_from_user()

def find_max(array):
    """Finds the maximum number in the array."""
    return max(array)

def main():
    array = read_array_from_user()
    max_num = find_max(array)
    print(f"The maximum number in the array is: {max_num}")

if __name__ == "__main__":
    main()
```
