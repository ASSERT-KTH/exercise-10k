## Exercise
Write a program that finds the missing number in a given sequence of consecutive numbers.

## Reference Solution
```python
def find_missing_number(sequence):
    """
    Finds the missing number in a given sequence of consecutive numbers.

    Args:
        sequence (list): A list of consecutive numbers with one number missing.

    Returns:
        int: The missing number in the sequence.
    """
    # Calculate the sum of the sequence if it were complete
    complete_sum = sum(range(min(sequence), max(sequence) + 1))
    
    # Calculate the actual sum of the sequence
    actual_sum = sum(sequence)
    
    # The missing number is the difference between the complete sum and the actual sum
    missing_number = complete_sum - actual_sum
    
    return missing_number

def main():
    # Example usage
    sequence = [1, 2, 3, 5, 6]
    missing_number = find_missing_number(sequence)
    print(f"The missing number in the sequence is: {missing_number}")

if __name__ == "__main__":
    main()
```
