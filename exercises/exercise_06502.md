## Exercise
Write a program to check the data type of a given input.

## Reference Solution
```python
def check_data_type(input_value):
    """
    Checks the data type of a given input.

    Args:
        input_value: The input value to check.

    Returns:
        The data type of the input value.
    """
    return type(input_value)


def main():
    # Test the function with different inputs
    inputs = [
        10,  # integer
        10.5,  # float
        "Hello",  # string
        True,  # boolean
        [1, 2, 3],  # list
        (1, 2, 3),  # tuple
        {"a": 1, "b": 2},  # dictionary
        {1, 2, 3},  # set
    ]

    for input_value in inputs:
        print(f"Input: {input_value}, Data Type: {check_data_type(input_value)}")


if __name__ == "__main__":
    main()
```
