## Exercise
Write a program that demonstrates explicit type conversion between different data types using functions like int(), float(), str().

## Reference Solution
```python
# explicit_type_conversion.py

def main():
    # Integer to Float
    int_to_float = float(10)
    print(f"Integer 10 converted to float: {int_to_float}")

    # Float to Integer
    float_to_int = int(10.5)
    print(f"Float 10.5 converted to integer: {float_to_int}")

    # String to Integer
    str_to_int = int("10")
    print(f"String '10' converted to integer: {str_to_int}")

    # String to Float
    str_to_float = float("10.5")
    print(f"String '10.5' converted to float: {str_to_float}")

    # Integer to String
    int_to_str = str(10)
    print(f"Integer 10 converted to string: {int_to_str}")

    # Float to String
    float_to_str = str(10.5)
    print(f"Float 10.5 converted to string: {float_to_str}")

    # Boolean to Integer
    bool_to_int = int(True)
    print(f"Boolean True converted to integer: {bool_to_int}")

    # Boolean to String
    bool_to_str = str(True)
    print(f"Boolean True converted to string: {bool_to_str}")

if __name__ == "__main__":
    main()
```
