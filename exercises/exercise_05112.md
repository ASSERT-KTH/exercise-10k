Perform explicit type conversion using casting or other mechanisms to change the data type of a variable.

```python
# Explicit type conversion in Python

# Convert a string to an integer
string_value = "123"
integer_value = int(string_value)
print(integer_value)  # Output: 123
print(type(integer_value))  # Output: <class 'int'>

# Convert an integer to a float
integer_value = 123
float_value = float(integer_value)
print(float_value)  # Output: 123.0
print(type(float_value))  # Output: <class 'float'>

# Convert a float to a string
float_value = 123.45
string_value = str(float_value)
print(string_value)  # Output: 123.45
print(type(string_value))  # Output: <class 'str'>
```

```python
// Explicit type conversion in Java

public class Main {
    public static void main(String[] args) {
        // Convert an integer to a double
        int integerValue = 123;
        double doubleValue = (double) integerValue;
        System.out.println(doubleValue);  // Output: 123.0
        System.out.println(doubleValue.getClass().getName());  // Output: java.lang.Double

        // Convert a double to an integer
        double doubleValue2 = 123.45;
        int integerValue2 = (int) doubleValue2;
        System.out.println(integerValue2);  // Output: 123
        System.out.println(integerValue2.getClass().getName());  // Output: java.lang.Integer

        // Convert a string to an integer
        String stringValue = "123";
        int integerValue3 = Integer.parseInt(stringValue);
        System.out.println(integerValue3);  // Output: 123
        System.out.println(integerValue3.getClass().getName());  // Output: java.lang.Integer
    }
}
```

```python
// Explicit type conversion in C++

#include <iostream>

int main() {
    // Convert an integer to a double
    int integerValue = 123;
    double doubleValue = static_cast<double>(integerValue);
    std::cout << doubleValue << std::endl;  // Output: 123.0
    std::cout << typeid(doubleValue).name() << std::endl;  // Output: d (double)

    // Convert a double to an integer
    double doubleValue2 = 123.45;
    int integerValue2 = static_cast<int>(doubleValue2);
    std::cout << integerValue2 << std::endl;  // Output: 123
    std::cout << typeid(integerValue2).name() << std::endl;  // Output: i (int)

    // Convert a string to an integer
    const char* stringValue = "123";
    int integerValue3 = std::stoi(stringValue);
    std::cout << integerValue3 << std::endl;  // Output: 123
    std::cout << typeid(integerValue3).name() << std::endl;  // Output: i (int)

    return 0;
}
```
