## Exercise
Write a program to calculate the simple interest on an investment

## Reference Solution
```python
def calculate_simple_interest(principal, rate, time):
    """
    Calculate the simple interest on an investment.

    Args:
        principal (float): The initial amount of money invested.
        rate (float): The interest rate as a decimal.
        time (int): The time period in years.

    Returns:
        float: The simple interest earned.
    """
    interest = principal * rate * time
    return interest

def main():
    print("Simple Interest Calculator")
    print("---------------------------")

    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))

    # Calculate simple interest
    interest = calculate_simple_interest(principal, rate, time)

    # Print results
    print(f"Simple Interest: ${interest:.2f}")
    print(f"Total Amount: ${principal + interest:.2f}")

if __name__ == "__main__":
    main()
```
