## Exercise
Create a function that calculates the perimeter of a rectangle given its length and width.

## Reference Solution
```python
def calculate_rectangle_perimeter(length: float, width: float) -> float:
    """
    Calculates the perimeter of a rectangle.

    Args:
        length (float): The length of the rectangle.
        width (float): The width of the rectangle.

    Returns:
        float: The perimeter of the rectangle.
    """
    if length <= 0 or width <= 0:
        raise ValueError("Length and width must be greater than zero.")

    perimeter = 2 * (length + width)
    return perimeter

# Example usage
def main():
    length = 5.0
    width = 3.0
    perimeter = calculate_rectangle_perimeter(length, width)
    print(f"The perimeter of the rectangle with length {length} and width {width} is {perimeter}.")

if __name__ == "__main__":
    main()
```
