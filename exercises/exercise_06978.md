## Exercise
Create a function that takes a list of numbers as an argument and returns the longest increasing subsequence.

## Reference Solution
```python
def longest_increasing_subsequence(nums):
    """
    Returns the longest increasing subsequence in a list of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        list: The longest increasing subsequence.
    """

    if not nums:
        return []

    # Initialize lists to store the length of the longest increasing subsequence
    # and the previous element in the subsequence
    lengths = [1] * len(nums)
    prev_elements = [None] * len(nums)

    # Compute the length of the longest increasing subsequence ending at each position
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[i] > nums[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                prev_elements[i] = j

    # Find the index of the last element in the longest increasing subsequence
    max_length_idx = max(range(len(lengths)), key=lambda i: lengths[i])

    # Reconstruct the longest increasing subsequence
    subsequence = []
    while max_length_idx is not None:
        subsequence.append(nums[max_length_idx])
        max_length_idx = prev_elements[max_length_idx]

    # Return the longest increasing subsequence in the correct order
    return subsequence[::-1]


# Example usage:
if __name__ == "__main__":
    nums = [10, 22, 9, 33, 21, 50, 41, 60, 80]
    print("Longest increasing subsequence:", longest_increasing_subsequence(nums))
```
