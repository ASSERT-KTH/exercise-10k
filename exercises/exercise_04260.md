## Exercise
Create a program that helps users plan their travel itinerary. The program should take the destination, departure date, and return date as input, and print the itinerary details.

## Reference Solution
```python
from datetime import datetime

def get_itinerary_details():
    # Get user input
    destination = input("Enter your destination: ")
    departure_date_str = input("Enter your departure date (YYYY-MM-DD): ")
    return_date_str = input("Enter your return date (YYYY-MM-DD): ")

    # Validate dates
    try:
        departure_date = datetime.strptime(departure_date_str, "%Y-%m-%d")
        return_date = datetime.strptime(return_date_str, "%Y-%m-%d")
    except ValueError:
        print("Invalid date format. Please use YYYY-MM-DD.")
        return

    # Print itinerary details
    print("\nItinerary Details:")
    print("--------------------")
    print(f"Destination: {destination}")
    print(f"Departure Date: {departure_date.strftime('%B %d, %Y')}")
    print(f"Return Date: {return_date.strftime('%B %d, %Y')}")

    # Calculate trip duration
    trip_duration = (return_date - departure_date).days
    print(f"Trip Duration: {trip_duration} days")

# Main program
def main():
    print("Travel Itinerary Planner")
    print("------------------------")
    get_itinerary_details()

if __name__ == "__main__":
    main()
```
