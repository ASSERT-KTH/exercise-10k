## Exercise
Write a program that demonstrates type casting in programming.

## Reference Solution
```python
# Type Casting Demonstration Program

def main():
    # Integer to Float
    integer_value = 10
    float_value = float(integer_value)
    print(f"Integer {integer_value} cast to float: {float_value}")

    # Float to Integer
    float_value = 10.5
    integer_value = int(float_value)
    print(f"Float {float_value} cast to integer: {integer_value}")

    # String to Integer
    string_value = "10"
    integer_value = int(string_value)
    print(f"String '{string_value}' cast to integer: {integer_value}")

    # String to Float
    string_value = "10.5"
    float_value = float(string_value)
    print(f"String '{string_value}' cast to float: {float_value}")

    # Integer to String
    integer_value = 10
    string_value = str(integer_value)
    print(f"Integer {integer_value} cast to string: {string_value}")

    # Float to String
    float_value = 10.5
    string_value = str(float_value)
    print(f"Float {float_value} cast to string: {string_value}")

if __name__ == "__main__":
    main()
```
