## Exercise
Write a program to reduce a given dictionary into a single value, such as the sum of all values.

## Reference Solution
```python
from functools import reduce

def reduce_dict(dictionary, reduction_func):
    """
    Reduces a dictionary into a single value.

    Args:
    dictionary (dict): The dictionary to be reduced.
    reduction_func (function): The function used to reduce the dictionary values.

    Returns:
    The reduced value.
    """
    # Check if the dictionary is empty
    if not dictionary:
        return None

    # Get all values from the dictionary
    values = dictionary.values()

    # Use the reduce function to apply the reduction function to all values
    reduced_value = reduce(reduction_func, values)

    return reduced_value

# Example usage
if __name__ == "__main__":
    # Define a dictionary
    example_dict = {"a": 10, "b": 20, "c": 30}

    # Define a reduction function (in this case, addition)
    def add(x, y):
        return x + y

    # Reduce the dictionary
    result = reduce_dict(example_dict, add)

    print("The sum of all values in the dictionary is:", result)
```
