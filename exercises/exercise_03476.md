## Exercise
Write a program that generates a short poem based on user-inputted words and themes.

## Reference Solution
```python
import random

# Define a dictionary of poem templates
poem_templates = {
    "nature": [
        "In the {adjective1} {noun1}, where {plural_noun1} play,",
        "The {adjective2} {noun2} shines bright,",
        "With {adjective3} {noun3} and {plural_noun2} in sight,",
        "The {noun4} whispers secrets of the night."
    ],
    "love": [
        "My {adjective1} {noun1} is {adjective2} and true,",
        "With {adjective3} {noun2} and a {adjective4} heart,",
        "In {adjective5} {noun3}, our love will never part,",
        "Together, our {noun4} will forever shine."
    ],
    "life": [
        "As I walk through {adjective1} {noun1},",
        "I see {plural_noun1} and {plural_noun2} everywhere,",
        "The {adjective2} {noun2} reminds me to be aware,",
        "Of the {noun3} that life has to share."
    ]
}

def generate_poem(theme, words):
    """Generate a poem based on the given theme and words."""
    poem = []
    template = random.choice(poem_templates[theme])
    
    # Split the template into lines
    lines = template.splitlines()
    
    # Replace placeholders with user-inputted words
    for line in lines:
        if "{adjective1}" in line:
            line = line.replace("{adjective1}", words["adjective1"])
        if "{adjective2}" in line:
            line = line.replace("{adjective2}", words["adjective2"])
        if "{adjective3}" in line:
            line = line.replace("{adjective3}", words["adjective3"])
        if "{adjective4}" in line:
            line = line.replace("{adjective4}", words["adjective4"])
        if "{adjective5}" in line:
            line = line.replace("{adjective5}", words["adjective5"])
        if "{noun1}" in line:
            line = line.replace("{noun1}", words["noun1"])
        if "{noun2}" in line:
            line = line.replace("{noun2}", words["noun2"])
        if "{noun3}" in line:
            line = line.replace("{noun3}", words["noun3"])
        if "{noun4}" in line:
            line = line.replace("{noun4}", words["noun4"])
        if "{plural_noun1}" in line:
            line = line.replace("{plural_noun1}", words["plural_noun1"])
        if "{plural_noun2}" in line:
            line = line.replace("{plural_noun2}", words["plural_noun2"])
        poem.append(line)
    
    return "\n".join(poem)

def main():
    print("Welcome to the Poem Generator!")
    theme = input("Choose a theme (nature, love, life): ")
    words = {
        "adjective1": input("Enter an adjective: "),
        "adjective2": input("Enter another adjective: "),
        "adjective3": input("Enter another adjective: "),
        "adjective4": input("Enter another adjective: "),
        "adjective5": input("Enter another adjective: "),
        "noun1": input("Enter a noun: "),
        "noun2": input("Enter another noun: "),
        "noun3": input("Enter another noun: "),
        "noun4": input("Enter another noun: "),
        "plural_noun1": input("Enter a plural noun: "),
        "plural_noun2": input("Enter another plural noun: ")
    }
    
    poem = generate_poem(theme, words)
    print("\nHere is your generated poem:")
    print(poem)

if __name__ == "__main__":
    main()
```
