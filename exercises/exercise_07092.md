## Exercise
Read a sentence, a word to replace, and a replacement word from the user, and replace all occurrences of the word

## Reference Solution
```python
def replace_word(sentence, word_to_replace, replacement_word):
    """
    Replaces all occurrences of a word in a sentence with a replacement word.

    Args:
        sentence (str): The sentence in which to replace the word.
        word_to_replace (str): The word to replace.
        replacement_word (str): The word to replace with.

    Returns:
        str: The sentence with all occurrences of the word replaced.
    """
    # Split the sentence into words
    words = sentence.split()

    # Replace all occurrences of the word
    replaced_words = [replacement_word if word.lower() == word_to_replace.lower() else word for word in words]

    # Join the words back into a sentence
    replaced_sentence = ' '.join(replaced_words)

    return replaced_sentence


def main():
    # Read the sentence, word to replace, and replacement word from the user
    sentence = input("Enter a sentence: ")
    word_to_replace = input("Enter a word to replace: ")
    replacement_word = input("Enter a replacement word: ")

    # Replace all occurrences of the word
    replaced_sentence = replace_word(sentence, word_to_replace, replacement_word)

    # Print the sentence with all occurrences of the word replaced
    print("Replaced sentence: ", replaced_sentence)


if __name__ == "__main__":
    main()
```
