## Exercise
Write a function to validate an ISBN-10 code

## Reference Solution

def validate_isbn10(isbn: str) -> bool:
    """
    Validates an ISBN-10 code.

    Args:
    isbn (str): The ISBN-10 code to validate.

    Returns:
    bool: True if the ISBN-10 code is valid, False otherwise.
    """
    # Remove any hyphens from the ISBN
    isbn = isbn.replace('-', '')

    # Check if the length of the ISBN is 10
    if len(isbn) != 10:
        return False

    # Initialize the sum of the weighted digits
    total = 0

    # Iterate over each character in the ISBN
    for i, char in enumerate(isbn):
        # Check if the character is a digit
        if char.isdigit():
            # Add the weighted digit to the total
            total += int(char) * (10 - i)
        # Check if the character is 'X' and it's the last character
        elif char.upper() == 'X' and i == 9:
            # Add 10 to the total (since 'X' represents 10)
            total += 10
        else:
            # If the character is not a digit or 'X', the ISBN is invalid
            return False

    # Check if the total is divisible by 11
    return total % 11 == 0


# Example usage
if __name__ == "__main__":
    print(validate_isbn10("0-306-40615-2"))  # True
    print(validate_isbn10("0-306-40615-1"))  # False
    print(validate_isbn10("0-306-40615-X"))  # True
    print(validate_isbn10("0-306-40615-3"))  # False
