## Exercise
Write a program that uses an if/else statement to manage a to-do list with reminders based on user input.

## Reference Solution
```python
# Importing the datetime module for date and time functionality
from datetime import datetime

class ToDoList:
    def __init__(self):
        self.tasks = {}

    def add_task(self, task_name, due_date):
        """Add a task to the to-do list."""
        if task_name in self.tasks:
            print("Task already exists. Please choose a different name.")
        else:
            self.tasks[task_name] = due_date
            print(f"Task '{task_name}' added successfully.")

    def view_tasks(self):
        """Display all tasks in the to-do list."""
        if not self.tasks:
            print("No tasks available.")
        else:
            print("To-Do List:")
            for task, due_date in self.tasks.items():
                print(f"{task}: {due_date}")

    def delete_task(self, task_name):
        """Delete a task from the to-do list."""
        if task_name in self.tasks:
            del self.tasks[task_name]
            print(f"Task '{task_name}' deleted successfully.")
        else:
            print("Task not found.")

    def check_reminders(self):
        """Check for tasks with due dates within the next 24 hours."""
        today = datetime.now()
        reminders = []
        for task, due_date in self.tasks.items():
            due_date_obj = datetime.strptime(due_date, "%Y-%m-%d %H:%M")
            time_diff = due_date_obj - today
            if time_diff.days <= 1 and time_diff.seconds >= 0:
                reminders.append((task, due_date))
        if reminders:
            print("Reminders:")
            for task, due_date in reminders:
                print(f"{task}: {due_date}")
        else:
            print("No reminders available.")

def main():
    todo_list = ToDoList()

    while True:
        print("\nTo-Do List Menu:")
        print("1. Add Task")
        print("2. View Tasks")
        print("3. Delete Task")
        print("4. Check Reminders")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            task_name = input("Enter task name: ")
            due_date = input("Enter due date (YYYY-MM-DD HH:MM): ")
            todo_list.add_task(task_name, due_date)
        elif choice == "2":
            todo_list.view_tasks()
        elif choice == "3":
            task_name = input("Enter task name: ")
            todo_list.delete_task(task_name)
        elif choice == "4":
            todo_list.check_reminders()
        elif choice == "5":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
