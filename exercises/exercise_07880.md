def check_loan_eligibility(age, income, credit_score):
    """
    Checks if a person is eligible for a loan based on their age, income, and credit score.

    Args:
        age (int): The age of the person.
        income (float): The income of the person.
        credit_score (int): The credit score of the person.

    Returns:
        bool: True if the person is eligible for a loan, False otherwise.
    """
    # Check if the person is old enough to apply for a loan
    if age >= 21:
        # Check if the person has a sufficient income
        if income >= 50000:
            # Check if the person has a good credit score
            if credit_score >= 600:
                return True
            else:
                print("Sorry, your credit score is too low.")
        else:
            print("Sorry, your income is too low.")
    else:
        print("Sorry, you are too young to apply for a loan.")
    return False

def main():
    # Example usage
    age = 25
    income = 60000
    credit_score = 700
    if check_loan_eligibility(age, income, credit_score):
        print("Congratulations, you are eligible for a loan!")
    else:
        print("Sorry, you are not eligible for a loan.")

if __name__ == "__main__":
    main()
