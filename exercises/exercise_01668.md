## Exercise
Develop a program that takes the total cost and the discount percentage as input and calculates the discounted price using if/else statements

## Reference Solution
```python
def calculate_discounted_price(total_cost, discount_percentage):
    """
    Calculate the discounted price based on the total cost and discount percentage.

    Args:
        total_cost (float): The total cost of the item.
        discount_percentage (float): The discount percentage.

    Returns:
        float: The discounted price.
    """
    if discount_percentage < 0 or discount_percentage > 100:
        print("Invalid discount percentage. Please enter a value between 0 and 100.")
        return None
    elif total_cost < 0:
        print("Invalid total cost. Please enter a non-negative value.")
        return None
    else:
        discount_amount = (discount_percentage / 100) * total_cost
        discounted_price = total_cost - discount_amount
        return discounted_price

def main():
    total_cost = float(input("Enter the total cost: "))
    discount_percentage = float(input("Enter the discount percentage: "))

    discounted_price = calculate_discounted_price(total_cost, discount_percentage)

    if discounted_price is not None:
        print(f"Discounted price: {discounted_price:.2f}")

if __name__ == "__main__":
    main()
```
