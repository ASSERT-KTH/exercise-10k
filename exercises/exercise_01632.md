## Exercise
Develop a program that takes a list and a target element as input and returns the index of the target element in the list.

## Reference Solution
```python
def find_index(lst, target):
    """
    Find the index of a target element in a list.

    Args:
        lst (list): The input list.
        target: The target element to find.

    Returns:
        int: The index of the target element if found, -1 otherwise.
    """
    try:
        return lst.index(target)
    except ValueError:
        return -1

def main():
    # Example usage
    lst = [1, 2, 3, 4, 5]
    target = 3
    index = find_index(lst, target)

    if index != -1:
        print(f"Target element {target} found at index {index}.")
    else:
        print(f"Target element {target} not found in the list.")

if __name__ == "__main__":
    main()
```
