## Exercise
Write a program to validate a password. Use a for loop to iterate over the characters in the password and check if it meets certain conditions (e.g. length, uppercase, lowercase, digits).

## Reference Solution
```python
def validate_password(password):
    """
    Validate a password based on length, uppercase, lowercase, and digits.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Initialize flags
    has_uppercase = False
    has_lowercase = False
    has_digit = False

    # Check length
    if len(password) < 8:
        return False

    # Iterate over characters in the password
    for char in password:
        # Check for uppercase
        if char.isupper():
            has_uppercase = True
        # Check for lowercase
        elif char.islower():
            has_lowercase = True
        # Check for digit
        elif char.isdigit():
            has_digit = True

    # Check if all conditions are met
    return has_uppercase and has_lowercase and has_digit


def main():
    # Example usage
    password = input("Enter a password: ")
    if validate_password(password):
        print("Password is valid.")
    else:
        print("Password is invalid. Please ensure it meets the following conditions:")
        print("* At least 8 characters long")
        print("* Contains at least one uppercase letter")
        print("* Contains at least one lowercase letter")
        print("* Contains at least one digit")


if __name__ == "__main__":
    main()
```
