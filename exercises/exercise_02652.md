## Exercise
Write a function to remove and return a value from a dictionary using the pop() method

## Reference Solution
```python
def remove_and_return(dictionary, key):
    """
    Removes and returns a value from a dictionary using the pop() method.

    Args:
        dictionary (dict): The dictionary from which to remove the value.
        key: The key of the value to be removed.

    Returns:
        The removed value if the key exists in the dictionary. Otherwise, raises a KeyError.
    """
    try:
        # Attempt to remove and return the value
        return dictionary.pop(key)
    except KeyError:
        # Handle the case where the key does not exist in the dictionary
        print(f"The key '{key}' does not exist in the dictionary.")
        return None


# Example usage
if __name__ == "__main__":
    example_dict = {"a": 1, "b": 2, "c": 3}
    print("Original dictionary:", example_dict)

    key_to_remove = "b"
    removed_value = remove_and_return(example_dict, key_to_remove)
    print(f"Removed value for key '{key_to_remove}': {removed_value}")
    print("Dictionary after removal:", example_dict)

    # Attempt to remove a non-existent key
    non_existent_key = "d"
    removed_value = remove_and_return(example_dict, non_existent_key)
    print(f"Removed value for key '{non_existent_key}': {removed_value}")
    print("Dictionary after attempted removal:", example_dict)
```
