## Exercise
Write a program that sorts a given list of numbers in ascending order using for loops.

## Reference Solution
```python
def sort_list(numbers):
    """
    Sorts a given list of numbers in ascending order.

    Args:
        numbers (list): The list of numbers to sort.

    Returns:
        list: The sorted list of numbers.
    """
    # Make a copy of the original list to avoid modifying it in place
    sorted_numbers = numbers[:]

    # Iterate over the list
    for i in range(len(sorted_numbers)):
        # Iterate over the unsorted portion of the list
        for j in range(i + 1, len(sorted_numbers)):
            # If the current number is greater than the next number, swap them
            if sorted_numbers[i] > sorted_numbers[j]:
                sorted_numbers[i], sorted_numbers[j] = sorted_numbers[j], sorted_numbers[i]

    return sorted_numbers


# Example usage
def main():
    numbers = [64, 34, 25, 12, 22, 11, 90]
    print("Original list:", numbers)
    sorted_numbers = sort_list(numbers)
    print("Sorted list:", sorted_numbers)


if __name__ == "__main__":
    main()
```
