## Exercise
Write a program that converts decimal numbers to binary using if/else statements

## Reference Solution
```python
def decimal_to_binary(n):
    """
    Converts a decimal number to binary.

    Args:
        n (int): The decimal number to convert.

    Returns:
        str: The binary representation of the decimal number.
    """
    if n == 0:
        return "0"
    elif n == 1:
        return "1"
    else:
        if n % 2 == 0:
            # If the number is even, divide by 2 and add 0 to the end
            return decimal_to_binary(n // 2) + "0"
        else:
            # If the number is odd, divide by 2 and add 1 to the end
            return decimal_to_binary(n // 2) + "1"

# Example usage:
def main():
    decimal_num = int(input("Enter a decimal number: "))
    binary_num = decimal_to_binary(decimal_num)
    print(f"The binary representation of {decimal_num} is: {binary_num}")

if __name__ == "__main__":
    main()
```
