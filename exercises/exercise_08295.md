## Exercise
Write a program that calculates the roots of a quadratic equation using the quadratic formula.

## Reference Solution
```python
import math
import cmath

# Function to calculate the roots of a quadratic equation
def calculate_roots(a, b, c):
    # Calculate the determinant
    determinant = (b**2) - (4*a*c)

    # Find two solutions
    root1 = (-b-math.sqrt(determinant))/(2*a)
    root2 = (-b+math.sqrt(determinant))/(2*a)

    return root1, root2

# Function to calculate the roots of a quadratic equation with complex roots
def calculate_complex_roots(a, b, c):
    # Calculate the determinant
    determinant = (b**2) - (4*a*c)

    # Find two solutions
    root1 = (-b-cmath.sqrt(determinant))/(2*a)
    root2 = (-b+cmath.sqrt(determinant))/(2*a)

    return root1, root2

# Main function
def main():
    a = float(input("Enter the coefficient of x^2: "))
    b = float(input("Enter the coefficient of x: "))
    c = float(input("Enter the constant term: "))

    determinant = (b**2) - (4*a*c)

    if determinant > 0:
        root1, root2 = calculate_roots(a, b, c)
        print("The roots of the equation are: ", root1, " and ", root2)
    elif determinant == 0:
        root1, root2 = calculate_roots(a, b, c)
        print("The root of the equation is: ", root1)
    else:
        root1, root2 = calculate_complex_roots(a, b, c)
        print("The roots of the equation are: ", root1, " and ", root2)

if __name__ == "__main__":
    main()
```
