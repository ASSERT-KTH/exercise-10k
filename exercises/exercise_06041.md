## Exercise
Create a program that takes in a mathematical operation and two numbers, then performs the operation based on the user's input using conditional statements.

## Reference Solution
```python
# Define a function to perform mathematical operations
def perform_operation(num1, num2, operation):
    """
    Perform a mathematical operation on two numbers.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.
    operation (str): The mathematical operation to perform. Can be '+', '-', '*', '/'.

    Returns:
    float: The result of the operation.
    """
    if operation == '+':
        # If the operation is '+', return the sum of the two numbers
        return num1 + num2
    elif operation == '-':
        # If the operation is '-', return the difference of the two numbers
        return num1 - num2
    elif operation == '*':
        # If the operation is '*', return the product of the two numbers
        return num1 * num2
    elif operation == '/':
        # If the operation is '/', return the quotient of the two numbers
        # Check for division by zero
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero")
    else:
        # If the operation is not recognized, raise a ValueError
        raise ValueError("Invalid operation. Must be '+', '-', '*', or '/'.")

# Define the main function
def main():
    # Get the two numbers and operation from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the mathematical operation (+, -, *, /): ")

    try:
        # Attempt to perform the operation
        result = perform_operation(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except ZeroDivisionError as e:
        # Handle division by zero
        print(f"Error: {e}")
    except ValueError as e:
        # Handle invalid operation
        print(f"Error: {e}")

# Run the main function
if __name__ == "__main__":
    main()
```
