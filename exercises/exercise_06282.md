## Exercise
Given a dictionary, access the value associated with a given key

## Reference Solution
```python
def get_value_by_key(dictionary, key):
    """
    Access the value associated with a given key in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key: The key to look up.

    Returns:
        The value associated with the key, or None if the key is not found.

    Raises:
        KeyError: If the key is not found in the dictionary and no default value is provided.
    """
    try:
        return dictionary[key]
    except KeyError:
        return None

# Example usage:
my_dict = {"name": "John", "age": 30}
key = "name"
value = get_value_by_key(my_dict, key)
print(f"The value associated with '{key}' is: {value}")
```

def get_value_by_key(dictionary, key, default=None):
    """
    Access the value associated with a given key in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key: The key to look up.
        default: The default value to return if the key is not found.

    Returns:
        The value associated with the key, or the default value if the key is not found.
    """
    return dictionary.get(key, default)

# Example usage:
my_dict = {"name": "John", "age": 30}
key = " occupation"
default_value = "Not specified"
value = get_value_by_key(my_dict, key, default_value)
print(f"The value associated with '{key}' is: {value}")
