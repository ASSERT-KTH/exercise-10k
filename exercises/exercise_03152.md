## Exercise
Write a program to find the first non-duplicate character in a string.

## Reference Solution
```python
def first_non_duplicate_char(s):
    """
    Find the first non-duplicate character in a string.

    Args:
        s (str): Input string.

    Returns:
        str: The first non-duplicate character, or None if no such character exists.
    """
    # Create a dictionary to store character frequencies
    char_freq = {}
    for char in s:
        if char in char_freq:
            char_freq[char] += 1
        else:
            char_freq[char] = 1

    # Iterate over the string to find the first non-duplicate character
    for char in s:
        if char_freq[char] == 1:
            return char

    # If no non-duplicate character is found, return None
    return None


# Example usage
def main():
    s = "hello"
    result = first_non_duplicate_char(s)
    if result:
        print(f"The first non-duplicate character is: {result}")
    else:
        print("No non-duplicate character found.")


if __name__ == "__main__":
    main()
```
