## Exercise
Write a program that finds the median of an array of integers.

## Reference Solution
```python
def find_median(arr):
    """
    Calculate the median of an array of integers.

    Args:
        arr (list): A list of integers.

    Returns:
        float: The median of the input array.
    """
    # Sort the array in ascending order
    sorted_arr = sorted(arr)

    # Get the length of the array
    n = len(sorted_arr)

    # If the array has an odd number of elements, return the middle value
    if n % 2 != 0:
        return sorted_arr[n // 2]
    # If the array has an even number of elements, return the average of the two middle values
    else:
        mid1 = sorted_arr[n // 2 - 1]
        mid2 = sorted_arr[n // 2]
        return (mid1 + mid2) / 2

# Example usage
def main():
    arr = [1, 3, 5, 7, 9]
    print("Median of the array:", find_median(arr))

    arr = [1, 3, 5, 7, 9, 11]
    print("Median of the array:", find_median(arr))

if __name__ == "__main__":
    main()
```
