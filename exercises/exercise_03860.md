## Exercise
Write a program that uses a while loop to find the longest common subsequence of two sequences.

## Reference Solution

def longest_common_subsequence(seq1, seq2):
    """
    Find the longest common subsequence of two sequences.

    Args:
    seq1 (str): The first sequence.
    seq2 (str): The second sequence.

    Returns:
    str: The longest common subsequence.
    """
    m, n = len(seq1), len(seq2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # Fill the dp table
    i, j = 1, 1
    while i <= m and j <= n:
        if seq1[i - 1] == seq2[j - 1]:
            dp[i][j] = dp[i - 1][j - 1] + 1
            i += 1
            j += 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            i += 1
        else:
            j += 1

    # Backtrack to find the lcs
    lcs = []
    i, j = m, n
    while i > 0 and j > 0:
        if seq1[i - 1] == seq2[j - 1]:
            lcs.append(seq1[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            i -= 1
        else:
            j -= 1

    # Return the lcs in the correct order
    return ''.join(reversed(lcs))


# Example usage
seq1 = "AGGTAB"
seq2 = "GXTXAYB"
lcs = longest_common_subsequence(seq1, seq2)
print(f"Longest common subsequence: {lcs}")
