## Exercise
Write a program to perform basic mathematical operations (e.g., exponentiation, roots, etc.)

## Reference Solution
```python
import math

def menu():
    print("Mathematical Operations Program")
    print("-------------------------------")
    print("1. Exponentiation")
    print("2. Roots")
    print("3. Logarithms")
    print("4. Trigonometric Functions")
    print("5. Exit")

def exponentiation():
    base = float(input("Enter base: "))
    exponent = float(input("Enter exponent: "))
    result = math.pow(base, exponent)
    print(f"{base}^{exponent} = {result}")

def roots():
    number = float(input("Enter number: "))
    root = float(input("Enter root: "))
    result = math.pow(number, 1/root)
    print(f"{root}th root of {number} = {result}")

def logarithms():
    number = float(input("Enter number: "))
    base = float(input("Enter base: "))
    result = math.log(number, base)
    print(f"log_{base}({number}) = {result}")

def trigonometric_functions():
    print("1. Sine")
    print("2. Cosine")
    print("3. Tangent")
    choice = int(input("Enter choice: "))
    angle = float(input("Enter angle in radians: "))
    if choice == 1:
        result = math.sin(angle)
        print(f"sin({angle}) = {result}")
    elif choice == 2:
        result = math.cos(angle)
        print(f"cos({angle}) = {result}")
    elif choice == 3:
        result = math.tan(angle)
        print(f"tan({angle}) = {result}")

def main():
    while True:
        menu()
        choice = int(input("Enter choice: "))
        if choice == 1:
            exponentiation()
        elif choice == 2:
            roots()
        elif choice == 3:
            logarithms()
        elif choice == 4:
            trigonometric_functions()
        elif choice == 5:
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
