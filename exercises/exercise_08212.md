## Exercise
Create a program that checks a student's grade based on their score using if/else statements

## Reference Solution
```python
def check_grade(score):
    """
    Checks a student's grade based on their score.

    Args:
        score (float): The student's score.

    Returns:
        str: The student's grade.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the student's score
    score = float(input("Enter the student's score: "))

    # Check if the score is valid
    if score < 0 or score > 100:
        print("Invalid score. Please enter a score between 0 and 100.")
    else:
        # Check the student's grade
        grade = check_grade(score)
        print(f"The student's grade is: {grade}")

if __name__ == "__main__":
    main()
```
