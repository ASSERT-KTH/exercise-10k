## Exercise
Develop a program that calculates the square root or cube root of a given number.

## Reference Solution
```python
import math

def calculate_root():
    print("Root Calculator")
    print("1. Square Root")
    print("2. Cube Root")

    choice = input("Enter your choice (1/2): ")

    if choice == "1":
        num = float(input("Enter a number: "))
        if num < 0:
            print("Error: Square root of negative number is not a real number.")
        else:
            print("The square root of", num, "is", math.sqrt(num))
    elif choice == "2":
        num = float(input("Enter a number: "))
        print("The cube root of", num, "is", round(num ** (1. / 3), 3))
    else:
        print("Invalid choice. Please enter 1 or 2.")

def main():
    calculate_root()
    again = input("Do you want to calculate again? (yes/no): ")
    if again.lower() == "yes":
        main()
    else:
        print("Goodbye!")

if __name__ == "__main__":
    main()
```
