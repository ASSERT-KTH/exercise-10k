## Exercise
Write a function that handles exceptions and returns a meaningful error message

## Reference Solution
```python
def divide_numbers(num1, num2):
    """
    Attempts to divide two numbers and returns the result or an error message if an exception occurs.

    Args:
        num1 (float): The dividend.
        num2 (float): The divisor.

    Returns:
        str: The result of the division or an error message.
    """
    try:
        # Attempt to divide num1 by num2
        result = num1 / num2
        return f"The result of {num1} divided by {num2} is {result}"
    except ZeroDivisionError:
        # Handle division by zero
        return "Error: Cannot divide by zero."
    except TypeError:
        # Handle non-numeric input
        return "Error: Both inputs must be numbers."
    except Exception as e:
        # Handle any other unexpected exceptions
        return f"An unexpected error occurred: {str(e)}"

# Example usage
if __name__ == "__main__":
    print(divide_numbers(10, 2))  # Successful division
    print(divide_numbers(10, 0))  # Division by zero
    print(divide_numbers(10, "a"))  # Non-numeric input
```
