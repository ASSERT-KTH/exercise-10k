## Exercise
Develop a program that converts between different color representations (e.g. RGB to HEX), taking the user's input as the color to convert.

## Reference Solution
```python
import re

def rgb_to_hex(rgb):
    """
    Convert RGB to HEX.
    
    Args:
        rgb (tuple): RGB values as a tuple (R, G, B)
    
    Returns:
        str: HEX representation of the color
    """
    return '#{:02x}{:02x}{:02x}'.format(rgb[0], rgb[1], rgb[2])

def hex_to_rgb(hex_color):
    """
    Convert HEX to RGB.
    
    Args:
        hex_color (str): HEX representation of the color
    
    Returns:
        tuple: RGB values as a tuple (R, G, B)
    """
    hex_color = hex_color.lstrip('#')
    return tuple(int(hex_color[i:i+2], 16) for i in (0, 2, 4))

def rgb_to_hsv(rgb):
    """
    Convert RGB to HSV.
    
    Args:
        rgb (tuple): RGB values as a tuple (R, G, B)
    
    Returns:
        tuple: HSV values as a tuple (H, S, V)
    """
    r, g, b = [x / 255.0 for x in rgb]
    mx = max(r, g, b)
    mn = min(r, g, b)
    df = mx - mn
    if mx == mn:
        h = 0
    elif mx == r:
        h = (60 * ((g-b)/df) + 360) % 360
    elif mx == g:
        h = (60 * ((b-r)/df) + 120) % 360
    else:
        h = (60 * ((r-g)/df) + 240) % 360
    if mx == 0:
        s = 0
    else:
        s = df / mx
    v = mx
    return (h, s, v)

def hsv_to_rgb(hsv):
    """
    Convert HSV to RGB.
    
    Args:
        hsv (tuple): HSV values as a tuple (H, S, V)
    
    Returns:
        tuple: RGB values as a tuple (R, G, B)
    """
    h, s, v = hsv
    h = h / 360.0
    i = int(h * 6.)
    f = h * 6. - i
    p = v * (1. - s)
    q = v * (1. - f * s)
    t = v * (1. - (1. - f) * s)
    i %= 6
    if i == 0:
        r, g, b = v, t, p
    elif i == 1:
        r, g, b = q, v, p
    elif i == 2:
        r, g, b = p, v, t
    elif i == 3:
        r, g, b = p, q, v
    elif i == 4:
        r, g, b = t, p, v
    else:
        r, g, b = v, p, q
    return (int(r * 255), int(g * 255), int(b * 255))

def main():
    print("Color Conversion Program")
    print("1. RGB to HEX")
    print("2. HEX to RGB")
    print("3. RGB to HSV")
    print("4. HSV to RGB")
    choice = input("Enter your choice: ")
    
    if choice == "1":
        rgb = input("Enter RGB values (R, G, B): ")
        rgb = tuple(map(int, rgb.split(', ')))
        print("HEX representation: ", rgb_to_hex(rgb))
    elif choice == "2":
        hex_color = input("Enter HEX color: ")
        print("RGB values: ", hex_to_rgb(hex_color))
    elif choice == "3":
        rgb = input("Enter RGB values (R, G, B): ")
        rgb = tuple(map(int, rgb.split(', ')))
        print("HSV values: ", rgb_to_hsv(rgb))
    elif choice == "4":
        hsv = input("Enter HSV values (H, S, V): ")
        hsv = tuple(map(float, hsv.split(', ')))
        print("RGB values: ", hsv_to_rgb(hsv))
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
