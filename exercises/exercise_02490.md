## Exercise
Write a program to generate all permutations of a given list of numbers

## Reference Solution
```python
def generate_permutations(nums):
    """
    Generate all permutations of a given list of numbers.

    Args:
    - nums (list): A list of numbers.

    Returns:
    - list: A list of lists, where each sublist is a permutation of the input list.
    """
    # Base case: if the list is empty or contains only one element, return the list itself
    if len(nums) <= 1:
        return [nums]

    # Initialize an empty list to store the permutations
    permutations = []

    # Iterate over each element in the list
    for i, num in enumerate(nums):
        # Remove the current element from the list
        remaining_nums = nums[:i] + nums[i+1:]

        # Recursively generate permutations of the remaining list
        for perm in generate_permutations(remaining_nums):
            # Add the current element to the beginning of each permutation
            permutations.append([num] + perm)

    # Return the list of permutations
    return permutations


# Example usage:
nums = [1, 2, 3]
permutations = generate_permutations(nums)
for i, perm in enumerate(permutations):
    print(f"Permutation {i+1}: {perm}")
```
