## Exercise
Write a program that takes two complex numbers as input and returns their sum.

## Reference Solution
```python
class ComplexNumber:
    def __init__(self, real, imag):
        """
        Initialize a complex number with real and imaginary parts.
        
        Args:
            real (float): The real part of the complex number.
            imag (float): The imaginary part of the complex number.
        """
        self.real = real
        self.imag = imag

    def __str__(self):
        """
        Return a string representation of the complex number.
        
        Returns:
            str: A string in the format 'a + bi' or 'a - bi' where a is the real part and b is the imaginary part.
        """
        sign = '+' if self.imag >= 0 else '-'
        return f'{self.real} {sign} {abs(self.imag)}i'


def add_complex_numbers(num1, num2):
    """
    Add two complex numbers.
    
    Args:
        num1 (ComplexNumber): The first complex number.
        num2 (ComplexNumber): The second complex number.
    
    Returns:
        ComplexNumber: The sum of the two complex numbers.
    """
    real_sum = num1.real + num2.real
    imag_sum = num1.imag + num2.imag
    return ComplexNumber(real_sum, imag_sum)


# Example usage:
if __name__ == "__main__":
    num1 = ComplexNumber(3, 2)
    num2 = ComplexNumber(1, 7)
    print(f'Complex Number 1: {num1}')
    print(f'Complex Number 2: {num2}')
    sum_num = add_complex_numbers(num1, num2)
    print(f'Sum: {sum_num}')
```
