## Exercise
Write a function to convert ogg files to mp3 using a given conversion tool.

## Reference Solution
```python
from pydub import AudioSegment
import os

def convert_ogg_to_mp3(ogg_file_path, mp3_file_path):
    """
    Converts an ogg file to an mp3 file.

    Args:
        ogg_file_path (str): The path to the ogg file.
        mp3_file_path (str): The path to the output mp3 file.

    Returns:
        None
    """
    # Check if the input file exists
    if not os.path.exists(ogg_file_path):
        raise FileNotFoundError("Input file not found")

    # Check if the output file already exists
    if os.path.exists(mp3_file_path):
        print("Output file already exists. Overwriting...")

    # Convert the ogg file to mp3
    sound = AudioSegment.from_ogg(ogg_file_path)
    sound.export(mp3_file_path, format="mp3")

# Example usage:
if __name__ == "__main__":
    ogg_file_path = "input.ogg"
    mp3_file_path = "output.mp3"
    convert_ogg_to_mp3(ogg_file_path, mp3_file_path)
```
