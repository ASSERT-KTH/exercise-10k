## Exercise
Implement a Vigenere cipher to encrypt and decrypt a given string

def generate_table():
    """
    Generates a Vigenere square table.
    """
    table = []
    for i in range(26):
        row = []
        for j in range(26):
            row.append(chr(((i + j) % 26) + 65))
        table.append(row)
    return table


def vigenere_encrypt(plaintext, keyword):
    """
    Encrypts a given string using the Vigenere cipher.

    Args:
    - plaintext (str): The string to be encrypted.
    - keyword (str): The keyword used for encryption.

    Returns:
    - str: The encrypted string.
    """
    table = generate_table()
    encrypted_text = ""
    keyword_index = 0
    keyword = keyword.upper()

    for char in plaintext:
        if char.isalpha():
            shift = ord(keyword[keyword_index]) - 65
            if char.isupper():
                encrypted_text += table[ord(char) - 65][shift]
            else:
                encrypted_text += table[ord(char.upper()) - 65][shift].lower()
            keyword_index = (keyword_index + 1) % len(keyword)
        else:
            encrypted_text += char

    return encrypted_text


def vigenere_decrypt(ciphertext, keyword):
    """
    Decrypts a given string using the Vigenere cipher.

    Args:
    - ciphertext (str): The string to be decrypted.
    - keyword (str): The keyword used for decryption.

    Returns:
    - str: The decrypted string.
    """
    table = generate_table()
    decrypted_text = ""
    keyword_index = 0
    keyword = keyword.upper()

    for char in ciphertext:
        if char.isalpha():
            shift = ord(keyword[keyword_index]) - 65
            if char.isupper():
                for i, row in enumerate(table):
                    if row[shift] == char:
                        decrypted_text += chr(i + 65)
                        break
            else:
                for i, row in enumerate(table):
                    if row[shift] == char.upper():
                        decrypted_text += chr(i + 65).lower()
                        break
            keyword_index = (keyword_index + 1) % len(keyword)
        else:
            decrypted_text += char

    return decrypted_text

    encrypted_text = vigenere_encrypt(plaintext, keyword)
    decrypted_text = vigenere_decrypt(encrypted_text, keyword)

    print(f"Plaintext: {plaintext}")
    print(f"Encrypted Text: {encrypted_text}")
