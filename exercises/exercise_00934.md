## Exercise
Write a program that creates a simple browser using if/else statement

## Reference Solution
```python
class SimpleBrowser:
    def __init__(self):
        self.history = []
        self.current_url = None

    def navigate(self, url):
        if url:
            self.history.append(self.current_url)
            self.current_url = url
            print(f"Navigated to {url}")
        else:
            print("Invalid URL")

    def go_back(self):
        if self.history:
            self.current_url, self.history[-1] = self.history[-1], None
            print(f"Went back to {self.current_url}")
        else:
            print("Cannot go back")

    def exit_browser(self):
        print("Exiting browser")
        return False

def main():
    browser = SimpleBrowser()
    while True:
        print("\nSimple Browser Menu:")
        print("1. Navigate to URL")
        print("2. Go back")
        print("3. Exit browser")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            url = input("Enter URL: ")
            browser.navigate(url)
        elif choice == "2":
            browser.go_back()
        elif choice == "3":
            if browser.exit_browser():
                continue
            else:
                break
        else:
            print("Invalid choice")

if __name__ == "__main__":
    main()
```
