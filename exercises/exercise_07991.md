## Exercise
Write a program that calculates the intersection of two lists of numbers.

## Reference Solution
```python
def intersection(list1, list2):
    """
    Calculate the intersection of two lists of numbers.

    Args:
        list1 (list): The first list of numbers.
        list2 (list): The second list of numbers.

    Returns:
        list: A new list containing the intersection of list1 and list2.
    """
    return [value for value in list1 if value in list2]

def main():
    # Example usage:
    list1 = [1, 2, 3, 4, 5]
    list2 = [4, 5, 6, 7, 8]
    print("List 1:", list1)
    print("List 2:", list2)
    print("Intersection:", intersection(list1, list2))

if __name__ == "__main__":
    main()
```
