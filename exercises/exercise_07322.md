## Exercise
Write a program that takes input from the user with delimiters and prints out the results.

## Reference Solution
```python
def get_user_input():
    """Get user input with delimiters."""
    user_input = input("Please enter a string with comma and space delimiters (e.g., 'apple, banana; orange'): ")
    return user_input

def parse_input(user_input):
    """Parse the user input and split it based on delimiters."""
    # Split the input based on comma and semicolon delimiters
    split_input = user_input.replace(';', ',').split(',')
    
    # Strip any leading or trailing whitespace from each item
    parsed_input = [item.strip() for item in split_input if item.strip()]
    
    return parsed_input

def print_results(parsed_input):
    """Print the parsed input."""
    print("Parsed Input:")
    for i, item in enumerate(parsed_input, start=1):
        print(f"{i}. {item}")

def main():
    user_input = get_user_input()
    parsed_input = parse_input(user_input)
    print_results(parsed_input)

if __name__ == "__main__":
    main()
```
