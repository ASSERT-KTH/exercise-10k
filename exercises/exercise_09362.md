## Exercise
Write a program that demonstrates the concept of variable scope.

## Reference Solution
```python
# Global variable
global_variable = "I am global"

def outer_function():
    # Enclosing variable
    enclosing_variable = "I am enclosing"

    def inner_function():
        # Local variable
        local_variable = "I am local"
        print("Inside inner_function:")
        print(f"global_variable: {global_variable}")
        print(f"enclosing_variable: {enclosing_variable}")
        print(f"local_variable: {local_variable}")

    inner_function()
    print("\nInside outer_function:")
    print(f"global_variable: {global_variable}")
    print(f"enclosing_variable: {enclosing_variable}")
    # Trying to access local_variable will raise an error
    # print(f"local_variable: {local_variable}")

outer_function()
print("\nOutside functions:")
print(f"global_variable: {global_variable}")
# Trying to access enclosing_variable or local_variable will raise an error
# print(f"enclosing_variable: {enclosing_variable}")
# print(f"local_variable: {local_variable}")
```
