## Exercise
Create a program that uses a dictionary to store student names as keys and their corresponding grades as values. The program should allow users to add, remove, and update student grades. It should also display all student grades and calculate the average grade.

## Reference Solution
```python
# Student Grades Program

# Initialize an empty dictionary to store student grades
student_grades = {}

def add_student_grade():
    """Add a student grade to the dictionary."""
    name = input("Enter student name: ")
    while True:
        try:
            grade = float(input("Enter student grade: "))
            if grade < 0 or grade > 100:
                print("Grade must be between 0 and 100.")
            else:
                student_grades[name] = grade
                print(f"Student {name} added with grade {grade}.")
                break
        except ValueError:
            print("Invalid grade. Please enter a number.")

def remove_student_grade():
    """Remove a student grade from the dictionary."""
    name = input("Enter student name: ")
    if name in student_grades:
        del student_grades[name]
        print(f"Student {name} removed.")
    else:
        print(f"Student {name} not found.")

def update_student_grade():
    """Update a student grade in the dictionary."""
    name = input("Enter student name: ")
    if name in student_grades:
        while True:
            try:
                grade = float(input("Enter new student grade: "))
                if grade < 0 or grade > 100:
                    print("Grade must be between 0 and 100.")
                else:
                    student_grades[name] = grade
                    print(f"Student {name} updated with grade {grade}.")
                    break
            except ValueError:
                print("Invalid grade. Please enter a number.")
    else:
        print(f"Student {name} not found.")

def display_student_grades():
    """Display all student grades."""
    if student_grades:
        print("Student Grades:")
        for name, grade in student_grades.items():
            print(f"{name}: {grade}")
    else:
        print("No student grades available.")

def calculate_average_grade():
    """Calculate the average student grade."""
    if student_grades:
        average_grade = sum(student_grades.values()) / len(student_grades)
        print(f"Average grade: {average_grade:.2f}")
    else:
        print("No student grades available.")

def main():
    while True:
        print("\nStudent Grades Menu:")
        print("1. Add student grade")
        print("2. Remove student grade")
        print("3. Update student grade")
        print("4. Display student grades")
        print("5. Calculate average grade")
        print("6. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            add_student_grade()
        elif choice == "2":
            remove_student_grade()
        elif choice == "3":
            update_student_grade()
        elif choice == "4":
            display_student_grades()
        elif choice == "5":
            calculate_average_grade()
        elif choice == "6":
            print("Exiting program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
