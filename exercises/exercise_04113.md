## Exercise
Write a program that demonstrates the concept of variable reassignment, including the effects on memory and variable scope.

## Reference Solution
```python
# Variable Reassignment Demo
def main():
    # Initial variable assignment
    x = 10
    print(f"Initial value of x: {x}")

    # Reassign the variable
    x = 20
    print(f"Value of x after reassignment: {x}")

    # Function scope
    def inner_function():
        # Local variable assignment
        y = 30
        print(f"Value of y inside inner_function: {y}")

        # Reassign the local variable
        y = 40
        print(f"Value of y inside inner_function after reassignment: {y}")

    inner_function()

    # Try to access the local variable outside the function
    try:
        print(f"Value of y outside inner_function: {y}")
    except NameError:
        print("Error: Variable y is not defined outside inner_function.")

    # Global variable assignment
    global z
    z = 50
    print(f"Value of global variable z: {z}")

    # Reassign the global variable
    z = 60
    print(f"Value of global variable z after reassignment: {z}")

if __name__ == "__main__":
    main()
```
