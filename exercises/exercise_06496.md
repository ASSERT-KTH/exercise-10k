## Exercise
Write a program that asks the user for a series of grades (A, B, C, D, or F) and then calculates and prints out the average grade.

## Reference Solution
```python
def get_grade_point(grade):
    """Converts a letter grade to a grade point."""
    grade_points = {
        'A': 4.0,
        'B': 3.0,
        'C': 2.0,
        'D': 1.0,
        'F': 0.0
    }
    return grade_points.get(grade.upper(), None)

def calculate_average_grade(grades):
    """Calculates the average grade."""
    total_grade_points = sum(grades)
    return total_grade_points / len(grades)

def main():
    print("Grade Average Calculator")
    print("-------------------------")

    grades = []
    while True:
        grade = input("Enter a grade (A, B, C, D, or F), or 'q' to quit: ")
        if grade.lower() == 'q':
            break
        grade_point = get_grade_point(grade)
        if grade_point is None:
            print("Invalid grade. Please try again.")
        else:
            grades.append(grade_point)

    if not grades:
        print("No grades entered.")
    else:
        average_grade = calculate_average_grade(grades)
        print(f"Average grade: {average_grade:.2f}")

if __name__ == "__main__":
    main()
```
