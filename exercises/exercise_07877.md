## Exercise
Write a program that implements a Caesar cipher using conditional statements.

## Reference Solution
```python
def caesar_cipher(text, shift, direction):
    """
    Encrypts or decrypts the given text using a Caesar cipher.

    Args:
        text (str): The text to be encrypted or decrypted.
        shift (int): The number of positions to shift the letters.
        direction (str): 'encrypt' or 'decrypt' to determine the direction of the shift.

    Returns:
        str: The encrypted or decrypted text.
    """
    result = ""

    for char in text:
        if char.isalpha():
            ascii_offset = 97 if char.islower() else 65
            if direction == 'encrypt':
                shifted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            elif direction == 'decrypt':
                shifted_char = chr((ord(char) - ascii_offset - shift) % 26 + ascii_offset)
            else:
                raise ValueError("Invalid direction. Please use 'encrypt' or 'decrypt'.")
            result += shifted_char
        else:
            result += char

    return result


def main():
    text = input("Enter the text to be encrypted or decrypted: ")
    shift = int(input("Enter the shift value: "))
    direction = input("Enter the direction ('encrypt' or 'decrypt'): ")

    result = caesar_cipher(text, shift, direction)
    print("Result:", result)


if __name__ == "__main__":
    main()
```
