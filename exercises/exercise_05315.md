## Exercise
Create a function that returns another function, allowing for function chaining

## Reference Solution
```python
def outer_function(value):
    """
    Outer function that returns an inner function.
    
    Args:
        value (int): Initial value.
    
    Returns:
        function: Inner function that can be chained.
    """
    def inner_function(operation=None, amount=0):
        """
        Inner function that performs operations on the initial value.
        
        Args:
            operation (str, optional): Operation to perform ('add', 'subtract', 'multiply', 'divide'). Defaults to None.
            amount (int, optional): Amount to use for the operation. Defaults to 0.
        
        Returns:
            function or int: Inner function if no operation is specified, otherwise the result of the operation.
        """
        nonlocal value  # Access the outer function's value
        
        if operation is None:
            # Return the inner function if no operation is specified
            return inner_function
        
        if operation == 'add':
            value += amount
        elif operation == 'subtract':
            value -= amount
        elif operation == 'multiply':
            value *= amount
        elif operation == 'divide':
            if amount != 0:
                value /= amount
            else:
                raise ZeroDivisionError("Cannot divide by zero")
        
        # Return the inner function to allow chaining
        return inner_function
    
    # Return the inner function with the initial value
    return inner_function


# Example usage:
if __name__ == "__main__":
    # Create a chainable function with an initial value of 10
    chain = outer_function(10)
    
    # Perform operations and chain them together
    result = chain('add', 5)('multiply', 2)('subtract', 3)
    
    # To get the final result, call the function without any arguments
    print(result())  # Output: 24.0
```
