## Exercise
Write a program that takes input from the user with a timeout and prints out the results.

## Reference Solution
```python
import signal

class TimeoutError(Exception): 
    pass 

def timeout_handler(signum, frame):
    raise TimeoutError

def input_with_timeout(prompt, timeout):
    """
    Get user input with a specified timeout.

    Args:
        prompt (str): The prompt to display to the user.
        timeout (int): The timeout in seconds.

    Returns:
        str: The user's input.
    """
    signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(timeout)

    try:
        return input(prompt)
    except TimeoutError:
        print("\nTimeout! No input received.")
        return None
    finally:
        signal.alarm(0)

def main():
    prompt = "Please enter your input: "
    timeout = 5  # seconds

    user_input = input_with_timeout(prompt, timeout)

    if user_input is not None:
        print(f"\nYou entered: {user_input}")

if __name__ == "__main__":
    main()
```
