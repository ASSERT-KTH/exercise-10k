## Exercise
Write a program that takes two positive integers as input and finds their greatest common divisor (GCD).

## Reference Solution
```python
import math

def find_gcd(num1, num2):
    """
    Calculate the greatest common divisor (GCD) of two positive integers.

    Args:
        num1 (int): The first positive integer.
        num2 (int): The second positive integer.

    Returns:
        int: The GCD of num1 and num2.
    """
    # Check if inputs are valid (positive integers)
    if not isinstance(num1, int) or not isinstance(num2, int) or num1 <= 0 or num2 <= 0:
        raise ValueError("Both inputs must be positive integers.")

    # Use the Euclidean algorithm to find the GCD
    while num2:
        num1, num2 = num2, num1 % num2

    return num1

# Alternatively, use the math.gcd function (Python 3.5+)
def find_gcd_builtin(num1, num2):
    """
    Calculate the greatest common divisor (GCD) of two positive integers using the math.gcd function.

    Args:
        num1 (int): The first positive integer.
        num2 (int): The second positive integer.

    Returns:
        int: The GCD of num1 and num2.
    """
    # Check if inputs are valid (positive integers)
    if not isinstance(num1, int) or not isinstance(num2, int) or num1 <= 0 or num2 <= 0:
        raise ValueError("Both inputs must be positive integers.")

    return math.gcd(num1, num2)

    gcd = find_gcd(num1, num2)
```
