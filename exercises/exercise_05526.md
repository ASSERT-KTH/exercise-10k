## Exercise
Write a program to perform implicit and explicit type casting between different data types.

## Reference Solution
```python
# Implicit Type Casting
def implicit_type_casting():
    """
    Implicit type casting occurs when Python automatically converts one data type into another.
    """
    # Implicit type casting from integer to float
    a = 5
    b = 2.0
    print("Implicit type casting from integer to float:")
    print(f"a = {a}, type(a) = {type(a)}")
    print(f"b = {b}, type(b) = {type(b)}")
    print(f"a + b = {a + b}, type(a + b) = {type(a + b)}")

    # Implicit type casting from integer to complex
    c = 5
    d = 2 + 3j
    print("\nImplicit type casting from integer to complex:")
    print(f"c = {c}, type(c) = {type(c)}")
    print(f"d = {d}, type(d) = {type(d)}")
    print(f"c + d = {c + d}, type(c + d) = {type(c + d)}")

# Explicit Type Casting
def explicit_type_casting():
    """
    Explicit type casting occurs when the programmer explicitly converts one data type into another.
    """
    # Explicit type casting from string to integer
    e = "123"
    print("\nExplicit type casting from string to integer:")
    print(f"e = {e}, type(e) = {type(e)}")
    print(f"int(e) = {int(e)}, type(int(e)) = {type(int(e))}")

    # Explicit type casting from integer to string
    f = 123
    print("\nExplicit type casting from integer to string:")
    print(f"f = {f}, type(f) = {type(f)}")
    print(f"str(f) = {str(f)}, type(str(f)) = {type(str(f))}")

    # Explicit type casting from float to integer
    g = 123.456
    print("\nExplicit type casting from float to integer:")
    print(f"g = {g}, type(g) = {type(g)}")
    print(f"int(g) = {int(g)}, type(int(g)) = {type(int(g))}")

# Main function
def main():
    print("Implicit and Explicit Type Casting in Python")
    print("-------------------------------------------")
    implicit_type_casting()
    explicit_type_casting()

if __name__ == "__main__":
    main()
```
