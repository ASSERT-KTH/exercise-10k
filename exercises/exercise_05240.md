## Exercise
Write a program that takes a user's preferences as input, and prints personalized recommendations based on their preferences.

## Reference Solution
```python
class RecommendationSystem:
    def __init__(self):
        self.movie_genres = {
            "action": ["Die Hard", "The Matrix", "John Wick"],
            "comedy": ["The Hangover", "Superbad", "Bridesmaids"],
            "romance": ["The Notebook", "Titanic", "La La Land"],
            "horror": ["The Shining", "The Exorcist", "Get Out"],
            "sci-fi": ["Star Wars", "The Lord of the Rings", "Blade Runner"]
        }
        
        self.music_genres = {
            "rock": ["Queen", "Led Zeppelin", "Guns N' Roses"],
            "pop": ["Taylor Swift", "Katy Perry", "Justin Bieber"],
            "hip-hop": ["Kendrick Lamar", "Drake", "Cardi B"],
            "electronic": ["Daft Punk", "The Chemical Brothers", "Skrillex"],
            "classical": ["Mozart", "Beethoven", "Chopin"]
        }
        
        self.book_genres = {
            "fiction": ["To Kill a Mockingbird", "The Great Gatsby", "Pride and Prejudice"],
            "non-fiction": ["Sapiens", "The Diary of a Young Girl", "The Immortal Life of Henrietta Lacks"],
            "mystery": ["And Then There Were None", "The Murder on the Orient Express", "The Girl with the Dragon Tattoo"],
            "thriller": ["The Silence of the Lambs", "The Shining", "Gone Girl"],
            "fantasy": ["The Lord of the Rings", "Harry Potter", "The Wheel of Time"]
        }

    def get_preferences(self):
        print("Please select your preferred genres:")
        print("1. Movie genres")
        print("2. Music genres")
        print("3. Book genres")
        
        choice = input("Enter your choice (1/2/3): ")
        
        if choice == "1":
            print("Select your preferred movie genres (separate with comma):")
            print("action, comedy, romance, horror, sci-fi")
            movie_genres = input("Enter your preferred movie genres: ").split(',')
            return self.get_movie_recommendations(movie_genres)
        elif choice == "2":
            print("Select your preferred music genres (separate with comma):")
            print("rock, pop, hip-hop, electronic, classical")
            music_genres = input("Enter your preferred music genres: ").split(',')
            return self.get_music_recommendations(music_genres)
        elif choice == "3":
            print("Select your preferred book genres (separate with comma):")
            print("fiction, non-fiction, mystery, thriller, fantasy")
            book_genres = input("Enter your preferred book genres: ").split(',')
            return self.get_book_recommendations(book_genres)
        else:
            print("Invalid choice. Please try again.")
            return self.get_preferences()

    def get_movie_recommendations(self, genres):
        print("Here are some movie recommendations based on your preferences:")
        for genre in genres:
            genre = genre.strip()
            if genre in self.movie_genres:
                print(f"{genre.capitalize()}: {', '.join(self.movie_genres[genre])}")

    def get_music_recommendations(self, genres):
        print("Here are some music recommendations based on your preferences:")
        for genre in genres:
            genre = genre.strip()
            if genre in self.music_genres:
                print(f"{genre.capitalize()}: {', '.join(self.music_genres[genre])}")

    def get_book_recommendations(self, genres):
        print("Here are some book recommendations based on your preferences:")
        for genre in genres:
            genre = genre.strip()
            if genre in self.book_genres:
                print(f"{genre.capitalize()}: {', '.join(self.book_genres[genre])}")

def main():
    recommendation_system = RecommendationSystem()
    recommendation_system.get_preferences()

if __name__ == "__main__":
    main()
```
