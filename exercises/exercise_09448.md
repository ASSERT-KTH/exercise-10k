## Exercise
Create a program that checks if a given year is a leap year and prints a corresponding message.

## Reference Solution
```python
def is_leap_year(year):
    """
    Checks if a given year is a leap year.
    
    A year is a leap year if it is evenly divisible by 4, except for end-of-century years, 
    which must be divisible by 400. This means that the year 2000 was a leap year, 
    although 1900 was not.

    Parameters:
    year (int): The year to check.

    Returns:
    bool: True if the year is a leap year, False otherwise.
    """
    return year % 4 == 0 and (year % 100 != 0 or year % 400 == 0)


def main():
    # Get the year from the user
    year = int(input("Enter a year: "))

    # Check if the year is a leap year
    if is_leap_year(year):
        print(f"{year} is a leap year.")
    else:
        print(f"{year} is not a leap year.")


if __name__ == "__main__":
    main()
```
