## Exercise
Create a program that takes a string as input and outputs the encrypted string using a Caesar cipher.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts a string using a Caesar cipher.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The number of places to shift the letters.

    Returns:
        str: The encrypted string.
    """
    result = ""

    for char in text:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            result += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            result += char

    return result

def main():
    text = input("Enter a string: ")
    shift = int(input("Enter the shift value: "))
    encrypted_text = caesar_cipher(text, shift)
    print("Encrypted text:", encrypted_text)

if __name__ == "__main__":
    main()
```
