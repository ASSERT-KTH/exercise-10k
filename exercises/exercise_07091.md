## Exercise
Read a sentence and a word from the user, and check if the word is present in the sentence

## Reference Solution
```python
def check_word_presence(sentence, word):
    """
    Checks if a word is present in a sentence.

    Args:
        sentence (str): The input sentence.
        word (str): The word to be searched.

    Returns:
        bool: True if the word is present, False otherwise.
    """
    # Convert both sentence and word to lower case to make the search case-insensitive
    sentence = sentence.lower()
    word = word.lower()

    # Split the sentence into words
    words = sentence.split()

    # Remove punctuation from each word
    words = [w.strip('.,!?;:"\'') for w in words]

    # Check if the word is present in the list of words
    return word in words

def main():
    # Get user input
    sentence = input("Enter a sentence: ")
    word = input("Enter a word: ")

    # Check if the word is present in the sentence
    if check_word_presence(sentence, word):
        print(f"The word '{word}' is present in the sentence.")
    else:
        print(f"The word '{word}' is not present in the sentence.")

if __name__ == "__main__":
    main()
```
