## Exercise
Write a program to unzip a list of tuples into separate lists.

## Reference Solution
```python
def unzip_tuples(tuples_list):
    """
    Unzips a list of tuples into separate lists.

    Args:
        tuples_list (list): A list of tuples.

    Returns:
        list: A list of lists, where each list contains the elements from the corresponding position in the input tuples.
    """
    return list(zip(*tuples_list))

def main():
    # Example usage
    tuples_list = [(1, 2, 3), (4, 5, 6), (7, 8, 9)]
    unzipped_lists = unzip_tuples(tuples_list)

    # Convert the unzipped tuples to lists
    unzipped_lists = [list(t) for t in unzipped_lists]

    print("Unzipped Lists:")
    for i, lst in enumerate(unzipped_lists):
        print(f"List {i+1}: {lst}")

if __name__ == "__main__":
    main()
```
