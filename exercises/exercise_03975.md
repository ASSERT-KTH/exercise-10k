## Exercise
Write a program that finds the minimum window substring that contains all characters of a given string. The program should take two strings as input and output the minimum window substring.

## Reference Solution
```python
from collections import defaultdict

def min_window(s, t):
    if not s or not t:
        return ""

    dict_t = defaultdict(int)
    for char in t:
        dict_t[char] += 1

    required = len(dict_t)
    left = 0
    min_len = float('inf')
    min_window = ""
    formed = 0

    window_counts = defaultdict(int)

    for right in range(len(s)):
        character = s[right]
        window_counts[character] += 1

        if character in dict_t and window_counts[character] == dict_t[character]:
            formed += 1

        while left <= right and formed == required:
            character = s[left]

            if right - left + 1 < min_len:
                min_len = right - left + 1
                min_window = s[left:right + 1]

            window_counts[character] -= 1
            if character in dict_t and window_counts[character] < dict_t[character]:
                formed -= 1

            left += 1

    return min_window

t = "ABC"
```
