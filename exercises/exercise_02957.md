## Exercise
Write a program to search for a pattern in a text using the Rabin-Karp algorithm.

## Reference Solution

def rabin_karp(text, pattern):
    """
    Search for a pattern in a text using the Rabin-Karp algorithm.

    Args:
        text (str): The text to search in.
        pattern (str): The pattern to search for.

    Returns:
        list: A list of indices where the pattern is found.
    """
    # Calculate the length of the text and pattern
    n = len(text)
    m = len(pattern)

    # Calculate the hash value of the pattern
    pattern_hash = 0
    for char in pattern:
        pattern_hash += ord(char)

    # Calculate the hash value of the first window of the text
    text_hash = 0
    for i in range(m):
        text_hash += ord(text[i])

    # Initialize a list to store the indices of the pattern
    indices = []

    # Slide the window over the text
    for i in range(n - m + 1):
        # Check if the hash values match
        if text_hash == pattern_hash:
            # Check if the strings match
            match = True
            for j in range(m):
                if text[i + j] != pattern[j]:
                    match = False
                    break
            if match:
                indices.append(i)

        # Update the hash value of the next window
        if i < n - m:
            text_hash -= ord(text[i])
            text_hash += ord(text[i + m])

    return indices

# Example usage
text = "abxabcabcaby"
pattern = "abc"
indices = rabin_karp(text, pattern)
print(f"Pattern '{pattern}' found at indices: {indices}")
