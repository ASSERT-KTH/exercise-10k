## Exercise
Design a program to calculate the square root and cube root of a number.

## Reference Solution
```python
import math

def calculate_roots(number):
    """
    Calculates the square root and cube root of a number.

    Args:
        number (float): The input number.

    Returns:
        tuple: A tuple containing the square root and cube root of the number.
    """
    if number < 0:
        raise ValueError("Input number cannot be negative")

    square_root = math.sqrt(number)
    cube_root = round(number ** (1/3), 3)  # Round to 3 decimal places

    return square_root, cube_root

def main():
    number = float(input("Enter a number: "))
    try:
        square_root, cube_root = calculate_roots(number)
        print(f"Square Root: {square_root}")
        print(f"Cube Root: {cube_root}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
