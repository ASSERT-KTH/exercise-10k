## Exercise
Write a program that calculates an object's speed given its distance and time.

## Reference Solution
```python
def calculate_speed(distance, time):
    """
    Calculates an object's speed given its distance and time.

    Args:
        distance (float): The distance traveled by the object.
        time (float): The time taken to travel the distance.

    Returns:
        float: The speed of the object.
    """
    if time == 0:
        raise ValueError("Time cannot be zero.")
    speed = distance / time
    return speed

def main():
    distance = float(input("Enter the distance traveled (in meters): "))
    time = float(input("Enter the time taken (in seconds): "))

    try:
        speed = calculate_speed(distance, time)
        print(f"The speed of the object is {speed} meters per second.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
