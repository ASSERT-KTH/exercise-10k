## Exercise
Create a program that takes two numbers and an arithmetic operation as input and performs the operation on the numbers. The operations to support are addition, subtraction, multiplication, and division. Handle invalid operations and division by zero errors.

## Reference Solution
```python
def calculate(num1, num2, operation):
    """
    Performs the given arithmetic operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to perform. Can be 'add', 'subtract', 'multiply', or 'divide'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operation is not one of the supported operations.
        ZeroDivisionError: If the operation is division and the second number is zero.
    """

    # Define the supported operations
    supported_operations = {
        'add': lambda x, y: x + y,
        'subtract': lambda x, y: x - y,
        'multiply': lambda x, y: x * y,
        'divide': lambda x, y: x / y
    }

    # Check if the operation is supported
    if operation not in supported_operations:
        raise ValueError(f"Unsupported operation: {operation}. Supported operations are: {list(supported_operations.keys())}")

    # Perform the operation
    try:
        result = supported_operations[operation](num1, num2)
    except ZeroDivisionError:
        raise ZeroDivisionError("Cannot divide by zero.")

    return result


def main():
    # Get the numbers and operation from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    print("Available operations:")
    print("1. Add")
    print("2. Subtract")
    print("3. Multiply")
    print("4. Divide")

    choice = int(input("Enter the number of your chosen operation: "))

    operations = {
        1: 'add',
        2: 'subtract',
        3: 'multiply',
        4: 'divide'
    }

    operation = operations.get(choice)

    if operation is None:
        print("Invalid operation choice.")
        return

    # Perform the operation and print the result
    try:
        result = calculate(num1, num2, operation)
        print(f"The result of {operation}ing {num1} and {num2} is {result}.")
    except (ValueError, ZeroDivisionError) as e:
        print(f"An error occurred: {e}")


if __name__ == "__main__":
    main()
```
