## Exercise
Write a program that finds the most frequent element in a list of numbers.

## Reference Solution
```python
from collections import Counter

def most_frequent_element(numbers):
    """
    Find the most frequent element in a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        The most frequent number in the list.
    """
    # Count the occurrences of each number
    count = Counter(numbers)

    # Find the number with the highest count
    most_frequent = max(count, key=count.get)

    return most_frequent

def main():
    # Example usage
    numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
    most_frequent = most_frequent_element(numbers)
    print(f"The most frequent element is {most_frequent}")

if __name__ == "__main__":
    main()
```
