## Exercise
Write a program to reverse a given string using conditional statements

## Reference Solution
```python
def reverse_string(input_str):
    """
    Reverses a given string using conditional statements.

    Args:
        input_str (str): The input string to be reversed.

    Returns:
        str: The reversed string.
    """
    reversed_str = ""
    i = len(input_str) - 1

    # Loop through the input string from the end to the start
    while i >= 0:
        # Append the current character to the reversed string
        reversed_str += input_str[i]
        # Decrement the index
        i -= 1

    return reversed_str

# Example usage
def main():
    input_str = "Hello World"
    print("Input String:", input_str)
    print("Reversed String:", reverse_string(input_str))

if __name__ == "__main__":
    main()
```
