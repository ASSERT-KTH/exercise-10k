## Exercise
Write a program to demonstrate the constraints of different data types, such as the minimum and maximum values that can be stored.

## Reference Solution
```python
import sys

def check_constraints():
    # Integer constraints
    print("Integer Constraints:")
    print("--------------------")
    print(f"Minimum integer value: {sys.maxsize * -1 - 1}")
    print(f"Maximum integer value: {sys.maxsize}")

    # Floating point constraints
    print("\nFloating Point Constraints:")
    print("---------------------------")
    print(f"Minimum floating point value: {sys.float_info.min}")
    print(f"Maximum floating point value: {sys.float_info.max}")
    print(f"Smallest difference between two floating point numbers: {sys.float_info.epsilon}")

    # Boolean constraints
    print("\nBoolean Constraints:")
    print("-------------------")
    print(f"Minimum boolean value: {False}")
    print(f"Maximum boolean value: {True}")

    # Complex number constraints
    print("\nComplex Number Constraints:")
    print("--------------------------")
    print(f"Minimum complex number value: {-sys.maxsize * 1j}")
    print(f"Maximum complex number value: {sys.maxsize * 1j}")

    # String constraints
    print("\nString Constraints:")
    print("------------------")
    print(f"Minimum string value: '' (empty string)")
    print(f"Maximum string value: arbitrary length (limited by memory)")

check_constraints()
```
