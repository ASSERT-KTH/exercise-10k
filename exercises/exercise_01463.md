## Exercise
Write a program that swaps the values of two variables without using a temporary variable.

## Reference Solution

def swap_variables_xor(a, b):
    """
    Swap two variables using the XOR operator.

    Args:
    a (int): The first variable.
    b (int): The second variable.

    Returns:
    tuple: A tuple containing the swapped values.
    """
    print("Before swap (XOR): a =", a, ", b =", b)
    a = a ^ b
    b = a ^ b
    a = a ^ b
    print("After swap (XOR): a =", a, ", b =", b)
    return a, b


def swap_variables_arithmetic(a, b):
    """
    Swap two variables using arithmetic operations.

    Args:
    a (int): The first variable.
    b (int): The second variable.

    Returns:
    tuple: A tuple containing the swapped values.
    """
    print("Before swap (Arithmetic): a =", a, ", b =", b)
    a = a + b
    b = a - b
    a = a - b
    print("After swap (Arithmetic): a =", a, ", b =", b)
    return a, b


def swap_variables_pythonic(a, b):
    """
    Swap two variables using Python's tuple packing and unpacking.

    Args:
    a (int): The first variable.
    b (int): The second variable.

    Returns:
    tuple: A tuple containing the swapped values.
    """
    print("Before swap (Pythonic): a =", a, ", b =", b)
    a, b = b, a
    print("After swap (Pythonic): a =", a, ", b =", b)
    return a, b
