## Exercise
Create a program that takes in two numbers and an arithmetic operation as input, then performs the operation and prints the result. The operations to support are addition, subtraction, multiplication, and division.

## Reference Solution
```python
def calculate(num1, num2, operation):
    """
    Perform arithmetic operations on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to perform. Supports 'add', 'subtract', 'multiply', 'divide'.

    Returns:
        float: The result of the operation.
    """

    # Define a dictionary to map operation names to their corresponding functions
    operations = {
        'add': lambda x, y: x + y,
        'subtract': lambda x, y: x - y,
        'multiply': lambda x, y: x * y,
        'divide': lambda x, y: x / y if y != 0 else float('inf')
    }

    # Check if the operation is supported
    if operation not in operations:
        raise ValueError("Unsupported operation. Supported operations are 'add', 'subtract', 'multiply', 'divide'.")

    # Perform the operation and return the result
    return operations[operation](num1, num2)


def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (add, subtract, multiply, divide): ").lower()

    try:
        # Calculate and print the result
        result = calculate(num1, num2, operation)
        if result == float('inf'):
            print("Error: Division by zero is not allowed.")
        else:
            print(f"The result of {operation}ing {num1} and {num2} is {result}.")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
