## Exercise
Write a program that takes a list as input and checks if it contains any duplicate elements.

## Reference Solution
```python
def has_duplicates(input_list):
    """
    Checks if a list contains any duplicate elements.

    Args:
        input_list (list): The input list to check for duplicates.

    Returns:
        bool: True if duplicates are found, False otherwise.
    """
    seen = set()
    for element in input_list:
        if element in seen:
            return True
        seen.add(element)
    return False

def main():
    # Example usage
    input_list = [1, 2, 3, 4, 5, 2]
    result = has_duplicates(input_list)
    print(f"List contains duplicates: {result}")

    input_list = [1, 2, 3, 4, 5]
    result = has_duplicates(input_list)
    print(f"List contains duplicates: {result}")

if __name__ == "__main__":
    main()
```
