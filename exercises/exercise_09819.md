## Exercise
Write a program that calculates the sum of all elements in an array using a for loop and a while loop. The program should first ask for the size of the array, then ask for each element, and finally print the sum of all elements.

## Reference Solution
```python
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n;
    cout << "Enter the size of the array: ";
    cin >> n;

    vector<int> v(n);
    cout << "Enter " << n << " elements: ";
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }

    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += v[i];
    }

    cout << "Sum of elements using for loop: " << sum << endl;

    sum = 0;
    int i = 0;
    while (i < n) {
        sum += v[i];
        i++;
    }

    cout << "Sum of elements using while loop: " << sum << endl;

    return 0;
}
```
