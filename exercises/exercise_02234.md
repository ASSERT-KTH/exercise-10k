## Exercise
Write a program that generates all permutations of a given list.

## Reference Solution
```python
import itertools

def generate_permutations(input_list):
    """
    Generate all permutations of a given list.

    Args:
        input_list (list): The input list for which permutations are to be generated.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    return list(itertools.permutations(input_list))

def main():
    # Example usage:
    input_list = [1, 2, 3]
    permutations = generate_permutations(input_list)

    print("Input List:", input_list)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i+1}: {permutation}")

if __name__ == "__main__":
    main()
```
