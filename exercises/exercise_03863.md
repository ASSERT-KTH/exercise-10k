## Exercise
Create a dictionary with names of countries as keys and their capitals as values. Update the capital of a specific country and print the updated dictionary.

## Reference Solution
```python
def update_capital(countries_dict, country, new_capital):
    """
    Updates the capital of a specific country in the dictionary.

    Args:
        countries_dict (dict): Dictionary with country names as keys and their capitals as values.
        country (str): Name of the country to update.
        new_capital (str): New capital of the country.

    Returns:
        dict: Updated dictionary with the new capital.
    """
    if country in countries_dict:
        countries_dict[country] = new_capital
    else:
        print(f"Country '{country}' not found in the dictionary.")
    return countries_dict


def main():
    # Create a dictionary with country names as keys and their capitals as values
    countries = {
        "France": "Paris",
        "Germany": "Berlin",
        "Italy": "Rome",
        "Spain": "Madrid",
        "Portugal": "Lisbon"
    }

    print("Original Dictionary:")
    print(countries)

    # Update the capital of a specific country
    updated_countries = update_capital(countries, "Portugal", "Porto")

    print("\nUpdated Dictionary:")
    print(updated_countries)


if __name__ == "__main__":
    main()
```
