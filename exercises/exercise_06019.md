## Exercise
Write a program that allows the user to play a text adventure game

## Reference Solution
```python
# Import the `random` module for generating random numbers
import random

# Define a dictionary to store the game's locations
locations = {
    "start": {
        "description": "You are standing at the entrance of a dark cave.",
        "items": ["torch"],
        "exits": ["cave", "forest"]
    },
    "cave": {
        "description": "You are inside the dark cave.",
        "items": ["sword"],
        "exits": ["start", "dragon"]
    },
    "forest": {
        "description": "You are in a dense forest.",
        "items": ["shield"],
        "exits": ["start", "village"]
    },
    "village": {
        "description": "You are in a small village.",
        "items": ["health potion"],
        "exits": ["forest", "dragon"]
    },
    "dragon": {
        "description": "You are facing a fierce dragon.",
        "items": [],
        "exits": ["cave", "village"]
    }
}

# Define a dictionary to store the player's inventory
player_inventory = []

# Define a variable to store the player's current location
current_location = "start"

# Define a function to display the game's menu
def display_menu():
    print("\n--- Menu ---")
    print("1. Look around")
    print("2. Take item")
    print("3. Go to another location")
    print("4. Fight the dragon")
    print("5. Quit game")

# Define a function to handle the player's choice
def handle_choice(choice):
    global current_location
    global player_inventory

    if choice == "1":
        print(locations[current_location]["description"])
    elif choice == "2":
        if locations[current_location]["items"]:
            item = locations[current_location]["items"][0]
            player_inventory.append(item)
            locations[current_location]["items"].remove(item)
            print(f"You took {item}.")
        else:
            print("There are no items to take.")
    elif choice == "3":
        print("Exits:")
        for i, exit in enumerate(locations[current_location]["exits"]):
            print(f"{i+1}. {exit}")
        exit_choice = input("Choose an exit: ")
        current_location = locations[current_location]["exits"][int(exit_choice) - 1]
    elif choice == "4":
        if current_location == "dragon":
            if "sword" in player_inventory:
                print("You defeated the dragon! Congratulations, you won!")
                return False
            else:
                print("You don't have a sword to fight the dragon.")
        else:
            print("There is no dragon to fight here.")
    elif choice == "5":
        return False
    return True

# Main game loop
while True:
    display_menu()
    choice = input("Choose an option: ")
    if not handle_choice(choice):
        break
```
