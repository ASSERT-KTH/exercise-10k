## Exercise
Create a program that converts numbers to their text equivalent using conditional statements.

## Reference Solution
```python
def convert_to_text(num):
    # Define the text equivalents for numbers 0-19
    ones = ['', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine', 'ten', 'eleven', 'twelve', 'thirteen', 'fourteen', 'fifteen', 'sixteen', 'seventeen', 'eighteen', 'nineteen']

    # Define the text equivalents for tens
    tens = ['', '', 'twenty', 'thirty', 'forty', 'fifty', 'sixty', 'seventy', 'eighty', 'ninety']

    # Check if the number is less than 20
    if num < 20:
        return ones[num]

    # Check if the number is less than 100
    elif num < 100:
        return tens[num // 10] + ('' if num % 10 == 0 else ' ' + ones[num % 10])

    # Check if the number is less than 1000
    elif num < 1000:
        return ones[num // 100] + ' hundred' + ('' if num % 100 == 0 else ' and ' + convert_to_text(num % 100))

    # Check if the number is less than 1000000
    elif num < 1000000:
        return convert_to_text(num // 1000) + ' thousand' + ('' if num % 1000 == 0 else ' ' + convert_to_text(num % 1000))

    # Check if the number is less than 1000000000
    elif num < 1000000000:
        return convert_to_text(num // 1000000) + ' million' + ('' if num % 1000000 == 0 else ' ' + convert_to_text(num % 1000000))

    # Check if the number is less than 1000000000000
    elif num < 1000000000000:
        return convert_to_text(num // 1000000000) + ' billion' + ('' if num % 1000000000 == 0 else ' ' + convert_to_text(num % 1000000000))

# Test the function
print(convert_to_text(123456789))
```
