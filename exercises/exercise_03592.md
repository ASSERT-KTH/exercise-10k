## Exercise
Create a program that converts an amount of money from one currency to another given the exchange rate.

## Reference Solution
```python
def convert_currency(amount, exchange_rate, from_currency, to_currency):
    """
    Converts an amount of money from one currency to another.

    Args:
        amount (float): The amount of money to convert.
        exchange_rate (float): The exchange rate between the two currencies.
        from_currency (str): The currency of the input amount.
        to_currency (str): The target currency.

    Returns:
        float: The equivalent amount in the target currency.
    """
    converted_amount = amount * exchange_rate
    return converted_amount

def main():
    # Get user input
    amount = float(input("Enter the amount of money: "))
    exchange_rate = float(input("Enter the exchange rate (1 {} = ? {}): ".format("USD", "EUR")))
    from_currency = input("Enter the currency of the input amount (e.g. USD): ")
    to_currency = input("Enter the target currency (e.g. EUR): ")

    # Convert currency
    converted_amount = convert_currency(amount, exchange_rate, from_currency, to_currency)

    # Print result
    print("{} {} is equal to {:.2f} {}".format(amount, from_currency, converted_amount, to_currency))

if __name__ == "__main__":
    main()
```
