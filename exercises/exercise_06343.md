```python
def convert_string_to_int(string):
    """
    Converts a string to an integer.

    Args:
        string (str): The string to convert.

    Returns:
        int: The converted integer.

    Raises:
        ValueError: If the string contains non-numeric characters.
    """
    try:
        return int(string)
    except ValueError:
        raise ValueError("The string contains non-numeric characters.")

def convert_int_to_string(integer):
    """
    Converts an integer to a string.

    Args:
        integer (int): The integer to convert.

    Returns:
        str: The converted string.
    """
    return str(integer)

# Example usage
string = "123"
integer = convert_string_to_int(string)
print(f"Converted '{string}' to {integer}")

integer = 123
string = convert_int_to_string(integer)
print(f"Converted {integer} to '{string}'")
```
