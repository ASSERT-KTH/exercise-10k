## Exercise
Create a program to calculate the bus fare based on the distance traveled using conditional statements

## Reference Solution
```python
def calculate_bus_fare(distance):
    """
    Calculate the bus fare based on the distance traveled.

    Args:
        distance (float): Distance traveled in kilometers.

    Returns:
        float: Bus fare.
    """
    if distance <= 5:
        # For distances up to 5 km, the fare is $2
        return 2.0
    elif distance <= 10:
        # For distances between 5 km and 10 km, the fare is $3
        return 3.0
    elif distance <= 20:
        # For distances between 10 km and 20 km, the fare is $5
        return 5.0
    else:
        # For distances above 20 km, the fare is $7
        return 7.0

def main():
    print("Bus Fare Calculator Program")
    print("--------------------------------")
    
    distance = float(input("Enter the distance traveled in kilometers: "))
    
    fare = calculate_bus_fare(distance)
    
    print(f"The bus fare for {distance} km is: ${fare:.2f}")

if __name__ == "__main__":
    main()
```
