## Exercise
Write a program that generates a random number and asks the user to guess it. After each guess, print out if the guess is higher or lower than the generated number.

## Reference Solution
```python
import random

def guessing_game():
    # Generate a random number between 1 and 100
    number_to_guess = random.randint(1, 100)
    guess = None
    attempts = 0

    print("Welcome to the number guessing game!")
    print("I'm thinking of a number between 1 and 100.")

    while guess != number_to_guess:
        # Ask the user for their guess
        try:
            guess = int(input("Take a guess: "))
        except ValueError:
            print("That's not a valid number!")
            continue

        # Check if the guess is in the correct range
        if guess < 1 or guess > 100:
            print("Please guess a number between 1 and 100.")
            continue

        # Increment the attempt counter
        attempts += 1

        # Check if the guess is higher or lower than the number to guess
        if guess < number_to_guess:
            print("Your guess is too low!")
        elif guess > number_to_guess:
            print("Your guess is too high!")

    # Print a success message
    print(f"Congratulations! You found the number in {attempts} attempts.")

if __name__ == "__main__":
    guessing_game()
```
