## Exercise
Implement a function to generate all permutations of a given list of elements.

## Reference Solution
```python
def generate_permutations(elements):
    """
    Generate all permutations of a given list of elements.

    Args:
        elements (list): The list of elements to generate permutations for.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    # Base case: if the list has only one element, return a list containing that element
    if len(elements) == 1:
        return [elements]

    permutations = []
    for i, element in enumerate(elements):
        # Get the remaining elements
        remaining_elements = elements[:i] + elements[i + 1:]
        
        # Generate permutations for the remaining elements
        for permutation in generate_permutations(remaining_elements):
            permutations.append([element] + permutation)

    return permutations

# Example usage
def main():
    elements = [1, 2, 3]
    permutations = generate_permutations(elements)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i + 1}: {permutation}")

if __name__ == "__main__":
    main()
```
