## Exercise
Develop a program that calculates the volume of different 3D shapes (cube, sphere, cylinder).

## Reference Solution
```python
import math

def calculate_cube_volume(side_length):
    """
    Calculate the volume of a cube.

    Args:
        side_length (float): The length of one side of the cube.

    Returns:
        float: The volume of the cube.
    """
    return side_length ** 3

def calculate_sphere_volume(radius):
    """
    Calculate the volume of a sphere.

    Args:
        radius (float): The radius of the sphere.

    Returns:
        float: The volume of the sphere.
    """
    return (4/3) * math.pi * (radius ** 3)

def calculate_cylinder_volume(radius, height):
    """
    Calculate the volume of a cylinder.

    Args:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.

    Returns:
        float: The volume of the cylinder.
    """
    return math.pi * (radius ** 2) * height

def main():
    while True:
        print("\n3D Shape Volume Calculator")
        print("1. Cube")
        print("2. Sphere")
        print("3. Cylinder")
        print("4. Quit")

        choice = input("Choose a shape (1-4): ")

        if choice == "1":
            side_length = float(input("Enter the side length of the cube: "))
            volume = calculate_cube_volume(side_length)
            print(f"The volume of the cube is: {volume:.2f} cubic units")
        elif choice == "2":
            radius = float(input("Enter the radius of the sphere: "))
            volume = calculate_sphere_volume(radius)
            print(f"The volume of the sphere is: {volume:.2f} cubic units")
        elif choice == "3":
            radius = float(input("Enter the radius of the cylinder: "))
            height = float(input("Enter the height of the cylinder: "))
            volume = calculate_cylinder_volume(radius, height)
            print(f"The volume of the cylinder is: {volume:.2f} cubic units")
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
