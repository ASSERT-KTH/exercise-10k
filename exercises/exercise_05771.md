## Exercise
Write a program that simulates a restaurant menu. The program should allow the user to order food and drinks, and calculate the total cost.

## Reference Solution
```python
class MenuItem:
    """Represents a menu item with a name and price."""
    def __init__(self, name, price):
        self.name = name
        self.price = price

class Restaurant:
    """Represents a restaurant with a menu."""
    def __init__(self):
        self.menu = {
            "Food": [
                MenuItem("Burger", 10.99),
                MenuItem("Pizza", 14.99),
                MenuItem("Salad", 8.99),
            ],
            "Drinks": [
                MenuItem("Soda", 2.99),
                MenuItem("Juice", 3.99),
                MenuItem("Water", 1.99),
            ],
        }
        self.order = []

    def display_menu(self):
        """Displays the restaurant menu."""
        print("Menu:")
        for category, items in self.menu.items():
            print(f"\n{category}:")
            for i, item in enumerate(items, start=1):
                print(f"{i}. {item.name}: ${item.price:.2f}")

    def add_to_order(self):
        """Adds an item to the user's order."""
        category = input("Enter category (Food/Drinks): ")
        if category in self.menu:
            print(f"\n{category}:")
            for i, item in enumerate(self.menu[category], start=1):
                print(f"{i}. {item.name}: ${item.price:.2f}")
            choice = int(input("Enter the number of your choice: ")) - 1
            if 0 <= choice < len(self.menu[category]):
                self.order.append(self.menu[category][choice])
                print(f"Added {self.menu[category][choice].name} to your order.")
            else:
                print("Invalid choice. Please try again.")
        else:
            print("Invalid category. Please try again.")

    def view_order(self):
        """Displays the user's order."""
        print("\nYour Order:")
        total = 0
        for i, item in enumerate(self.order, start=1):
            print(f"{i}. {item.name}: ${item.price:.2f}")
            total += item.price
        print(f"\nTotal: ${total:.2f}")

    def checkout(self):
        """Checks out the user's order."""
        self.view_order()
        print("\nThank you for your order!")

def main():
    restaurant = Restaurant()
    while True:
        print("\nRestaurant Menu Simulator")
        print("1. Display Menu")
        print("2. Add to Order")
        print("3. View Order")
        print("4. Checkout")
        print("5. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            restaurant.display_menu()
        elif choice == "2":
            restaurant.add_to_order()
        elif choice == "3":
            restaurant.view_order()
        elif choice == "4":
            restaurant.checkout()
            break
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
