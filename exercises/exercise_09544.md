## Exercise
Write a program that asks the user for their name and age, then prints out a greeting message with their name and tells them how old they will be in 5 years.

## Reference Solution
```python
# Get user's name and age
def get_user_info():
    name = input("Please enter your name: ")
    while True:
        try:
            age = int(input("Please enter your age: "))
            break
        except ValueError:
            print("Invalid input. Age should be a number.")

    return name, age

# Print greeting and future age
def print_greeting(name, age):
    future_age = age + 5
    print(f"Hello, {name}!")
    print(f"In 5 years, you will be {future_age} years old.")

# Main function
def main():
    name, age = get_user_info()
    print_greeting(name, age)

if __name__ == "__main__":
    main()
```
