## Exercise
Create a function that uses tail recursion to calculate the factorial of a number

## Reference Solution

def tail_recursive_factorial(n, accumulator=1):
    """
    Calculate the factorial of a number using tail recursion.

    Args:
        n (int): The number for which the factorial is to be calculated.
        accumulator (int, optional): The accumulator for the factorial. Defaults to 1.

    Returns:
        int: The factorial of the given number.
    """
    # Base case: when n is 0 or 1, return the accumulator
    if n <= 1:
        return accumulator
    # Recursive case: call the function with n-1 and the updated accumulator
    else:
        return tail_recursive_factorial(n-1, n * accumulator)


# Example usage
def main():
    num = 5
    result = tail_recursive_factorial(num)
    print(f"The factorial of {num} is {result}")


if __name__ == "__main__":
    main()
