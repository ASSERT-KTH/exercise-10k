## Exercise
Write a program that generates a random fortune based on user-inputted name and birthdate.

## Reference Solution
```python
import random
from datetime import datetime

def generate_fortune(name, birthdate):
    """
    Generate a random fortune based on user-inputted name and birthdate.

    Args:
        name (str): The user's name.
        birthdate (str): The user's birthdate in YYYY-MM-DD format.

    Returns:
        str: The generated fortune.
    """
    # Calculate the name length and birthdate day of the year
    name_length = len(name)
    birth_date = datetime.strptime(birthdate, '%Y-%m-%d')
    day_of_year = birth_date.timetuple().tm_yday

    # Define a list of fortunes
    fortunes = [
        "You will have a great day!",
        "Good things are coming your way.",
        "Be careful, challenges are ahead.",
        "Your luck is about to change.",
        "You will meet someone special soon.",
        "A new opportunity is on the horizon.",
        "Your hard work will pay off.",
        "You will find success and happiness.",
        "A surprise is waiting for you.",
        "Your dreams will come true."
    ]

    # Use the name length and day of the year to determine the fortune
    index = (name_length + day_of_year) % len(fortunes)
    return fortunes[index]

def main():
    # Get the user's name and birthdate
    name = input("Please enter your name: ")
    birthdate = input("Please enter your birthdate (YYYY-MM-DD): ")

    # Generate and print the fortune
    fortune = generate_fortune(name, birthdate)
    print("Your fortune is:")
    print(fortune)

if __name__ == "__main__":
    main()
```
