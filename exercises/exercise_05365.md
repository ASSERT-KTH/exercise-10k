## Exercise
Write a program to sort an array in ascending or descending order.

## Reference Solution
```python
def sort_array(array, order):
    """
    Sorts an array in ascending or descending order.

    Args:
        array (list): The input array to be sorted.
        order (str): The order of sorting, either 'asc' for ascending or 'desc' for descending.

    Returns:
        list: The sorted array.
    """
    if order == 'asc':
        return sorted(array)
    elif order == 'desc':
        return sorted(array, reverse=True)
    else:
        raise ValueError("Invalid order. Please use 'asc' for ascending or 'desc' for descending.")

# Example usage:
array = [64, 34, 25, 12, 22, 11, 90]
print("Original array:", array)

print("Sorted array in ascending order:", sort_array(array, 'asc'))
print("Sorted array in descending order:", sort_array(array, 'desc'))
```

def bubble_sort(array, order):
    """
    Sorts an array in ascending or descending order using Bubble Sort.

    Args:
        array (list): The input array to be sorted.
        order (str): The order of sorting, either 'asc' for ascending or 'desc' for descending.

    Returns:
        list: The sorted array.
    """
    n = len(array)

    for i in range(n):
        for j in range(0, n - i - 1):
            if order == 'asc':
                if array[j] > array[j + 1]:
                    array[j], array[j + 1] = array[j + 1], array[j]
            elif order == 'desc':
                if array[j] < array[j + 1]:
                    array[j], array[j + 1] = array[j + 1], array[j]
            else:
                raise ValueError("Invalid order. Please use 'asc' for ascending or 'desc' for descending.")

    return array

# Example usage:
array = [64, 34, 25, 12, 22, 11, 90]
print("Original array:", array)

print("Sorted array in ascending order:", bubble_sort(array[:], 'asc'))
print("Sorted array in descending order:", bubble_sort(array[:], 'desc'))
