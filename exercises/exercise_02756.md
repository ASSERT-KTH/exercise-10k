## Exercise
Write a program that calculates the tip for a meal given the total cost and tip percentage.

## Reference Solution
```python
def calculate_tip(total_cost, tip_percentage):
    """
    Calculates the tip for a meal.

    Args:
        total_cost (float): The total cost of the meal.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The calculated tip.
    """
    tip = (total_cost / 100) * tip_percentage
    return tip

def main():
    print("Tip Calculator Program")
    print("------------------------")

    total_cost = float(input("Enter the total cost of the meal: $"))
    tip_percentage = float(input("Enter the tip percentage: "))

    tip = calculate_tip(total_cost, tip_percentage)
    total_amount = total_cost + tip

    print(f"Total Cost: ${total_cost:.2f}")
    print(f"Tip ({tip_percentage}%): ${tip:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
