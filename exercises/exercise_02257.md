## Exercise
Write a program that takes a string and a substring as input and returns whether the string contains the substring.

## Reference Solution
```python
def check_substring(main_string: str, sub_string: str) -> bool:
    """
    Checks if the main string contains the substring.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        bool: True if the main string contains the substring, False otherwise.
    """
    return sub_string in main_string


def main():
    # Example usage
    main_string = input("Enter the main string: ")
    sub_string = input("Enter the substring: ")

    result = check_substring(main_string, sub_string)

    if result:
        print(f'"{sub_string}" is found in "{main_string}"')
    else:
        print(f'"{sub_string}" is not found in "{main_string}"')


if __name__ == "__main__":
    main()
```
