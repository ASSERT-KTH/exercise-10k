    """

    Args:
        input_string (str): The input string to be encoded.

    Returns:
        str: The input string with HTML entities encoded.
    """

    # First, we import the html module which has a function called escape() that can be used to encode HTML entities.
    
    # Now, we define our function encode_html_entities that takes a string as input.
    
    # We use the html.escape() function to replace special characters with their corresponding HTML entities.
    encoded_string = html.escape(input_string)

    return encoded_string
