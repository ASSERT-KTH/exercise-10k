## Exercise
Write a program that calculates the area and perimeter of a rectangle using functions. The program should take the length and width of the rectangle as input and return the calculated area and perimeter.

## Reference Solution
```python
#include <iostream>

using namespace std;

// Function to calculate the area of a rectangle
void area(int l, int w, int &a) {
    a = l * w;
}

// Function to calculate the perimeter of a rectangle
void perimeter(int l, int w, int &p) {
    p = 2 * (l + w);
}

int main() {
    int l, w, a, p;

    // Get user input for length and width
    cout << "Enter the length of the rectangle: ";
    cin >> l;
    cout << "Enter the width of the rectangle: ";
    cin >> w;

    // Calculate the area and perimeter
    area(l, w, a);
    perimeter(l, w, p);

    // Display the results
    cout << "The area of the rectangle is: " << a << endl;
    cout << "The perimeter of the rectangle is: " << p << endl;

    return 0;
}
```
