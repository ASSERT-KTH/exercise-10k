## Exercise
Create a function that treats another function as a first-class citizen

## Reference Solution
```python
def add(a, b):
    """A simple function to add two numbers."""
    return a + b

def subtract(a, b):
    """A simple function to subtract two numbers."""
    return a - b

def calculator(func, a, b):
    """
    A function that treats another function as a first-class citizen.
    
    Args:
        func (function): A function that takes two arguments.
        a (int): The first number.
        b (int): The second number.
    
    Returns:
        result: The result of the function applied to a and b.
    """
    return func(a, b)

# Example usage:
if __name__ == "__main__":
    # Passing a function as an argument
    result_add = calculator(add, 5, 3)
    print(f"5 + 3 = {result_add}")

    # Passing a different function as an argument
    result_subtract = calculator(subtract, 5, 3)
    print(f"5 - 3 = {result_subtract}")

    # Assigning a function to a variable
    operation = add
    result_add = operation(5, 3)
    print(f"5 + 3 = {result_add}")

    # Returning a function from a function
    def get_operation(op):
        if op == "add":
            return add
        elif op == "subtract":
            return subtract

    operation = get_operation("add")
    result_add = operation(5, 3)
    print(f"5 + 3 = {result_add}")
```
