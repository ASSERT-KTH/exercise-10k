## Exercise
Create a function to find the minimum window substring that contains all characters of another string.

## Reference Solution

from collections import defaultdict

def min_window(s, t):
    """
    Find the minimum window substring in string `s` that contains all characters of string `t`.

    Args:
    s (str): The string to search for the minimum window substring.
    t (str): The string containing the characters to search for.

    Returns:
    str: The minimum window substring. If no such substring exists, returns an empty string.
    """
    if not s or not t:
        return ""

    # Create a dictionary to store the frequency of characters in string `t`
    t_count = defaultdict(int)
    for char in t:
        t_count[char] += 1

    # Initialize variables to keep track of the minimum window substring
    required_chars = len(t_count)
    formed_chars = 0

    # Initialize variables to keep track of the window boundaries
    window_counts = defaultdict(int)
    min_window = float("inf")
    min_window_substring = ""

    # Initialize the window boundaries
    left = 0

    # Iterate over the string `s` to find the minimum window substring
    for right in range(len(s)):
        character = s[right]
        window_counts[character] += 1

        # If the frequency of the current character in the window is equal to the frequency in string `t`,
        # increment the `formed_chars` count
        if character in t_count and window_counts[character] == t_count[character]:
            formed_chars += 1

        # While the window contains all characters of string `t` and the left boundary is not at the start of the string,
        # try to shrink the window from the left
        while left <= right and formed_chars == required_chars:
            character = s[left]

            # If the current window is smaller than the minimum window found so far, update the minimum window
            if right - left + 1 < min_window:
                min_window = right - left + 1
                min_window_substring = s[left:right + 1]

            # Decrement the frequency of the character at the left boundary and move the left boundary to the right
            window_counts[character] -= 1
            if character in t_count and window_counts[character] < t_count[character]:
                formed_chars -= 1

            left += 1

    return min_window_substring


# Example usage:
s = "ADOBECODEBANC"
t = "ABC"
print(min_window(s, t))  # Output: "BANC"
