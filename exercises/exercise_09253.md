## Exercise
Write a program that defines and calls multiple functions with the same name but different parameters.

## Reference Solution
```python
from typing import Union

# Define a function with the same name but different parameters
def add(a: Union[int, str], b: Union[int, str]) -> Union[int, str]:
    if isinstance(a, int) and isinstance(b, int):
        # If both parameters are integers, return their sum
        return a + b
    elif isinstance(a, str) and isinstance(b, str):
        # If both parameters are strings, return their concatenation
        return a + b
    else:
        # If the parameters are of different types, raise a TypeError
        raise TypeError("Both parameters must be of the same type")

# Define a main function to test the add function
def main():
    # Call the add function with integer parameters
    result1 = add(5, 7)
    print(f"5 + 7 = {result1}")

    # Call the add function with string parameters
    result2 = add("Hello", " World")
    print(f"'Hello' + ' World' = {result2}")

    # Try calling the add function with parameters of different types
    try:
        result3 = add(5, "Hello")
        print(f"5 + 'Hello' = {result3}")
    except TypeError as e:
        print(f"Error: {e}")

# Call the main function to start the program
if __name__ == "__main__":
    main()
```
