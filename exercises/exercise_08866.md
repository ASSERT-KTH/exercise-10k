## Exercise
Write a function to calculate the sum of amicable numbers less than a given limit

## Reference Solution
```python
def sum_of_proper_divisors(n):
    """Calculate the sum of proper divisors of a number."""
    divisors_sum = 1  # 1 is a divisor of every number
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            divisors_sum += i
            if i != n // i:  # Check if divisors are different
                divisors_sum += n // i
    return divisors_sum

def sum_of_amicable_numbers(limit):
    """Calculate the sum of amicable numbers less than a given limit."""
    amicable_numbers = set()
    for i in range(1, limit):
        sum_of_divisors_i = sum_of_proper_divisors(i)
        if sum_of_divisors_i < limit:
            sum_of_divisors_j = sum_of_proper_divisors(sum_of_divisors_i)
            if sum_of_divisors_j == i and i != sum_of_divisors_i:
                amicable_numbers.add(i)
                amicable_numbers.add(sum_of_divisors_i)
    return sum(amicable_numbers)

# Example usage:
limit = 10000
result = sum_of_amicable_numbers(limit)
print(f"The sum of amicable numbers less than {limit} is: {result}")
```
