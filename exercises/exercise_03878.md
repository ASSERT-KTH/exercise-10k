## Exercise
Create two dictionaries with names of students as keys and their grades as values. Calculate the symmetric difference of the two dictionaries and print the resulting dictionary.

## Reference Solution
```python
def symmetric_difference(dict1, dict2):
    """
    Calculate the symmetric difference of two dictionaries.
    
    The symmetric difference of two dictionaries is a new dictionary with keys that are in exactly one of the input dictionaries.
    
    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.
    
    Returns:
        dict: The symmetric difference of the two input dictionaries.
    """
    # Calculate the symmetric difference of the keys
    symmetric_diff_keys = set(dict1.keys()).symmetric_difference(set(dict2.keys()))
    
    # Create a new dictionary with the symmetric difference keys and their corresponding values
    symmetric_diff_dict = {key: dict1.get(key, dict2.get(key)) for key in symmetric_diff_keys}
    
    return symmetric_diff_dict

# Example usage
students1 = {
    "John": 85,
    "Alice": 90,
    "Bob": 78,
    "Charlie": 92
}

students2 = {
    "John": 85,
    "Alice": 91,
    "David": 88,
    "Eve": 95
}

symmetric_diff = symmetric_difference(students1, students2)

print("Symmetric Difference:")
for key, value in symmetric_diff.items():
    print(f"{key}: {value}")
```
