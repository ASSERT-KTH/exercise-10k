## Exercise
Create a program that calculates the sine, cosine, and tangent of a given angle.

## Reference Solution
```python
import math

def calculate_trigonometry(angle):
    """
    Calculate the sine, cosine, and tangent of a given angle.

    Args:
        angle (float): The angle in degrees.

    Returns:
        tuple: A tuple containing the sine, cosine, and tangent of the angle.
    """
    # Convert the angle from degrees to radians
    angle_in_radians = math.radians(angle)

    # Calculate the sine, cosine, and tangent of the angle
    sine = math.sin(angle_in_radians)
    cosine = math.cos(angle_in_radians)
    tangent = math.tan(angle_in_radians)

    return sine, cosine, tangent


def main():
    # Get the angle from the user
    angle = float(input("Enter an angle in degrees: "))

    # Calculate the trigonometry of the angle
    sine, cosine, tangent = calculate_trigonometry(angle)

    # Print the results
    print(f"The sine of {angle} degrees is {sine:.4f}")
    print(f"The cosine of {angle} degrees is {cosine:.4f}")
    print(f"The tangent of {angle} degrees is {tangent:.4f}")


if __name__ == "__main__":
    main()
```
