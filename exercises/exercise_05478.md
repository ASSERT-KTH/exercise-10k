## Exercise
Write a program that takes an array and a condition as input and partitions the array into two parts: elements that satisfy the condition and elements that do not.

## Reference Solution
```python
function partition<T>(arr: T[], condition: (arg: T) => boolean): [T[], T[]] {
  return arr.reduce(
    ([left, right], current) => {
      if (condition(current)) {
        left.push(current);
      } else {
        right.push(current);
      }
      return [left, right];
    },
    [[], []]
  );
}

// Test the function
const arr = [1, 2, 3, 4, 5];
const condition = (n: number) => n % 2 === 0;
const [even, odd] = partition(arr, condition);
console.log(even); // [2, 4]
console.log(odd); // [1, 3, 5]
```
