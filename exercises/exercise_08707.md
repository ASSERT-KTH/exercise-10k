## Exercise
Create a simple timer that counts down from a given time.

## Reference Solution
```python
import time

def countdown(t):
    """
    Counts down from the given time in seconds.

    Args:
        t (int): Time in seconds.
    """
    while t:
        mins, secs = divmod(t, 60)
        timer = '{:02d}:{:02d}'.format(mins, secs)
        print(timer, end="\r")
        time.sleep(1)
        t -= 1
    print('Time Up!!')

# Example usage
def main():
    t = input("Enter the time in seconds: ")

    # Validate input
    try:
        t = int(t)
        if t <= 0:
            print("Please enter a positive integer.")
        else:
            countdown(t)
    except ValueError:
        print("Invalid input. Please enter a positive integer.")

if __name__ == "__main__":
    main()
```
