## Exercise
Create a program that stores student grades in a dictionary, where the keys are the student names and the values are lists of grades. The program should be able to add a new student, add a grade to an existing student, calculate the average grade of a student, and display all students' grades.

## Reference Solution
```python
class StudentGradebook:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.gradebook = {}

    def add_student(self, name):
        """
        Add a new student to the gradebook.

        Args:
            name (str): The name of the student.
        """
        if name not in self.gradebook:
            self.gradebook[name] = []
            print(f"Student '{name}' added to the gradebook.")
        else:
            print(f"Student '{name}' already exists in the gradebook.")

    def add_grade(self, name, grade):
        """
        Add a grade to an existing student.

        Args:
            name (str): The name of the student.
            grade (float): The grade to be added.
        """
        if name in self.gradebook:
            self.gradebook[name].append(grade)
            print(f"Grade '{grade}' added to student '{name}'.")
        else:
            print(f"Student '{name}' does not exist in the gradebook.")

    def calculate_average(self, name):
        """
        Calculate the average grade of a student.

        Args:
            name (str): The name of the student.

        Returns:
            float: The average grade of the student.
        """
        if name in self.gradebook:
            if self.gradebook[name]:
                return sum(self.gradebook[name]) / len(self.gradebook[name])
            else:
                print(f"Student '{name}' has no grades.")
                return None
        else:
            print(f"Student '{name}' does not exist in the gradebook.")
            return None

    def display_grades(self):
        """
        Display all students' grades.
        """
        for name, grades in self.gradebook.items():
            print(f"Student: {name}")
            if grades:
                print(f"Grades: {grades}")
                print(f"Average: {sum(grades) / len(grades)}")
            else:
                print("No grades.")
            print()


def main():
    gradebook = StudentGradebook()

    while True:
        print("1. Add a new student")
        print("2. Add a grade to an existing student")
        print("3. Calculate the average grade of a student")
        print("4. Display all students' grades")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter the student's name: ")
            gradebook.add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            grade = float(input("Enter the grade: "))
            gradebook.add_grade(name, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            average = gradebook.calculate_average(name)
            if average is not None:
                print(f"The average grade of {name} is {average}.")
        elif choice == "4":
            gradebook.display_grades()
        elif choice == "5":
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
