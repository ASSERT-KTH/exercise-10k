## Exercise
Create a simple quiz that asks the user a series of questions and keeps track of the score.

## Reference Solution
```python
# Import the required modules
import random

# Define a dictionary with quiz questions and answers
quiz_questions = {
    "What is the capital of France?": "Paris",
    "Which planet is known as the Red Planet?": "Mars",
    "Who painted the Starry Night?": "Vincent van Gogh",
    "What is the largest planet in our solar system?": "Jupiter",
    "Who wrote Romeo and Juliet?": "William Shakespeare"
}

def run_quiz():
    # Initialize the score
    score = 0

    # Ask each question and check the answer
    for question, answer in quiz_questions.items():
        user_answer = input(question + " ")
        if user_answer.lower() == answer.lower():
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer is {answer}.\n")

    # Print the final score
    print(f"Quiz complete! Your final score is {score} out of {len(quiz_questions)}.")

def main():
    print("Welcome to the quiz!")
    run_quiz()

if __name__ == "__main__":
    main()
```
