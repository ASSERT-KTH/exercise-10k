## Exercise
Write a program to use multithreading to perform multiple tasks concurrently.

## Reference Solution
```python
import threading
import time
import random

# Function to perform a task
def perform_task(task_id):
    print(f"Task {task_id} started")
    time.sleep(random.randint(1, 3))  # Simulate some work
    print(f"Task {task_id} finished")

# Create and start threads
def main():
    num_tasks = 5
    threads = []

    for i in range(num_tasks):
        thread = threading.Thread(target=perform_task, args=(i,))
        threads.append(thread)
        thread.start()

    # Wait for all threads to finish
    for thread in threads:
        thread.join()

    print("All tasks finished")

if __name__ == "__main__":
    main()
```
