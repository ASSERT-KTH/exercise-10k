## Exercise
Create a program that uses a while loop to estimate the square root of a given number.

## Reference Solution
```python
def estimate_square_root(number, precision=0.00001):
    """
    Estimates the square root of a given number using the Babylonian method.

    Args:
        number (float): The number to estimate the square root of.
        precision (float, optional): The desired precision of the estimate. Defaults to 0.00001.

    Returns:
        float: The estimated square root of the given number.
    """
    if number < 0:
        raise ValueError("Cannot estimate square root of negative number")

    # Initial guess
    guess = number / 2.0

    while True:
        # Calculate the next guess using the Babylonian method
        next_guess = (guess + number / guess) / 2.0

        # Check if the desired precision is reached
        if abs(guess - next_guess) < precision:
            return next_guess

        # Update the guess for the next iteration
        guess = next_guess


def main():
    # Example usage
    number = float(input("Enter a number: "))
    estimated_square_root = estimate_square_root(number)
    print(f"The estimated square root of {number} is {estimated_square_root}")


if __name__ == "__main__":
    main()
```
