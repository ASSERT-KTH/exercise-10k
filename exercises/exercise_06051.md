## Exercise
Create a program that simulates a game show, allowing contestants to answer questions and win prizes using conditional statements.

## Reference Solution
```python
# Game Show Simulator

# Import the random module to generate random questions
import random

# Define a dictionary with questions, answers, and prizes
questions = {
    "What is the capital of France?": {"A": "Paris", "B": "London", "C": "Berlin", "D": "Rome", "correct": "A"},
    "Who painted the Mona Lisa?": {"A": "Leonardo da Vinci", "B": "Michelangelo", "C": "Raphael", "D": "Caravaggio", "correct": "A"},
    "What is the largest planet in our solar system?": {"A": "Earth", "B": "Saturn", "C": "Jupiter", "D": "Uranus", "correct": "C"},
}

# Define a function to play the game
def play_game():
    # Initialize the player's score and prize
    score = 0
    prize = 0

    # Loop through each question
    for question, answers in questions.items():
        # Print the question and answers
        print(question)
        for option, answer in answers.items():
            if option != "correct":
                print(f"{option}: {answer}")

        # Get the player's answer
        player_answer = input("Enter your answer (A, B, C, D): ")

        # Check if the player's answer is correct
        if player_answer.upper() == answers["correct"]:
            print("Correct!")
            score += 1
            # Increase the prize amount
            prize += 1000
            print(f"You've won ${prize} so far!")
        else:
            print(f"Sorry, that's incorrect. The correct answer was {answers[answers['correct']]}")
            print(f"You've won ${prize} so far!")

    # Print the final score and prize
    print(f"Game over! Your final score is {score} out of {len(questions)}")
    print(f"Congratulations, you've won a total of ${prize}!")

# Call the play_game function to start the game
play_game()
```
