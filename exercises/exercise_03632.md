## Exercise
Write a program to find the longest common prefix of two given strings.

## Reference Solution
```python
def longest_common_prefix(str1, str2):
    """
    Returns the longest common prefix of two given strings.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        str: The longest common prefix.
    """
    prefix = ""
    for char1, char2 in zip(str1, str2):
        if char1 == char2:
            prefix += char1
        else:
            break
    return prefix

# Example usage
def main():
    str1 = "hello"
    str2 = "hello world"
    print(f"Longest common prefix of '{str1}' and '{str2}': {longest_common_prefix(str1, str2)}")

    str1 = "apple"
    str2 = "apply"
    print(f"Longest common prefix of '{str1}' and '{str2}': {longest_common_prefix(str1, str2)}")

if __name__ == "__main__":
    main()
```
