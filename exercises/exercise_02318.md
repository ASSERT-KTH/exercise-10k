## Exercise
Write a program that takes in a value and unit (e.g. length, weight) and converts it to a different unit (e.g. from inches to centimeters).

## Reference Solution

def length_conversion(value, from_unit, to_unit):
    """
    Convert length from one unit to another.

    Args:
        value (float): The value to convert.
        from_unit (str): The unit to convert from.
        to_unit (str): The unit to convert to.

    Returns:
        float: The converted value.
    """
    conversions = {
        'inches': {'centimeters': value * 2.54},
        'centimeters': {'inches': value / 2.54},
        'feet': {'meters': value * 0.3048},
        'meters': {'feet': value / 0.3048}
    }

    if from_unit not in conversions or to_unit not in conversions[from_unit]:
        raise ValueError("Unsupported conversion")

    return conversions[from_unit][to_unit]


def weight_conversion(value, from_unit, to_unit):
    """
    Convert weight from one unit to another.

    Args:
        value (float): The value to convert.
        from_unit (str): The unit to convert from.
        to_unit (str): The unit to convert to.

    Returns:
        float: The converted value.
    """
    conversions = {
        'pounds': {'kilograms': value * 0.453592},
        'kilograms': {'pounds': value / 0.453592}
    }

    if from_unit not in conversions or to_unit not in conversions[from_unit]:
        raise ValueError("Unsupported conversion")

    return conversions[from_unit][to_unit]


def temperature_conversion(value, from_unit, to_unit):
    """
    Convert temperature from one unit to another.

    Args:
        value (float): The value to convert.
        from_unit (str): The unit to convert from.
        to_unit (str): The unit to convert to.

    Returns:
        float: The converted value.
    """
    conversions = {
        'fahrenheit': {'celsius': (value - 32) * 5/9},
        'celsius': {'fahrenheit': (value * 9/5) + 32}
    }

    if from_unit not in conversions or to_unit not in conversions[from_unit]:
        raise ValueError("Unsupported conversion")

    return conversions[from_unit][to_unit]


def main():
    print("Unit Conversion Program")
    print("1. Length Conversion")

    print("3. Temperature Conversion")

    choice = int(input("Enter your choice: "))

    if choice == 1:
        value = float(input("Enter the value: "))
        from_unit = input("Enter the unit to convert from (inches, centimeters, feet, meters): ")
        to_unit = input("Enter the unit to convert to (inches, centimeters, feet, meters): ")
        print(f"{value} {from_unit} is equal to {length_conversion(value, from_unit, to_unit)} {to_unit}")

    elif choice == 2:
        value = float(input("Enter the value: "))
        from_unit = input("Enter the unit to convert from (pounds, kilograms): ")
        to_unit = input("Enter the unit to convert to (pounds, kilograms): ")
        print(f"{value} {from_unit} is equal to {weight_conversion(value, from_unit, to_unit)} {to_unit}")

    elif choice == 3:
        value = float(input("Enter the value: "))
        from_unit = input("Enter the unit to convert from (fahrenheit, celsius): ")
        to_unit = input("Enter the unit to convert to (fahrenheit, celsius): ")
        print(f"{value} {from_unit} is equal to {temperature_conversion(value, from_unit, to_unit)} {to_unit}")

    else:
        print("Invalid choice")


if __name__ == "__main__":
    main()
