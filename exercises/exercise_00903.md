## Exercise
Create a program that converts a given variable from one data type to another and prints out the converted value.

## Reference Solution
```python
def convert_data_type(var, target_type):
    """
    Converts a variable from one data type to another.

    Args:
        var (any): The variable to be converted.
        target_type (type): The target data type.

    Returns:
        any: The converted variable.

    Raises:
        ValueError: If the conversion is not possible.
    """
    try:
        if target_type == int:
            return int(var)
        elif target_type == float:
            return float(var)
        elif target_type == str:
            return str(var)
        elif target_type == bool:
            if var.lower() == 'true':
                return True
            elif var.lower() == 'false':
                return False
            else:
                raise ValueError("Cannot convert to bool")
        else:
            raise ValueError("Unsupported target type")
    except ValueError as e:
        print(f"Error: {e}")
        return None


def main():
    # Test the function with different data types
    var_int = 10
    var_float = 10.5
    var_str = "10"
    var_bool = "True"

    print("Converting int to float:", convert_data_type(var_int, float))
    print("Converting float to int:", convert_data_type(var_float, int))
    print("Converting str to int:", convert_data_type(var_str, int))
    print("Converting str to float:", convert_data_type(var_str, float))
    print("Converting str to bool:", convert_data_type(var_bool, bool))

if __name__ == "__main__":
    main()
```
