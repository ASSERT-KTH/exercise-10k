## Exercise
Write a program to find the shortest path between two nodes in a weighted graph using Bellman-Ford algorithm.

## Reference Solution
```python
class Graph:
    def __init__(self, num_vertices):
        self.num_vertices = num_vertices
        self.edges = []

    def add_edge(self, u, v, weight):
        self.edges.append((u, v, weight))

    def bellman_ford(self, source):
        # Initialize distances and predecessors
        distances = [float('inf')] * self.num_vertices
        distances[source] = 0
        predecessors = [None] * self.num_vertices

        # Relax edges repeatedly
        for _ in range(self.num_vertices - 1):
            for u, v, weight in self.edges:
                if distances[u] + weight < distances[v]:
                    distances[v] = distances[u] + weight
                    predecessors[v] = u

        # Check for negative weight cycles
        for u, v, weight in self.edges:
            if distances[u] + weight < distances[v]:
                print("Negative weight cycle detected")
                return None

        return distances, predecessors

    def get_shortest_path(self, source, target):
        distances, predecessors = self.bellman_ford(source)
        if distances is None:
            return None

        path = []
        current = target
        while current is not None:
            path.append(current)
            current = predecessors[current]
        path.reverse()

        return path, distances[target]


# Example usage
if __name__ == "__main__":
    graph = Graph(5)
    graph.add_edge(0, 1, -1)
    graph.add_edge(0, 2, 4)
    graph.add_edge(1, 2, 3)
    graph.add_edge(1, 3, 2)
    graph.add_edge(1, 4, 2)
    graph.add_edge(3, 2, 5)
    graph.add_edge(3, 1, 1)
    graph.add_edge(4, 3, -3)

    source = 0
    target = 4
    path, distance = graph.get_shortest_path(source, target)

    if path is not None:
        print(f"Shortest path from {source} to {target}: {path}")
        print(f"Distance: {distance}")
```
