## Exercise
Write a program that takes two numbers and an operator as input and performs the specified operation.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    This function performs arithmetic operations based on the operator provided.
    
    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The arithmetic operator. Can be '+', '-', '*', '/'.
    
    Returns:
        float: The result of the operation.
    """

    # Check if the operator is valid
    if operator not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operator. Only '+', '-', '*', '/' are allowed.")

    # Perform the operation based on the operator
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operator = input("Enter the operator (+, -, *, /): ")

    try:
        # Calculate and print the result
        result = calculate(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
