## Exercise
Write a program to check if a given linked list is cyclic or not.

## Reference Solution
```python
class Node:
    """Represents a node in the linked list."""
    def __init__(self, data):
        self.data = data
        self.next = None

def is_cyclic(head):
    """
    Checks if the linked list is cyclic.

    Args:
        head (Node): The head node of the linked list.

    Returns:
        bool: True if the linked list is cyclic, False otherwise.
    """
    if head is None or head.next is None:
        return False

    # Initialize two pointers, one moving twice as fast as the other
    tortoise = head
    hare = head

    while hare is not None and hare.next is not None:
        # Move the tortoise one step at a time
        tortoise = tortoise.next
        # Move the hare two steps at a time
        hare = hare.next.next

        # If the tortoise and hare meet, the linked list is cyclic
        if tortoise == hare:
            return True

    return False

def main():
    # Create a cyclic linked list
    node1 = Node(1)
    node2 = Node(2)
    node3 = Node(3)
    node4 = Node(4)

    node1.next = node2
    node2.next = node3
    node3.next = node4
    node4.next = node2  # Create a cycle

    print(is_cyclic(node1))  # Output: True

    # Create a non-cyclic linked list
    node5 = Node(5)
    node6 = Node(6)
    node7 = Node(7)

    node5.next = node6
    node6.next = node7

    print(is_cyclic(node5))  # Output: False

if __name__ == "__main__":
    main()
```
