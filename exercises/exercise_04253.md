## Exercise
Develop a program that allows users to reserve hotel rooms. The program should take the room type, check-in date, and check-out date as input, and print the reservation details.

## Reference Solution
```python
from datetime import datetime

class HotelRoomReservation:
    def __init__(self):
        self.room_types = {
            "single": 100,
            "double": 150,
            "suite": 200
        }
        self.reservations = []

    def make_reservation(self):
        print("Available room types: ")
        for i, (room_type, price) in enumerate(self.room_types.items()):
            print(f"{i+1}. {room_type.capitalize()} - ${price}")

        room_type_choice = input("Enter the number of your preferred room type: ")
        room_type = list(self.room_types.keys())[int(room_type_choice) - 1]

        check_in_date = input("Enter your check-in date (YYYY-MM-DD): ")
        check_out_date = input("Enter your check-out date (YYYY-MM-DD): ")

        try:
            check_in_date = datetime.strptime(check_in_date, "%Y-%m-%d")
            check_out_date = datetime.strptime(check_out_date, "%Y-%m-%d")
        except ValueError:
            print("Invalid date format. Please use YYYY-MM-DD.")
            return

        if check_in_date >= check_out_date:
            print("Check-in date must be before check-out date.")
            return

        reservation = {
            "room_type": room_type,
            "check_in_date": check_in_date,
            "check_out_date": check_out_date
        }

        self.reservations.append(reservation)

        self.print_reservation_details(reservation)

    def print_reservation_details(self, reservation):
        print("\nReservation Details:")
        print(f"Room Type: {reservation['room_type'].capitalize()}")
        print(f"Check-in Date: {reservation['check_in_date'].strftime('%Y-%m-%d')}")
        print(f"Check-out Date: {reservation['check_out_date'].strftime('%Y-%m-%d')}")
        print(f"Total Cost: ${self.room_types[reservation['room_type']] * (reservation['check_out_date'] - reservation['check_in_date']).days}")


def main():
    hotel_reservation = HotelRoomReservation()
    while True:
        print("\nHotel Room Reservation System")
        print("1. Make a reservation")
        print("2. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            hotel_reservation.make_reservation()
        elif choice == "2":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
