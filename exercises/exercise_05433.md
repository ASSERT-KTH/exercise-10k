## Exercise
Create a program that converts color from RGB to HEX or vice versa. If the user chooses to convert from RGB to HEX, use the formula HEX = '#%02x%02x%02x' % (R, G, B). If the user chooses to convert from HEX to RGB, use the formula R, G, B = int(HEX[1:3], 16), int(HEX[3:5], 16), int(HEX[5:7], 16).

## Reference Solution
```python
def rgb_to_hex(r, g, b):
    """
    Converts RGB color to HEX format.

    Args:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)

    Returns:
        str: HEX color code
    """
    return '#%02x%02x%02x' % (r, g, b)


def hex_to_rgb(hex_color):
    """
    Converts HEX color to RGB format.

    Args:
        hex_color (str): HEX color code

    Returns:
        tuple: RGB color components
    """
    hex_color = hex_color.lstrip('#')
    return int(hex_color[0:2], 16), int(hex_color[2:4], 16), int(hex_color[4:6], 16)


def main():
    while True:
        print("Color Converter Menu:")
        print("1. RGB to HEX")
        print("2. HEX to RGB")
        print("3. Quit")
        
        choice = input("Choose an option: ")

        if choice == "1":
            r = int(input("Enter red component (0-255): "))
            g = int(input("Enter green component (0-255): "))
            b = int(input("Enter blue component (0-255): "))
            print(f"HEX color: {rgb_to_hex(r, g, b)}")

        elif choice == "2":
            hex_color = input("Enter HEX color code (e.g., #ffffff): ")
            r, g, b = hex_to_rgb(hex_color)
            print(f"RGB color: ({r}, {g}, {b})")

        elif choice == "3":
            print("Goodbye!")
            break

        else:
            print("Invalid option. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
