## Exercise
Write a program to search for an element in a list.

## Reference Solution
```python
def search_element(lst, target):
    """
    Searches for an element in a list and returns its index if found.

    Args:
        lst (list): The list to search in.
        target: The element to search for.

    Returns:
        int: The index of the target element if found.
        str: A message indicating that the element is not in the list if not found.
    """
    try:
        index = lst.index(target)
        return index
    except ValueError:
        return "Element not found in the list"

def main():
    # Example usage
    my_list = [1, 2, 3, 4, 5]
    target_element = 3
    result = search_element(my_list, target_element)

    if isinstance(result, int):
        print(f"Element {target_element} found at index {result}")
    else:
        print(result)

if __name__ == "__main__":
    main()
```
