## Exercise
Given two strings, find the longest common substring and return it.

## Reference Solution

def longest_common_substring(s1, s2):
    """
    Find the longest common substring between two strings.

    Args:
        s1 (str): The first string.
        s2 (str): The second string.

    Returns:
        str: The longest common substring.
    """
    m = len(s1)
    n = len(s2)

    # Create a 2D table to store the lengths of common substrings
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # Initialize variables to store the maximum length and ending position
    max_length = 0
    end = 0

    # Fill the table in a bottom-up manner
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
                if dp[i][j] > max_length:
                    max_length = dp[i][j]
                    end = i
            else:
                dp[i][j] = 0

    # Extract the longest common substring from the table
    return s1[end - max_length: end]

# Example usage
s1 = "abcdef"
s2 = "zbcdfg"
print(longest_common_substring(s1, s2))  # Output: "bcd"
