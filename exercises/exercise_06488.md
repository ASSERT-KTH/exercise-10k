## Exercise
Write a program that generates a random number between 1 and 10, and then asks the user to guess the number. After each guess, print out a hint, such as 'too high' or 'too low'.

## Reference Solution
```python
import random

def get_random_number(min_value, max_value):
    """Generate a random number between min_value and max_value"""
    return random.randint(min_value, max_value)

def get_user_guess(min_value, max_value):
    """Get the user's guess and validate the input"""
    while True:
        try:
            guess = int(input(f"Guess a number between {min_value} and {max_value}: "))
            if min_value <= guess <= max_value:
                return guess
            else:
                print("Invalid input. Please enter a number within the specified range.")
        except ValueError:
            print("Invalid input. Please enter a whole number.")

def play_game(min_value, max_value):
    """Play the random number guessing game"""
    number_to_guess = get_random_number(min_value, max_value)
    attempts = 0

    while True:
        user_guess = get_user_guess(min_value, max_value)
        attempts += 1

        if user_guess < number_to_guess:
            print("Too low!")
        elif user_guess > number_to_guess:
            print("Too high!")
        else:
            print(f"Congratulations! You found the number in {attempts} attempts.")
            break

def main():
    min_value = 1
    max_value = 10
    play_game(min_value, max_value)

if __name__ == "__main__":
    main()
```
