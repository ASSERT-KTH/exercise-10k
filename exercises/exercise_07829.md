## Exercise
Create a program that simulates a phonebook using a dictionary. The program should allow users to add, remove, and search for contacts. When adding a contact, the program should prompt for the name and phone number. When removing a contact, the program should prompt for the name of the contact to remove. When searching for a contact, the program should prompt for the name and display the corresponding phone number if found.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self):
        """Add a new contact to the phonebook"""
        name = input("Enter the name of the contact: ")
        phone_number = input("Enter the phone number of the contact: ")
        self.contacts[name] = phone_number
        print(f"Contact '{name}' added successfully!")

    def remove_contact(self):
        """Remove a contact from the phonebook"""
        name = input("Enter the name of the contact to remove: ")
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact '{name}' removed successfully!")
        else:
            print(f"Contact '{name}' not found!")

    def search_contact(self):
        """Search for a contact in the phonebook"""
        name = input("Enter the name of the contact to search: ")
        if name in self.contacts:
            print(f"Phone number of '{name}': {self.contacts[name]}")
        else:
            print(f"Contact '{name}' not found!")

    def display_contacts(self):
        """Display all contacts in the phonebook"""
        if not self.contacts:
            print("No contacts in the phonebook!")
        else:
            print("Contacts:")
            for name, phone_number in self.contacts.items():
                print(f"{name}: {phone_number}")


def main():
    phonebook = Phonebook()
    while True:
        print("\nPhonebook Menu:")
        print("1. Add Contact")
        print("2. Remove Contact")
        print("3. Search Contact")
        print("4. Display Contacts")
        print("5. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            phonebook.add_contact()
        elif choice == "2":
            phonebook.remove_contact()
        elif choice == "3":
            phonebook.search_contact()
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
