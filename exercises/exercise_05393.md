## Exercise
Create a function that calculates the total points earned by a football team.

def calculate_football_points(touchdowns, field_goals, extra_points, two_point_conversions):
    """
    Calculate the total points earned by a football team.

    Parameters:
    touchdowns (int): Number of touchdowns scored.
    field_goals (int): Number of field goals scored.
    extra_points (int): Number of extra points scored.
    two_point_conversions (int): Number of two-point conversions scored.

    Returns:
    int: Total points earned by the team.
    """
    # Points per touchdown (including extra point)
    TOUCHDOWN_POINTS = 7
    # Points per field goal
    FIELD_GOAL_POINTS = 3
    # Points per extra point
    EXTRA_POINT_POINTS = 1
    # Points per two-point conversion
    TWO_POINT_CONVERSION_POINTS = 2

    # Calculate points from touchdowns
    touchdown_points = touchdowns * TOUCHDOWN_POINTS
    # Calculate points from field goals
    field_goal_points = field_goals * FIELD_GOAL_POINTS
    # Calculate points from extra points
    extra_point_points = extra_points * EXTRA_POINT_POINTS
    # Calculate points from two-point conversions
    two_point_conversion_points = two_point_conversions * TWO_POINT_CONVERSION_POINTS

    # Calculate total points
    total_points = touchdown_points + field_goal_points + extra_point_points + two_point_conversion_points

    return total_points

# Example usage:
if __name__ == "__main__":
    touchdowns = 3
    field_goals = 2
    extra_points = 1
    two_point_conversions = 1
    total_points = calculate_football_points(touchdowns, field_goals, extra_points, two_point_conversions)
    print(f"Total points earned: {total_points}")
