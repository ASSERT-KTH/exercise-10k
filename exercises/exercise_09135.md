## Exercise
Create a program that takes in a value as input and determines its data type. The program should then print out the data type of the input value.

## Reference Solution
```python
def determine_data_type(input_value):
    """
    Determine the data type of the input value.

    Args:
        input_value: The value to determine the data type for.

    Returns:
        None
    """
    data_type = type(input_value)
    print(f"The data type of the input value is: {data_type.__name__}")


def main():
    # Get user input
    user_input = input("Please enter a value: ")

    # Try to convert the input to different data types
    try:
        # Try to convert to integer
        input_value = int(user_input)
    except ValueError:
        try:
            # Try to convert to float
            input_value = float(user_input)
        except ValueError:
            # If not a number, assume it's a string
            input_value = user_input

    # Determine the data type of the input value
    determine_data_type(input_value)


if __name__ == "__main__":
    main()
```
