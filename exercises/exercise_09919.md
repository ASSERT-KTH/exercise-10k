## Exercise
Write a program that displays the current weather for a given location.

## Reference Solution
```python
import requests
import json

def get_current_weather(api_key, location):
    """
    Fetches the current weather for a given location.

    Args:
        api_key (str): OpenWeatherMap API key
        location (str): City name or zip code

    Returns:
        dict: Current weather data
    """
    base_url = "http://api.openweathermap.org/data/2.5/weather"
    params = {
        "q": location,
        "appid": api_key,
        "units": "metric"  # Use metric units for temperature
    }

    response = requests.get(base_url, params=params)
    response.raise_for_status()  # Raise an exception for bad status codes

    return response.json()

def print_weather(weather_data):
    """
    Prints the current weather data in a human-readable format.

    Args:
        weather_data (dict): Current weather data
    """
    print(f"Weather in {weather_data['name']}:")
    print(f"Temperature: {weather_data['main']['temp']}Â°C")
    print(f"Humidity: {weather_data['main']['humidity']}%")
    print(f"Weather condition: {weather_data['weather'][0]['description']}")

def main():
    api_key = "YOUR_API_KEY_HERE"  # Replace with your OpenWeatherMap API key
    location = input("Enter a city name or zip code: ")

    weather_data = get_current_weather(api_key, location)
    print_weather(weather_data)

if __name__ == "__main__":
    main()
```
