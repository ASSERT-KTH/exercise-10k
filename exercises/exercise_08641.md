## Exercise
Create a function that checks if a given string contains a given substring.

## Reference Solution
```python
def check_substring(main_string: str, sub_string: str) -> bool:
    """
    Checks if a given string contains a specified substring.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        bool: True if the substring is found, False otherwise.
    """
    return sub_string in main_string

# Example usage:
def main():
    main_str = "Hello, World!"
    sub_str = "World"
    print(f"Is '{sub_str}' in '{main_str}'? {check_substring(main_str, sub_str)}")

    main_str = "Hello, Universe!"
    sub_str = "World"
    print(f"Is '{sub_str}' in '{main_str}'? {check_substring(main_str, sub_str)}")

if __name__ == "__main__":
    main()
```
