## Exercise
Create a program that uses a dictionary to store student grades. The program should allow the user to add a student's name and their corresponding grade. It should also be able to calculate the average grade of all students and display the student with the highest grade.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.student_grades = {}

    def add_student(self, name, grade):
        """
        Add a student's name and their corresponding grade to the dictionary.

        Args:
            name (str): Student's name
            grade (float): Student's grade
        """
        self.student_grades[name] = grade
        print(f"Student '{name}' added with grade {grade}.")

    def calculate_average_grade(self):
        """
        Calculate the average grade of all students.

        Returns:
            float: Average grade of all students
        """
        if not self.student_grades:
            print("No students available.")
            return None
        return sum(self.student_grades.values()) / len(self.student_grades)

    def get_highest_grade_student(self):
        """
        Get the student with the highest grade.

        Returns:
            tuple: Student's name and their highest grade
        """
        if not self.student_grades:
            print("No students available.")
            return None
        return max(self.student_grades.items(), key=lambda x: x[1])

    def display_grades(self):
        """
        Display all student grades.
        """
        if not self.student_grades:
            print("No students available.")
            return
        for name, grade in self.student_grades.items():
            print(f"Student: {name}, Grade: {grade}")


def main():
    student_grades = StudentGrades()

    while True:
        print("\n1. Add Student")
        print("2. Calculate Average Grade")
        print("3. Get Student with Highest Grade")
        print("4. Display Student Grades")
        print("5. Exit")
        
        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter student's name: ")
            grade = float(input("Enter student's grade: "))
            student_grades.add_student(name, grade)
        elif choice == "2":
            average_grade = student_grades.calculate_average_grade()
            if average_grade is not None:
                print(f"Average grade: {average_grade}")
        elif choice == "3":
            highest_grade_student = student_grades.get_highest_grade_student()
            if highest_grade_student is not None:
                print(f"Student with highest grade: {highest_grade_student[0]}, Grade: {highest_grade_student[1]}")
        elif choice == "4":
            student_grades.display_grades()
        elif choice == "5":
            break
        else:
            print("Invalid option. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
