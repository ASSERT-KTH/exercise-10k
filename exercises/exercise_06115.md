## Exercise
Write a program that takes an array of integers and a predicate function as input, and returns a new array containing only the integers for which the predicate function returns True.

## Reference Solution
```python
// Helper functions
function isEven(n) {
  return n % 2 === 0;
}

function isOdd(n) {
  return n % 2 !== 0;
}

function isPositive(n) {
  return n > 0;
}

function isNegative(n) {
  return n < 0;
}

function isZero(n) {
  return n === 0;
}

// filterIntegers function
function filterIntegers(integers, predicate) {
  var result = [];
  for (var i = 0; i < integers.length; i++) {
    if (predicate(integers[i])) {
      result.push(integers[i]);
    }
  }
  return result;
}

// Example usage
var numbers = [1, 2, 3, 4, 5, 6];

console.log(filterIntegers(numbers, isEven));
console.log(filterIntegers(numbers, isOdd));
console.log(filterIntegers(numbers, isPositive));
console.log(filterIntegers(numbers, isNegative));
console.log(filterIntegers(numbers, isZero));
```
