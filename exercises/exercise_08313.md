    try:
        data_int = int(data)
        print(f"The integer version of {data} is {data_int}")
    except ValueError:
        print(f"Error: unable to convert {data} to integer")

    try:
        data_float = float(data)
        print(f"The float version of {data} is {data_float}")
    except ValueError:
        print(f"Error: unable to convert {data} to float")

    data_str = str(data)
    print(f"The string version of {data} is {data_str}")
