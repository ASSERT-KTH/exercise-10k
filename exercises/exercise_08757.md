## Exercise
Write a function to split a given string into a list of substrings.

## Reference Solution
```python
def split_string(input_string, separator=None):
    """
    Splits a given string into a list of substrings.

    Args:
        input_string (str): The input string to be split.
        separator (str, optional): The separator to split the string by. Defaults to None.

    Returns:
        list: A list of substrings.
    """
    if separator is None:
        # If no separator is provided, split the string into individual characters
        return list(input_string)
    else:
        # Split the string using the provided separator
        return input_string.split(separator)

# Example usage
if __name__ == "__main__":
    input_str = "hello,world,this,is,a,test"
    separator = ","
    print(split_string(input_str, separator))  # Output: ['hello', 'world', 'this', 'is', 'a', 'test']

    input_str = "HelloWorld"
    print(split_string(input_str))  # Output: ['H', 'e', 'l', 'l', 'o', 'W', 'o', 'r', 'l', 'd']
```
