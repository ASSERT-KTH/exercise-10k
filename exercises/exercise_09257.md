## Exercise
Write a program that resizes an array to a new size and prints the result.

## Reference Solution
```python
import numpy as np

def resize_array(array, new_size):
    """
    Resizes the given array to the specified new size.

    Args:
        array (list): The input array to be resized.
        new_size (int): The new size of the array.

    Returns:
        list: The resized array.
    """

    # Convert the input list to a numpy array
    array = np.array(array)

    # Calculate the number of elements to add or remove
    num_elements_to_add = new_size - len(array)

    # If the new size is larger, add zeros to the end of the array
    if num_elements_to_add > 0:
        array = np.pad(array, (0, num_elements_to_add), mode='constant')

    # If the new size is smaller, remove elements from the end of the array
    elif num_elements_to_add < 0:
        array = array[:new_size]

    return array.tolist()

def main():
    # Example usage
    original_array = [1, 2, 3, 4, 5]
    new_size = 7

    resized_array = resize_array(original_array, new_size)

    print("Original Array:", original_array)
    print("Resized Array:", resized_array)

if __name__ == "__main__":
    main()
```
