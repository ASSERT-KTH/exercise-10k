## Exercise
Create a program to solve the 0/1 knapsack problem using dynamic programming.

def knapsack(weights, values, capacity):
    """
    Solves the 0/1 knapsack problem using dynamic programming.

    Args:
        weights (list): List of item weights.
        values (list): List of item values.
        capacity (int): Knapsack capacity.

    Returns:
        int: Maximum value achievable within the capacity.
        list: List of items to include in the knapsack.
    """
    n = len(weights)
    dp = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]

    # Fill the dp table
    for i in range(1, n + 1):
        for w in range(1, capacity + 1):
            if weights[i - 1] <= w:
                dp[i][w] = max(values[i - 1] + dp[i - 1][w - weights[i - 1]], dp[i - 1][w])
            else:
                dp[i][w] = dp[i - 1][w]

    # Reconstruct the solution
    max_value = dp[n][capacity]
    included_items = []
    w = capacity
    for i in range(n, 0, -1):
        if dp[i][w] != dp[i - 1][w]:
            included_items.append(i - 1)
            w -= weights[i - 1]

    return max_value, included_items[::-1]


# Example usage
weights = [2, 3, 1, 4]
values = [10, 5, 8, 15]
capacity = 7

max_value, included_items = knapsack(weights, values, capacity)
print(f"Maximum value: {max_value}")
print(f"Included items: {included_items}")
