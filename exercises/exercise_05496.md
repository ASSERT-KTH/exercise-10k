## Exercise
Write a program to check if a given graph contains a cycle.

## Reference Solution

from typing import Dict, List

def has_cycle(graph: Dict[int, List[int]]) -> bool:
    """
    Checks if a given graph contains a cycle.

    Args:
    graph: A dictionary representing the graph, where each key is a node and its corresponding value is a list of adjacent nodes.

    Returns:
    bool: True if the graph contains a cycle, False otherwise.
    """
    visited = set()
    recursion_stack = set()

    def dfs(node: int) -> bool:
        visited.add(node)
        recursion_stack.add(node)

        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                if dfs(neighbor):
                    return True
            elif neighbor in recursion_stack:
                return True

        recursion_stack.remove(node)
        return False

    for node in graph:
        if node not in visited:
            if dfs(node):
                return True

    return False


# Example usage
if __name__ == "__main__":
    # Create a sample graph with a cycle
    graph_with_cycle = {
        0: [1, 2],
        1: [2],
        2: [0, 3],
        3: [3]
    }

    # Create a sample graph without a cycle
    graph_without_cycle = {
        0: [1, 2],
        1: [2],
        2: [],
        3: []
    }

    print("Graph with cycle:", has_cycle(graph_with_cycle))  # Expected output: True
    print("Graph without cycle:", has_cycle(graph_without_cycle))  # Expected output: False
