## Exercise
Write a program to demonstrate the basics of database management using a database system.

## Reference Solution
```python
import sqlite3

# Connect to the database
def connect_to_database(db_name):
    """Connect to the SQLite database"""
    try:
        conn = sqlite3.connect(db_name)
        print(f"Connected to {db_name} database")
        return conn
    except sqlite3.Error as e:
        print(f"Error connecting to database: {e}")

# Create a table
def create_table(conn, table_name):
    """Create a table in the database"""
    try:
        cursor = conn.cursor()
        query = f"""CREATE TABLE IF NOT EXISTS {table_name} (
                    id INTEGER PRIMARY KEY,
                    name TEXT NOT NULL,
                    age INTEGER NOT NULL
                )"""
        cursor.execute(query)
        conn.commit()
        print(f"Table {table_name} created")
    except sqlite3.Error as e:
        print(f"Error creating table: {e}")

# Insert data into the table
def insert_data(conn, table_name, data):
    """Insert data into the table"""
    try:
        cursor = conn.cursor()
        query = f"""INSERT INTO {table_name} (name, age) VALUES (?, ?)"""
        cursor.execute(query, data)
        conn.commit()
        print(f"Data inserted into {table_name} table")
    except sqlite3.Error as e:
        print(f"Error inserting data: {e}")

# Update data in the table
def update_data(conn, table_name, id, data):
    """Update data in the table"""
    try:
        cursor = conn.cursor()
        query = f"""UPDATE {table_name} SET name = ?, age = ? WHERE id = ?"""
        cursor.execute(query, (*data, id))
        conn.commit()
        print(f"Data updated in {table_name} table")
    except sqlite3.Error as e:
        print(f"Error updating data: {e}")

# Delete data from the table
def delete_data(conn, table_name, id):
    """Delete data from the table"""
    try:
        cursor = conn.cursor()
        query = f"""DELETE FROM {table_name} WHERE id = ?"""
        cursor.execute(query, (id,))
        conn.commit()
        print(f"Data deleted from {table_name} table")
    except sqlite3.Error as e:
        print(f"Error deleting data: {e}")

# Query the database
def query_database(conn, table_name):
    """Query the database"""
    try:
        cursor = conn.cursor()
        query = f"""SELECT * FROM {table_name}"""
        cursor.execute(query)
        rows = cursor.fetchall()
        print(f"Data from {table_name} table:")
        for row in rows:
            print(row)
    except sqlite3.Error as e:
        print(f"Error querying database: {e}")

# Main function
def main():
    db_name = "example.db"
    table_name = "users"

    # Connect to the database
    conn = connect_to_database(db_name)

    # Create a table
    create_table(conn, table_name)

    # Insert data into the table
    data = ("John Doe", 30)
    insert_data(conn, table_name, data)

    # Update data in the table
    id = 1
    data = ("Jane Doe", 31)
    update_data(conn, table_name, id, data)

    # Query the database
    query_database(conn, table_name)

    # Delete data from the table
    id = 1
    delete_data(conn, table_name, id)

    # Query the database again
    query_database(conn, table_name)

    # Close the connection
    conn.close()

if __name__ == "__main__":
    main()
```
