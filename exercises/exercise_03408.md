## Exercise
Write a program to remove the last inserted key-value pair from a given dictionary.

## Reference Solution
```python
from collections import OrderedDict

def remove_last_inserted(dictionary):
    """
    Remove the last inserted key-value pair from a dictionary.

    Args:
        dictionary (dict): The dictionary from which to remove the last inserted key-value pair.

    Returns:
        dict: The dictionary with the last inserted key-value pair removed.
    """
    # Convert the dictionary to an OrderedDict to maintain insertion order
    ordered_dict = OrderedDict(dictionary)

    # Check if the dictionary is not empty
    if ordered_dict:
        # Remove the last inserted key-value pair
        ordered_dict.popitem(last=True)

    # Convert the OrderedDict back to a regular dictionary
    updated_dict = dict(ordered_dict)

    return updated_dict

# Example usage
if __name__ == "__main__":
    # Create a dictionary
    dictionary = {"a": 1, "b": 2, "c": 3}

    # Print the original dictionary
    print("Original dictionary:")
    print(dictionary)

    # Remove the last inserted key-value pair
    updated_dict = remove_last_inserted(dictionary)

    # Print the updated dictionary
    print("Updated dictionary:")
    print(updated_dict)
```
