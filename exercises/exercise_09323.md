## Exercise
Write a program to demonstrate the correct variable naming convention in programming.

## Reference Solution
```python
# Correct variable naming convention
def calculate_area(length, width):
    """
    Calculate the area of a rectangle.

    Args:
        length (float): The length of the rectangle.
        width (float): The width of the rectangle.

    Returns:
        float: The area of the rectangle.
    """
    area = length * width
    return area

# Incorrect variable naming convention (for demonstration purposes only)
def calculate_area1(x, y):
    """
    Calculate the area of a rectangle.

    Args:
        x (float): The length of the rectangle.
        y (float): The width of the rectangle.

    Returns:
        float: The area of the rectangle.
    """
    area1 = x * y
    return area1

# Example usage
if __name__ == "__main__":
    length = 10.0
    width = 5.0
    area = calculate_area(length, width)
    print(f"The area of the rectangle is: {area}")

    # Using the incorrect function for demonstration purposes
    x = 10.0
    y = 5.0
    area1 = calculate_area1(x, y)
    print(f"The area of the rectangle (incorrect convention) is: {area1}")
```
