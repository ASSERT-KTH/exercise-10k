## Exercise
Create a program that validates a time using if/else statements.

## Reference Solution
```python
def validate_time(hour, minute, second):
    """
    Validate a time in the format HH:MM:SS.

    Args:
        hour (int): The hour of the day (0-23).
        minute (int): The minute of the hour (0-59).
        second (int): The second of the minute (0-59).

    Returns:
        bool: True if the time is valid, False otherwise.
    """
    if not isinstance(hour, int) or not isinstance(minute, int) or not isinstance(second, int):
        return False  # Time components must be integers

    if hour < 0 or hour > 23:
        return False  # Hour must be between 0 and 23

    if minute < 0 or minute > 59:
        return False  # Minute must be between 0 and 59

    if second < 0 or second > 59:
        return False  # Second must be between 0 and 59

    return True  # Time is valid


def main():
    hour = int(input("Enter the hour (0-23): "))
    minute = int(input("Enter the minute (0-59): "))
    second = int(input("Enter the second (0-59): "))

    if validate_time(hour, minute, second):
        print(f"{hour:02d}:{minute:02d}:{second:02d} is a valid time.")
    else:
        print(f"{hour:02d}:{minute:02d}:{second:02d} is not a valid time.")


if __name__ == "__main__":
    main()
```
