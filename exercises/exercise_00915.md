## Exercise
Create a program that detects the data type of a given variable and prints out a message indicating its type.

## Reference Solution
```python
def detect_data_type(var):
    """
    Detects the data type of a given variable and prints out a message indicating its type.

    Args:
        var: The variable to check the data type of.

    Returns:
        None
    """
    data_type = type(var).__name__
    print(f"The variable is of type {data_type}")


# Test the function with different data types
if __name__ == "__main__":
    # Integer
    detect_data_type(10)

    # Float
    detect_data_type(10.5)

    # String
    detect_data_type("Hello, World!")

    # Boolean
    detect_data_type(True)

    # List
    detect_data_type([1, 2, 3, 4, 5])

    # Dictionary
    detect_data_type({"name": "John", "age": 30})

    # Set
    detect_data_type({1, 2, 3, 4, 5})

    # Tuple
    detect_data_type((1, 2, 3, 4, 5))
```
