## Exercise
Write a program that demonstrates the use of tuple operations in programming.

## Reference Solution
```python
def main():
    # Creating tuples
    print("## Creating Tuples")
    tuple1 = (1, 2, 3, 4, 5)
    print(f"Tuple 1: {tuple1}")
    
    tuple2 = ("apple", "banana", "cherry")
    print(f"Tuple 2: {tuple2}")

    # Indexing tuples
    print("\n## Indexing Tuples")
    print(f"First element of Tuple 1: {tuple1[0]}")
    print(f"Last element of Tuple 2: {tuple2[-1]}")

    # Slicing tuples
    print("\n## Slicing Tuples")
    print(f"Middle elements of Tuple 1: {tuple1[1:3]}")
    print(f"All elements of Tuple 2 except the first: {tuple2[1:]}")

    # Concatenating tuples
    print("\n## Concatenating Tuples")
    tuple3 = tuple1 + tuple2
    print(f"Tuple 3 (concatenation of Tuple 1 and Tuple 2): {tuple3}")

    # Tuple multiplication
    print("\n## Tuple Multiplication")
    tuple4 = tuple1 * 2
    print(f"Tuple 4 (Tuple 1 multiplied by 2): {tuple4}")

    # Tuple membership
    print("\n## Tuple Membership")
    print(f"Is 3 in Tuple 1? {3 in tuple1}")
    print(f"Is 'orange' in Tuple 2? {'orange' in tuple2}")

    # Tuple length
    print("\n## Tuple Length")
    print(f"Length of Tuple 1: {len(tuple1)}")
    print(f"Length of Tuple 2: {len(tuple2)}")

    # Tuple max and min
    print("\n## Tuple Max and Min")
    print(f"Max value in Tuple 1: {max(tuple1)}")
    print(f"Min value in Tuple 1: {min(tuple1)}")

if __name__ == "__main__":
    main()
```
