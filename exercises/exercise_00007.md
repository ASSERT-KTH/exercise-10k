## Exercise
Perform common string operations such as concatenation, slicing, and indexing.

## Reference Solution
```python
# String Concatenation
def string_concatenation():
    """Concatenate two strings using the '+' operator."""
    str1 = "Hello"
    str2 = "World"
    result = str1 + " " + str2
    print("Concatenation:", result)

# String Slicing
def string_slicing():
    """Slice a string to extract a subset of characters."""
    str1 = "HelloWorld"
    # Extract the first 5 characters
    slice1 = str1[:5]
    print("Slice 1:", slice1)
    # Extract the last 5 characters
    slice2 = str1[-5:]
    print("Slice 2:", slice2)

# String Indexing
def string_indexing():
    """Access individual characters in a string using indexing."""
    str1 = "HelloWorld"
    # Access the first character
    first_char = str1[0]
    print("First Character:", first_char)
    # Access the last character
    last_char = str1[-1]
    print("Last Character:", last_char)

# String Repetition
def string_repetition():
    """Repeat a string using the '*' operator."""
    str1 = "Hello"
    repeated_str = str1 * 3
    print("Repeated String:", repeated_str)

# String Uppercase and Lowercase Conversion
def string_case_conversion():
    """Convert a string to uppercase or lowercase using the upper() and lower() methods."""
    str1 = "HelloWorld"
    uppercase_str = str1.upper()
    print("Uppercase String:", uppercase_str)
    lowercase_str = str1.lower()
    print("Lowercase String:", lowercase_str)

# Main function with example usage
def main():
    print("String Operations:")
    string_concatenation()
    string_slicing()
    string_indexing()
    string_repetition()
    string_case_conversion()

if __name__ == "__main__":
    main()
```
