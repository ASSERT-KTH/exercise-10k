## Exercise
Write a program that checks if two words are anagrams using a while loop.

## Reference Solution
```python
def are_anagrams(word1, word2):
    """
    Checks if two words are anagrams of each other.

    Args:
        word1 (str): The first word.
        word2 (str): The second word.

    Returns:
        bool: True if the words are anagrams, False otherwise.
    """
    # Convert the words to lowercase and remove any whitespace
    word1 = word1.replace(" ", "").lower()
    word2 = word2.replace(" ", "").lower()

    # If the words have different lengths, they cannot be anagrams
    if len(word1) != len(word2):
        return False

    # Sort the characters in each word
    sorted_word1 = ""
    sorted_word2 = ""
    word1_list = list(word1)
    word2_list = list(word2)
    while word1_list:
        min_index1 = 0
        min_index2 = 0
        for i in range(len(word1_list)):
            if word1_list[i] < word1_list[min_index1]:
                min_index1 = i
        sorted_word1 += word1_list.pop(min_index1)
    while word2_list:
        for i in range(len(word2_list)):
            if word2_list[i] < word2_list[min_index2]:
                min_index2 = i
        sorted_word2 += word2_list.pop(min_index2)

    # Compare the sorted words
    return sorted_word1 == sorted_word2


def main():
    # Example usage:
    word1 = "listen"
    word2 = "silent"
    if are_anagrams(word1, word2):
        print(f'"{word1}" and "{word2}" are anagrams.')
    else:
        print(f'"{word1}" and "{word2}" are not anagrams.')


if __name__ == "__main__":
    main()
```
