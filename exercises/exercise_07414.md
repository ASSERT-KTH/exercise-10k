## Exercise
Create a program that stores student grades in a dictionary. The dictionary should have student names as keys and lists of grades as values. The program should prompt the user to input a student's name and a grade, then add the grade to the student's list of grades in the dictionary. If the student is not already in the dictionary, the program should add them with the given grade as their first grade.

## Reference Solution
```python
# Initialize an empty dictionary to store student grades
student_grades = {}

def add_grade():
    """
    Prompts the user to input a student's name and a grade, 
    then adds the grade to the student's list of grades in the dictionary.
    """
    # Prompt the user to input a student's name
    student_name = input("Please enter the student's name: ")

    # Prompt the user to input a grade
    while True:
        try:
            grade = float(input("Please enter the student's grade: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Check if the student is already in the dictionary
    if student_name in student_grades:
        # If the student is already in the dictionary, append the grade to their list of grades
        student_grades[student_name].append(grade)
    else:
        # If the student is not in the dictionary, add them with the given grade as their first grade
        student_grades[student_name] = [grade]

    # Print a confirmation message
    print(f"Grade added for {student_name}.")

def print_grades():
    """
    Prints out the grades for all students in the dictionary.
    """
    # Check if the dictionary is empty
    if not student_grades:
        print("No grades have been entered.")
    else:
        # Print out the grades for all students
        for student, grades in student_grades.items():
            print(f"{student}: {grades}")

def main():
    # Run the program in a loop until the user chooses to quit
    while True:
        print("\nOptions:")
        print("1. Add a grade")
        print("2. Print all grades")
        print("3. Quit")
        
        # Prompt the user to choose an option
        choice = input("Please choose an option: ")
        
        # Handle the user's choice
        if choice == "1":
            add_grade()
        elif choice == "2":
            print_grades()
        elif choice == "3":
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
