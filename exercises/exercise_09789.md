## Exercise
Write a program that takes a dictionary as input and processes it, then outputs the result.

## Reference Solution
```python
def process_dictionary(input_dict):
    """
    Process a dictionary by counting the occurrences of each value.

    Args:
        input_dict (dict): The input dictionary to process.

    Returns:
        dict: A new dictionary with value counts.
    """
    value_counts = {}
    for key, value in input_dict.items():
        if value in value_counts:
            value_counts[value] += 1
        else:
            value_counts[value] = 1
    return value_counts

def main():
    # Example usage
    input_dict = {
        "a": 1,
        "b": 2,
        "c": 1,
        "d": 3,
        "e": 2,
        "f": 1
    }
    result_dict = process_dictionary(input_dict)
    print("Input Dictionary:")
    print(input_dict)
    print("\nResult Dictionary (Value Counts):")
    print(result_dict)

if __name__ == "__main__":
    main()
```
