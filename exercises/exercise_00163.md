## Exercise
Swap the values of two variables without using a temporary variable.

def swap_variables_arithmetic(a, b):
    """
    Swap the values of two variables without using a temporary variable.
    
    This method uses arithmetic operations to swap the values.
    However, it may cause an overflow for large numbers.
    """
    print("Before swap: a =", a, ", b =", b)
    a = a + b  # Add a and b and store the result in a
    b = a - b  # Subtract b from a (which is a + b) and store the result in b
    a = a - b  # Subtract b (which is the original a) from a (which is a + b) and store the result in a
    print("After swap: a =", a, ", b =", b)

# Example usage:
swap_variables_arithmetic(5, 10)

def swap_variables_xor(a, b):

    Swap the values of two variables without using a temporary variable.
    
    This method uses the XOR operator to swap the values.
    It works by using the properties of XOR: a ^ a = 0 and a ^ 0 = a.

    a = a ^ b  # XOR a and b and store the result in a
    b = a ^ b  # XOR a (which is a ^ b) and b, and store the result in b
    a = a ^ b  # XOR a (which is a ^ b) and b (which is the original a), and store the result in a
    print("After swap: a =", a, ", b =", b)

# Example usage:
swap_variables_xor(5, 10)

def swap_variables_tuple(a, b):
