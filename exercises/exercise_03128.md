## Exercise
Write a program that implements a Caesar Cipher, which shifts each letter in a string by a specified number of positions.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    This function implements a Caesar Cipher, shifting each letter in the input string by the specified number of positions.

    Parameters:
    text (str): The input string to be encrypted.
    shift (int): The number of positions to shift each letter.

    Returns:
    str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character and add the shift
            result += chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character and add the shift
            result += chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
        # If the character is not a letter, add it to the result as is
        else:
            result += char

    return result

def main():
    text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_cipher(text, shift)
    print(f"Text: {text}")
    print(f"Shift: {shift}")
    print(f"Encrypted Text: {encrypted_text}")

if __name__ == "__main__":
    main()
```
