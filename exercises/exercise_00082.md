## Exercise
Given an array of integers, find the maximum contiguous subarray

## Reference Solution

def max_contiguous_subarray(arr):
    """
    Find the maximum contiguous subarray in a given array of integers.

    Args:
        arr (list): A list of integers.

    Returns:
        list: The maximum contiguous subarray.
    """
    if not arr:
        return []

    max_current = max_global = arr[0]
    start = end = temp_start = 0

    for i in range(1, len(arr)):
        if arr[i] > max_current + arr[i]:
            max_current = arr[i]
            temp_start = i
        else:
            max_current += arr[i]

        if max_current > max_global:
            max_global = max_current
            start = temp_start
            end = i

    return arr[start:end+1]

# Example usage
arr = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
print(max_contiguous_subarray(arr))  # Output: [4, -1, 2, 1]
