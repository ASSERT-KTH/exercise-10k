## Exercise
Write a program that plays a tournament of Rock Paper Scissors with the user

## Reference Solution
```python
import random

def get_computer_choice():
    """Get the computer's random choice"""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def determine_winner(user_choice, computer_choice):
    """Determine the winner of a single match"""
    if user_choice == computer_choice:
        return 'tie'
    if (user_choice == 'rock' and computer_choice == 'scissors') or \
       (user_choice == 'paper' and computer_choice == 'rock') or \
       (user_choice == 'scissors' and computer_choice == 'paper'):
        return 'user'
    return 'computer'

def play_tournament(num_matches):
    """Play a tournament of Rock Paper Scissors"""
    user_wins = 0
    computer_wins = 0
    ties = 0

    for i in range(num_matches):
        print(f"\nMatch {i+1}:")
        user_choice = input("Enter your choice (rock, paper, or scissors): ").lower()
        while user_choice not in ['rock', 'paper', 'scissors']:
            user_choice = input("Invalid choice. Please enter rock, paper, or scissors: ").lower()
        computer_choice = get_computer_choice()
        print(f"\nComputer chose: {computer_choice}")

        winner = determine_winner(user_choice, computer_choice)
        if winner == 'tie':
            print("It's a tie!")
            ties += 1
        elif winner == 'user':
            print("You win this match!")
            user_wins += 1
        else:
            print("Computer wins this match!")
            computer_wins += 1

    print(f"\nTournament Results:")
    print(f"User wins: {user_wins}")
    print(f"Computer wins: {computer_wins}")
    print(f"Ties: {ties}")

    if user_wins > computer_wins:
        print("You win the tournament! Congratulations!")
    elif computer_wins > user_wins:
        print("Computer wins the tournament! Better luck next time!")
    else:
        print("It's a tie tournament! Well played!")

def main():
    num_matches = int(input("Enter the number of matches in the tournament: "))
    play_tournament(num_matches)

if __name__ == "__main__":
    main()
```
