## Exercise
Write a program that checks the data type of a given variable and prints out the type.

## Reference Solution
```python
def check_data_type(var):
    """
    This function checks the data type of a given variable and prints out the type.
    
    Parameters:
    var (any): The variable to be checked.
    """
    print(f"The data type of the given variable is: {type(var)}")

# Testing the function with different data types
def main():
    # Integer
    var1 = 10
    print("Variable 1: ", var1)
    check_data_type(var1)

    # Float
    var2 = 10.5
    print("Variable 2: ", var2)
    check_data_type(var2)

    # String
    var3 = "Hello, World!"
    print("Variable 3: ", var3)
    check_data_type(var3)

    # Boolean
    var4 = True
    print("Variable 4: ", var4)
    check_data_type(var4)

    # List
    var5 = [1, 2, 3, 4, 5]
    print("Variable 5: ", var5)
    check_data_type(var5)

    # Dictionary
    var6 = {"name": "John", "age": 30}
    print("Variable 6: ", var6)
    check_data_type(var6)

if __name__ == "__main__":
    main()
```
