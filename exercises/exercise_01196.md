## Exercise
Create a program that calculates the ticket price based on the passenger's age and ticket type.

## Reference Solution
```python
class TicketPriceCalculator:
    def __init__(self):
        self.ticket_types = {
            'economy': 100,
            'premium_economy': 200,
            'business': 500,
            'first_class': 1000
        }
        self.age_discounts = {
            'child': 0.5,
            'senior': 0.75
        }

    def calculate_ticket_price(self, age, ticket_type):
        """
        Calculates the ticket price based on the passenger's age and ticket type.

        Args:
            age (int): The passenger's age.
            ticket_type (str): The type of ticket (economy, premium_economy, business, first_class).

        Returns:
            float: The calculated ticket price.
        """
        base_price = self.ticket_types.get(ticket_type)
        if base_price is None:
            raise ValueError("Invalid ticket type")

        if age < 12:
            discount = self.age_discounts['child']
        elif age > 65:
            discount = self.age_discounts['senior']
        else:
            discount = 1

        ticket_price = base_price * discount
        return ticket_price


def main():
    calculator = TicketPriceCalculator()

    print("Ticket Price Calculator")
    print("------------------------")

    age = int(input("Enter passenger's age: "))
    print("Select ticket type:")
    print("1. Economy ($100)")
    print("2. Premium Economy ($200)")
    print("3. Business ($500)")
    print("4. First Class ($1000)")
    choice = int(input("Enter choice (1/2/3/4): "))

    ticket_types = {
        1: 'economy',
        2: 'premium_economy',
        3: 'business',
        4: 'first_class'
    }

    ticket_type = ticket_types.get(choice)
    if ticket_type is None:
        print("Invalid choice")
        return

    ticket_price = calculator.calculate_ticket_price(age, ticket_type)
    print(f"Ticket price: ${ticket_price:.2f}")


if __name__ == "__main__":
    main()
```
