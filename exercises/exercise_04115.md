## Exercise
Write a program that asks the user for input, validates the input to ensure it is of the correct data type, and prints out an error message if the input is invalid.

## Reference Solution
```python
def get_int_input(prompt):
    """
    Asks the user for integer input and validates it.

    Args:
        prompt (str): The prompt to display to the user.

    Returns:
        int: The user's input as an integer.
    """
    while True:
        try:
            return int(input(prompt))
        except ValueError:
            print("Error: Invalid input. Please enter a valid integer.")

def get_float_input(prompt):
    """
    Asks the user for float input and validates it.

    Args:
        prompt (str): The prompt to display to the user.

    Returns:
        float: The user's input as a float.
    """
    while True:
        try:
            return float(input(prompt))
        except ValueError:
            print("Error: Invalid input. Please enter a valid number.")

def get_string_input(prompt):
    """
    Asks the user for string input and validates it.

    Args:
        prompt (str): The prompt to display to the user.

    Returns:
        str: The user's input as a string.
    """
    while True:
        user_input = input(prompt)
        if user_input:  # Check if the input is not empty
            return user_input
        else:
            print("Error: Invalid input. Please enter a non-empty string.")

def main():
    print("Input Validation Program")
    print("-------------------------")

    # Get integer input
    int_input = get_int_input("Enter an integer: ")
    print(f"You entered: {int_input}")

    # Get float input
    float_input = get_float_input("Enter a float: ")
    print(f"You entered: {float_input}")

    # Get string input
    string_input = get_string_input("Enter a string: ")
    print(f"You entered: {string_input}")

if __name__ == "__main__":
    main()
```
