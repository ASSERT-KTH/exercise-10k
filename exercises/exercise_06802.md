## Exercise
Create a program that uses a dictionary to store student names as keys and their corresponding grades as values. The program should allow users to add students, update grades, and calculate the average grade of all students.

## Reference Solution
```python
# Student Grades Program

# Initialize an empty dictionary to store student grades
student_grades = {}

def add_student():
    """Add a new student to the dictionary"""
    name = input("Enter student name: ")
    grade = float(input("Enter student grade: "))
    student_grades[name] = grade
    print(f"Student '{name}' added with grade {grade}")

def update_grade():
    """Update the grade of an existing student"""
    name = input("Enter student name: ")
    if name in student_grades:
        new_grade = float(input("Enter new grade: "))
        student_grades[name] = new_grade
        print(f"Grade of '{name}' updated to {new_grade}")
    else:
        print(f"Student '{name}' not found")

def calculate_average():
    """Calculate the average grade of all students"""
    if not student_grades:
        print("No students in the dictionary")
        return
    average_grade = sum(student_grades.values()) / len(student_grades)
    print(f"Average grade: {average_grade:.2f}")

def main():
    while True:
        print("\nStudent Grades Program")
        print("1. Add student")
        print("2. Update grade")
        print("3. Calculate average grade")
        print("4. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            add_student()
        elif choice == "2":
            update_grade()
        elif choice == "3":
            calculate_average()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
