## Exercise
Create a function that takes in a dictionary and returns True if the dictionary contains all required keys, and False otherwise

## Reference Solution
```python
def check_required_keys(dictionary, required_keys):
    """
    This function checks if a dictionary contains all required keys.

    Args:
        dictionary (dict): The dictionary to be checked.
        required_keys (list): A list of required keys.

    Returns:
        bool: True if the dictionary contains all required keys, False otherwise.
    """

    # First, we need to check if the input dictionary is actually a dictionary
    if not isinstance(dictionary, dict):
        raise TypeError("The input dictionary must be a dict.")

    # Then, we need to check if the required keys are a list
    if not isinstance(required_keys, list):
        raise TypeError("The required keys must be a list.")

    # Now, we can iterate over each required key
    for key in required_keys:
        # For each required key, we check if it exists in the dictionary
        # If it doesn't exist, we immediately return False
        if key not in dictionary:
            return False

    # If we've checked all required keys and haven't returned False, then the dictionary contains all required keys
    # So, we return True
    return True


# Example usage:
dictionary = {"name": "John", "age": 30, "city": "New York"}
required_keys = ["name", "age", "city"]
print(check_required_keys(dictionary, required_keys))  # Output: True

dictionary = {"name": "John", "age": 30}
required_keys = ["name", "age", "city"]
print(check_required_keys(dictionary, required_keys))  # Output: False
```
