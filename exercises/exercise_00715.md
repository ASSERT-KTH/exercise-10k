## Exercise
Write a function to find the least common value in a dictionary

## Reference Solution
```python
from collections import Counter

def least_common_value(d):
    """
    Find the key with the least common value in a dictionary.

    Args:
        d (dict): Input dictionary.

    Returns:
        tuple: A tuple containing the key and its corresponding value.
    """
    # Count the occurrences of each value
    value_counts = Counter(d.values())

    # Find the minimum count
    min_count = min(value_counts.values())

    # Find the keys with the minimum count
    least_common_keys = [key for key, value in d.items() if value_counts[value] == min_count]

    # Return the first key with the least common value
    return least_common_keys[0], d[least_common_keys[0]]

# Example usage
if __name__ == "__main__":
    d = {'a': 1, 'b': 2, 'c': 1, 'd': 3, 'e': 2, 'f': 4}
    key, value = least_common_value(d)
    print(f"The key with the least common value is: {key} with value {value}")
```
