## Exercise
Write a program that generates a random quote based on user input

## Reference Solution
```python
import random

# Dictionary of quotes, categorized by theme
quotes = {
    "inspiration": [
        "Believe you can and you're halfway there. - Theodore Roosevelt",
        "The only way to do great work is to love what you do. - Steve Jobs",
        "Success is not final, failure is not fatal: It is the courage to continue that counts. - Winston Churchill"
    ],
    "motivation": [
        "You miss 100% of the shots you don't take. - Wayne Gretzky",
        "I have not failed. I've just found 10,000 ways that won't work. - Thomas Edison",
        "Don't watch the clock; do what it does. Keep going. - Sam Levenson"
    ],
    "humor": [
        "I told my wife she was drawing her eyebrows too high. She looked surprised. - Steve Martin",
        "Why don't scientists trust atoms? Because they make up everything. - Unknown",
        "I'm reading a book about anti-gravity. It's impossible to put down. - Steven Wright"
    ]
}

def get_quote(category):
    """Returns a random quote from the specified category"""
    return random.choice(quotes.get(category, ["No quotes found in this category"]))

def main():
    print("Quote Generator")
    print("----------------")
    categories = list(quotes.keys())
    print("Available categories:")
    for i, category in enumerate(categories):
        print(f"{i+1}. {category.capitalize()}")
    
    choice = input("Enter the number of your chosen category: ")
    try:
        choice = int(choice)
        if 1 <= choice <= len(categories):
            category = categories[choice - 1]
            print(f"\nHere's a random quote from the {category} category:")
            print(get_quote(category))
        else:
            print("Invalid choice. Please try again.")
    except ValueError:
        print("Invalid input. Please enter a number.")

if __name__ == "__main__":
    main()
```
