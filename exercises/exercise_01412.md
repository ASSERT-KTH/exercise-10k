## Exercise
Create a program that asks the user a series of questions and keeps track of the score. If the user answers a question correctly, add a point to the score.

## Reference Solution
```python
# Define a dictionary with questions, options, and answers
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "Answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "Answer": "C"
    },
    "Who painted the Starry Night?": {
        "A": "Leonardo da Vinci",
        "B": "Vincent van Gogh",
        "C": "Pablo Picasso",
        "D": "Claude Monet",
        "Answer": "B"
    }
}

def quiz_program():
    # Initialize the score
    score = 0

    # Iterate over the questions
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "Answer":
                print(f"{option}: {value}")
        answer = input("Enter your answer (A, B, C, D): ")
        if answer.upper() == options["Answer"]:
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer is {options['Answer']}.\n")

    # Print the final score
    print(f"Quiz complete! Your final score is {score} out of {len(questions)}.")

# Run the quiz program
if __name__ == "__main__":
    quiz_program()
```
