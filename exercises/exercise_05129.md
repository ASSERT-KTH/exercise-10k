## Exercise
Write a program to find the missing number in an array of consecutive integers.

## Reference Solution
```python
def find_missing_number(nums):
    """
    Find the missing number in an array of consecutive integers.

    Args:
        nums (list): A list of consecutive integers with one number missing.

    Returns:
        int: The missing number in the array.
    """
    n = len(nums) + 1  # The total number of integers in the sequence
    max_num = max(nums)  # The maximum number in the array
    min_num = min(nums)  # The minimum number in the array

    # Calculate the sum of the first n natural numbers
    total_sum = (n * (min_num + max_num)) // 2

    # Calculate the sum of the numbers in the array
    array_sum = sum(nums)

    # The missing number is the difference between the total sum and the array sum
    missing_number = total_sum - array_sum

    return missing_number


# Example usage:
def main():
    nums = [1, 2, 3, 5]
    missing_number = find_missing_number(nums)
    print(f"The missing number in the array is: {missing_number}")


if __name__ == "__main__":
    main()
```
