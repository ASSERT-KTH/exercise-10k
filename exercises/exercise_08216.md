## Exercise
Create a program that finds the maximum number among three given numbers using if/else statements

## Reference Solution
```python
def find_max(a, b, c):
    """
    Finds the maximum number among three given numbers.

    Args:
        a (int): The first number.
        b (int): The second number.
        c (int): The third number.

    Returns:
        int: The maximum number among a, b, and c.
    """
    if a >= b and a >= c:
        return a
    elif b >= a and b >= c:
        return b
    else:
        return c

# Example usage:
def main():
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))
    num3 = int(input("Enter the third number: "))

    max_num = find_max(num1, num2, num3)
    print("The maximum number is:", max_num)

if __name__ == "__main__":
    main()
```
