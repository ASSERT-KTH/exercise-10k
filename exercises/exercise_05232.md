## Exercise
Write a program that takes a time in standard format (HH:MM AM/PM) as input, and prints the equivalent time in military format (HHMM).

## Reference Solution
```python
import re
from datetime import datetime

def convert_to_military_time(time_str):
    """Converts time from standard format (HH:MM AM/PM) to military format (HHMM)"""
    # Regular expression to match the input time format
    pattern = r"(\d{1,2}):(\d{2}) (AM|PM)"
    match = re.match(pattern, time_str)
    
    if match:
        hours, minutes, am_pm = match.groups()
        hours = int(hours)
        minutes = int(minutes)
        
        # Convert to 24-hour format
        if am_pm == "PM" and hours != 12:
            hours += 12
        elif am_pm == "AM" and hours == 12:
            hours = 0
        
        # Format the output time
        military_time = f"{hours:02d}{minutes:02d}"
        return military_time
    else:
        return "Invalid input time format"

def main():
    time_str = input("Enter time in standard format (HH:MM AM/PM): ")
    military_time = convert_to_military_time(time_str)
    print(f"Military time: {military_time}")

if __name__ == "__main__":
    main()
```
