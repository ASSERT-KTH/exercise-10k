## Exercise
Create a program that performs arithmetic operations based on user input. The program should take two numbers and an operator as input and print the result. The operators to support are +, -, *, /, and %. The program should handle invalid inputs and division by zero errors.

## Reference Solution
```python
def get_user_input():
    """
    Retrieves two numbers and an operator from the user.
    
    Returns:
        tuple: A tuple containing two numbers and an operator.
    """

    # Get the first number from the user
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Get the operator from the user
    while True:
        operator = input("Enter the operator (+, -, *, /, %): ")
        if operator in ['+', '-', '*', '/', '%']:
            break
        else:
            print("Invalid operator. Please enter one of the following: +, -, *, /, %")

    # Get the second number from the user
    while True:
        try:
            num2 = float(input("Enter the second number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    return num1, num2, operator


def perform_arithmetic_operation(num1, num2, operator):
    """
    Performs an arithmetic operation based on the provided operator.
    
    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use for the operation.
    
    Returns:
        float: The result of the arithmetic operation.
    """

    # Perform the arithmetic operation based on the operator
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return num1 / num2
    elif operator == '%':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return num1 % num2


def main():
    # Get user input
    num1, num2, operator = get_user_input()

    try:
        # Perform the arithmetic operation
        result = perform_arithmetic_operation(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except ZeroDivisionError as e:
        print(str(e))


if __name__ == "__main__":
    main()
```
