## Exercise
Write a program to perform basic list operations (append, insert, remove, sort) on a given list

## Reference Solution
```python
def main():
    # Initialize a list
    my_list = [1, 3, 5, 7, 9]

    print("Initial List:", my_list)

    # Append an element to the list
    def append_to_list(lst, element):
        """Append an element to the end of the list."""
        lst.append(element)
        return lst

    my_list = append_to_list(my_list, 11)
    print("List after appending 11:", my_list)

    # Insert an element at a specific position
    def insert_at_position(lst, element, position):
        """Insert an element at a specific position in the list."""
        lst.insert(position, element)
        return lst

    my_list = insert_at_position(my_list, 2, 1)
    print("List after inserting 2 at position 1:", my_list)

    # Remove an element from the list
    def remove_from_list(lst, element):
        """Remove the first occurrence of an element from the list."""
        if element in lst:
            lst.remove(element)
        return lst

    my_list = remove_from_list(my_list, 3)
    print("List after removing 3:", my_list)

    # Sort the list in ascending order
    def sort_list(lst):
        """Sort the list in ascending order."""
        lst.sort()
        return lst

    my_list = sort_list(my_list)
    print("List after sorting:", my_list)


if __name__ == "__main__":
    main()
```
