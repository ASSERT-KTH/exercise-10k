## Exercise
Write a function to find the maximum contiguous subarray sum within a one-dimensional array of numbers.

## Reference Solution
```python
def max_contiguous_subarray_sum(array):
    """
    Calculate the maximum contiguous subarray sum within a one-dimensional array of numbers.

    Args:
        array (list): A list of integers.

    Returns:
        int: The maximum contiguous subarray sum.
    """
    if not array:
        raise ValueError("Array cannot be empty")

    max_current = max_global = array[0]
    for i in range(1, len(array)):
        max_current = max(array[i], max_current + array[i])
        if max_current > max_global:
            max_global = max_current

    return max_global

# Example usage
if __name__ == "__main__":
    array = [-2, -3, 4, -1, -2, 1, 5, -3]
    max_sum = max_contiguous_subarray_sum(array)
    print(f"Maximum contiguous subarray sum: {max_sum}")
```
