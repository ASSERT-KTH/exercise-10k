## Exercise
Write a program that performs basic bitwise operations (AND, OR, XOR, NOT) using variables.

#include <stdio.h>

int main() {
    int x = 5;
    int y = 3;

    printf("x = %d\n", x);
    printf("y = %d\n", y);

    printf("x & y = %d\n", x & y);
    printf("x | y = %d\n", x | y);
    printf("x ^ y = %d\n", x ^ y);
    printf("~x = %d\n", ~x);

    return 0;
}
