## Exercise
Create a program that stores student grades in a dictionary. The program should allow users to add, remove, and update student grades. The dictionary keys should be the student names and the values should be another dictionary with the subject as the key and the grade as the value.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        self.grades = {}

    def add_student(self, name):
        """Add a student to the dictionary"""
        if name not in self.grades:
            self.grades[name] = {}
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def remove_student(self, name):
        """Remove a student from the dictionary"""
        if name in self.grades:
            del self.grades[name]
            print(f"Student '{name}' removed successfully.")
        else:
            print(f"Student '{name}' not found.")

    def add_grade(self, name, subject, grade):
        """Add a grade for a student"""
        if name in self.grades:
            if subject in self.grades[name]:
                print(f"Grade for '{subject}' already exists for '{name}'.")
            else:
                self.grades[name][subject] = grade
                print(f"Grade for '{subject}' added for '{name}' successfully.")
        else:
            print(f"Student '{name}' not found.")

    def remove_grade(self, name, subject):
        """Remove a grade for a student"""
        if name in self.grades:
            if subject in self.grades[name]:
                del self.grades[name][subject]
                print(f"Grade for '{subject}' removed for '{name}' successfully.")
            else:
                print(f"No grade found for '{subject}' for '{name}'.")
        else:
            print(f"Student '{name}' not found.")

    def update_grade(self, name, subject, grade):
        """Update a grade for a student"""
        if name in self.grades:
            if subject in self.grades[name]:
                self.grades[name][subject] = grade
                print(f"Grade for '{subject}' updated for '{name}' successfully.")
            else:
                print(f"No grade found for '{subject}' for '{name}'.")
        else:
            print(f"Student '{name}' not found.")

    def display_grades(self):
        """Display all grades"""
        for name, subjects in self.grades.items():
            print(f"Grades for {name}:")
            for subject, grade in subjects.items():
                print(f"{subject}: {grade}")
            print()

def main():
    grades = StudentGrades()

    while True:
        print("1. Add student")
        print("2. Remove student")
        print("3. Add grade")
        print("4. Remove grade")
        print("5. Update grade")
        print("6. Display grades")
        print("7. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter student name: ")
            grades.add_student(name)
        elif choice == "2":
            name = input("Enter student name: ")
            grades.remove_student(name)
        elif choice == "3":
            name = input("Enter student name: ")
            subject = input("Enter subject: ")
            grade = input("Enter grade: ")
            grades.add_grade(name, subject, grade)
        elif choice == "4":
            name = input("Enter student name: ")
            subject = input("Enter subject: ")
            grades.remove_grade(name, subject)
        elif choice == "5":
            name = input("Enter student name: ")
            subject = input("Enter subject: ")
            grade = input("Enter new grade: ")
            grades.update_grade(name, subject, grade)
        elif choice == "6":
            grades.display_grades()
        elif choice == "7":
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
