## Exercise
Generate an arithmetic progression sequence up to a given number

## Reference Solution

def generate_arithmetic_progression(first_term, common_difference, max_value):
    """
    Generate an arithmetic progression sequence up to a given number.

    Args:
        first_term (int): The first term of the sequence.
        common_difference (int): The common difference between each term.
        max_value (int): The maximum value up to which the sequence should be generated.

    Returns:
        list: The generated arithmetic progression sequence.
    """
    sequence = [first_term]
    next_term = first_term + common_difference

    while next_term <= max_value:
        sequence.append(next_term)
        next_term += common_difference

    return sequence


# Example usage:
first_term = 2
common_difference = 3
max_value = 20

sequence = generate_arithmetic_progression(first_term, common_difference, max_value)
print("Arithmetic Progression Sequence:", sequence)
