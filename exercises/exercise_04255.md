## Exercise
Develop a program that allows students to register for university courses. The program should take the student ID, course ID, and semester as input, and print the registration details.

## Reference Solution
```python
class Course:
    def __init__(self, course_id, course_name, semester):
        self.course_id = course_id
        self.course_name = course_name
        self.semester = semester

class Student:
    def __init__(self, student_id, name):
        self.student_id = student_id
        self.name = name
        self.courses = []

    def register_course(self, course):
        self.courses.append(course)
        print(f"Student {self.name} (ID: {self.student_id}) has registered for course {course.course_name} (ID: {course.course_id}) in semester {course.semester}.")

    def print_registration_details(self):
        print(f"Registration details for student {self.name} (ID: {self.student_id}):")
        for course in self.courses:
            print(f"Course ID: {course.course_id}, Course Name: {course.course_name}, Semester: {course.semester}")


def main():
    student_id = input("Enter student ID: ")
    name = input("Enter student name: ")
    student = Student(student_id, name)

    while True:
        print("\n1. Register for a course")
        print("2. Print registration details")
        print("3. Exit")
        choice = input("Enter your choice: ")

        if choice == "1":
            course_id = input("Enter course ID: ")
            course_name = input("Enter course name: ")
            semester = input("Enter semester: ")
            course = Course(course_id, course_name, semester)
            student.register_course(course)
        elif choice == "2":
            student.print_registration_details()
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
