## Exercise
Write a program to evaluate a complex arithmetic expression involving multiple operators and operands.

def evaluate_expression(expression):
    # Initialize an empty stack
    stack = []

    # Iterate over the expression from left to right
    for char in expression:
        # If the character is a digit or a decimal point, treat it as part of a number
        if char.isdigit() or char == '.':
            if stack and isinstance(stack[-1], str) and (stack[-1].replace('.', '', 1).isdigit()):
                stack[-1] += char
            else:
                stack.append(char)
        # If the character is an operator, push it onto the stack
        elif char in '+-*/':
            stack.append(char)
        # If the character is a closing parenthesis, evaluate the expression inside the parentheses
        elif char == ')':
            # Initialize an empty list to store the operands and operators inside the parentheses
            temp_list = []
            # Pop elements from the stack until we find the opening parenthesis
            while stack and stack[-1] != '(':
                temp_list.append(stack.pop())
            # Remove the opening parenthesis from the stack
            if stack and stack[-1] == '(':
                stack.pop()
            # Evaluate the expression inside the parentheses
            temp_expression = ''.join(reversed(temp_list))
            result = eval(temp_expression)
            # Push the result onto the stack
            stack.append(str(result))

    # Evaluate the final expression
    result = eval(''.join(stack))

    return result


# Test the function with a complex arithmetic expression
expression = "10 + 20 * 3 - 4 / 2"
print("Expression:", expression)
print("Result:", evaluate_expression(expression))
