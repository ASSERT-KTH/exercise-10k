## Exercise
Write a program that asks the user for a Japanese word and determines whether it is affected by the rendaku (sequential voicing) rule, which changes the initial consonant of the second part of a compound word from voiceless to voiced.

def is_rendaku_affected(word):
    """
    Determines whether a Japanese word is affected by the rendaku rule.
    
    Args:
    word (str): A Japanese word in romaji or kana.
    
    Returns:
    bool: True if the word is affected by the rendaku rule, False otherwise.
    """
    # Define the rendaku rule changes
    rendaku_changes = {
        'k': 'g',
        't': 'd',
        'p': 'b',
        's': 'z',
        'h': 'b',  # Only applies to some words
        'm': 'm',  # Not changed by rendaku
        'y': 'y',  # Not changed by rendaku
        'r': 'r',  # Not changed by rendaku
        'n': 'n',  # Not changed by rendaku
        'w': 'w',  # Not changed by rendaku
    }
    
    # Convert word to kana if it's in romaji
    # Note: This is a simplified conversion and may not cover all cases
    romaji_to_kana = {
        'ka': 'か', 'ki': 'き', 'ku': 'く', 'ke': 'け', 'ko': 'こ',
        'ta': 'た', 'ti': 'ち', 'tu': 'つ', 'te': 'て', 'to': 'と',
        'pa': 'ぱ', 'pi': 'ぴ', 'pu': 'ぷ', 'pe': 'ぺ', 'po': 'ぽ',
        'sa': 'さ', 'si': 'し', 'su': 'す', 'se': 'せ', 'so': 'そ',
        'ha': 'は', 'hi': 'ひ', 'hu': 'ふ', 'he': 'へ', 'ho': 'ほ',
        'ma': 'ま', 'mi': 'み', 'mu': 'む', 'me': 'め', 'mo': 'も',
        'ya': 'や', 'yu': 'ゆ', 'yo': 'よ',
        'ra': 'ら', 'ri': 'り', 'ru': 'る', 're': 'れ', 'ro': 'ろ',
        'wa': 'わ', 'wo': 'を', 'n': 'ん',
    }
    if any(char in 'abcdefghijklmnopqrstuvwxyz' for char in word):
        # Convert romaji to kana
        kana_word = ''
        i = 0
        while i < len(word):
            if word[i:i+2] in romaji_to_kana:
                kana_word += romaji_to_kana[word[i:i+2]]
                i += 2
            elif word[i] in romaji_to_kana:
                kana_word += romaji_to_kana[word[i]]
                i += 1
            else:
                i += 1
        word = kana_word
    
    # Check if the word is a compound word
    if len(word) < 4:
        return False
    
    # Check if the second part of the word starts with a voiceless consonant
    second_part = word[2:]
    if second_part[0] not in 'かきくけこさしすせそたちつてとはひふへほ':
        return False
    
    # Check if the word is affected by the rendaku rule
    for change in rendaku_changes:
        if second_part[0] == change:
            return True
    
    return False

def main():
    word = input("Enter a Japanese word: ")
    if is_rendaku_affected(word):
        print(f"The word '{word}' is affected by the rendaku rule.")
    else:
        print(f"The word '{word}' is not affected by the rendaku rule.")

if __name__ == "__main__":
    main()
