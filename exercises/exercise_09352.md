## Exercise
Write a program that converts a string to an integer and vice versa.

## Reference Solution
```python
def string_to_int(s):
    """
    Converts a string to an integer.
    
    Args:
        s (str): The string to be converted.
    
    Returns:
        int: The integer representation of the string.
    """
    try:
        return int(s)
    except ValueError:
        print("Invalid input. Please enter a string that can be converted to an integer.")
        return None


def int_to_string(n):
    """
    Converts an integer to a string.
    
    Args:
        n (int): The integer to be converted.
    
    Returns:
        str: The string representation of the integer.
    """
    return str(n)


def main():
    # Example usage:
    s = "123"
    print(f"String '{s}' converted to integer: {string_to_int(s)}")

    n = 456
    print(f"Integer {n} converted to string: {int_to_string(n)}")

    # Error handling example:
    invalid_s = "abc"
    print(f"String '{invalid_s}' converted to integer: {string_to_int(invalid_s)}")


if __name__ == "__main__":
    main()
```
