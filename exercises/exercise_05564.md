## Exercise
Write a program that takes a story template with placeholders for nouns, verbs, and adjectives, and asks the user to input words to fill in the placeholders.

## Reference Solution
```python
# Program to fill in a story template with user input

# Story template with placeholders for nouns, verbs, and adjectives
story_template = '''
Once upon a time, in a {adjective1} land, there lived a {adjective2} {noun1}. 
This {noun1} loved to {verb1} in the {adjective3} sun. 
One day, while {verb2}, the {noun1} stumbled upon a {adjective4} {noun2}. 
The {noun1} was so {adjective5} that it decided to {verb3} the {noun2} home.
'''

def get_user_input():
    # Dictionary to store user input
    user_input = {}

    # Ask user to input words to fill in the placeholders
    user_input['adjective1'] = input("Enter an adjective (e.g., beautiful, happy, tall): ")
    user_input['adjective2'] = input("Enter an adjective (e.g., big, fluffy, green): ")
    user_input['noun1'] = input("Enter a noun (e.g., cat, dog, elephant): ")
    user_input['verb1'] = input("Enter a verb ending in -ing (e.g., running, jumping, singing): ")
    user_input['adjective3'] = input("Enter an adjective (e.g., bright, warm, sunny): ")
    user_input['verb2'] = input("Enter a verb ending in -ing (e.g., walking, reading, dancing): ")
    user_input['adjective4'] = input("Enter an adjective (e.g., shiny, colorful, soft): ")
    user_input['noun2'] = input("Enter a noun (e.g., ball, book, toy): ")
    user_input['adjective5'] = input("Enter an adjective (e.g., excited, happy, surprised): ")
    user_input['verb3'] = input("Enter a verb (e.g., take, bring, carry): ")

    return user_input

def fill_in_template(template, user_input):
    # Fill in the story template with user input
    story = template.format(**user_input)

    return story

def main():
    print("Welcome to the story generator!")
    user_input = get_user_input()
    story = fill_in_template(story_template, user_input)

    print("\nHere's your story:")
    print(story)

if __name__ == "__main__":
    main()
```
