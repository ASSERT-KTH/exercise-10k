## Exercise
Create a program that checks if an ISBN number is valid or not. If the ISBN number contains 13 digits and the check digit is correct, it is valid. If the ISBN number does not meet these conditions, it is invalid.

## Reference Solution
```python
def is_valid_isbn(isbn: str) -> bool:
    """
    Checks if the given ISBN number is valid or not.
    An ISBN number is valid if it contains 13 digits and its check digit is correct.
    
    Args:
        isbn (str): The ISBN number to be checked.
    
    Returns:
        bool: True if the ISBN number is valid, False otherwise.
    """

    # Remove any hyphens from the ISBN number
    isbn = isbn.replace("-", "")

    # Check if the ISBN number contains exactly 13 digits
    if not isbn.isdigit() or len(isbn) != 13:
        return False

    # Calculate the check digit
    check_digit = 0
    for i in range(12):
        digit = int(isbn[i])
        if i % 2 == 0:
            check_digit += digit
        else:
            check_digit += digit * 3

    # Calculate the remainder of the check digit when divided by 10
    remainder = check_digit % 10

    # If the remainder is 0, the check digit is 0, otherwise it's 10 - remainder
    check_digit = 0 if remainder == 0 else 10 - remainder

    # Check if the calculated check digit matches the last digit of the ISBN number
    return check_digit == int(isbn[-1])


def main():
    # Test the function
    isbn_numbers = ["978-3-16-148410-0", "9783161484100", "978-3-16-148410-1", "1234567890123"]
    for isbn in isbn_numbers:
        print(f"ISBN: {isbn}, Valid: {is_valid_isbn(isbn)}")


if __name__ == "__main__":
    main()
```
