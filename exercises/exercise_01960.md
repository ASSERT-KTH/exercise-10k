## Exercise
Write a program that takes two lists as input and returns the linear regression line that best fits the data.

## Reference Solution
```python
import numpy as np
import matplotlib.pyplot as plt

def linear_regression(x, y):
    """
    Calculate the linear regression line that best fits the data.

    Parameters:
    x (list): Independent variable data
    y (list): Dependent variable data

    Returns:
    tuple: (slope, intercept) of the linear regression line
    """
    # Convert input lists to numpy arrays
    x = np.array(x)
    y = np.array(y)

    # Calculate the coefficients (slope and intercept) using numpy's polyfit function
    coefficients = np.polyfit(x, y, 1)

    return coefficients

def plot_regression_line(x, y, coefficients):
    """
    Plot the data points and the linear regression line.

    Parameters:
    x (list): Independent variable data
    y (list): Dependent variable data
    coefficients (tuple): (slope, intercept) of the linear regression line
    """
    # Generate x values for the regression line
    x_regression = np.linspace(x.min(), x.max(), 100)

    # Calculate corresponding y values for the regression line
    y_regression = coefficients[0] * x_regression + coefficients[1]

    # Plot the data points and the regression line
    plt.scatter(x, y, label='Data Points')
    plt.plot(x_regression, y_regression, label='Linear Regression Line', color='red')
    plt.xlabel('Independent Variable')
    plt.ylabel('Dependent Variable')
    plt.legend()
    plt.show()

# Example usage
x = [1, 2, 3, 4, 5]
y = [2, 3, 5, 7, 11]

coefficients = linear_regression(x, y)
print(f"Linear Regression Line: y = {coefficients[0]:.2f}x + {coefficients[1]:.2f}")

plot_regression_line(x, y, coefficients)
```
