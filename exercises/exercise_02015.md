## Exercise
Write a program that takes a student's grades as input and returns their average grade using conditional statements.

## Reference Solution
```python
#include <stdio.h>

int main() {
    int n, i, nota;
    float media = 0;

    printf("Entre com o numero de notas: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++) {
        printf("Entre com a nota %d: ", i);
        scanf("%d", &nota);
        media += nota;
    }

    media = media / n;

    if (media >= 7) {
        printf("Aprovado com media %.2f\n", media);
    } else if (media >= 3) {
        printf("Prova final com media %.2f\n", media);
    } else {
        printf("Reprovado com media %.2f\n", media);
    }

    return 0;
}
```
