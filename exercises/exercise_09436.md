## Exercise
Create a function that uses coroutines to asynchronously compute the sum of a list of numbers

## Reference Solution
```python
import asyncio

async def sum_numbers(numbers):
    """
    Asynchronously computes the sum of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The sum of the numbers in the list.
    """
    # Initialize the sum to 0
    total_sum = 0
    
    # Iterate over the numbers in the list
    for num in numbers:
        # Simulate an asynchronous operation (e.g., I/O-bound task)
        await asyncio.sleep(0.1)
        
        # Add the current number to the total sum
        total_sum += num
    
    # Return the total sum
    return total_sum

async def main():
    # Define a list of numbers
    numbers = [1, 2, 3, 4, 5]
    
    # Asynchronously compute the sum of the numbers
    result = await sum_numbers(numbers)
    
    # Print the result
    print(f"The sum of the numbers is: {result}")

# Run the main coroutine
asyncio.run(main())
```
