## Exercise
Write a program that calculates the mean, median, and mode of a list of numbers using for loops.

## Reference Solution
```python
def calculate_mean(numbers):
    """
    Calculate the mean of a list of numbers.
    """
    total = 0
    for num in numbers:
        total += num
    return total / len(numbers)

def calculate_median(numbers):
    """
    Calculate the median of a list of numbers.
    """
    # First, we need to sort the list
    for i in range(len(numbers)):
        for j in range(i + 1, len(numbers)):
            if numbers[i] > numbers[j]:
                # Swap the numbers
                numbers[i], numbers[j] = numbers[j], numbers[i]
    
    # If the list has an odd number of elements, the median is the middle element
    if len(numbers) % 2 == 1:
        return numbers[len(numbers) // 2]
    # If the list has an even number of elements, the median is the average of the two middle elements
    else:
        mid1 = numbers[len(numbers) // 2 - 1]
        mid2 = numbers[len(numbers) // 2]
        return (mid1 + mid2) / 2

def calculate_mode(numbers):
    """
    Calculate the mode of a list of numbers.
    """
    # Create a dictionary to store the frequency of each number
    frequency = {}
    for num in numbers:
        if num in frequency:
            frequency[num] += 1
        else:
            frequency[num] = 1
    
    # Find the number with the highest frequency
    max_frequency = 0
    modes = []
    for num, freq in frequency.items():
        if freq > max_frequency:
            max_frequency = freq
            modes = [num]
        elif freq == max_frequency:
            modes.append(num)
    
    return modes

def main():
    numbers = [1, 2, 3, 4, 4, 5, 5, 5]
    print(f"Mean: {calculate_mean(numbers)}")
    print(f"Median: {calculate_median(numbers)}")
    print(f"Mode: {calculate_mode(numbers)}")

if __name__ == "__main__":
    main()
```
