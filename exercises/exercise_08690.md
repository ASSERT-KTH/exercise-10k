## Exercise
Write a program that solves a simple maze and finds the shortest path to the exit.

## Reference Solution

from collections import deque

def solve_maze(maze):
    """
    Solves the maze using BFS and returns the shortest path to the exit.
    """
    # Find the entrance
    entrance = None
    for i in range(len(maze)):
        for j in range(len(maze[i])):
            if maze[i][j] == 'E':
                entrance = (i, j)
                break
        if entrance:
            break

    # Directions to move in the maze
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

    # Queue for BFS
    queue = deque([(entrance, [])])

    # Set to keep track of visited cells
    visited = set([entrance])

    while queue:
        (x, y), path = queue.popleft()

        # If we've reached the exit, return the path
        if maze[x][y] == 'X':
            return path + [(x, y)]

        # Explore neighbors
        for dx, dy in directions:
            nx, ny = x + dx, y + dy

            # If the neighbor is within bounds and not a wall
            if (0 <= nx < len(maze)) and (0 <= ny < len(maze[nx])) and maze[nx][ny] != '#':
                # If the neighbor has not been visited
                if (nx, ny) not in visited:
                    queue.append(((nx, ny), path + [(x, y)]))
                    visited.add((nx, ny))

    # If there's no path to the exit
    return None

def print_maze(maze, path=None):
    """
    Prints the maze with the path highlighted.
    """
    for i in range(len(maze)):
        for j in range(len(maze[i])):
            if path and (i, j) in path:
                print('*', end=' ')
            else:
                print(maze[i][j], end=' ')
        print()

# Example usage
maze = [
    ['#', '#', '#', '#', '#', '#', '#'],
    ['#', 'E', ' ', ' ', ' ', ' ', '#'],
    ['#', '#', '#', ' ', '#', ' ', '#'],
    ['#', ' ', ' ', ' ', '#', ' ', '#'],
    ['#', ' ', '#', ' ', '#', ' ', '#'],
    ['#', ' ', ' ', ' ', ' ', 'X', '#'],
    ['#', '#', '#', '#', '#', '#', '#']
]

path = solve_maze(maze)
print("Path to the exit:")
print_maze(maze, path)
