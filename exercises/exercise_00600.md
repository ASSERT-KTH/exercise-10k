## Exercise
Write a program that takes a string of words as input and returns the most frequent word in the string.

## Reference Solution
```python
from collections import Counter
import re

def most_frequent_word(input_string):
    """
    Returns the most frequent word in the input string.

    Args:
        input_string (str): The input string of words.

    Returns:
        str: The most frequent word in the string.
    """
    # Convert the string to lowercase and remove punctuation
    cleaned_string = re.sub(r'[^\w\s]', '', input_string.lower())

    # Split the string into words
    words = cleaned_string.split()

    # Count the frequency of each word
    word_counts = Counter(words)

    # Return the most common word
    return word_counts.most_common(1)[0][0]

def main():
    # Example usage
    input_string = "This is a test string. The string is just a test."
    most_common = most_frequent_word(input_string)
    print(f"The most frequent word is: {most_common}")

if __name__ == "__main__":
    main()
```
