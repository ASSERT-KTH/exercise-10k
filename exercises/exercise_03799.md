## Exercise
Create a function that finds the minimum number of operations needed to transform one string into another.

## Reference Solution

def levenshtein_distance(s1, s2):
    """
    Calculate the Levenshtein distance between two strings.
    
    Args:
        s1 (str): The first string.
        s2 (str): The second string.
    
    Returns:
        int: The minimum number of operations needed to transform s1 into s2.
    """
    
    # Initialize a matrix to store the Levenshtein distances between substrings
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    # Initialize the base cases
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    
    # Fill in the rest of the matrix
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            cost = 0 if s1[i - 1] == s2[j - 1] else 1
            dp[i][j] = min(dp[i - 1][j] + 1,      # Deletion
                           dp[i][j - 1] + 1,      # Insertion
                           dp[i - 1][j - 1] + cost)  # Substitution or no operation
    
    # The Levenshtein distance is stored in the bottom-right corner of the matrix
    return dp[m][n]

# Example usage
if __name__ == "__main__":
    s1 = "kitten"
    s2 = "sitting"
    distance = levenshtein_distance(s1, s2)
    print(f"The minimum number of operations needed to transform '{s1}' into '{s2}' is {distance}.")
