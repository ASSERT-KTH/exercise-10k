## Exercise
Write a program that uses escape sequences to insert special characters into a string.

## Reference Solution
```python
def main():
    # Using escape sequence to insert a newline character
    newline_string = "Hello\nWorld"
    print("Newline String:")
    print(newline_string)

    # Using escape sequence to insert a tab character
    tab_string = "Name\tAge"
    print("\nTab String:")
    print(tab_string)

    # Using escape sequence to insert a backslash character
    backslash_string = "C:\\\\Windows"
    print("\nBackslash String:")
    print(backslash_string)

    # Using escape sequence to insert a single quote character
    single_quote_string = "It\'s a beautiful day"
    print("\nSingle Quote String:")
    print(single_quote_string)

    # Using escape sequence to insert a double quote character
    double_quote_string = "He said, \"Hello, World!\""
    print("\nDouble Quote String:")
    print(double_quote_string)

if __name__ == "__main__":
    main()
```
