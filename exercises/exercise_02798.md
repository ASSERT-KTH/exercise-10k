## Exercise
Write a program that takes two arrays as input and returns a new array with the elements of the two arrays paired together.

## Reference Solution
```python
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int l[10], m[10];
    int n, i, j, k;

    printf("Enter the number of elements in the first array: ");
    scanf("%d", &n);

    printf("Enter the elements of the first array:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &l[i]);
    }

    printf("Enter the number of elements in the second array: ");
    scanf("%d", &j);

    printf("Enter the elements of the second array:\n");
    for (k = 0; k < j; k++) {
        scanf("%d", &m[k]);
    }

    printf("First array: ");
    for (i = 0; i < n; i++) {
        printf("%d ", l[i]);
    }
    printf("\n");

    printf("Second array: ");
    for (k = 0; k < j; k++) {
        printf("%d ", m[k]);
    }
    printf("\n");

    printf("Paired array: ");
    for (i = 0; i < n && i < j; i++) {
        printf("(%d, %d) ", l[i], m[i]);
    }
    printf("\n");

    return 0;
}
```
