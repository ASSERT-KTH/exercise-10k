## Exercise
Write a program to demonstrate how memory is allocated for variables of different data types.

## Reference Solution
```python
import sys

def print_memory_allocation():
    # Integer
    integer_var = 10
    print(f"Memory allocated for integer {integer_var}: {sys.getsizeof(integer_var)} bytes")

    # Float
    float_var = 10.5
    print(f"Memory allocated for float {float_var}: {sys.getsizeof(float_var)} bytes")

    # String
    string_var = "Hello, World!"
    print(f"Memory allocated for string '{string_var}': {sys.getsizeof(string_var)} bytes")

    # Boolean
    bool_var = True
    print(f"Memory allocated for boolean {bool_var}: {sys.getsizeof(bool_var)} bytes")

    # List
    list_var = [1, 2, 3, 4, 5]
    print(f"Memory allocated for list {list_var}: {sys.getsizeof(list_var)} bytes")

    # Dictionary
    dict_var = {"name": "John", "age": 30}
    print(f"Memory allocated for dictionary {dict_var}: {sys.getsizeof(dict_var)} bytes")

    # Tuple
    tuple_var = (1, 2, 3, 4, 5)
    print(f"Memory allocated for tuple {tuple_var}: {sys.getsizeof(tuple_var)} bytes")

    # Set
    set_var = {1, 2, 3, 4, 5}
    print(f"Memory allocated for set {set_var}: {sys.getsizeof(set_var)} bytes")

# Call the function
print_memory_allocation()
```
