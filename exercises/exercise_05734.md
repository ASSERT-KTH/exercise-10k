## Exercise
Create a function that takes a function as an argument and returns a new function that wraps the input function with additional functionality.

# Import the time module to measure execution time
import time
from functools import wraps

# Define a function that takes a function as an argument and returns a new function
def timer_decorator(func):
    """
    A decorator function that measures the execution time of the input function.
    
    Args:
        func (function): The input function to be wrapped.
    
    Returns:
        function: A new function that wraps the input function with additional functionality.
    """
    
    # Use the @wraps decorator to preserve the metadata of the input function
    @wraps(func)
    def wrapper(*args, **kwargs):
        # Record the start time before calling the input function
        start_time = time.time()
        
        # Try to call the input function with the given arguments
        try:
            result = func(*args, **kwargs)
        except Exception as e:
            # If an exception occurs, print the error message and re-raise the exception
            print(f"An error occurred: {e}")
            raise
        
        # Record the end time after the input function has finished executing
        end_time = time.time()
        
        # Calculate the execution time by subtracting the start time from the end time
        execution_time = end_time - start_time
        
        # Print the execution time
        print(f"Function '{func.__name__}' executed in {execution_time:.4f} seconds.")
        
        # Return the result of the input function
        return result
    
    # Return the wrapper function
    return wrapper

# Example usage:
@timer_decorator
def example_function():
    # Simulate some work
    time.sleep(1)
    print("Example function executed.")

# Call the example function
example_function()
