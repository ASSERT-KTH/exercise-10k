## Exercise
Write a program that takes a list as input, and returns its mean, median, and standard deviation.

## Reference Solution
```python
import numpy as np

def calculate_statistics(input_list):
    """
    Calculate the mean, median, and standard deviation of a list.

    Args:
    input_list (list): A list of numbers.

    Returns:
    tuple: A tuple containing the mean, median, and standard deviation.
    """
    # Calculate the mean
    mean = np.mean(input_list)
    
    # Calculate the median
    median = np.median(input_list)
    
    # Calculate the standard deviation
    std_dev = np.std(input_list)
    
    return mean, median, std_dev

def main():
    # Test the function with a sample list
    input_list = [1, 2, 3, 4, 5]
    mean, median, std_dev = calculate_statistics(input_list)
    
    print(f"Input List: {input_list}")
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Standard Deviation: {std_dev}")

if __name__ == "__main__":
    main()
```
