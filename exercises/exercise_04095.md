## Exercise
Write an async function that waits for 1 second and then returns a message

## Reference Solution
```python
/**
 * Asynchronously waits for a specified amount of time before returning a message.
 * @param {number} waitTimeMs - The time to wait in milliseconds.
 * @param {string} message - The message to return after waiting.
 * @returns {Promise<string>} A promise that resolves with the message after waiting.
 */
async function waitForMessage(waitTimeMs, message) {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve(message);
        }, waitTimeMs);
    });
}

// Example usage
async function main() {
    try {
        const message = await waitForMessage(1000, "Hello, World!");
        console.log(message);
    } catch (error) {
        console.error("An error occurred:", error);
    }
}

main();
```
