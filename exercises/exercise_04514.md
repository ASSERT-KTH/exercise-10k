## Exercise
Create a program that plays Tic-Tac-Toe against the user. The user can choose to play as X or O.

## Reference Solution
```python
import random

def print_board(board):
    """Print the Tic Tac Toe board"""
    print(f" {board[0]} | {board[1]} | {board[2]}")
    print("---+---+---")
    print(f" {board[3]} | {board[4]} | {board[5]}")
    print("---+---+---")
    print(f" {board[6]} | {board[7]} | {board[8]}")

def check_win(board):
    """Check if there is a winner"""
    win_conditions = [(0, 1, 2), (3, 4, 5), (6, 7, 8), (0, 3, 6), (1, 4, 7), (2, 5, 8), (0, 4, 8), (2, 4, 6)]
    for condition in win_conditions:
        if board[condition[0]] == board[condition[1]] == board[condition[2]] != " ":
            return board[condition[0]]
    if " " not in board:
        return "Tie"
    return False

def computer_move(board, computer_symbol):
    """Make a move for the computer"""
    possible_moves = [i for i, x in enumerate(board) if x == " "]
    move = random.choice(possible_moves)
    board[move] = computer_symbol

def main():
    """Main game loop"""
    board = [" "] * 9
    print("Welcome to Tic Tac Toe!")
    print("Do you want to play as X or O?")
    user_symbol = input("Enter X or O: ").upper()
    while user_symbol not in ["X", "O"]:
        user_symbol = input("Invalid input. Enter X or O: ").upper()
    computer_symbol = "O" if user_symbol == "X" else "X"

    while True:
        print_board(board)
        move = input("Enter your move (1-9): ")
        while not move.isdigit() or not 1 <= int(move) <= 9:
            move = input("Invalid input. Enter your move (1-9): ")
        move = int(move) - 1
        if board[move] != " ":
            print("Invalid move, try again.")
            continue
        board[move] = user_symbol
        result = check_win(board)
        if result:
            print_board(board)
            if result == "Tie":
                print("It's a tie!")
            else:
                print(f"{result} wins!")
            break
        computer_move(board, computer_symbol)
        result = check_win(board)
        if result:
            print_board(board)
            if result == "Tie":
                print("It's a tie!")
            else:
                print(f"{result} wins!")
            break

if __name__ == "__main__":
    main()
```
