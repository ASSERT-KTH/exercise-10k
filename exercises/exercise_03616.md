## Exercise
Write a function that invokes another function with a variable number of arguments

## Reference Solution
```python
def invoke_function(func, *args, **kwargs):
    """
    Invokes a function with a variable number of arguments.

    Args:
        func (function): The function to be invoked.
        *args: Variable number of non-keyword arguments.
        **kwargs: Variable number of keyword arguments.

    Returns:
        The result of invoking func with the provided arguments.
    """
    return func(*args, **kwargs)


# Example usage:
def add(a, b):
    """Returns the sum of two numbers."""
    return a + b


def greet(name, message="Hello"):
    """Prints a greeting message."""
    print(f"{message}, {name}!")


def main():
    # Invoke the add function with two arguments
    result = invoke_function(add, 3, 5)
    print(f"Result: {result}")

    # Invoke the greet function with two arguments
    invoke_function(greet, "John", message="Hi")

    # Invoke the greet function with one argument (using default message)
    invoke_function(greet, "Alice")


if __name__ == "__main__":
    main()
```
