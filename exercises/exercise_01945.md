## Exercise
Write a program that takes an array, an index, and a value as input and replaces the element at that index with the given value.

## Reference Solution
```python
#include <iostream>
#include <vector>
#include <string>

using namespace std;

vector<string> replaceElement(vector<string> array, int index, string value) {
    array[index] = value;
    return array;
}

int main() {
    int n;
    cin >> n;
    vector<string> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    int idx;
    cin >> idx;
    string val;
    cin >> val;
    vector<string> result = replaceElement(a, idx, val);
    for (int i = 0; i < n; i++) {
        cout << result[i] << " ";
    }
    return 0;
}
```
