## Exercise
Write a program that calculates the car insurance premium based on the driver's age and number of accidents.

## Reference Solution
```python
def calculate_premium(age, accidents):
    """
    Calculates the car insurance premium based on the driver's age and number of accidents.

    Args:
        age (int): The driver's age.
        accidents (int): The number of accidents the driver has been involved in.

    Returns:
        float: The car insurance premium.
    """
    base_premium = 500.0  # Base premium
    age_factor = 0.0  # Age factor
    accident_factor = 0.0  # Accident factor

    # Calculate age factor
    if age < 25:
        age_factor = 0.2  # 20% increase for drivers under 25
    elif age >= 65:
        age_factor = 0.1  # 10% increase for drivers over 65

    # Calculate accident factor
    if accidents == 0:
        accident_factor = -0.1  # 10% discount for no accidents
    elif accidents == 1:
        accident_factor = 0.05  # 5% increase for one accident
    elif accidents > 1:
        accident_factor = 0.1  # 10% increase for multiple accidents

    # Calculate premium
    premium = base_premium * (1 + age_factor + accident_factor)

    return premium

def main():
    # Get user input
    age = int(input("Enter your age: "))
    accidents = int(input("Enter the number of accidents you've been involved in: "))

    # Calculate premium
    premium = calculate_premium(age, accidents)

    # Print result
    print(f"Your car insurance premium is: ${premium:.2f}")

if __name__ == "__main__":
    main()
```
