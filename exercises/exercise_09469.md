## Exercise
Write a program that scrambles a given word and asks the user to unscramble it.

## Reference Solution
```python
import random

def scramble_word(word):
    """
    Scrambles a given word.
    
    Parameters:
    word (str): The word to be scrambled.
    
    Returns:
    str: The scrambled word.
    """
    word_list = list(word)
    random.shuffle(word_list)
    return ''.join(word_list)

def main():
    """
    The main function of the program.
    """
    word = input("Enter a word: ")
    scrambled_word = scramble_word(word)
    
    print(f"\nUnscramble the word: {scrambled_word}")
    user_answer = input("Enter your answer: ")
    
    if user_answer.lower() == word.lower():
        print("Correct! The unscrambled word is indeed", word)
    else:
        print(f"Incorrect. The correct answer is {word}")

if __name__ == "__main__":
    main()
```
