## Exercise
Write a function to use the copy() method to create a copy of a dictionary.

## Reference Solution
```python
def create_dict_copy(original_dict):
    """
    Creates a shallow copy of a dictionary using the copy() method.

    Args:
        original_dict (dict): The dictionary to be copied.

    Returns:
        dict: A shallow copy of the original dictionary.
    """
    # Create a copy of the original dictionary
    dict_copy = original_dict.copy()

    return dict_copy

# Example usage:
original_dict = {"name": "John", "age": 30, "city": "New York"}
print("Original Dictionary:")
print(original_dict)

dict_copy = create_dict_copy(original_dict)
print("\nCopied Dictionary:")
print(dict_copy)

# Modify the original dictionary to demonstrate that the copy is shallow
original_dict["name"] = "Jane"
print("\nOriginal Dictionary after modification:")
print(original_dict)
print("\nCopied Dictionary after modification:")
print(dict_copy)
```
