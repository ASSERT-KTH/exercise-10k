## Exercise
Write a program that helps students schedule courses using conditional statements, considering factors such as course availability, time conflicts, and prerequisites.

## Reference Solution
```python
# Define a dictionary to store course information
courses = {
    "Math 101": {"available": True, "time": "Monday 10:00 AM", "prerequisites": []},
    "Math 102": {"available": True, "time": "Tuesday 11:00 AM", "prerequisites": ["Math 101"]},
    "Science 101": {"available": True, "time": "Wednesday 2:00 PM", "prerequisites": []},
    "Science 102": {"available": True, "time": "Thursday 3:00 PM", "prerequisites": ["Science 101"]},
    "English 101": {"available": True, "time": "Monday 10:00 AM", "prerequisites": []},
}

# Define a dictionary to store student schedules
student_schedules = {}

def schedule_course(student_name, course_name):
    """
    Schedules a course for a student.

    Args:
        student_name (str): The name of the student.
        course_name (str): The name of the course.

    Returns:
        bool: True if the course is scheduled successfully, False otherwise.
    """
    if course_name not in courses:
        print(f"Error: {course_name} is not available.")
        return False

    if not courses[course_name]["available"]:
        print(f"Error: {course_name} is not available.")
        return False

    if course_name in student_schedules.get(student_name, []):
        print(f"Error: You are already enrolled in {course_name}.")
        return False

    for scheduled_course in student_schedules.get(student_name, []):
        if courses[scheduled_course]["time"] == courses[course_name]["time"]:
            print(f"Error: Time conflict with {scheduled_course}.")
            return False

    for prerequisite in courses[course_name]["prerequisites"]:
        if prerequisite not in student_schedules.get(student_name, []):
            print(f"Error: You need to take {prerequisite} before taking {course_name}.")
            return False

    if student_name not in student_schedules:
        student_schedules[student_name] = []

    student_schedules[student_name].append(course_name)
    print(f"Success: {course_name} has been scheduled for {student_name}.")
    return True

def main():
    while True:
        print("\nCourse Scheduling Menu:")
        print("1. Schedule a course")
        print("2. View student schedule")
        print("3. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            student_name = input("Enter your name: ")
            course_name = input("Enter the course name: ")
            schedule_course(student_name, course_name)
        elif choice == "2":
            student_name = input("Enter your name: ")
            if student_name in student_schedules:
                print(f"\n{student_name}'s Schedule:")
                for course in student_schedules[student_name]:
                    print(course)
            else:
                print(f"Error: {student_name} has no scheduled courses.")
        elif choice == "3":
            break
        else:
            print("Error: Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
