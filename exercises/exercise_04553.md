## Exercise
Create a program that finds the longest word in a given string.

## Reference Solution
```python
def find_longest_word(input_string):
    """
    Finds the longest word in a given string.

    Args:
        input_string (str): The input string to search for the longest word.

    Returns:
        str: The longest word found in the input string.
    """
    # Split the input string into words
    words = input_string.split()

    # Initialize the longest word and its length
    longest_word = ""
    max_length = 0

    # Iterate over each word in the list of words
    for word in words:
        # Remove punctuation from the word
        word = word.strip(".,!?:;")

        # Check if the current word is longer than the longest word found so far
        if len(word) > max_length:
            # Update the longest word and its length
            longest_word = word
            max_length = len(word)

    # Return the longest word
    return longest_word

def main():
    # Example usage
    input_string = "This is a sample string with multiple words"
    longest_word = find_longest_word(input_string)
    print(f"The longest word is: {longest_word}")

if __name__ == "__main__":
    main()
```
