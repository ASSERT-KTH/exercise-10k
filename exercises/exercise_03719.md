## Exercise
Create a function that generates a random graph of a specified size.

## Reference Solution
```python
import networkx as nx
import random
import matplotlib.pyplot as plt

def generate_random_graph(num_nodes, num_edges):
    """
    Generate a random graph with the specified number of nodes and edges.

    Args:
    - num_nodes (int): The number of nodes in the graph.
    - num_edges (int): The number of edges in the graph.

    Returns:
    - A NetworkX graph object representing the random graph.
    """
    # Check if the number of edges is valid
    if num_edges > num_nodes * (num_nodes - 1) / 2:
        raise ValueError("Too many edges for the given number of nodes")

    # Create an empty graph
    G = nx.Graph()

    # Add nodes to the graph
    for i in range(num_nodes):
        G.add_node(i)

    # Add edges to the graph
    edges = set()
    while len(edges) < num_edges:
        node1 = random.randint(0, num_nodes - 1)
        node2 = random.randint(0, num_nodes - 1)
        # Ensure the edge is not self-referential and not already added
        if node1 != node2 and (node1, node2) not in edges and (node2, node1) not in edges:
            edges.add((node1, node2))
            G.add_edge(node1, node2)

    return G

# Example usage
if __name__ == "__main__":
    num_nodes = 10
    num_edges = 15
    G = generate_random_graph(num_nodes, num_edges)
    print("Nodes:", list(G.nodes()))
    print("Edges:", list(G.edges()))

    # Draw the graph
    nx.draw(G, with_labels=True)
    plt.show()
```
