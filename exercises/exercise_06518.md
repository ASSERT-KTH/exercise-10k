## Exercise
Write a program to cast a value from one data type to another (int, float, str, etc.).

## Reference Solution
```python
def cast_data_type(value, target_type):
    """
    Casts a value from one data type to another.

    Args:
        value (any): The value to be casted.
        target_type (type): The target data type.

    Returns:
        any: The casted value.
    """
    try:
        if target_type == int:
            return int(value)
        elif target_type == float:
            return float(value)
        elif target_type == str:
            return str(value)
        elif target_type == bool:
            return bool(value)
        else:
            raise ValueError("Unsupported target data type")
    except ValueError as e:
        print(f"Error: {e}")
        return None

# Example usage
if __name__ == "__main__":
    # Test cases
    test_cases = [
        ("123", int),
        (123.45, float),
        ("hello", str),
        (True, bool),
        ("true", bool),
        (123, str),
        (123.45, int),
        ("123.45", float)
    ]

    for value, target_type in test_cases:
        result = cast_data_type(value, target_type)
        print(f"Casting {value} to {target_type.__name__}: {result}")
```
