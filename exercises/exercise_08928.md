## Exercise
Write a program to check if a given linked list has a cycle.

## Reference Solution
```python
class Node:
    """Represents a node in the linked list."""
    def __init__(self, x):
        self.val = x
        self.next = None

def has_cycle(head):
    """
    Detects whether a cycle exists in the linked list.

    Args:
        head (Node): The head of the linked list.

    Returns:
        bool: True if a cycle exists, False otherwise.
    """
    if head is None or head.next is None:
        return False

    # Initialize slow and fast pointers
    slow = head
    fast = head.next

    while slow != fast:
        if fast is None or fast.next is None:
            return False

        # Move slow pointer one step at a time
        slow = slow.next
        # Move fast pointer two steps at a time
        fast = fast.next.next

    return True

# Example usage:
if __name__ == "__main__":
    # Create a linked list with a cycle
    node1 = Node(1)
    node2 = Node(2)
    node3 = Node(3)
    node4 = Node(4)

    node1.next = node2
    node2.next = node3
    node3.next = node4
    node4.next = node2  # Create a cycle

    print(has_cycle(node1))  # Output: True

    # Create a linked list without a cycle
    node5 = Node(5)
    node6 = Node(6)
    node7 = Node(7)

    node5.next = node6
    node6.next = node7

    print(has_cycle(node5))  # Output: False
```
