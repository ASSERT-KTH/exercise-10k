## Exercise
Write a function to find the symmetric difference of two dictionaries

def symmetric_difference(dict1, dict2):
    """
    Calculate the symmetric difference of two dictionaries.

    Args:
    dict1 (dict): The first dictionary.
    dict2 (dict): The second dictionary.

    Returns:
    dict: A new dictionary containing the symmetric difference of dict1 and dict2.
    """
    symmetric_diff = {}
    all_keys = set(dict1.keys()) | set(dict2.keys())

    for key in all_keys:
        value1 = dict1.get(key)
        value2 = dict2.get(key)

        if (value1 is None) ^ (value2 is None):
            symmetric_diff[key] = value1 if value1 is not None else value2
        elif value1 != value2:
            symmetric_diff[key] = value1

    return symmetric_diff

# Example usage
if __name__ == "__main__":
    dict1 = {'a': 1, 'b': 2, 'c': 3}
    dict2 = {'b': 20, 'c': 3, 'd': 4}

    result = symmetric_difference(dict1, dict2)
    print(result)  # Output: {'a': 1, 'b': 2, 'd': 4}
