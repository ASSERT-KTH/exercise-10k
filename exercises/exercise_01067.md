## Exercise
Write a program that uses a while loop to simulate a coin toss and keep track of the number of heads and tails.

## Reference Solution
```python
import random

def coin_toss_simulator():
    """
    Simulates a coin toss and keeps track of the number of heads and tails.
    """
    heads = 0
    tails = 0
    num_tosses = 0

    while True:
        # Simulate a coin toss
        toss = random.choice(["Heads", "Tails"])
        num_tosses += 1

        # Update the count
        if toss == "Heads":
            heads += 1
        else:
            tails += 1

        # Print the result
        print(f"Toss {num_tosses}: {toss}")

        # Check if the user wants to continue
        response = input("Do you want to toss again? (yes/no): ")
        if response.lower() != "yes":
            break

    # Print the final results
    print(f"\nTotal Tosses: {num_tosses}")
    print(f"Heads: {heads} ({heads / num_tosses * 100:.2f}%)")
    print(f"Tails: {tails} ({tails / num_tosses * 100:.2f}%)")

if __name__ == "__main__":
    coin_toss_simulator()
```
