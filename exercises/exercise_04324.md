## Exercise
Write a program to find the middle element of an array.

## Reference Solution
```python
def find_middle_element(array):
    """
    Find the middle element of an array.

    Args:
        array (list): The input array.

    Returns:
        The middle element(s) of the array.
    """
    n = len(array)
    middle_index = n // 2

    if n % 2 == 0:
        # If the array has an even number of elements, return the two middle elements
        return array[middle_index - 1:middle_index + 1]
    else:
        # If the array has an odd number of elements, return the single middle element
        return array[middle_index]

# Example usage
def main():
    # Test the function with an odd-length array
    array1 = [1, 2, 3, 4, 5]
    print("Middle element of array1:", find_middle_element(array1))

    # Test the function with an even-length array
    array2 = [1, 2, 3, 4]
    print("Middle elements of array2:", find_middle_element(array2))

if __name__ == "__main__":
    main()
```
