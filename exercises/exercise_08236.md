## Exercise
Create a program that manages a product inventory using if/else statements

## Reference Solution
```python
# Define a dictionary to store the product inventory
product_inventory = {}

def display_menu():
    """Display the main menu"""
    print("\nProduct Inventory Management Program")
    print("-------------------------------------")
    print("1. Add product")
    print("2. Remove product")
    print("3. Update product quantity")
    print("4. Display product inventory")
    print("5. Exit")

def add_product():
    """Add a new product to the inventory"""
    product_name = input("Enter product name: ")
    quantity = int(input("Enter product quantity: "))
    price = float(input("Enter product price: "))
    
    if product_name in product_inventory:
        print("Product already exists. Please update the quantity instead.")
    else:
        product_inventory[product_name] = {"quantity": quantity, "price": price}
        print("Product added successfully.")

def remove_product():
    """Remove a product from the inventory"""
    product_name = input("Enter product name: ")
    
    if product_name in product_inventory:
        del product_inventory[product_name]
        print("Product removed successfully.")
    else:
        print("Product not found in the inventory.")

def update_product_quantity():
    """Update the quantity of a product in the inventory"""
    product_name = input("Enter product name: ")
    new_quantity = int(input("Enter new quantity: "))
    
    if product_name in product_inventory:
        product_inventory[product_name]["quantity"] = new_quantity
        print("Product quantity updated successfully.")
    else:
        print("Product not found in the inventory.")

def display_product_inventory():
    """Display the current product inventory"""
    if not product_inventory:
        print("Inventory is empty.")
    else:
        print("\nCurrent Product Inventory:")
        for product, info in product_inventory.items():
            print(f"Product: {product}, Quantity: {info['quantity']}, Price: {info['price']}")

def main():
    while True:
        display_menu()
        choice = input("Enter your choice: ")
        
        if choice == "1":
            add_product()
        elif choice == "2":
            remove_product()
        elif choice == "3":
            update_product_quantity()
        elif choice == "4":
            display_product_inventory()
        elif choice == "5":
            print("Exiting the program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
