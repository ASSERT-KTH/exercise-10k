## Exercise
Write a program that takes two strings as input and returns True if they are anagrams of each other, False otherwise.

## Reference Solution
```python
def are_anagrams(str1, str2):
    """
    Returns True if str1 and str2 are anagrams, False otherwise.
    
    :param str1: First input string
    :param str2: Second input string
    :return: Boolean indicating whether the strings are anagrams
    """
    # Remove any whitespace and convert to lowercase
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()
    
    # Check if sorted versions of the strings are equal
    return sorted(str1) == sorted(str2)

# Example usage
def main():
    str1 = "listen"
    str2 = "silent"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")
    
    str1 = "hello"
    str2 = "world"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")

if __name__ == "__main__":
    main()
```
