## Exercise
Create a function that uses regular expressions to validate and extract data from strings.

## Reference Solution
```python
import re

def extract_data(input_string, pattern):
    """
    Validate and extract data from input string based on the provided pattern.

    Args:
        input_string (str): The string to extract data from.
        pattern (str): The regular expression pattern to match.

    Returns:
        list: A list of extracted data. If no data is found, an empty list is returned.
    """

    try:
        # Compile the regular expression pattern
        regex = re.compile(pattern)

        # Search for matches in the input string
        matches = regex.findall(input_string)

        return matches

    except re.error as e:
        print(f"Invalid regular expression: {e}")
        return []

# Example usage
if __name__ == "__main__":
    input_string = "Hello, my email is john.doe@example.com and my phone number is 123-456-7890."
    email_pattern = r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b"
    phone_pattern = r"\b\d{3}-\d{3}-\d{4}\b"

    emails = extract_data(input_string, email_pattern)
    phone_numbers = extract_data(input_string, phone_pattern)

    print("Extracted Emails:", emails)
    print("Extracted Phone Numbers:", phone_numbers)
```
