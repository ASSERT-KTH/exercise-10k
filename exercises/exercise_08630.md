## Exercise
Write a program that takes a string as input and returns a list of all permutations.

## Reference Solution
```python
import itertools

def generate_permutations(input_string):
    """
    Generates all permutations of the input string.

    Args:
        input_string (str): The input string.

    Returns:
        list: A list of all permutations of the input string.
    """
    # Use itertools.permutations to generate all permutations
    permutations = itertools.permutations(input_string)

    # Convert the permutations to a list of strings
    permutation_list = [''.join(p) for p in permutations]

    return permutation_list

def main():
    # Example usage
    input_string = "abc"
    permutations = generate_permutations(input_string)

    print("Input String:", input_string)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"{i+1}. {permutation}")

if __name__ == "__main__":
    main()
```
