## Exercise
Write a program that generates all anagrams of a given word using a nested for loop.

## Reference Solution
```python
from itertools import permutations

def generate_anagrams(word):
    """
    Generate all anagrams of a given word.

    Args:
    word (str): The input word.

    Returns:
    list: A list of all anagrams of the input word.
    """
    # Generate all permutations of the word
    perms = permutations(word)

    # Convert each permutation to a string and add to the result list
    anagrams = [''.join(p) for p in perms]

    return anagrams

def main():
    word = input("Enter a word: ")
    anagrams = generate_anagrams(word)

    # Print each anagram on a new line
    for anagram in anagrams:
        print(anagram)

if __name__ == "__main__":
    main()
```

```python
def generate_anagrams(word):
    """
    Generate all anagrams of a given word.

    Args:
    word (str): The input word.

    Returns:
    list: A list of all anagrams of the input word.
    """
    # Base case: if the word has only one character, return it
    if len(word) == 1:
        return [word]

    # Initialize result list
    anagrams = []

    # For each character in the word
    for i, char in enumerate(word):
        # Get the rest of the word
        rest = word[:i] + word[i+1:]

        # Generate all anagrams of the rest of the word
        for anagram in generate_anagrams(rest):
            # Add the current character to the beginning of each anagram
            anagrams.append(char + anagram)

    return anagrams
```

```python
def generate_anagrams(word):
    """
    Generate all anagrams of a given word.

    Args:
    word (str): The input word.

    Returns:
    list: A list of all anagrams of the input word.
    """
    # Convert the word to a list of characters
    chars = list(word)

    # Initialize result list
    anagrams = []

    # For each possible permutation of the characters
    for i in range(len(chars)):
        for j in range(len(chars)):
            if j == i:
                continue
            for k in range(len(chars)):
                if k == i or k == j:
                    continue
                # For the remaining characters
                remaining = [c for index, c in enumerate(chars) if index not in [i, j, k]]
                # Generate all permutations of the remaining characters
                for p in permutations(remaining):
                    # Add the current character to the beginning of each permutation
                    anagram = chars[i] + chars[j] + chars[k] + ''.join(p)
                    anagrams.append(anagram)

    return anagrams

    anagrams = generate_anagrams(word)

    # Print each anagram on a new line
    for anagram in anagrams:
```
