## Exercise
Read two positive integers from the user and calculate their greatest common divisor

## Reference Solution
```python
import math

def calculate_gcd(num1, num2):
    """
    Calculate the greatest common divisor of two numbers using the Euclidean algorithm.
    
    Args:
        num1 (int): The first positive integer.
        num2 (int): The second positive integer.
    
    Returns:
        int: The greatest common divisor of num1 and num2.
    """
    while num2 != 0:
        num1, num2 = num2, num1 % num2
    return num1

def main():
    # Read two positive integers from the user
    num1 = int(input("Enter the first positive integer: "))
    num2 = int(input("Enter the second positive integer: "))

    # Validate user input
    if num1 <= 0 or num2 <= 0:
        print("Error: Both numbers must be positive integers.")
        return

    # Calculate the GCD using the Euclidean algorithm
    gcd = calculate_gcd(num1, num2)

    # Alternative method using the math.gcd function
    gcd_alternative = math.gcd(num1, num2)

    # Print the results
    print(f"GCD of {num1} and {num2} (Euclidean algorithm): {gcd}")
    print(f"GCD of {num1} and {num2} (math.gcd function): {gcd_alternative}")

if __name__ == "__main__":
    main()
```
