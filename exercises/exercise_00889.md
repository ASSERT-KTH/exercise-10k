## Exercise
Create a program that asks the user for a number and prints out the next number in the sequence: 1, 2, 4, 8, 16, ...

## Reference Solution
```python
def get_next_number_in_sequence(n):
    """
    Returns the next number in the sequence 1, 2, 4, 8, 16, ...

    Args:
        n (int): The current number in the sequence.

    Returns:
        int: The next number in the sequence.
    """
    return n * 2

def main():
    # Ask the user for a number in the sequence
    while True:
        try:
            num = int(input("Enter a number in the sequence (1, 2, 4, 8, 16, ...): "))
            if num <= 0 or (num & (num - 1) != 0):
                print("Invalid input. Please enter a positive power of 2.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid integer.")

    # Print the next number in the sequence
    next_num = get_next_number_in_sequence(num)
    print(f"The next number in the sequence is: {next_num}")

if __name__ == "__main__":
    main()
```
