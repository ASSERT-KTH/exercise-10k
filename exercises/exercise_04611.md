## Exercise
Write a program to calculate the compound interest given the principal amount, rate, and time.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, n=1):
    """
    Calculate the compound interest.

    Args:
        principal (float): The initial amount of money.
        rate (float): The interest rate (in decimal form, e.g. 4% = 0.04).
        time (int): The time period in years.
        n (int, optional): The number of times interest is compounded per year. Defaults to 1.

    Returns:
        float: The amount of money after the given time period, including interest.
    """
    amount = principal * (1 + rate/n) ** (n*time)
    return amount

def main():
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))
    n = int(input("Enter the number of times interest is compounded per year (default=1): ") or 1)

    amount = calculate_compound_interest(principal, rate, time, n)

    print(f"The amount after {time} years is: ${amount:.2f}")
    print(f"The compound interest is: ${amount - principal:.2f}")

if __name__ == "__main__":
    main()
```
