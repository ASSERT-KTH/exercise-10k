## Exercise
Write a program that books a ticket using if/else statements

## Reference Solution
```python
class TicketBooking:
    def __init__(self):
        self.seats = {
            "A1": False,
            "A2": False,
            "B1": False,
            "B2": False,
            "C1": False,
            "C2": False,
        }

    def display_seats(self):
        print("Available Seats:")
        for seat, booked in self.seats.items():
            if not booked:
                print(f"{seat}")

    def book_seat(self):
        self.display_seats()
        seat = input("Enter the seat number you want to book: ")
        if seat in self.seats:
            if not self.seats[seat]:
                name = input("Enter your name: ")
                email = input("Enter your email: ")
                phone = input("Enter your phone number: ")
                self.seats[seat] = True
                print(f"Seat {seat} booked successfully for {name}!")
            else:
                print("Seat already booked. Please choose another seat.")
        else:
            print("Invalid seat number. Please try again.")

    def cancel_booking(self):
        seat = input("Enter the seat number you want to cancel: ")
        if seat in self.seats:
            if self.seats[seat]:
                self.seats[seat] = False
                print(f"Booking for seat {seat} cancelled successfully!")
            else:
                print("Seat is not booked. No need to cancel.")
        else:
            print("Invalid seat number. Please try again.")

def main():
    booking = TicketBooking()
    while True:
        print("\n1. Book a seat")
        print("2. Cancel a booking")
        print("3. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            booking.book_seat()
        elif choice == "2":
            booking.cancel_booking()
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
