## Exercise
Write a program that takes a string and a number as input and returns the string with the correct plural form.

## Reference Solution
```python
def pluralize(s, n):
    """
    Returns the string s with the correct plural form based on the number n.
    
    Parameters:
    s (str): The input string.
    n (int): The number to determine the plural form.
    
    Returns:
    str: The string s with the correct plural form.
    """
    if n == 1:
        return s
    else:
        # Check if the string already ends with 's' or 'z' to avoid adding another 's'
        if s.endswith(('s', 'z', 'x', 'sh', 'ch')):
            return f"{s}es" if not s.endswith('is') else f"{s[:-2]}es"
        # Check if the string ends with 'y' and the letter before 'y' is a consonant
        elif s.endswith('y') and s[-2].lower() not in 'aeiou':
            return f"{s[:-1]}ies"
        else:
            return f"{s}s"

def main():
    # Example usage:
    print(pluralize("cat", 1))  # Output: cat
    print(pluralize("cat", 2))  # Output: cats
    print(pluralize("bus", 1))  # Output: bus
    print(pluralize("bus", 2))  # Output: buses
    print(pluralize("city", 1))  # Output: city
    print(pluralize("city", 2))  # Output: cities

if __name__ == "__main__":
    main()
```
