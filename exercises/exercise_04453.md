## Exercise
Create a program that encrypts and decrypts messages using a Caesar Cipher. The user should be able to input a message and a shift value, and the program should then print out the encrypted or decrypted message.

## Reference Solution
```python
def caesar_cipher(message, shift, mode):
    """
    Encrypts or decrypts a message using a Caesar Cipher.

    Parameters:
    message (str): The message to be encrypted or decrypted.
    shift (int): The number of positions to shift the letters.
    mode (str): 'encrypt' or 'decrypt' mode.

    Returns:
    str: The encrypted or decrypted message.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in message:
        if char.isalpha():
            # Find the position of the character in the alphabet
            index = alphabet.index(char.lower())
            
            # Apply the shift
            if mode == 'encrypt':
                new_index = (index + shift) % 26
            elif mode == 'decrypt':
                new_index = (index - shift) % 26
            else:
                raise ValueError("Invalid mode. Use 'encrypt' or 'decrypt'.")

            # Replace the character with the shifted one
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            # If the character is not a letter, leave it as is
            result += char

    return result


def main():
    while True:
        print("\nCaesar Cipher Program")
        print("1. Encrypt a message")
        print("2. Decrypt a message")
        print("3. Quit")
        
        choice = input("Choose an option: ")

        if choice == '1':
            message = input("Enter the message to encrypt: ")
            shift = int(input("Enter the shift value: "))
            encrypted_message = caesar_cipher(message, shift, 'encrypt')
            print(f"Encrypted message: {encrypted_message}")
        elif choice == '2':
            message = input("Enter the message to decrypt: ")
            shift = int(input("Enter the shift value: "))
            decrypted_message = caesar_cipher(message, shift, 'decrypt')
            print(f"Decrypted message: {decrypted_message}")
        elif choice == '3':
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
