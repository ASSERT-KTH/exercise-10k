## Exercise
Write a program that takes a list of numbers as input and returns the second-largest value.

## Reference Solution
```python
def second_largest(numbers):
    """
    Returns the second-largest number in a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The second-largest number, or None if the list has less than two unique numbers.
    """
    # Remove duplicates and sort the list in descending order
    unique_numbers = sorted(set(numbers), reverse=True)

    # Check if the list has at least two unique numbers
    if len(unique_numbers) < 2:
        return None

    # Return the second-largest number
    return unique_numbers[1]

# Example usage
def main():
    numbers = [10, 20, 30, 40, 50]
    result = second_largest(numbers)
    print(f"Second-largest number: {result}")

    numbers = [10, 10, 10]
    result = second_largest(numbers)
    print(f"Second-largest number: {result}")

if __name__ == "__main__":
    main()
```
