## Exercise
Write a program to manage a phonebook for the user.

## Reference Solution
```python
class Contact:
    def __init__(self, name, phone_number):
        self.name = name
        self.phone_number = phone_number

class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone_number):
        """Add a new contact to the phonebook"""
        if name in self.contacts:
            print("Contact already exists. Please update the existing contact.")
        else:
            self.contacts[name] = Contact(name, phone_number)
            print("Contact added successfully.")

    def remove_contact(self, name):
        """Remove a contact from the phonebook"""
        if name in self.contacts:
            del self.contacts[name]
            print("Contact removed successfully.")
        else:
            print("Contact not found.")

    def update_contact(self, name, new_name=None, new_phone_number=None):
        """Update an existing contact in the phonebook"""
        if name in self.contacts:
            if new_name:
                self.contacts[new_name] = self.contacts.pop(name)
                self.contacts[new_name].name = new_name
            if new_phone_number:
                self.contacts[name].phone_number = new_phone_number
            print("Contact updated successfully.")
        else:
            print("Contact not found.")

    def search_contact(self, name):
        """Search for a contact in the phonebook"""
        if name in self.contacts:
            print(f"Name: {self.contacts[name].name}, Phone Number: {self.contacts[name].phone_number}")
        else:
            print("Contact not found.")

    def display_contacts(self):
        """Display all contacts in the phonebook"""
        if self.contacts:
            for contact in self.contacts.values():
                print(f"Name: {contact.name}, Phone Number: {contact.phone_number}")
        else:
            print("No contacts in the phonebook.")

def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook Management Menu:")
        print("1. Add Contact")
        print("2. Remove Contact")
        print("3. Update Contact")
        print("4. Search Contact")
        print("5. Display Contacts")
        print("6. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter contact name: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter contact name: ")
            new_name = input("Enter new contact name (press enter to skip): ")
            new_phone_number = input("Enter new contact phone number (press enter to skip): ")
            phonebook.update_contact(name, new_name or None, new_phone_number or None)
        elif choice == "4":
            name = input("Enter contact name: ")
            phonebook.search_contact(name)
        elif choice == "5":
            phonebook.display_contacts()
        elif choice == "6":
            print("Exiting phonebook management program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
