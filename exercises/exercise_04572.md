## Exercise
Write a function that takes two dictionaries and returns a new dictionary with the key-value pairs that exist in either dictionary but not in both

def symmetric_dict_diff(dict1, dict2):
    """
    Calculate the symmetric difference between two dictionaries.
    
    Args:
    dict1 (dict): The first dictionary.
    dict2 (dict): The second dictionary.
    
    Returns:
    dict: A new dictionary containing key-value pairs that exist in either dictionary but not in both.
    """
    # Calculate the symmetric difference of the dictionary keys
    diff_keys = set(dict1.keys()) ^ set(dict2.keys())
    
    # Initialize the result dictionary
    result = {}
    
    # Iterate over the keys in dict1 and add them to the result if they are in the diff_keys
    for key in dict1:
        if key in diff_keys:
            result[key] = dict1[key]
    
    # Iterate over the keys in dict2 and add them to the result if they are in the diff_keys
    for key in dict2:
        if key in diff_keys:
            result[key] = dict2[key]
    
    return result

    diff_keys = set(dict1.keys()) ^ set(dict2.keys())
    
    # Use dictionary comprehensions to create the result dictionary
    result = {key: dict1[key] for key in dict1 if key in diff_keys}
    result.update({key: dict2[key] for key in dict2 if key in diff_keys})
    
    return result
