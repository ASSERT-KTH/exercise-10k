## Exercise
Write a program that generates a flower pattern using a while loop.

## Reference Solution
```python
import turtle

def draw_petal(turtle, radius, angle):
    """
    Draw a petal of the flower.

    Args:
        turtle (turtle.Turtle): The turtle object to use for drawing.
        radius (int): The radius of the petal.
        angle (int): The angle of the petal.
    """
    turtle.circle(radius, angle)
    turtle.left(180 - angle)
    turtle.circle(radius, angle)

def draw_flower(turtle, num_petals, radius, angle):
    """
    Draw a flower with the specified number of petals.

    Args:
        turtle (turtle.Turtle): The turtle object to use for drawing.
        num_petals (int): The number of petals in the flower.
        radius (int): The radius of each petal.
        angle (int): The angle of each petal.
    """
    i = 0
    while i < num_petals:
        draw_petal(turtle, radius, angle)
        turtle.left(360 / num_petals)
        i += 1

def main():
    # Create a new turtle screen and set its background color
    screen = turtle.Screen()
    screen.bgcolor("white")

    # Create a new turtle object
    my_turtle = turtle.Turtle()
    my_turtle.speed(1)  # Set the speed of the turtle

    # Draw a flower with 10 petals
    draw_flower(my_turtle, 10, 100, 60)

    # Keep the window open until it is closed by the user
    turtle.done()

if __name__ == "__main__":
    main()
```
