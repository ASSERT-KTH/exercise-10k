## Exercise
Create a function that takes in a nested dictionary and returns a new flattened dictionary containing the same key-value pairs

## Reference Solution
```python
def flatten_dict(nested_dict, parent_key='', separator='_'):
    """
    Flattens a nested dictionary into a new dictionary.

    Args:
    - nested_dict (dict): The dictionary to be flattened.
    - parent_key (str, optional): The key of the parent dictionary. Defaults to ''.
    - separator (str, optional): The separator used to join keys. Defaults to '_'.

    Returns:
    - dict: The flattened dictionary.
    """
    flattened_dict = {}
    for key, value in nested_dict.items():
        new_key = f"{parent_key}{separator}{key}" if parent_key else key
        if isinstance(value, dict):
            # Recursively call the function for nested dictionaries
            flattened_dict.update(flatten_dict(value, new_key, separator))
        else:
            # Add the key-value pair to the flattened dictionary
            flattened_dict[new_key] = value
    return flattened_dict

# Example usage:
if __name__ == "__main__":
    nested_dict = {
        'a': 1,
        'b': 2,
        'c': {
            'd': 3,
            'e': 4,
            'f': {
                'g': 5,
                'h': 6
            }
        }
    }
    flattened_dict = flatten_dict(nested_dict)
    print(flattened_dict)
```
