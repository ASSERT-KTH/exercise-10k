## Exercise
Write a function that checks if a string is numeric.

## Reference Solution
```python
def is_numeric(s):
    """
    Checks if a string is numeric.

    Args:
        s (str): The input string to check.

    Returns:
        bool: True if the string is numeric, False otherwise.
    """
    try:
        float(s)
        return True
    except ValueError:
        return False

# Example usage:
if __name__ == "__main__":
    print(is_numeric("123"))  # True
    print(is_numeric("123.45"))  # True
    print(is_numeric("abc"))  # False
    print(is_numeric("123abc"))  # False
```

```python
import re

def is_numeric_regex(s):
    """
    Checks if a string is numeric using regular expressions.

    Args:
        s (str): The input string to check.

    Returns:
        bool: True if the string is numeric, False otherwise.
    """
    pattern = r"^[-+]?[0-9]+\.?[0-9]*$"
    return bool(re.match(pattern, s))

# Example usage:
if __name__ == "__main__":
    print(is_numeric_regex("123"))  # True
    print(is_numeric_regex("123.45"))  # True
    print(is_numeric_regex("abc"))  # False
    print(is_numeric_regex("123abc"))  # False
```
