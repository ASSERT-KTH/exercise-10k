## Exercise
Write a function to reduce dictionary values to a single value

## Reference Solution
```python
from functools import reduce
import operator

def reduce_dict_values(dictionary, reduction_operation):
    """
    Reduce dictionary values to a single value.

    Args:
    - dictionary (dict): Input dictionary.
    - reduction_operation (str): Reduction operation to apply. Can be 'sum', 'product', 'min', 'max'.

    Returns:
    - The reduced value.

    Raises:
    - ValueError: If reduction_operation is not one of 'sum', 'product', 'min', 'max'.
    """

    # Define the reduction operations
    reduction_operations = {
        'sum': operator.add,
        'product': operator.mul,
        'min': operator.lt,
        'max': operator.gt
    }

    # Check if the reduction operation is valid
    if reduction_operation not in reduction_operations:
        raise ValueError("Invalid reduction operation. Must be one of 'sum', 'product', 'min', 'max'.")

    # Get the reduction operation function
    if reduction_operation in ['min', 'max']:
        if reduction_operation == 'min':
            return min(dictionary.values())
        else:
            return max(dictionary.values())
    else:
        op = reduction_operations[reduction_operation]
        return reduce(op, dictionary.values())

# Example usage
if __name__ == "__main__":
    dictionary = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

    print("Sum:", reduce_dict_values(dictionary, 'sum'))  # Output: 10
    print("Product:", reduce_dict_values(dictionary, 'product'))  # Output: 24
    print("Min:", reduce_dict_values(dictionary, 'min'))  # Output: 1
    print("Max:", reduce_dict_values(dictionary, 'max'))  # Output: 4
```
