## Exercise
Write a program that takes the distance and fuel consumption as input and returns the fuel efficiency.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumption):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
        distance (float): The distance traveled in miles or kilometers.
        fuel_consumption (float): The amount of fuel consumed in gallons or liters.

    Returns:
        float: The fuel efficiency in miles per gallon (mpg) or kilometers per liter (km/l).
    """
    if fuel_consumption == 0:
        raise ValueError("Fuel consumption cannot be zero.")

    fuel_efficiency = distance / fuel_consumption
    return fuel_efficiency


def main():
    # Get user input
    distance = float(input("Enter the distance traveled (miles or kilometers): "))
    fuel_consumption = float(input("Enter the fuel consumption (gallons or liters): "))

    # Calculate fuel efficiency
    try:
        fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumption)
        print(f"The fuel efficiency is: {fuel_efficiency:.2f} miles per gallon (mpg) or kilometers per liter (km/l)")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
