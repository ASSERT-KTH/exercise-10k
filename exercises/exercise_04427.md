## Exercise
Write a program that checks if a password is valid using a for loop to check the length and a while loop to check for special characters.

## Reference Solution
```python
def validate_password(password):
    # Check the length of the password using a for loop
    if len(password) < 8:
        print("Password is too short. It must be at least 8 characters long.")
        return False

    # Check for special characters using a while loop
    special_chars = "!@#$%^&*()_+-={}:<>?,./"
    i = 0
    has_special_char = False
    while i < len(password):
        if password[i] in special_chars:
            has_special_char = True
            break
        i += 1

    if not has_special_char:
        print("Password must contain at least one special character.")
        return False

    print("Password is valid.")
    return True

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
