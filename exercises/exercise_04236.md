## Exercise
Write a program to simulate a basketball game, including scoring and timing.

## Reference Solution
```python
import random
import time

class Team:
    def __init__(self, name):
        self.name = name
        self.score = 0

class Game:
    def __init__(self, team1, team2):
        self.team1 = team1
        self.team2 = team2
        self.quarter = 1
        self.time_remaining = 720  # 12 minutes in seconds

    def simulate_quarter(self):
        print(f"Quarter {self.quarter}:")
        self.time_remaining = 720
        while self.time_remaining > 0:
            # Simulate a possession
            if random.random() < 0.5:
                team = self.team1
            else:
                team = self.team2
            # Simulate a shot
            if random.random() < 0.4:
                points = 2
            elif random.random() < 0.7:
                points = 3
            else:
                points = 0
            team.score += points
            print(f"{team.name} scored {points} points.")
            # Update time remaining
            self.time_remaining -= random.randint(10, 30)
            # Print current score and time remaining
            print(f"Score: {self.team1.name} {self.team1.score}, {self.team2.name} {self.team2.score}")
            print(f"Time remaining: {self.time_remaining // 60}:{self.time_remaining % 60:02d}")
            time.sleep(1)
        self.quarter += 1

    def simulate_game(self):
        while self.quarter <= 4:
            self.simulate_quarter()
            print("\n")
        # Print final score
        print(f"Final score: {self.team1.name} {self.team1.score}, {self.team2.name} {self.team2.score}")
        # Determine winner
        if self.team1.score > self.team2.score:
            print(f"{self.team1.name} wins!")
        elif self.team2.score > self.team1.score:
            print(f"{self.team2.name} wins!")
        else:
            print("It's a tie!")

# Example usage
team1 = Team("Lakers")
team2 = Team("Celtics")
game = Game(team1, team2)
game.simulate_game()
```
