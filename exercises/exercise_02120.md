## Exercise
Write a function to create a new dictionary containing only the key-value pairs where the value meets a certain condition

## Reference Solution
```python
def filter_dict(dictionary, condition):
    """
    Creates a new dictionary containing only the key-value pairs where the value meets a certain condition.

    Args:
        dictionary (dict): The input dictionary to be filtered.
        condition (function): A function that takes one argument and returns a boolean value.

    Returns:
        dict: A new dictionary containing only the key-value pairs where the value meets the condition.
    """

    # Initialize an empty dictionary to store the filtered key-value pairs
    filtered_dict = {}

    # Iterate over each key-value pair in the input dictionary
    for key, value in dictionary.items():
        
        # Check if the value meets the condition by calling the condition function with the value as an argument
        if condition(value):
            
            # If the condition is met, add the key-value pair to the filtered dictionary
            filtered_dict[key] = value
    
    # Return the filtered dictionary
    return filtered_dict

# Example usage:
# Define a condition function that checks if a number is even
def is_even(num):
    return num % 2 == 0

# Create a dictionary
numbers = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

# Filter the dictionary using the condition function
filtered_numbers = filter_dict(numbers, is_even)

# Print the filtered dictionary
print(filtered_numbers)  # Output: {'b': 2, 'd': 4}
```
