## Exercise
Develop a program that allows two players to play Rock, Paper, Scissors. The program should determine the winner based on the game's rules.

## Reference Solution
```python
import random

def get_computer_choice():
    """Get the computer's random choice."""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def get_player_choice(player_name):
    """Get the player's choice."""
    while True:
        choice = input(f"{player_name}, enter your choice (rock, paper, or scissors): ")
        if choice.lower() in ['rock', 'paper', 'scissors']:
            return choice.lower()
        else:
            print("Invalid choice. Please try again.")

def determine_winner(player1_choice, player2_choice):
    """Determine the winner based on the game's rules."""
    if player1_choice == player2_choice:
        return "It's a tie!"
    if (player1_choice == 'rock' and player2_choice == 'scissors') or \
       (player1_choice == 'scissors' and player2_choice == 'paper') or \
       (player1_choice == 'paper' and player2_choice == 'rock'):
        return "Player 1 wins!"
    else:
        return "Player 2 wins!"

def play_game():
    """Play the Rock, Paper, Scissors game."""
    player1_name = input("Enter Player 1's name: ")
    player2_name = input("Enter Player 2's name: ")
    
    player1_choice = get_player_choice(player1_name)
    player2_choice = get_player_choice(player2_name)
    
    print(f"\n{player1_name} chose: {player1_choice}")
    print(f"{player2_name} chose: {player2_choice}\n")
    
    print(determine_winner(player1_choice, player2_choice))

def play_against_computer():
    """Play the Rock, Paper, Scissors game against the computer."""
    player_name = input("Enter your name: ")
    
    player_choice = get_player_choice(player_name)
    computer_choice = get_computer_choice()
    
    print(f"\n{player_name} chose: {player_choice}")
    print(f"Computer chose: {computer_choice}\n")
    
    print(determine_winner(player_choice, computer_choice))

def main():
    print("Welcome to Rock, Paper, Scissors!")
    print("1. Play against another player")
    print("2. Play against the computer")
    
    choice = input("Enter your choice (1 or 2): ")
    
    if choice == '1':
        play_game()
    elif choice == '2':
        play_against_computer()
    else:
        print("Invalid choice.")

if __name__ == "__main__":
    main()
```
