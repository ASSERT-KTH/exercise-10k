Create a program to solve a linear equation of the form ax + b = c.

def solve_linear_equation(a, b, c):
    """
    Solves a linear equation of the form ax + b = c.
    
    Args:
    a (float): Coefficient of x.
    b (float): Constant term on the left side.
    c (float): Constant term on the right side.
    
    Returns:
    float: Solution for x.
    """
    if a == 0:
        if b == c:
            return "Infinite solutions"
        else:
            return "No solution"
    else:
        return (c - b) / a

def main():
    # Get user input
    a = float(input("Enter coefficient a: "))
    b = float(input("Enter coefficient b: "))
    c = float(input("Enter coefficient c: "))

    # Solve the equation
    solution = solve_linear_equation(a, b, c)

    # Print the solution
    print("The solution is:", solution)

if __name__ == "__main__":
    main()
