## Exercise
Write a function to compress a given string by removing consecutive duplicates

## Reference Solution
```python
def compress_string(s: str) -> str:
    """
    Compress a given string by removing consecutive duplicates.

    Args:
    s (str): The input string.

    Returns:
    str: The compressed string.
    """
    if not s:  # Check if the string is empty
        return s

    compressed = [s[0]]  # Initialize the compressed list with the first character
    for char in s[1:]:  # Iterate over the string starting from the second character
        if char != compressed[-1]:  # If the current character is different from the last one in the compressed list
            compressed.append(char)  # Add it to the compressed list

    return ''.join(compressed)  # Join the compressed list into a string and return it


# Example usage:
if __name__ == "__main__":
    s = "aaaabbbccc"
    print(f"Original string: {s}")
    print(f"Compressed string: {compress_string(s)}")
```
