## Exercise
Write a program that demonstrates type coercion techniques.

## Reference Solution
```python
def main():
    # Implicit Type Coercion
    print("Implicit Type Coercion:")
    print(5 + 3.5)  # Adding an integer and a float
    print("Hello" + 5)  # Trying to add a string and an integer (will throw an error)

    # Explicit Type Coercion
    print("\nExplicit Type Coercion:")
    print(int(3.5))  # Converting a float to an integer
    print(float(5))  # Converting an integer to a float
    print(str(5))  # Converting an integer to a string

    # Using Type Coercion Functions
    print("\nUsing Type Coercion Functions:")
    print(bool(5))  # Converting an integer to a boolean
    print(complex(5, 3))  # Converting integers to a complex number

    # Real-World Example
    print("\nReal-World Example:")
    user_age = input("Please enter your age: ")
    try:
        age = int(user_age)  # Trying to convert the input to an integer
        print(f"Your age is: {age}")
    except ValueError:
        print("Invalid input. Please enter a valid age.")

if __name__ == "__main__":
    main()
```
