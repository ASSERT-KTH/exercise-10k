## Exercise
Write a program that analyzes a given text and reports its statistics (e.g. word count, average word length).

## Reference Solution
```python
import re
from collections import Counter

def analyze_text(text):
    """
    Analyze the given text and report its statistics.

    Args:
        text (str): The text to be analyzed.

    Returns:
        dict: A dictionary containing the text statistics.
    """
    # Remove punctuation and convert to lowercase
    text = re.sub(r'[^\w\s]', '', text).lower()

    # Split the text into words
    words = text.split()

    # Calculate the word count
    word_count = len(words)

    # Calculate the average word length
    total_chars = sum(len(word) for word in words)
    avg_word_length = total_chars / word_count if word_count > 0 else 0

    # Find the most common words
    word_freq = Counter(words)
    most_common_words = word_freq.most_common(10)

    # Create a dictionary to store the text statistics
    stats = {
        'word_count': word_count,
        'avg_word_length': avg_word_length,
        'most_common_words': most_common_words
    }

    return stats

def main():
    # Example usage
    text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
    stats = analyze_text(text)

    print("Text Statistics:")
    print("----------------")
    print(f"Word Count: {stats['word_count']}")
    print(f"Average Word Length: {stats['avg_word_length']:.2f}")
    print("Most Common Words:")
    for word, freq in stats['most_common_words']:
        print(f"{word}: {freq}")

if __name__ == "__main__":
    main()
```
