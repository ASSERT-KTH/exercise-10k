## Exercise
Write a function that takes a flattened dictionary and returns a nested dictionary

## Reference Solution
```python
def flatten_to_nested(flattened_dict, separator='.'):
    """
    Converts a flattened dictionary to a nested dictionary.

    Args:
        flattened_dict (dict): The flattened dictionary to convert.
        separator (str): The separator used to separate nested keys. Defaults to '.'.

    Returns:
        dict: The converted nested dictionary.
    """

    nested_dict = {}

    for key, value in flattened_dict.items():
        # Split the key into nested keys
        nested_keys = key.split(separator)

        # Initialize the current dictionary
        current_dict = nested_dict

        # Iterate over the nested keys
        for nested_key in nested_keys[:-1]:
            # If the nested key does not exist, create it
            if nested_key not in current_dict:
                current_dict[nested_key] = {}

            # Update the current dictionary
            current_dict = current_dict[nested_key]

        # Set the value for the last nested key
        current_dict[nested_keys[-1]] = value

    return nested_dict


# Example usage
if __name__ == "__main__":
    flattened_dict = {
        "a.b.c": 1,
        "a.b.d": 2,
        "a.e": 3,
        "f.g.h": 4,
    }

    nested_dict = flatten_to_nested(flattened_dict)

    print("Flattened Dictionary:")
    print(flattened_dict)

    print("\nNested Dictionary:")
    print(nested_dict)
```
