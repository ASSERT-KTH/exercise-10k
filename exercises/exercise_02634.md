## Exercise
Create a program that takes a question and answer as input and generates a quiz game.

## Reference Solution
```python
class QuizGame:
    def __init__(self):
        self.questions = {}

    def add_question(self, question, answer):
        """Add a question and answer to the quiz game"""
        self.questions[question] = answer
        print(f"Question '{question}' added successfully!")

    def view_questions(self):
        """View all questions in the quiz game"""
        if not self.questions:
            print("No questions added yet!")
        else:
            for question, answer in self.questions.items():
                print(f"Question: {question}")
                print(f"Answer: {answer}")
                print("------------------------")

    def play_quiz(self):
        """Play the quiz game"""
        if not self.questions:
            print("No questions added yet!")
        else:
            score = 0
            for question, answer in self.questions.items():
                user_answer = input(f"Question: {question}\nYour answer: ")
                if user_answer.lower() == answer.lower():
                    print("Correct answer!\n")
                    score += 1
                else:
                    print(f"Incorrect answer. The correct answer was {answer}.\n")
            print(f"Quiz finished! Your final score is {score} out of {len(self.questions)}.")

    def check_score(self, score):
        """Check the user's score"""
        if score == len(self.questions):
            print("Congratulations! You scored perfectly!")
        elif score / len(self.questions) >= 0.5:
            print("Good job! You scored above average.")
        else:
            print("Better luck next time! You scored below average.")


def main():
    quiz_game = QuizGame()

    while True:
        print("\nQuiz Game Menu:")
        print("1. Add Question")
        print("2. View Questions")
        print("3. Play Quiz")
        print("4. Exit")
        choice = input("Enter your choice: ")

        if choice == "1":
            question = input("Enter your question: ")
            answer = input("Enter the answer: ")
            quiz_game.add_question(question, answer)
        elif choice == "2":
            quiz_game.view_questions()
        elif choice == "3":
            quiz_game.play_quiz()
            score = sum(1 for q, a in quiz_game.questions.items() if input(f"Question: {q}\nYour answer: ").lower() == a.lower())
            quiz_game.check_score(score)
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
