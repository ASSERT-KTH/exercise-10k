## Exercise
Create a program that takes in a 3D shape and calculates its surface area using conditional statements.

## Reference Solution
```python
import math

def calculate_surface_area(shape, **kwargs):
    """
    Calculate the surface area of a 3D shape.

    Args:
        shape (str): The type of shape (cube, sphere, cylinder, cone).
        **kwargs: Shape-specific parameters (e.g., side_length, radius, height).

    Returns:
        float: The surface area of the shape.
    """

    if shape == "cube":
        # Calculate surface area of a cube
        return 6 * kwargs["side_length"] ** 2

    elif shape == "sphere":
        # Calculate surface area of a sphere
        return 4 * math.pi * kwargs["radius"] ** 2

    elif shape == "cylinder":
        # Calculate surface area of a cylinder
        return 2 * math.pi * kwargs["radius"] * (kwargs["height"] + kwargs["radius"])

    elif shape == "cone":
        # Calculate surface area of a cone
        slant_height = math.sqrt(kwargs["radius"] ** 2 + kwargs["height"] ** 2)
        return math.pi * kwargs["radius"] * (kwargs["radius"] + slant_height)

    else:
        raise ValueError("Invalid shape. Supported shapes: cube, sphere, cylinder, cone")


# Example usage
if __name__ == "__main__":
    print("Surface area of a cube (side length 5):", calculate_surface_area("cube", side_length=5))
    print("Surface area of a sphere (radius 3):", calculate_surface_area("sphere", radius=3))
    print("Surface area of a cylinder (radius 2, height 6):", calculate_surface_area("cylinder", radius=2, height=6))
    print("Surface area of a cone (radius 1, height 4):", calculate_surface_area("cone", radius=1, height=4))
```
