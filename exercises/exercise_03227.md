## Exercise
Write a program to check if a given string is a palindrome or not.

## Reference Solution
```python
def is_palindrome(s: str) -> bool:
    """
    Checks if a given string is a palindrome or not.

    Args:
        s (str): The input string to check.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    s = ''.join(c for c in s if c.isalnum()).lower()  # Remove non-alphanumeric characters and convert to lowercase
    return s == s[::-1]  # Compare the string with its reverse

def main():
    # Example usage
    strings_to_check = [
        "radar",
        "hello",
        "A man, a plan, a canal: Panama",
        "Not a palindrome"
    ]

    for s in strings_to_check:
        print(f"'{s}' is a palindrome: {is_palindrome(s)}")

if __name__ == "__main__":
    main()
```
