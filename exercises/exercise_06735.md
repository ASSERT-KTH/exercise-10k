import pandas as pd
from faker import Faker

def generate_input_data(num_records):
    """
    Generate input data programmatically.

    Args:
        num_records (int): Number of records to generate.

    Returns:
        pd.DataFrame: DataFrame containing the generated input data.
    """
    fake = Faker()

    data = {
        "id": range(1, num_records + 1),
        "name": [fake.name() for _ in range(num_records)],
        "email": [fake.email() for _ in range(num_records)],
        "phone_number": [fake.phone_number() for _ in range(num_records)],
        "address": [fake.address() for _ in range(num_records)],
    }

    df = pd.DataFrame(data)
    return df

# Example usage
if __name__ == "__main__":
    num_records = 10
    input_data = generate_input_data(num_records)
    print(input_data)
