## Exercise
Write a function that takes a dictionary and a key as input and returns True if the key is present in the dictionary and False otherwise using the in operator

def check_key(dictionary, key):
    """
    Checks if a key is present in a dictionary.

    Args:
    dictionary (dict): The dictionary to search in.
    key (any): The key to search for.

    Returns:
    bool: True if the key is present, False otherwise.
    """
    return key in dictionary

# Example usage
if __name__ == "__main__":
