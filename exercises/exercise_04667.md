## Exercise
Write a program that takes a string as input and compresses it by replacing repeated characters with a single character and a count.

## Reference Solution
```python
def compress_string(input_str):
    """
    Compress a string by replacing repeated characters with a single character and a count.

    Args:
        input_str (str): The input string to be compressed.

    Returns:
        str: The compressed string.
    """
    if not input_str:
        return ""

    compressed_str = []
    count = 1

    for i in range(1, len(input_str)):
        if input_str[i] == input_str[i - 1]:
            count += 1
        else:
            compressed_str.append(input_str[i - 1] + str(count))
            count = 1

    compressed_str.append(input_str[-1] + str(count))
    compressed_str = "".join(compressed_str)

    # If the compressed string is longer than the original string, return the original string
    if len(compressed_str) > len(input_str):
        return input_str

    return compressed_str


def main():
    input_str = input("Enter a string: ")
    compressed_str = compress_string(input_str)
    print("Compressed string:", compressed_str)


if __name__ == "__main__":
    main()
```
