def boolean_operations():

    x = True

    print("x or y:", x or y)

    # NOT operation (not x)
    print("not x:", not x)

    # XOR operation (x != y)
    print("x != y:", x != y)

    # Example usage: checking if a number is within a range
    num = 5
    print("Is num between 1 and 10?", 1 <= num <= 10)

    # Example usage: checking if a string contains a substring
    string = "Hello World"

# Call the function to demonstrate boolean operations
boolean_operations()
