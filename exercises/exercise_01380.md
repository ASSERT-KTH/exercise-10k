## Exercise
Create a program to perform complex calculations involving multiple arithmetic operations and functions.

## Reference Solution
```python
import math

def add(x, y):
    """Add two numbers."""
    return x + y

def subtract(x, y):
    """Subtract two numbers."""
    return x - y

def multiply(x, y):
    """Multiply two numbers."""
    return x * y

def divide(x, y):
    """Divide two numbers."""
    if y == 0:
        raise ValueError("Cannot divide by zero.")
    return x / y

def power(x, y):
    """Raise a number to a power."""
    return x ** y

def sqrt(x):
    """Calculate the square root of a number."""
    if x < 0:
        raise ValueError("Cannot calculate square root of negative number.")
    return math.sqrt(x)

def calculate(expression):
    """Evaluate a mathematical expression."""
    try:
        result = eval(expression)
        return result
    except Exception as e:
        raise ValueError("Invalid expression: " + str(e))

def main():
    print("Complex Calculator")
    print("------------------")

    while True:
        print("1. Basic Arithmetic Operations")
        print("2. Power")
        print("3. Square Root")
        print("4. Evaluate Expression")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            print("Basic Arithmetic Operations")
            print("-------------------------")
            print("1. Addition")
            print("2. Subtraction")
            print("3. Multiplication")
            print("4. Division")

            operation = input("Choose an operation: ")

            if operation == "1":
                x = float(input("Enter first number: "))
                y = float(input("Enter second number: "))
                print(f"{x} + {y} = {add(x, y)}")
            elif operation == "2":
                x = float(input("Enter first number: "))
                y = float(input("Enter second number: "))
                print(f"{x} - {y} = {subtract(x, y)}")
            elif operation == "3":
                x = float(input("Enter first number: "))
                y = float(input("Enter second number: "))
                print(f"{x} * {y} = {multiply(x, y)}")
            elif operation == "4":
                x = float(input("Enter first number: "))
                y = float(input("Enter second number: "))
                try:
                    print(f"{x} / {y} = {divide(x, y)}")
                except ValueError as e:
                    print(str(e))
            else:
                print("Invalid operation.")

        elif choice == "2":
            x = float(input("Enter base number: "))
            y = float(input("Enter exponent: "))
            print(f"{x} ^ {y} = {power(x, y)}")

        elif choice == "3":
            x = float(input("Enter number: "))

            except ValueError as e:
                print(str(e))

        elif choice == "4":
            expression = input("Enter mathematical expression: ")
            try:
                print(f"{expression} = {calculate(expression)}")
            except ValueError as e:
                print(str(e))

        elif choice == "5":
            print("Goodbye!")
            break

        else:
            print("Invalid choice.")

if __name__ == "__main__":
    main()
```
