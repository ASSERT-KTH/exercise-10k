## Exercise
Create a program that takes four numbers as input and prints out the biggest of the four.

## Reference Solution
```python
def find_biggest(num1, num2, num3, num4):
    """
    This function takes four numbers as input and returns the biggest one.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        num3 (float): The third number.
        num4 (float): The fourth number.

    Returns:
        float: The biggest of the four numbers.
    """
    return max(num1, num2, num3, num4)


def main():
    # Get the four numbers from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    num3 = float(input("Enter the third number: "))
    num4 = float(input("Enter the fourth number: "))

    # Find and print the biggest number
    biggest = find_biggest(num1, num2, num3, num4)
    print(f"The biggest number is: {biggest}")


if __name__ == "__main__":
    main()
```
