## Exercise
Write a program that takes a string and a pattern as input, and returns whether the string matches the pattern or not.

## Reference Solution
```python
import re

def match_string(input_string, pattern):
    """
    Matches a given string against a pattern.

    Args:
    input_string (str): The string to be matched.
    pattern (str): The pattern to match against.

    Returns:
    bool: True if the string matches the pattern, False otherwise.
    """

    # Check if the pattern is an exact match
    if pattern.startswith('^') and pattern.endswith('$'):
        # Remove the '^' and '$' characters from the pattern
        pattern = pattern[1:-1]
        return re.fullmatch(pattern, input_string) is not None

    # Check if the pattern is a prefix match
    elif pattern.startswith('^'):
        # Remove the '^' character from the pattern
        pattern = pattern[1:]
        return re.match(pattern, input_string) is not None

    # Check if the pattern is a suffix match
    elif pattern.endswith('$'):
        # Remove the '$' character from the pattern
        pattern = pattern[:-1]
        return re.search(pattern + '$', input_string) is not None

    # Default to substring matching
    else:
        return re.search(pattern, input_string) is not None


def main():
    input_string = input("Enter a string: ")
    pattern = input("Enter a pattern: ")

    if match_string(input_string, pattern):
        print("The string matches the pattern.")
    else:
        print("The string does not match the pattern.")


if __name__ == "__main__":
    main()
```
