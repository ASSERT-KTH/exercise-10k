## Exercise
Create a program to display a calendar for a given month and year, using nested for loops to handle each week.

## Reference Solution
```python
import calendar

def display_calendar(year, month):
    # Create a calendar object
    cal = calendar.monthcalendar(year, month)

    # Print the month and year
    print(calendar.month_name[month], year)
    print("Mo Tu We Th Fr Sa Su")

    # Iterate over each week in the month
    for week in cal:
        # Iterate over each day in the week
        for day in week:
            # If the day is 0, it means it's not in the current month, so print a space
            if day == 0:
                print("  ", end=" ")
            # Otherwise, print the day
            else:
                print(f"{day:2}", end=" ")
        # Move to the next line after each week
        print()

# Test the function
year = int(input("Enter the year: "))
month = int(input("Enter the month (1-12): "))
display_calendar(year, month)
```
