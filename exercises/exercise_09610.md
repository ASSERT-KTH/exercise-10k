## Exercise
Write a program that simulates a library, allowing the user to add, remove, and search for books

## Reference Solution
```python
# library_simulator.py

class Book:
    def __init__(self, id, title, author, publication_year):
        self.id = id
        self.title = title
        self.author = author
        self.publication_year = publication_year

class Library:
    def __init__(self):
        self.books = {}

    def add_book(self, id, title, author, publication_year):
        """Add a new book to the library"""
        if id in self.books:
            print("Book with this ID already exists.")
        else:
            self.books[id] = Book(id, title, author, publication_year)
            print("Book added successfully.")

    def remove_book(self, id):
        """Remove a book from the library"""
        if id in self.books:
            del self.books[id]
            print("Book removed successfully.")
        else:
            print("Book with this ID does not exist.")

    def search_book(self, query):
        """Search for books by title, author, or publication year"""
        results = []
        for book in self.books.values():
            if query.lower() in book.title.lower() or query.lower() in book.author.lower() or query.lower() in str(book.publication_year):
                results.append(book)
        if results:
            print("Search results:")
            for book in results:
                print(f"ID: {book.id}, Title: {book.title}, Author: {book.author}, Publication Year: {book.publication_year}")
        else:
            print("No matching books found.")

    def list_books(self):
        """List all books in the library"""
        if self.books:
            print("Library Catalog:")
            for book in self.books.values():
                print(f"ID: {book.id}, Title: {book.title}, Author: {book.author}, Publication Year: {book.publication_year}")
        else:
            print("Library is empty.")

def main():
    library = Library()
    while True:
        print("\nLibrary Simulator Menu:")
        print("1. Add a book")
        print("2. Remove a book")
        print("3. Search for a book")
        print("4. List all books")
        print("5. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            id = input("Enter book ID: ")
            title = input("Enter book title: ")
            author = input("Enter book author: ")
            publication_year = int(input("Enter book publication year: "))
            library.add_book(id, title, author, publication_year)
        elif choice == "2":
            id = input("Enter book ID: ")
            library.remove_book(id)
        elif choice == "3":
            query = input("Enter search query: ")
            library.search_book(query)
        elif choice == "4":
            library.list_books()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
