## Exercise
Create a program that takes in an amount in one currency and converts it to another currency using conditional statements.

def currency_converter(amount, from_currency, to_currency):
    """
    Converts an amount from one currency to another.

    Args:
        amount (float): The amount to convert.
        from_currency (str): The currency to convert from.
        to_currency (str): The currency to convert to.

    Returns:
        float: The converted amount.
    """

    # Define conversion rates
    conversion_rates = {
        'USD': {'EUR': 0.88, 'GBP': 0.76, 'INR': 74.83, 'AUD': 1.31},
        'EUR': {'USD': 1.14, 'GBP': 0.86, 'INR': 84.93, 'AUD': 1.49},
        'GBP': {'USD': 1.32, 'EUR': 1.16, 'INR': 98.36, 'AUD': 1.73},
        'INR': {'USD': 0.013, 'EUR': 0.012, 'GBP': 0.010, 'AUD': 0.018},
        'AUD': {'USD': 0.76, 'EUR': 0.67, 'GBP': 0.58, 'INR': 56.85}
    }

    # Check if from_currency and to_currency are valid
    if from_currency not in conversion_rates or to_currency not in conversion_rates:
        return "Invalid currency"

    # Check if from_currency and to_currency are the same
    if from_currency == to_currency:
        return amount

    # Convert amount
    if from_currency == 'USD':
        if to_currency == 'EUR':
            return amount * conversion_rates['USD']['EUR']
        elif to_currency == 'GBP':
            return amount * conversion_rates['USD']['GBP']
        elif to_currency == 'INR':
            return amount * conversion_rates['USD']['INR']
        elif to_currency == 'AUD':
            return amount * conversion_rates['USD']['AUD']
    elif from_currency == 'EUR':
        if to_currency == 'USD':
            return amount * conversion_rates['EUR']['USD']
        elif to_currency == 'GBP':
            return amount * conversion_rates['EUR']['GBP']
        elif to_currency == 'INR':
            return amount * conversion_rates['EUR']['INR']
        elif to_currency == 'AUD':
            return amount * conversion_rates['EUR']['AUD']
    elif from_currency == 'GBP':
        if to_currency == 'USD':
            return amount * conversion_rates['GBP']['USD']
        elif to_currency == 'EUR':
            return amount * conversion_rates['GBP']['EUR']
        elif to_currency == 'INR':
            return amount * conversion_rates['GBP']['INR']
        elif to_currency == 'AUD':
            return amount * conversion_rates['GBP']['AUD']
    elif from_currency == 'INR':
        if to_currency == 'USD':
            return amount * conversion_rates['INR']['USD']
        elif to_currency == 'EUR':
            return amount * conversion_rates['INR']['EUR']
        elif to_currency == 'GBP':
            return amount * conversion_rates['INR']['GBP']
        elif to_currency == 'AUD':
            return amount * conversion_rates['INR']['AUD']
    elif from_currency == 'AUD':
        if to_currency == 'USD':
            return amount * conversion_rates['AUD']['USD']
        elif to_currency == 'EUR':
            return amount * conversion_rates['AUD']['EUR']
        elif to_currency == 'GBP':
            return amount * conversion_rates['AUD']['GBP']
        elif to_currency == 'INR':
            return amount * conversion_rates['AUD']['INR']

def main():
    amount = float(input("Enter the amount: "))
    from_currency = input("Enter the currency to convert from (USD, EUR, GBP, INR, AUD): ").upper()
    to_currency = input("Enter the currency to convert to (USD, EUR, GBP, INR, AUD): ").upper()

    result = currency_converter(amount, from_currency, to_currency)
    if isinstance(result, str):
        print(result)
    else:
        print(f"{amount} {from_currency} is equal to {result} {to_currency}")

if __name__ == "__main__":
    main()
