## Exercise
Create a program that uses a while loop to generate the first 10 numbers in the Fibonacci sequence.

## Reference Solution
```python
# Initialize the first two numbers in the sequence
a, b = 0, 1

# Print the first two numbers
print(a, b, end=' ')

# Initialize a counter to keep track of the number of terms
i = 2

# Generate the next 8 numbers in the sequence
while i < 10:
    # Calculate the next number as the sum of the previous two
    next_number = a + b
    
    # Print the next number
    print(next_number, end=' ')
    
    # Update the previous two numbers for the next iteration
    a, b = b, next_number
    
    # Increment the counter
    i += 1
```

```python
def generate_fibonacci(n):
    """
    Generate the first n numbers in the Fibonacci sequence.
    
    Args:
        n (int): The number of terms to generate.
    
    Returns:
        None
    """
    # Initialize the first two numbers in the sequence
    a, b = 0, 1

    # Print the first two numbers
    print(a, b, end=' ')

    # Initialize a counter to keep track of the number of terms
    i = 2

    # Generate the next n-2 numbers in the sequence
    while i < n:
        # Calculate the next number as the sum of the previous two
        next_number = a + b
        
        # Print the next number
        print(next_number, end=' ')
        
        # Update the previous two numbers for the next iteration
        a, b = b, next_number
        
        # Increment the counter
        i += 1


def main():
    n = 10  # Change this to generate a different number of terms
    print(f"First {n} numbers in the Fibonacci sequence:")
    generate_fibonacci(n)


if __name__ == "__main__":
    main()
```
