## Exercise
Create a function that calculates the average grade of a list of students.

## Reference Solution
```python
def calculate_average_grade(students):
    """
    Calculate the average grade for each student in the list.

    Args:
    students (list): A list of dictionaries, where each dictionary contains a student's name and grades.

    Returns:
    dict: A dictionary with student names as keys and their average grades as values.
    """
    average_grades = {}
    for student in students:
        name = student['name']
        grades = student['grades']
        average_grade = sum(grades) / len(grades)
        average_grades[name] = average_grade
    return average_grades

# Example usage:
students = [
    {'name': 'John', 'grades': [90, 80, 70]},
    {'name': 'Alice', 'grades': [95, 85, 75]},
    {'name': 'Bob', 'grades': [92, 82, 72]}
]

average_grades = calculate_average_grade(students)
for name, average_grade in average_grades.items():
    print(f"{name}: {average_grade:.2f}")
```
