## Exercise
Write a function that uses error handling to catch and handle exceptions

## Reference Solution
```python
def divide_numbers(num1, num2):
    """
    Divide two numbers and handle potential exceptions.

    Args:
        num1 (float): The dividend.
        num2 (float): The divisor.

    Returns:
        float: The quotient of num1 and num2.

    Raises:
        TypeError: If either num1 or num2 is not a number.
        ZeroDivisionError: If num2 is zero.
    """

    try:
        # Attempt to divide num1 by num2
        quotient = num1 / num2
        return quotient

    except TypeError:
        # Handle TypeError if either num1 or num2 is not a number
        print("Error: Both inputs must be numbers.")
        return None

    except ZeroDivisionError:
        # Handle ZeroDivisionError if num2 is zero
        print("Error: Cannot divide by zero.")
        return None

    except Exception as e:
        # Handle any other unexpected exceptions
        print(f"An unexpected error occurred: {e}")
        return None


# Example usage
if __name__ == "__main__":
    print(divide_numbers(10, 2))  # Expected output: 5.0
    print(divide_numbers(10, 0))  # Expected output: Error: Cannot divide by zero.
    print(divide_numbers("ten", 2))  # Expected output: Error: Both inputs must be numbers.
```
