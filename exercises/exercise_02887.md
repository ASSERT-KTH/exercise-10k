## Exercise
Write a program to generate the terms of an arithmetic progression (e.g., 2, 5, 8, 11) based on user input.

## Reference Solution
```python
def generate_arithmetic_progression(first_term, common_difference, num_terms):
    """
    Generates the terms of an arithmetic progression.

    Args:
        first_term (int): The first term of the progression.
        common_difference (int): The common difference between terms.
        num_terms (int): The number of terms to generate.

    Returns:
        list: A list of terms in the arithmetic progression.
    """
    progression = [first_term + i * common_difference for i in range(num_terms)]
    return progression

def main():
    # Get user input
    first_term = int(input("Enter the first term: "))
    common_difference = int(input("Enter the common difference: "))
    num_terms = int(input("Enter the number of terms: "))

    # Generate the arithmetic progression
    progression = generate_arithmetic_progression(first_term, common_difference, num_terms)

    # Print the progression
    print("Arithmetic Progression:")
    print(progression)

if __name__ == "__main__":
    main()
```
