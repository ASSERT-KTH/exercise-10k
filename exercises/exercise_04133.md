## Exercise
Check the data type of a variable and perform actions based on the type

## Reference Solution
```python
def check_data_type(var):
    """
    Check the data type of a variable and perform actions based on the type.
    
    Args:
        var: The variable to check.
    
    Returns:
        None
    """
    if isinstance(var, int):
        print(f"The variable is an integer: {var}")
        # Perform action for integer
        print(f"Square of {var}: {var ** 2}")
    elif isinstance(var, float):
        print(f"The variable is a float: {var}")
        # Perform action for float
        print(f"Square root of {var}: {var ** 0.5}")
    elif isinstance(var, str):
        print(f"The variable is a string: {var}")
        # Perform action for string
        print(f"Uppercase: {var.upper()}")
    elif isinstance(var, list):
        print(f"The variable is a list: {var}")
        # Perform action for list
        print(f"Length of list: {len(var)}")
    else:
        print(f"Unsupported data type: {type(var)}")

# Example usage
check_data_type(10)  # integer
check_data_type(10.5)  # float
check_data_type("hello")  # string
check_data_type([1, 2, 3])  # list
```
