## Exercise
Write a program to take a string input from the user and print it to the screen, using regular expressions to validate the input.

## Reference Solution
```python
import re

def validate_input(input_str):
    """
    Validate the input string using regular expressions.

    Args:
        input_str (str): The input string to be validated.

    Returns:
        bool: True if the input is valid, False otherwise.
    """
    # Define the regular expression pattern
    pattern = r"^[a-zA-Z0-9\s]+$"

    # Check if the input string matches the pattern
    if re.match(pattern, input_str):
        return True
    else:
        return False

def main():
    # Get the input string from the user
    input_str = input("Please enter a string: ")

    # Validate the input string
    if validate_input(input_str):
        # Print the input string if it's valid
        print("Valid input:", input_str)
    else:
        # Print an error message if the input is invalid
        print("Invalid input. Please enter a string containing only letters, numbers, and spaces.")

if __name__ == "__main__":
    main()
```
