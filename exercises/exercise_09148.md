## Exercise
Create a variable inside a function and access it outside the function, then print out the variable's value.

## Reference Solution
```python
def my_function():
    """Define a variable inside the function and return it."""
    my_variable = "Hello, World!"
    return my_variable

# Access the variable outside the function
variable_value = my_function()
print(variable_value)  # Output: Hello, World!
```

```python
my_variable = None  # Initialize the global variable

def my_function():
    """Define a global variable inside the function and modify it."""
    global my_variable
    my_variable = "Hello, World!"

# Call the function to modify the global variable
my_function()

# Access the global variable outside the function
print(my_variable)  # Output: Hello, World!
```

```python
class MyClass:
    """Define a class with a variable as an attribute."""
    def __init__(self):
        self.my_variable = None

    def my_function(self):
        """Modify the class attribute."""
        self.my_variable = "Hello, World!"

# Create an instance of the class
obj = MyClass()

# Call the function to modify the class attribute
obj.my_function()

# Access the class attribute outside the function
print(obj.my_variable)  # Output: Hello, World!
```
