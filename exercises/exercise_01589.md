## Exercise
Write a program to calculate a student's grade based on their scores on different assignments.

## Reference Solution
```python
class Student:
    def __init__(self, name):
        self.name = name
        self.assignments = {}

    def add_assignment(self, name, score, weight):
        """
        Add an assignment to the student's record.

        Args:
            name (str): Name of the assignment.
            score (float): Score achieved on the assignment (out of 100).
            weight (float): Weight of the assignment in the overall grade.
        """
        self.assignments[name] = {'score': score, 'weight': weight}

    def calculate_grade(self):
        """
        Calculate the student's overall grade.

        Returns:
            float: The student's overall grade (out of 100).
        """
        total_weight = sum(assignment['weight'] for assignment in self.assignments.values())
        if total_weight != 1:
            raise ValueError("Weights must add up to 1")

        total_score = sum(assignment['score'] * assignment['weight'] for assignment in self.assignments.values())
        return total_score


def main():
    student_name = input("Enter student's name: ")
    student = Student(student_name)

    num_assignments = int(input("Enter number of assignments: "))

    for i in range(num_assignments):
        assignment_name = input(f"Enter name of assignment {i+1}: ")
        score = float(input(f"Enter score for assignment {i+1} (out of 100): "))
        weight = float(input(f"Enter weight of assignment {i+1} (out of 1): "))
        student.add_assignment(assignment_name, score, weight)

    grade = student.calculate_grade()
    print(f"{student_name}'s overall grade is: {grade:.2f}")


if __name__ == "__main__":
    main()
```
