## Exercise
Create a program that stores student grades in a dictionary. The dictionary keys should be the student names and the values should be another dictionary with the subject names as keys and the grades as values. The program should allow the user to add a student, add a subject grade for a student, and print all the grades for a student.

## Reference Solution
```python
class StudentGradebook:
    def __init__(self):
        self.student_grades = {}

    def add_student(self, name):
        """Add a student to the gradebook."""
        if name not in self.student_grades:
            self.student_grades[name] = {}
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists in the gradebook.")

    def add_subject_grade(self, name, subject, grade):
        """Add a subject grade for a student."""
        if name in self.student_grades:
            if subject not in self.student_grades[name]:
                self.student_grades[name][subject] = grade
                print(f"Grade for '{subject}' added successfully for student '{name}'.")
            else:
                print(f"Subject '{subject}' already exists for student '{name}'.")
        else:
            print(f"Student '{name}' does not exist in the gradebook. Please add the student first.")

    def print_student_grades(self, name):
        """Print all the grades for a student."""
        if name in self.student_grades:
            print(f"Grades for student '{name}':")
            for subject, grade in self.student_grades[name].items():
                print(f"{subject}: {grade}")
        else:
            print(f"Student '{name}' does not exist in the gradebook.")


def main():
    gradebook = StudentGradebook()

    while True:
        print("\nStudent Gradebook Menu:")
        print("1. Add a student")
        print("2. Add a subject grade for a student")
        print("3. Print all the grades for a student")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter the student's name: ")
            gradebook.add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            subject = input("Enter the subject name: ")
            grade = input("Enter the grade: ")
            gradebook.add_subject_grade(name, subject, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            gradebook.print_student_grades(name)
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
