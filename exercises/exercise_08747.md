## Exercise
Cache variables to improve performance and reduce computation time.

## Reference Solution
```python
import functools

def cache_results(func):
    """Cache the results of a function."""
    cache = {}
    @functools.wraps(func)
    def wrapper(*args):
        if args in cache:
            return cache[args]
        result = func(*args)
        cache[args] = result
        return result
    return wrapper

@cache_results
def expensive_computation(x):
    """Simulate an expensive computation."""
    import time
    time.sleep(2)  # Simulate a 2-second computation
    return x * x

print(expensive_computation(5))  # Takes 2 seconds
print(expensive_computation(5))  # Returns immediately from cache
```
