## Exercise
Create a program to calculate the greatest common divisor (GCD) of two or more integers using the Euclidean algorithm.

## Reference Solution
```python
import math

def gcd_euclidean(a, b):
    """Calculate the GCD of two integers using the Euclidean algorithm."""
    while b != 0:
        a, b = b, a % b
    return abs(a)

def gcd_multiple(*args):
    """Calculate the GCD of multiple integers."""
    result = args[0]
    for num in args[1:]:
        result = gcd_euclidean(result, num)
    return result

def main():
    # Test the function with two integers
    num1 = 48
    num2 = 18
    print(f"GCD of {num1} and {num2}: {gcd_euclidean(num1, num2)}")

    # Test the function with multiple integers
    numbers = [12, 24, 36, 48]
    print(f"GCD of {numbers}: {gcd_multiple(*numbers)}")

if __name__ == "__main__":
    main()
```
