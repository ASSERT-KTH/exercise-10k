## Exercise
Write a program to take a set of questions and answers as input and ask the user to answer the questions, keeping track of the score

## Reference Solution
```python
# question_answerer.py

class QuestionAnswerer:
    def __init__(self, questions):
        """
        Initialize the QuestionAnswerer with a dictionary of questions and answers.

        Args:
            questions (dict): A dictionary where the keys are the questions and the values are the answers.
        """
        self.questions = questions
        self.score = 0

    def ask_questions(self):
        """
        Ask the user to answer the questions and keep track of the score.
        """
        for question, answer in self.questions.items():
            user_answer = input(question + " ")
            if user_answer.lower() == answer.lower():
                print("Correct!\n")
                self.score += 1
            else:
                print(f"Sorry, that's incorrect. The correct answer is {answer}.\n")

    def print_score(self):
        """
        Print the final score.
        """
        print(f"Your final score is {self.score} out of {len(self.questions)}")


def main():
    # Example usage
    questions = {
        "What is the capital of France?": "Paris",
        "What is the largest planet in our solar system?": "Jupiter",
        "Who wrote the book 'To Kill a Mockingbird'?": "Harper Lee"
    }

    question_answerer = QuestionAnswerer(questions)
    print("Welcome to the quiz! Please answer the following questions:")
    question_answerer.ask_questions()
    question_answerer.print_score()


if __name__ == "__main__":
    main()
```
