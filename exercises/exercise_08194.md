## Exercise
Write a program that parses an XML file using a while loop.

## Reference Solution
```python
import xml.etree.ElementTree as ET

def parse_xml(file_name):
    # Parse the XML file
    tree = ET.parse(file_name)
    root = tree.getroot()

    # Initialize the current element
    current_element = root

    # Initialize a stack to keep track of the current element's children
    stack = list(current_element)

    # Iterate over the elements using a while loop
    while stack:
        current_element = stack.pop(0)

        # Print the element's tag and text
        print(f"Tag: {current_element.tag}, Text: {current_element.text}")

        # Add the current element's children to the stack
        if len(current_element) > 0:
            stack = current_element + stack

# Call the function
parse_xml('example.xml')
```
