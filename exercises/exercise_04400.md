Use tree data structure with variables

```python
class TreeNode:
    """Represents a node in the tree."""
    def __init__(self, value):
        self.value = value  # Store the node value
        self.children = []  # Store child nodes

class Tree:
    """Represents the tree itself."""
    def __init__(self, root):
        self.root = TreeNode(root)  # Initialize the tree with a root node

    def add_child(self, parent_value, child_value):
        """Add a child node to the tree."""
        parent_node = self.find_node(self.root, parent_value)
        if parent_node:
            parent_node.children.append(TreeNode(child_value))

    def find_node(self, node, value):
        """Find a node in the tree by its value."""
        if node.value == value:
            return node
        for child in node.children:
            found_node = self.find_node(child, value)
            if found_node:
                return found_node
        return None

    def print_tree(self, node, level=0):
        """Print the tree structure."""
        print(' ' * 4 * level + '->', node.value)
        level += 1
        for child in node.children:
            self.print_tree(child, level)


# Example usage:
tree = Tree("Root")
tree.add_child("Root", "Child 1")
tree.add_child("Root", "Child 2")
tree.add_child("Child 1", "Grandchild 1")
tree.add_child("Child 1", "Grandchild 2")
tree.add_child("Child 2", "Grandchild 3")

print("Tree structure:")
tree.print_tree(tree.root)
```
