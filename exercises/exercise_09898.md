## Exercise
Write a program that generates the Fibonacci sequence up to a given number.

## Reference Solution
```python
def fibonacci(n):
    """
    Generate the Fibonacci sequence up to the nth number.

    Args:
        n (int): The number of terms in the sequence.

    Returns:
        list: A list of integers representing the Fibonacci sequence.
    """
    sequence = [0, 1]
    while len(sequence) < n:
        sequence.append(sequence[-1] + sequence[-2])
    return sequence

def fibonacci_up_to(max_value):
    """
    Generate the Fibonacci sequence up to a given maximum value.

    Args:
        max_value (int): The maximum value in the sequence.

    Returns:
        list: A list of integers representing the Fibonacci sequence.
    """
    sequence = [0, 1]
    while sequence[-1] + sequence[-2] <= max_value:
        sequence.append(sequence[-1] + sequence[-2])
    return sequence

# Example usage:
if __name__ == "__main__":
    n = 10
    max_value = 100
    print(f"Fibonacci sequence up to {n} numbers: {fibonacci(n)}")
    print(f"Fibonacci sequence up to {max_value}: {fibonacci_up_to(max_value)}")
```
