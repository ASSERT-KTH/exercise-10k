## Exercise
Write a program that translates English words into Pig Latin.

## Reference Solution
```python
def translate_to_pig_latin(word):
    """
    Translates an English word into Pig Latin.

    Args:
        word (str): The English word to translate.

    Returns:
        str: The Pig Latin translation of the word.
    """
    vowels = 'aeiou'
    if word[0].lower() in vowels:
        return word + 'way'
    else:
        consonant_cluster = ''
        for char in word:
            if char.lower() not in vowels:
                consonant_cluster += char
            else:
                break
        return word[len(consonant_cluster):] + consonant_cluster + 'ay'

def main():
    print("Pig Latin Translator")
    print("--------------------")
    while True:
        word = input("Enter an English word (or 'quit' to exit): ")
        if word.lower() == 'quit':
            break
        print(f"The Pig Latin translation of '{word}' is: {translate_to_pig_latin(word)}")

if __name__ == "__main__":
    main()
```
