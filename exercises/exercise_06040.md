Write a function that simulates a real-world scenario, such as calculating the area of a room or the cost of an item. Use the function to calculate the result of the scenario.

def calculate_item_cost(base_price, tax_rate, discount_rate):
    """
    Calculate the cost of an item after applying tax and discount.

    Args:
        base_price (float): The base price of the item.
        tax_rate (float): The tax rate as a decimal (e.g., 8% = 0.08).
        discount_rate (float): The discount rate as a decimal (e.g., 10% = 0.10).

    Returns:
        float: The final cost of the item.
    """
    # Calculate the tax amount
    tax_amount = base_price * tax_rate
    
    # Calculate the discount amount
    discount_amount = base_price * discount_rate
    
    # Calculate the final cost
    final_cost = base_price + tax_amount - discount_amount
    
    return final_cost

# Example usage
base_price = 100.0  # Base price of the item
tax_rate = 0.08  # 8% tax rate
discount_rate = 0.10  # 10% discount rate

final_cost = calculate_item_cost(base_price, tax_rate, discount_rate)
print(f"Final cost: ${final_cost:.2f}")
