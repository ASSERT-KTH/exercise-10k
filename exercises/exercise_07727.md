## Exercise
Find all anagrams of a given string in a larger string.

## Reference Solution
```python
from typing import List

def find_anagrams(s: str, p: str) -> List[int]:
    """
    Finds all anagrams of string p in string s.

    Args:
    s (str): The larger string to search in.
    p (str): The string for which anagrams are to be found.

    Returns:
    List[int]: A list of indices where anagrams of p are found in s.
    """
    p_sorted = sorted(p)  # Reference sorted string
    anagram_indices = []  # List to store indices of anagrams

    # Iterate over the larger string using a sliding window
    for i in range(len(s) - len(p) + 1):
        window = s[i:i + len(p)]  # Current window
        window_sorted = sorted(window)  # Sorted characters in the window

        # Check if the sorted window matches the reference sorted string
        if window_sorted == p_sorted:
            anagram_indices.append(i)  # Store the index if it's an anagram

    return anagram_indices

# Example usage:
s = "abxaba"
p = "ab"
print(find_anagrams(s, p))  # Output: [0, 3, 4]
```
