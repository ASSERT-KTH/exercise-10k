## Exercise
Write a program that calculates the least common multiple (LCM) of two given numbers.

## Reference Solution
```python
import math

def calculate_lcm(num1: int, num2: int) -> int:
    """
    Calculate the least common multiple (LCM) of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The LCM of num1 and num2.
    """
    # Calculate the GCD of num1 and num2
    gcd = math.gcd(num1, num2)

    # Calculate the LCM using the formula: LCM(a, b) = |a*b| / GCD(a, b)
    lcm = abs(num1 * num2) // gcd

    return lcm

def main():
    # Example usage:
    num1 = 12
    num2 = 15

    lcm = calculate_lcm(num1, num2)
    print(f"The LCM of {num1} and {num2} is: {lcm}")

if __name__ == "__main__":
    main()
```
