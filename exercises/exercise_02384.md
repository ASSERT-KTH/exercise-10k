## Exercise
Write a program that plays a guessing game with the user, where the user has to guess a random number between 1 and 100 using a while loop.

## Reference Solution
```python
import random

def guessing_game():
    # Generate a random number between 1 and 100
    number_to_guess = random.randint(1, 100)
    # Initialize the number of tries the player has made
    tries = 0

    print("Welcome to the number guessing game!")
    print("I'm thinking of a number between 1 and 100.")
    print("Try to guess the number in as few attempts as possible.\n")

    while True:
        # Ask the player for their guess
        user_guess = input("Take a guess: ")
        
        # Check if the user wants to quit
        if user_guess.lower() == "quit":
            print("Okay, I was thinking of number {}.".format(number_to_guess))
            break
        
        # Try to convert the user's guess to an integer
        try:
            user_guess = int(user_guess)
        except ValueError:
            print("That's not a valid number!")
            continue

        # Increment the number of tries the player has made
        tries += 1

        # Check if the user's guess is higher or lower than the number to guess
        if user_guess < number_to_guess:
            print("Too low! Try again.")
        elif user_guess > number_to_guess:
            print("Too high! Try again.")
        else:
            print("Congratulations! You found the number in {} tries.".format(tries))
            break

if __name__ == "__main__":
    guessing_game()
```
