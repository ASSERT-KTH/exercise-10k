## Exercise
Write a program that demonstrates the concept of constant variables.

## Reference Solution
```python
# Define a constant variable
PI = 3.14159

def main():
    # Attempt to change the value of the constant variable
    try:
        PI = 2.71828
    except Exception as e:
        # Since Python does not have a built-in constant variable type, we'll use a NameError to simulate an error
        print("Error: Cannot change the value of a constant variable.")

    # Print the original value of the constant variable
    print("The value of PI is:", PI)

if __name__ == "__main__":
    main()
```

```python
class Constants:
    def __init__(self):
        self._PI = 3.14159

    @property
    def PI(self):
        return self._PI

    @PI.setter
    def PI(self, value):
        raise ValueError("Cannot change the value of a constant variable.")

def main():
    constants = Constants()
    print("The value of PI is:", constants.PI)

    # Attempt to change the value of the constant variable
    try:
        constants.PI = 2.71828
    except ValueError as e:
        print("Error:", e)

if __name__ == "__main__":
    main()
```
