## Exercise
Write a program that converts a given time from one time zone to another

## Reference Solution
```python
from datetime import datetime
import pytz

def convert_time_zone(date_string, from_tz, to_tz, date_format="%Y-%m-%d %H:%M:%S"):
    """
    Convert a given time from one time zone to another.

    Args:
    - date_string (str): The date and time to convert.
    - from_tz (str): The time zone of the input date and time.
    - to_tz (str): The time zone to convert to.
    - date_format (str): The format of the input date and time.

    Returns:
    - converted_date (str): The converted date and time in the target time zone.
    """

    # Parse the input date and time
    from_date = datetime.strptime(date_string, date_format)

    # Set the time zone of the input date and time
    from_date_tz = pytz.timezone(from_tz)
    from_date_tz_aware = from_date_tz.localize(from_date)

    # Convert to the target time zone
    to_date_tz = pytz.timezone(to_tz)
    to_date = from_date_tz_aware.astimezone(to_date_tz)

    # Format the converted date and time
    converted_date = to_date.strftime(date_format)

    return converted_date

def main():
    # Example usage
    date_string = "2022-01-01 12:00:00"
    from_tz = "America/New_York"
    to_tz = "Europe/London"

    converted_date = convert_time_zone(date_string, from_tz, to_tz)
    print(f"Converted date and time in {to_tz}: {converted_date}")

if __name__ == "__main__":
    main()
```
