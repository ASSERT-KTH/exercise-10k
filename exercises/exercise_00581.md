## Exercise
Write a program that takes an array of numbers as input and returns the maximum number in the array.

def find_max_number(array):
    if not array:
        return None
    return max(array)

numbers = [4, 2, 9, 6, 5, 1]

def find_max_number(array):
    if not array:
        return None
    max_number = array[0]
    for num in array:
        if num > max_number:
            max_number = num
    return max_number

# Example usage:
numbers = [4, 2, 9, 6, 5, 1]
max_number = find_max_number(numbers)
print("Max number:", max_number)
