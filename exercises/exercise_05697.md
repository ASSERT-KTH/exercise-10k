## Exercise
Create a program that takes a phone number as input and checks if it's valid using conditional statements.

## Reference Solution
```python
import re

def validate_phone_number(phone_number):
    """
    Validate a phone number.

    Args:
        phone_number (str): The phone number to validate.

    Returns:
        bool: True if the phone number is valid, False otherwise.
    """
    # Check if the phone number contains only digits, hyphens, and parentheses
    if not re.match(r'^[\d\-\(\)\s]+$', phone_number):
        return False

    # Remove all non-digit characters
    digits = re.sub(r'\D', '', phone_number)

    # Check if the phone number has exactly 10 digits
    if len(digits) != 10:
        return False

    # Check if the phone number is in the correct format
    if not re.match(r'^\d{3}-\d{3}-\d{4}$', phone_number) and not re.match(r'^\(\d{3}\) \d{3}-\d{4}$', phone_number):
        return False

    return True

def main():
    phone_number = input("Enter a phone number: ")
    if validate_phone_number(phone_number):
        print("The phone number is valid.")
    else:
        print("The phone number is not valid.")

if __name__ == "__main__":
    main()
```
