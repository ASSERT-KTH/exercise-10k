## Exercise
Create a function to generate all permutations of a given string

## Reference Solution
```python
def generate_permutations(s):
    """
    Generates all permutations of a given string.

    Args:
        s (str): The input string.

    Returns:
        list: A list of all permutations of the input string.
    """
    if len(s) == 1:
        return [s]

    permutations = []
    for i, char in enumerate(s):
        remaining_string = s[:i] + s[i+1:]
        for perm in generate_permutations(remaining_string):
            permutations.append(char + perm)

    return permutations

# Example usage:
def main():
    input_string = "abc"
    permutations = generate_permutations(input_string)
    print("Permutations of '{}'".format(input_string))
    for i, perm in enumerate(permutations):
        print("{}. {}".format(i+1, perm))

if __name__ == "__main__":
    main()
```
