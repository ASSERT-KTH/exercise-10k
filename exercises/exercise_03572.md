## Exercise
Write a function that validates a dictionary against a given schema

## Reference Solution
```python
import jsonschema

def validate_dict(schema, data):
    """
    Validates a dictionary against a given schema.

    Args:
    - schema (dict): The schema to validate against.
    - data (dict): The dictionary to be validated.

    Returns:
    - bool: True if the dictionary is valid, False otherwise.
    - list: A list of validation errors.
    """
    try:
        jsonschema.validate(instance=data, schema=schema)
        return True, []
    except jsonschema.exceptions.ValidationError as err:
        return False, [err.message]

# Example usage:
if __name__ == "__main__":
    schema = {
        "type": "object",
        "properties": {
            "name": {"type": "string"},
            "age": {"type": "integer"}
        },
        "required": ["name", "age"]
    }

    data = {
        "name": "John Doe",
        "age": 30
    }

    is_valid, errors = validate_dict(schema, data)
    print(f"Is valid: {is_valid}")
    print(f"Errors: {errors}")

    data = {
        "name": "Jane Doe"
    }

    is_valid, errors = validate_dict(schema, data)
    print(f"Is valid: {is_valid}")
    print(f"Errors: {errors}")
```
