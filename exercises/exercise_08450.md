## Exercise
Write a program that takes in a bill amount and calculates the corresponding tip using conditional statements.

## Reference Solution
```python
def calculate_tip(bill_amount, service_quality):
    """
    Calculate the tip based on the bill amount and service quality.

    Args:
        bill_amount (float): The total bill amount.
        service_quality (str): The quality of service, which can be 'good', 'average', or 'bad'.

    Returns:
        float: The calculated tip amount.
    """
    if service_quality == 'good':
        tip_percentage = 0.2  # 20% tip for good service
    elif service_quality == 'average':
        tip_percentage = 0.15  # 15% tip for average service
    elif service_quality == 'bad':
        tip_percentage = 0.1  # 10% tip for bad service
    else:
        raise ValueError("Invalid service quality. Please choose from 'good', 'average', or 'bad'.")

    tip_amount = bill_amount * tip_percentage
    return tip_amount


def main():
    bill_amount = float(input("Enter the bill amount: $"))
    print("Choose the service quality:")
    print("1. Good")
    print("2. Average")
    print("3. Bad")
    choice = int(input("Enter your choice (1/2/3): "))

    if choice == 1:
        service_quality = 'good'
    elif choice == 2:
        service_quality = 'average'
    elif choice == 3:
        service_quality = 'bad'
    else:
        print("Invalid choice. Please choose from 1, 2, or 3.")
        return

    tip_amount = calculate_tip(bill_amount, service_quality)
    print(f"Tip amount: ${tip_amount:.2f}")
    print(f"Total amount: ${bill_amount + tip_amount:.2f}")


if __name__ == "__main__":
    main()
```
