## Exercise
Use the Sieve of Eratosthenes algorithm to find all prime numbers up to a given limit

## Reference Solution

def sieve_of_eratosthenes(limit):
    """
    Finds all prime numbers up to a given limit using the Sieve of Eratosthenes algorithm.

    Args:
        limit (int): The upper limit for finding prime numbers.

    Returns:
        list: A list of all prime numbers up to the given limit.
    """
    # Create a boolean array, prime, of size limit+1
    prime = [True] * (limit + 1)
    prime[0] = prime[1] = False

    # Iterate from 2 to sqrt(limit)
    for p in range(2, int(limit ** 0.5) + 1):
        # If p is prime, mark as composite all the multiples of p
        if prime[p]:
            for i in range(p * p, limit + 1, p):
                prime[i] = False

    # Return a list of all prime numbers in the range
    return [p for p in range(2, limit + 1) if prime[p]]

# Example usage
limit = 100
primes = sieve_of_eratosthenes(limit)
print("Prime numbers up to", limit, ":", primes)
