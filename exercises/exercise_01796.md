Write a function to find the symmetric difference of two dictionaries and return a new dictionary containing only the key-value pairs that exist in either dictionary but not in both.

def symmetric_difference(dict1, dict2):
    """
    Returns a new dictionary containing only the key-value pairs that exist in either dictionary but not in both.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A new dictionary containing the symmetric difference of dict1 and dict2.
    """
    # Create a new dictionary to store the symmetric difference
    symmetric_diff = {}

    # Iterate over the key-value pairs in dict1
    for key, value in dict1.items():
        # If the key is not in dict2, add it to the symmetric difference
        if key not in dict2:
            symmetric_diff[key] = value

    # Iterate over the key-value pairs in dict2
    for key, value in dict2.items():
        # If the key is not in dict1, add it to the symmetric difference
        if key not in dict1:
            symmetric_diff[key] = value

    # Return the symmetric difference
    return symmetric_diff

def symmetric_difference(dict1, dict2):
