## Exercise
Create a program that uses functions to perform basic arithmetic operations such as addition, subtraction, multiplication, and division. The program should take two numbers and an operation as input, and then use a function to calculate and return the result.

## Reference Solution
```python
# Define a function for addition
def add(num1, num2):
    """Return the sum of two numbers."""
    return num1 + num2

# Define a function for subtraction
def subtract(num1, num2):
    """Return the difference of two numbers."""
    return num1 - num2

# Define a function for multiplication
def multiply(num1, num2):
    """Return the product of two numbers."""
    return num1 * num2

# Define a function for division
def divide(num1, num2):
    """
    Return the quotient of two numbers.

    Raises:
        ZeroDivisionError: If the second number is zero.
    """
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero.")
    return num1 / num2

# Define a function to get user input for the operation
def get_operation():
    """Return the operation chosen by the user."""
    operations = {
        "1": "Addition",
        "2": "Subtraction",
        "3": "Multiplication",
        "4": "Division"
    }
    print("Choose an operation:")
    for key, value in operations.items():
        print(f"{key}: {value}")
    while True:
        choice = input("Enter the number of your chosen operation: ")
        if choice in operations:
            return choice
        else:
            print("Invalid choice. Please try again.")

# Define a function to get user input for numbers
def get_numbers():
    """Return two numbers entered by the user."""
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            num2 = float(input("Enter the second number: "))
            return num1, num2
        except ValueError:
            print("Invalid input. Please enter a number.")

# Define a function to perform the chosen operation
def perform_operation(num1, num2, operation):
    """Return the result of the chosen operation."""
    operations = {
        "1": add,
        "2": subtract,
        "3": multiply,
        "4": divide
    }
    if operation == "4" and num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero.")
    return operations[operation](num1, num2)

# Define the main function
def main():
    num1, num2 = get_numbers()
    operation = get_operation()
    try:
        result = perform_operation(num1, num2, operation)
        print(f"The result is: {result}")
    except ZeroDivisionError as e:
        print(e)

# Call the main function
if __name__ == "__main__":
    main()
```
