## Exercise
Create a function that finds all combinations of numbers that sum up to a target value.

## Reference Solution

def combination_sum(candidates, target):
    """
    Finds all combinations of numbers that sum up to the target value.

    Args:
        candidates (list): A list of numbers.
        target (int): The target value.

    Returns:
        list: A list of lists, where each sublist is a combination of numbers that sum up to the target value.
    """
    def backtrack(remain, comb, start):
        if remain == 0:
            result.append(list(comb))
            return
        elif remain < 0:
            return
        for i in range(start, len(candidates)):
            comb.append(candidates[i])
            backtrack(remain - candidates[i], comb, i)
            comb.pop()

    result = []
    backtrack(target, [], 0)
    return result

# Example usage:
candidates = [2, 3, 5]
target = 8
result = combination_sum(candidates, target)
print(result)
