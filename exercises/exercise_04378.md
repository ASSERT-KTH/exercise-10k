## Exercise
Write a program that aggregates the values associated with a given key in multiple dictionaries.

## Reference Solution
```python
def aggregate_values(dicts, key):
    """
    Aggregates the values associated with a given key in multiple dictionaries.

    Args:
        dicts (list): A list of dictionaries.
        key (str): The key to aggregate values for.

    Returns:
        list: A list of values associated with the given key in the dictionaries.
    """
    aggregated_values = [d.get(key) for d in dicts if key in d]
    return aggregated_values

# Example usage:
dicts = [
    {"a": 1, "b": 2},
    {"a": 3, "c": 4},
    {"a": 5, "b": 6},
    {"c": 7}
]

key = "a"
print(aggregate_values(dicts, key))  # Output: [1, 3, 5]

key = "b"
print(aggregate_values(dicts, key))  # Output: [2, 6]

key = "c"
print(aggregate_values(dicts, key))  # Output: [4, 7]
```

```python
from collections import defaultdict

def aggregate_values(dicts, key):
    aggregated_values = defaultdict(list)
    for d in dicts:
        if key in d:
            aggregated_values[key].append(d[key])
    return aggregated_values.get(key, [])

# Example usage:
dicts = [
    {"a": 1, "b": 2},
    {"a": 3, "c": 4},
    {"a": 5, "b": 6},
    {"c": 7}
```
