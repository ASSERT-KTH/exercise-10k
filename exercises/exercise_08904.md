## Exercise
Write a program to implement binary search on a sorted list.

## Reference Solution

def binary_search(sorted_list, target):
    """
    Searches for the target element in a sorted list using binary search.

    Args:
    - sorted_list (list): A sorted list of elements.
    - target: The target element to search for.

    Returns:
    - The index of the target element if found, -1 otherwise.
    """
    low = 0
    high = len(sorted_list) - 1

    while low <= high:
        mid = (low + high) // 2
        if sorted_list[mid] == target:
            return mid
        elif sorted_list[mid] < target:
            low = mid + 1
        else:
            high = mid - 1

    return -1


# Example usage
if __name__ == "__main__":
    sorted_list = [2, 5, 8, 12, 16, 23, 38, 56, 72, 91]
    target = 23

    result = binary_search(sorted_list, target)

    if result != -1:
        print(f"Target {target} found at index {result}.")
    else:
        print(f"Target {target} not found in the list.")
