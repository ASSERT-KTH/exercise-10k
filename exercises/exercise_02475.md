Create a function that compares two given strings lexicographically.

```python
def compare_strings_lexicographically(str1, str2):
    """
    This function compares two given strings lexicographically.
    
    Args:
        str1 (str): The first string to be compared.
        str2 (str): The second string to be compared.
    
    Returns:
        int: A negative integer if str1 is lexicographically smaller than str2,
             a positive integer if str1 is lexicographically larger than str2,
             and zero if both strings are equal.
    """

    # We use the built-in comparison operators in Python to compare the strings
    # These operators compare strings lexicographically by default
    
    # If str1 is lexicographically smaller than str2, return -1
    if str1 < str2:
        return -1
    
    # If str1 is lexicographically larger than str2, return 1
    elif str1 > str2:
        return 1
    
    # If both strings are equal, return 0
    else:
        return 0

# Example usage:
print(compare_strings_lexicographically("apple", "banana"))  # Output: -1
print(compare_strings_lexicographically("banana", "apple"))  # Output: 1
print(compare_strings_lexicographically("apple", "apple"))   # Output: 0
```
