## Exercise
Design a program to perform integer division (e.g., 17 divided by 5 is 3) and calculate the remainder.

## Reference Solution
```python
#include <iostream>
using namespace std;

// Function to perform integer division and calculate remainder
void divideAndCalculateRemainder(int dividend, int divisor) {
    if (divisor == 0) {
        cout << "Error: Division by zero is not allowed." << endl;
        return;
    }

    int quotient = dividend / divisor;
    int remainder = dividend % divisor;

    cout << "Division: " << dividend << " / " << divisor << " = " << quotient << endl;
    cout << "Remainder: " << remainder << endl;
}

int main() {
    int dividend, divisor;

    cout << "Enter the dividend: ";
    cin >> dividend;

    cout << "Enter the divisor: ";
    cin >> divisor;

    divideAndCalculateRemainder(dividend, divisor);

    return 0;
}
```
