## Exercise
Write a program that asks for two strings and checks if the first string is a subsequence of the second string.

## Reference Solution
```python
def is_subsequence(str1, str2):
    """
    Checks if str1 is a subsequence of str2.

    Args:
        str1 (str): The potential subsequence string.
        str2 (str): The parent string.

    Returns:
        bool: True if str1 is a subsequence of str2, False otherwise.
    """
    i = 0  # Index for str1
    j = 0  # Index for str2

    while i < len(str1) and j < len(str2):
        if str1[i] == str2[j]:
            i += 1
        j += 1

    return i == len(str1)


def main():
    # Get input strings from user
    str1 = input("Enter the first string: ")
    str2 = input("Enter the second string: ")

    # Check if str1 is a subsequence of str2
    if is_subsequence(str1, str2):
        print(f"'{str1}' is a subsequence of '{str2}'.")
    else:
        print(f"'{str1}' is not a subsequence of '{str2}'.")


if __name__ == "__main__":
    main()
```
