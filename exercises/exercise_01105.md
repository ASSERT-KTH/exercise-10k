## Exercise
Write a program to subtract two fractions and display the result

def subtract_fractions(numerator1, denominator1, numerator2, denominator2):
    """
    Subtracts two fractions and returns the result.

    Args:
        numerator1 (int): The numerator of the first fraction.
        denominator1 (int): The denominator of the first fraction.
        numerator2 (int): The numerator of the second fraction.
        denominator2 (int): The denominator of the second fraction.

    Returns:
        tuple: A tuple containing the numerator and denominator of the result.
    """
    # Calculate the least common multiple (LCM) of the denominators
    def lcm(a, b):
        from math import gcd
        return a * b // gcd(a, b)

    lcm_value = lcm(denominator1, denominator2)

    # Convert the fractions to have the LCM as the denominator
    numerator1_converted = numerator1 * (lcm_value // denominator1)
    numerator2_converted = numerator2 * (lcm_value // denominator2)

    # Subtract the fractions
    result_numerator = numerator1_converted - numerator2_converted
    result_denominator = lcm_value

    # Simplify the result
    def gcd(a, b):
        while b:
            a, b = b, a % b
        return a

    gcd_value = gcd(result_numerator, result_denominator)
    result_numerator //= gcd_value
    result_denominator //= gcd_value

    return result_numerator, result_denominator


def main():
    # Get the input from the user
    numerator1 = int(input("Enter the numerator of the first fraction: "))
    denominator1 = int(input("Enter the denominator of the first fraction: "))
    numerator2 = int(input("Enter the numerator of the second fraction: "))
    denominator2 = int(input("Enter the denominator of the second fraction: "))

    # Subtract the fractions
    result_numerator, result_denominator = subtract_fractions(numerator1, denominator1, numerator2, denominator2)

    # Display the result
    print(f"The result of subtracting {numerator2}/{denominator2} from {numerator1}/{denominator1} is {result_numerator}/{result_denominator}")


if __name__ == "__main__":
    main()
