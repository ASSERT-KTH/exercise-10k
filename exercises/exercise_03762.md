## Exercise
Write a program to access the value of a specific key in a dictionary.

## Reference Solution
```python
def access_dict_key(dictionary, key):
    """
    Access the value of a specific key in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key (any): The key to access.

    Returns:
        any: The value associated with the key if it exists, otherwise None.
    """
    try:
        # Attempt to access the key value
        return dictionary[key]
    except KeyError:
        # Handle the case where the key does not exist
        print(f"The key '{key}' does not exist in the dictionary.")
        return None

# Example usage
if __name__ == "__main__":
    # Define a sample dictionary
    sample_dict = {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }

    # Access a key value
    key_to_access = "name"
    value = access_dict_key(sample_dict, key_to_access)

    if value is not None:
        print(f"The value of the key '{key_to_access}' is: {value}")

    # Attempt to access a non-existent key
    non_existent_key = "country"
    access_dict_key(sample_dict, non_existent_key)
```
