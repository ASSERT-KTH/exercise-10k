## Exercise
Write a program that performs basic dictionary operations such as key-value pair access and modification.

## Reference Solution
```python
# Define a dictionary
my_dict = {"name": "John", "age": 30, "city": "New York"}

# Accessing a key-value pair
def access_key_value_pair(dictionary, key):
    """
    Access a key-value pair in the dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key (str): The key to access.

    Returns:
        The value associated with the key if it exists, otherwise None.
    """
    return dictionary.get(key)

# Modifying a key-value pair
def modify_key_value_pair(dictionary, key, value):
    """
    Modify a key-value pair in the dictionary.

    Args:
        dictionary (dict): The dictionary to modify.
        key (str): The key to modify.
        value (any): The new value to associate with the key.
    """
    dictionary[key] = value

# Adding a new key-value pair
def add_key_value_pair(dictionary, key, value):
    """
    Add a new key-value pair to the dictionary.

    Args:
        dictionary (dict): The dictionary to modify.
        key (str): The new key to add.
        value (any): The value to associate with the new key.
    """
    dictionary[key] = value

# Removing a key-value pair
def remove_key_value_pair(dictionary, key):
    """
    Remove a key-value pair from the dictionary.

    Args:
        dictionary (dict): The dictionary to modify.
        key (str): The key to remove.
    """
    if key in dictionary:
        del dictionary[key]

# Main function
def main():
    print("Original dictionary:", my_dict)

    # Access a key-value pair
    print("Accessing 'name' key:", access_key_value_pair(my_dict, "name"))

    # Modify a key-value pair
    modify_key_value_pair(my_dict, "age", 31)
    print("Modified dictionary:", my_dict)

    # Add a new key-value pair
    add_key_value_pair(my_dict, "country", "USA")
    print("Dictionary after adding a new key-value pair:", my_dict)

    # Remove a key-value pair
    remove_key_value_pair(my_dict, "city")
    print("Dictionary after removing a key-value pair:", my_dict)

if __name__ == "__main__":
    main()
```
