## Exercise
Create a function that takes in two numbers and an operator, then performs the specified operation and returns the result.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    Perform the specified arithmetic operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The arithmetic operator. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not one of '+', '-', '*', or '/'.
        ZeroDivisionError: If the operator is '/' and num2 is 0.
    """
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return num1 / num2
    else:
        raise ValueError("Invalid operator. Must be one of '+', '-', '*', or '/'")

# Example usage
if __name__ == "__main__":
    print(calculate(10, 2, '+'))  # Output: 12
    print(calculate(10, 2, '-'))  # Output: 8
    print(calculate(10, 2, '*'))  # Output: 20
    print(calculate(10, 2, '/'))  # Output: 5.0
    try:
        print(calculate(10, 0, '/'))  # Raises ZeroDivisionError
    except ZeroDivisionError as e:
        print(e)
    try:
        print(calculate(10, 2, '%'))  # Raises ValueError
    except ValueError as e:
        print(e)
```

```python
def calculate(num1, num2, operator):
    """
    Perform the specified arithmetic operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The arithmetic operator. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not one of '+', '-', '*', or '/'.
        ZeroDivisionError: If the operator is '/' and num2 is 0.
    """
    operations = {
        '+': lambda x, y: x + y,
        '-': lambda x, y: x - y,
        '*': lambda x, y: x * y,
        '/': lambda x, y: x / y if y != 0 else float('inf')
    }

    if operator in operations:
        if operator == '/' and num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        result = operations[operator](num1, num2)
        if result == float('inf'):
            raise ZeroDivisionError("Cannot divide by zero")
        return result
    else:
        raise ValueError("Invalid operator. Must be one of '+', '-', '*', or '/'")

# Example usage
if __name__ == "__main__":
    print(calculate(10, 2, '+'))  # Output: 12
    print(calculate(10, 2, '-'))  # Output: 8
    print(calculate(10, 2, '*'))  # Output: 20
    print(calculate(10, 2, '/'))  # Output: 5.0
    try:
        print(calculate(10, 0, '/'))  # Raises ZeroDivisionError
    except ZeroDivisionError as e:
        print(e)
    try:
        print(calculate(10, 2, '%'))  # Raises ValueError
    except ValueError as e:
        print(e)
```
