## Exercise
Create a dictionary with names of animals as keys and their habitats as values. Calculate and print the number of key-value pairs in the dictionary.

## Reference Solution
```python
# Define the dictionary
animal_habitats = {
    "Lion": "Savannah",
    "Penguin": "Antarctica",
    "Koala": "Eucalyptus Forest",
    "Giraffe": "Savannah",
    "Octopus": "Coral Reef",
    "Polar Bear": "Arctic",
    "Kangaroo": "Grassland",
    "Butterfly": "Tropical Forest"
}

# Calculate and print the number of key-value pairs in the dictionary
def print_dict_length(dictionary):
    """Print the number of key-value pairs in a dictionary."""
    length = len(dictionary)
    print(f"The dictionary has {length} key-value pairs.")

# Main function with example usage
def main():
    print("Animal Habitats Dictionary:")
    print(animal_habitats)
    print_dict_length(animal_habitats)

if __name__ == "__main__":
    main()
```
