## Exercise
Write a program to inspect the attributes and methods of a variable, including its type, value, and memory location.

## Reference Solution
```python
# Import the necessary modules
import sys

def inspect_variable(var):
    """
    Inspect the attributes and methods of a variable.

    Args:
        var: The variable to be inspected.

    Returns:
        None
    """
    print(f"**Variable Type:** {type(var)}")
    print(f"**Variable Value:** {var}")
    print(f"**Variable Memory Location:** {id(var)}")
    print(f"**Variable Size in Bytes:** {sys.getsizeof(var)}")

    # Get the attributes and methods of the variable
    print("\n**Attributes and Methods:**")
    for attr in dir(var):
        if not attr.startswith('__') and not attr.endswith('__'):
            print(f"- {attr}")

# Example usage:
if __name__ == "__main__":
    var = "Hello, World!"
    inspect_variable(var)

    var = 123
    inspect_variable(var)

    var = [1, 2, 3, 4, 5]
    inspect_variable(var)

    var = {"a": 1, "b": 2}
    inspect_variable(var)
```
