## Exercise
Write a program that analyzes a given text and provides statistics such as word count and sentence count.

## Reference Solution
```python
import re

def analyze_text(text):
    """
    Analyzes the given text and returns word count and sentence count.
    
    Args:
        text (str): The text to be analyzed.
    
    Returns:
        dict: A dictionary containing word count and sentence count.
    """
    # Remove leading and trailing whitespace
    text = text.strip()
    
    # Split the text into words
    words = re.findall(r'\b\w+\b', text.lower())
    
    # Split the text into sentences
    sentences = re.split(r'[.!?]\s*', text)
    sentences = [sentence for sentence in sentences if sentence]  # Remove empty strings
    
    # Calculate word count and sentence count
    word_count = len(words)
    sentence_count = len(sentences)
    
    # Return the statistics as a dictionary
    return {
        'word_count': word_count,
        'sentence_count': sentence_count
    }

def main():
    # Example usage
    text = "Hello, world! This is a test. How many words and sentences are there?"
    stats = analyze_text(text)
    print("Word Count:", stats['word_count'])
    print("Sentence Count:", stats['sentence_count'])

if __name__ == "__main__":
    main()
```
