## Exercise
Write a program to demonstrate implicit type conversion, where the data type of a variable is automatically converted.

## Reference Solution
```python
def implicit_type_conversion_demo():
    """
    Demonstrates implicit type conversion in Python.
    """
    # Integer to Float
    int_val = 10
    float_val = 5.5
    result = int_val + float_val
    print(f"Integer to Float: {int_val} + {float_val} = {result} ({type(result).__name__})")

    # Integer to Complex
    complex_val = 3 + 2j
    result = int_val + complex_val
    print(f"Integer to Complex: {int_val} + {complex_val} = {result} ({type(result).__name__})")

    # Float to Complex
    result = float_val + complex_val
    print(f"Float to Complex: {float_val} + {complex_val} = {result} ({type(result).__name__})")

    # Boolean to Integer
    bool_val = True
    result = int_val + bool_val
    print(f"Boolean to Integer: {int_val} + {bool_val} = {result} ({type(result).__name__})")

if __name__ == "__main__":
    implicit_type_conversion_demo()
```
