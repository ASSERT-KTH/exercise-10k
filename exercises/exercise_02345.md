## Exercise
Write a program to take a story with missing words as input and ask the user to fill in the words, then print out the completed story

## Reference Solution
```python
def get_user_input(prompt):
    """Get user input for a missing word."""
    while True:
        user_input = input(prompt)
        if user_input:
            return user_input
        else:
            print("Please enter a word.")

def complete_story(story):
    """Complete the story by filling in the missing words."""
    completed_story = story
    missing_word_index = 0
    while True:
        start_index = completed_story.find("_____")
        if start_index == -1:
            break
        end_index = start_index + 6
        prompt = f"Enter a word for the {missing_word_index + 1} missing word: "
        user_input = get_user_input(prompt)
        completed_story = completed_story[:start_index] + user_input + completed_story[end_index:]
        missing_word_index += 1
    return completed_story

def main():
    story = "Once upon a time, there was a _____ who lived in a _____ house. The _____ had a _____ best friend who loved to play _____."
    print("Story with missing words:")
    print(story)
    completed_story = complete_story(story)
    print("\nCompleted story:")
    print(completed_story)

if __name__ == "__main__":
    main()
```
