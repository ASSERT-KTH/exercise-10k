## Exercise
Create a function that implements the quickselect algorithm to find the k-th smallest element in a list of numbers. The function should take a list and a value of k as arguments and return the k-th smallest element.

def quickselect(lst, k):
    """
    This function implements the quickselect algorithm to find the k-th smallest element in a list of numbers.

    Args:
        lst (list): A list of numbers.
        k (int): The position of the element to be found.

    Returns:
        int: The k-th smallest element in the list.

    Raises:
        ValueError: If k is less than 1 or greater than the length of the list.
    """

    # Check if k is within the valid range
    if k < 1 or k > len(lst):
        raise ValueError("k is out of range")

    # Call the helper function to start the quickselect process
    return quickselect_helper(lst, k, 0, len(lst) - 1)


def quickselect_helper(lst, k, left, right):
    # If the list contains only one element, return that element
    if left == right:
        return lst[left]

    # Select a pivot index
    pivot_index = partition(lst, left, right)

    # If the pivot index is equal to k-1, return the pivot element
    if k - 1 == pivot_index:
        return lst[pivot_index]

    # If the pivot index is greater than k-1, recurse on the left sublist
    elif k - 1 < pivot_index:
        return quickselect_helper(lst, k, left, pivot_index - 1)

    # If the pivot index is less than k-1, recurse on the right sublist
    else:
        return quickselect_helper(lst, k, pivot_index + 1, right)


def partition(lst, left, right):
    # Select the middle element as the pivot
    pivot = lst[(left + right) // 2]

    # Move all elements smaller than the pivot to the left of the pivot
    # and all elements greater than the pivot to the right of the pivot
    i = left - 1
    j = right + 1
    while True:
        i += 1
        while lst[i] < pivot:
            i += 1

        j -= 1
        while lst[j] > pivot:
            j -= 1

        if i >= j:
            return j

        # Swap the elements at indices i and j
        lst[i], lst[j] = lst[j], lst[i]
