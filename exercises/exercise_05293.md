## Exercise
Create a program that plays a game of Rock, Paper, Scissors with the user. Use a for loop to iterate over the possible moves.

## Reference Solution
```python
import random

def game():
    # List of possible moves
    moves = ['rock', 'paper', 'scissors']

    # Dictionary to determine the winning move
    winning_moves = {
        'rock': 'scissors',
        'paper': 'rock',
        'scissors': 'paper'
    }

    # Initialize score
    user_score = 0
    computer_score = 0

    # Play 5 rounds
    for i in range(5):
        # Get user move
        user_move = input("Enter your move (rock, paper, or scissors): ").lower()

        # Validate user move
        while user_move not in moves:
            user_move = input("Invalid move. Please enter rock, paper, or scissors: ").lower()

        # Get computer move
        computer_move = random.choice(moves)

        # Determine the winner
        if winning_moves[user_move] == computer_move:
            print(f"\nYou chose {user_move}, computer chose {computer_move}. You win this round!")
            user_score += 1
        elif winning_moves[computer_move] == user_move:
            print(f"\nYou chose {user_move}, computer chose {computer_move}. Computer wins this round!")
            computer_score += 1
        else:
            print(f"\nYou chose {user_move}, computer chose {computer_move}. It's a tie!")

        # Print current score
        print(f"Score - You: {user_score}, Computer: {computer_score}\n")

    # Determine the overall winner
    if user_score > computer_score:
        print("Congratulations, you won the game!")
    elif user_score < computer_score:
        print("Sorry, the computer won the game.")
    else:
        print("It's a tie game!")

if __name__ == "__main__":
    game()
```
