## Exercise
Write a program to slice a list and print the sliced elements.

## Reference Solution
```python
def slice_list(input_list, start, end):
    """
    Slice a list based on start and end indices.

    Args:
        input_list (list): The input list to be sliced.
        start (int): The start index of the slice.
        end (int): The end index of the slice.

    Returns:
        list: The sliced list.
    """
    try:
        # Slice the list using list slicing syntax
        sliced_list = input_list[start:end]
        return sliced_list
    except IndexError:
        print("Error: Invalid start or end index.")
        return None

def main():
    # Example usage
    input_list = [1, 2, 3, 4, 5, 6, 7, 8, 9]
    start = 2
    end = 6

    print("Input List:", input_list)
    print("Start Index:", start)
    print("End Index:", end)

    sliced_list = slice_list(input_list, start, end)

    if sliced_list is not None:
        print("Sliced List:", sliced_list)

if __name__ == "__main__":
    main()
```
