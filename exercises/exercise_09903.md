## Exercise
Write a program that calculates the minimum number of coins needed to make change for a given amount.

## Reference Solution
```python
def min_coins(coins, amount):
    """
    Calculate the minimum number of coins needed to make change for a given amount.

    Args:
        coins (list): List of coin denominations.
        amount (int): Amount to make change for.

    Returns:
        int: Minimum number of coins needed.
    """
    # Create a table to store the minimum number of coins needed for each amount
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # Base case: 0 coins needed for amount 0

    # Fill the table using dynamic programming
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] = min(dp[i], dp[i - coin] + 1)

    # If no solution is found, return -1
    return dp[amount] if dp[amount] != float('inf') else -1


# Example usage
if __name__ == "__main__":
    coins = [1, 5, 10, 25, 50]  # Available coin denominations
    amount = 37  # Amount to make change for

    min_coins_needed = min_coins(coins, amount)
    if min_coins_needed != -1:
        print(f"Minimum coins needed: {min_coins_needed}")
    else:
        print("No solution found")
```
