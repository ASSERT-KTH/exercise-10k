## Exercise
Write a program to calculate the distance between two points in a 2D space.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the Euclidean distance between two points in a 2D space.

    Args:
        point1 (tuple): The coordinates of the first point (x1, y1).
        point2 (tuple): The coordinates of the second point (x2, y2).

    Returns:
        float: The Euclidean distance between the two points.
    """
    x1, y1 = point1
    x2, y2 = point2
    return math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)


def main():
    # Example usage
    point1 = (1, 2)  # coordinates of the first point
    point2 = (4, 6)  # coordinates of the second point

    distance = calculate_distance(point1, point2)
    print(f"The distance between {point1} and {point2} is: {distance:.2f}")


if __name__ == "__main__":
    main()
```
