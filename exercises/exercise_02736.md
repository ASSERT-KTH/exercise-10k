## Exercise
Write a program to find the first missing positive integer in an array of integers

## Reference Solution
```python
def first_missing_positive(nums):
    """
    Find the first missing positive integer in an array of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The first missing positive integer.
    """
    # Remove non-positive integers and duplicates
    num_set = set(num for num in nums if num > 0)

    # Initialize the first missing positive integer to 1
    missing = 1

    # Iterate through the positive integers
    while missing in num_set:
        # If the current integer is in the set, increment it
        missing += 1

    return missing


# Example usage
def main():
    # Test case 1: [1, 2, 0]
    print(first_missing_positive([1, 2, 0]))  # Output: 3

    # Test case 2: [3, 4, -1, 1]
    print(first_missing_positive([3, 4, -1, 1]))  # Output: 2

    # Test case 3: [7, 8, 9, 11, 12]
    print(first_missing_positive([7, 8, 9, 11, 12]))  # Output: 1


if __name__ == "__main__":
    main()
```
