## Exercise
Write a function to check if a string consists only of letters.

## Reference Solution
```python
def check_letters_only(input_string: str) -> bool:
    """
    Checks if a string consists only of letters.

    Args:
        input_string (str): The input string to be checked.

    Returns:
        bool: True if the string consists only of letters, False otherwise.
    """
    return input_string.isalpha()

# Example usage
def main():
    print(check_letters_only("HelloWorld"))  # Returns: True
    print(check_letters_only("Hello World"))  # Returns: False
    print(check_letters_only("Hello123"))  # Returns: False

if __name__ == "__main__":
    main()
```
