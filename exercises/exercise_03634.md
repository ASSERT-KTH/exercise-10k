## Exercise
Write a program to find the first occurrence of a given character in a string.

## Reference Solution
```python
def find_first_occurrence(string, char):
    """
    Find the first occurrence of a given character in a string.

    Args:
        string (str): The input string.
        char (str): The character to find.

    Returns:
        int: The index of the first occurrence of the character. Returns -1 if the character is not found.
    """
    try:
        return string.index(char)
    except ValueError:
        return -1

def main():
    string = input("Enter a string: ")
    char = input("Enter a character: ")

    index = find_first_occurrence(string, char)

    if index != -1:
        print(f"The first occurrence of '{char}' is at index {index}.")
    else:
        print(f"The character '{char}' is not found in the string.")

if __name__ == "__main__":
    main()
```
