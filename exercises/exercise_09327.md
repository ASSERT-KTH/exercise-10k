## Exercise
Write a program to declare and use char variables.

## Reference Solution
```python
#include <stdio.h>

int main() {
    // Declare char variables
    char initial = 'A';
    char grade = 'B';

    // Print the values of char variables
    printf("The initial is: %c\n", initial);
    printf("The grade is: %c\n", grade);

    // Use char variables in a conditional statement
    if (grade == 'A') {
        printf("Grade A\n");
    } else if (grade == 'B') {
        printf("Grade B\n");
    } else {
        printf("Other grade\n");
    }

    // Get user input and store it in a char variable
    char response;
    printf("Do you like the program? (y/n): ");
    scanf(" %c", &response);

    // Use the char variable in a conditional statement
    if (response == 'y' || response == 'Y') {
        printf("Thank you!\n");
    } else if (response == 'n' || response == 'N') {
        printf("Sorry to hear that.\n");
    } else {
        printf("Invalid response.\n");
    }

    return 0;
}
```
