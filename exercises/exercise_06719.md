## Exercise
Write a program that prints out a complex console-based user interface

## Reference Solution
```python
import os
import time

def clear_screen():
    """Clears the console screen."""
    os.system('cls' if os.name == 'nt' else 'clear')

def print_header():
    """Prints the header of the user interface."""
    print("+" + "-" * 78 + "+")
    print("|" + " " * 78 + "|")
    print("|" + "Console-Based User Interface".center(78) + "|")
    print("|" + " " * 78 + "|")
    print("+" + "-" * 78 + "+")

def print_menu():
    """Prints the menu of the user interface."""
    print("+" + "-" * 78 + "+")
    print("|" + "Menu".ljust(39) + "|" + "Description".rjust(39) + "|")
    print("+" + "-" * 78 + "+")
    print("|" + "1. View Profile".ljust(39) + "|" + "View your profile information.".rjust(39) + "|")
    print("|" + "2. Edit Profile".ljust(39) + "|" + "Edit your profile information.".rjust(39) + "|")
    print("|" + "3. View Settings".ljust(39) + "|" + "View your application settings.".rjust(39) + "|")
    print("|" + "4. Edit Settings".ljust(39) + "|" + "Edit your application settings.".rjust(39) + "|")
    print("|" + "5. Exit".ljust(39) + "|" + "Exit the application.".rjust(39) + "|")
    print("+" + "-" * 78 + "+")

def print_footer():
    """Prints the footer of the user interface."""
    print("+" + "-" * 78 + "+")
    print("|" + " " * 78 + "|")
    print("|" + "Copyright 2024".center(78) + "|")
    print("|" + " " * 78 + "|")
    print("+" + "-" * 78 + "+")

def main():
    while True:
        clear_screen()
        print_header()
        print_menu()
        print_footer()
        choice = input("Enter your choice: ")
        if choice == "1":
            print("View Profile")
            time.sleep(2)
        elif choice == "2":
            print("Edit Profile")
            time.sleep(2)
        elif choice == "3":
            print("View Settings")
            time.sleep(2)
        elif choice == "4":
            print("Edit Settings")
            time.sleep(2)
        elif choice == "5":
            print("Exiting the application.")
            time.sleep(2)
            break
        else:
            print("Invalid choice. Please try again.")
            time.sleep(2)

if __name__ == "__main__":
    main()
```
