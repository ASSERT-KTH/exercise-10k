## Exercise
Create a function that translates a given string into Pig Latin.

def translate_to_pig_latin(word):
    """
    Translates a given word into Pig Latin.
    
    Parameters:
    word (str): The word to be translated.
    
    Returns:
    str: The translated word in Pig Latin.
    """
    # Define vowels
    vowels = 'aeiou'
    
    # Check if the word starts with a vowel
    if word[0].lower() in vowels:
        # If it starts with a vowel, append 'way'
        return word + 'way'
    else:
        # If it starts with a consonant, move it to the end and append 'ay'
        for i, char in enumerate(word):
            if char.lower() in vowels:
                # Find the index of the first vowel
                return word[i:] + word[:i] + 'ay'
        # If the word contains no vowels, just append 'ay'
        return word + 'ay'

def translate_sentence_to_pig_latin(sentence):
    """
    Translates a given sentence into Pig Latin.
    
    Parameters:
    sentence (str): The sentence to be translated.
    
    Returns:
    str: The translated sentence in Pig Latin.
    """
    # Split the sentence into words
    words = sentence.split()
    
    # Translate each word into Pig Latin
    translated_words = [translate_to_pig_latin(word) for word in words]
    
    # Join the translated words back into a sentence
    return ' '.join(translated_words)

# Example usage
if __name__ == "__main__":
    word = "hello"
    sentence = "hello world"
    print(f"The Pig Latin translation of '{word}' is: {translate_to_pig_latin(word)}")
    print(f"The Pig Latin translation of '{sentence}' is: {translate_sentence_to_pig_latin(sentence)}")
