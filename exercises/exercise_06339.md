## Exercise
Write a program that takes two arrays as input, finds their intersection, and returns the intersection array.

## Reference Solution
```python
#include <iostream>

int main() {
    int n, m;
    std::cin >> n >> m;
    int arr1[n], arr2[m];
    for (int i = 0; i < n; i++) {
        std::cin >> arr1[i];
    }
    for (int i = 0; i < m; i++) {
        std::cin >> arr2[i];
    }
    int l = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (arr1[i] == arr2[j]) {
                std::cout << arr1[i] << " ";
                l++;
                break;
            }
        }
    }
    if (l == 0) {
        std::cout << -1;
    }
    return 0;
}
```
