## Exercise
Create a program that uses functions to perform basic arithmetic operations. The program should have separate functions for addition, subtraction, multiplication, and division. Each function should take two numbers as arguments and return the result of the operation.

## Reference Solution
```python
# Define a function for addition
def add(num1, num2):
    """
    This function adds two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.

    Returns:
        float: The sum of num1 and num2.
    """
    return num1 + num2

# Define a function for subtraction
def subtract(num1, num2):
    """
    This function subtracts num2 from num1.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.

    Returns:
        float: The difference of num1 and num2.
    """
    return num1 - num2

# Define a function for multiplication
def multiply(num1, num2):
    """
    This function multiplies two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.

    Returns:
        float: The product of num1 and num2.
    """
    return num1 * num2

# Define a function for division
def divide(num1, num2):
    """
    This function divides num1 by num2.

    Args:
        num1 (float): The dividend.
        num2 (float): The divisor.

    Returns:
        float: The quotient of num1 and num2.

    Raises:
        ZeroDivisionError: If num2 is zero.
    """
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return num1 / num2

# Define a main function to test the arithmetic functions
def main():
    print("Basic Arithmetic Operations")
    print("---------------------------")

    # Test the addition function
    num1 = 10
    num2 = 5
    result = add(num1, num2)
    print(f"{num1} + {num2} = {result}")

    # Test the subtraction function
    num1 = 10
    num2 = 5
    result = subtract(num1, num2)
    print(f"{num1} - {num2} = {result}")

    # Test the multiplication function
    num1 = 10
    num2 = 5
    result = multiply(num1, num2)
    print(f"{num1} * {num2} = {result}")

    # Test the division function
    num1 = 10
    num2 = 5
    result = divide(num1, num2)
    print(f"{num1} / {num2} = {result}")

    # Test division by zero
    try:
        num1 = 10
        num2 = 0
        result = divide(num1, num2)
        print(f"{num1} / {num2} = {result}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
