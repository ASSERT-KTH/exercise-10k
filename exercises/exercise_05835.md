## Exercise
Write a function to parse a given URL and extract its components

## Reference Solution
```python
from urllib.parse import urlparse

def parse_url(url):
    """
    Parse a given URL and extract its components.

    Args:
        url (str): The URL to be parsed.

    Returns:
        dict: A dictionary containing the URL's components.
    """
    parsed_url = urlparse(url)
    url_components = {
        "scheme": parsed_url.scheme,
        "netloc": parsed_url.netloc,
        "path": parsed_url.path,
        "params": parsed_url.params,
        "query": parsed_url.query,
        "fragment": parsed_url.fragment
    }
    return url_components

# Example usage
if __name__ == "__main__":
    url = "https://www.example.com/path/to/resource?param1=value1&param2=value2#anchor"
    components = parse_url(url)
    for component, value in components.items():
        print(f"{component}: {value}")
```
