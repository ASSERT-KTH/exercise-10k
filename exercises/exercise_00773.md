## Exercise
Write a program that uses string indexing to extract a substring from a given string.

## Reference Solution
```python
def extract_substring(input_string, start_index, end_index):
    """
    Extracts a substring from the input string using string indexing.

    Args:
        input_string (str): The input string from which to extract the substring.
        start_index (int): The starting index of the substring.
        end_index (int): The ending index of the substring.

    Returns:
        str: The extracted substring.
    """
    # Check if the start index is less than the end index
    if start_index > end_index:
        raise ValueError("Start index cannot be greater than end index")

    # Check if the start and end indices are within the bounds of the string
    if start_index < 0 or end_index > len(input_string):
        raise IndexError("Indices out of range")

    # Extract the substring using string indexing
    substring = input_string[start_index:end_index]

    return substring


def main():
    # Example usage
    input_string = "Hello, World!"
    start_index = 7
    end_index = 12

    try:
        substring = extract_substring(input_string, start_index, end_index)
        print(f"Extracted substring: {substring}")
    except (ValueError, IndexError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
