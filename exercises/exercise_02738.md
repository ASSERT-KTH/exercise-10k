## Exercise
Write a program to find the maximum depth of a binary tree

## Reference Solution
```python
class Node:
    """Represents a node in a binary tree."""
    def __init__(self, value, left=None, right=None):
        self.value = value
        self.left = left
        self.right = right

class BinaryTree:
    """Represents a binary tree."""
    def __init__(self, root=None):
        self.root = root

    def max_depth(self, node):
        """
        Calculates the maximum depth of the binary tree.

        Args:
        node (Node): The current node.

        Returns:
        int: The maximum depth of the binary tree.
        """
        if node is None:
            return 0
        else:
            left_depth = self.max_depth(node.left)
            right_depth = self.max_depth(node.right)
            return max(left_depth, right_depth) + 1

    def find_max_depth(self):
        """
        Finds the maximum depth of the binary tree.

        Returns:
        int: The maximum depth of the binary tree.
        """
        return self.max_depth(self.root)


# Example usage:
if __name__ == "__main__":
    # Create nodes
    node1 = Node(1)
    node2 = Node(2)
    node3 = Node(3)
    node4 = Node(4)
    node5 = Node(5)
    node6 = Node(6)
    node7 = Node(7)

    # Create binary tree
    node1.left = node2
    node1.right = node3
    node2.left = node4
    node2.right = node5
    node3.left = node6
    node3.right = node7

    # Create binary tree object
    tree = BinaryTree(node1)

    # Find maximum depth
    max_depth = tree.find_max_depth()

    print("Maximum depth:", max_depth)
```
