## Exercise
Write a program that takes a string as input and returns the number of consonants in the string.

## Reference Solution
```python
def count_consonants(input_string):
    """
    Counts the number of consonants in a given string.

    Args:
        input_string (str): The input string to count consonants from.

    Returns:
        int: The number of consonants in the input string.
    """
    # Define a set of vowels for efficient lookups
    vowels = set('aeiouAEIOU')

    # Initialize a counter for consonants
    consonant_count = 0

    # Iterate over each character in the input string
    for char in input_string:
        # Check if the character is an alphabet letter and not a vowel
        if char.isalpha() and char not in vowels:
            # Increment the consonant counter
            consonant_count += 1

    # Return the total consonant count
    return consonant_count


# Example usage
def main():
    input_string = input("Enter a string: ")
    consonant_count = count_consonants(input_string)
    print(f"Number of consonants: {consonant_count}")


if __name__ == "__main__":
    main()
```
