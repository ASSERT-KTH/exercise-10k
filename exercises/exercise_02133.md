## Exercise
Create a function that takes a 2D array as input and returns a new 1D array containing all elements from the 2D array.

## Reference Solution

def flatten_2d_array(arr):
    """
    This function takes a 2D array as input and returns a new 1D array containing all elements from the 2D array.

    Args:
        arr (list): A 2D list of elements.

    Returns:
        list: A 1D list of elements.
    """

    # Initialize an empty list to store the result
    result = []

    # Iterate over each row in the 2D array
    for row in arr:
        # For each row, extend the result list with its elements
        # This effectively flattens the 2D array into a 1D array
        result.extend(row)

    # Return the resulting 1D array
    return result

# Alternatively, you can use list comprehension to achieve the same result in a more concise way
def flatten_2d_array_concise(arr):
    """

    Args:
        arr (list): A 2D list of elements.

    Returns:
        list: A 1D list of elements.
    """

    # Use list comprehension to iterate over each row in the 2D array and each element in the row
    # This will create a new 1D list containing all elements from the 2D array
    return [element for row in arr for element in row]
