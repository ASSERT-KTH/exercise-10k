## Exercise
Create a program that implements a to-do list app using a for loop.

## Reference Solution
```python
# to_do_list.py

class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        """Add a task to the to-do list."""
        self.tasks.append(task)
        print(f"Task '{task}' added successfully.")

    def view_tasks(self):
        """View all tasks in the to-do list."""
        if not self.tasks:
            print("No tasks available.")
            return

        print("Available Tasks:")
        for index, task in enumerate(self.tasks, start=1):
            print(f"{index}. {task}")

    def delete_task(self, task_number):
        """Delete a task from the to-do list."""
        try:
            task_number = int(task_number)
            if task_number <= 0:
                raise ValueError
            task = self.tasks.pop(task_number - 1)
            print(f"Task '{task}' deleted successfully.")
        except (IndexError, ValueError):
            print("Invalid task number.")

    def mark_completed(self, task_number):
        """Mark a task as completed."""
        try:
            task_number = int(task_number)
            if task_number <= 0:
                raise ValueError
            task = self.tasks[task_number - 1]
            self.tasks[task_number - 1] = f"[Completed] {task}"
            print(f"Task '{task}' marked as completed.")
        except (IndexError, ValueError):
            print("Invalid task number.")


def main():
    to_do_list = ToDoList()

    while True:
        print("\nTo-Do List App")
        print("1. Add Task")
        print("2. View Tasks")
        print("3. Delete Task")
        print("4. Mark Task as Completed")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            task = input("Enter a task: ")
            to_do_list.add_task(task)
        elif choice == "2":
            to_do_list.view_tasks()
        elif choice == "3":
            to_do_list.view_tasks()
            task_number = input("Enter the task number to delete: ")
            to_do_list.delete_task(task_number)
        elif choice == "4":
            to_do_list.view_tasks()
            task_number = input("Enter the task number to mark as completed: ")
            to_do_list.mark_completed(task_number)
        elif choice == "5":
            print("Exiting the app.")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
