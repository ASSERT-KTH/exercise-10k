## Exercise
Create a program that calculates the total bill given the subtotal and the tax percentage.

## Reference Solution
```python
def calculate_total_bill(subtotal, tax_percentage):
    """
    Calculates the total bill given the subtotal and the tax percentage.

    Args:
        subtotal (float): The subtotal of the bill.
        tax_percentage (float): The tax percentage.

    Returns:
        float: The total bill.
    """
    tax_amount = subtotal * (tax_percentage / 100)
    total_bill = subtotal + tax_amount
    return total_bill

def main():
    print("Tax Calculator Program")
    print("------------------------")

    # Get the subtotal from the user
    while True:
        try:
            subtotal = float(input("Enter the subtotal: $"))
            if subtotal < 0:
                print("Subtotal cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tax percentage from the user
    while True:
        try:
            tax_percentage = float(input("Enter the tax percentage (%): "))
            if tax_percentage < 0:
                print("Tax percentage cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the total bill
    total_bill = calculate_total_bill(subtotal, tax_percentage)

    # Print the result
    print(f"Subtotal: ${subtotal:.2f}")
    print(f"Tax ({tax_percentage}%): ${subtotal * (tax_percentage / 100):.2f}")
    print(f"Total Bill: ${total_bill:.2f}")

if __name__ == "__main__":
    main()
```
