## Exercise
Write a program that uses a for loop to check whether a given password meets certain criteria, such as length and character types.

## Reference Solution
```python
def check_password(password):
    # Initialize flags for password criteria
    has_uppercase = False
    has_lowercase = False
    has_digit = False
    has_special_char = False

    # Check password length
    if len(password) < 8:
        print("Password is too short. It should be at least 8 characters.")
        return False

    # Check each character in the password
    for char in password:
        if char.isupper():
            has_uppercase = True
        elif char.islower():
            has_lowercase = True
        elif char.isdigit():
            has_digit = True
        elif not char.isalnum():
            has_special_char = True

    # Check if password meets all criteria
    if not has_uppercase:
        print("Password should have at least one uppercase letter.")
    if not has_lowercase:
        print("Password should have at least one lowercase letter.")
    if not has_digit:
        print("Password should have at least one digit.")
    if not has_special_char:
        print("Password should have at least one special character.")

    # Return True if password meets all criteria
    if has_uppercase and has_lowercase and has_digit and has_special_char:
        print("Password is valid.")
        return True
    else:
        print("Password is invalid.")
        return False


# Example usage:
password = input("Enter a password: ")
check_password(password)
```
