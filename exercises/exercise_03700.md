## Exercise
Write a program to calculate the average speed given the total distance and time taken.

## Reference Solution
```python
def calculate_average_speed(distance, time):
    """
    Calculate the average speed given the total distance and time taken.

    Args:
        distance (float): Total distance traveled in kilometers.
        time (float): Time taken in hours.

    Returns:
        float: Average speed in kilometers per hour.
    """
    if time <= 0:
        raise ValueError("Time must be greater than zero.")
    average_speed = distance / time
    return average_speed


def main():
    distance = float(input("Enter the total distance traveled (in kilometers): "))
    time = float(input("Enter the time taken (in hours): "))

    try:
        average_speed = calculate_average_speed(distance, time)
        print(f"The average speed is {average_speed:.2f} km/h.")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
