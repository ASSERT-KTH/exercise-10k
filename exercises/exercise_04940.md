## Exercise
Write a program that calculates the variance of a list of numbers. The program should use the formula variance = sum((x - mean) ** 2) / (n - 1) to calculate the variance.

## Reference Solution
```python
# Define a function to calculate the mean of a list of numbers
def mean(list):
    s = 0
    for i in list:
        s += i
    return s / len(list)

# Define a function to calculate the variance of a list of numbers
def variance(list):
    mean_value = mean(list)
    s = 0
    for i in list:
        s += (i - mean_value) ** 2
    return s / (len(list) - 1)

# Define a function to get the list of numbers from the user
def get_list():
    while True:
        try:
            n = int(input("Enter the number of values: "))
            break
        except ValueError:
            print("Please enter a valid integer.")
    list = []
    for i in range(n):
        while True:
            try:
                num = float(input(f"Enter value {i+1}: "))
                list.append(num)
                break
            except ValueError:
                print("Please enter a valid number.")
    return list

# Main program
list = get_list()
print("The variance is:", variance(list))
```
