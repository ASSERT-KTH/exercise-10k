## Exercise
Write a program that takes a string and a list of characters as input, and returns the shortest substring that contains all the characters.

## Reference Solution

from collections import Counter

def shortest_substring(s, chars):
    """
    Returns the shortest substring of `s` that contains all characters in `chars`.

    Args:
        s (str): The input string.
        chars (list): A list of characters.

    Returns:
        str: The shortest substring that contains all characters in `chars`. If no such substring exists, returns an empty string.
    """
    if not s or not chars:
        return ""

    char_count = Counter(chars)
    required_chars = len(char_count)

    left = 0
    min_length = float('inf')
    min_substring = ""

    formed_chars = 0
    window_counts = {}

    for right in range(len(s)):
        character = s[right]
        window_counts[character] = window_counts.get(character, 0) + 1

        if character in char_count and window_counts[character] == char_count[character]:
            formed_chars += 1

        while left <= right and formed_chars == required_chars:
            character = s[left]

            if right - left + 1 < min_length:
                min_length = right - left + 1
                min_substring = s[left:right + 1]

            window_counts[character] -= 1
            if character in char_count and window_counts[character] < char_count[character]:
                formed_chars -= 1

            left += 1

    return min_substring

# Example usage:
s = "abcba"
chars = ['a', 'b', 'c']
print(shortest_substring(s, chars))  # Output: "abc"
