## Exercise
Write a program that takes input from the user with escapes and prints out the results.

## Reference Solution

def print_escape_sequences():
    """
    Take input from the user and print out the results, handling escape sequences.
    """
    user_input = input("Please enter a string with escape sequences: ")
    
    print("You entered: {}".format(user_input))
    print("Escape sequences will be interpreted as follows:")
    print("  \\n: newline")
    print("  \\t: tab")
    print("  \\\\": backslash")
    print("  \\': single quote")
    print("  \\\": double quote")

    # Print the interpreted string
    print("Interpreted string: {}".format(user_input.encode('utf-8').decode('unicode_escape')))

if __name__ == "__main__":
    print_escape_sequences()
