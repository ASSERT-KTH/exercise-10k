## Exercise
Perform basic dictionary operations (add, remove, access) and print out the resulting dictionary.

## Reference Solution
```python
def main():
    # Create a dictionary
    my_dict = {"name": "John", "age": 30, "city": "New York"}

    # Print the initial dictionary
    print("Initial dictionary:")
    print(my_dict)

    # Add a new element to the dictionary
    my_dict["country"] = "USA"
    print("\nDictionary after adding a new element:")
    print(my_dict)

    # Remove an element from the dictionary
    del my_dict["age"]
    print("\nDictionary after removing an element:")
    print(my_dict)

    # Access an element from the dictionary
    print("\nAccessing an element from the dictionary:")
    print("Name:", my_dict["name"])
    print("City:", my_dict["city"])

    # Check if a key exists in the dictionary
    print("\nChecking if a key exists in the dictionary:")
    if "country" in my_dict:
        print("The key 'country' exists in the dictionary.")
    else:
        print("The key 'country' does not exist in the dictionary.")

    # Get all keys and values from the dictionary
    print("\nGetting all keys and values from the dictionary:")
    print("Keys:", list(my_dict.keys()))
    print("Values:", list(my_dict.values()))

if __name__ == "__main__":
    main()
```
