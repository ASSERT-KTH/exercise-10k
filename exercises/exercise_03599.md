## Exercise
Write a program that calculates the speed given the distance and time.

## Reference Solution
```python
def calculate_speed(distance, time):
    """
    Calculate the speed given the distance and time.

    Args:
        distance (float): The distance traveled in meters.
        time (float): The time taken to travel the distance in seconds.

    Returns:
        float: The speed in meters per second.
    """
    if time == 0:
        raise ValueError("Time cannot be zero")
    speed = distance / time
    return speed

def main():
    # Example usage
    distance = float(input("Enter the distance in meters: "))
    time = float(input("Enter the time in seconds: "))

    try:
        speed = calculate_speed(distance, time)
        print(f"The speed is {speed} meters per second")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
