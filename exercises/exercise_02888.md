## Exercise
Create a program to generate the terms of a geometric progression (e.g., 2, 6, 18, 54) based on user input.

## Reference Solution
```python
def geometric_progression(first_term, common_ratio, num_terms):
    """
    Generates the terms of a geometric progression.

    Args:
        first_term (float): The first term of the progression.
        common_ratio (float): The common ratio of the progression.
        num_terms (int): The number of terms to generate.

    Returns:
        list: A list of terms in the geometric progression.
    """
    progression = [first_term]
    for i in range(1, num_terms):
        progression.append(progression[i-1] * common_ratio)
    return progression

def main():
    # Get user input
    first_term = float(input("Enter the first term: "))
    common_ratio = float(input("Enter the common ratio: "))
    num_terms = int(input("Enter the number of terms: "))

    # Generate and print the geometric progression
    progression = geometric_progression(first_term, common_ratio, num_terms)
    print("Geometric Progression:")
    for i, term in enumerate(progression):
        print(f"Term {i+1}: {term}")

if __name__ == "__main__":
    main()
```
