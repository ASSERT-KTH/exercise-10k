## Exercise
Write a program that finds the unique elements in two lists, i.e., elements that are in either list but not both.

## Reference Solution
```python
def unique_elements(list1, list2):
    """
    Returns a set of unique elements in two lists, i.e., elements that are in either list but not both.
    
    Parameters:
    list1 (list): The first list.
    list2 (list): The second list.
    
    Returns:
    set: A set of unique elements.
    """
    set1 = set(list1)
    set2 = set(list2)
    return (set1 - set2) | (set2 - set1)

def main():
    # Example usage
    list1 = [1, 2, 3, 4, 5]
    list2 = [4, 5, 6, 7, 8]
    unique = unique_elements(list1, list2)
    print("Unique elements:", unique)

if __name__ == "__main__":
    main()
```
