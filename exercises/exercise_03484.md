## Exercise
Create a program that takes two numbers as input and returns their quotient.

## Reference Solution
```python
def calculate_quotient(dividend: float, divisor: float) -> float:
    """
    Calculate the quotient of two numbers.

    Args:
    - dividend (float): The number being divided.
    - divisor (float): The number by which we are dividing.

    Returns:
    - float: The quotient of the two numbers.

    Raises:
    - ZeroDivisionError: If the divisor is zero.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend / divisor


def main():
    try:
        # Get user input
        dividend = float(input("Enter the dividend: "))
        divisor = float(input("Enter the divisor: "))

        # Calculate and display the quotient
        quotient = calculate_quotient(dividend, divisor)
        print(f"The quotient of {dividend} and {divisor} is {quotient}")

    except ZeroDivisionError as e:
        print(f"Error: {e}")
    except ValueError:
        print("Error: Invalid input. Please enter a number.")


if __name__ == "__main__":
    main()
```
