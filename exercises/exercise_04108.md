## Exercise
Write a program that demonstrates basic string operations, including concatenation, indexing, and slicing.

## Reference Solution
```python
def main():
    # Define two strings
    str1 = "Hello"
    str2 = "World"

    # Concatenation
    print("Concatenation:")
    result = str1 + " " + str2
    print(result)

    # Indexing
    print("\nIndexing:")
    print("str1[0]:", str1[0])
    print("str1[1]:", str1[1])
    print("str1[-1]:", str1[-1])

    # Slicing
    print("\nSlicing:")
    print("str1[1:3]:", str1[1:3])
    print("str1[1:]:", str1[1:])
    print("str1[:3]:", str1[:3])
    print("str1[:]:", str1[:])

if __name__ == "__main__":
    main()
```
