## Exercise
Write a program that demonstrates the use of type casting to convert between different data types.

## Reference Solution
```python
# Demonstration of Type Casting in Python

def main():
    # Integer to Float
    integer_value = 10
    float_value = float(integer_value)
    print(f"Integer {integer_value} converted to Float: {float_value}")

    # Float to Integer
    float_value = 10.5
    integer_value = int(float_value)
    print(f"Float {float_value} converted to Integer: {integer_value}")

    # String to Integer
    string_value = "10"
    integer_value = int(string_value)
    print(f"String '{string_value}' converted to Integer: {integer_value}")

    # Integer to String
    integer_value = 10
    string_value = str(integer_value)
    print(f"Integer {integer_value} converted to String: '{string_value}'")

    # Boolean to Integer
    boolean_value = True
    integer_value = int(boolean_value)
    print(f"Boolean {boolean_value} converted to Integer: {integer_value}")

    # Integer to Boolean
    integer_value = 10
    boolean_value = bool(integer_value)
    print(f"Integer {integer_value} converted to Boolean: {boolean_value}")

if __name__ == "__main__":
    main()
```
