## Exercise
Write a program that takes two strings as input and returns an integer value indicating the lexicographic order of the strings.

## Reference Solution
```python
#include <iostream>
#include <string>

using namespace std;

int lexicographicOrder(const string& lhs, const string& rhs) {
    if (lhs < rhs) {
        return -1;
    } else if (lhs > rhs) {
        return 1;
    } else {
        return 0;
    }
}

int main() {
    string input1, input2;
    cout << "Enter first string: ";
    cin >> input1;
    cout << "Enter second string: ";
    cin >> input2;

    int result = lexicographicOrder(input1, input2);
    cout << "Lexicographic order result: " << result << endl;

    return 0;
}
```
