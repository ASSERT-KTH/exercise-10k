## Exercise
Write a function to check if a given string is a subsequence of another string.

## Reference Solution
```python
def is_subsequence(main_string, sub_string):
    """
    Checks if a given string is a subsequence of another string.

    Args:
        main_string (str): The main string to check against.
        sub_string (str): The string to check if it's a subsequence.

    Returns:
        bool: True if sub_string is a subsequence of main_string, False otherwise.
    """
    main_index = 0
    sub_index = 0

    while main_index < len(main_string) and sub_index < len(sub_string):
        if main_string[main_index] == sub_string[sub_index]:
            sub_index += 1
        main_index += 1

    return sub_index == len(sub_string)

# Example usage
if __name__ == "__main__":
    main_str = "hello world"
    sub_str = "hwo"
    print(f"Is '{sub_str}' a subsequence of '{main_str}'? {is_subsequence(main_str, sub_str)}")
```
