## Exercise
Write a program that takes a list and a predicate as input, and returns two lists: one containing the elements that satisfy the predicate, and one containing the elements that do not.

## Reference Solution
```python
function split_list(list, predicate) {
  const true_list = [];
  const false_list = [];

  for (const element of list) {
    if (predicate(element)) {
      true_list.push(element);
    } else {
      false_list.push(element);
    }
  }

  return [true_list, false_list];
}

// Example usage:
function is_even(x) {
  return x % 2 === 0;
}

const numbers = [1, 2, 3, 4, 5, 6];

const [even_numbers, odd_numbers] = split_list(numbers, is_even);

console.log("Even numbers:", even_numbers);
console.log("Odd numbers:", odd_numbers);
```
