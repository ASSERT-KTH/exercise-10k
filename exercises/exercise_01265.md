## Exercise
Write a program to sort an array in ascending or descending order.

## Reference Solution
```python
def sort_array(array, order):
    """
    Sorts an array in ascending or descending order.

    Args:
        array (list): The input array to be sorted.
        order (str): The order of sorting, either 'asc' or 'desc'.

    Returns:
        list: The sorted array.
    """
    if order == 'asc':
        return sorted(array)
    elif order == 'desc':
        return sorted(array, reverse=True)
    else:
        raise ValueError("Invalid order. Please specify 'asc' or 'desc'.")

def main():
    array = [64, 34, 25, 12, 22, 11, 90]
    print("Original array:", array)

    # Sort in ascending order
    sorted_array_asc = sort_array(array, 'asc')
    print("Sorted array in ascending order:", sorted_array_asc)

    # Sort in descending order
    sorted_array_desc = sort_array(array, 'desc')
    print("Sorted array in descending order:", sorted_array_desc)

if __name__ == "__main__":
    main()
```
