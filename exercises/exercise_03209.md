## Exercise
Write a program that takes a list of numbers as input and prints the sum of their squares.

## Reference Solution
```python
# Include the math library
use Math::Complex;

# Define a subroutine called square
sub square {
    # Calculate the square of a number
    my $num = shift;
    return $num * $num;
}

# Define a subroutine called sum_of_squares
sub sum_of_squares {
    # Initialize a variable $sum to 0
    my $sum = 0;
    # Iterate over the list of numbers passed as arguments
    foreach my $num (@_) {
        # Add the square of the current number to $sum
        $sum += square($num);
    }
    # Return the final value of $sum
    return $sum;
}

# Define a subroutine called main
sub main {
    # Define a list of numbers
    my @numbers = (1, 2, 3, 4, 5);
    # Call the sum_of_squares subroutine with the list of numbers
    my $result = sum_of_squares(@numbers);
    # Print the result
    print "The sum of the squares is: $result\n";
}

# Call the main subroutine
main();
```
