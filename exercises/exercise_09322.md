## Exercise
Write a program to convert a string to an integer and a float to a string.

## Reference Solution
```python
# Function to convert a string to an integer
def convert_string_to_int(input_string):
    try:
        # Attempt to convert the string to an integer
        result = int(input_string)
        return result
    except ValueError:
        # Handle the case where the string cannot be converted to an integer
        print("Error: The input string cannot be converted to an integer.")
        return None

# Function to convert a float to a string
def convert_float_to_string(input_float):
    try:
        # Attempt to convert the float to a string
        result = str(input_float)
        return result
    except Exception as e:
        # Handle any exceptions that may occur during the conversion
        print(f"An error occurred: {e}")
        return None

# Main function with example usage
def main():
    # Convert a string to an integer
    input_string = "123"
    result = convert_string_to_int(input_string)
    if result is not None:
        print(f"The integer representation of '{input_string}' is: {result}")

    # Convert a float to a string
    input_float = 123.45
    result = convert_float_to_string(input_float)
    if result is not None:
        print(f"The string representation of {input_float} is: {result}")

# Run the main function
if __name__ == "__main__":
    main()
```
