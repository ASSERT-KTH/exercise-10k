## Exercise
Create a program that uses a dictionary to store student grades. The program should allow the user to add a student, add a grade for a student, and calculate the average grade for a student. The dictionary should use the student's name as the key and a list of grades as the value.

## Reference Solution
```python
# student_grades.py

# Initialize an empty dictionary to store student grades
student_grades = {}

def add_student(name):
    """
    Add a new student to the dictionary.
    
    Args:
        name (str): The name of the student.
    """
    if name not in student_grades:
        student_grades[name] = []
        print(f"Student '{name}' added successfully.")
    else:
        print(f"Student '{name}' already exists.")

def add_grade(name, grade):
    """
    Add a grade for a student.
    
    Args:
        name (str): The name of the student.
        grade (float): The grade to be added.
    """
    if name in student_grades:
        student_grades[name].append(grade)
        print(f"Grade '{grade}' added for student '{name}'.")
    else:
        print(f"Student '{name}' does not exist. Please add the student first.")

def calculate_average(name):
    """
    Calculate the average grade for a student.
    
    Args:
        name (str): The name of the student.
    
    Returns:
        float: The average grade for the student.
    """
    if name in student_grades:
        if student_grades[name]:
            average = sum(student_grades[name]) / len(student_grades[name])
            print(f"The average grade for student '{name}' is {average:.2f}.")
            return average
        else:
            print(f"No grades found for student '{name}'.")
    else:
        print(f"Student '{name}' does not exist.")

def main():
    while True:
        print("\nStudent Grades Menu:")
        print("1. Add a student")
        print("2. Add a grade for a student")
        print("3. Calculate the average grade for a student")
        print("4. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            name = input("Enter the student's name: ")
            add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            grade = float(input("Enter the grade: "))
            add_grade(name, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            calculate_average(name)
        elif choice == "4":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
