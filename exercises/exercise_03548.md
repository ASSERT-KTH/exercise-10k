## Exercise
Write a program that takes a list and an aggregation function as input, and returns the aggregated value.

## Reference Solution
```python
def aggregate(lst, agg_func):
    """
    This function aggregates a list using a specified aggregation function.

    Args:
        lst (list): The input list to be aggregated.
        agg_func (function): The aggregation function to be used.

    Returns:
        The aggregated value.
    """
    if not lst:
        raise ValueError("Input list is empty")

    # Check if the input is a list
    if not isinstance(lst, list):
        raise TypeError("Input must be a list")

    # Check if the aggregation function is callable
    if not callable(agg_func):
        raise TypeError("Aggregation function must be callable")

    return agg_func(lst)


# Example usage
if __name__ == "__main__":
    import statistics

    numbers = [1, 2, 3, 4, 5]

    # Calculate the mean
    mean = aggregate(numbers, statistics.mean)
    print(f"Mean: {mean}")

    # Calculate the median
    median = aggregate(numbers, statistics.median)
    print(f"Median: {median}")

    # Calculate the sum
    sum_numbers = aggregate(numbers, sum)
    print(f"Sum: {sum_numbers}")
```
