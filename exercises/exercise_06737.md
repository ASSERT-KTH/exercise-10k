## Exercise
Write a program that monitors input data in real-time

## Reference Solution
```python
import threading
import time

class DataMonitor:
    def __init__(self):
        self.data = []
        self.lock = threading.Lock()

    def monitor_data(self):
        while True:
            user_input = input("Enter data (or 'quit' to exit): ")
            if user_input.lower() == 'quit':
                break
            with self.lock:
                self.data.append(user_input)
                print(f"Received data: {user_input}")

    def process_data(self):
        while True:
            with self.lock:
                if self.data:
                    print(f"Processing data: {self.data.pop(0)}")
            time.sleep(1)  # Wait for 1 second before checking again

def main():
    monitor = DataMonitor()

    # Create and start a thread to monitor input data
    monitor_thread = threading.Thread(target=monitor.monitor_data)
    monitor_thread.start()

    # Create and start a thread to process the monitored data
    process_thread = threading.Thread(target=monitor.process_data)
    process_thread.daemon = True  # Stop the thread when the main program exits
    process_thread.start()

    # Wait for the monitor thread to finish
    monitor_thread.join()

if __name__ == "__main__":
    main()
```
