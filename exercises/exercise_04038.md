## Exercise
Create a function to convert an integer to a Roman numeral

## Reference Solution
```python
def int_to_roman(num: int) -> str:
    """
    Converts an integer to a Roman numeral.

    Args:
    num (int): The integer to be converted.

    Returns:
    str: The Roman numeral representation of the input integer.

    Raises:
    ValueError: If the input integer is not in the range 1-3999.
    """
    if not 1 <= num <= 3999:
        raise ValueError("Input integer must be in the range 1-3999")

    # Dictionary mapping Roman numerals to their integer values
    roman_numerals = {
        1000: 'M',
        900: 'CM',
        500: 'D',
        400: 'CD',
        100: 'C',
        90: 'XC',
        50: 'L',
        40: 'XL',
        10: 'X',
        9: 'IX',
        5: 'V',
        4: 'IV',
        1: 'I'
    }

    result = ''
    for value, numeral in roman_numerals.items():
        # Subtract the largest possible Roman numeral value from the input integer
        while num >= value:
            num -= value
            result += numeral

    return result


# Example usage
if __name__ == "__main__":
    print(int_to_roman(4))  # Should print IV
    print(int_to_roman(9))  # Should print IX
    print(int_to_roman(13))  # Should print XIII
    print(int_to_roman(44))  # Should print XLIV
    print(int_to_roman(1000))  # Should print M
```
