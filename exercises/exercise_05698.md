## Exercise
Write a program that takes a credit card number as input and checks if it's valid using conditional statements.

## Reference Solution
```python
def validate_credit_card(card_number):
    # Convert card number to a list of integers
    card_number = [int(x) for x in str(card_number)]
    
    # Reverse the card number
    card_number = card_number[::-1]
    
    # Double every second digit
    for i in range(1, len(card_number), 2):
        card_number[i] *= 2
        # Subtract 9 from numbers over 9
        if card_number[i] > 9:
            card_number[i] -= 9
    
    # If the sum of the digits is divisible by 10, the card number is valid
    if sum(card_number) % 10 == 0:
        return True
    else:
        return False


def main():
    card_number = input("Enter your credit card number: ")
    
    # Check if the input is a valid integer
    if card_number.isdigit():
        # Check if the card number is valid
        if validate_credit_card(card_number):
            print("The credit card number is valid.")
        else:
            print("The credit card number is not valid.")
    else:
        print("Invalid input. Please enter a valid credit card number.")


if __name__ == "__main__":
    main()
```
