## Exercise
Write a function to check if a given key is present in a dictionary.

## Reference Solution

def check_key(dictionary, key):
    """
    Check if a given key is present in a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.

    Returns:
        bool: True if the key is present, False otherwise.
    """
    return key in dictionary

# Example usage:
my_dict = {"name": "John", "age": 30, "city": "New York"}

def check_key(dictionary, key):
    """
    Check if a given key is present in a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.

    Returns:
        bool: True if the key is present, False otherwise.
    """
    if not isinstance(dictionary, dict):
        raise TypeError("Input is not a dictionary")
    return key in dictionary

# Example usage:
my_dict = {"name": "John", "age": 30, "city": "New York"}
print(check_key(my_dict, "name"))  # Output: True
print(check_key(my_dict, "country"))  # Output: False
print(check_key("not a dictionary", "key"))  # Raises TypeError
