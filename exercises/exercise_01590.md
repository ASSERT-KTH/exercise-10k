## Exercise
Create a program to scramble a word and have the user unscramble it.

## Reference Solution
```python
import random

# List of words to scramble
words = ["apple", "banana", "cherry", "date", "elderberry"]

def scramble_word(word):
    """Scrambles a word by shuffling its characters."""
    word_list = list(word)
    random.shuffle(word_list)
    return "".join(word_list)

def play_game():
    """Plays the word scramble game."""
    word = random.choice(words)
    scrambled_word = scramble_word(word)
    print(f"Unscramble the word: {scrambled_word}")
    user_answer = input("Enter your answer: ")
    if user_answer.lower() == word:
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {word}.")

def main():
    """Main function to start the game."""
    print("Welcome to the Word Scramble Game!")
    play_again = "y"
    while play_again.lower() == "y":
        play_game()
        play_again = input("Do you want to play again? (y/n): ")
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
```
