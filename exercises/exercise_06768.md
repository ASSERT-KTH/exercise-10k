## Exercise
Write a program to create a weather program that displays the current weather and forecast for a given location.

## Reference Solution
```python
import requests
import json

def get_current_weather(api_key, location):
    """
    Retrieves the current weather for a given location.

    Args:
        api_key (str): OpenWeatherMap API key
        location (str): Location (city, state, country)

    Returns:
        dict: Current weather data
    """
    base_url = "http://api.openweathermap.org/data/2.5/weather"
    params = {
        "q": location,
        "units": "metric",
        "appid": api_key
    }
    response = requests.get(base_url, params=params)
    return response.json()

def get_forecast(api_key, location):
    """
    Retrieves the forecast for a given location.

    Args:
        api_key (str): OpenWeatherMap API key
        location (str): Location (city, state, country)

    Returns:
        dict: Forecast data
    """
    base_url = "http://api.openweathermap.org/data/2.5/forecast"
    params = {
        "q": location,
        "units": "metric",
        "appid": api_key
    }
    response = requests.get(base_url, params=params)
    return response.json()

def main():
    api_key = "YOUR_OPENWEATHERMAP_API_KEY"  # replace with your API key
    location = input("Enter location (city, state, country): ")

    current_weather = get_current_weather(api_key, location)
    forecast = get_forecast(api_key, location)

    print("Current Weather:")
    print(f"Weather: {current_weather['weather'][0]['description']}")
    print(f"Temperature: {current_weather['main']['temp']}°C")
    print(f"Humidity: {current_weather['main']['humidity']}%")

    print("\nForecast:")
    for i, forecast_data in enumerate(forecast['list'][:5]):
        print(f"Day {i+1}:")
        print(f"Weather: {forecast_data['weather'][0]['description']}")
        print(f"Temperature: {forecast_data['main']['temp']}°C")
        print(f"Humidity: {forecast_data['main']['humidity']}%")

if __name__ == "__main__":
    main()
```
