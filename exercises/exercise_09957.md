## Exercise
Write a program that attempts to convert a string to an integer and handles potential errors.

## Reference Solution
```python
def convert_string_to_int(input_string):
    """
    Attempts to convert a string to an integer and handles potential errors.

    Args:
        input_string (str): The string to be converted to an integer.

    Returns:
        int: The converted integer if successful, otherwise None.
    """
    try:
        # Attempt to convert the string to an integer
        converted_int = int(input_string)
        return converted_int
    except ValueError:
        # Handle ValueError exception if the string cannot be converted to an integer
        print(f"Error: '{input_string}' cannot be converted to an integer.")
        return None
    except Exception as e:
        # Handle any other unexpected exceptions
        print(f"An unexpected error occurred: {str(e)}")
        return None


def main():
    # Example usage
    input_strings = ["123", "abc", "456.78", "def"]

    for input_string in input_strings:
        result = convert_string_to_int(input_string)
        if result is not None:
            print(f"Converted '{input_string}' to integer: {result}")


if __name__ == "__main__":
    main()
```
