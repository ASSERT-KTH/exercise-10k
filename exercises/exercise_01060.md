## Exercise
Write a program that takes a list as input, and returns the average of the elements of the list.

## Reference Solution
```python
def calculate_average(input_list):
    """
    Calculate the average of the elements in a list.

    Args:
        input_list (list): A list of numbers.

    Returns:
        float: The average of the elements in the list.

    Raises:
        ValueError: If the input list is empty.
        TypeError: If the input list contains non-numeric values.
    """
    if not input_list:
        raise ValueError("Input list is empty")

    try:
        average = sum(input_list) / len(input_list)
        return average
    except TypeError:
        raise TypeError("Input list contains non-numeric values")


def main():
    # Example usage:
    numbers = [1, 2, 3, 4, 5]
    try:
        average = calculate_average(numbers)
        print(f"The average of the list is: {average}")
    except (ValueError, TypeError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
