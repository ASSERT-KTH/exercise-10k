## Exercise
Write a program that takes a story with missing words as input and asks the user to fill in the missing words.

## Reference Solution
```python
def get_user_input(prompt):
    """Gets user input for a specific prompt."""
    while True:
        user_input = input(prompt)
        if user_input:
            return user_input
        else:
            print("Please enter a value.")

def fill_in_story(story):
    """Fills in the missing words in a story."""
    print("Please fill in the missing words in the story:")
    words = story.split()
    filled_in_words = []
    for word in words:
        if word.startswith("{") and word.endswith("}"):
            word_type = word[1:-1]
            user_input = get_user_input(f"Enter a {word_type}: ")
            filled_in_words.append(user_input)
        else:
            filled_in_words.append(word)
    return " ".join(filled_in_words)

def main():
    """Main function."""
    story = input("Enter a story with missing words (use {word_type} for missing words): ")
    filled_in_story_text = fill_in_story(story)
    print("\nFilled-in story:")
    print(filled_in_story_text)

if __name__ == "__main__":
    main()
```
