## Exercise
Write a function that checks if a list is empty.

## Reference Solution
```python
def is_list_empty(input_list):
    """
    Checks if a list is empty.

    Args:
        input_list (list): The list to be checked.

    Returns:
        bool: True if the list is empty, False otherwise.
    """
    return len(input_list) == 0

# Example usage:
def main():
    empty_list = []
    non_empty_list = [1, 2, 3]

    print(is_list_empty(empty_list))  # Expected output: True
    print(is_list_empty(non_empty_list))  # Expected output: False

if __name__ == "__main__":
    main()
```
