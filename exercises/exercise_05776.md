## Exercise
Create a program that calculates the average grade of a student based on their scores.

## Reference Solution
```python
def calculate_average_grade():
    """
    Calculates the average grade of a student based on their scores.
    """
    num_subjects = int(input("Enter the number of subjects: "))

    total_score = 0
    subject_scores = {}

    for i in range(num_subjects):
        subject = input(f"Enter subject {i+1} name: ")
        score = float(input(f"Enter {subject} score (out of 100): "))

        # Validate score
        while score < 0 or score > 100:
            print("Invalid score. Please enter a score between 0 and 100.")
            score = float(input(f"Enter {subject} score (out of 100): "))

        total_score += score
        subject_scores[subject] = score

    average_grade = total_score / num_subjects

    print("\nSubject Scores:")
    for subject, score in subject_scores.items():
        print(f"{subject}: {score:.2f}")

    print(f"\nAverage Grade: {average_grade:.2f}")

    # Determine grade based on average score
    if average_grade >= 90:
        print("Grade: A")
    elif average_grade >= 80:
        print("Grade: B")
    elif average_grade >= 70:
        print("Grade: C")
    elif average_grade >= 60:
        print("Grade: D")
    else:
        print("Grade: F")

if __name__ == "__main__":
    calculate_average_grade()
```
