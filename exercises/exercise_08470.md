## Exercise
Write a program to calculate a person's age given their birth date and the current date

## Reference Solution
```python
from datetime import date

def calculate_age(birth_date):
    """
    Calculate the age of a person given their birth date and the current date.

    Args:
        birth_date (date): The birth date of the person.

    Returns:
        int: The age of the person.
    """
    today = date.today()
    age = today.year - birth_date.year - ((today.month, today.day) < (birth_date.month, birth_date.day))
    return age

def main():
    # Example usage
    birth_date = date(1990, 1, 1)  # Replace with the desired birth date
    age = calculate_age(birth_date)
    print(f"The person's age is: {age} years")

    # Get birth date from user
    year = int(input("Enter birth year: "))
    month = int(input("Enter birth month (1-12): "))
    day = int(input("Enter birth day (1-31): "))

    birth_date = date(year, month, day)
    age = calculate_age(birth_date)
    print(f"The person's age is: {age} years")

if __name__ == "__main__":
    main()
```
