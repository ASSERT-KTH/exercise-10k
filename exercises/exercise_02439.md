## Exercise
Develop a program to play Tic-Tac-Toe against the computer, using a while loop to handle turns and a for loop to check for wins.

## Reference Solution
```python
import random

# Initialize the game board
board = [' ' for _ in range(9)]

def print_board():
    """
    Print the current state of the game board.
    """
    print(f"{board[0]} | {board[1]} | {board[2]}")
    print("--+---+--")
    print(f"{board[3]} | {board[4]} | {board[5]}")
    print("--+---+--")
    print(f"{board[6]} | {board[7]} | {board[8]}")

def check_win(player):
    """
    Check if the given player has won the game.

    Args:
        player (str): The player to check for (either 'X' or 'O').

    Returns:
        bool: True if the player has won, False otherwise.
    """
    win_conditions = [(0, 1, 2), (3, 4, 5), (6, 7, 8), (0, 3, 6), (1, 4, 7), (2, 5, 8), (0, 4, 8), (2, 4, 6)]
    for condition in win_conditions:
        if board[condition[0]] == board[condition[1]] == board[condition[2]] == player:
            return True
    return False

def computer_move():
    """
    Make a random move for the computer player.
    """
    available_moves = [i for i, x in enumerate(board) if x == ' ']
    if available_moves:
        move = random.choice(available_moves)
        board[move] = 'O'

def main():
    current_player = 'X'
    while True:
        print_board()
        if current_player == 'X':
            move = input("Enter your move (1-9): ")
            if board[int(move) - 1] != ' ':
                print("Invalid move, try again.")
                continue
            board[int(move) - 1] = 'X'
            if check_win('X'):
                print_board()
                print("You win!")
                break
        else:
            computer_move()
            if check_win('O'):
                print_board()
                print("Computer wins!")
                break
        current_player = 'O' if current_player == 'X' else 'X'

if __name__ == "__main__":
    main()
```
