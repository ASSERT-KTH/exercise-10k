## Exercise
Write a program that takes two inputs, the user's choice and the computer's choice, and determines the winner based on the game's rules.

#include <iostream>
#include <string>

// Function to determine the winner
std::string find_winner(std::string user_choice, std::string computer_choice) {
  // Check for invalid inputs
  if (user_choice != "rock" && user_choice != "paper" && user_choice != "scissors") {
    return "Invalid user choice. Please enter 'rock', 'paper', or 'scissors'.";
  }
  if (computer_choice != "rock" && computer_choice != "paper" && computer_choice != "scissors") {
    return "Invalid computer choice. Please enter 'rock', 'paper', or 'scissors'.";
  }

  // Determine the winner
  if (user_choice == computer_choice) {
    return "It's a tie!";
  }
  if ((user_choice == "rock" && computer_choice == "scissors") ||
      (user_choice == "scissors" && computer_choice == "paper") ||
      (user_choice == "paper" && computer_choice == "rock")) {
    return "User wins!";
  }
  return "Computer wins!";
}

int main() {
  std::string user_choice, computer_choice;
  std::cout << "Enter user's choice: ";
  std::cin >> user_choice;
  std::cout << "Enter computer's choice: ";
  std::cin >> computer_choice;

  std::string winner = find_winner(user_choice, computer_choice);
  std::cout << "The winner is: " << winner << std::endl;

  return 0;
}
