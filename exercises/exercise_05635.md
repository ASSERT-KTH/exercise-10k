## Exercise
Write a program to perform a topological sort on a given directed acyclic graph (DAG).

## Reference Solution

from collections import defaultdict, deque

def topological_sort(graph):
    """
    Performs a topological sort on a given DAG.

    Args:
        graph (dict): A dictionary representing the DAG, where each key is a vertex and its corresponding value is a list of its neighbors.

    Returns:
        list: A list of vertices in topologically sorted order.
    """

    # Initialize a dictionary to store the in-degree of each vertex
    in_degree = defaultdict(int)

    # Calculate the in-degree of each vertex
    for vertex in graph:
        for neighbor in graph[vertex]:
            in_degree[neighbor] += 1

    # Initialize a queue with vertices having in-degree 0
    queue = deque([vertex for vertex in graph if in_degree[vertex] == 0])

    # Initialize an empty list to store the sorted vertices
    sorted_vertices = []

    # Perform the topological sort
    while queue:
        vertex = queue.popleft()
        sorted_vertices.append(vertex)

        # Decrease the in-degree of each neighbor by 1
        for neighbor in graph[vertex]:
            in_degree[neighbor] -= 1

            # Add the neighbor to the queue if its in-degree becomes 0
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # Check if the graph has a cycle (i.e., it's not a DAG)
    if len(sorted_vertices) != len(graph):
        raise ValueError("The graph has a cycle and is not a DAG")

    return sorted_vertices


# Example usage
if __name__ == "__main__":
    graph = {

        'B': ['D', 'E'],
        'C': ['F'],
        'D': [],
        'E': ['F'],
        'F': []
    }

    sorted_vertices = topological_sort(graph)
    print("Topologically sorted vertices:", sorted_vertices)
