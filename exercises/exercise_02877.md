## Exercise
Write a function to replace all occurrences of a specified character with another character in a given string

## Reference Solution
```python
def replace_character(input_string: str, char_to_replace: str, replacement_char: str) -> str:
    """
    Replaces all occurrences of a specified character with another character in a given string.

    Args:
        input_string (str): The input string to modify.
        char_to_replace (str): The character to replace.
        replacement_char (str): The character to replace with.

    Returns:
        str: The modified string with replaced characters.
    """
    if len(char_to_replace) != 1 or len(replacement_char) != 1:
        raise ValueError("Both char_to_replace and replacement_char must be single characters")
    
    return input_string.replace(char_to_replace, replacement_char)

# Example usage:
def main():
    input_str = "Hello, World!"
    char_to_replace = "o"
    replacement_char = "0"
    
    print(f"Original String: {input_str}")
    print(f"Character to Replace: {char_to_replace}")
    print(f"Replacement Character: {replacement_char}")
    
    modified_str = replace_character(input_str, char_to_replace, replacement_char)
    print(f"Modified String: {modified_str}")

if __name__ == "__main__":
    main()
```
