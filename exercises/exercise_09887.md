## Exercise
Write a program that takes a word as input and finds all possible anagrams.

## Reference Solution
```python
import itertools

def generate_anagrams(word):
    """
    Generate all possible anagrams of a given word.

    Args:
        word (str): The input word.

    Returns:
        list: A list of all possible anagrams.
    """
    # Generate all permutations of the word
    perms = itertools.permutations(word)

    # Convert permutations to a list of strings and remove duplicates
    anagrams = set("".join(p) for p in perms)

    return list(anagrams)

def main():
    # Get the input word from the user
    word = input("Enter a word: ")

    # Generate and print all possible anagrams
    anagrams = generate_anagrams(word)
    print("Anagrams:")
    for i, anagram in enumerate(anagrams):
        print(f"{i+1}. {anagram}")

if __name__ == "__main__":
    main()
```
