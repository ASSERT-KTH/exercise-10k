def check_data_type(var):
    if isinstance(var, int):
        print("The data type of the variable is: integer")
    elif isinstance(var, float):
        print("The data type of the variable is: float")
    elif isinstance(var, str):
        print("The data type of the variable is: string")
    elif isinstance(var, bool):
        print("The data type of the variable is: boolean")
    elif isinstance(var, list):
        print("The data type of the variable is: list")
    elif isinstance(var, tuple):
        print("The data type of the variable is: tuple")
    elif isinstance(var, dict):
        print("The data type of the variable is: dictionary")
    elif isinstance(var, set):
        print("The data type of the variable is: set")
    else:
        print("Unknown data type")

# Test the function with different data types
check_data_type(10)  # integer
check_data_type(10.5)  # float
check_data_type("Hello")  # string
check_data_type(True)  # boolean
