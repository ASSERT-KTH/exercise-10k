## Exercise
Create a program that takes a story with blanks as input and fills them with user-provided words.

## Reference Solution
```python
def get_word(prompt):
    """Get a word from the user."""
    while True:
        word = input(prompt)
        if word:  # Check if the word is not empty
            return word
        print("Please enter a word.")

def fill_story(story):
    """Fill the story with user-provided words."""
    words = {}
    for i, blank in enumerate(story["blanks"]):
        word = get_word(f"Enter a {blank['type']}: ")
        words[blank['name']] = word
    filled_story = story["text"]
    for blank, word in words.items():
        filled_story = filled_story.replace(f"{{{blank}}}", word)
    return filled_story

def main():
    # Example story with blanks
    story = {
        "text": "Once upon a time, there was a {name} who lived in a {location}. "
                "This {name} had a {adjective} {pet} and loved to eat {food}.",
        "blanks": [
            {"name": "name", "type": "name"},
            {"name": "location", "type": "location"},
            {"name": "adjective", "type": "adjective"},
            {"name": "pet", "type": "pet"},
            {"name": "food", "type": "food"}
        ]
    }
    filled_story = fill_story(story)
    print("Filled Story:")
    print(filled_story)

if __name__ == "__main__":
    main()
```
