## Exercise
Write a program to calculate the modulus (remainder) of two integers.

## Reference Solution
```python
def calculate_modulus(dividend: int, divisor: int) -> int:
    """
    Calculate the modulus (remainder) of two integers.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        int: The remainder of the division.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend % divisor

def main():
    # Example usage
    dividend = 17
    divisor = 5
    try:
        modulus = calculate_modulus(dividend, divisor)
        print(f"The modulus of {dividend} and {divisor} is: {modulus}")
    except ZeroDivisionError as e:
        print(e)

if __name__ == "__main__":
    main()
```
