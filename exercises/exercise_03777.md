## Exercise
Aggregate a dictionary by a common key.

## Reference Solution
```python
# Sample dictionary
data = [
    {'id': 1, 'name': 'John', 'age': 25},
    {'id': 1, 'name': 'John', 'city': 'New York'},
    {'id': 2, 'name': 'Jane', 'age': 30},
    {'id': 2, 'name': 'Jane', 'city': 'Chicago'}
]

# Aggregate the dictionary by 'id'
aggregated_data = {}
for item in data:
    if item['id'] not in aggregated_data:
        aggregated_data[item['id']] = {}
    aggregated_data[item['id']].update({k: v for k, v in item.items() if k != 'id'})

print(aggregated_data)
```

```python
import pandas as pd

# Sample dictionary
data = [
    {'id': 1, 'name': 'John', 'age': 25},
    {'id': 1, 'name': 'John', 'city': 'New York'},
    {'id': 2, 'name': 'Jane', 'age': 30},
    {'id': 2, 'name': 'Jane', 'city': 'Chicago'}
]

# Convert the dictionary to a pandas DataFrame
df = pd.DataFrame(data)

# Define a custom function to aggregate the data
def aggregate_data(group):
    aggregated_data = {}
    for index, row in group.iterrows():
        for column, value in row.items():
            if column != 'id':
                aggregated_data[column] = value
    return aggregated_data

# Aggregate the DataFrame by 'id'
aggregated_df = df.groupby('id').apply(aggregate_data)

print(aggregated_df)
```
