## Exercise
Write a program to count the occurrences of each ASCII character in a given string.

## Reference Solution
```python
def count_ascii_chars(input_string):
    # Initialize a dictionary to store ASCII character counts
    ascii_char_counts = {}

    # Iterate over each character in the input string
    for char in input_string:
        # Get the ASCII value of the character
        ascii_value = ord(char)

        # Check if the ASCII value is already in the dictionary
        if ascii_value in ascii_char_counts:
            # If it is, increment the count
            ascii_char_counts[ascii_value] += 1
        else:
            # If it's not, add it to the dictionary with a count of 1
            ascii_char_counts[ascii_value] = 1

    return ascii_char_counts

def print_ascii_char_counts(ascii_char_counts):
    # Iterate over each ASCII value and count in the dictionary
    for ascii_value, count in ascii_char_counts.items():
        # Print the ASCII value, the corresponding character, and the count
        print(f"ASCII Value: {ascii_value}, Character: '{chr(ascii_value)}', Count: {count}")

def main():
    # Get the input string from the user
    input_string = input("Enter a string: ")

    # Count the occurrences of each ASCII character
    ascii_char_counts = count_ascii_chars(input_string)

    # Print the counts
    print_ascii_char_counts(ascii_char_counts)

if __name__ == "__main__":
    main()
```
