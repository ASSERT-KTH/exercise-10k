## Exercise
Create a program that calculates the selling price of a product given the cost price and markup percentage.

## Reference Solution
```python
def calculate_selling_price(cost_price, markup_percentage):
    """
    Calculates the selling price of a product given the cost price and markup percentage.

    Args:
        cost_price (float): The cost price of the product.
        markup_percentage (float): The markup percentage.

    Returns:
        float: The selling price of the product.
    """
    selling_price = cost_price + (cost_price * (markup_percentage / 100))
    return selling_price

def main():
    print("Selling Price Calculator")
    print("-------------------------")

    # Get user input
    cost_price = float(input("Enter the cost price: $"))
    markup_percentage = float(input("Enter the markup percentage (%): "))

    # Calculate selling price
    selling_price = calculate_selling_price(cost_price, markup_percentage)

    # Display result
    print(f"Selling Price: ${selling_price:.2f}")

if __name__ == "__main__":
    main()
```
