## Exercise
Write a program to calculate the average grade given a list of grades.

## Reference Solution
```python
def calculate_average(grades):
    """
    Calculate the average grade from a list of grades.

    Args:
        grades (list): A list of grades (numeric values)

    Returns:
        float: The average grade
    """
    try:
        return sum(grades) / len(grades)
    except ZeroDivisionError:
        return 0  # Return 0 if the list is empty

def main():
    # Example usage
    grades = [90, 85, 95, 92, 88]
    average = calculate_average(grades)
    print(f"Average grade: {average:.2f}")

    # Ask user for input
    user_grades = input("Enter a list of grades separated by spaces: ")
    try:
        user_grades = [float(grade) for grade in user_grades.split()]
        user_average = calculate_average(user_grades)
        print(f"Average grade: {user_average:.2f}")
    except ValueError:
        print("Invalid input. Please enter numeric grades only.")

if __name__ == "__main__":
    main()
```
