## Exercise
Write a program that finds the minimum window of a given string that contains all characters of another given string.

## Reference Solution

```python
from collections import Counter

def min_window(s, t):
    """
    Find the minimum window of string `s` that contains all characters of string `t`.

    Args:
    s (str): The string to search in.
    t (str): The string containing the characters to search for.

    Returns:
    str: The minimum window of `s` that contains all characters of `t`. If no such window exists, returns an empty string.
    """
    if not s or not t:
        return ""

    t_count = Counter(t)
    required_chars = len(t_count)
    left = 0
    min_len = float('inf')
    min_window = ""

    for right in range(len(s)):
        if s[right] in t_count:
            t_count[s[right]] -= 1
            if t_count[s[right]] == 0:
                required_chars -= 1

        while required_chars == 0:
            if right - left + 1 < min_len:
                min_len = right - left + 1
                min_window = s[left:right + 1]

            if s[left] in t_count:
                t_count[s[left]] += 1
                if t_count[s[left]] > 0:
                    required_chars += 1
            left += 1

    return min_window

# Example usage:
s = "ADOBECODEBANC"
t = "ABC"
print(min_window(s, t))  # Output: "BANC"
```
