## Exercise
Write a program that takes nested input from the user and prints out the results.

## Reference Solution
```python
class NestedInput:
    def __init__(self):
        self.nested_list = []

    def get_input(self):
        while True:
            user_input = input("Enter a value (or 'done' to finish): ")
            if user_input.lower() == 'done':
                break
            elif user_input.lower() == 'new':
                self.nested_list.append(self.get_nested_input())
            else:
                self.nested_list.append(user_input)

    def get_nested_input(self):
        print("Enter values for the nested list (or 'done' to finish):")
        nested_list = []
        while True:
            user_input = input("Enter a value (or 'done' to finish): ")
            if user_input.lower() == 'done':
                break
            elif user_input.lower() == 'new':
                nested_list.append(self.get_nested_input())
            else:
                nested_list.append(user_input)
        return nested_list

    def print_results(self):
        print("\nResults:")
        self.print_nested_list(self.nested_list)

    def print_nested_list(self, nested_list, indent=0):
        for item in nested_list:
            if isinstance(item, list):
                print("  " * indent + "Nested list:")
                self.print_nested_list(item, indent + 1)
            else:
                print("  " * indent + str(item))


def main():
    nested_input = NestedInput()
    print("Enter values (or 'new' to create a nested list, or 'done' to finish):")
    nested_input.get_input()
    nested_input.print_results()


if __name__ == "__main__":
    main()
```
