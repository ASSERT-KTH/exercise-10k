## Exercise
Write a program that asks for a string and a character, and prints out the index of the character in the string.

## Reference Solution
```python
def get_index():
    """
    Asks for a string and a character, and prints out the index of the character in the string.
    """
    string = input("Please enter a string: ")
    char = input("Please enter a character: ")

    # Check if the character is in the string
    if char in string:
        # Find the index of the character
        index = string.find(char)
        print(f"The character '{char}' is found at index {index}.")
    else:
        print(f"The character '{char}' is not found in the string.")

def main():
    get_index()

if __name__ == "__main__":
    main()
```

```python
def get_indices():
    """
    Asks for a string and a character, and prints out the indices of the character in the string.
    """
    string = input("Please enter a string: ")
    char = input("Please enter a character: ")

    # Check if the character is in the string
    if char in string:
        # Find the indices of the character
        indices = [i for i, c in enumerate(string) if c == char]
        print(f"The character '{char}' is found at indices {indices}.")
    else:
        print(f"The character '{char}' is not found in the string.")

def main():
    get_indices()

if __name__ == "__main__":
    main()
```
