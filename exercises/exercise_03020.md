## Exercise
Write a program to rotate a given string by a specified amount.

## Reference Solution
```python
def rotate_string(input_string, rotation_amount, direction='clockwise'):
    """
    Rotate a given string by a specified amount.

    Args:
        input_string (str): The string to be rotated.
        rotation_amount (int): The amount of rotation.
        direction (str, optional): The direction of rotation. Defaults to 'clockwise'.

    Returns:
        str: The rotated string.
    """
    if direction == 'clockwise':
        return input_string[-rotation_amount:] + input_string[:-rotation_amount]
    elif direction == 'counterclockwise':
        return input_string[rotation_amount:] + input_string[:rotation_amount]
    else:
        raise ValueError("Invalid direction. Please specify 'clockwise' or 'counterclockwise'.")

def main():
    input_string = "Hello, World!"
    rotation_amount = 5
    direction = 'clockwise'

    rotated_string = rotate_string(input_string, rotation_amount, direction)
    print(f"Original String: {input_string}")
    print(f"Rotated String ({direction}): {rotated_string}")

if __name__ == "__main__":
    main()
```
