## Exercise
Write a program that takes the principal amount, interest rate, and time as input, and returns the simple interest.

## Reference Solution
```python
def calculate_simple_interest(principal, rate, time):
    """
    Calculate the simple interest.

    Args:
    principal (float): The principal amount.
    rate (float): The interest rate (in percent).
    time (float): The time period (in years).

    Returns:
    float: The simple interest.
    """
    # Convert the interest rate from percent to a decimal
    rate = rate / 100
    
    # Calculate the simple interest
    simple_interest = principal * rate * time
    
    return simple_interest

def main():
    # Get the principal amount, interest rate, and time from the user
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in percent): "))
    time = float(input("Enter the time period (in years): "))

    # Calculate and display the simple interest
    interest = calculate_simple_interest(principal, rate, time)
    print(f"The simple interest is: {interest}")

if __name__ == "__main__":
    main()
```
