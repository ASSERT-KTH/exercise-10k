## Exercise
Write a program that takes an ISBN number as input and prints out whether it is valid or not

## Reference Solution
```python
def is_valid_isbn(isbn):
    """
    Checks whether an ISBN number is valid.
    
    Args:
    isbn (str): The ISBN number to be checked.
    
    Returns:
    bool: True if the ISBN is valid, False otherwise.
    """
    
    # Remove hyphens from the ISBN
    isbn = isbn.replace("-", "")
    
    # Check if the ISBN is 10 or 13 digits long
    if len(isbn) == 10:
        return is_valid_isbn_10(isbn)
    elif len(isbn) == 13:
        return is_valid_isbn_13(isbn)
    else:
        return False


def is_valid_isbn_10(isbn):
    """
    Checks whether a 10-digit ISBN number is valid.
    
    Args:
    isbn (str): The 10-digit ISBN number to be checked.
    
    Returns:
    bool: True if the ISBN is valid, False otherwise.
    """
    
    # Check if the ISBN contains only digits and a possible 'X' at the end
    if not isbn[:-1].isdigit() or (isbn[-1].upper() != 'X' and not isbn[-1].isdigit()):
        return False
    
    # Calculate the checksum
    checksum = 0
    for i in range(10):
        if isbn[i].upper() == 'X':
            checksum += 10 * (10 - i)
        else:
            checksum += int(isbn[i]) * (10 - i)
    
    # Check if the checksum is divisible by 11
    return checksum % 11 == 0


def is_valid_isbn_13(isbn):
    """
    Checks whether a 13-digit ISBN number is valid.
    
    Args:
    isbn (str): The 13-digit ISBN number to be checked.
    
    Returns:
    bool: True if the ISBN is valid, False otherwise.
    """
    
    # Check if the ISBN contains only digits
    if not isbn.isdigit():
        return False
    
    # Calculate the checksum
    checksum = 0
    for i in range(13):
        digit = int(isbn[i])
        if i % 2 == 0:
            checksum += digit
        else:
            checksum += digit * 3
    
    # Check if the checksum is divisible by 10
    return checksum % 10 == 0


def main():
    isbn = input("Enter an ISBN number: ")
    if is_valid_isbn(isbn):
        print("The ISBN is valid.")
    else:
        print("The ISBN is not valid.")


if __name__ == "__main__":
    main()
```
