## Exercise
Write a program that calculates the total cost of items based on their prices, quantities, and tax rate provided by the user.

## Reference Solution
```python
# Define a function to calculate the total cost
def calculate_total_cost():
    # Initialize the total cost
    total_cost = 0

    # Get the number of items from the user
    num_items = int(input("Enter the number of items: "))

    # Get the price, quantity, and tax rate for each item
    for i in range(num_items):
        price = float(input(f"Enter the price of item {i+1}: $"))
        quantity = int(input(f"Enter the quantity of item {i+1}: "))
        tax_rate = float(input(f"Enter the tax rate for item {i+1} (in %): "))

        # Calculate the subtotal for the current item
        subtotal = price * quantity

        # Calculate the tax for the current item
        tax = subtotal * (tax_rate / 100)

        # Calculate the total cost for the current item
        item_total = subtotal + tax

        # Add the total cost for the current item to the overall total cost
        total_cost += item_total

        # Print the details for the current item
        print(f"Item {i+1} Details:")
        print(f"Price: ${price:.2f}")
        print(f"Quantity: {quantity}")
        print(f"Tax Rate: {tax_rate}%")
        print(f"Subtotal: ${subtotal:.2f}")
        print(f"Tax: ${tax:.2f}")
        print(f"Total: ${item_total:.2f}")
        print()

    # Print the overall total cost
    print(f"Overall Total Cost: ${total_cost:.2f}")

# Call the function to calculate the total cost
calculate_total_cost()
```
