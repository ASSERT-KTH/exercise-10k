```python
# Define a string variable

```

```python
# Define a list of string variables
fruits = ["apple", "banana", "cherry"]

# Join the list of strings into a single string
str1 = ",".join(fruits)

```

```python
# Define a string variable
str1 = "Hello, World!"

str2 = str1.replace("World", "Earth")

```

```python

age = 30

# Format the string using the format() method

# Format the string using f-strings
str2 = f"My name is {name} and I am {age} years old."

print(str1)  # Output: "My name is John and I am 30 years old."
```
