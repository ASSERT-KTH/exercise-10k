## Exercise
Create a function that takes in two numbers and an operator, then performs the specified calculation. The operators can be '+', '-', '*', or '/'. Handle any potential division by zero errors.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    Perform a calculation based on the given operator.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the calculation.

    Raises:
        ValueError: If the operator is not one of the supported operators.
        ZeroDivisionError: If the operator is '/' and num2 is 0.
    """

    # Check if the operator is valid
    if operator not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operator. Supported operators are '+', '-', '*', and '/'.")

    # Perform the calculation based on the operator
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


# Example usage
def main():
    print(calculate(10, 2, '+'))  # Output: 12
    print(calculate(10, 2, '-'))  # Output: 8
    print(calculate(10, 2, '*'))  # Output: 20
    print(calculate(10, 2, '/'))  # Output: 5.0

    # Test division by zero
    try:
        print(calculate(10, 0, '/'))
    except ZeroDivisionError as e:
        print(e)  # Output: Cannot divide by zero.

    # Test invalid operator
    try:
        print(calculate(10, 2, '%'))
    except ValueError as e:
        print(e)  # Output: Invalid operator. Supported operators are '+', '-', '*', and '/'.


if __name__ == "__main__":
    main()
```
