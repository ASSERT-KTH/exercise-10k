## Exercise
Write a program that takes a graph as input and returns the number of connected components.

from typing import Dict, List

def count_connected_components(graph: Dict[int, List[int]]) -> int:
    """
    Counts the number of connected components in an undirected graph.

    Args:
    graph: A dictionary representing the adjacency list of the graph.
           Each key is a node, and its corresponding value is a list of its neighbors.

    Returns:
    The number of connected components in the graph.
    """

    # Initialize a set to keep track of visited nodes
    visited = set()

    # Initialize a variable to count the number of connected components
    count = 0

    # Iterate over each node in the graph
    for node in graph:
        # If the node has not been visited yet, it's a new connected component
        if node not in visited:
            # Increment the count of connected components
            count += 1

            # Perform a DFS from the current node to mark all nodes in the same component as visited
            dfs(graph, node, visited)

    # Return the total count of connected components
    return count


def dfs(graph: Dict[int, List[int]], node: int, visited: set) -> None:
    """
    Performs a Depth-First Search (DFS) from a given node in the graph.

    Args:
    graph: A dictionary representing the adjacency list of the graph.
           Each key is a node, and its corresponding value is a list of its neighbors.
    node: The current node being visited.
    visited: A set of nodes that have already been visited.
    """

    # Mark the current node as visited
    visited.add(node)

    # Iterate over each neighbor of the current node
    for neighbor in graph[node]:
        # If the neighbor has not been visited yet, recursively visit it
        if neighbor not in visited:
            dfs(graph, neighbor, visited)


# Example usage:
graph = {
    0: [1, 2],
    1: [0, 2],
    2: [0, 1],
    3: [4],
    4: [3],
    5: []
}

connected_components = count_connected_components(graph)
print(f"Number of connected components: {connected_components}")
