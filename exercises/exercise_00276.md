## Exercise
Write a program that calculates the future population of a city given its current population, growth rate, and time period.

## Reference Solution
```python
def calculate_future_population(current_population, growth_rate, time_period):
    """
    Calculates the future population of a city.

    Args:
        current_population (int): The current population of the city.
        growth_rate (float): The growth rate of the city as a decimal.
        time_period (int): The time period in years.

    Returns:
        int: The future population of the city.
    """
    future_population = current_population * (1 + growth_rate) ** time_period
    return round(future_population)

def main():
    print("Population Growth Calculator")
    print("--------------------------------")

    # Get user input
    current_population = int(input("Enter the current population: "))
    growth_rate = float(input("Enter the growth rate (in %): ")) / 100
    time_period = int(input("Enter the time period (in years): "))

    # Calculate future population
    future_population = calculate_future_population(current_population, growth_rate, time_period)

    # Print result
    print(f"The future population in {time_period} years will be approximately {future_population} people.")

if __name__ == "__main__":
    main()
```
