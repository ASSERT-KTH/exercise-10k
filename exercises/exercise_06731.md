## Exercise
Write a program that filters input data based on certain criteria

## Reference Solution
```python
def filter_data(data, criteria):
    """
    Filters input data based on certain criteria.

    Args:
        data (list): A list of dictionaries, where each dictionary represents a data point.
        criteria (dict): A dictionary where the keys are the fields to filter by and the values are the desired values.

    Returns:
        list: A list of dictionaries that match the specified criteria.
    """
    filtered_data = []
    for data_point in data:
        matches_criteria = True
        for field, value in criteria.items():
            if field not in data_point or data_point[field] != value:
                matches_criteria = False
                break
        if matches_criteria:
            filtered_data.append(data_point)
    return filtered_data

# Example usage
data = [
    {"name": "John", "age": 25, "city": "New York"},
    {"name": "Jane", "age": 30, "city": "New York"},
    {"name": "Bob", "age": 25, "city": "Chicago"},
    {"name": "Alice", "age": 30, "city": "Chicago"}
]

criteria = {"age": 25, "city": "New York"}

filtered_data = filter_data(data, criteria)
print(filtered_data)
```
