## Exercise
Write a program that generates four random numbers and asks the user to make an equation that equals 24 using basic arithmetic operations.

## Reference Solution
```python
import random
import operator

# Define the basic arithmetic operations
operations = {
    operator.add: '+',
    operator.sub: '-',
    operator.mul: '*',
    operator.truediv: '/'
}

def generate_numbers():
    """Generate four random numbers between 1 and 10"""
    return [random.randint(1, 10) for _ in range(4)]

def get_user_equation(numbers):
    """Ask the user to input an equation"""
    print("Create an equation that equals 24 using the numbers:", numbers)
    equation = input("Enter your equation: ")
    return equation

def evaluate_equation(equation):
    """Evaluate the user's equation"""
    try:
        result = eval(equation)
        return result
    except ZeroDivisionError:
        print("Error: Division by zero is not allowed.")
        return None
    except Exception as e:
        print("Error: Invalid equation.")
        return None

def check_solution(result):
    """Check if the equation equals 24"""
    if result == 24:
        print(" Congratulations! Your equation is correct.")
    else:
        print("Sorry, your equation does not equal 24.")

def main():
    numbers = generate_numbers()
    equation = get_user_equation(numbers)
    result = evaluate_equation(equation)
    if result is not None:
        check_solution(result)

if __name__ == "__main__":
    main()
```
