## Exercise
Write a program to calculate the solutions of a quadratic equation

import math

def calculate_quadratic_solutions(a, b, c):
    """
    Calculate the solutions of a quadratic equation.

    Args:
        a (float): Coefficient of x^2
        b (float): Coefficient of x
        c (float): Constant term

    Returns:
        tuple: Solutions of the quadratic equation
    """
    determinant = b**2 - 4*a*c

    if determinant > 0:
        # Two distinct real roots
        root1 = (-b + math.sqrt(determinant)) / (2*a)
        root2 = (-b - math.sqrt(determinant)) / (2*a)
        return (root1, root2)
    elif determinant == 0:
        # One repeated real root
        root = -b / (2*a)
        return (root, root)
    else:
        # Two complex roots
        real_part = -b / (2*a)
        imaginary_part = math.sqrt(-determinant) / (2*a)
        root1 = complex(real_part, imaginary_part)
        root2 = complex(real_part, -imaginary_part)
        return (root1, root2)

def main():
    # Example usage
    a = float(input("Enter coefficient of x^2 (a): "))
    b = float(input("Enter coefficient of x (b): "))
    c = float(input("Enter constant term (c): "))

    solutions = calculate_quadratic_solutions(a, b, c)

    print("Solutions of the quadratic equation:")
    if solutions[0] == solutions[1]:
        print(f"x = {solutions[0]}")
    else:
        print(f"x1 = {solutions[0]}")
        print(f"x2 = {solutions[1]}")

if __name__ == "__main__":
    main()
