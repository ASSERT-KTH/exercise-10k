## Exercise
Write a program to keep track of game scores using conditional statements to determine the winner

## Reference Solution
```python
def game_score_tracker():
    """
    Tracks game scores and determines the winner.
    """

    # Initialize team names and scores
    team1_name = input("Enter Team 1 name: ")
    team2_name = input("Enter Team 2 name: ")

    team1_score = int(input(f"Enter {team1_name} score: "))
    team2_score = int(input(f"Enter {team2_name} score: "))

    # Determine the winner using conditional statements
    if team1_score > team2_score:
        winner = team1_name
    elif team2_score > team1_score:
        winner = team2_name
    else:
        winner = "It's a tie!"

    # Print the results
    print("\nGame Results:")
    print(f"{team1_name} score: {team1_score}")
    print(f"{team2_name} score: {team2_score}")
    print(f"The winner is: {winner}")

def main():
    game_score_tracker()

if __name__ == "__main__":
    main()
```
