## Exercise
Write a program that demonstrates the mutation of variables (e.g., changing the value of a list or dictionary).

## Reference Solution
```python
def main():
    # Mutation of Lists
    print("### Mutation of Lists ###")
    my_list = [1, 2, 3, 4, 5]
    print("Original List:", my_list)

    # Append a new element to the list
    my_list.append(6)
    print("List after append:", my_list)

    # Modify an existing element in the list
    my_list[0] = 10
    print("List after modification:", my_list)

    # Mutation of Dictionaries
    print("\n### Mutation of Dictionaries ###")
    my_dict = {"name": "John", "age": 30}
    print("Original Dictionary:", my_dict)

    # Add a new key-value pair to the dictionary
    my_dict["city"] = "New York"
    print("Dictionary after adding a new key-value pair:", my_dict)

    # Modify an existing key-value pair in the dictionary
    my_dict["age"] = 31
    print("Dictionary after modification:", my_dict)

    # Delete a key-value pair from the dictionary
    del my_dict["city"]
    print("Dictionary after deletion:", my_dict)

if __name__ == "__main__":
    main()
```
