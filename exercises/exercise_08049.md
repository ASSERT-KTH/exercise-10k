## Exercise
Write a program that uses an if/else statement to manage personal finances based on user input.

## Reference Solution
```python
def personal_finance_manager():
    """
    Provides financial recommendations based on user input.
    """
    print("Personal Finance Manager")
    print("-------------------------")

    # Get user input
    income = float(input("Enter your monthly income: $"))
    expenses = float(input("Enter your monthly expenses: $"))
    savings = float(input("Enter your current savings: $"))

    # Calculate disposable income
    disposable_income = income - expenses

    # Provide financial recommendations
    if disposable_income > 0:
        print("\nYou have a positive disposable income.")
        if savings < disposable_income * 3:
            print("Consider saving more. Aim to save at least 3 months' worth of expenses.")
        else:
            print("You have a good savings cushion. Consider investing or paying off high-interest debt.")
    elif disposable_income < 0:
        print("\nYou have a negative disposable income.")
        print("Consider reducing expenses or increasing income to avoid debt.")
    else:
        print("\nYou have a break-even disposable income.")
        print("Consider building an emergency fund to avoid financial stress.")

    # Provide additional recommendations
    if income < 50000:
        print("\nConsider taking advantage of tax-advantaged savings options, such as a Roth IRA.")
    elif income > 100000:
        print("\nConsider consulting a financial advisor to optimize your investment strategy.")

def main():
    personal_finance_manager()

if __name__ == "__main__":
    main()
```
