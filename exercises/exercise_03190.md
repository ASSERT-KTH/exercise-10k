## Exercise
Write a program that takes a temperature and a conversion type as input (e.g., Celsius to Fahrenheit). Use if/else statements to perform the correct conversion.

## Reference Solution
```python
def convert_temperature(temp, conversion_type):
    """
    Converts temperature based on the given conversion type.

    Args:
    temp (float): The temperature value to convert.
    conversion_type (str): The type of conversion to perform. Can be 'C2F' (Celsius to Fahrenheit), 'F2C' (Fahrenheit to Celsius), 'K2C' (Kelvin to Celsius), 'C2K' (Celsius to Kelvin), 'F2K' (Fahrenheit to Kelvin), or 'K2F' (Kelvin to Fahrenheit).

    Returns:
    float: The converted temperature value.
    """
    if conversion_type == 'C2F':  # Celsius to Fahrenheit
        return temp * 9/5 + 32
    elif conversion_type == 'F2C':  # Fahrenheit to Celsius
        return (temp - 32) * 5/9
    elif conversion_type == 'K2C':  # Kelvin to Celsius
        return temp - 273.15
    elif conversion_type == 'C2K':  # Celsius to Kelvin
        return temp + 273.15
    elif conversion_type == 'F2K':  # Fahrenheit to Kelvin
        return (temp - 32) * 5/9 + 273.15
    elif conversion_type == 'K2F':  # Kelvin to Fahrenheit
        return (temp - 273.15) * 9/5 + 32
    else:
        raise ValueError("Invalid conversion type. Supported types: C2F, F2C, K2C, C2K, F2K, K2F")

def main():
    temp = float(input("Enter the temperature value: "))
    print("Choose a conversion type:")
    print("1. Celsius to Fahrenheit (C2F)")
    print("2. Fahrenheit to Celsius (F2C)")
    print("3. Kelvin to Celsius (K2C)")
    print("4. Celsius to Kelvin (C2K)")
    print("5. Fahrenheit to Kelvin (F2K)")
    print("6. Kelvin to Fahrenheit (K2F)")
    choice = input("Enter the choice number: ")
    
    conversion_types = {
        '1': 'C2F',
        '2': 'F2C',
        '3': 'K2C',
        '4': 'C2K',
        '5': 'F2K',
        '6': 'K2F'
    }
    
    if choice in conversion_types:
        result = convert_temperature(temp, conversion_types[choice])
        print(f"The converted temperature is {result:.2f} degrees")
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
