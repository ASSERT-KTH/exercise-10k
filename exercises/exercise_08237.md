## Exercise
Write a program that displays customer information using if/else statements

## Reference Solution
```python
class Customer:
    def __init__(self, name, email, subscription_status):
        self.name = name
        self.email = email
        self.subscription_status = subscription_status

    def display_info(self):
        if self.subscription_status == "premium":
            print(f"**Premium Customer**")
            print(f"Name: {self.name}")
            print(f"Email: {self.email}")
            print(f"Subscription Status: {self.subscription_status}")
            print(f"Access to exclusive content: Yes")
            print(f"Priority customer support: Yes")
        elif self.subscription_status == "standard":
            print(f"**Standard Customer**")
            print(f"Name: {self.name}")
            print(f"Email: {self.email}")
            print(f"Subscription Status: {self.subscription_status}")
            print(f"Access to exclusive content: No")
            print(f"Priority customer support: No")
        else:
            print(f"Invalid subscription status: {self.subscription_status}")

# Example usage
if __name__ == "__main__":
    customer1 = Customer("John Doe", "john@example.com", "premium")
    customer2 = Customer("Jane Doe", "jane@example.com", "standard")
    customer3 = Customer("Bob Smith", "bob@example.com", "invalid")

    print("Customer 1:")
    customer1.display_info()
    print("\nCustomer 2:")
    customer2.display_info()
    print("\nCustomer 3:")
    customer3.display_info()
```
