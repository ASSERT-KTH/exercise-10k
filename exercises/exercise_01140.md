## Exercise
Write a program that takes a CSV file as input and analyzes its data (e.g. average, median, mode)

## Reference Solution
```python
import pandas as pd
from scipy import stats

def analyze_data(csv_file):
    """
    Analyze the data in a CSV file.

    Parameters:
    csv_file (str): Path to the CSV file.

    Returns:
    dict: A dictionary containing the analysis results.
    """
    # Load the CSV file
    df = pd.read_csv(csv_file)

    # Calculate the average
    average = df.mean().to_dict()

    # Calculate the median
    median = df.median().to_dict()

    # Calculate the mode
    mode = df.mode().iloc[0].to_dict()

    # Calculate the standard deviation
    std_dev = df.std().to_dict()

    # Create a dictionary with the analysis results
    analysis_results = {
        "average": average,
        "median": median,
        "mode": mode,
        "standard_deviation": std_dev
    }

    return analysis_results

def main():
    # Get the CSV file path from the user
    csv_file = input("Enter the CSV file path: ")

    # Analyze the data
    analysis_results = analyze_data(csv_file)

    # Print the analysis results
    print("Analysis Results:")
    for key, value in analysis_results.items():
        print(f"{key.capitalize().replace('_', ' ')}:")
        for column, result in value.items():
            print(f"  {column}: {result}")

if __name__ == "__main__":
    main()
```
