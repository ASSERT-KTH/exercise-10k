## Exercise
Write a program that uses an if/else statement to recommend products based on user profiles.

## Reference Solution
```python
def get_user_profile():
    """Get user profile information"""
    age = int(input("Enter your age: "))
    income = int(input("Enter your annual income: $"))
    interests = input("Enter your interests (e.g., sports, music, travel): ")
    return age, income, interests

def recommend_product(age, income, interests):
    """Recommend a product based on user profile"""
    if age < 25 and income < 50000:
        # Young adult with low income, recommend affordable electronics
        if "gaming" in interests.lower():
            return "Gaming Console"
        else:
            return "Budget Smartphone"
    elif age < 40 and income < 100000:
        # Adult with medium income, recommend fashion products
        if "fashion" in interests.lower():
            return "Designer Clothing"
        else:
            return "Accessories"
    elif age >= 40 and income >= 100000:
        # Adult with high income, recommend high-end electronics or travel
        if "travel" in interests.lower():
            return "Luxury Vacation Package"
        else:
            return "High-End Laptop"
    else:
        # Default recommendation
        return "Home Decor"

def main():
    """Main program function"""
    age, income, interests = get_user_profile()
    recommended_product = recommend_product(age, income, interests)
    print(f"Based on your profile, we recommend: {recommended_product}")

if __name__ == "__main__":
    main()
```
