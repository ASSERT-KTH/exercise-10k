## Exercise
Develop a program to convert volume from one unit to another. If the input unit is 'liters' and the output unit is 'gallons', multiply the volume by 0.2642.

## Reference Solution
```python
def convert_volume(volume, input_unit, output_unit):
    """
    Converts volume from one unit to another.

    Args:
        volume (float): The volume to convert.
        input_unit (str): The unit of the input volume.
        output_unit (str): The unit to convert to.

    Returns:
        float: The converted volume.
    """

    # Define conversion factors
    conversion_factors = {
        'liters': {'gallons': 0.2642, 'milliliters': 1000, 'pints': 2.1134},
        'gallons': {'liters': 3.7854, 'milliliters': 3785.41, 'pints': 8},
        'milliliters': {'liters': 0.001, 'gallons': 0.0002642, 'pints': 0.0021134},
        'pints': {'liters': 0.4732, 'gallons': 0.125, 'milliliters': 473.18}
    }

    # Check if input and output units are supported
    if input_unit not in conversion_factors or output_unit not in conversion_factors:
        raise ValueError("Unsupported unit")

    # Convert volume
    if input_unit == output_unit:
        return volume
    else:
        return volume * conversion_factors[input_unit][output_unit]


def main():
    print("Volume Conversion Program")
    print("----------------------------")

    volume = float(input("Enter volume: "))
    input_unit = input("Enter input unit (liters, gallons, milliliters, pints): ")
    output_unit = input("Enter output unit (liters, gallons, milliliters, pints): ")

    try:
        converted_volume = convert_volume(volume, input_unit, output_unit)
        print(f"{volume} {input_unit} is equal to {converted_volume} {output_unit}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
