## Exercise
Write a program to calculate the amount of each ingredient needed to make a mixture given the total amount and percentage of each ingredient

## Reference Solution
```python
def calculate_ingredients(total_amount, ingredients):
    """
    Calculate the amount of each ingredient needed to make a mixture.

    Args:
        total_amount (float): The total amount of the mixture.
        ingredients (dict): A dictionary where the keys are the ingredient names and the values are the percentages.

    Returns:
        dict: A dictionary where the keys are the ingredient names and the values are the calculated amounts.
    """
    calculated_ingredients = {}
    for ingredient, percentage in ingredients.items():
        calculated_ingredients[ingredient] = (percentage / 100) * total_amount
    return calculated_ingredients

def main():
    # Example usage
    total_amount = float(input("Enter the total amount of the mixture: "))
    num_ingredients = int(input("Enter the number of ingredients: "))

    ingredients = {}
    for i in range(num_ingredients):
        ingredient_name = input(f"Enter the name of ingredient {i+1}: ")
        percentage = float(input(f"Enter the percentage of {ingredient_name}: "))
        ingredients[ingredient_name] = percentage

    calculated_ingredients = calculate_ingredients(total_amount, ingredients)

    print("\nCalculated Ingredients:")
    for ingredient, amount in calculated_ingredients.items():
        print(f"{ingredient}: {amount:.2f}")

if __name__ == "__main__":
    main()
```
