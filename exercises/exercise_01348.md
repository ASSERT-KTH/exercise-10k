## Exercise
Write a program to perform basic dictionary operations (add, remove, update, retrieve) on a given dictionary

## Reference Solution
```python
class DictionaryOperations:
    def __init__(self, dictionary=None):
        """
        Initialize the dictionary.

        Args:
            dictionary (dict, optional): The dictionary to operate on. Defaults to None.
        """
        self.dictionary = dictionary if dictionary else {}

    def add(self, key, value):
        """
        Add a new key-value pair to the dictionary.

        Args:
            key (any): The key to add.
            value (any): The value to add.
        """
        self.dictionary[key] = value

    def remove(self, key):
        """
        Remove a key-value pair from the dictionary.

        Args:
            key (any): The key to remove.

        Raises:
            KeyError: If the key does not exist in the dictionary.
        """
        if key in self.dictionary:
            del self.dictionary[key]
        else:
            raise KeyError(f"Key '{key}' not found in the dictionary.")

    def update(self, key, value):
        """
        Update the value of an existing key in the dictionary.

        Args:
            key (any): The key to update.
            value (any): The new value.

        Raises:
            KeyError: If the key does not exist in the dictionary.
        """
        if key in self.dictionary:
            self.dictionary[key] = value
        else:
            raise KeyError(f"Key '{key}' not found in the dictionary.")

    def retrieve(self, key):
        """
        Retrieve the value of a key from the dictionary.

        Args:
            key (any): The key to retrieve.

        Returns:
            any: The value associated with the key.

        Raises:
            KeyError: If the key does not exist in the dictionary.
        """
        if key in self.dictionary:
            return self.dictionary[key]
        else:
            raise KeyError(f"Key '{key}' not found in the dictionary.")

    def display(self):
        """
        Display the current state of the dictionary.
        """
        print(self.dictionary)


# Example usage
if __name__ == "__main__":
    dictionary = DictionaryOperations()

    # Add key-value pairs
    dictionary.add("name", "John Doe")
    dictionary.add("age", 30)
    dictionary.display()  # Output: {'name': 'John Doe', 'age': 30}

    # Update a key
    dictionary.update("age", 31)
    dictionary.display()  # Output: {'name': 'John Doe', 'age': 31}

    # Retrieve a key
    print(dictionary.retrieve("name"))  # Output: John Doe

    # Remove a key
    dictionary.remove("age")
    dictionary.display()  # Output: {'name': 'John Doe'}

    # Attempt to retrieve a non-existent key
    try:
        dictionary.retrieve("non_existent_key")
    except KeyError as e:
        print(e)  # Output: Key 'non_existent_key' not found in the dictionary.
```
