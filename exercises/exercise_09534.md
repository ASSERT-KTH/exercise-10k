## Exercise
Create a program that simulates a phonebook using a dictionary. The program should allow users to add, remove, and search for contacts. The dictionary keys should be the contact names and the values should be the contact phone numbers.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        """
        Initialize an empty phonebook.
        """
        self.phonebook = {}

    def add_contact(self, name, phone_number):
        """
        Add a new contact to the phonebook.

        Args:
            name (str): The name of the contact.
            phone_number (str): The phone number of the contact.
        """
        if name in self.phonebook:
            print(f"A contact with the name '{name}' already exists.")
        else:
            self.phonebook[name] = phone_number
            print(f"Contact '{name}' added successfully.")

    def remove_contact(self, name):
        """
        Remove a contact from the phonebook.

        Args:
            name (str): The name of the contact to remove.
        """
        if name in self.phonebook:
            del self.phonebook[name]
            print(f"Contact '{name}' removed successfully.")
        else:
            print(f"No contact found with the name '{name}'.")

    def search_contact(self, name):
        """
        Search for a contact in the phonebook.

        Args:
            name (str): The name of the contact to search for.
        """
        if name in self.phonebook:
            print(f"The phone number of '{name}' is {self.phonebook[name]}.")
        else:
            print(f"No contact found with the name '{name}'.")

    def display_contacts(self):
        """
        Display all contacts in the phonebook.
        """
        if not self.phonebook:
            print("The phonebook is empty.")
        else:
            print("Contacts:")
            for name, phone_number in self.phonebook.items():
                print(f"{name}: {phone_number}")


def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook Menu:")
        print("1. Add contact")
        print("2. Remove contact")
        print("3. Search contact")
        print("4. Display contacts")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter the contact name: ")
            phone_number = input("Enter the contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter the contact name to remove: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter the contact name to search: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Exiting the phonebook.")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
