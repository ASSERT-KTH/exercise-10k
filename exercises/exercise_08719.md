## Exercise
Ask the user for a list of numbers and print out the sorted list.

## Reference Solution
```python
def get_numbers_from_user():
    """Gets a list of numbers from the user."""
    user_input = input("Please enter a list of numbers separated by spaces: ")
    try:
        numbers = [float(num) for num in user_input.split()]
        return numbers
    except ValueError:
        print("Invalid input. Please enter numbers only.")
        return get_numbers_from_user()

def sort_numbers(numbers):
    """Sorts a list of numbers in ascending order."""
    return sorted(numbers)

def main():
    numbers = get_numbers_from_user()
    sorted_numbers = sort_numbers(numbers)
    print("Sorted list of numbers:")
    print(sorted_numbers)

if __name__ == "__main__":
    main()
```
