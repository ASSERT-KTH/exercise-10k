## Exercise
Write a program that transforms a given word into another word by changing one character at a time, with each resulting word being a valid word.

## Reference Solution
```python
import collections
import nltk

# Download the word list if not already downloaded
nltk.download('words')

def word_ladder(start_word, end_word):
    """
    Transforms the start_word into the end_word by changing one character at a time,
    with each resulting word being a valid word.

    Args:
        start_word (str): The starting word.
        end_word (str): The ending word.

    Returns:
        list: A list of words representing the transformation from start_word to end_word.
    """

    # Get the list of valid words
    word_list = set(w.lower() for w in nltk.corpus.words.words())

    # Create a queue for BFS, containing the starting word
    queue = collections.deque([[start_word]])

    # Create a set to store visited words
    visited = set()

    while queue:
        # Get the current path
        path = queue.popleft()
        word = path[-1]

        # If the current word is the end_word, return the path
        if word == end_word:
            return path

        # Mark the current word as visited
        visited.add(word)

        # Generate all possible words by changing one character
        for i in range(len(word)):
            for c in 'abcdefghijklmnopqrstuvwxyz':
                next_word = word[:i] + c + word[i+1:]

                # Check if the next word is valid and not visited
                if next_word in word_list and next_word not in visited:
                    # Create a new path and add it to the queue
                    new_path = list(path)
                    new_path.append(next_word)
                    queue.append(new_path)

    # If no path is found, return None
    return None

def main():
    start_word = "cold"
    end_word = "warm"
    result = word_ladder(start_word, end_word)

    if result:
        print("Word Ladder from '{}' to '{}':".format(start_word, end_word))
        print(" -> ".join(result))
    else:
        print("No word ladder found from '{}' to '{}'".format(start_word, end_word))

if __name__ == "__main__":
    main()
```
