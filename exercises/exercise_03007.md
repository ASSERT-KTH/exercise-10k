## Exercise
Write a program to search for a substring within a given string.

## Reference Solution
```python
def search_substring(main_string, sub_string):
    """
    Searches for a substring within a given string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the first occurrence of the substring. Returns -1 if the substring is not found.
    """
    try:
        # Use the index method to find the substring
        index = main_string.index(sub_string)
        return index
    except ValueError:
        # If the substring is not found, return -1
        return -1

def main():
    # Example usage
    main_string = "Hello, World!"
    sub_string = "World"
    index = search_substring(main_string, sub_string)

    if index != -1:
        print(f"'{sub_string}' found at index {index} in '{main_string}'")
    else:
        print(f"'{sub_string}' not found in '{main_string}'")

if __name__ == "__main__":
    main()
```
