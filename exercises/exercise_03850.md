## Exercise
Write a program that uses a for loop to print each element in a linked list.

## Reference Solution
```python
class Node:
    """Represents an element in the linked list."""
    def __init__(self, data=None):
        self.data = data
        self.next = None

class LinkedList:
    """Manages the linked list."""
    def __init__(self):
        self.head = None

    def append(self, data):
        """Adds a new node to the end of the list."""
        if not self.head:
            self.head = Node(data)
        else:
            current = self.head
            while current.next:
                current = current.next
            current.next = Node(data)

    def __iter__(self):
        """Allows iteration over the linked list."""
        current = self.head
        while current:
            yield current.data
            current = current.next

    def print_list(self):
        """Prints all elements in the linked list."""
        for data in self:
            print(data)

# Example usage:
if __name__ == "__main__":
    linked_list = LinkedList()
    linked_list.append(1)
    linked_list.append(2)
    linked_list.append(3)
    linked_list.append(4)
    linked_list.append(5)
    
    print("Linked List Elements:")
    linked_list.print_list()
```
