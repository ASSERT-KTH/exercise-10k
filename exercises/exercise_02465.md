## Exercise
Create a function that finds the index of a given substring within a larger string.

## Reference Solution
```python
def find_substring(main_string, sub_string):
    """
    Finds the index of a given substring within a larger string.

    Args:
        main_string (str): The larger string to search in.
        sub_string (str): The substring to search for.

    Returns:
        int: The index of the substring if found, -1 otherwise.
    """
    try:
        # Attempt to find the substring
        index = main_string.find(sub_string)
        return index
    except Exception as e:
        # Handle any exceptions that may occur
        print(f"An error occurred: {e}")
        return None


# Example usage
if __name__ == "__main__":
    main_str = "Hello, world!"
    sub_str = "world"
    index = find_substring(main_str, sub_str)
    
    if index != -1:
        print(f"The substring '{sub_str}' is found at index {index}.")
    else:
        print(f"The substring '{sub_str}' is not found in the main string.")
```
