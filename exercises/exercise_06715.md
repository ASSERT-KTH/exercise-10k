## Exercise
Write a program that reads input from a file and processes it

## Reference Solution
```python
def process_file(input_file):
    """
    Reads input from a file, processes it, and prints the results.

    Args:
        input_file (str): Path to the input file
    """
    try:
        with open(input_file, 'r') as file:
            # Read the file line by line
            lines = file.readlines()

            # Process each line (e.g., strip whitespace, convert to uppercase)
            processed_lines = [line.strip().upper() for line in lines]

            # Print the processed lines
            for line in processed_lines:
                print(line)

    except FileNotFoundError:
        print(f"Error: File '{input_file}' not found.")
    except Exception as e:
        print(f"An error occurred: {e}")

def main():
    input_file = 'input.txt'  # Replace with your input file path
    process_file(input_file)

if __name__ == '__main__':
    main()
```
