## Exercise
Write a function to check if a dictionary is a subset of another dictionary

## Reference Solution
```python
def is_dict_subset(subset_dict, superset_dict):
    """
    Checks if a dictionary is a subset of another dictionary.

    Args:
        subset_dict (dict): The dictionary to check.
        superset_dict (dict): The dictionary to check against.

    Returns:
        bool: True if subset_dict is a subset of superset_dict, False otherwise.
    """
    return all(key in superset_dict and superset_dict[key] == value for key, value in subset_dict.items())

# Example usage:
if __name__ == "__main__":
    superset_dict = {"a": 1, "b": 2, "c": 3}
    subset_dict = {"a": 1, "b": 2}
    print(is_dict_subset(subset_dict, superset_dict))  # Output: True

    superset_dict = {"a": 1, "b": 2, "c": 3}
    subset_dict = {"a": 1, "b": 3}
    print(is_dict_subset(subset_dict, superset_dict))  # Output: False
```
