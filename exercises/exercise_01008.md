## Exercise
Write a program that takes a string as input and returns all possible permutations of the string.

## Reference Solution
```python
def get_permutations(input_str):
    """
    Generates all possible permutations of a given input string.

    Args:
        input_str (str): The input string.

    Returns:
        list: A list of all possible permutations of the input string.
    """
    # Base case: if the string is empty or has only one character
    if len(input_str) <= 1:
        return [input_str]

    # Initialize an empty list to store the permutations
    permutations = []

    # Iterate over each character in the string
    for i, char in enumerate(input_str):
        # Get the remaining characters
        remaining_str = input_str[:i] + input_str[i + 1:]

        # Generate permutations for the remaining characters
        for perm in get_permutations(remaining_str):
            # Append the current character to each permutation
            permutations.append(char + perm)

    return permutations


def main():
    # Example usage
    input_str = "abc"
    permutations = get_permutations(input_str)
    print("Permutations of '{}'".format(input_str))
    for i, perm in enumerate(permutations):
        print("{}. {}".format(i + 1, perm))


if __name__ == "__main__":
    main()
```
