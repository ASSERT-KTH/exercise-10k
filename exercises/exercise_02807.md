## Exercise
Write a program to ask the user for a string and count the number of occurrences of each character.

## Reference Solution
```python
def count_character_frequencies(input_string):
    """
    Counts the frequency of each character in a given string.

    Args:
        input_string (str): The input string to count character frequencies from.

    Returns:
        dict: A dictionary where keys are characters and values are their frequencies.
    """
    frequency_dict = {}
    for char in input_string:
        if char in frequency_dict:
            frequency_dict[char] += 1
        else:
            frequency_dict[char] = 1
    return frequency_dict


def main():
    # Ask the user for a string
    input_string = input("Please enter a string: ")

    # Count character frequencies
    frequency_dict = count_character_frequencies(input_string)

    # Display the results
    print("Character Frequencies:")
    for char, frequency in frequency_dict.items():
        print(f"{char}: {frequency}")


if __name__ == "__main__":
    main()
```
