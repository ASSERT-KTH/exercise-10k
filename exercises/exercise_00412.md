## Exercise
Write a program to find all subsets of an array.

## Reference Solution
```python
def generate_subsets(array):
    """
    Generate all possible subsets of an array.

    Args:
    array (list): The input array.

    Returns:
    list: A list of lists, where each sublist is a subset of the input array.
    """
    # Initialize an empty list to store the subsets
    subsets = [[]]

    # Iterate over each element in the input array
    for element in array:
        # Calculate the number of existing subsets
        num_subsets = len(subsets)

        # Iterate over the existing subsets
        for i in range(num_subsets):
            # Create a new subset by appending the current element to an existing subset
            new_subset = subsets[i] + [element]

            # Add the new subset to the list of subsets
            subsets.append(new_subset)

    return subsets

def main():
    # Example usage
    array = [1, 2, 3]
    subsets = generate_subsets(array)

    print("Subsets of the array:")
    for i, subset in enumerate(subsets):
        print(f"Subset {i+1}: {subset}")

if __name__ == "__main__":
    main()
```
