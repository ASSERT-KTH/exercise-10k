## Exercise
Create a program that stores and retrieves movie information using conditional statements, allowing users to search for movies based on title, genre, and release year.

## Reference Solution
```python
# Movie Database Program

# Define a list to store movie information
movies = [
    {"title": "The Shawshank Redemption", "genre": "Drama", "year": 1994},
    {"title": "The Godfather", "genre": "Crime", "year": 1972},
    {"title": "The Dark Knight", "genre": "Action", "year": 2008},
    {"title": "12 Angry Men", "genre": "Drama", "year": 1957},
    {"title": "Schindler's List", "genre": "History", "year": 1993},
    {"title": "The Lord of the Rings: The Return of the King", "genre": "Adventure", "year": 2003},
    {"title": "Pulp Fiction", "genre": "Crime", "year": 1994},
    {"title": "The Lord of the Rings: The Fellowship of the Ring", "genre": "Adventure", "year": 2001},
    {"title": "Forrest Gump", "genre": "Drama", "year": 1994},
    {"title": "Inception", "genre": "Action", "year": 2010}
]

# Function to search for movies by title
def search_by_title():
    title = input("Enter movie title: ")
    found_movies = [movie for movie in movies if title.lower() in movie["title"].lower()]
    if found_movies:
        print("Search results:")
        for movie in found_movies:
            print(f"Title: {movie['title']}, Genre: {movie['genre']}, Year: {movie['year']}")
    else:
        print("No movies found with that title.")

# Function to search for movies by genre
def search_by_genre():
    genre = input("Enter movie genre: ")
    found_movies = [movie for movie in movies if genre.lower() in movie["genre"].lower()]
    if found_movies:
        print("Search results:")
        for movie in found_movies:
            print(f"Title: {movie['title']}, Genre: {movie['genre']}, Year: {movie['year']}")
    else:
        print("No movies found with that genre.")

# Function to search for movies by release year
def search_by_year():
    year = int(input("Enter movie release year: "))
    found_movies = [movie for movie in movies if movie["year"] == year]
    if found_movies:
        print("Search results:")
        for movie in found_movies:
            print(f"Title: {movie['title']}, Genre: {movie['genre']}, Year: {movie['year']}")
    else:
        print("No movies found with that release year.")

# Main function
def main():
    while True:
        print("\nMovie Database Program")
        print("1. Search for movies by title")
        print("2. Search for movies by genre")
        print("3. Search for movies by release year")
        print("4. Exit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            search_by_title()
        elif choice == "2":
            search_by_genre()
        elif choice == "3":
            search_by_year()
        elif choice == "4":
            print("Exiting program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

# Run the main function
if __name__ == "__main__":
    main()
```
