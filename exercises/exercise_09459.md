## Exercise
Write a program that generates a random number within a given range.

## Reference Solution
```python
import random

def generate_random_number(min_value, max_value):
    """
    Generates a random number within a given range.

    Args:
        min_value (int): The minimum value of the range (inclusive).
        max_value (int): The maximum value of the range (inclusive).

    Returns:
        int: A random number within the given range.
    """
    if min_value > max_value:
        raise ValueError("Min value cannot be greater than max value")

    return random.randint(min_value, max_value)


def main():
    # Example usage
    min_value = 1
    max_value = 100
    random_number = generate_random_number(min_value, max_value)
    print(f"Random number within the range [{min_value}, {max_value}]: {random_number}")


if __name__ == "__main__":
    main()
```
