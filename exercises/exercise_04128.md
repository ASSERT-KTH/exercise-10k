## Exercise
Create a set to store unique student IDs, then print out the number of unique IDs

## Reference Solution
```python
def count_unique_student_ids(student_ids):
    """
    Creates a set to store unique student IDs and returns the number of unique IDs.

    Args:
        student_ids (list): A list of student IDs.

    Returns:
        int: The number of unique student IDs.
    """
    # Create a set from the list of student IDs to store unique IDs
    unique_ids = set(student_ids)
    
    # Return the number of unique IDs
    return len(unique_ids)


def main():
    # Example list of student IDs
    student_ids = [1, 2, 2, 3, 4, 4, 5, 6, 6, 7]
    
    # Count the number of unique student IDs
    num_unique_ids = count_unique_student_ids(student_ids)
    
    # Print the number of unique IDs
    print("Number of unique student IDs:", num_unique_ids)


if __name__ == "__main__":
    main()
```
