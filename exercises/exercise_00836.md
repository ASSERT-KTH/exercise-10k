## Exercise
Given a string, find the longest increasing subsequence of characters and return it.

## Reference Solution

def longest_increasing_subsequence(s):
    """
    Find the longest increasing subsequence of characters in a string.

    Args:
        s (str): The input string.

    Returns:
        str: The longest increasing subsequence of characters.
    """
    if not s:
        return ""

    # Initialize arrays to store the length of the longest increasing subsequence
    # and the previous character in the subsequence
    dp = [1] * len(s)
    prev = [-1] * len(s)

    # Initialize the maximum length and the index of the last character in the subsequence
    max_length = 1
    max_index = 0

    # Compute the length of the longest increasing subsequence ending at each position
    for i in range(1, len(s)):
        for j in range(i):
            if s[i] > s[j] and dp[i] < dp[j] + 1:
                dp[i] = dp[j] + 1
                prev[i] = j
        if dp[i] > max_length:
            max_length = dp[i]
            max_index = i

    # Reconstruct the longest increasing subsequence
    subsequence = []
    while max_index != -1:
        subsequence.append(s[max_index])
        max_index = prev[max_index]

    # Return the longest increasing subsequence in the correct order
    return "".join(reversed(subsequence))


# Example usage
s = "abcde"
print(longest_increasing_subsequence(s))  # Output: "abcde"

s = "zyxwv"
print(longest_increasing_subsequence(s))  # Output: "z"

s = "abcdefg"
print(longest_increasing_subsequence(s))  # Output: "abcdefg"
