## Exercise
Write a function to create a dictionary from a string where each word is a key and its frequency is the value

## Reference Solution
```python
def word_frequency(string):
    """
    Create a dictionary from a string where each word is a key and its frequency is the value.

    Args:
        string (str): The input string.

    Returns:
        dict: A dictionary where each key is a unique word and its corresponding value is the frequency.
    """
    # Convert the string to lowercase and split it into words
    words = string.lower().split()

    # Initialize an empty dictionary to store the word frequencies
    frequency_dict = {}

    # Iterate over each word in the list of words
    for word in words:
        # Remove any punctuation from the word
        word = ''.join(e for e in word if e.isalnum())
        
        # If the word is already in the dictionary, increment its count
        if word in frequency_dict:
            frequency_dict[word] += 1
        # Otherwise, add the word to the dictionary with a count of 1
        else:
            frequency_dict[word] = 1

    return frequency_dict

# Example usage
def main():
    string = "This is a test string. This string is just a test."
    print(word_frequency(string))

if __name__ == "__main__":
    main()
```
