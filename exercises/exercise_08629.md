## Exercise
Write a program that takes a string as input and returns a list of all subsequences.

## Reference Solution
```python
def generate_subsequences(input_string):
    """
    Generates all possible subsequences from a given input string.

    Args:
        input_string (str): The input string.

    Returns:
        list: A list of all subsequences.
    """
    # Base case: if the input string is empty, return a list with an empty string
    if not input_string:
        return [""]

    # Recursive case: get all subsequences of the input string without the first character
    subsequences = generate_subsequences(input_string[1:])

    # Add the first character to each subsequence
    new_subsequences = [input_string[0] + subsequence for subsequence in subsequences]

    # Combine the new subsequences with the old ones
    return subsequences + new_subsequences


def main():
    input_string = "abc"
    subsequences = generate_subsequences(input_string)

    print("Input String:", input_string)
    print("Subsequences:")
    for i, subsequence in enumerate(subsequences):
        print(f"{i+1}. {subsequence}")


if __name__ == "__main__":
    main()
```
