## Exercise
Write a program to validate if a given dictionary contains all required keys.

## Reference Solution
```python
def validate_dict_keys(dictionary, required_keys):
    """
    Validate if a dictionary contains all required keys.

    Args:
        dictionary (dict): The dictionary to be validated.
        required_keys (list): A list of required keys.

    Returns:
        bool: True if all required keys are present, False otherwise.
    """
    for key in required_keys:
        if key not in dictionary:
            return False
    return True

def main():
    # Example usage
    dictionary = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }
    required_keys = ["name", "age", "city"]

    if validate_dict_keys(dictionary, required_keys):
        print("Dictionary contains all required keys.")
    else:
        print("Dictionary is missing some required keys.")

if __name__ == "__main__":
    main()
```
