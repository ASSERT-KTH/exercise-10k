## Exercise
Create a program that takes in a list of variables and their corresponding data types, then validates if the given values match their respective data types. The program should handle basic data types such as integers, floats, and strings.

## Reference Solution
```python
def validate_data_types(variables: dict) -> dict:
    """
    This function takes in a dictionary of variables and their corresponding data types,
    then validates if the given values match their respective data types.

    Args:
        variables (dict): A dictionary where keys are variable names and values are tuples of (value, data_type).

    Returns:
        dict: A dictionary where keys are variable names and values are boolean indicating whether the data type is valid.
    """

    # Initialize an empty dictionary to store the validation results
    validation_results = {}

    # Iterate over each variable in the input dictionary
    for variable, (value, data_type) in variables.items():
        # Initialize a flag to assume the data type is valid
        is_valid = True

        # Check the data type of the value and compare it with the given data type
        if data_type == 'int':
            # If the data type is int, check if the value can be converted to an integer
            try:
                int(value)
            except ValueError:
                # If the value cannot be converted to an integer, set the flag to False
                is_valid = False
        elif data_type == 'float':
            # If the data type is float, check if the value can be converted to a float
            try:
                float(value)
            except ValueError:
                # If the value cannot be converted to a float, set the flag to False
                is_valid = False
        elif data_type == 'str':
            # If the data type is str, check if the value is a string
            if not isinstance(value, str):
                # If the value is not a string, set the flag to False
                is_valid = False
        else:
            # If the data type is not supported, set the flag to False
            is_valid = False

        # Store the validation result in the dictionary
        validation_results[variable] = is_valid

    # Return the dictionary of validation results
    return validation_results


# Example usage:
variables = {
    'var1': ('10', 'int'),
    'var2': ('20.5', 'float'),
    'var3': ('hello', 'str'),
    'var4': ('30', 'float'),  # This should fail validation
    'var5': ('world', 'int')  # This should fail validation
}

validation_results = validate_data_types(variables)

for variable, is_valid in validation_results.items():
    print(f"Variable {variable} is valid: {is_valid}")
```
