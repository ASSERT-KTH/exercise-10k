## Exercise
Create a program that simulates a coffee shop. The program should take in the order and output the updated order based on the following conditions: if the user orders a coffee, add the coffee to the order, if the user orders a pastry, add the pastry to the order, if the user checks the order, output the current order.

## Reference Solution
```python
class CoffeeShop:
    def __init__(self):
        self.order = []

    def add_coffee(self, coffee_type):
        """Add a coffee to the order."""
        self.order.append(f"Coffee: {coffee_type}")

    def add_pastry(self, pastry_type):
        """Add a pastry to the order."""
        self.order.append(f"Pastry: {pastry_type}")

    def check_order(self):
        """Output the current order."""
        if not self.order:
            print("Your order is empty.")
        else:
            print("Your current order is:")
            for i, item in enumerate(self.order, start=1):
                print(f"{i}. {item}")


def main():
    coffee_shop = CoffeeShop()

    while True:
        print("\nCoffee Shop Menu:")
        print("1. Order a coffee")
        print("2. Order a pastry")
        print("3. Check your order")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            coffee_type = input("Enter the type of coffee (e.g., latte, cappuccino, espresso): ")
            coffee_shop.add_coffee(coffee_type)
        elif choice == "2":
            pastry_type = input("Enter the type of pastry (e.g., croissant, muffin, cake): ")
            coffee_shop.add_pastry(pastry_type)
        elif choice == "3":
            coffee_shop.check_order()
        elif choice == "4":
            print("Thank you for visiting our coffee shop!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
