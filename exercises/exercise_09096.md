## Exercise
Create a program that takes a sequence of numbers as input and uses a loop to identify the longest increasing or decreasing subsequence. The program should output the type of subsequence (increasing or decreasing) and its length.

## Reference Solution

def longest_subsequence(sequence):
    """
    This function identifies the longest increasing or decreasing subsequence in a given sequence of numbers.

    Args:
    sequence (list): A list of integers.

    Returns:
    tuple: A tuple containing the type of the longest subsequence and its length.
    """

    # Check if the input sequence is empty
    if not sequence:
        return None, 0

    # Initialize variables to store the lengths of the longest increasing and decreasing subsequences
    longest_increasing = [1] * len(sequence)
    longest_decreasing = [1] * len(sequence)

    # Iterate over the sequence to fill the longest_increasing and longest_decreasing lists
    for i in range(1, len(sequence)):
        for j in range(i):
            # If the current number is greater than the previous number, update the longest increasing subsequence
            if sequence[i] > sequence[j]:
                longest_increasing[i] = max(longest_increasing[i], longest_increasing[j] + 1)
            # If the current number is less than the previous number, update the longest decreasing subsequence
            elif sequence[i] < sequence[j]:
                longest_decreasing[i] = max(longest_decreasing[i], longest_decreasing[j] + 1)

    # Find the maximum length of the longest increasing and decreasing subsequences
    max_increasing = max(longest_increasing)
    max_decreasing = max(longest_decreasing)

    # Determine the type of the longest subsequence
    if max_increasing >= max_decreasing:
        return "increasing", max_increasing
    else:
        return "decreasing", max_decreasing


# Example usage
sequence = [1, 2, 3, 4, 5, 2, 1]
subsequence_type, length = longest_subsequence(sequence)
print(f"The longest subsequence is {subsequence_type} with a length of {length}.")
