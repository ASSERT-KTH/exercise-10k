## Exercise
Write a program that checks whether a string is a palindrome.

## Reference Solution
```python
def is_palindrome(s: str) -> bool:
    """
    Checks whether a string is a palindrome.

    Args:
        s (str): The input string.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    s = ''.join(c for c in s if c.isalnum()).lower()  # remove non-alphanumeric characters and convert to lowercase
    return s == s[::-1]  # compare the string with its reverse

def main():
    # example usage
    strings = ["radar", "hello", "A man, a plan, a canal: Panama"]
    for s in strings:
        print(f"'{s}': {is_palindrome(s)}")

if __name__ == "__main__":
    main()
```
