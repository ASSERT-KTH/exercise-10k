## Exercise
Write a function that implements a Caesar cipher, shifting each character in the input string by a specified number of positions.

## Reference Solution
```python
def caesar_cipher(input_string: str, shift: int, direction: str = 'forward') -> str:
    """
    Shifts each character in the input string by a specified number of positions.

    Args:
        input_string (str): The string to be encrypted or decrypted.
        shift (int): The number of positions to shift each character.
        direction (str, optional): The direction of the shift. Defaults to 'forward'.

    Returns:
        str: The encrypted or decrypted string.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in input_string:
        if char.isalpha():
            # Determine the shift direction
            if direction == 'backward':
                shift_amount = -shift
            else:
                shift_amount = shift

            # Find the position of the character in the alphabet
            position = alphabet.index(char.lower())

            # Shift the character and wrap around the alphabet if necessary
            new_position = (position + shift_amount) % len(alphabet)

            # Preserve the original case
            if char.isupper():
                result += alphabet[new_position].upper()
            else:
                result += alphabet[new_position]
        else:
            # Non-alphabet characters are not shifted
            result += char

    return result

# Example usage:
if __name__ == "__main__":
    input_string = "Hello, World!"
    shift = 3

    encrypted_string = caesar_cipher(input_string, shift)
    decrypted_string = caesar_cipher(encrypted_string, shift, direction='backward')

    print(f"Input String: {input_string}")
    print(f"Encrypted String: {encrypted_string}")
    print(f"Decrypted String: {decrypted_string}")
```
