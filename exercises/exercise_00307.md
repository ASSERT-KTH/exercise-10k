## Exercise
Write a function that searches for a specific element in a given array and returns its index.

## Reference Solution
```python
def search_element(array, target):
    """
    Searches for a specific element in a given array and returns its index.

    Args:
        array (list): The array to search in.
        target: The element to search for.

    Returns:
        int: The index of the target element if found, -1 otherwise.
    """
    try:
        # Use the index method to find the index of the target element
        index = array.index(target)
        return index
    except ValueError:
        # If the target element is not found, return -1
        return -1

# Example usage
if __name__ == "__main__":
    array = [1, 2, 3, 4, 5]
    target = 3
    index = search_element(array, target)
    if index != -1:
        print(f"Element {target} found at index {index}.")
    else:
        print(f"Element {target} not found in the array.")
```
