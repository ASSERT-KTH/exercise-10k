## Exercise
Write a program to calculate the compound interest on an investment

## Reference Solution
```python
def calculate_compound_interest(principal, interest_rate, time_period, compounding_frequency):
    """
    Calculate compound interest on an investment.

    Args:
        principal (float): The initial investment amount.
        interest_rate (float): The annual interest rate as a decimal.
        time_period (int): The time period of the investment in years.
        compounding_frequency (int): The number of times interest is compounded per year.

    Returns:
        float: The total amount after the given time period, including interest.
    """
    amount = principal * (1 + interest_rate / compounding_frequency) ** (compounding_frequency * time_period)
    return amount

def main():
    print("Compound Interest Calculator")
    print("--------------------------------")

    principal = float(input("Enter the principal amount: $"))
    interest_rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time_period = int(input("Enter the time period (in years): "))
    compounding_frequency = int(input("Enter the compounding frequency (per year): "))

    amount = calculate_compound_interest(principal, interest_rate, time_period, compounding_frequency)
    interest = amount - principal

    print(f"\nTotal Amount: ${amount:.2f}")
    print(f"Compound Interest: ${interest:.2f}")
    print(f"Interest Rate: {interest_rate * 100}%")
    print(f"Time Period: {time_period} years")
    print(f"Compounding Frequency: {compounding_frequency} times per year")

if __name__ == "__main__":
    main()
```
