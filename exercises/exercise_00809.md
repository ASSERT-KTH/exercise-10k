## Exercise
Understand the concept of variable scope and print the value of a variable from a different scope.

## Reference Solution
```python
# Global scope
x = 10

def outer_function():
    # Enclosing scope
    y = 20

    def inner_function():
        # Local scope
        z = 30
        print("Accessing variables from different scopes:")
        print("Global scope: x =", x)
        print("Enclosing scope: y =", y)
        print("Local scope: z =", z)

    inner_function()

outer_function()
```

```python
x = 10

def modify_global():
    x = 20  # Creates a new local variable x

modify_global()
print(x)  # Still prints 10
```

```python
x = 10

def modify_global():
    global x
    x = 20

modify_global()
print(x)  # Prints 20
```

```python
def outer_function():
    y = 20

    def inner_function():
        nonlocal y
        y = 30

    inner_function()
    print(y)  # Prints 30

outer_function()
```
