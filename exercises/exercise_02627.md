## Exercise
Write a program that takes a movie title and rating as input and generates a review.

## Reference Solution
```python
import random

def generate_review(title, rating):
    """
    Generate a movie review based on the title and rating.

    Args:
        title (str): The title of the movie.
        rating (int): The rating of the movie (1-5).

    Returns:
        str: A generated movie review.
    """

    # Define a list of possible review phrases for each rating
    reviews = {
        1: [
            "was a major disappointment.",
            "failed to impress.",
            "was a waste of time.",
            "had too many plot holes.",
            "lacked any real substance."
        ],
        2: [
            "was just okay.",
            "had some potential, but fell short.",
            "was a bit of a snooze fest.",
            "had too much repetition.",
            "needed more character development."
        ],
        3: [
            "was a solid effort.",
            "had some great moments.",
            "was entertaining, but not memorable.",
            "had a interesting plot, but poor execution.",
            "was a fun watch, but lacked depth."
        ],
        4: [
            "was a real treat.",
            "exceeded my expectations.",
            "had a great cast and crew.",
            "was a must-see for fans of the genre.",
            "was a well-crafted film."
        ],
        5: [
            "was a masterpiece.",
            "was a game-changer.",
            "was a work of art.",
            "was a cinematic experience like no other.",
            "was a film that will be remembered for years to come."
        ]
    }

    # Check if the rating is within the valid range
    if rating < 1 or rating > 5:
        return "Invalid rating. Please enter a rating between 1 and 5."

    # Select a random review phrase based on the rating
    review_phrase = random.choice(reviews[rating])

    # Generate the review
    review = f"\"{title}\" {review_phrase}"

    return review

def main():
    # Get the movie title and rating from the user
    title = input("Enter the movie title: ")
    rating = int(input("Enter the movie rating (1-5): "))

    # Generate and print the review
    print(generate_review(title, rating))

if __name__ == "__main__":
    main()
```
