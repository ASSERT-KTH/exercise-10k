## Exercise
Write a program that checks if a given list of numbers is sorted in ascending order.

## Reference Solution
```python
def is_sorted(num_list):
    """
    Checks if a given list of numbers is sorted in ascending order.

    Args:
        num_list (list): A list of numbers.

    Returns:
        bool: True if the list is sorted, False otherwise.
    """
    return num_list == sorted(num_list)


def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    print(f"Is {numbers} sorted? {is_sorted(numbers)}")

    numbers = [5, 4, 3, 2, 1]
    print(f"Is {numbers} sorted? {is_sorted(numbers)}")


if __name__ == "__main__":
    main()
```

```python
def is_sorted(num_list):
    """
    Checks if a given list of numbers is sorted in ascending order.

    Args:
        num_list (list): A list of numbers.

    Returns:
        bool: True if the list is sorted, False otherwise.
    """
    return all(num_list[i] <= num_list[i + 1] for i in range(len(num_list) - 1))
```
