## Exercise
Create a program that simulates a simple quiz show and asks the user a series of questions using if statements to determine the winner

## Reference Solution
```python
# Define the quiz questions and answers
quiz_questions = {
    "What is the capital of France?": "Paris",
    "What is the largest planet in our solar system?": "Jupiter",
    "Who painted the Starry Night?": "Vincent van Gogh",
    "What is the chemical symbol for gold?": "Au",
    "Who wrote Romeo and Juliet?": "William Shakespeare"
}

def quiz_game():
    # Initialize the score
    score = 0

    # Ask the quiz questions
    for question, answer in quiz_questions.items():
        user_answer = input(question + " ")
        if user_answer.lower() == answer.lower():
            print("Correct answer!\n")
            score += 1
        else:
            print(f"Sorry, the correct answer is {answer}.\n")

    # Determine the winner
    if score >= 3:
        print(f"Congratulations, you won the game with a score of {score} out of {len(quiz_questions)}!")
    else:
        print(f"Sorry, you lost the game with a score of {score} out of {len(quiz_questions)}. Better luck next time!")

# Run the quiz game
quiz_game()
```
