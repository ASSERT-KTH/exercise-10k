## Exercise
Write a program to calculate the median of a set of numbers.

## Reference Solution
```python
def calculate_median(numbers):
    """
    Calculate the median of a set of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The median of the numbers.
    """
    # Sort the list of numbers
    numbers.sort()

    # Get the length of the list
    n = len(numbers)

    # If the list is empty, return None
    if n == 0:
        return None

    # If the list has an odd number of elements, return the middle value
    if n % 2 == 1:
        return numbers[n // 2]

    # If the list has an even number of elements, return the average of the two middle values
    else:
        mid1 = numbers[n // 2 - 1]
        mid2 = numbers[n // 2]
        return (mid1 + mid2) / 2

# Example usage
def main():
    numbers = [1, 3, 5, 7, 9]
    median = calculate_median(numbers)
    print(f"The median of {numbers} is {median}")

    numbers = [1, 2, 3, 4, 5, 6]
    median = calculate_median(numbers)
    print(f"The median of {numbers} is {median}")

if __name__ == "__main__":
    main()
```
