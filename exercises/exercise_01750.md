## Exercise
Write a program to delete a specific element from an array.

## Reference Solution
```python
def delete_element(array, target):
    """
    Deletes the first occurrence of a target element from an array.

    Args:
    array (list): The input array.
    target: The element to be deleted.

    Returns:
    list: The updated array after deletion.
    """
    if target in array:
        array.remove(target)
    return array

# Example usage:
array = [1, 2, 3, 4, 5, 2, 6]
target = 2
print("Original array:", array)
print("Target element:", target)
print("Updated array:", delete_element(array, target))
```

```python
def delete_all_occurrences(array, target):
    """
    Deletes all occurrences of a target element from an array.

    Args:
    array (list): The input array.
    target: The element to be deleted.

    Returns:
    list: The updated array after deletion.
    """
    return [element for element in array if element != target]

# Example usage:
array = [1, 2, 3, 4, 5, 2, 6]
target = 2
print("Original array:", array)
print("Target element:", target)
print("Updated array:", delete_all_occurrences(array, target))
```
