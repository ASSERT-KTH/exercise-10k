## Exercise
Create a program that takes two numbers and an arithmetic operation as input and performs the operation. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
def perform_arithmetic_operation(num1, num2, operation):
    """
    Performs an arithmetic operation on two numbers.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.
    operation (str): The arithmetic operation to perform. Can be '+', '-', '*', or '/'.

    Returns:
    float: The result of the operation.

    Raises:
    ValueError: If the operation is not one of '+', '-', '*', or '/'.
    ZeroDivisionError: If the operation is '/' and num2 is zero.
    """

    # Check if the operation is valid
    if operation not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operation. Must be one of '+', '-', '*', or '/'.")

    # Perform the operation
    if operation == '+':
        return num1 + num2
    elif operation == '-':
        return num1 - num2
    elif operation == '*':
        return num1 * num2
    elif operation == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


def main():
    # Get the numbers and operation from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, /): ")

    try:
        # Perform the operation and print the result
        result = perform_arithmetic_operation(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
