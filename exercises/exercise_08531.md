## Exercise
Write a program to calculate the total cost of an item after applying a quantity discount.

## Reference Solution
```python
def calculate_total_cost(price, quantity, discount_thresholds):
    """
    Calculate the total cost of an item after applying a quantity discount.

    Args:
        price (float): The price of a single item.
        quantity (int): The quantity of items purchased.
        discount_thresholds (dict): A dictionary of discount thresholds and corresponding discounts.

    Returns:
        float: The total cost of the items after applying the discount.
    """
    total_cost = price * quantity

    # Apply discount based on quantity
    for threshold, discount in discount_thresholds.items():
        if quantity >= threshold:
            total_cost *= (1 - discount / 100)
            break

    return total_cost

def main():
    # Define discount thresholds and corresponding discounts
    discount_thresholds = {
        10: 5,  # 5% discount for 10 or more items
        20: 10,  # 10% discount for 20 or more items
        50: 15   # 15% discount for 50 or more items
    }

    # Get user input
    price = float(input("Enter the price of a single item: "))
    quantity = int(input("Enter the quantity of items purchased: "))

    # Calculate total cost
    total_cost = calculate_total_cost(price, quantity, discount_thresholds)

    # Print result
    print(f"Total cost: ${total_cost:.2f}")

if __name__ == "__main__":
    main()
```
