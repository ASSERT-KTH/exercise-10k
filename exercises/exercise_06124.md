## Exercise
Create a game of Rock Paper Scissors where the user can play against the computer.

## Reference Solution
```python
import random

class RockPaperScissors:
    def __init__(self):
        self.user_score = 0
        self.computer_score = 0

    def play_round(self):
        """Play a round of Rock Paper Scissors"""
        user_choice = input("Enter your choice (rock, paper, scissors): ").lower()
        while user_choice not in ["rock", "paper", "scissors"]:
            user_choice = input("Invalid choice. Please enter rock, paper, or scissors: ").lower()

        computer_choice = random.choice(["rock", "paper", "scissors"])
        print(f"\nComputer chose: {computer_choice}")

        if user_choice == computer_choice:
            print(f"Both players selected {user_choice}. It's a tie!")
        elif user_choice == "rock":
            if computer_choice == "scissors":
                print("Rock smashes scissors! You win this round.")
                self.user_score += 1
            else:
                print("Paper covers rock! You lose this round.")
                self.computer_score += 1
        elif user_choice == "paper":
            if computer_choice == "rock":
                print("Paper covers rock! You win this round.")
                self.user_score += 1
            else:
                print("Scissors cuts paper! You lose this round.")
                self.computer_score += 1
        elif user_choice == "scissors":
            if computer_choice == "paper":
                print("Scissors cuts paper! You win this round.")
                self.user_score += 1
            else:
                print("Rock smashes scissors! You lose this round.")
                self.computer_score += 1

        print(f"\nScore - You: {self.user_score}, Computer: {self.computer_score}\n")

    def play_game(self):
        """Play the game until the user decides to quit"""
        while True:
            self.play_round()
            play_again = input("Do you want to play again? (yes/no): ").lower()
            while play_again not in ["yes", "no"]:
                play_again = input("Invalid choice. Please enter yes or no: ").lower()
            if play_again != "yes":
                break

        print("\nFinal Score:")
        print(f"You: {self.user_score}")
        print(f"Computer: {self.computer_score}")

        if self.user_score > self.computer_score:
            print("You win the game! Congratulations!")
        elif self.user_score < self.computer_score:
            print("Computer wins the game! Better luck next time!")
        else:
            print("It's a tie game!")

if __name__ == "__main__":
    game = RockPaperScissors()
    game.play_game()
```
