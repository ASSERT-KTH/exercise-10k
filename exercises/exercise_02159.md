## Exercise
Write a program that prints out a table with columns for name, age, and city.

## Reference Solution
```python
from tabulate import tabulate

def print_table(data):
    """
    Print a table with columns for name, age, and city.

    Args:
        data (list): A list of dictionaries containing name, age, and city data.
    """
    # Define the table headers
    headers = ["Name", "Age", "City"]

    # Extract the data into a list of lists
    table_data = [[person["name"], person["age"], person["city"]] for person in data]

    # Print the table
    print(tabulate(table_data, headers, tablefmt="grid"))

def main():
    # Example data
    data = [
        {"name": "John Doe", "age": 30, "city": "New York"},
        {"name": "Jane Smith", "age": 25, "city": "Los Angeles"},
        {"name": "Bob Johnson", "age": 40, "city": "Chicago"}
    ]

    # Print the table
    print_table(data)

if __name__ == "__main__":
    main()
```
