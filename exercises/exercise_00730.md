## Exercise
Write a program that takes a list of numbers as input and outputs their mean, median, and range.

## Reference Solution
```python
import statistics

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and range of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the mean, median, and range of the input list.
    """
    mean = statistics.mean(numbers)
    median = statistics.median(numbers)
    range_ = max(numbers) - min(numbers)
    return mean, median, range_

def main():
    # Get the list of numbers from the user
    numbers = input("Enter a list of numbers separated by spaces: ")
    numbers = [float(num) for num in numbers.split()]

    # Calculate the statistics
    mean, median, range_ = calculate_statistics(numbers)

    # Print the results
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Range: {range_}")

if __name__ == "__main__":
    main()
```
