## Exercise
Create a function to check if a given string is in title case or not

## Reference Solution
```python
def is_title_case(input_string: str) -> bool:
    """
    Checks if a given string is in title case or not.

    Args:
        input_string (str): The string to be checked.

    Returns:
        bool: True if the string is in title case, False otherwise.
    """
    # Split the string into words
    words = input_string.split()

    # Check each word
    for word in words:
        # If the word is not in title case, return False
        if not word.istitle():
            return False

    # If all words are in title case, return True
    return True

# Example usage:
def main():
    print(is_title_case("Hello World"))  # True
    print(is_title_case("hello world"))  # False
    print(is_title_case("HELLO WORLD"))  # False
    print(is_title_case("Hello"))  # True

if __name__ == "__main__":
    main()
```
