## Exercise
Write a program to remove duplicate elements from an array.

## Reference Solution
```python
def remove_duplicates(array):
    # Convert the array to a set, which automatically removes duplicates
    unique_set = set(array)
    
    # Convert the set back to a list
    unique_list = list(unique_set)
    
    return unique_list

# Test the function
array = [1, 2, 2, 3, 4, 4, 5, 6, 6, 7, 8, 8, 9]
print("Original array:", array)
print("Array without duplicates:", remove_duplicates(array))
```

```python
def remove_duplicates_ordered(array):
    seen = set()
    unique_list = [x for x in array if not (x in seen or seen.add(x))]
    return unique_list

# Test the function
array = [1, 2, 2, 3, 4, 4, 5, 6, 6, 7, 8, 8, 9]
print("Original array:", array)
print("Array without duplicates:", remove_duplicates_ordered(array))
```
