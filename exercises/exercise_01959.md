## Exercise
Write a program that takes two arrays as input and returns the correlation between them.

## Reference Solution
```python
import numpy as np

def calculate_correlation(array1, array2):
    """
    Calculate the correlation between two arrays.

    Args:
    array1 (list): The first input array.
    array2 (list): The second input array.

    Returns:
    float: The correlation between the two input arrays.
    """
    # Convert input lists to NumPy arrays
    array1 = np.array(array1)
    array2 = np.array(array2)

    # Check if the input arrays have the same length
    if len(array1) != len(array2):
        raise ValueError("Input arrays must have the same length")

    # Calculate the correlation using NumPy's corrcoef function
    correlation_coefficient = np.corrcoef(array1, array2)[0, 1]

    return correlation_coefficient

# Example usage
if __name__ == "__main__":
    # Define two example arrays
    array1 = [1, 2, 3, 4, 5]
    array2 = [2, 3, 5, 7, 11]

    # Calculate and print the correlation
    correlation = calculate_correlation(array1, array2)
    print(f"The correlation between the two arrays is: {correlation}")
```
