## Exercise
Develop a program to simulate a shopping cart, allowing users to add, remove, and view items, using a while loop.

## Reference Solution
```python
class ShoppingCart:
    def __init__(self):
        self.cart = {}

    def add_item(self, item, quantity):
        """Add an item to the cart"""
        if item in self.cart:
            self.cart[item] += quantity
        else:
            self.cart[item] = quantity
        print(f"Added {quantity} {item}(s) to the cart.")

    def remove_item(self, item, quantity):
        """Remove an item from the cart"""
        if item in self.cart:
            if self.cart[item] <= quantity:
                del self.cart[item]
                print(f"Removed {item} from the cart.")
            else:
                self.cart[item] -= quantity
                print(f"Removed {quantity} {item}(s) from the cart.")
        else:
            print(f"{item} is not in the cart.")

    def view_cart(self):
        """Display the current state of the cart"""
        if not self.cart:
            print("The cart is empty.")
        else:
            print("Current Cart:")
            for item, quantity in self.cart.items():
                print(f"{item}: {quantity}")


def main():
    cart = ShoppingCart()
    while True:
        print("\nShopping Cart Menu:")
        print("1. Add item to cart")
        print("2. Remove item from cart")
        print("3. View cart")
        print("4. Exit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            item = input("Enter the item name: ")
            quantity = int(input("Enter the quantity: "))
            cart.add_item(item, quantity)
        elif choice == "2":
            item = input("Enter the item name: ")
            quantity = int(input("Enter the quantity to remove: "))
            cart.remove_item(item, quantity)
        elif choice == "3":
            cart.view_cart()
        elif choice == "4":
            print("Exiting the shopping cart simulator. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
