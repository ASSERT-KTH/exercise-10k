## Exercise
Given a string, scramble the words by rearranging the characters in each word.

## Reference Solution
```python
import random

def scramble_words(input_string):
    """
    Scramble the words in a given string by rearranging the characters in each word.

    Args:
        input_string (str): The input string to scramble.

    Returns:
        str: The scrambled string.
    """
    words = input_string.split()
    scrambled_words = []

    for word in words:
        # Convert the word to a list of characters
        char_list = list(word)

        # Shuffle the list of characters
        random.shuffle(char_list)

        # Join the scrambled characters back into a string
        scrambled_word = ''.join(char_list)

        scrambled_words.append(scrambled_word)

    # Join the scrambled words back into a string
    scrambled_string = ' '.join(scrambled_words)

    return scrambled_string

# Example usage
input_string = "Hello World this is a test"
scrambled_string = scramble_words(input_string)
print(scrambled_string)
```
