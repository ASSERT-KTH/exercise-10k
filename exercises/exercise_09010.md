## Exercise
Write a function that implements a basic tree data structure.

## Reference Solution
```python
class Node:
    """Represents a node in the tree."""
    def __init__(self, value):
        self.value = value
        self.children = []

class Tree:
    """Represents the tree itself."""
    def __init__(self, root):
        self.root = Node(root)

    def insert(self, parent_value, new_value):
        """Inserts a new node into the tree."""
        new_node = Node(new_value)
        parent_node = self.find(parent_value)
        if parent_node:
            parent_node.children.append(new_node)

    def find(self, value, node=None):
        """Finds a node with the given value in the tree."""
        if not node:
            node = self.root
        if node.value == value:
            return node
        for child in node.children:
            found_node = self.find(value, child)
            if found_node:
                return found_node
        return None

    def preorder_traversal(self, node=None):
        """Performs a pre-order traversal of the tree."""
        if not node:
            node = self.root
        print(node.value, end=" ")
        for child in node.children:
            self.preorder_traversal(child)

    def inorder_traversal(self, node=None):
        """Performs an in-order traversal of the tree."""
        if not node:
            node = self.root
        if node.children:
            self.inorder_traversal(node.children[0])
        print(node.value, end=" ")
        for child in node.children[1:]:
            self.inorder_traversal(child)

    def postorder_traversal(self, node=None):
        """Performs a post-order traversal of the tree."""
        if not node:
            node = self.root
        for child in node.children:
            self.postorder_traversal(child)
        print(node.value, end=" ")

    def search(self, value):
        """Searches for a value in the tree."""
        return self.find(value) is not None


# Example usage:
if __name__ == "__main__":
    tree = Tree("A")
    tree.insert("A", "B")
    tree.insert("A", "C")
    tree.insert("B", "D")
    tree.insert("B", "E")
    tree.insert("C", "F")

    print("Pre-order traversal:")
    tree.preorder_traversal()  # Output: A B D E C F
    print("\nIn-order traversal:")
    tree.inorder_traversal()  # Note: In-order traversal doesn't apply well to non-binary trees
    print("\nPost-order traversal:")
    tree.postorder_traversal()  # Output: D E B F C A

    print("\nSearching for value 'D':")
    print(tree.search("D"))  # Output: True
    print("Searching for value 'G':")
    print(tree.search("G"))  # Output: False
```
