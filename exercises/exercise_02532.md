## Exercise
Implement a shortest path algorithm to find the minimum distance between two nodes in a graph

## Reference Solution
```python
import sys
import heapq

def dijkstra(graph, source, target):
    """
    Dijkstra's shortest path algorithm implementation.

    Args:
    graph (dict): Adjacency list representation of the graph.
    source (node): Source node.
    target (node): Target node.

    Returns:
    distance (float): Minimum distance between the source and target nodes.
    path (list): Shortest path from the source to the target node.
    """
    # Initialize distances and previous nodes
    distances = {node: sys.maxsize for node in graph}
    previous = {node: None for node in graph}
    distances[source] = 0

    # Priority queue to store nodes to be processed
    priority_queue = [(0, source)]

    while priority_queue:
        # Extract the node with the minimum distance
        current_distance, current_node = heapq.heappop(priority_queue)

        # If the current node is the target node, construct the path
        if current_node == target:
            path = []
            while current_node is not None:
                path.append(current_node)
                current_node = previous[current_node]
            path.reverse()
            return distances[target], path

        # If the current distance is greater than the known distance, skip
        if current_distance > distances[current_node]:
            continue

        # Iterate over the neighbors of the current node
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight

            # If a shorter path is found, update the distance and previous node
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                previous[neighbor] = current_node
                heapq.heappush(priority_queue, (distance, neighbor))

    # If there is no path to the target node, return infinity and an empty path
    return sys.maxsize, []

# Example usage
graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

source_node = 'A'
target_node = 'D'

distance, path = dijkstra(graph, source_node, target_node)
print(f"Minimum distance from {source_node} to {target_node}: {distance}")
print(f"Shortest path from {source_node} to {target_node}: {path}")
```
