## Exercise
Create a program that simulates a bank transaction system. The program should ask the user for their account type (checking or savings) and the transaction type (deposit or withdrawal). Based on the account type and transaction type, the program should apply the corresponding fees and calculate the new balance. If the user tries to withdraw more money than they have in their account, the program should display an error message.

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_type, balance=0):
        self.account_type = account_type
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        print(f"Deposit of ${amount} successful. New balance: ${self.balance:.2f}")

    def withdraw(self, amount):
        if amount > self.balance:
            print("Error: Insufficient funds.")
        else:
            if self.account_type == "checking":
                fee = 0.01  # 1% fee for checking accounts
            elif self.account_type == "savings":
                fee = 0.005  # 0.5% fee for savings accounts

            total_amount = amount + (amount * fee)
            self.balance -= total_amount
            print(f"Withdrawal of ${amount} successful. Fee: ${amount * fee:.2f}. New balance: ${self.balance:.2f}")


def main():
    print("Welcome to the bank transaction system!")
    account_type = input("Enter your account type (checking or savings): ").lower()
    while account_type not in ["checking", "savings"]:
        account_type = input("Invalid account type. Please enter checking or savings: ").lower()

    initial_balance = float(input("Enter your initial balance: $"))
    account = BankAccount(account_type, initial_balance)

    while True:
        print("\nTransaction options:")
        print("1. Deposit")
        print("2. Withdrawal")
        print("3. Check balance")
        print("4. Exit")
        choice = input("Enter your choice: ")

        if choice == "1":
            amount = float(input("Enter the amount to deposit: $"))
            account.deposit(amount)
        elif choice == "2":
            amount = float(input("Enter the amount to withdraw: $"))
            account.withdraw(amount)
        elif choice == "3":
            print(f"Current balance: ${account.balance:.2f}")
        elif choice == "4":
            print("Thank you for using the bank transaction system!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
