## Exercise
Write a program to find the most frequent character in a given string.

## Reference Solution
```python
from collections import Counter

def most_frequent_char(s):
    """
    Find the most frequent character in a given string.

    Args:
        s (str): The input string.

    Returns:
        tuple: A tuple containing the most frequent character and its frequency.
    """
    # Remove whitespace and convert to lowercase
    s = s.replace(" ", "").lower()

    # Count the occurrences of each character
    char_count = Counter(s)

    # Find the most frequent character
    most_frequent = char_count.most_common(1)[0]

    return most_frequent

def main():
    # Example usage
    s = "Hello, World!"
    most_frequent = most_frequent_char(s)
    print(f"The most frequent character in '{s}' is '{most_frequent[0]}' with a frequency of {most_frequent[1]}.")

if __name__ == "__main__":
    main()
```
