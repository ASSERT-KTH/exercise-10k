## Exercise
Write a program to analyze a given text and print out statistics such as word count, character count, and average word length.

## Reference Solution
```python
def analyze_text(text):
    """
    Analyze a given text and print out statistics such as word count, character count, and average word length.

    Args:
        text (str): The text to be analyzed.

    Returns:
        None
    """

    # Remove leading and trailing whitespaces
    text = text.strip()

    # Calculate word count
    words = text.split()
    word_count = len(words)

    # Calculate character count
    character_count = len(text)

    # Calculate average word length
    total_word_length = sum(len(word) for word in words)
    average_word_length = total_word_length / word_count if word_count > 0 else 0

    # Print the statistics
    print("Text Analysis Statistics:")
    print("-------------------------")
    print(f"Word Count: {word_count}")
    print(f"Character Count: {character_count}")
    print(f"Average Word Length: {average_word_length:.2f}")

def main():
    # Example usage
    text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Nunc accumsan sem ut ligula scelerisque sollicitudin. Ut at sagittis augue. Praesent quis rhoncus justo. Aliquam erat volutpat. Donec sit amet felis quis est ullamcorper placerat. Proin blandit eu arcu eu facilisis. In hac habitasse platea dictumst."
    analyze_text(text)

if __name__ == "__main__":
    main()
```
