## Exercise
Create a Mad Libs game using variables and string manipulation.

## Reference Solution
```python
# Import the required library
import random

# Define a function to get user input
def get_user_input(word_type):
    """Get user input for a specific word type"""
    while True:
        user_input = input(f"Enter a {word_type}: ")
        if user_input:
            return user_input
        else:
            print("Please enter a word.")

# Define a function to play the game
def play_game():
    """Play the Mad Libs game"""
    print("Welcome to Mad Libs!")

    # Get user input for different word types
    noun = get_user_input("noun")
    verb = get_user_input("verb")
    adjective = get_user_input("adjective")
    place = get_user_input("place")
    person = get_user_input("person")

    # Define a story with missing words
    story = f"One day, a very {adjective} {noun} was walking down the street. " \
            f"Suddenly, it started to {verb} and ran all the way to the {place}. " \
            f"There, it met a {person} who was also very {adjective}. " \
            f"They became fast friends and had many more {adjective} adventures together."

    # Print the completed story
    print("\nHere's your completed story:")
    print(story)

# Run the game
if __name__ == "__main__":
    play_game()
```
