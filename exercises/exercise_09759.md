## Exercise
Write a program to calculate the compound interest

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, frequency):
    """
    Calculate compound interest.

    Args:
        principal (float): Initial amount of money
        rate (float): Annual interest rate (in decimal)
        time (int): Time period in years
        frequency (int): Compounding frequency per year

    Returns:
        float: Compound interest
    """
    return principal * (1 + rate / frequency) ** (frequency * time)

def main():
    print("Compound Interest Calculator")
    print("----------------------------")

    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period in years: "))
    frequency = int(input("Enter the compounding frequency per year (e.g., 1 for annually, 12 for monthly): "))

    interest = calculate_compound_interest(principal, rate, time, frequency)
    print(f"The compound interest is: ${interest:.2f}")

if __name__ == "__main__":
    main()
```
