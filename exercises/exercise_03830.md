## Exercise
Write a program that takes user input for different parts of speech (e.g., noun, verb, adjective, etc.) and uses them to fill in the blanks in a story.

## Reference Solution
```python
# Import the required libraries
import random

# Define a function to get user input for parts of speech
def get_user_input():
    """
    Get user input for different parts of speech.
    
    Returns:
        dict: A dictionary containing the user's input for each part of speech.
    """
    print("Please provide the following words:")
    
    # Get user input for each part of speech
    user_input = {
        "noun1": input("Noun (e.g., book, car, tree): "),
        "noun2": input("Plural noun (e.g., cats, dogs, cities): "),
        "verb1": input("Verb ending in -ing (e.g., running, dancing, singing): "),
        "verb2": input("Verb (e.g., jump, read, write): "),
        "adjective1": input("Adjective (e.g., happy, tall, green): "),
        "adjective2": input("Adjective (e.g., big, soft, loud): "),
        "adverb": input("Adverb (e.g., quickly, loudly, wisely): "),
        "place": input("Place (e.g., park, beach, city): ")
    }
    
    return user_input

# Define a function to fill in the blanks in the story
def fill_in_the_blanks(user_input):
    """
    Fill in the blanks in the story with the user's input.
    
    Args:
        user_input (dict): A dictionary containing the user's input for each part of speech.
    
    Returns:
        str: The completed story.
    """
    # Define the story with blanks
    story = """
    Yesterday, I saw a very {adjective1} {noun1} in the {place}. It was {verb1} down the street, and I couldn't help but {verb2} after it. As I was {verb2}, I stumbled upon a group of {noun2} who were {verb1} and having a great time. They invited me to join them, and we spent the rest of the day {verb1} and laughing {adverb}. It was a truly {adjective2} experience, and I'm so glad I got to be a part of it.
    """
    
    # Fill in the blanks with the user's input
    completed_story = story.format(**user_input)
    
    return completed_story

# Define the main function
def main():
    # Get user input for parts of speech
    user_input = get_user_input()
    
    # Fill in the blanks in the story
    completed_story = fill_in_the_blanks(user_input)
    
    # Print the completed story
    print("\nHere is the completed story:")
    print(completed_story)

# Run the main function
if __name__ == "__main__":
    main()
```
