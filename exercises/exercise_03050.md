## Exercise
Create a class called Person with attributes for name, age, and occupation. Then, create a function to calculate the person's status based on their age.

## Reference Solution
```python
class Person:
    def __init__(self, name: str, age: int, occupation: str):
        """
        Initialize a Person object.

        Args:
        - name (str): The person's name.
        - age (int): The person's age.
        - occupation (str): The person's occupation.
        """
        self.name = name
        self.age = age
        self.occupation = occupation

    def calculate_status(self) -> str:
        """
        Calculate the person's status based on their age.

        Returns:
        - str: The person's status (e.g., "Child", "Teenager", "Adult", "Senior").
        """
        if self.age < 13:
            return "Child"
        elif self.age < 20:
            return "Teenager"
        elif self.age < 65:
            return "Adult"
        else:
            return "Senior"


# Example usage
def main():
    person1 = Person("John Doe", 25, "Software Engineer")
    person2 = Person("Jane Doe", 70, "Retiree")

    print(f"{person1.name}'s status: {person1.calculate_status()}")
    print(f"{person2.name}'s status: {person2.calculate_status()}")


if __name__ == "__main__":
    main()
```
