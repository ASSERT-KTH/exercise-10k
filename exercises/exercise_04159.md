Write a function to apply a function to each value in a dictionary.

def apply_function_to_dict_values(dictionary, func):
    """
    Applies a given function to each value in a dictionary.

    Args:
    - dictionary (dict): The dictionary to apply the function to.
    - func (function): The function to apply to each value.

    Returns:
    - dict: A new dictionary with the function applied to each value.
    """
    return {key: func(value) for key, value in dictionary.items()}

```python
def apply_function_to_dict_values(dictionary, func):
    """
    Applies a given function to each value in a dictionary.

    Args:
    - dictionary (dict): The dictionary to apply the function to.
    - func (function): The function to apply to each value.

    Returns:
    - dict: A new dictionary with the function applied to each value.
    """
    result = {}
    for key, value in dictionary.items():
        try:
            result[key] = func(value)
        except Exception as e:
            print(f"Error applying function to {key}: {e}")
    return result

# Example usage:
if __name__ == "__main__":
    # Define a dictionary
    my_dict = {"a": 1, "b": 2, "c": "three"}

    # Define a function to double a number
    def double(x):
        return x * 2

    # Apply the function to the dictionary
    result = apply_function_to_dict_values(my_dict, double)

    # Print the result
    print(result)
```
