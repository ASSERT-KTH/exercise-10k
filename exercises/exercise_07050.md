## Exercise
Write a program that simulates a game of Rock, Paper, Scissors.

## Reference Solution
```python
import random

def get_computer_move():
    """Returns a random move from the computer."""
    moves = ['rock', 'paper', 'scissors']
    return random.choice(moves)

def get_user_move():
    """Asks the user for their move and returns it."""
    while True:
        move = input("Enter your move (rock, paper, or scissors): ").lower()
        if move in ['rock', 'paper', 'scissors']:
            return move
        print("Invalid move. Please try again.")

def determine_winner(player1_move, player2_move):
    """Determines the winner based on the moves of the two players."""
    if player1_move == player2_move:
        return "It's a tie!"
    if (player1_move == 'rock' and player2_move == 'scissors') or \
       (player1_move == 'scissors' and player2_move == 'paper') or \
       (player1_move == 'paper' and player2_move == 'rock'):
        return "Player 1 wins!"
    return "Player 2 wins!"

def play_single_player():
    """Plays a game of Rock, Paper, Scissors against the computer."""
    user_move = get_user_move()
    computer_move = get_computer_move()
    print(f"\nYou chose {user_move}, computer chose {computer_move}.")
    print(determine_winner(user_move, computer_move))

def play_multi_player():
    """Plays a game of Rock, Paper, Scissors between two human players."""
    player1_move = get_user_move()
    player2_move = get_user_move()
    print(f"\nPlayer 1 chose {player1_move}, Player 2 chose {player2_move}.")
    print(determine_winner(player1_move, player2_move))

def main():
    """Asks the user to choose a game mode and starts the game."""
    print("Welcome to Rock, Paper, Scissors!")
    while True:
        print("\nChoose a game mode:")
        print("1. Play against the computer")
        print("2. Play against another player")
        print("3. Quit")
        choice = input("Enter your choice (1, 2, or 3): ")
        if choice == '1':
            play_single_player()
        elif choice == '2':
            play_multi_player()
        elif choice == '3':
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
