## Exercise
Write a program that checks the lifetime of a given variable.

## Reference Solution
```python
import weakref
import gc

class Variable:
    def __init__(self, name):
        self.name = name

    def __del__(self):
        print(f"Variable {self.name} has been destroyed")

def check_variable_lifetime(variable_name):
    # Create a variable
    variable = Variable(variable_name)
    print(f"Variable {variable_name} has been created")

    # Create a weak reference to the variable
    weak_ref = weakref.ref(variable)

    # Delete the strong reference to the variable
    del variable

    # Check if the variable still exists
    if weak_ref() is not None:
        print(f"Variable {variable_name} still exists")
    else:
        print(f"Variable {variable_name} has been garbage collected")

    # Force garbage collection
    gc.collect()

    # Check again if the variable still exists
    if weak_ref() is not None:
        print(f"Variable {variable_name} still exists after garbage collection")
    else:
        print(f"Variable {variable_name} has been garbage collected after garbage collection")

# Example usage
check_variable_lifetime("my_variable")
```
