## Exercise
Write a program that takes a password as input and prints out whether it is strong or not. A strong password should contain at least 8 characters, including uppercase and lowercase letters, and numbers.

## Reference Solution
```python
import re

def check_password_strength(password):
    """
    Checks whether a password is strong or not.

    A strong password should contain at least 8 characters, including:
    - Uppercase letters
    - Lowercase letters
    - Numbers

    Args:
        password (str): The password to check.

    Returns:
        bool: True if the password is strong, False otherwise.
    """
    # Define the regular expression pattern for a strong password
    pattern = r"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).{8,}$"

    # Check if the password matches the pattern
    if re.match(pattern, password):
        return True
    else:
        return False

def main():
    # Get the password from the user
    password = input("Enter a password: ")

    # Check the password strength
    if check_password_strength(password):
        print("The password is strong.")
    else:
        print("The password is not strong. It should contain at least 8 characters, including uppercase and lowercase letters, and numbers.")

if __name__ == "__main__":
    main()
```
