## Exercise
Write a program that plays a game of Rock, Paper, Scissors with the user, where the user can choose to play again using a while loop.

## Reference Solution
```python
import random

def get_computer_move():
    """Get the computer's move."""
    moves = ["rock", "paper", "scissors"]
    return random.choice(moves)

def determine_winner(user_move, computer_move):
    """Determine the winner of the game."""
    if user_move == computer_move:
        return "It's a tie!"
    if (user_move == "rock" and computer_move == "scissors") or \
       (user_move == "scissors" and computer_move == "paper") or \
       (user_move == "paper" and computer_move == "rock"):
        return "You win!"
    return "Computer wins!"

def play_game():
    """Play a game of Rock, Paper, Scissors."""
    user_move = input("Enter your move (rock, paper, or scissors): ").lower()
    while user_move not in ["rock", "paper", "scissors"]:
        user_move = input("Invalid move. Please enter rock, paper, or scissors: ").lower()
    computer_move = get_computer_move()
    print(f"\nYou chose {user_move}, computer chose {computer_move}.\n")
    print(determine_winner(user_move, computer_move))

def main():
    """Play games of Rock, Paper, Scissors until the user chooses to stop."""
    play_again = "y"
    while play_again.lower() == "y":
        play_game()
        play_again = input("\nDo you want to play again? (y/n): ")
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
```
