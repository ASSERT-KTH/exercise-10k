## Exercise
Create a program that removes duplicate words from a given string.

## Reference Solution
```python
def remove_duplicate_words(input_string):
    """
    Removes duplicate words from a given string, preserving the original order.
    
    Args:
        input_string (str): The input string.
    
    Returns:
        str: The string with duplicate words removed.
    """
    # Split the string into words
    words = input_string.split()
    
    # Use a set to keep track of unique words
    unique_words = set()
    result = []
    
    # Iterate over the words and add unique ones to the result
    for word in words:
        if word.lower() not in [w.lower() for w in unique_words]:
            unique_words.add(word)
            result.append(word)
    
    # Join the result back into a string
    return ' '.join(result)


def main():
    # Example usage
    input_string = "This is a test string with duplicate words like this and this"
    print("Original String:")
    print(input_string)
    print("String without duplicates:")
    print(remove_duplicate_words(input_string))


if __name__ == "__main__":
    main()
```
