## Exercise
Write a program to reduce an array to a single value using a reduction operation and print the result.

## Reference Solution
```python
import numpy as np

def reduce_array(arr, operation):
    """
    Reduces an array to a single value using a reduction operation.

    Args:
        arr (list): The input array.
        operation (str): The reduction operation to apply. Can be 'sum', 'product', or 'average'.

    Returns:
        float: The result of the reduction operation.
    """
    if operation == 'sum':
        return np.sum(arr)
    elif operation == 'product':
        return np.prod(arr)
    elif operation == 'average':
        return np.mean(arr)
    else:
        raise ValueError("Invalid operation. Must be 'sum', 'product', or 'average'.")

def main():
    # Example usage
    arr = [1, 2, 3, 4, 5]
    operations = ['sum', 'product', 'average']

    for operation in operations:
        result = reduce_array(arr, operation)
        print(f"Result of {operation} operation: {result}")

if __name__ == "__main__":
    main()
```
