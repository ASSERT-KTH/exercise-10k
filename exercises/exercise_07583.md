## Exercise
Write a program that converts a variable from one data type to another.

## Reference Solution
```python
# Function to convert a variable to a different data type
def convert_data_type(var, data_type):
    try:
        if data_type == "int":
            return int(var)
        elif data_type == "float":
            return float(var)
        elif data_type == "str":
            return str(var)
        elif data_type == "bool":
            return bool(var)
        else:
            print("Invalid data type.")
            return None
    except ValueError:
        print("Invalid input for the given data type.")
        return None

# Test the function with different data types
if __name__ == "__main__":
    # Convert a string to an integer
    var1 = "123"
    print(f"Original value: {var1}, Original type: {type(var1)}")
    converted_var1 = convert_data_type(var1, "int")
    print(f"Converted value: {converted_var1}, Converted type: {type(converted_var1)}")

    # Convert a number to a string
    var2 = 456
    print(f"\nOriginal value: {var2}, Original type: {type(var2)}")
    converted_var2 = convert_data_type(var2, "str")
    print(f"Converted value: {converted_var2}, Converted type: {type(converted_var2)}")

    # Convert a float to an integer
    var3 = 7.89
    print(f"\nOriginal value: {var3}, Original type: {type(var3)}")
    converted_var3 = convert_data_type(var3, "int")
    print(f"Converted value: {converted_var3}, Converted type: {type(converted_var3)}")

    # Convert a string to a boolean
    var4 = "True"
    print(f"\nOriginal value: {var4}, Original type: {type(var4)}")
    converted_var4 = convert_data_type(var4, "bool")
    print(f"Converted value: {converted_var4}, Converted type: {type(converted_var4)}")
```
