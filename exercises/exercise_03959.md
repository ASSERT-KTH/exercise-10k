## Exercise
Write a program to calculate the grade of a student based on their scores.

## Reference Solution
```python
# Define a function to calculate the grade
def calculate_grade(math_score, science_score, english_score):
    """
    Calculate the grade based on the scores.

    Args:
        math_score (float): Score in Mathematics
        science_score (float): Score in Science
        english_score (float): Score in English

    Returns:
        str: The calculated grade
    """
    # Calculate the total score
    total_score = math_score + science_score + english_score

    # Calculate the average score
    average_score = total_score / 3

    # Determine the grade based on the average score
    if average_score >= 90:
        return "A"
    elif average_score >= 80:
        return "B"
    elif average_score >= 70:
        return "C"
    elif average_score >= 60:
        return "D"
    else:
        return "F"

# Define a function to get the scores from the user
def get_scores():
    """
    Get the scores from the user.

    Returns:
        tuple: A tuple containing the scores in Mathematics, Science, and English
    """
    math_score = float(input("Enter the score in Mathematics: "))
    science_score = float(input("Enter the score in Science: "))
    english_score = float(input("Enter the score in English: "))
    return math_score, science_score, english_score

# Define the main function
def main():
    print("Student Grade Calculator")
    print("------------------------")
    math_score, science_score, english_score = get_scores()
    grade = calculate_grade(math_score, science_score, english_score)
    print(f"Grade: {grade}")

# Run the main function
if __name__ == "__main__":
    main()
```
