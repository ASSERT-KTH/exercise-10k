## Exercise
Create a program that takes a password as input and checks if it meets certain criteria (e.g. length, digits, letters).

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the following criteria:
    - Minimum length of 8 characters
    - At least one digit
    - At least one letter (both uppercase and lowercase)
    - At least one special character

    Args:
        password (str): The password to validate

    Returns:
        bool: True if the password is valid, False otherwise
    """

    # Check minimum length
    if len(password) < 8:
        return False

    # Check for digits
    if not re.search(r"\d", password):
        return False

    # Check for letters (both uppercase and lowercase)
    if not re.search(r"[a-z]", password) or not re.search(r"[A-Z]", password):
        return False

    # Check for special characters
    if not re.search(r"[!@#$%^&*()_+=-{};:'<>,./?]", password):
        return False

    return True

def main():
    password = input("Enter a password: ")
    if validate_password(password):
        print("Password is valid")
    else:
        print("Password is not valid. Please ensure it meets the following criteria:")
        print("- Minimum length of 8 characters")
        print("- At least one digit")
        print("- At least one letter (both uppercase and lowercase)")
        print("- At least one special character")

if __name__ == "__main__":
    main()
```
