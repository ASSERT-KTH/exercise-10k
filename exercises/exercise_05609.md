## Exercise
Develop a program that asks the user a series of trivia questions, taking the user's answers as input and keeping track of the score.

## Reference Solution
```python
# Import the required modules
import random

# Define a dictionary with trivia questions, options, and answers
trivia_questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "Answer": "B"
    },
    "Which planet is the largest in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "Answer": "C"
    },
    "Who painted the famous painting 'The Starry Night'?": {
        "A": "Leonardo da Vinci",
        "B": "Vincent van Gogh",
        "C": "Pablo Picasso",
        "D": "Claude Monet",
        "Answer": "B"
    },
    "What is the largest mammal on Earth?": {
        "A": "Blue whale",
        "B": "African elephant",
        "C": "Hippopotamus",
        "D": "Giraffe",
        "Answer": "A"
    },
    "Which music group's original members were John, Paul, George, and Ringo?": {
        "A": "The Rolling Stones",
        "B": "The Beatles",
        "C": "The Who",
        "D": "The Beach Boys",
        "Answer": "B"
    }
}

def trivia_game():
    # Initialize the score
    score = 0
    
    # Shuffle the questions to randomize the order
    questions = list(trivia_questions.items())
    random.shuffle(questions)
    
    # Ask each question and check the answer
    for question, options in questions:
        print(f"\nQuestion: {question}")
        for option, value in options.items():
            if option != "Answer":
                print(f"{option}: {value}")
        
        # Get the user's answer
        answer = input("Enter your answer (A, B, C, D): ").upper()
        
        # Check if the answer is correct
        if answer == options["Answer"]:
            print("Correct answer!")
            score += 1
        else:
            print(f"Sorry, the correct answer is {options['Answer']}.")
    
    # Print the final score
    print(f"\nGame over! Your final score is {score} out of {len(questions)}.")

if __name__ == "__main__":
    trivia_game()
```
