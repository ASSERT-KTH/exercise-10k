## Exercise
Write a program that checks if a string is a palindrome using a while loop.

## Reference Solution
```python
def is_palindrome(s):
    """
    Checks if a string is a palindrome using a while loop.

    Args:
        s (str): The input string.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    s = s.lower()  # Convert to lowercase for case-insensitive comparison
    left = 0
    right = len(s) - 1

    while left < right:
        if s[left] != s[right]:
            return False
        left += 1
        right -= 1

    return True

# Example usage
def main():
    strings_to_check = ["radar", "hello", "level", "python"]

    for s in strings_to_check:
        result = is_palindrome(s)
        print(f"'{s}' is a palindrome: {result}")

if __name__ == "__main__":
    main()
```
