## Exercise
Create a program that uses a while loop to repeatedly ask the user for their password until it meets a given set of criteria (e.g. length, characters).

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the given criteria.
    
    Args:
        password (str): The password to validate.
    
    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the regular expression patterns for each criterion
    patterns = {
        'uppercase': r'[A-Z]',
        'lowercase': r'[a-z]',
        'digit': r'\d',
        'special_char': r'[^A-Za-z0-9]'
    }
    
    # Check if the password meets each criterion
    if (len(password) < 8 or
        not re.search(patterns['uppercase'], password) or
        not re.search(patterns['lowercase'], password) or
        not re.search(patterns['digit'], password) or
        not re.search(patterns['special_char'], password)):
        return False
    
    return True

def main():
    password = None
    
    # Repeatedly ask the user for their password until it meets the criteria
    while password is None or not validate_password(password):
        password = input("Enter your password: ")
        
        if not validate_password(password):
            print("Invalid password. Please ensure that it meets the following criteria:")
            print("* At least 8 characters long")
            print("* Contains at least one uppercase letter")
            print("* Contains at least one lowercase letter")
            print("* Contains at least one digit")
            print("* Contains at least one special character")
    
    print("Password accepted.")

if __name__ == "__main__":
    main()
```
