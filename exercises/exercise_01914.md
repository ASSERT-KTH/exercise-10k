## Exercise
Calculate the GPA given a list of grades and their respective credits.

## Reference Solution
```python
def calculate_gpa(grades):
    """
    Calculate the GPA given a list of grades and their respective credits.

    Args:
        grades (list): A list of tuples, where each tuple contains a grade and its credit.

    Returns:
        float: The calculated GPA.

    """
    # Define a dictionary to map grades to grade points
    grade_points = {
        'A': 4.0,
        'A-': 3.7,
        'B+': 3.3,
        'B': 3.0,
        'B-': 2.7,
        'C+': 2.3,
        'C': 2.0,
        'C-': 1.7,
        'D+': 1.3,
        'D': 1.0,
        'D-': 0.7,
        'F': 0.0
    }

    # Initialize variables to store total grade points and total credits
    total_grade_points = 0
    total_credits = 0

    # Iterate over each grade and credit
    for grade, credit in grades:
        # Convert grade to uppercase for case-insensitive comparison
        grade = grade.upper()

        # Check if grade is valid
        if grade not in grade_points:
            raise ValueError(f"Invalid grade: {grade}")

        # Add grade points and credits to total
        total_grade_points += grade_points[grade] * credit
        total_credits += credit

    # Check if there are any credits
    if total_credits == 0:
        raise ValueError("No credits provided")

    # Calculate GPA
    gpa = total_grade_points / total_credits

    return gpa


# Example usage:
grades = [
    ('A', 3),
    ('B', 4),
    ('C', 3),
    ('A-', 2)
]

gpa = calculate_gpa(grades)
print(f"GPA: {gpa:.2f}")
```
