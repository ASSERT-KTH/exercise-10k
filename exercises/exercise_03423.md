## Exercise
Write a function that takes a string and an integer as input and returns the Caesar cipher-encrypted string

def caesar_cipher_encrypt(text: str, shift: int) -> str:
    """
    Encrypts the input text using the Caesar cipher with the given shift.

    Args:
    text (str): The input text to be encrypted.
    shift (int): The number of positions to shift the alphabet.

    Returns:
    str: The encrypted text.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character, and add the shift
            result += chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character, and add the shift
            result += chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
        # If the character is not a letter, leave it unchanged
        else:
            result += char

    return result

# Example usage
if __name__ == "__main__":
    text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_cipher_encrypt(text, shift)
    print(f"Text: {text}")
    print(f"Shift: {shift}")
    print(f"Encrypted Text: {encrypted_text}")
