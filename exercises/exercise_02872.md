## Exercise
Create a function to find the longest substring of a given string without repeating characters

## Reference Solution

def longest_substring_without_repeating_chars(s: str) -> str:
    """
    Find the longest substring of a given string without repeating characters.

    Args:
    s (str): The input string.

    Returns:
    str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    left = 0
    max_length = 0
    max_substring = ""
    char_index_map = {}

    for right in range(len(s)):
        if s[right] in char_index_map:
            left = max(left, char_index_map[s[right]] + 1)

        char_index_map[s[right]] = right
        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring

# Example usage:
print(longest_substring_without_repeating_chars("abcabcbb"))  # Output: "abc"
print(longest_substring_without_repeating_chars("bbbbb"))  # Output: "b"
print(longest_substring_without_repeating_chars("pwwkew"))  # Output: "wke"
print(longest_substring_without_repeating_chars(""))  # Output: ""
