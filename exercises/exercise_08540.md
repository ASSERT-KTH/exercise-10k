## Exercise
Create a program to calculate the surface area of a cone given its radius and height.

## Reference Solution
```python
import math

def calculate_cone_surface_area(radius, height):
    """
    Calculate the surface area of a cone given its radius and height.

    Args:
        radius (float): The radius of the base of the cone.
        height (float): The height of the cone.

    Returns:
        float: The surface area of the cone.
    """
    # Calculate the slant height of the cone
    slant_height = math.sqrt(radius**2 + height**2)

    # Calculate the surface area of the cone
    surface_area = math.pi * radius**2 + math.pi * radius * slant_height

    return surface_area

def main():
    # Get the radius and height of the cone from the user
    radius = float(input("Enter the radius of the cone: "))
    height = float(input("Enter the height of the cone: "))

    # Calculate the surface area of the cone
    surface_area = calculate_cone_surface_area(radius, height)

    # Print the result
    print(f"The surface area of the cone is: {surface_area:.2f}")

if __name__ == "__main__":
    main()
```
