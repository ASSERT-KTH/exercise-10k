## Exercise
Create a program to calculate the momentum given the mass and velocity.

## Reference Solution
```python
def calculate_momentum(mass, velocity):
    """
    Calculate the momentum of an object given its mass and velocity.

    Args:
        mass (float): The mass of the object in kg.
        velocity (float): The velocity of the object in m/s.

    Returns:
        float: The momentum of the object in kg*m/s.
    """
    momentum = mass * velocity
    return momentum

def main():
    print("Momentum Calculator")
    print("--------------------")

    mass = float(input("Enter the mass of the object (in kg): "))
    velocity = float(input("Enter the velocity of the object (in m/s): "))

    momentum = calculate_momentum(mass, velocity)

    print(f"The momentum of the object is {momentum} kg*m/s")

if __name__ == "__main__":
    main()
```
