## Exercise
Write a program to define and use a class with attributes and methods.

## Reference Solution
```python
# Define a class called Vehicle
class Vehicle:
    # Constructor method to initialize the attributes
    def __init__(self, brand, model, year, color):
        self.brand = brand
        self.model = model
        self.year = year
        self.color = color
        self.mileage = 0

    # Method to display the vehicle's details
    def display_details(self):
        print(f"Brand: {self.brand}")
        print(f"Model: {self.model}")
        print(f"Year: {self.year}")
        print(f"Color: {self.color}")
        print(f"Mileage: {self.mileage} miles")

    # Method to update the vehicle's mileage
    def update_mileage(self, new_mileage):
        if new_mileage >= self.mileage:
            self.mileage = new_mileage
        else:
            print("You can't roll back the mileage.")

    # Method to describe the vehicle
    def describe_vehicle(self):
        print(f"This vehicle is a {self.year} {self.color} {self.brand} {self.model} with {self.mileage} miles.")


# Create an instance of the Vehicle class
my_vehicle = Vehicle("Toyota", "Corolla", 2015, "Silver")

# Display the vehicle's details
my_vehicle.display_details()

# Update the vehicle's mileage
my_vehicle.update_mileage(50000)
my_vehicle.display_details()

# Describe the vehicle
my_vehicle.describe_vehicle()
```
