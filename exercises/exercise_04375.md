## Exercise
Write a program that normalizes a dictionary by converting its keys to a standard case and removing any redundant or unnecessary key-value pairs.

## Reference Solution
```python
import string
from collections import defaultdict

def normalize_dict(input_dict):
    normalized_dict = {}
    for key in input_dict:
        # Remove punctuation from the key
        key_no_punct = key.translate(str.maketrans('', '', string.punctuation))
        
        # Convert key to lowercase
        lower_key = key_no_punct.lower()
        
        # If the key is already in the normalized dictionary, append the value to the existing value
        if lower_key in normalized_dict:
            if isinstance(normalized_dict[lower_key], list):
                normalized_dict[lower_key].append(input_dict[key])
            else:
                normalized_dict[lower_key] = [normalized_dict[lower_key], input_dict[key]]
        else:
            normalized_dict[lower_key] = input_dict[key]
    
    return normalized_dict

# Example usage
input_dict = {
    "Name": "John Doe",
    "Email": "john@example.com",
    "email": "john2@example.com",
    "Address": "123 Main St",
    "address": "456 Elm St"
}

normalized_dict = normalize_dict(input_dict)
print(normalized_dict)
```
