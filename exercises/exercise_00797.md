## Exercise
Write a program to calculate the regression line given two lists of numbers.

## Reference Solution
```python
import numpy as np
import matplotlib.pyplot as plt

def calculate_regression_line(x, y):
    """
    Calculate the linear regression line given two lists of numbers.

    Args:
        x (list): List of x-coordinates.
        y (list): List of y-coordinates.

    Returns:
        tuple: A tuple containing the slope (m) and intercept (b) of the regression line.
    """
    # Convert lists to numpy arrays
    x = np.array(x)
    y = np.array(y)

    # Calculate the mean of x and y
    mean_x = np.mean(x)
    mean_y = np.mean(y)

    # Calculate the deviations from the mean
    dev_x = x - mean_x
    dev_y = y - mean_y

    # Calculate the slope (m) and intercept (b) of the regression line
    m = np.sum(dev_x * dev_y) / np.sum(dev_x ** 2)
    b = mean_y - m * mean_x

    return m, b

def plot_regression_line(x, y, m, b):
    """
    Plot the regression line.

    Args:
        x (list): List of x-coordinates.
        y (list): List of y-coordinates.
        m (float): Slope of the regression line.
        b (float): Intercept of the regression line.
    """
    # Generate x-values for the regression line
    x_reg = np.linspace(min(x), max(x), 100)

    # Calculate y-values for the regression line
    y_reg = m * x_reg + b

    # Plot the data points and regression line
    plt.scatter(x, y, label='Data Points')
    plt.plot(x_reg, y_reg, label='Regression Line', color='red')
    plt.xlabel('x')
    plt.ylabel('y')
    plt.title('Linear Regression Line')
    plt.legend()
    plt.show()

def main():
    # Example usage
    x = [1, 2, 3, 4, 5]
    y = [2, 3, 5, 7, 11]

    m, b = calculate_regression_line(x, y)
    print(f'Regression Line: y = {m:.2f}x + {b:.2f}')

    plot_regression_line(x, y, m, b)

if __name__ == '__main__':
    main()
```
