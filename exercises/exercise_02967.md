## Exercise
Write a program that counts the number of occurrences of each word in a given text. The text should be input by the user.

## Reference Solution
```python
import re
from collections import Counter

def count_word_occurrences(text):
    """
    Count the occurrences of each word in the given text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary where keys are words and values are their occurrences.
    """
    # Convert text to lowercase and remove punctuation
    text = text.lower()
    text = re.sub(r'[^\w\s]', '', text)

    # Split text into words
    words = text.split()

    # Count word occurrences
    word_counts = Counter(words)

    return word_counts

def main():
    # Get text input from user
    text = input("Please enter a text: ")

    # Count word occurrences
    word_counts = count_word_occurrences(text)

    # Print word occurrences
    print("Word Occurrences:")
    for word, count in word_counts.most_common():
        print(f"{word}: {count}")

if __name__ == "__main__":
    main()
```
