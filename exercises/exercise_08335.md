## Exercise
Write a program that takes a user input and checks its data type. The program should then print out the data type of the input.

## Reference Solution
```python
def check_data_type(user_input):
    """Checks the data type of the user input."""
    if isinstance(user_input, str):
        return "string"
    elif isinstance(user_input, int):
        return "integer"
    elif isinstance(user_input, float):
        return "float"
    elif isinstance(user_input, bool):
        return "boolean"
    elif isinstance(user_input, list):
        return "list"
    elif isinstance(user_input, tuple):
        return "tuple"
    elif isinstance(user_input, dict):
        return "dictionary"
    elif isinstance(user_input, set):
        return "set"
    else:
        return "unknown"

def main():
    """Gets user input and checks its data type."""
    user_input = input("Please enter something: ")
    
    # Try to convert the input to an integer or float
    try:
        user_input = int(user_input)
    except ValueError:
        try:
            user_input = float(user_input)
        except ValueError:
            # If the input can't be converted to an integer or float, 
            # it will be treated as a string
            pass
    
    data_type = check_data_type(user_input)
    
    print(f"The data type of the input is: {data_type}")

if __name__ == "__main__":
    main()
```
