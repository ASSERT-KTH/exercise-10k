## Exercise
Write a program that checks the data type of a variable and converts it to another data type.

## Reference Solution

def check_and_convert(var):
    """
    Checks the data type of a variable and converts it to another data type.

    Args:
    var: The variable to check and convert.

    Returns:
    A tuple containing the original data type and the converted data type.
    """
    # Check the original data type
    original_type = type(var)
    print(f"Original data type: {original_type.__name__}")

    # Convert to another data type
    if original_type == int:
        converted_var = float(var)
        converted_type = type(converted_var)
    elif original_type == float:
        converted_var = int(var)
        converted_type = type(converted_var)
    elif original_type == str:
        try:
            converted_var = int(var)
            converted_type = type(converted_var)
        except ValueError:
            try:
                converted_var = float(var)
                converted_type = type(converted_var)
            except ValueError:
                print("Cannot convert string to numeric type.")
                return None
    else:
        print("Unsupported data type.")
        return None

    print(f"Converted data type: {converted_type.__name__}")
    return original_type, converted_type


# Example usage
var1 = 10
var2 = 10.5
var3 = "10"
var4 = "10.5"
var5 = "hello"

print("Checking var1:")
check_and_convert(var1)

print("\nChecking var2:")
check_and_convert(var2)

print("\nChecking var3:")
check_and_convert(var3)

print("\nChecking var4:")
check_and_convert(var4)

print("\nChecking var5:")
check_and_convert(var5)
