## Exercise
Create a program that asks the user a series of questions and keeps track of the score. The program should ask the user for their answer and then tell them if it is correct or not.

## Reference Solution
```python
# Import the required modules
import random

# Define the questions, options, and answers
questions = [
    {
        "question": "What is the capital of France?",
        "options": ["Berlin", "Paris", "London", "Rome"],
        "answer": "Paris"
    },
    {
        "question": "What is the largest planet in our solar system?",
        "options": ["Earth", "Saturn", "Jupiter", "Uranus"],
        "answer": "Jupiter"
    },
    {
        "question": "Who painted the Mona Lisa?",
        "options": ["Leonardo da Vinci", "Michelangelo", "Raphael", "Caravaggio"],
        "answer": "Leonardo da Vinci"
    },
    {
        "question": "What is the chemical symbol for gold?",
        "options": ["Ag", "Au", "Hg", "Pb"],
        "answer": "Au"
    },
    {
        "question": "What is the largest mammal?",
        "options": ["Blue whale", "African elephant", "Hippopotamus", "Giraffe"],
        "answer": "Blue whale"
    }
]

# Initialize the score
score = 0

# Loop through each question
for i, question in enumerate(questions, start=1):
    print(f"\nQuestion {i}: {question['question']}")
    
    # Shuffle the options
    random_options = question["options"].copy()
    random.shuffle(random_options)
    
    # Print the options
    for j, option in enumerate(random_options, start=1):
        print(f"{j}. {option}")
    
    # Get the user's answer
    while True:
        try:
            user_answer_index = int(input("\nEnter the number of your answer: "))
            if 1 <= user_answer_index <= len(random_options):
                break
            else:
                print("Invalid option number. Please try again.")
        except ValueError:
            print("Invalid input. Please enter a number.")
    
    # Check if the user's answer is correct
    user_answer = random_options[user_answer_index - 1]
    if user_answer == question["answer"]:
        print("Correct!")
        score += 1
    else:
        print(f"Incorrect. The correct answer is {question['answer']}.")

# Print the final score
print(f"\nYour final score is {score} out of {len(questions)}")
```
