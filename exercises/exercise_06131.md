## Exercise
Create a program that simulates a tournament of Rock Paper Scissors games between multiple players.

## Reference Solution
```python
import random

class Player:
    """Represents a player in the tournament."""
    def __init__(self, name):
        self.name = name
        self.wins = 0
        self.losses = 0
        self.ties = 0

    def play(self):
        """Returns the player's move (rock, paper, or scissors)."""
        # Randomly select a move
        moves = ['rock', 'paper', 'scissors']
        return random.choice(moves)

    def update_record(self, result):
        """Updates the player's win/loss record."""
        if result == 'win':
            self.wins += 1
        elif result == 'loss':
            self.losses += 1
        else:
            self.ties += 1

def determine_winner(player1_move, player2_move):
    """Determines the winner of a game."""
    if player1_move == player2_move:
        return 'tie'
    if (player1_move == 'rock' and player2_move == 'scissors') or \
       (player1_move == 'paper' and player2_move == 'rock') or \
       (player1_move == 'scissors' and player2_move == 'paper'):
        return 'player1'
    return 'player2'

def play_game(player1, player2):
    """Plays a game between two players."""
    player1_move = player1.play()
    player2_move = player2.play()
    print(f"{player1.name} plays {player1_move}, {player2.name} plays {player2_move}")
    winner = determine_winner(player1_move, player2_move)
    if winner == 'tie':
        print("It's a tie!")
        player1.update_record('tie')
        player2.update_record('tie')
    elif winner == 'player1':
        print(f"{player1.name} wins!")
        player1.update_record('win')
        player2.update_record('loss')
    else:
        print(f"{player2.name} wins!")
        player1.update_record('loss')
        player2.update_record('win')

def play_tournament(players):
    """Plays a round-robin tournament between multiple players."""
    for i in range(len(players)):
        for j in range(i + 1, len(players)):
            play_game(players[i], players[j])

def print_standings(players):
    """Prints the standings after the tournament."""
    print("Standings:")
    for player in players:
        print(f"{player.name}: {player.wins} wins, {player.losses} losses, {player.ties} ties")

def main():
    # Create players
    num_players = 5
    players = [Player(f"Player {i+1}") for i in range(num_players)]

    # Play tournament
    play_tournament(players)

    # Print standings
    print_standings(players)

if __name__ == "__main__":
    main()
```
