## Exercise
Write a program that rotates the elements of a given array to the right by a specified number of positions. For example, if the array is [1, 2, 3, 4, 5] and the number of positions is 2, the resulting array should be [4, 5, 1, 2, 3]. The program should handle arrays of any length and any number of positions.

## Reference Solution

#include <iostream>

using namespace std;

int main() {
    int arr[100], n, k, i, j;

    cout << "Enter the number of elements in the array: ";
    cin >> n;

    cout << "Enter the elements of the array: ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    cout << "Enter the number of positions to rotate: ";
    cin >> k;

    for (i = n - k; i < n; i++) {
        cout << arr[i] << " ";
    }

    for (i = 0; i < n - k; i++) {
        cout << arr[i] << " ";
    }

    cout << endl;

    return 0;
}
