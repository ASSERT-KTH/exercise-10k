## Exercise
Create a program that performs basic arithmetic operations (addition, subtraction, multiplication, division) based on user input. The program should ask the user to enter two numbers and an operator (+, -, *, /) and then perform the specified operation.

## Reference Solution
```python
def get_user_input():
    """
    Get user input for two numbers and an operator.
    """
    try:
        num1 = float(input("Enter the first number: "))
        num2 = float(input("Enter the second number: "))
        operator = input("Enter the operator (+, -, *, /): ")
        return num1, num2, operator
    except ValueError:
        print("Invalid input. Please enter a number.")
        return get_user_input()


def perform_arithmetic_operation(num1, num2, operator):
    """
    Perform arithmetic operation based on user input.
    """
    if operator == "+":
        return num1 + num2
    elif operator == "-":
        return num1 - num2
    elif operator == "*":
        return num1 * num2
    elif operator == "/":
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero.")
    else:
        raise ValueError("Invalid operator. Please enter +, -, *, /.")


def main():
    num1, num2, operator = get_user_input()
    try:
        result = perform_arithmetic_operation(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ZeroDivisionError, ValueError) as e:
        print(e)


if __name__ == "__main__":
    main()
```
