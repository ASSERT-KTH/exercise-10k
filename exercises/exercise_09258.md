## Exercise
Write a program that prints the keys of a dictionary and handles potential errors.

## Reference Solution
```python
def print_dict_keys(input_dict):
    """
    Prints the keys of a dictionary.

    Args:
        input_dict (dict): The dictionary to print keys from.

    Raises:
        TypeError: If the input is not a dictionary.
    """
    try:
        # Check if the input is a dictionary
        if not isinstance(input_dict, dict):
            raise TypeError("Input must be a dictionary")

        # Print the keys of the dictionary
        for key in input_dict.keys():
            print(key)

    except TypeError as e:
        print(f"Error: {e}")

    except Exception as e:
        print(f"An unexpected error occurred: {e}")

# Example usage
my_dict = {"name": "John", "age": 30, "city": "New York"}
print_dict_keys(my_dict)

# Error handling example
print_dict_keys("Not a dictionary")
```
