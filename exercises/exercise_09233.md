## Exercise
Write a program that converts a string to an integer and a float to a string.

## Reference Solution
```python
def convert_string_to_int(input_str):
    """
    Converts a string to an integer.
    
    Args:
        input_str (str): The string to be converted.
    
    Returns:
        int: The integer equivalent of the input string.
    
    Raises:
        ValueError: If the input string cannot be converted to an integer.
    """
    try:
        return int(input_str)
    except ValueError:
        print("Error: The input string cannot be converted to an integer.")
        return None

def convert_float_to_str(input_float):
    """
    Converts a float to a string.
    
    Args:
        input_float (float): The float to be converted.
    
    Returns:
        str: The string equivalent of the input float.
    """
    return str(input_float)

def main():
    # Example usage:
    input_str = "123"
    input_float = 45.67
    
    # Convert string to integer
    result_int = convert_string_to_int(input_str)
    print(f"String '{input_str}' converted to integer: {result_int}")
    
    # Convert float to string
    result_str = convert_float_to_str(input_float)
    print(f"Float '{input_float}' converted to string: {result_str}")

if __name__ == "__main__":
    main()
```
