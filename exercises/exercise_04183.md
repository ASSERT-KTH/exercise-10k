## Exercise
Create a function that plays a game of Rock, Paper, Scissors against the user.

## Reference Solution
```python
import random

def play_rock_paper_scissors():
    """
    Play a game of Rock, Paper, Scissors against the user.
    """
    while True:
        # Get user's choice
        user_choice = input("Enter your choice (rock, paper, scissors) or 'q' to quit: ").lower()

        # Check if user wants to quit
        if user_choice == 'q':
            break

        # Validate user's choice
        if user_choice not in ['rock', 'paper', 'scissors']:
            print("Invalid choice. Please enter rock, paper, or scissors.")
            continue

        # Get computer's choice
        choices = ['rock', 'paper', 'scissors']
        computer_choice = random.choice(choices)

        # Print choices
        print(f"\nUser chose: {user_choice}")
        print(f"Computer chose: {computer_choice}\n")

        # Determine winner
        if user_choice == computer_choice:
            print(f"Both players selected {user_choice}. It's a tie!")
        elif user_choice == 'rock':
            if computer_choice == 'scissors':
                print("Rock smashes scissors! User wins!")
            else:
                print("Paper covers rock! Computer wins.")
        elif user_choice == 'paper':
            if computer_choice == 'rock':
                print("Paper covers rock! User wins!")
            else:
                print("Scissors cuts paper! Computer wins.")
        elif user_choice == 'scissors':
            if computer_choice == 'paper':
                print("Scissors cuts paper! User wins!")
            else:
                print("Rock smashes scissors! Computer wins.")

        # Ask user to play again
        play_again = input("Play again? (y/n): ").lower()
        if play_again != 'y':
            break

# Example usage
if __name__ == "__main__":
    play_rock_paper_scissors()
```
