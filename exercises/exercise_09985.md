## Exercise
Use a floating point variable to store a person's height and weight, then calculate their BMI

## Reference Solution
```python
def calculate_bmi(height, weight):
    """
    Calculate the BMI of a person given their height and weight.

    Args:
        height (float): The height of the person in meters.
        weight (float): The weight of the person in kilograms.

    Returns:
        float: The calculated BMI.
    """
    bmi = weight / (height ** 2)
    return bmi

def classify_bmi(bmi):
    """
    Classify the BMI into different categories.

    Args:
        bmi (float): The calculated BMI.

    Returns:
        str: The corresponding BMI category.
    """
    if bmi < 18.5:
        return "Underweight"
    elif bmi < 25:
        return "Normal"
    elif bmi < 30:
        return "Overweight"
    else:
        return "Obese"

def main():
    # Input height and weight
    height = float(input("Enter your height in meters: "))
    weight = float(input("Enter your weight in kilograms: "))

    # Calculate BMI
    bmi = calculate_bmi(height, weight)

    # Classify BMI
    classification = classify_bmi(bmi)

    # Print results
    print(f"Your BMI is: {bmi:.2f}")
    print(f"Your BMI classification is: {classification}")

if __name__ == "__main__":
    main()
```
