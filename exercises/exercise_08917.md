Write a program to find the longest increasing subsequence of a given list of numbers.

def longest_increasing_subsequence(sequence):
    """
    Find the longest increasing subsequence of a given list of numbers.

    Args:
        sequence (list): A list of integers.

    Returns:
        list: The longest increasing subsequence.
    """
    if not sequence:
        return []

    # Initialize a list to store the length of the LIS ending at each position
    lengths = [1] * len(sequence)

    # Initialize a list to store the previous element in the LIS ending at each position
    previous = [None] * len(sequence)

    # Compute the length of the LIS ending at each position
    for i in range(1, len(sequence)):
        for j in range(i):
            if sequence[i] > sequence[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                previous[i] = j

    # Find the position of the end of the LIS
    max_length_idx = max(range(len(lengths)), key=lambda x: lengths[x])

    # Reconstruct the LIS
    lis = []
    while max_length_idx is not None:
        lis.append(sequence[max_length_idx])
        max_length_idx = previous[max_length_idx]

    # Return the LIS in the correct order
    return lis[::-1]

    print("Longest increasing subsequence:", longest_increasing_subsequence(sequence))
