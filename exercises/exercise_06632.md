## Exercise
Create a function that uses memoization to cache its results

## Reference Solution
```python
from functools import wraps

def memoize(func):
    """Memoization decorator to cache function results"""
    cache = dict()
    
    @wraps(func)
    def memoized_func(*args):
        if args in cache:
            return cache[args]
        result = func(*args)
        cache[args] = result
        return result
    
    return memoized_func

@memoize
def fibonacci(n):
    """Calculate the nth Fibonacci number"""
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# Example usage:
if __name__ == "__main__":
    for i in range(10):
        print(f"Fibonacci({i}) = {fibonacci(i)}")
```
