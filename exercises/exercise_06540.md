## Exercise
Write a function that returns the leaderboard for a given game.

## Reference Solution
```python
def generate_leaderboard(player_scores):
    """
    Generate a leaderboard for a given game.

    Args:
        player_scores (dict): A dictionary where the keys are player names and the values are their scores.

    Returns:
        list: A sorted list of tuples, where each tuple contains the player's name and their score.
    """
    # Sort the player scores in descending order
    sorted_scores = sorted(player_scores.items(), key=lambda x: x[1], reverse=True)

    return sorted_scores

# Example usage:
def main():
    # Define the player scores
    player_scores = {
        "John": 100,
        "Alice": 200,
        "Bob": 50,
        "Charlie": 150
    }

    # Generate the leaderboard
    leaderboard = generate_leaderboard(player_scores)

    # Print the leaderboard
    print("Leaderboard:")
    for i, (player, score) in enumerate(leaderboard):
        print(f"{i+1}. {player}: {score}")

if __name__ == "__main__":
    main()
```
