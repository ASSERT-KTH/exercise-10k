## Exercise
Write a program that takes a string of digits as input and returns the string formatted as a phone number.

## Reference Solution
```python
def format_phone_number(digits):
    """
    Formats a string of digits as a phone number.

    Args:
        digits (str): A string of digits.

    Returns:
        str: The formatted phone number.

    Raises:
        ValueError: If the input string contains non-digit characters or has an invalid length.
    """
    if not digits.isdigit():
        raise ValueError("Input string must contain only digits")
    if len(digits) != 10:
        raise ValueError("Input string must be 10 digits long")

    # Format the phone number as (XXX) XXX-XXXX
    formatted_number = f"({digits[:3]}) {digits[3:6]}-{digits[6:]}"

    return formatted_number

# Example usage:
def main():
    digits = input("Enter a 10-digit phone number: ")
    try:
        formatted_number = format_phone_number(digits)
        print(f"Formatted phone number: {formatted_number}")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
