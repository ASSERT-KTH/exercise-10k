## Exercise
Create a program that reads a list of numbers from the user, calculates the mean, median, and standard deviation, and prints out these statistics.

## Reference Solution
```python
import numpy as np

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and standard deviation of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the mean, median, and standard deviation.
    """
    mean = np.mean(numbers)
    median = np.median(numbers)
    std_dev = np.std(numbers)

    return mean, median, std_dev

def main():
    # Read a list of numbers from the user
    numbers = input("Enter a list of numbers (space-separated): ")
    numbers = [float(num) for num in numbers.split()]

    # Calculate the statistics
    mean, median, std_dev = calculate_statistics(numbers)

    # Print out the statistics
    print("Statistics:")
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Standard Deviation: {std_dev}")

if __name__ == "__main__":
    main()
```
