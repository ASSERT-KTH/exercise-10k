## Exercise
Write a function to update a dictionary value by its key

## Reference Solution
```python
def update_dict_value(dictionary, key, new_value):
    """
    Updates a dictionary value by its key.

    Args:
        dictionary (dict): The dictionary to be updated.
        key (any): The key of the value to be updated.
        new_value (any): The new value to be assigned to the key.

    Returns:
        dict: The updated dictionary.

    Raises:
        KeyError: If the key does not exist in the dictionary.
    """
    if key in dictionary:
        dictionary[key] = new_value
    else:
        raise KeyError(f"The key '{key}' does not exist in the dictionary.")
    return dictionary

# Example usage:
if __name__ == "__main__":
    my_dict = {"name": "John", "age": 30}
    print("Original dictionary:", my_dict)

    updated_dict = update_dict_value(my_dict, "age", 31)
    print("Updated dictionary:", updated_dict)

    # Trying to update a non-existent key
    try:
        update_dict_value(my_dict, "city", "New York")
    except KeyError as e:
        print("Error:", e)
```
