## Exercise
Create a program that allows the user to add and remove expenses and track their budget.

## Reference Solution
```python
class BudgetTracker:
    def __init__(self):
        self.expenses = {}
        self.total_budget = 0
        self.total_expenses = 0

    def add_expense(self, name, amount):
        """Add a new expense to the tracker"""
        self.expenses[name] = amount
        self.total_expenses += amount
        print(f"Expense '{name}' added successfully.")

    def remove_expense(self, name):
        """Remove an existing expense from the tracker"""
        if name in self.expenses:
            amount = self.expenses[name]
            del self.expenses[name]
            self.total_expenses -= amount
            print(f"Expense '{name}' removed successfully.")
        else:
            print(f"Expense '{name}' not found.")

    def set_budget(self, amount):
        """Set the total budget"""
        self.total_budget = amount
        print(f"Budget set to ${amount:.2f}.")

    def track_expenses(self):
        """Print the current expenses and budget status"""
        print("\nCurrent Expenses:")
        for name, amount in self.expenses.items():
            print(f"{name}: ${amount:.2f}")
        print(f"\nTotal Expenses: ${self.total_expenses:.2f}")
        print(f"Total Budget: ${self.total_budget:.2f}")
        if self.total_expenses > self.total_budget:
            print("You are over budget.")
        elif self.total_expenses < self.total_budget:
            print("You are under budget.")
        else:
            print("You are at budget.")

def main():
    tracker = BudgetTracker()

    while True:
        print("\nBudget Tracker Menu:")
        print("1. Add Expense")
        print("2. Remove Expense")
        print("3. Set Budget")
        print("4. Track Expenses")
        print("5. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter expense name: ")
            amount = float(input("Enter expense amount: $"))
            tracker.add_expense(name, amount)
        elif choice == "2":
            name = input("Enter expense name: ")
            tracker.remove_expense(name)
        elif choice == "3":
            amount = float(input("Enter budget amount: $"))
            tracker.set_budget(amount)
        elif choice == "4":
            tracker.track_expenses()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
