def calculate_median(array):
    """
    Calculate the median of all elements in a given array.

    Args:
        array (list): A list of numbers.

    Returns:
        float: The median of the array.
    """
    sorted_array = sorted(array)
    n = len(sorted_array)

    if n % 2 == 0:
        # Even number of elements
        median = (sorted_array[n // 2 - 1] + sorted_array[n // 2]) / 2
    else:
        # Odd number of elements
        median = sorted_array[n // 2]

    return median

def main():
    # Example usage
    array = [1, 3, 5, 2, 4]
    median = calculate_median(array)
    print(f"Median: {median}")

if __name__ == "__main__":
    main()
