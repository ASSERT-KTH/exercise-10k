## Exercise
Create a function that generates all possible anagrams of a given word.

## Reference Solution
```python
import itertools

def generate_anagrams(word):
    """
    Generate all possible anagrams of a given word.

    Args:
    word (str): The input word.

    Returns:
    list: A list of all possible anagrams.
    """
    # Remove duplicate characters by converting the word to a set
    # Then, convert it back to a list and sort it
    unique_chars = sorted(list(set(word)))
    
    # Generate all permutations of the unique characters
    anagrams = [''.join(p) for p in itertools.permutations(word)]
    
    # Return the list of anagrams
    return anagrams

# Example usage:
word = "abc"
anagrams = generate_anagrams(word)
print(f"Anagrams of '{word}':")
for anagram in anagrams:
    print(anagram)
```

def generate_unique_anagrams(word):
    """
    Generate all possible anagrams of a given word without duplicate characters.

    Args:
    word (str): The input word.

    Returns:
    list: A list of all possible anagrams without duplicate characters.
    """
    # Remove duplicate characters by converting the word to a set
    unique_chars = set(word)
    
    # Generate all permutations of the unique characters
    anagrams = set(''.join(p) for p in itertools.permutations(word))
    
    # Return the list of anagrams
    return list(anagrams)

# Example usage:
