## Exercise
Write a program to validate an IP address.

## Reference Solution
```python
import ipaddress

def validate_ip_address(ip):
    """
    Validate an IP address.

    Args:
        ip (str): The IP address to validate.

    Returns:
        bool: True if the IP address is valid, False otherwise.
    """
    try:
        ipaddress.ip_address(ip)
        return True
    except ValueError:
        return False

def main():
    # Example usage
    ip_addresses = [
        "192.168.1.1",  # Valid IPv4 address
        "2001:0db8:85a3:0000:0000:8a2e:0370:7334",  # Valid IPv6 address
        "256.1.1.1",  # Invalid IPv4 address
        "2001:0db8:85a3:0000:0000:8a2e:0370:733g"  # Invalid IPv6 address
    ]

    for ip in ip_addresses:
        if validate_ip_address(ip):
            print(f"{ip} is a valid IP address")
        else:
            print(f"{ip} is not a valid IP address")

if __name__ == "__main__":
    main()
```
