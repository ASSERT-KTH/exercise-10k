## Exercise
Write a program that implements a Caesar Cipher using a while loop.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts or decrypts the input text using a Caesar Cipher.

    Args:
        text (str): The input text to be encrypted or decrypted.
        shift (int): The number of positions to shift each letter.

    Returns:
        str: The encrypted or decrypted text.
    """
    result = ""
    i = 0
    while i < len(text):
        char = text[i]
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            result += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            result += char
        i += 1
    return result

def main():
    text = input("Enter the text to be encrypted or decrypted: ")
    shift = int(input("Enter the shift value: "))
    direction = input("Do you want to (E)ncrypt or (D)ecrypt? ")

    if direction.upper() == "E":
        print("Encrypted text:", caesar_cipher(text, shift))
    elif direction.upper() == "D":
        print("Decrypted text:", caesar_cipher(text, -shift))
    else:
        print("Invalid direction. Please enter E or D.")

if __name__ == "__main__":
    main()
```
