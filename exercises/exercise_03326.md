## Exercise
Write a program that takes an array of integers as input and modifies it by doubling each element.

## Reference Solution
```python
#include <iostream>

// Function to double the elements of an array
void doubleArray(int arr[], int size) {
  for (int i = 0; i < size; i++) {
    arr[i] *= 2;
  }
}

// Function to print an array
void printArray(int arr[], int size) {
  for (int i = 0; i < size; i++) {
    std::cout << arr[i] << " ";
  }
  std::cout << std::endl;
}

int main() {
  int arr[] = {1, 2, 3, 4, 5};
  int size = sizeof(arr) / sizeof(arr[0]);

  std::cout << "Original array: ";
  printArray(arr, size);

  doubleArray(arr, size);

  std::cout << "Array after doubling: ";
  printArray(arr, size);

  return 0;
}
```
