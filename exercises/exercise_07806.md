## Exercise
Write a program that converts a distance from one unit to another (e.g. from miles to kilometers).

## Reference Solution
```python
def convert_distance(value, from_unit, to_unit):
    """
    Convert a distance from one unit to another.

    Args:
        value (float): The distance value to convert.
        from_unit (str): The unit of the input distance.
        to_unit (str): The unit to convert the distance to.

    Returns:
        float: The converted distance value.
    """

    # Define conversion factors
    conversion_factors = {
        'miles': {'kilometers': 1.60934, 'meters': 1609.34, 'yards': 1760, 'feet': 5280},
        'kilometers': {'miles': 0.621371, 'meters': 1000, 'yards': 1093.61, 'feet': 3280.84},
        'meters': {'kilometers': 0.001, 'miles': 0.000621371, 'yards': 1.09361, 'feet': 3.28084},
        'yards': {'miles': 0.000568182, 'kilometers': 0.0009144, 'meters': 0.9144, 'feet': 3},
        'feet': {'miles': 0.000189394, 'kilometers': 0.0003048, 'meters': 0.3048, 'yards': 0.333333}
    }

    # Check if the input units are valid
    if from_unit not in conversion_factors or to_unit not in conversion_factors:
        raise ValueError("Invalid unit")

    # Convert the distance
    if from_unit == to_unit:
        return value
    else:
        return value * conversion_factors[from_unit][to_unit]

def main():
    print("Distance Conversion Program")
    print("----------------------------")

    # Get user input
    value = float(input("Enter the distance value: "))
    from_unit = input("Enter the unit of the input distance (miles, kilometers, meters, yards, feet): ")
    to_unit = input("Enter the unit to convert the distance to (miles, kilometers, meters, yards, feet): ")

    # Convert the distance
    try:
        converted_value = convert_distance(value, from_unit, to_unit)
        print(f"{value} {from_unit} is equal to {converted_value} {to_unit}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
