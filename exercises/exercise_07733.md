## Exercise
Write a function that takes an array of integers as input and returns the reversed array. The function should not use any built-in reverse functions.

## Reference Solution
```python
def reverse_array(arr):
    """
    Reverses the input array without using built-in reverse functions.

    Args:
        arr (list): The input array of integers.

    Returns:
        list: The reversed array.
    """
    left = 0
    right = len(arr) - 1

    while left < right:
        # Swap the elements at the left and right indices
        arr[left], arr[right] = arr[right], arr[left]
        # Move the left and right indices towards the center
        left += 1
        right -= 1

    return arr

# Example usage
def main():
    arr = [1, 2, 3, 4, 5]
    print("Original array:", arr)
    reversed_arr = reverse_array(arr)
    print("Reversed array:", reversed_arr)

if __name__ == "__main__":
    main()
```
