## Exercise
Write a program that accesses a value in a dictionary by its key

## Reference Solution
```python
# Define a dictionary
my_dict = {
    "name": "John",
    "age": 30,
    "city": "New York"
}

# Function to access a dictionary value by key
def access_dict_value(dictionary, key):
    """
    Access a value in a dictionary by its key.

    Args:
        dictionary (dict): The dictionary to access.
        key (str): The key of the value to access.

    Returns:
        The value associated with the given key if it exists, otherwise None.
    """
    return dictionary.get(key)

# Example usage
if __name__ == "__main__":
    key = "name"
    value = access_dict_value(my_dict, key)
    
    if value is not None:
        print(f"The value for key '{key}' is: {value}")
    else:
        print(f"Key '{key}' not found in the dictionary.")
```
