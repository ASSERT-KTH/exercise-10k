## Exercise
Write a program to validate a password based on the following rules: at least 8 characters long, contains at least one uppercase letter, one lowercase letter, and one number.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the given rules.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the rules
    rules = [
        lambda x: len(x) >= 8,  # At least 8 characters long
        lambda x: any(c.isupper() for c in x),  # At least one uppercase letter
        lambda x: any(c.islower() for c in x),  # At least one lowercase letter
        lambda x: any(c.isdigit() for c in x)  # At least one number
    ]

    # Check each rule
    for rule in rules:
        if not rule(password):
            return False

    return True

def main():
    # Test the function
    passwords = [
        "weak",  # Too short
        "Strong",  # No number
        "Strong1",  # Valid
        "weak123"  # No uppercase letter
    ]

    for password in passwords:
        print(f"Password: {password}, Valid: {validate_password(password)}")

if __name__ == "__main__":
    main()
```
