## Exercise
Create a program to calculate the speed given the distance and time.

## Reference Solution
```python
def calculate_speed(distance, time):
    """
    Calculate the speed of an object.

    Args:
        distance (float): The distance traveled in meters.
        time (float): The time taken in seconds.

    Returns:
        float: The speed of the object in meters per second.
    """
    if time == 0:
        raise ValueError("Time cannot be zero.")
    speed = distance / time
    return speed

def main():
    print("Speed Calculator Program")
    print("------------------------")

    distance = float(input("Enter the distance traveled (in meters): "))
    time = float(input("Enter the time taken (in seconds): "))

    try:
        speed = calculate_speed(distance, time)
        print(f"The speed of the object is {speed} meters per second.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
