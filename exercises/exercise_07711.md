## Exercise
Implement a Caesar cipher, which shifts each letter in a given string by a specified number of positions down the alphabet.

## Reference Solution
```python
def caesar_cipher(text: str, shift: int, direction: str = 'encrypt') -> str:
    """
    Encrypts or decrypts a given text using the Caesar cipher.

    Args:
    - text (str): The input text to be encrypted or decrypted.
    - shift (int): The number of positions to shift the letters.
    - direction (str): The direction of the shift. Defaults to 'encrypt'.

    Returns:
    - str: The encrypted or decrypted text.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in text:
        if char.isalpha():
            # Find the position of the character in the alphabet
            position = alphabet.index(char.lower())
            
            if direction == 'encrypt':
                # Shift the character to the right
                new_position = (position + shift) % 26
            elif direction == 'decrypt':
                # Shift the character to the left
                new_position = (position - shift) % 26
            else:
                raise ValueError("Invalid direction. Use 'encrypt' or 'decrypt'.")

            # Preserve the original case of the character
            if char.isupper():
                result += alphabet[new_position].upper()
            else:
                result += alphabet[new_position]
        else:
            # Non-alphabet characters are not shifted
            result += char

    return result

def main():
    text = "Hello, World!"
    shift = 3

    encrypted_text = caesar_cipher(text, shift, direction='encrypt')
    decrypted_text = caesar_cipher(encrypted_text, shift, direction='decrypt')

    print(f"Original Text: {text}")
    print(f"Encrypted Text: {encrypted_text}")
    print(f"Decrypted Text: {decrypted_text}")

if __name__ == "__main__":
    main()
```
