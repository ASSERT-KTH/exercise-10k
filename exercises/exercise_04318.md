## Exercise
Write a function that handles errors using try-except blocks

## Reference Solution
```python
def divide_numbers(num1, num2):
    """
    Divide two numbers and handle potential errors.

    Args:
        num1 (float): The dividend.
        num2 (float): The divisor.

    Returns:
        float: The result of the division.

    Raises:
        TypeError: If either num1 or num2 is not a number.
        ZeroDivisionError: If num2 is zero.
    """
    try:
        # Attempt to divide num1 by num2
        result = num1 / num2
        return result
    except TypeError:
        # Handle the case where either num1 or num2 is not a number
        print("Error: Both inputs must be numbers.")
        return None
    except ZeroDivisionError:
        # Handle the case where num2 is zero
        print("Error: Cannot divide by zero.")
        return None
    except Exception as e:
        # Catch any other unexpected errors
        print(f"An unexpected error occurred: {e}")
        return None

# Example usage
if __name__ == "__main__":
    print(divide_numbers(10, 2))  # Expected output: 5.0
    print(divide_numbers(10, 0))  # Expected output: Error: Cannot divide by zero.
    print(divide_numbers(10, "a"))  # Expected output: Error: Both inputs must be numbers.
```
