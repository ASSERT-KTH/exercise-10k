## Exercise
Create a program that simulates a chatbot conversation with the user.

## Reference Solution
```python
import random

# Define a dictionary of chatbot responses
responses = {
    'hello': ['Hi, how are you?', 'Hello! What can I help you with today?', 'Hey, what\'s up?'],
    'how are you': ['I\'m doing well, thanks for asking!', 'I\'m good, thanks!', 'I\'m great, thanks!'],
    'what is your name': ['My name is Chatbot!', 'I\'m an AI assistant, but you can call me Chatbot.', 'I don\'t have a personal name, but I\'m here to help you!'],
    'default': ['I didn\'t understand that. Can you please rephrase?', 'Sorry, I didn\'t get that. Can you try again?', 'I\'m not sure what you mean. Can you explain?']
}

def get_response(user_input):
    """Get a chatbot response based on user input"""
    user_input = user_input.lower()
    for keyword in responses:
        if keyword in user_input:
            return random.choice(responses[keyword])
    return random.choice(responses['default'])

def main():
    print("Welcome to the chatbot conversation simulator!")
    print("Type 'quit' to exit the conversation.")
    while True:
        user_input = input("You: ")
        if user_input.lower() == 'quit':
            break
        print("Chatbot: ", get_response(user_input))

if __name__ == "__main__":
    main()
```
