## Exercise
Create a program that allows the user to add and remove items from a to-do list. The program should store the list in memory and print it out after each action.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)
        print(f"Task '{task}' added to the list.")

    def remove_task(self, task):
        if task in self.tasks:
            self.tasks.remove(task)
            print(f"Task '{task}' removed from the list.")
        else:
            print(f"Task '{task}' not found in the list.")

    def print_list(self):
        print("\nCurrent To-Do List:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")


def main():
    todo = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add task")
        print("2. Remove task")
        print("3. Print list")
        print("4. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            task = input("Enter a task: ")
            todo.add_task(task)
            todo.print_list()
        elif choice == "2":
            task = input("Enter a task to remove: ")
            todo.remove_task(task)
            todo.print_list()
        elif choice == "3":
            todo.print_list()
        elif choice == "4":
            print("Exiting the program.")
            break
        else:
            print("Invalid option. Please try again.")

if __name__ == "__main__":
    main()
```
