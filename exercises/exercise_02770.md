## Exercise
Write a program that takes a password as input and checks if it meets certain criteria (e.g. length, characters, numbers).

## Reference Solution
```python
import re

def check_password(password):
    """
    Checks a password against the following criteria:
    - Minimum length of 8 characters
    - At least one uppercase letter
    - At least one lowercase letter
    - At least one number
    - At least one special character

    Args:
        password (str): The password to check

    Returns:
        bool: True if the password meets all criteria, False otherwise
    """

    # Define the regular expression patterns for each criterion
    patterns = {
        'uppercase': r'[A-Z]',
        'lowercase': r'[a-z]',
        'number': r'\d',
        'special_char': r'[^A-Za-z0-9]'
    }

    # Check each criterion
    if len(password) < 8:
        print("Password must be at least 8 characters long.")
        return False

    for criterion, pattern in patterns.items():
        if not re.search(pattern, password):
            print(f"Password must contain at least one {criterion} character.")
            return False

    print("Password meets all criteria.")
    return True

def main():
    password = input("Enter a password: ")
    check_password(password)

if __name__ == "__main__":
    main()
```
