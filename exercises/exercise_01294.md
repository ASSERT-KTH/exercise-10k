## Exercise
Write a function that takes a dictionary and a predicate function, and returns a new dictionary containing only the key-value pairs for which the predicate function returns True.

## Reference Solution
```python
def filter_dict(dictionary, predicate):
    """
    This function filters a dictionary based on a given predicate function.
    
    Args:
        dictionary (dict): The dictionary to be filtered.
        predicate (function): A function that takes a key-value pair and returns a boolean value.
        
    Returns:
        dict: A new dictionary containing only the key-value pairs for which the predicate function returns True.
    """
    
    # Initialize an empty dictionary to store the filtered key-value pairs
    filtered_dict = {}
    
    # Iterate over each key-value pair in the input dictionary
    for key, value in dictionary.items():
        # Apply the predicate function to the current key-value pair
        if predicate(key, value):
            # If the predicate function returns True, add the key-value pair to the filtered dictionary
            filtered_dict[key] = value
    
    # Return the filtered dictionary
    return filtered_dict

# Example usage:
# Define a predicate function that checks if the value is greater than 10
def is_greater_than_10(key, value):
    return value > 10

# Create a sample dictionary
sample_dict = {'a': 5, 'b': 15, 'c': 20, 'd': 8}

# Filter the dictionary using the predicate function
filtered_dict = filter_dict(sample_dict, is_greater_than_10)

# Print the filtered dictionary
print(filtered_dict)  # Output: {'b': 15, 'c': 20}
```
