## Exercise
Write a program to calculate the standard deviation of a set of numbers.

## Reference Solution
```python
import math

def calculate_mean(numbers):
    """Calculate the mean of a set of numbers"""
    return sum(numbers) / len(numbers)

def calculate_variance(numbers, mean):
    """Calculate the variance of a set of numbers"""
    variance = sum((x - mean) ** 2 for x in numbers) / len(numbers)
    return variance

def calculate_standard_deviation(variance):
    """Calculate the standard deviation of a set of numbers"""
    return math.sqrt(variance)

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    mean = calculate_mean(numbers)
    variance = calculate_variance(numbers, mean)
    standard_deviation = calculate_standard_deviation(variance)
    
    print("Mean:", mean)
    print("Variance:", variance)
    print("Standard Deviation:", standard_deviation)

if __name__ == "__main__":
    main()
```
