## Exercise
Write a program that takes a range of numbers as input, calculates the prime numbers within that range, and prints them out.

// Author: [Your Name]

#include <iostream>
using namespace std;

int main() {
    int x, y;
    cin >> x >> y;
    for (int i = x; i <= y; i++) {
        bool flag = true;
        for (int j = 2; j <= i / 2; j++) {
            if (i % j == 0) {
                flag = false;
                break;
            }
        }
        if (flag && i > 1) {
            cout << i << endl;
        }
    }
    return 0;
}
