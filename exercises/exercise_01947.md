## Exercise
Write a program that takes an array and an index as input and removes the element at that index from the array.

## Reference Solution
```python
#include <stdio.h>

int main() {
    int n, i, k;
    int a[100];

    printf("Enter the size of the array: ");
    scanf("%d", &n);

    printf("Enter the elements of the array: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    printf("Enter the index to remove: ");
    scanf("%d", &k);

    printf("Original array: ");
    for (i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");

    if (k >= 1 && k <= n) {
        for (i = k - 1; i < n - 1; i++) {
            a[i] = a[i + 1];
        }
        n--;
    }

    printf("Array after removing the element at index %d: ", k);
    for (i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");

    return 0;
}
```
