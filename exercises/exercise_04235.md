## Exercise
Write a program to calculate the score of a student based on their grades.

## Reference Solution
```python
# Define a dictionary to store the grades and their corresponding scores
grade_scores = {
    'A': 90,
    'B': 80,
    'C': 70,
    'D': 60,
    'F': 0
}

def calculate_score(grades):
    """
    Calculate the total score based on the grades.

    Args:
        grades (dict): A dictionary where the keys are the subject names and the values are the grades.

    Returns:
        float: The total score.
    """
    total_score = 0
    total_subjects = len(grades)

    for grade in grades.values():
        if grade in grade_scores:
            total_score += grade_scores[grade]
        else:
            print(f"Invalid grade: {grade}")

    return total_score / total_subjects if total_subjects > 0 else 0

def main():
    # Get the student's name
    student_name = input("Enter the student's name: ")

    # Get the grades
    num_subjects = int(input("Enter the number of subjects: "))
    grades = {}

    for i in range(num_subjects):
        subject_name = input(f"Enter the name of subject {i+1}: ")
        grade = input(f"Enter the grade for {subject_name}: ").upper()
        grades[subject_name] = grade

    # Calculate the score
    score = calculate_score(grades)

    # Display the result
    print(f"Student Name: {student_name}")
    print(f"Total Score: {score:.2f}%")

if __name__ == "__main__":
    main()
```
