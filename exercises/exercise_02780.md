## Exercise
Write a program that takes a story with placeholders as input and asks the user to fill in the placeholders.

## Reference Solution
```python
def get_user_input(story):
    """
    Asks the user to fill in the placeholders in the story.

    Args:
        story (str): The story with placeholders.

    Returns:
        dict: A dictionary where the keys are the placeholder names and the values are the user's input.
    """
    # Find all placeholders in the story
    placeholders = []
    start = 0
    while True:
        start = story.find('{', start)
        if start == -1:
            break
        end = story.find('}', start)
        if end == -1:
            break
        placeholder = story[start+1:end]
        placeholders.append(placeholder)
        start = end + 1

    # Ask the user to fill in the placeholders
    user_input = {}
    for placeholder in placeholders:
        value = input(f"Enter a value for {placeholder}: ")
        user_input[placeholder] = value

    return user_input

def replace_placeholders(story, user_input):
    """
    Replaces the placeholders in the story with the user's input.

    Args:
        story (str): The story with placeholders.
        user_input (dict): A dictionary where the keys are the placeholder names and the values are the user's input.

    Returns:
        str: The story with the placeholders replaced.
    """
    for placeholder, value in user_input.items():
        story = story.replace('{' + placeholder + '}', value)

    return story

def main():
    story = input("Enter a story with placeholders (e.g., {name}, {place}, etc.): ")
    user_input = get_user_input(story)
    filled_story = replace_placeholders(story, user_input)
    print("Here is the filled-in story:")
    print(filled_story)

if __name__ == "__main__":
    main()
```
