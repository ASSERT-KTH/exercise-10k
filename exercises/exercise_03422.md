## Exercise
Write a function that takes a string as input and returns all palindromic substrings

## Reference Solution
```python
def find_palindromic_substrings(s):
    """
    Finds all palindromic substrings in a given string.

    Args:
    s (str): The input string.

    Returns:
    list: A list of all palindromic substrings.
    """
    def is_palindrome(substring):
        """Checks if a substring is a palindrome."""
        return substring == substring[::-1]

    palindromic_substrings = set()
    for length in range(1, len(s) + 1):
        for i in range(len(s) - length + 1):
            substring = s[i:i + length]
            if is_palindrome(substring):
                palindromic_substrings.add(substring)

    return list(palindromic_substrings)


# Example usage
if __name__ == "__main__":
    input_string = "abba"
    palindromic_substrings = find_palindromic_substrings(input_string)
    print("Palindromic substrings:", palindromic_substrings)
```
