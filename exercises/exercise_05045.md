## Exercise
Create a program that performs modular arithmetic operations on two numbers.

## Reference Solution
```python
def add_mod(a, b, mod):
    """
    Perform modular addition.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulus.

    Returns:
        int: The result of (a + b) mod mod.
    """
    return (a + b) % mod


def sub_mod(a, b, mod):
    """
    Perform modular subtraction.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulus.

    Returns:
        int: The result of (a - b) mod mod.
    """
    return (a - b) % mod


def mul_mod(a, b, mod):
    """
    Perform modular multiplication.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulus.

    Returns:
        int: The result of (a * b) mod mod.
    """
    return (a * b) % mod


def exp_mod(a, b, mod):
    """
    Perform modular exponentiation.

    Args:
        a (int): The base.
        b (int): The exponent.
        mod (int): The modulus.

    Returns:
        int: The result of a^b mod mod.
    """
    result = 1
    while b > 0:
        if b % 2 == 1:
            result = (result * a) % mod
        a = (a * a) % mod
        b //= 2
    return result


def main():
    print("Modular Arithmetic Program")
    print("-------------------------------")

    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))
    mod = int(input("Enter the modulus: "))

    print("\nOperations:")
    print("1. Addition")
    print("2. Subtraction")
    print("3. Multiplication")
    print("4. Exponentiation")

    choice = int(input("Enter your choice (1-4): "))

    if choice == 1:
        result = add_mod(num1, num2, mod)
        print(f"\n({num1} + {num2}) mod {mod} = {result}")
    elif choice == 2:
        result = sub_mod(num1, num2, mod)
        print(f"\n({num1} - {num2}) mod {mod} = {result}")
    elif choice == 3:
        result = mul_mod(num1, num2, mod)
        print(f"\n({num1} * {num2}) mod {mod} = {result}")
    elif choice == 4:
        result = exp_mod(num1, num2, mod)
        print(f"\n{num1}^{num2} mod {mod} = {result}")
    else:
        print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
