## Exercise
Create a program that generates a random number between 1 and 10, and asks the user to guess it. If the user's guess is incorrect, provide a hint (e.g., 'too high' or 'too low').

## Reference Solution
```python
import random

def get_user_guess():
    """Get the user's guess."""
    while True:
        try:
            guess = int(input("Guess a number between 1 and 10: "))
            if 1 <= guess <= 10:
                return guess
            else:
                print("Please enter a number between 1 and 10.")
        except ValueError:
            print("Invalid input. Please enter a number.")

def get_hint(guess, target):
    """Provide a hint based on the user's guess."""
    if guess < target:
        return "too low"
    elif guess > target:
        return "too high"
    else:
        return "correct"

def main():
    # Generate a random number between 1 and 10
    target = random.randint(1, 10)
    
    # Initialize the number of attempts
    attempts = 0
    
    while True:
        # Get the user's guess
        guess = get_user_guess()
        
        # Increment the number of attempts
        attempts += 1
        
        # Get the hint
        hint = get_hint(guess, target)
        
        # Print the hint
        if hint == "correct":
            print(f"Congratulations! You guessed the number in {attempts} attempts.")
            break
        else:
            print(f"Your guess is {hint}. Try again!")

if __name__ == "__main__":
    main()
```
