```python
public class DataTypeConversion {
    public static void main(String[] args) {
        // Integer to Float
        int intVal = 10;
        float floatVal = (float) intVal;
        System.out.println(floatVal);  // Output: 10.0

        // Float to Integer
        floatVal = 10.5f;
        intVal = (int) floatVal;
        System.out.println(intVal);  // Output: 10

        // Integer to String
        intVal = 10;
        String strVal = Integer.toString(intVal);
        System.out.println(strVal);  // Output: "10"

        // Float to String
        floatVal = 10.5f;
        strVal = Float.toString(floatVal);
        System.out.println(strVal);  // Output: "10.5"

        // String to Integer
        strVal = "10";
        intVal = Integer.parseInt(strVal);
        System.out.println(intVal);  // Output: 10

        // String to Float
        strVal = "10.5";
        floatVal = Float.parseFloat(strVal);
        System.out.println(floatVal);  // Output: 10.5

        // Boolean to Integer
        boolean boolVal = true;
        intVal = boolVal ? 1 : 0;
        System.out.println(intVal);  // Output: 1

        // Boolean to String
        boolVal = true;
        strVal = Boolean.toString(boolVal);
        System.out.println(strVal);  // Output: "true"
    }
}
```
