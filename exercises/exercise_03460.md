## Exercise
Create a program that helps users manage their personal finances, where the user can track expenses, income, and budget

## Reference Solution
```python
import datetime

class Transaction:
    def __init__(self, date, type, category, amount, description):
        self.date = date
        self.type = type
        self.category = category
        self.amount = amount
        self.description = description

class Budget:
    def __init__(self):
        self.income = 0
        self.expenses = 0
        self.categories = {}

    def add_category(self, category, amount):
        self.categories[category] = amount

    def update_income(self, amount):
        self.income += amount

    def update_expenses(self, amount):
        self.expenses += amount

class FinanceManager:
    def __init__(self):
        self.transactions = []
        self.budget = Budget()

    def add_transaction(self, date, type, category, amount, description):
        transaction = Transaction(date, type, category, amount, description)
        self.transactions.append(transaction)
        if type == "income":
            self.budget.update_income(amount)
        else:
            self.budget.update_expenses(amount)

    def edit_transaction(self, index, date, type, category, amount, description):
        self.transactions[index].date = date
        self.transactions[index].type = type
        self.transactions[index].category = category
        self.transactions[index].amount = amount
        self.transactions[index].description = description

    def delete_transaction(self, index):
        del self.transactions[index]

    def view_transactions(self):
        for i, transaction in enumerate(self.transactions):
            print(f"{i+1}. {transaction.date} - {transaction.type} - {transaction.category} - {transaction.amount} - {transaction.description}")

    def view_budget(self):
        print(f"Income: {self.budget.income}")
        print(f"Expenses: {self.budget.expenses}")
        print(f"Categories: {self.budget.categories}")

def main():
    finance_manager = FinanceManager()

    while True:
        print("\n1. Add Transaction")
        print("2. Edit Transaction")
        print("3. Delete Transaction")
        print("4. View Transactions")
        print("5. View Budget")
        print("6. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            date = input("Enter date (YYYY-MM-DD): ")
            type = input("Enter type (income/expense): ")
            category = input("Enter category: ")
            amount = float(input("Enter amount: "))
            description = input("Enter description: ")
            finance_manager.add_transaction(date, type, category, amount, description)
        elif choice == "2":
            finance_manager.view_transactions()
            index = int(input("Enter the index of the transaction to edit: ")) - 1
            date = input("Enter new date (YYYY-MM-DD): ")
            type = input("Enter new type (income/expense): ")
            category = input("Enter new category: ")
            amount = float(input("Enter new amount: "))
            description = input("Enter new description: ")
            finance_manager.edit_transaction(index, date, type, category, amount, description)
        elif choice == "3":
            finance_manager.view_transactions()
            index = int(input("Enter the index of the transaction to delete: ")) - 1
            finance_manager.delete_transaction(index)
        elif choice == "4":
            finance_manager.view_transactions()
        elif choice == "5":
            finance_manager.view_budget()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
