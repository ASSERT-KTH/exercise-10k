## Exercise
Convert a string to an integer and vice versa, then print out the results.

## Reference Solution
```python
def convert_string_to_int(input_str):
    """Converts a string to an integer."""
    try:
        return int(input_str)
    except ValueError:
        print("Error: Input string is not a valid integer.")
        return None

def convert_int_to_string(input_int):
    """Converts an integer to a string."""
    return str(input_int)

def main():
    # String to integer conversion
    input_str = "123"
    output_int = convert_string_to_int(input_str)
    if output_int is not None:
        print(f"String '{input_str}' converted to integer: {output_int}")

    # Integer to string conversion
    input_int = 456
    output_str = convert_int_to_string(input_int)
    print(f"Integer {input_int} converted to string: '{output_str}'")

if __name__ == "__main__":
    main()
```
