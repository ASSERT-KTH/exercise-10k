## Exercise
Write a program that demonstrates the concept of floating point precision by performing arithmetic operations on floating point numbers.

## Reference Solution
```python
def floating_point_precision():
    """
    Demonstrates the concept of floating point precision by performing arithmetic operations on floating point numbers.
    """
    # Define two floating point numbers
    num1 = 0.1
    num2 = 0.2

    # Perform addition
    sum_result = num1 + num2
    print(f"Sum of {num1} and {num2}: {sum_result}")

    # Perform subtraction
    diff_result = num1 - num2
    print(f"Difference of {num1} and {num2}: {diff_result}")

    # Perform multiplication
    product_result = num1 * num2
    print(f"Product of {num1} and {num2}: {product_result}")

    # Perform division
    quotient_result = num1 / num2
    print(f"Quotient of {num1} and {num2}: {quotient_result}")

    # Check for equality
    if num1 + num2 == 0.3:
        print("Equality check: 0.1 + 0.2 equals 0.3")
    else:
        print("Equality check: 0.1 + 0.2 does not equal 0.3")

    # Check for rounding errors
    print("Rounding errors:")
    print(f"0.1 + 0.2 = {num1 + num2}")
    print(f"0.1 + 0.2 == 0.3: {num1 + num2 == 0.3}")

if __name__ == "__main__":
    floating_point_precision()
```
