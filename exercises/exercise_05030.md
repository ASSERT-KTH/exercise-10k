## Exercise
Write a program that simulates a flight reservation system using for loops and while loops.

## Reference Solution
```python
class Flight:
    def __init__(self, flight_number, departure, arrival, seats):
        self.flight_number = flight_number
        self.departure = departure
        self.arrival = arrival
        self.seats = seats
        self.bookings = []

class FlightReservationSystem:
    def __init__(self):
        self.flights = []
        self.bookings = {}

    def add_flight(self, flight):
        self.flights.append(flight)

    def view_flights(self):
        print("Available Flights:")
        for i, flight in enumerate(self.flights):
            print(f"{i+1}. Flight Number: {flight.flight_number}, Departure: {flight.departure}, Arrival: {flight.arrival}, Seats: {flight.seats}")

    def book_flight(self, flight_number, passenger_name):
        for flight in self.flights:
            if flight.flight_number == flight_number:
                if len(flight.bookings) < flight.seats:
                    flight.bookings.append(passenger_name)
                    if passenger_name not in self.bookings:
                        self.bookings[passenger_name] = []
                    self.bookings[passenger_name].append(flight_number)
                    print(f"Flight {flight_number} booked successfully for {passenger_name}.")
                    return
                else:
                    print(f"Flight {flight_number} is fully booked.")
                    return
        print(f"Flight {flight_number} not found.")

    def cancel_booking(self, passenger_name, flight_number):
        if passenger_name in self.bookings:
            if flight_number in self.bookings[passenger_name]:
                for flight in self.flights:
                    if flight.flight_number == flight_number:
                        flight.bookings.remove(passenger_name)
                        self.bookings[passenger_name].remove(flight_number)
                        print(f"Booking for flight {flight_number} cancelled successfully for {passenger_name}.")
                        return
                print(f"Flight {flight_number} not found.")
                return
            else:
                print(f"{passenger_name} has no booking for flight {flight_number}.")
                return
        else:
            print(f"{passenger_name} has no bookings.")
            return

    def view_bookings(self, passenger_name):
        if passenger_name in self.bookings:
            print(f"Bookings for {passenger_name}:")
            for flight_number in self.bookings[passenger_name]:
                for flight in self.flights:
                    if flight.flight_number == flight_number:
                        print(f"Flight Number: {flight_number}, Departure: {flight.departure}, Arrival: {flight.arrival}")
        else:
            print(f"{passenger_name} has no bookings.")

def main():
    system = FlightReservationSystem()

    # Add flights
    flight1 = Flight("AA101", "New York", "Los Angeles", 100)
    flight2 = Flight("AA102", "Los Angeles", "New York", 100)
    system.add_flight(flight1)
    system.add_flight(flight2)

    while True:
        print("\nFlight Reservation System Menu:")
        print("1. View Flights")
        print("2. Book Flight")
        print("3. Cancel Booking")
        print("4. View Bookings")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            system.view_flights()
        elif choice == "2":
            system.view_flights()
            flight_number = input("Enter flight number: ")
            passenger_name = input("Enter passenger name: ")
            system.book_flight(flight_number, passenger_name)
        elif choice == "3":
            passenger_name = input("Enter passenger name: ")
            flight_number = input("Enter flight number: ")
            system.cancel_booking(passenger_name, flight_number)
        elif choice == "4":
            passenger_name = input("Enter passenger name: ")
            system.view_bookings(passenger_name)
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
