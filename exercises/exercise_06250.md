## Exercise
Write a function that takes a dictionary as input and returns a copy of the dictionary

## Reference Solution
```python
def copy_dictionary(input_dict):
    """
    Creates a copy of the input dictionary.

    Args:
    input_dict (dict): The dictionary to be copied.

    Returns:
    dict: A copy of the input dictionary.
    """
    return input_dict.copy()

# Example usage
original_dict = {"a": 1, "b": 2, "c": 3}
copied_dict = copy_dictionary(original_dict)

print("Original Dictionary:", original_dict)
print("Copied Dictionary:", copied_dict)

# Modify the original dictionary to demonstrate the copy
original_dict["d"] = 4
print("After modifying the original dictionary:")
print("Original Dictionary:", original_dict)
print("Copied Dictionary:", copied_dict)
```

```python
import copy

def copy_dictionary(input_dict):
    """
    Creates a deep copy of the input dictionary.

    Args:
    input_dict (dict): The dictionary to be copied.

    Returns:
    dict: A deep copy of the input dictionary.
    """
    return copy.deepcopy(input_dict)

# Example usage
original_dict = {"a": 1, "b": [2, 3], "c": {"d": 4}}
copied_dict = copy_dictionary(original_dict)

print("Original Dictionary:", original_dict)
print("Copied Dictionary:", copied_dict)

# Modify the original dictionary to demonstrate the deep copy
original_dict["b"].append(5)
original_dict["c"]["e"] = 6
print("After modifying the original dictionary:")
print("Original Dictionary:", original_dict)
print("Copied Dictionary:", copied_dict)
```
