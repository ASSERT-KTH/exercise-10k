## Exercise
Find the maximum contiguous subarray sum of an array of integers. The function should take in an array of integers as input and return the maximum subarray sum.

def max_subarray_sum(nums):
    """
    This function finds the maximum contiguous subarray sum of an array of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The maximum subarray sum.
    """
    
    # Check if the input list is empty
    if not nums:
        return 0

    # Initialize the maximum current sum and the maximum global sum to the first element of the array
    # This is because the maximum subarray sum could be the first element itself if all other elements are negative
    max_current = max_global = nums[0]
    
    # Iterate over the array starting from the second element (index 1)
    for num in nums[1:]:
        
        # Update the maximum current sum to be the maximum of the current number and the sum of the current number and the previous maximum current sum
        # This is because the maximum subarray sum ending at the current position could be either the current number itself or the sum of the current number and the previous maximum subarray sum
        max_current = max(num, max_current + num)
        
        # Update the maximum global sum to be the maximum of the current maximum global sum and the maximum current sum
        # This is because the maximum subarray sum could be the maximum of the current maximum subarray sum and the maximum subarray sum ending at the current position
        max_global = max(max_global, max_current)
    
    # Return the maximum global sum
    return max_global

# Example usage:
nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
print(max_subarray_sum(nums))  # Output: 6
