## Exercise
Write a function to determine the winner of a Rock, Paper, Scissors game given the choices of two players.

## Reference Solution
```python
def determine_winner(player1_choice, player2_choice):
    """
    Determine the winner of a Rock, Paper, Scissors game.

    Args:
        player1_choice (str): The choice of player 1. Can be 'Rock', 'Paper', or 'Scissors'.
        player2_choice (str): The choice of player 2. Can be 'Rock', 'Paper', or 'Scissors'.

    Returns:
        str: The winner of the game. Can be 'Player 1', 'Player 2', or 'Tie'.
    """

    # Define the winning conditions
    winning_conditions = {
        'Rock': 'Scissors',
        'Paper': 'Rock',
        'Scissors': 'Paper'
    }

    # Check if the choices are valid
    if player1_choice not in winning_conditions or player2_choice not in winning_conditions:
        raise ValueError("Invalid choice. Choices must be 'Rock', 'Paper', or 'Scissors'.")

    # Check for a tie
    if player1_choice == player2_choice:
        return 'Tie'

    # Determine the winner
    if winning_conditions[player1_choice] == player2_choice:
        return 'Player 1'
    else:
        return 'Player 2'

# Example usage
def main():
    print(determine_winner('Rock', 'Scissors'))  # Output: Player 1
    print(determine_winner('Paper', 'Rock'))  # Output: Player 1
    print(determine_winner('Scissors', 'Paper'))  # Output: Player 1
    print(determine_winner('Rock', 'Rock'))  # Output: Tie
    print(determine_winner('Paper', 'Scissors'))  # Output: Player 2

if __name__ == "__main__":
    main()
```
