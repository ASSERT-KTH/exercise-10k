## Exercise
Write a program that allows the user to play a text-based adventure game

## Reference Solution
```python
# Import the `random` module for generating random numbers
import random

# Define a dictionary to store the game world
game_world = {
    "start": {
        "description": "You are at the entrance of a dark forest.",
        "exits": {
            "north": "forest",
            "south": "village"
        },
        "items": ["sword", "shield"]
    },
    "forest": {
        "description": "You are in a dense forest.",
        "exits": {
            "south": "start",
            "east": "cave"
        },
        "items": ["apple", "water"]
    },
    "village": {
        "description": "You are in a small village.",
        "exits": {
            "north": "start",
            "west": "shop"
        },
        "items": ["gold", "food"]
    },
    "cave": {
        "description": "You are in a dark cave.",
        "exits": {
            "west": "forest"
        },
        "items": ["diamond", "ruby"]
    },
    "shop": {
        "description": "You are in a shop.",
        "exits": {
            "east": "village"
        },
        "items": ["sword", "shield", "armor"]
    }
}

# Define a dictionary to store the player's inventory
player_inventory = {
    "items": [],
    "gold": 0
}

# Define a function to display the game world
def display_game_world(location):
    print(game_world[location]["description"])
    print("Exits:")
    for exit, destination in game_world[location]["exits"].items():
        print(f"- {exit}: {destination}")
    print("Items:")
    for item in game_world[location]["items"]:
        print(f"- {item}")

# Define a function to handle player input
def handle_player_input(location, command):
    if command == "quit":
        print("Thanks for playing!")
        return False
    elif command == "inventory":
        print("Your inventory:")
        for item in player_inventory["items"]:
            print(f"- {item}")
        print(f"Gold: {player_inventory['gold']}")
    elif command.startswith("go "):
        direction = command[3:]
        if direction in game_world[location]["exits"]:
            return game_world[location]["exits"][direction]
        else:
            print("You cannot go that way.")
    elif command.startswith("take "):
        item = command[5:]
        if item in game_world[location]["items"]:
            player_inventory["items"].append(item)
            game_world[location]["items"].remove(item)
            print(f"You took the {item}.")
        else:
            print("There is no such item here.")
    elif command.startswith("buy "):
        item = command[4:]
        if item in game_world[location]["items"]:
            if player_inventory["gold"] >= 10:
                player_inventory["items"].append(item)
                game_world[location]["items"].remove(item)
                player_inventory["gold"] -= 10
                print(f"You bought the {item}.")
            else:
                print("You do not have enough gold.")
        else:
            print("There is no such item here.")
    else:
        print("Invalid command.")
    return location

# Define a function to play the game
def play_game():
    current_location = "start"
    while True:
        display_game_world(current_location)
        command = input("> ").lower()
        current_location = handle_player_input(current_location, command)
        if current_location is False:
            break

# Play the game
play_game()
```
