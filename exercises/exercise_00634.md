## Exercise
Write a program that calculates the monthly payment of a loan given the principal amount, interest rate, and loan term.

## Reference Solution
```python
def calculate_monthly_payment(principal, interest_rate, loan_term):
    """
    Calculate the monthly payment of a loan.

    Args:
    - principal (float): The principal amount of the loan.
    - interest_rate (float): The annual interest rate of the loan as a decimal.
    - loan_term (int): The loan term in years.

    Returns:
    - float: The monthly payment amount.
    """
    monthly_interest_rate = interest_rate / 12
    number_of_payments = loan_term * 12

    monthly_payment = principal * monthly_interest_rate * (1 + monthly_interest_rate) ** number_of_payments / (
                (1 + monthly_interest_rate) ** number_of_payments - 1)

    return round(monthly_payment, 2)


def main():
    principal = float(input("Enter the principal amount: $"))
    interest_rate = float(input("Enter the annual interest rate as a percentage (e.g., 6 for 6%): ")) / 100
    loan_term = int(input("Enter the loan term in years: "))

    monthly_payment = calculate_monthly_payment(principal, interest_rate, loan_term)

    print(f"The monthly payment is: ${monthly_payment}")


if __name__ == "__main__":
    main()
```
