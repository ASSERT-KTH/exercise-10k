## Exercise
Create type-safe functions that can handle multiple data types while preventing type-related errors.

## Reference Solution
```python
def add_values(a, b):
    """
    This function adds two values together.
    
    Args:
    a (int or float): The first number to add.
    b (int or float): The second number to add.
    
    Returns:
    int or float: The sum of a and b.
    """
    
    # Check if both a and b are numbers
    if not isinstance(a, (int, float)) or not isinstance(b, (int, float)):
        raise TypeError("Both values must be numbers.")
    
    # If both checks pass, then we can safely add the values
    return a + b


def multiply_values(a, b):
    """
    This function multiplies two values together.
    
    Args:
    a (int or float): The first number to multiply.
    b (int or float): The second number to multiply.
    
    Returns:
    int or float: The product of a and b.
    """
    
    # Check if both a and b are numbers
    if not isinstance(a, (int, float)) or not isinstance(b, (int, float)):
        raise TypeError("Both values must be numbers.")
    
    # If both checks pass, then we can safely multiply the values
    return a * b


def concatenate_strings(a, b):
    """
    This function concatenates two strings together.
    
    Args:
    a (str): The first string to concatenate.
    b (str): The second string to concatenate.
    
    Returns:
    str: The concatenation of a and b.
    """
    
    # Check if both a and b are strings
    if not isinstance(a, str) or not isinstance(b, str):
        raise TypeError("Both values must be strings.")
    
    # If both checks pass, then we can safely concatenate the strings
    return a + b


# Test the functions
print(add_values(5, 7))  # Outputs: 12
print(multiply_values(5, 7))  # Outputs: 35
print(concatenate_strings("Hello", "World"))  # Outputs: "HelloWorld"
```
