    Checks if a specific student is in the dictionary.

    Args:
        grades_dict (dict): Dictionary with student names as keys and grades as values.
        student_name (str): Name of the student to check.

    Returns:
        bool: True if the student is in the dictionary, False otherwise.
    """

    # Check if the student name is in the dictionary keys
    return student_name in grades_dict


# Example usage:
grades_dict = {
    "John Doe": 85,
    "Jane Doe": 90,
    "Bob Smith": 78
}

student_to_check = "Jane Doe"

if check_student(grades_dict, student_to_check):
