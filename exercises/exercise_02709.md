## Exercise
Write a program to count the frequency of each word in a given text.

## Reference Solution
```python
import re
from collections import Counter

def count_word_frequency(text):
    """
    Counts the frequency of each word in the given text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary where the keys are the unique words and the values are their frequencies.
    """
    # Convert text to lowercase
    text = text.lower()

    # Remove punctuation and special characters
    text = re.sub(r'[^\w\s]', '', text)

    # Split text into words
    words = text.split()

    # Count word frequency
    frequency = Counter(words)

    return frequency

def main():
    # Example usage
    text = "This is an example sentence. This sentence is just an example."
    frequency = count_word_frequency(text)

    print("Word Frequency:")
    for word, count in frequency.most_common():
        print(f"{word}: {count}")

if __name__ == "__main__":
    main()
```
