## Exercise
Write a program that calculates the greatest common divisor (GCD) of two numbers.

## Reference Solution
```python
import math

def calculate_gcd(num1, num2):
    """
    Calculate the greatest common divisor (GCD) of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The GCD of num1 and num2.
    """
    while num2 != 0:
        num1, num2 = num2, num1 % num2
    return abs(num1)

def main():
    # Example usage:
    num1 = 48
    num2 = 18
    gcd = calculate_gcd(num1, num2)
    print(f"The GCD of {num1} and {num2} is {gcd}.")

    # Using the math library's gcd function for comparison:
    math_gcd = math.gcd(num1, num2)
    print(f"The math library's GCD of {num1} and {num2} is {math_gcd}.")

if __name__ == "__main__":
    main()
```
