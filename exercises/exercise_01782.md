Write a function to merge two dictionaries into one, handling duplicate keys by summing their values.

```python
def merge_dicts(dict1, dict2):
    """
    Merge two dictionaries into one, handling duplicate keys by summing their values.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:
        dict: A new dictionary containing the merged key-value pairs.
    """
    merged_dict = dict1.copy()  # Create a copy of the first dictionary
    for key, value in dict2.items():
        if key in merged_dict:
            # If the key is already present, sum the values
            merged_dict[key] += value
        else:
            # If the key is not present, add it to the merged dictionary
            merged_dict[key] = value
    return merged_dict

# Example usage
dict1 = {'a': 10, 'b': 20, 'c': 30}
dict2 = {'b': 40, 'c': 50, 'd': 60}

merged = merge_dicts(dict1, dict2)
print(merged)  # Output: {'a': 10, 'b': 60, 'c': 80, 'd': 60}
```

```python
def merge_dicts(dict1, dict2):
    """
    Merge two dictionaries into one, handling duplicate keys by summing their values.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:
        dict: A new dictionary containing the merged key-value pairs.
    """
    all_keys = set(dict1.keys()) | set(dict2.keys())
    return {key: dict1.get(key, 0) + dict2.get(key, 0) for key in all_keys}

# Example usage
dict1 = {'a': 10, 'b': 20, 'c': 30}
dict2 = {'b': 40, 'c': 50, 'd': 60}

merged = merge_dicts(dict1, dict2)
```
