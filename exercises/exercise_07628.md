## Exercise
Create a program that uses a while loop to simulate a game where the user has to guess a randomly chosen color.

## Reference Solution
```python
import random

def color_guessing_game():
    # List of possible colors
    colors = ["red", "green", "blue", "yellow", "purple", "orange"]

    # Randomly select a color
    target_color = random.choice(colors)

    # Initialize the number of attempts
    attempts = 0

    print("Welcome to the color guessing game!")
    print("I'm thinking of a color. Try to guess it!")

    while True:
        # Ask the user for their guess
        user_guess = input("Enter your guess: ").lower()

        # Increment the number of attempts
        attempts += 1

        # Check if the user's guess is correct
        if user_guess == target_color:
            print(f"Congratulations! You guessed the color in {attempts} attempts.")
            break
        else:
            print("Incorrect guess. Try again!")

            # Provide a hint if the user's guess is not correct
            if user_guess in colors:
                if target_color in ["red", "orange", "yellow"]:
                    print("Hint: The color I'm thinking of is a warm color.")
                elif target_color in ["green", "blue", "purple"]:
                    print("Hint: The color I'm thinking of is a cool color.")
            else:
                print("Hint: The color I'm thinking of is one of the following: red, green, blue, yellow, purple, orange")

if __name__ == "__main__":
    color_guessing_game()
```
