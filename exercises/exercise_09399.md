## Exercise
Write a program that calculates the hyperbolic sine, cosine, and tangent of a given number.

## Reference Solution
```python
import math

def calculate_hyperbolic_functions(x):
    # Calculate hyperbolic sine
    sinh_x = (math.exp(x) - math.exp(-x)) / 2
    # Calculate hyperbolic cosine
    cosh_x = (math.exp(x) + math.exp(-x)) / 2
    # Calculate hyperbolic tangent
    if cosh_x == 0:
        print("Error: Division by zero in calculating hyperbolic tangent.")
        return None, None, None
    else:
        tanh_x = sinh_x / cosh_x
    return sinh_x, cosh_x, tanh_x

def main():
    num = float(input("Enter a number: "))
    sinh_x, cosh_x, tanh_x = calculate_hyperbolic_functions(num)
    if sinh_x is not None:
        print(f"The hyperbolic sine of {num} is: {sinh_x}")
        print(f"The hyperbolic cosine of {num} is: {cosh_x}")
        print(f"The hyperbolic tangent of {num} is: {tanh_x}")

if __name__ == "__main__":
    main()
```
