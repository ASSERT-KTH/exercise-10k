## Exercise
Write a program that asks the user for a list of events with their start and end times, and then prints out a schedule of the events for the day.

## Reference Solution
```python
from datetime import datetime

class Event:
    def __init__(self, name, start_time, end_time):
        self.name = name
        self.start_time = datetime.strptime(start_time, "%H:%M")
        self.end_time = datetime.strptime(end_time, "%H:%M")

    def __str__(self):
        return f"{self.name}: {self.start_time.strftime('%H:%M')} - {self.end_time.strftime('%H:%M')}"

def get_events_from_user():
    events = []
    num_events = int(input("Enter the number of events: "))

    for i in range(num_events):
        name = input(f"Enter the name of event {i+1}: ")
        start_time = input(f"Enter the start time of event {i+1} (HH:MM): ")
        end_time = input(f"Enter the end time of event {i+1} (HH:MM): ")

        events.append(Event(name, start_time, end_time))

    return events

def print_schedule(events):
    events.sort(key=lambda x: x.start_time)

    for event in events:
        print(event)

def main():
    events = get_events_from_user()
    print("\nSchedule for the day:")
    print_schedule(events)

if __name__ == "__main__":
    main()
```
