## Exercise
Read a positive integer from the user and calculate the sum of its digits

## Reference Solution
```python
def calculate_digit_sum(n):
    """
    Calculate the sum of digits of a given integer.

    Args:
        n (int): The input integer.

    Returns:
        int: The sum of digits of the input integer.
    """
    return sum(int(digit) for digit in str(n))

def main():
    # Read a positive integer from the user
    while True:
        try:
            num = int(input("Enter a positive integer: "))
            if num > 0:
                break
            else:
                print("Please enter a positive integer.")
        except ValueError:
            print("Invalid input. Please enter an integer.")

    # Calculate and print the sum of its digits
    digit_sum = calculate_digit_sum(num)
    print(f"The sum of the digits of {num} is: {digit_sum}")

if __name__ == "__main__":
    main()
```
