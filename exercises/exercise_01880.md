## Exercise
Write a program to apply multiple transformations to each element of a list and print the resulting list.

## Reference Solution
```python
def apply_transformations(input_list, transformations):
    """
    Applies multiple transformations to each element of a list and returns the resulting list.

    Args:
        input_list (list): The input list of elements.
        transformations (list): A list of transformation functions.

    Returns:
        list: The resulting list after applying all transformations.
    """
    result = []
    for element in input_list:
        transformed_element = element
        for transformation in transformations:
            transformed_element = transformation(transformed_element)
        result.append(transformed_element)
    return result

# Example transformations
def double(x):
    """Doubles the input value."""
    return x * 2

def add_one(x):
    """Adds one to the input value."""
    return x + 1

def square(x):
    """Squares the input value."""
    return x ** 2

# Example usage
input_list = [1, 2, 3, 4, 5]
transformations = [double, add_one, square]

result = apply_transformations(input_list, transformations)
print("Input List:", input_list)
print("Transformations:", [func.__name__ for func in transformations])
print("Resulting List:", result)
```
