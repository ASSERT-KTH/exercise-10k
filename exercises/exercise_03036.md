## Exercise
Write a program that takes a string and a width as input, and returns the string justified to the specified width.

## Reference Solution
```python
def justify_words(string, width):
    words = string.split()
    justified_string = ''
    line = ''

    for word in words:
        if len(line) + len(word) + 1 <= width:
            line += word + ' '
        else:
            justified_string += justify(line, width) + '\n'
            line = word + ' '

    justified_string += justify(line, width) + '\n'
    return justified_string

def justify(line, width):
    words = line.split()
    num_words = len(words)
    num_spaces = width - sum(len(word) for word in words)

    if num_words == 1:
        return words[0] + ' ' * num_spaces

    space_per_word = num_spaces // (num_words - 1)
    extra_spaces = num_spaces % (num_words - 1)

    justified_line = ''
    for i, word in enumerate(words):
        justified_line += word
        if i < num_words - 1:
            justified_line += ' ' * (space_per_word + (1 if i < extra_spaces else 0))
        if i == num_words - 1 and num_words > 1:
            justified_line += ' ' * (space_per_word - extra_spaces)

    return justified_line

# Example usage
string = "This is a test string to be justified"
width = 20
print(justify_words(string, width))
```
