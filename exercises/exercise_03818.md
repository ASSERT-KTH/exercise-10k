## Exercise
Create a program that uses a for loop to perform a series of calculations on a list of numbers.

## Reference Solution
```python
def perform_calculations(numbers):
    """
    Performs a series of calculations on a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        dict: A dictionary containing the results of the calculations.
    """
    results = {
        'sum': 0,
        'product': 1,
        'mean': 0,
        'max': float('-inf'),
        'min': float('inf')
    }

    for num in numbers:
        # Calculate the sum
        results['sum'] += num

        # Calculate the product
        results['product'] *= num

        # Update the maximum value
        if num > results['max']:
            results['max'] = num

        # Update the minimum value
        if num < results['min']:
            results['min'] = num

    # Calculate the mean
    results['mean'] = results['sum'] / len(numbers)

    return results


def main():
    # Example list of numbers
    numbers = [1, 2, 3, 4, 5]

    # Perform the calculations
    results = perform_calculations(numbers)

    # Print the results
    print("Results:")
    print(f"Sum: {results['sum']}")
    print(f"Product: {results['product']}")
    print(f"Mean: {results['mean']}")
    print(f"Max: {results['max']}")
    print(f"Min: {results['min']}")


if __name__ == "__main__":
    main()
```
