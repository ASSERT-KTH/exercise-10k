## Exercise
Write a function to scramble the letters in a given word

## Reference Solution
```python
import random

def scramble_word(word):
    """
    Scramble the letters in a given word.

    Args:
        word (str): The word to scramble.

    Returns:
        str: The scrambled word.
    """
    # Convert the word to a list of characters
    char_list = list(word)
    
    # Shuffle the characters in the list
    random.shuffle(char_list)
    
    # Join the characters back into a string
    scrambled_word = ''.join(char_list)
    
    return scrambled_word

# Example usage
def main():
    word = "hello"
    print(f"Original word: {word}")
    print(f"Scrambled word: {scramble_word(word)}")

if __name__ == "__main__":
    main()
```
