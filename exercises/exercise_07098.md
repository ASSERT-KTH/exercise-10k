## Exercise
Read two positive integers from the user and calculate their least common multiple

## Reference Solution
```python
import math

def calculate_lcm(num1, num2):
    """
    Calculate the least common multiple of two numbers.
    
    Args:
        num1 (int): The first number.
        num2 (int): The second number.
    
    Returns:
        int: The least common multiple of num1 and num2.
    """
    # Calculate the GCD using the Euclidean algorithm
    gcd = math.gcd(num1, num2)
    
    # Calculate the LCM using the formula
    lcm = (num1 * num2) // gcd
    
    return lcm

def main():
    # Read two positive integers from the user
    num1 = int(input("Enter the first positive integer: "))
    num2 = int(input("Enter the second positive integer: "))
    
    # Check if the input numbers are positive
    if num1 <= 0 or num2 <= 0:
        print("Both numbers must be positive.")
    else:
        # Calculate the LCM
        lcm = calculate_lcm(num1, num2)
        
        # Print the result
        print(f"The least common multiple of {num1} and {num2} is: {lcm}")

if __name__ == "__main__":
    main()
```
