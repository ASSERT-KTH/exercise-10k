## Exercise
Write a program that demonstrates the concept of pass by reference by passing a variable to a function and modifying it.

## Reference Solution
```python
# Python program to demonstrate the concept of pass by reference

def modify_list(lst):
    """
    This function modifies the input list by appending a new element.
    """
    lst.append(10)
    print("Inside function: ", lst)

def main():
    # Create a list
    my_list = [1, 2, 3]
    print("Original list: ", my_list)

    # Pass the list to the function
    modify_list(my_list)
    print("List after modification: ", my_list)

    # Create a new integer variable
    my_int = 5
    print("Original integer: ", my_int)

    # Try to modify the integer in a function
    def modify_int(x):
        x = x + 10
        print("Inside function: ", x)

    modify_int(my_int)
    print("Integer after 'modification': ", my_int)

if __name__ == "__main__":
    main()
```
