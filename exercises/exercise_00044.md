## Exercise
Implement the Sieve of Eratosthenes algorithm to find all prime numbers up to a given number

## Reference Solution
```python
def sieve_of_eratosthenes(limit):
    """
    Returns a list of all prime numbers up to the given limit.

    :param limit: The upper limit for finding prime numbers.
    :return: A list of prime numbers.
    """
    # Create a boolean array, prime, of size limit+1
    prime = [True] * (limit + 1)
    prime[0] = prime[1] = False

    # Iterate from 2 to sqrt(limit)
    for p in range(2, int(limit ** 0.5) + 1):
        # If p is a prime, mark as composite all the multiples of p
        if prime[p]:
            for i in range(p * p, limit + 1, p):
                prime[i] = False

    # Return a list of all prime numbers in the range
    return [p for p in range(2, limit + 1) if prime[p]]


def main():
    limit = 50  # Example limit
    primes = sieve_of_eratosthenes(limit)
    print(f"Prime numbers up to {limit}: {primes}")


if __name__ == "__main__":
    main()
```
