## Exercise
Write a program that takes the weight and height of a person as input and returns the BMI.

## Reference Solution
```python
def calculate_bmi(weight, height):
    """
    Calculate the BMI of a person.

    Args:
    weight (float): The weight of the person in kg.
    height (float): The height of the person in meters.

    Returns:
    float: The calculated BMI.
    """
    if height <= 0 or weight <= 0:
        raise ValueError("Height and weight must be positive numbers")

    bmi = weight / (height ** 2)
    return bmi

def get_bmi_category(bmi):
    """
    Get the BMI category.

    Args:
    bmi (float): The calculated BMI.

    Returns:
    str: The corresponding BMI category.
    """
    if bmi < 18.5:
        return "Underweight"
    elif bmi < 25:
        return "Normal weight"
    elif bmi < 30:
        return "Overweight"
    else:
        return "Obese"

def main():
    weight = float(input("Enter your weight in kg: "))
    height = float(input("Enter your height in meters: "))

    bmi = calculate_bmi(weight, height)
    category = get_bmi_category(bmi)

    print(f"Your BMI is: {bmi:.2f}")
    print(f"Your BMI category is: {category}")

if __name__ == "__main__":
    main()
```
