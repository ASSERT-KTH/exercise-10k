## Exercise
Write a program that merges two sorted lists of integers into a single sorted list. The program should take two sorted lists as input and output the merged sorted list.

def mergeSortedLists(list1, list2):
    merged_list = []
    index1 = 0
    index2 = 0

    while index1 < len(list1) and index2 < len(list2):
        if list1[index1] < list2[index2]:
            merged_list.append(list1[index1])
            index1 += 1
        else:
            merged_list.append(list2[index2])
            index2 += 1

    while index1 < len(list1):
        merged_list.append(list1[index1])
        index1 += 1

    while index2 < len(list2):
        merged_list.append(list2[index2])
        index2 += 1

    return merged_list

list1 = [1, 3, 5]
list2 = [2, 4, 6]
result = mergeSortedLists(list1, list2)
print(result)
