## Exercise
Create a program that implements a game of Rock, Paper, Scissors Lizard Spock using if/else statements

## Reference Solution
```python
import random

def game():
    while True:
        # Display game rules
        print("Game Rules:")
        print("  - Rock crushes Scissors")
        print("  - Scissors cuts Paper")
        print("  - Paper covers Rock")
        print("  - Rock crushes Lizard")
        print("  - Lizard poisons Spock")
        print("  - Spock smashes Scissors")
        print("  - Scissors decapitates Lizard")
        print("  - Lizard eats Paper")
        print("  - Paper disproves Spock")
        print("  - Spock vaporizes Rock")

        # Get user choice
        user_choice = input("Enter your choice (Rock, Paper, Scissors, Lizard, Spock): ").lower()

        # Validate user input
        while user_choice not in ["rock", "paper", "scissors", "lizard", "spock"]:
            user_choice = input("Invalid choice. Please enter Rock, Paper, Scissors, Lizard, or Spock: ").lower()

        # Get computer choice
        choices = ["rock", "paper", "scissors", "lizard", "spock"]
        computer_choice = random.choice(choices)

        # Display user and computer choices
        print(f"\nUser choice: {user_choice.capitalize()}")
        print(f"Computer choice: {computer_choice.capitalize()}\n")

        # Determine game outcome
        if user_choice == computer_choice:
            print("It's a tie!")
        elif (
            (user_choice == "rock" and computer_choice in ["scissors", "lizard"])
            or (user_choice == "scissors" and computer_choice in ["paper", "lizard"])
            or (user_choice == "paper" and computer_choice in ["rock", "spock"])
            or (user_choice == "lizard" and computer_choice in ["spock", "paper"])
            or (user_choice == "spock" and computer_choice in ["scissors", "rock"])
        ):
            print("User wins!")
        else:
            print("Computer wins!")

        # Play again?
        play_again = input("\nDo you want to play again? (yes/no): ").lower()
        while play_again not in ["yes", "no"]:
            play_again = input("Invalid input. Please enter yes or no: ").lower()
        if play_again == "no":
            break

if __name__ == "__main__":
    game()
```
