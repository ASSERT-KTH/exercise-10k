## Exercise
Create a function that takes a string of parentheses as input and returns True if they are valid, False otherwise

## Reference Solution
```python
def validate_parentheses(s: str) -> bool:
    """
    Validate a string of parentheses.

    Args:
    s (str): The string of parentheses to validate.

    Returns:
    bool: True if the string is valid, False otherwise.
    """
    stack = []
    parentheses_map = {")": "(", "}": "{", "]": "["}

    for char in s:
        if char in parentheses_map.values():
            # If the character is an open parenthesis, push it onto the stack
            stack.append(char)
        elif char in parentheses_map.keys():
            # If the character is a close parenthesis, check if the stack is empty or the top of the stack does not match
            if not stack or stack.pop() != parentheses_map[char]:
                return False

    # If the stack is empty after iterating over the string, the string is valid
    return not stack


# Example usage
if __name__ == "__main__":
    print(validate_parentheses("()"))  # True
    print(validate_parentheses("()[]{}"))  # True
    print(validate_parentheses("(]"))  # False
    print(validate_parentheses("([)]"))  # False
    print(validate_parentheses("{[]}"))  # True
```
