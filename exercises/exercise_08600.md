## Exercise
Write a function to sort a dictionary by its keys or values

def sort_dict(input_dict, by='key', reverse=False):
    """
    Sorts a dictionary by its keys or values.

    Args:
        input_dict (dict): The dictionary to be sorted.
        by (str, optional): The attribute to sort by. Defaults to 'key'.
                             Can be 'key' or 'value'.
        reverse (bool, optional): Whether to sort in reverse order. Defaults to False.

    Returns:
        dict: The sorted dictionary.
    """

    if by not in ['key', 'value']:
        raise ValueError("Invalid 'by' parameter. It should be 'key' or 'value'.")

    if by == 'key':
        return dict(sorted(input_dict.items(), key=lambda item: item[0], reverse=reverse))
    elif by == 'value':
        return dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))


# Example usage:
if __name__ == "__main__":

    sorted_by_key_reverse = sort_dict(example_dict, by='key', reverse=True)
    print("\nSorted by Key in Reverse Order:")
    print(sorted_by_key_reverse)

    sorted_by_value_reverse = sort_dict(example_dict, by='value', reverse=True)
    print("\nSorted by Value in Reverse Order:")
    print(sorted_by_value_reverse)
