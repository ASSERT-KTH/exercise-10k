## Exercise
Create a Caesar Cipher program using variables and string manipulation.

```python
def caesar_cipher(text, shift, direction):
    """
    Encrypts or decrypts the input text using a Caesar Cipher.

    Args:
        text (str): The input text to be encrypted or decrypted.
        shift (int): The number of positions to shift the letters.
        direction (str): 'encrypt' or 'decrypt' to specify the operation.

    Returns:
        str: The encrypted or decrypted text.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in text:
        if char.isalpha():
            # Find the position of the character in the alphabet
            index = alphabet.index(char.lower())

            if direction == 'encrypt':
                # Shift the character to the right
                new_index = (index + shift) % 26
            elif direction == 'decrypt':
                # Shift the character to the left
                new_index = (index - shift) % 26

            # Append the shifted character to the result
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            # Non-alphabetic characters are not shifted
            result += char

    return result


def main():
    text = input("Enter the text: ")
    shift = int(input("Enter the shift value: "))
    direction = input("Enter 'encrypt' or 'decrypt': ")

    result = caesar_cipher(text, shift, direction)
    print("Result:", result)


if __name__ == "__main__":
    main()
```
