## Exercise
Create a program that mixes two colors and prints out the result.

def mix_colors(color1, color2):
    """
    Mix two colors and return the resulting color.

    Args:
        color1 (str): The first color.
        color2 (str): The second color.

    Returns:
        str: The resulting mixed color.
    """
    # Define a dictionary to store color combinations
    color_combinations = {
        "red": {"blue": "purple", "yellow": "orange", "green": "brown"},
        "blue": {"red": "purple", "yellow": "green", "green": "teal"},
        "yellow": {"red": "orange", "blue": "green", "green": "lime"},
        "green": {"red": "brown", "blue": "teal", "yellow": "lime"},
        "purple": {"red": "pink", "blue": "indigo", "yellow": "brown", "green": "gray"},
        "orange": {"red": "pink", "blue": "purple", "yellow": "brown", "green": "brown"},
        "brown": {"red": "maroon", "blue": "gray", "yellow": "tan", "green": "olive"},
        "teal": {"red": "brown", "blue": "indigo", "yellow": "green", "green": "blue"},
        "lime": {"red": "orange", "blue": "green", "yellow": "yellow", "green": "green"},
        "pink": {"red": "red", "blue": "purple", "yellow": "orange", "green": "brown"},
        "indigo": {"red": "purple", "blue": "blue", "yellow": "green", "green": "teal"},
        "gray": {"red": "brown", "blue": "blue", "yellow": "green", "green": "gray"},
        "maroon": {"red": "red", "blue": "purple", "yellow": "orange", "green": "brown"},
        "tan": {"red": "brown", "blue": "gray", "yellow": "yellow", "green": "green"},
        "olive": {"red": "brown", "blue": "green", "yellow": "green", "green": "green"},
    }

    # Check if the colors are in the dictionary
    if color1 in color_combinations and color2 in color_combinations[color1]:
        return color_combinations[color1][color2]
    elif color2 in color_combinations and color1 in color_combinations[color2]:
        return color_combinations[color2][color1]
    else:
        return "Invalid color combination"

def main():
    # Get the two colors from the user
    color1 = input("Enter the first color: ").lower()
    color2 = input("Enter the second color: ").lower()

    # Mix the colors and print the result
    result = mix_colors(color1, color2)
    print(f"Mixed {color1} and {color2}: {result}")

if __name__ == "__main__":
    main()
