## Exercise
Create a program to update the value of a specific key in a given dictionary.

## Reference Solution
```python
def update_dict_key(dictionary, key, new_value):
    """
    Updates the value of a specific key in a given dictionary.

    Args:
        dictionary (dict): The dictionary to be updated.
        key (any): The key to be updated.
        new_value (any): The new value for the key.

    Returns:
        dict: The updated dictionary.
    """

    # Check if the key exists in the dictionary
    if key in dictionary:
        # Update the value of the key
        dictionary[key] = new_value
        return dictionary
    else:
        # If the key does not exist, add it to the dictionary
        dictionary[key] = new_value
        return dictionary


def main():
    # Create a dictionary
    my_dict = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }

    # Print the original dictionary
    print("Original Dictionary:")
    print(my_dict)

    # Update the value of a specific key
    updated_dict = update_dict_key(my_dict, "age", 31)

    # Print the updated dictionary
    print("\nUpdated Dictionary:")
    print(updated_dict)


if __name__ == "__main__":
    main()
```
