## Exercise
Write a program that checks if one list is a subset of another.

## Reference Solution
```python
def is_subset(list1, list2):
    """
    Checks if list1 is a subset of list2.

    Args:
        list1 (list): The list to check.
        list2 (list): The list to check against.

    Returns:
        bool: True if list1 is a subset of list2, False otherwise.
    """
    return set(list1).issubset(set(list2))

def main():
    # Example usage
    list1 = [1, 2, 3]
    list2 = [1, 2, 3, 4, 5]
    print(f"Is {list1} a subset of {list2}? {is_subset(list1, list2)}")

    list1 = [1, 2, 6]
    list2 = [1, 2, 3, 4, 5]
    print(f"Is {list1} a subset of {list2}? {is_subset(list1, list2)}")

if __name__ == "__main__":
    main()
```
