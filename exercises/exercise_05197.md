## Exercise
Write a program that asks the user for their location and then prints out the weather forecast for the next 5 days.

## Reference Solution
```python
import requests
import json

def get_weather_forecast(api_key, location):
    """
    Get the weather forecast for the next 5 days.

    Args:
        api_key (str): API key from OpenWeatherMap
        location (str): User's location (city, state, country)

    Returns:
        dict: Weather forecast data
    """
    base_url = "http://api.openweathermap.org/data/2.5/forecast"
    params = {
        "q": location,
        "appid": api_key,
        "units": "metric"
    }
    response = requests.get(base_url, params=params)
    return response.json()

def print_weather_forecast(forecast_data):
    """
    Print the weather forecast for the next 5 days.

    Args:
        forecast_data (dict): Weather forecast data
    """
    for day in range(5):
        date = forecast_data["list"][day]["dt_txt"]
        temp = forecast_data["list"][day]["main"]["temp"]
        feels_like = forecast_data["list"][day]["main"]["feels_like"]
        humidity = forecast_data["list"][day]["main"]["humidity"]
        weather = forecast_data["list"][day]["weather"][0]["description"]
        print(f"Day {day+1}:")
        print(f"  Date: {date}")
        print(f"  Temperature: {temp}°C")
        print(f"  Feels like: {feels_like}°C")
        print(f"  Humidity: {humidity}%")
        print(f"  Weather: {weather}")
        print()

def main():
    api_key = input("Enter your OpenWeatherMap API key: ")
    location = input("Enter your location (city, state, country): ")
    forecast_data = get_weather_forecast(api_key, location)
    print_weather_forecast(forecast_data)

if __name__ == "__main__":
    main()
```
