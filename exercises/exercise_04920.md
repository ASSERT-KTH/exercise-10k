## Exercise
Write a program that takes a time as input and prints whether it is time to wake up (current time is equal to or later than the alarm time) or not.

## Reference Solution
```python
import datetime

def check_alarm_time(alarm_time):
    """
    Check if it's time to wake up based on the current time and alarm time.

    Args:
        alarm_time (str): The alarm time in the format 'HH:MM'.

    Returns:
        bool: True if it's time to wake up, False otherwise.
    """

    # Get the current time
    current_time = datetime.datetime.now()

    # Parse the alarm time
    alarm_hour, alarm_minute = map(int, alarm_time.split(':'))

    # Create a datetime object for the alarm time
    alarm_datetime = current_time.replace(hour=alarm_hour, minute=alarm_minute, second=0, microsecond=0)

    # If the alarm time is earlier than the current time, assume it's for the next day
    if alarm_datetime < current_time:
        alarm_datetime += datetime.timedelta(days=1)

    # Check if it's time to wake up
    if current_time >= alarm_datetime:
        print("It's time to wake up!")
        return True
    else:
        print("Not time to wake up yet.")
        return False

def main():
    alarm_time = input("Enter the alarm time (HH:MM): ")
    check_alarm_time(alarm_time)

if __name__ == "__main__":
    main()
```
