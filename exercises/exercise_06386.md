## Exercise
Develop a program that takes in three numbers and prints out the maximum number. Use if/else statements to compare the numbers and find the maximum

## Reference Solution
```python
def find_max(a, b, c):
    """
    Finds the maximum number among three input numbers.

    Args:
        a (int): The first number.
        b (int): The second number.
        c (int): The third number.

    Returns:
        int: The maximum number.
    """
    if a >= b and a >= c:
        return a
    elif b >= a and b >= c:
        return b
    else:
        return c

def main():
    # Get input numbers from user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    num3 = float(input("Enter the third number: "))

    # Find and print the maximum number
    max_num = find_max(num1, num2, num3)
    print(f"The maximum number is: {max_num}")

if __name__ == "__main__":
    main()
```
