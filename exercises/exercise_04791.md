## Exercise
Write a program that asks the user for two numbers and an operator and then performs the operation.

## Reference Solution
```python
def calculator():
    """
    Asks the user for two numbers and an operator, then performs the operation.
    """
    # Ask the user for the first number
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Ask the user for the operator
    while True:
        operator = input("Enter the operator (+, -, *, /): ")
        if operator in ['+', '-', '*', '/']:
            break
        else:
            print("Invalid operator. Please enter one of the following: +, -, *, /")

    # Ask the user for the second number
    while True:
        try:
            num2 = float(input("Enter the second number: "))
            if operator == '/' and num2 == 0:
                print("Cannot divide by zero. Please enter a non-zero number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Perform the operation
    if operator == '+':
        result = num1 + num2
    elif operator == '-':
        result = num1 - num2
    elif operator == '*':
        result = num1 * num2
    elif operator == '/':
        result = num1 / num2

    # Print the result
    print(f"{num1} {operator} {num2} = {result}")

def main():
    calculator()
    while True:
        response = input("Do you want to perform another operation? (yes/no): ")
        if response.lower() == "yes":
            calculator()
        elif response.lower() == "no":
            break
        else:
            print("Invalid input. Please enter yes or no.")

if __name__ == "__main__":
    main()
```
