## Exercise
Create a program that uses a while loop to play a quiz game with the user. The program should ask the user a series of questions and keep track of the score.

## Reference Solution
```python
# Define a dictionary to store the questions, answers, and options
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Who painted the Mona Lisa?": {
        "A": "Leonardo da Vinci",
        "B": "Michelangelo",
        "C": "Raphael",
        "D": "Caravaggio",
        "answer": "A"
    }
}

def play_quiz():
    # Initialize the score
    score = 0
    
    # Initialize a counter to keep track of the current question
    question_number = 1
    
    # Loop through each question
    for question, options in questions.items():
        print(f"\nQuestion {question_number}: {question}")
        
        # Print the options
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        
        # Get the user's answer
        user_answer = input("Enter your answer (A, B, C, D): ")
        
        # Check if the user's answer is correct
        if user_answer.upper() == options["answer"]:
            print("Correct!")
            score += 1
        else:
            print(f"Incorrect. The correct answer is {options['answer']}.")
        
        # Increment the question number
        question_number += 1
    
    # Print the final score
    print(f"\nGame over! Your final score is {score} out of {len(questions)}.")

def main():
    play_again = "y"
    while play_again.lower() == "y":
        play_quiz()
        play_again = input("Would you like to play again? (y/n): ")
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
```
