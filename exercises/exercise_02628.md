## Exercise
Create a program that takes a player's name and score as input and generates a leaderboard entry.

## Reference Solution
```python
class LeaderboardEntry:
    def __init__(self, name, score):
        """
        Initialize a leaderboard entry with a player's name and score.

        Args:
            name (str): The player's name.
            score (int): The player's score.
        """
        self.name = name
        self.score = score

    def __str__(self):
        """
        Return a string representation of the leaderboard entry.

        Returns:
            str: A string in the format "Name: score".
        """
        return f"{self.name}: {self.score}"


class Leaderboard:
    def __init__(self):
        """
        Initialize a leaderboard.
        """
        self.entries = []

    def add_entry(self, name, score):
        """
        Add a leaderboard entry to the leaderboard.

        Args:
            name (str): The player's name.
            score (int): The player's score.
        """
        entry = LeaderboardEntry(name, score)
        self.entries.append(entry)

    def print_leaderboard(self):
        """
        Print the leaderboard in descending order of scores.
        """
        self.entries.sort(key=lambda x: x.score, reverse=True)
        for i, entry in enumerate(self.entries):
            print(f"{i+1}. {entry}")


def main():
    leaderboard = Leaderboard()

    while True:
        print("\nOptions:")
        print("1. Add a leaderboard entry")
        print("2. Print the leaderboard")
        print("3. Quit")
        
        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter player's name: ")
            score = int(input("Enter player's score: "))
            leaderboard.add_entry(name, score)
        elif choice == "2":
            leaderboard.print_leaderboard()
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
