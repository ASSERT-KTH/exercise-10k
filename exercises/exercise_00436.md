## Exercise
Ask the user for a string and print out the number of vowels in it.

## Reference Solution
```python
def count_vowels(input_string):
    """
    Counts the number of vowels in a given string.

    Args:
        input_string (str): The input string to count vowels from.

    Returns:
        int: The number of vowels in the input string.
    """
    vowels = 'aeiou'
    return sum(1 for char in input_string.lower() if char in vowels)

def main():
    # Ask the user for a string
    user_string = input("Please enter a string: ")

    # Count and print the number of vowels
    vowel_count = count_vowels(user_string)
    print(f"The string '{user_string}' contains {vowel_count} vowels.")

if __name__ == "__main__":
    main()
```
