## Exercise
Write a program that takes two lists as input, and returns a boolean indicating whether the first list is a subset of the second.

## Reference Solution
```python
def is_subset(list1, list2):
    """
    Checks if list1 is a subset of list2.
    
    Args:
        list1 (list): The list to check.
        list2 (list): The list to check against.
    
    Returns:
        bool: True if list1 is a subset of list2, False otherwise.
    """
    return set(list1).issubset(set(list2))

# Example usage:
list1 = [1, 2, 3]
list2 = [1, 2, 3, 4, 5]
print(is_subset(list1, list2))  # Output: True

list1 = [1, 2, 3]
list2 = [1, 2, 4, 5]
print(is_subset(list1, list2))  # Output: False
```

def is_subset_ordered(list1, list2):
    """
    Checks if list1 is a subset of list2, preserving the original order.
    
    Args:
        list1 (list): The list to check.
        list2 (list): The list to check against.
    
    Returns:
        bool: True if list1 is a subset of list2, False otherwise.
    """
    i = 0
    for elem in list1:
        found = False
        while i < len(list2):
            if elem == list2[i]:
                found = True
                i += 1
                break
            i += 1
        if not found:
            return False
    return True

# Example usage:
list1 = [1, 2, 3]
list2 = [1, 2, 3, 4, 5]
print(is_subset_ordered(list1, list2))  # Output: True

list1 = [1, 2, 3]
list2 = [1, 2, 4, 5]
print(is_subset_ordered(list1, list2))  # Output: False
