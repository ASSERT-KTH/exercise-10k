## Exercise
Create a dictionary with names of products as keys and their prices as values. Calculate the total price of all products and print the result.

## Reference Solution
```python
def calculate_total_price(product_prices):
    """
    Calculate the total price of all products.

    Args:
        product_prices (dict): Dictionary with product names as keys and their prices as values.

    Returns:
        float: Total price of all products.
    """
    total_price = sum(product_prices.values())
    return total_price


def main():
    # Create a dictionary with product names as keys and their prices as values
    product_prices = {
        "Apple": 1.00,
        "Banana": 0.50,
        "Orange": 1.50,
        "Milk": 2.00,
        "Bread": 1.25
    }

    # Calculate the total price of all products
    total_price = calculate_total_price(product_prices)

    # Print the product prices and total price
    print("Product Prices:")
    for product, price in product_prices.items():
        print(f"{product}: ${price:.2f}")

    print(f"\nTotal Price: ${total_price:.2f}")


if __name__ == "__main__":
    main()
```
