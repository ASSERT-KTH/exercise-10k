## Exercise
Write a program to understand the concept of NaN (Not a Number) and how to check for it.

## Reference Solution
```python
import numpy as np
import math

def check_nan(num):
    """Check if a number is NaN."""
    return np.isnan(num)

def main():
    # Create a NaN value
    nan_value = float('nan')

    # Create a valid number
    valid_number = 5.0

    # Check if the values are NaN
    print("Is NaN value NaN?", check_nan(nan_value))
    print("Is valid number NaN?", check_nan(valid_number))

    # Example of operation that produces NaN
    print("0/0 =", 0/0)
    print("Is 0/0 NaN?", check_nan(0/0))

    # Example of operation that does not produce NaN
    print("5/2 =", 5/2)
    print("Is 5/2 NaN?", check_nan(5/2))

if __name__ == "__main__":
    main()
```
