## Exercise
Create a program that simulates a database query system where users can add, remove, and query data.

## Reference Solution
```python
class Database:
    def __init__(self):
        self.data = {}

    def add(self, id, value):
        """Add a new entry to the database"""
        self.data[id] = value
        print(f"Added entry with ID {id} and value {value}")

    def remove(self, id):
        """Remove an entry from the database"""
        if id in self.data:
            del self.data[id]
            print(f"Removed entry with ID {id}")
        else:
            print(f"Entry with ID {id} not found")

    def query(self, id):
        """Query the database for a specific entry"""
        if id in self.data:
            print(f"Entry with ID {id} has value {self.data[id]}")
        else:
            print(f"Entry with ID {id} not found")

    def query_all(self):
        """Query the database for all entries"""
        print("All entries:")
        for id, value in self.data.items():
            print(f"ID: {id}, Value: {value}")


def main():
    db = Database()

    while True:
        print("\nDatabase Query System Simulator")
        print("1. Add entry")
        print("2. Remove entry")
        print("3. Query entry")
        print("4. Query all entries")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            id = input("Enter ID: ")
            value = input("Enter value: ")
            db.add(id, value)
        elif choice == "2":
            id = input("Enter ID: ")
            db.remove(id)
        elif choice == "3":
            id = input("Enter ID: ")
            db.query(id)
        elif choice == "4":
            db.query_all()
        elif choice == "5":
            print("Exiting program")
            break
        else:
            print("Invalid choice. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
