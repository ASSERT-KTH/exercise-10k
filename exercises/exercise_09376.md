## Exercise
Write a program that demonstrates data type inference techniques.

## Reference Solution
```python
def infer_data_type(value):
    """
    Infer the data type of a given value.

    Args:
        value (any): The value to infer the data type for.

    Returns:
        str: The inferred data type.
    """
    if isinstance(value, int):
        return "Integer"
    elif isinstance(value, float):
        return "Float"
    elif isinstance(value, str):
        return "String"
    elif isinstance(value, bool):
        return "Boolean"
    elif isinstance(value, list):
        return "List"
    elif isinstance(value, dict):
        return "Dictionary"
    elif isinstance(value, tuple):
        return "Tuple"
    else:
        return "Unknown"

def infer_list_data_type(values):
    """
    Infer the data type of a list of values.

    Args:
        values (list): The list of values to infer the data type for.

    Returns:
        str: The inferred data type.
    """
    if not values:
        return "Empty List"

    data_types = set()
    for value in values:
        data_types.add(infer_data_type(value))

    if len(data_types) == 1:
        return list(data_types)[0]
    else:
        return "Mixed Data Types"

def main():
    # Test infer_data_type function
    print("Infer Data Type Test:")
    print(f"Value: 10, Data Type: {infer_data_type(10)}")
    print(f"Value: 3.14, Data Type: {infer_data_type(3.14)}")
    print(f"Value: 'Hello', Data Type: {infer_data_type('Hello')}")
    print(f"Value: True, Data Type: {infer_data_type(True)}")
    print(f"Value: [1, 2, 3], Data Type: {infer_data_type([1, 2, 3])}")
    print(f"Value: {'a': 1, 'b': 2}, Data Type: {infer_data_type({'a': 1, 'b': 2})}")
    print(f"Value: (1, 2, 3), Data Type: {infer_data_type((1, 2, 3))}")

    # Test infer_list_data_type function
    print("\nInfer List Data Type Test:")
    print(f"Values: [1, 2, 3], Data Type: {infer_list_data_type([1, 2, 3])}")
    print(f"Values: [1.1, 2.2, 3.3], Data Type: {infer_list_data_type([1.1, 2.2, 3.3])}")
    print(f"Values: ['a', 'b', 'c'], Data Type: {infer_list_data_type(['a', 'b', 'c'])}")
    print(f"Values: [True, False, True], Data Type: {infer_list_data_type([True, False, True])}")
    print(f"Values: [1, 'a', 3.14], Data Type: {infer_list_data_type([1, 'a', 3.14])}")
    print(f"Values: [], Data Type: {infer_list_data_type([])}")

if __name__ == "__main__":
    main()
```
