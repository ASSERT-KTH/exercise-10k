## Exercise
Write a program that takes a location and temperature as input and generates a weather forecast.

## Reference Solution
```python
import random

class WeatherForecast:
    def __init__(self, location, temperature):
        self.location = location
        self.temperature = temperature

    def generate_forecast(self):
        forecast = {
            "location": self.location,
            "temperature": self.temperature,
            "condition": self.generate_condition(),
            "precipitation": self.generate_precipitation(),
            "wind": self.generate_wind()
        }
        return forecast

    def generate_condition(self):
        if self.temperature < 0:
            return "Snowy"
        elif self.temperature < 15:
            return "Cloudy"
        elif self.temperature < 25:
            return "Sunny"
        else:
            return "Hot"

    def generate_precipitation(self):
        if random.random() < 0.3:
            return "Rain"
        elif random.random() < 0.2:
            return "Snow"
        else:
            return "None"

    def generate_wind(self):
        wind_speed = random.randint(0, 20)
        if wind_speed < 5:
            return "Calm"
        elif wind_speed < 10:
            return "Light Breeze"
        elif wind_speed < 15:
            return "Moderate Wind"
        else:
            return "Strong Wind"

def main():
    location = input("Enter a location: ")
    temperature = int(input("Enter a temperature (C): "))

    forecast = WeatherForecast(location, temperature)
    weather_forecast = forecast.generate_forecast()

    print("Weather Forecast:")
    print(f"Location: {weather_forecast['location']}")
    print(f"Temperature: {weather_forecast['temperature']}C")
    print(f"Condition: {weather_forecast['condition']}")
    print(f"Precipitation: {weather_forecast['precipitation']}")
    print(f"Wind: {weather_forecast['wind']}")

if __name__ == "__main__":
    main()
```
