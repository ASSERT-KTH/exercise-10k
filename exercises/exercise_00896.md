## Exercise
Create a program that asks the user for a student's name and grades in multiple subjects, and prints out the student's average grade and a message indicating whether they passed or failed.

## Reference Solution
```python
# Define a function to get student information
def get_student_info():
    # Get student's name
    name = input("Enter student's name: ")

    # Get student's grades
    grades = {}
    while True:
        subject = input("Enter subject name (or 'done' to finish): ")
        if subject.lower() == 'done':
            break
        grade = float(input("Enter grade for {}: ".format(subject)))
        grades[subject] = grade

    return name, grades

# Define a function to calculate the student's average grade
def get_average_grade(grades):
    total_grade = sum(grades.values())
    average_grade = total_grade / len(grades)
    return average_grade

# Define a function to determine if the student passed or failed
def did_student_pass(average_grade):
    if average_grade >= 60:
        return True
    else:
        return False

# Define a function to print the student's report
def print_report(name, average_grade, did_pass):
    if did_pass:
        print("{}'s average grade is {:.2f}. They passed!".format(name, average_grade))
    else:
        print("{}'s average grade is {:.2f}. They failed.".format(name, average_grade))

# Main program
def main():
    name, grades = get_student_info()
    average_grade = get_average_grade(grades)
    did_pass = did_student_pass(average_grade)
    print_report(name, average_grade, did_pass)

# Run the main program
if __name__ == '__main__':
    main()
```
