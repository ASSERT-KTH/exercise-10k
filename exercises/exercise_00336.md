## Exercise
Write a program to compare variables of different data types using comparison operators.

## Reference Solution
```python
# Comparison of variables of different data types

def compare_variables():
    # Integer variables
    int1 = 10
    int2 = 20

    # Float variables
    float1 = 10.5
    float2 = 20.5

    # String variables
    str1 = "Hello"
    str2 = "World"

    # Boolean variables
    bool1 = True
    bool2 = False

    # Comparison of integer variables
    print("Comparison of integer variables:")
    print(f"{int1} == {int2}: {int1 == int2}")
    print(f"{int1} != {int2}: {int1 != int2}")
    print(f"{int1} > {int2}: {int1 > int2}")
    print(f"{int1} < {int2}: {int1 < int2}")
    print(f"{int1} >= {int2}: {int1 >= int2}")
    print(f"{int1} <= {int2}: {int1 <= int2}")

    # Comparison of float variables
    print("\nComparison of float variables:")
    print(f"{float1} == {float2}: {float1 == float2}")
    print(f"{float1} != {float2}: {float1 != float2}")
    print(f"{float1} > {float2}: {float1 > float2}")
    print(f"{float1} < {float2}: {float1 < float2}")
    print(f"{float1} >= {float2}: {float1 >= float2}")
    print(f"{float1} <= {float2}: {float1 <= float2}")

    # Comparison of string variables
    print("\nComparison of string variables:")
    print(f"{str1} == {str2}: {str1 == str2}")
    print(f"{str1} != {str2}: {str1 != str2}")
    print(f"{str1} > {str2}: {str1 > str2}")
    print(f"{str1} < {str2}: {str1 < str2}")
    print(f"{str1} >= {str2}: {str1 >= str2}")
    print(f"{str1} <= {str2}: {str1 <= str2}")

    # Comparison of boolean variables
    print("\nComparison of boolean variables:")
    print(f"{bool1} == {bool2}: {bool1 == bool2}")
    print(f"{bool1} != {bool2}: {bool1 != bool2}")

    # Comparison of variables of different data types
    print("\nComparison of variables of different data types:")
    print(f"{int1} == {float1}: {int1 == float1}")
    print(f"{int1} == {str1}: {int1 == str1}")
    print(f"{int1} == {bool1}: {int1 == bool1}")

if __name__ == "__main__":
    compare_variables()
```
