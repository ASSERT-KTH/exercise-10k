## Exercise
Write a program to demonstrate the difference between local and global variables

## Reference Solution
```python
# Global variable
x = 10

def local_variable_example():
    # Local variable
    y = 20
    print("Local variable y:", y)

def global_variable_example():
    global x  # Accessing global variable
    x = 30  # Modifying global variable
    print("Global variable x after modification:", x)

def global_and_local_variable_example():
    x = 40  # This is a local variable, not the global one
    print("Local variable x in global_and_local_variable_example:", x)

def main():
    print("Initial global variable x:", x)
    local_variable_example()
    global_variable_example()
    print("Global variable x after global_variable_example:", x)
    global_and_local_variable_example()
    print("Global variable x after global_and_local_variable_example:", x)

if __name__ == "__main__":
    main()
```
