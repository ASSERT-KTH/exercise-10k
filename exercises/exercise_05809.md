## Exercise
Write a program that uses a while loop to check whether a given string is palindromic or not.

## Reference Solution
```python
def is_palindrome(s):
    """
    Checks if a given string is palindromic.
    
    Args:
        s (str): The input string.
    
    Returns:
        bool: True if the string is palindromic, False otherwise.
    """
    s = s.lower()  # Convert to lowercase for case-insensitive comparison
    i = 0
    j = len(s) - 1
    
    while i < j:
        if s[i] != s[j]:
            return False
        i += 1
        j -= 1
    
    return True

def main():
    # Example usage
    strings = ["radar", "python", "level", "hello"]
    
    for s in strings:
        if is_palindrome(s):
            print(f"'{s}' is a palindrome.")
        else:
            print(f"'{s}' is not a palindrome.")

if __name__ == "__main__":
    main()
```
