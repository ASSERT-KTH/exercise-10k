## Exercise
Write a program that uses a while loop to solve a maze, finding the shortest path from the start to the end.

## Reference Solution
```python
from collections import deque

# Define the maze
maze = [
    ['#', '#', '#', '#', '#', '#', '#', '#'],
    ['#', 'S', ' ', ' ', ' ', ' ', ' ', '#'],
    ['#', '#', '#', ' ', '#', '#', ' ', '#'],
    ['#', ' ', ' ', ' ', ' ', '#', ' ', '#'],
    ['#', ' ', '#', '#', ' ', '#', ' ', '#'],
    ['#', ' ', '#', ' ', ' ', ' ', ' ', '#'],
    ['#', ' ', '#', ' ', '#', '#', 'E', '#'],
    ['#', '#', '#', '#', '#', '#', '#', '#']
]

# Define the possible movements (up, down, left, right)
movements = [(0, 1), (0, -1), (1, 0), (-1, 0)]

# Find the start and end positions
start = None
end = None
for y, row in enumerate(maze):
    for x, cell in enumerate(row):
        if cell == 'S':
            start = (x, y)
        elif cell == 'E':
            end = (x, y)

# Use a queue to keep track of the cells to visit
queue = deque([(start, [start])])

# Use a set to keep track of the visited cells
visited = set()

while queue:
    # Get the current cell and the path that led to it
    (x, y), path = queue.popleft()

    # If this is the end cell, print the path and break
    if (x, y) == end:
        print("Path found:", path)
        break

    # Mark the current cell as visited
    visited.add((x, y))

    # Check all possible movements from the current cell
    for dx, dy in movements:
        nx, ny = x + dx, y + dy

        # Check if the new cell is within the maze boundaries and is not a wall
        if (0 <= nx < len(maze[0])) and (0 <= ny < len(maze)) and maze[ny][nx] != '#':
            # Check if the new cell has not been visited before
            if (nx, ny) not in visited:
                # Add the new cell to the queue
                queue.append(((nx, ny), path + [(nx, ny)]))

# If no path is found, print a message
if not queue:
    print("No path found")
```
