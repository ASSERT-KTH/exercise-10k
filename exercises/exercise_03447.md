## Exercise
Write a program that converts temperature from Celsius to Fahrenheit or vice versa, based on the user's choice

## Reference Solution
```python
def celsius_to_fahrenheit(celsius):
    """Converts temperature from Celsius to Fahrenheit."""
    return (celsius * 9/5) + 32


def fahrenheit_to_celsius(fahrenheit):
    """Converts temperature from Fahrenheit to Celsius."""
    return (fahrenheit - 32) * 5/9


def main():
    while True:
        print("\nTemperature Conversion Program")
        print("-------------------------------")
        print("1. Convert Celsius to Fahrenheit")
        print("2. Convert Fahrenheit to Celsius")
        print("3. Quit")
        
        choice = input("Enter your choice (1/2/3): ")

        if choice == "1":
            celsius = float(input("Enter temperature in Celsius: "))
            print(f"{celsius}째C is equal to {celsius_to_fahrenheit(celsius)}째F")
        elif choice == "2":
            fahrenheit = float(input("Enter temperature in Fahrenheit: "))
            print(f"{fahrenheit}째F is equal to {fahrenheit_to_celsius(fahrenheit)}째C")
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
