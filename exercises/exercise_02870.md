## Exercise
Create a function to count the frequency of each word in a given string

## Reference Solution
```python
def word_frequency_counter(input_string):
    """
    Counts the frequency of each word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where the keys are the unique words in the string and the values are their corresponding frequencies.
    """
    # Convert the string to lower case to ensure the word count is case-insensitive
    input_string = input_string.lower()
    
    # Split the string into words
    words = input_string.split()
    
    # Remove punctuation from each word
    words = [word.strip('.,!?"\'') for word in words]
    
    # Create a dictionary to store the word frequencies
    frequency_dict = {}
    
    # Iterate over each word in the list of words
    for word in words:
        # If the word is already in the dictionary, increment its count
        if word in frequency_dict:
            frequency_dict[word] += 1
        # If the word is not in the dictionary, add it with a count of 1
        else:
            frequency_dict[word] = 1
    
    return frequency_dict

# Example usage:
input_str = "This is a test string. This string is just a test."
print(word_frequency_counter(input_str))
```

```python
from collections import Counter
import re

def word_frequency_counter(input_string):
    """
    Counts the frequency of each word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where the keys are the unique words in the string and the values are their corresponding frequencies.
    """
    # Convert the string to lower case to ensure the word count is case-insensitive
    input_string = input_string.lower()
    
    # Remove punctuation from the string
    input_string = re.sub(r'[.,!?"\']', '', input_string)
    
    # Split the string into words
    words = input_string.split()
    
    # Create a dictionary to store the word frequencies
    frequency_dict = Counter(words)
    
    return dict(frequency_dict)

# Example usage:
input_str = "This is a test string. This string is just a test."
print(word_frequency_counter(input_str))
```
