## Exercise
Create a program that takes two numbers and an arithmetic operation as input and returns the result of the operation. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
def calculate(num1, num2, operation):
    """
    Perform arithmetic operations on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to be performed.
                         Can be '+', '-', '*', '/' or 'add', 'subtract', 'multiply', 'divide'.

    Returns:
        float: The result of the operation.
    """

    # Define a dictionary to map operation names to their corresponding functions
    operations = {
        '+': lambda x, y: x + y,
        'add': lambda x, y: x + y,
        '-': lambda x, y: x - y,
        'subtract': lambda x, y: x - y,
        '*': lambda x, y: x * y,
        'multiply': lambda x, y: x * y,
        '/': lambda x, y: x / y if y != 0 else float('inf'),
        'divide': lambda x, y: x / y if y != 0 else float('inf')
    }

    # Check if the operation is valid
    if operation in operations:
        # Perform the operation and return the result
        return operations[operation](num1, num2)
    else:
        # Raise an error if the operation is not valid
        raise ValueError("Invalid operation. Supported operations are '+', '-', '*', '/' or 'add', 'subtract', 'multiply', 'divide'.")

def main():
    # Take input from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, / or add, subtract, multiply, divide): ")

    # Calculate and print the result
    try:
        result = calculate(num1, num2, operation)
        print(f"The result of {operation} {num1} and {num2} is {result}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
