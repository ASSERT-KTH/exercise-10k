## Exercise
Write a function to filter key-value pairs from a dictionary based on a condition

## Reference Solution
```python
def filter_dictionary(input_dict, condition):
    """
    Filters key-value pairs from a dictionary based on a condition.

    Args:
        input_dict (dict): The input dictionary.
        condition (function): A function that takes a key-value pair and returns True if the pair satisfies the condition.

    Returns:
        dict: A new dictionary with the filtered key-value pairs.
    """
    return {key: value for key, value in input_dict.items() if condition(key, value)}

# Example usage
if __name__ == "__main__":
    # Define a dictionary
    input_dict = {
        "apple": 5,
        "banana": 10,
        "orange": 7,
        "grape": 3
    }

    # Define a condition function
    def condition(key, value):
        return value > 5

    # Filter the dictionary
    filtered_dict = filter_dictionary(input_dict, condition)

    # Print the filtered dictionary
    print(filtered_dict)
```
