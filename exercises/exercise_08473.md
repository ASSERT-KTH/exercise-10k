## Exercise
Write a program to calculate the day of the week given a date

def calculate_day_of_week(year, month, day):
    """
    Calculate the day of the week given a date using Zeller's Congruence algorithm.

    Args:
        year (int): The year of the date.
        month (int): The month of the date (1-12).
        day (int): The day of the date (1-31).

    Returns:
        str: The day of the week (e.g. "Monday", "Tuesday", etc.).
    """
    t = [0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4]
    year -= month < 3
    return ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"][
        (year + int(year / 4) - int(year / 100) + int(year / 400) + t[month - 1] + day) % 7
    ]

def main():
    year = int(input("Enter the year: "))
    month = int(input("Enter the month (1-12): "))
    day = int(input("Enter the day (1-31): "))

    day_of_week = calculate_day_of_week(year, month, day)
    print(f"The day of the week is: {day_of_week}")

if __name__ == "__main__":
    main()
