## Exercise
Find the next greater element for each element in an array

## Reference Solution

def next_greater_element(arr):
    """
    Find the next greater element for each element in an array.

    Args:
        arr (list): The input array.

    Returns:
        list: The next greater element for each element in the array.
    """
    stack = []
    result = [-1] * len(arr)

    for i in range(len(arr)):
        # Pop all elements from the stack that are smaller than the current element
        while stack and arr[stack[-1]] < arr[i]:
            # Update the next greater element for the popped element
            result[stack.pop()] = arr[i]
        # Push the current element onto the stack
        stack.append(i)

    return result

# Example usage
arr = [4, 5, 2, 25]
print(next_greater_element(arr))  # Output: [5, 25, 25, -1]
