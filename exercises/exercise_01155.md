## Exercise
Write a program that generates random art using Turtle graphics.

## Reference Solution
```python
import turtle
import random

def draw_square(turt, length, color):
    """
    Draw a square with the given length and color.
    
    Args:
        turt (turtle.Turtle): The turtle object to use for drawing.
        length (int): The length of the square.
        color (str): The color of the square.
    """
    turt.fillcolor(color)
    turt.begin_fill()
    for _ in range(4):
        turt.forward(length)
        turt.right(90)
    turt.end_fill()

def draw_circle(turt, radius, color):
    """
    Draw a circle with the given radius and color.
    
    Args:
        turt (turtle.Turtle): The turtle object to use for drawing.
        radius (int): The radius of the circle.
        color (str): The color of the circle.
    """
    turt.fillcolor(color)
    turt.begin_fill()
    turt.circle(radius)
    turt.end_fill()

def main():
    # Set up the screen
    window = turtle.Screen()
    window.setup(800, 600)
    window.bgcolor((random.random(), random.random(), random.random()))

    # Create a new turtle object
    turt = turtle.Turtle()
    turt.speed(0)

    # Draw 50 random shapes
    for _ in range(50):
        shape = random.choice(['square', 'circle'])
        if shape == 'square':
            draw_square(turt, random.randint(10, 100), (random.random(), random.random(), random.random()))
        else:
            draw_circle(turt, random.randint(10, 50), (random.random(), random.random(), random.random()))

        # Move to a new random location
        turt.penup()
        turt.goto(random.randint(-300, 300), random.randint(-200, 200))
        turt.pendown()

        # Rotate to a new random angle
        turt.left(random.randint(0, 360))

    # Keep the window open
    turtle.done()

if __name__ == "__main__":
    main()
```
