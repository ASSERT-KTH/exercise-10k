## Exercise
Write a program to check if a given sequence of parentheses is valid

## Reference Solution
```python
def is_valid_parentheses(sequence):
    """
    Checks if a given sequence of parentheses is valid.

    Args:
        sequence (str): The sequence of parentheses to check.

    Returns:
        bool: True if the sequence is valid, False otherwise.
    """
    stack = []
    mapping = {")": "(", "}": "{", "]": "["}

    for char in sequence:
        if char in mapping.values():
            stack.append(char)
        elif char in mapping.keys():
            if stack == [] or mapping[char] != stack.pop():
                return False

    return stack == []

# Example usage
if __name__ == "__main__":
    sequences = [
        "((()))",  # Valid
        "(()",     # Invalid
        "{[()]}",
        "{[()]",
        "{[]}",
        "{[}",
    ]

    for sequence in sequences:
        print(f"Sequence: {sequence}, Valid: {is_valid_parentheses(sequence)}")
```
