## Exercise
Implement a function to perform a topological sort on a directed acyclic graph (DAG).

## Reference Solution

from collections import defaultdict, deque

def topological_sort(graph):
    """
    Performs a topological sort on a directed acyclic graph (DAG).

    Args:
    graph: A dictionary representing the adjacency list of the graph.
           Each key is a node, and its corresponding value is a list of its neighbors.

    Returns:
    A list of nodes in topological order.
    """

    # Initialize the in-degree dictionary
    in_degree = defaultdict(int)
    
    # Calculate the in-degree of each node
    for node in graph:
        for neighbor in graph[node]:
            in_degree[neighbor] += 1
    
    # Initialize the queue with nodes having an in-degree of 0
    queue = deque([node for node in graph if in_degree[node] == 0])
    
    # Initialize the result list
    result = []
    
    # Perform the topological sort
    while queue:
        node = queue.popleft()
        result.append(node)
        
        # Decrease the in-degree of each neighbor
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)
    
    # Check for cycles in the graph
    if len(result) != len(graph):
        raise ValueError("The graph contains a cycle and cannot be topologically sorted")
    
    return result

# Example usage
if __name__ == "__main__":
    graph = {
        'A': ['B', 'C'],
        'B': ['D'],
        'C': ['D'],
        'D': []
    }

    try:
        sorted_nodes = topological_sort(graph)
        print("Topologically sorted nodes:", sorted_nodes)
    except ValueError as e:
        print(e)
