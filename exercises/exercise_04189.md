## Exercise
Create a function that generates a weather forecast based on the current temperature and humidity.

```python
def generate_weather_forecast(temperature: float, humidity: float) -> str:
    """
    Generates a weather forecast based on the current temperature and humidity.

    Args:
        temperature (float): The current temperature in degrees Celsius.
        humidity (float): The current humidity as a percentage.

    Returns:
        str: A string describing the weather forecast.
    """

    # Define the temperature ranges for each forecast type
    if temperature < 0:
        temperature_forecast = "Freezing"
    elif temperature < 15:
        temperature_forecast = "Cold"
    elif temperature < 25:
        temperature_forecast = "Mild"
    elif temperature < 30:
        temperature_forecast = "Warm"
    else:
        temperature_forecast = "Hot"

    # Define the humidity ranges for each forecast type
    if humidity < 30:
        humidity_forecast = "Dry"
    elif humidity < 60:
        humidity_forecast = "Moderate"
    elif humidity < 80:
        humidity_forecast = "Humid"
    else:
        humidity_forecast = "Very Humid"

    # Combine the temperature and humidity forecasts
    forecast = f"The weather will be {temperature_forecast} and {humidity_forecast}."

    return forecast

# Example usage
if __name__ == "__main__":
    temperature = 22
    humidity = 50
    print(generate_weather_forecast(temperature, humidity))
```
