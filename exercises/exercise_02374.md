## Exercise
Write a program that calculates the distance between two points in a 2D or 3D space.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.

    Args:
        point1 (list): The coordinates of the first point.
        point2 (list): The coordinates of the second point.

    Returns:
        float: The distance between the two points.
    """
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of dimensions")

    return math.sqrt(sum((a - b) ** 2 for a, b in zip(point1, point2)))

def main():
    # Example usage:
    point1_2d = [1, 2]
    point2_2d = [4, 6]
    point1_3d = [1, 2, 3]
    point2_3d = [4, 5, 6]

    print(f"Distance between {point1_2d} and {point2_2d}: {calculate_distance(point1_2d, point2_2d)}")
    print(f"Distance between {point1_3d} and {point2_3d}: {calculate_distance(point1_3d, point2_3d)}")

if __name__ == "__main__":
    main()
```
