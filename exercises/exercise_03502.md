## Exercise
Write a program that converts a string into an integer. The string will be a valid integer representation.

## Reference Solution
```python
def string_to_int(s):
    """
    Converts a string into an integer.

    Args:
        s (str): A valid integer representation as a string.

    Returns:
        int: The integer equivalent of the input string.

    Raises:
        ValueError: If the input string is not a valid integer representation.
    """
    try:
        # Attempt to convert the string to an integer
        return int(s)
    except ValueError:
        # If the conversion fails, raise a ValueError with a meaningful message
        raise ValueError("Invalid integer representation")

def main():
    # Example usage
    input_str = "123"
    try:
        result = string_to_int(input_str)
        print(f"The integer equivalent of '{input_str}' is: {result}")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
