## Exercise
Create a function that uses memoization to store the results of expensive function calls

## Reference Solution
```python
from functools import wraps

def memoize(func):
    """Decorator to memoize function calls"""
    cache = dict()

    @wraps(func)
    def memoized_func(*args):
        if args in cache:
            return cache[args]
        else:
            result = func(*args)
            cache[args] = result
            return result

    return memoized_func

# Example usage:
@memoize
def expensive_function(x):
    """Simulate an expensive function call"""
    import time
    time.sleep(2)  # simulate expensive computation
    return x * x

def main():
    print(expensive_function(2))  # takes 2 seconds
    print(expensive_function(2))  # returns cached result instantly
    print(expensive_function(3))  # takes 2 seconds
    print(expensive_function(3))  # returns cached result instantly

if __name__ == "__main__":
    main()
```
