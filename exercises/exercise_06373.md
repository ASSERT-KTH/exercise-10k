## Exercise
Write a program that takes an array and an element as input and returns a boolean indicating whether the element is in the array

## Reference Solution
```python
def check_element(array, element):
    """
    Checks if an element exists in an array.

    Args:
        array (list): The input array.
        element: The element to search for.

    Returns:
        bool: True if the element is in the array, False otherwise.
    """
    return element in array

# Example usage:
def main():
    array = [1, 2, 3, 4, 5]
    element = 3
    result = check_element(array, element)
    print(f"Is {element} in the array? {result}")

    element = 6
    result = check_element(array, element)
    print(f"Is {element} in the array? {result}")

if __name__ == "__main__":
    main()
```
