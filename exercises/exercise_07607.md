## Exercise
Write a program that uses comparison operators to compare two variables.

## Reference Solution
```python
# Define two variables
var1 = 10
var2 = 20

# Comparison operators
def compare_variables(var1, var2):
    print("Comparison Results:")
    print("-------------------")
    
    # Equal to operator (==)
    print(f"Is {var1} equal to {var2}? {var1 == var2}")
    
    # Not equal to operator (!=)
    print(f"Is {var1} not equal to {var2}? {var1 != var2}")
    
    # Greater than operator (>)
    print(f"Is {var1} greater than {var2}? {var1 > var2}")
    
    # Less than operator (<)
    print(f"Is {var1} less than {var2}? {var1 < var2}")
    
    # Greater than or equal to operator (>=)
    print(f"Is {var1} greater than or equal to {var2}? {var1 >= var2}")
    
    # Less than or equal to operator (<=)
    print(f"Is {var1} less than or equal to {var2}? {var1 <= var2}")

# Main function
def main():
    compare_variables(var1, var2)

# Run the main function
if __name__ == "__main__":
    main()
```
