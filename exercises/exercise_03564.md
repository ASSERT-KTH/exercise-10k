## Exercise
Update a dictionary with new key-value pairs and print the updated dictionary

## Reference Solution
```python
# Define the initial dictionary
initial_dict = {
    "name": "John",
    "age": 30,
    "city": "New York"
}

# Define the new key-value pairs to update the dictionary
new_pairs = {
    "country": "USA",
    "occupation": "Software Engineer"
}

# Update the dictionary with the new key-value pairs
def update_dictionary(initial_dict, new_pairs):
    """
    Updates a dictionary with new key-value pairs.

    Args:
        initial_dict (dict): The initial dictionary.
        new_pairs (dict): The new key-value pairs to update the dictionary.

    Returns:
        dict: The updated dictionary.
    """
    # Use the update() method to add the new key-value pairs
    initial_dict.update(new_pairs)
    return initial_dict

# Example usage
updated_dict = update_dictionary(initial_dict, new_pairs)

# Print the updated dictionary
print("Updated Dictionary:")
for key, value in updated_dict.items():
    print(f"{key}: {value}")
```
