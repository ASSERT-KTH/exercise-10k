## Exercise
Create a program that implements a To-Do List app using if/else statements

## Reference Solution
```python
# To-Do List App

class ToDoList:
    def __init__(self):
        self.tasks = []

    def display_tasks(self):
        """Display all tasks in the To-Do List"""
        if len(self.tasks) == 0:
            print("No tasks in the list.")
        else:
            print("Tasks in the list:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")

    def add_task(self, task):
        """Add a new task to the To-Do List"""
        if task not in self.tasks:
            self.tasks.append(task)
            print(f"Task '{task}' added successfully.")
        else:
            print(f"Task '{task}' already exists in the list.")

    def remove_task(self, task_number):
        """Remove a task from the To-Do List"""
        try:
            task_number = int(task_number)
            if task_number <= 0:
                print("Invalid task number. Please enter a positive integer.")
            elif task_number > len(self.tasks):
                print("Task number out of range.")
            else:
                task = self.tasks.pop(task_number - 1)
                print(f"Task '{task}' removed successfully.")
        except ValueError:
            print("Invalid task number. Please enter an integer.")

    def update_task(self, task_number, new_task):
        """Update a task in the To-Do List"""
        try:
            task_number = int(task_number)
            if task_number <= 0:
                print("Invalid task number. Please enter a positive integer.")
            elif task_number > len(self.tasks):
                print("Task number out of range.")
            else:
                old_task = self.tasks[task_number - 1]
                self.tasks[task_number - 1] = new_task
                print(f"Task '{old_task}' updated to '{new_task}'.")
        except ValueError:
            print("Invalid task number. Please enter an integer.")

def main():
    todo_list = ToDoList()

    while True:
        print("\nTo-Do List App Menu:")
        print("1. Display Tasks")
        print("2. Add Task")
        print("3. Remove Task")
        print("4. Update Task")
        print("5. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            todo_list.display_tasks()
        elif choice == "2":
            task = input("Enter a new task: ")
            todo_list.add_task(task)
        elif choice == "3":
            task_number = input("Enter the task number to remove: ")
            todo_list.remove_task(task_number)
        elif choice == "4":
            task_number = input("Enter the task number to update: ")
            new_task = input("Enter the new task: ")
            todo_list.update_task(task_number, new_task)
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
