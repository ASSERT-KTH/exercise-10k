## Exercise
Create a program that takes an integer as input and checks if it's even or odd using conditional statements.

#include <iostream>
using namespace std;

int main() {
    int numero;
    cout << "Ingrese un numero: ";
    cin >> numero;

    if (numero % 2 == 0) {
        cout << "El numero es par";
    } else {
        cout << "El numero es impar";
    }

    return 0;
}
