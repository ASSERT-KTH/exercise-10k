## Exercise
Count the number of occurrences of a specific letter in a given string.

## Reference Solution
```python
def count_letter(string, letter):
    """
    Counts the occurrences of a specific letter in a given string.

    Args:
        string (str): The input string.
        letter (str): The letter to count.

    Returns:
        int: The number of occurrences of the letter.
    """
    return string.lower().count(letter.lower())

def main():
    string = input("Enter a string: ")
    letter = input("Enter a letter: ")
    
    if len(letter) != 1:
        print("Please enter a single letter.")
    else:
        count = count_letter(string, letter)
        print(f"The letter '{letter}' occurs {count} times in the string.")

if __name__ == "__main__":
    main()
```
