## Exercise
Write a program that takes a list and a slice specification as input, and returns the sliced list.

## Reference Solution
```python
def slice_list(input_list, slice_spec):
    """
    Slice a list based on a slice specification.

    Args:
        input_list (list): The input list to be sliced.
        slice_spec (str): A string representing the slice specification.
            It should be in the format 'start:stop:step', where 'start' and 'stop' are optional and 'step' is optional.
            If 'start' is omitted, it defaults to 0. If 'stop' is omitted, it defaults to the end of the list.
            If 'step' is omitted, it defaults to 1.

    Returns:
        list: The sliced list.
    """

    # Split the slice specification into start, stop, and step
    parts = slice_spec.split(':')

    # Initialize start, stop, and step
    start = None
    stop = None
    step = None

    # Parse the slice specification
    if len(parts) == 1:
        # Only one part, it's the stop
        stop = int(parts[0])
    elif len(parts) == 2:
        # Two parts, they're start and stop
        if parts[0]:
            start = int(parts[0])
        stop = int(parts[1])
    elif len(parts) == 3:
        # Three parts, they're start, stop, and step
        if parts[0]:
            start = int(parts[0])
        if parts[1]:
            stop = int(parts[1])
        if parts[2]:
            step = int(parts[2])

    # Create a slice object
    if start is None:
        start = 0
    if stop is None:
        stop = len(input_list)
    if step is None:
        step = 1

    # Slice the list
    sliced_list = input_list[start:stop:step]

    return sliced_list

print(slice_list(input_list, '::2'))  # Output: [1, 3, 5, 7, 9]
```
