## Exercise
Create a dictionary with names as keys and ages as values. Write a function to look up the age of a person given their name.

## Reference Solution
```python
# Create a dictionary with names as keys and ages as values
people = {
    "John": 25,
    "Alice": 30,
    "Bob": 35,
    "Charlie": 20
}

def look_up_age(name):
    """
    Lookup the age of a person given their name.

    Args:
        name (str): The name of the person.

    Returns:
        int: The age of the person if found, otherwise None.
    """
    return people.get(name)

# Example usage:
def main():
    print("People and their ages:")
    for name, age in people.items():
        print(f"{name}: {age}")

    name_to_lookup = "Alice"
    age = look_up_age(name_to_lookup)
    if age is not None:
        print(f"The age of {name_to_lookup} is {age}.")
    else:
        print(f"No age found for {name_to_lookup}.")

if __name__ == "__main__":
    main()
```
