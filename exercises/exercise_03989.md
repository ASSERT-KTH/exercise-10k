## Exercise
Write a program that finds the most frequent character in a given string.

## Reference Solution
```python
def most_frequent_char(s):
    """
    Finds the most frequent character in a given string.

    Args:
        s (str): The input string.

    Returns:
        str: The most frequent character.
    """
    # Create a dictionary to count character occurrences
    char_count = {}
    for char in s:
        if char in char_count:
            char_count[char] += 1
        else:
            char_count[char] = 1

    # Find the character with the highest count
    max_count = max(char_count.values())
    most_frequent = [char for char, count in char_count.items() if count == max_count]

    # Return the most frequent character(s)
    if len(most_frequent) == 1:
        return most_frequent[0]
    else:
        return most_frequent

# Example usage
def main():
    s = "hello world"
    print("Most frequent character(s):", most_frequent_char(s))

if __name__ == "__main__":
    main()
```
