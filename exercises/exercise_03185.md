## Exercise
Create a program that takes a password as input and uses if/else statements to check if it meets certain criteria (length, characters, etc.).

## Reference Solution
```python
def validate_password(password):
    """
    Validate a password based on the following criteria:
    - Length: at least 8 characters
    - Characters: at least one uppercase letter, one lowercase letter, and one special character
    - Digits: at least one digit
    """
    errors = []

    # Check length
    if len(password) < 8:
        errors.append("Password should be at least 8 characters long.")

    # Check characters
    if not any(char.isupper() for char in password):
        errors.append("Password should have at least one uppercase letter.")
    if not any(char.islower() for char in password):
        errors.append("Password should have at least one lowercase letter.")
    if not any(not char.isalnum() for char in password):
        errors.append("Password should have at least one special character.")

    # Check digits
    if not any(char.isdigit() for char in password):
        errors.append("Password should have at least one digit.")

    if errors:
        return False, errors
    else:
        return True, None

def main():
    password = input("Enter a password: ")
    is_valid, errors = validate_password(password)

    if is_valid:
        print("Password is valid.")
    else:
        print("Password is not valid. Please fix the following errors:")
        for error in errors:
            print(error)

if __name__ == "__main__":
    main()
```
