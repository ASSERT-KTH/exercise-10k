## Exercise
Write a program that takes a list and a reduction function as input, and returns the result of applying the reduction function to all elements in the list.

## Reference Solution
```python
def reduction(list, func):
    result = list[0]
    i = 1
    while i < len(list):
        result = func(result, list[i])
        i += 1
    return result

def find_max(a, b):
    return a if a > b else b

def find_min(a, b):
    return a if a < b else b

def find_sum(a, b):
    return a + b

def find_product(a, b):
    return a * b

def find_concat(a, b):
    return a + b

def main():
    # Test cases
    print(reduction([1, 2, 3, 4, 5], find_max))  # Output: 5
    print(reduction([1, 2, 3, 4, 5], find_min))  # Output: 1
    print(reduction([1, 2, 3, 4, 5], find_sum))  # Output: 15
    print(reduction([1, 2, 3, 4, 5], find_product))  # Output: 120
    print(reduction(["hello", "world"], find_concat))  # Output: "helloworld"

if __name__ == "__main__":
    main()
```
