## Exercise
Write a function to encode a given string using the ROT13 algorithm

def rot13_encode(input_string: str) -> str:
    """
    Encode a given string using the ROT13 algorithm.

    Args:
    input_string (str): The string to be encoded.

    Returns:
    str: The encoded string.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    encoded_string = ''

    for char in input_string:
        if char.isalpha():
            # Find the position in the alphabet for the character
            position = alphabet.index(char.lower())
            # Calculate the new position by adding 13 and taking the modulus
            new_position = (position + 13) % 26
            # If the original character was uppercase, convert the encoded character to uppercase
            if char.isupper():
                encoded_string += alphabet[new_position].upper()
            else:
                encoded_string += alphabet[new_position]
        else:
            # If the character is not a letter, leave it unchanged
            encoded_string += char

    return encoded_string

# Example usage:
if __name__ == "__main__":
    input_str = "Hello, World!"
    encoded_str = rot13_encode(input_str)
    print(f"Input: {input_str}")
    print(f"Encoded: {encoded_str}")
