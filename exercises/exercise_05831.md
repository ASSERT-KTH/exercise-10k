## Exercise
Write a function to justify a given text by aligning the words to the left, right, or center

def justify_text(text, width, alignment='left'):
    """
    Justify a given text by aligning the words to the left, right, or center.

    Args:
        text (str): The input text.
        width (int): The width of the justified text.
        alignment (str, optional): The alignment of the text. Defaults to 'left'.

    Returns:
        str: The justified text.
    """

    # Split the text into words
    words = text.split()

    # Initialize the justified text
    justified_text = ''

    # Initialize the current line
    line = ''

    # Iterate over the words
    for word in words:
        # If adding the word to the current line would exceed the width
        if len(line) + len(word) + 1 > width:
            # Justify the current line
            justified_text += justify_line(line, width, alignment) + '\n'
            # Reset the current line
            line = word
        else:
            # Add the word to the current line
            if line:
                line += ' '
            line += word

    # Justify the last line
    justified_text += justify_line(line, width, alignment)

    return justified_text


def justify_line(line, width, alignment):
    """
    Justify a single line of text.

    Args:
        line (str): The input line.
        width (int): The width of the justified line.
        alignment (str): The alignment of the line.

    Returns:
        str: The justified line.
    """

    # Split the line into words
    words = line.split()

    # Calculate the number of spaces needed
    num_spaces = width - sum(len(word) for word in words)

    # Justify the line based on the alignment
    if alignment == 'left':
        return line + ' ' * num_spaces
    elif alignment == 'right':
        return ' ' * num_spaces + line
    elif alignment == 'center':
        left_spaces = num_spaces // 2
        right_spaces = num_spaces - left_spaces
        return ' ' * left_spaces + line + ' ' * right_spaces
