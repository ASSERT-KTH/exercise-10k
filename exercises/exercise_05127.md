## Exercise
Write a program to find all duplicate elements in an array.

## Reference Solution
```python
def find_duplicates(array):
    """
    Finds all duplicate elements in a given array.

    Args:
        array (list): The input array.

    Returns:
        list: A list of duplicate elements.
    """
    seen = set()
    duplicates = set()

    for num in array:
        if num in seen:
            duplicates.add(num)
        seen.add(num)

    return list(duplicates)


def main():
    # Example usage
    array = [1, 2, 3, 2, 4, 5, 6, 2, 3]
    duplicates = find_duplicates(array)

    print("Array:", array)
    print("Duplicate elements:", duplicates)


if __name__ == "__main__":
    main()
```
