## Exercise
Write a program that asks for a list of ASCII values and decodes them into a string.

## Reference Solution
```python
def ascii_to_string(ascii_values):
    """
    This function takes a list of ASCII values and decodes them into a string.

    Args:
        ascii_values (list): A list of ASCII values.

    Returns:
        str: The decoded string.
    """
    return ''.join(chr(value) for value in ascii_values)


def main():
    # Get the list of ASCII values from the user
    ascii_values = input("Enter a list of ASCII values separated by spaces: ")

    # Convert the input string into a list of integers
    try:
        ascii_values = [int(value) for value in ascii_values.split()]
    except ValueError:
        print("Invalid input. Please enter a list of integers separated by spaces.")
        return

    # Decode the ASCII values into a string
    decoded_string = ascii_to_string(ascii_values)

    # Print the decoded string
    print("The decoded string is:", decoded_string)


if __name__ == "__main__":
    main()
```
