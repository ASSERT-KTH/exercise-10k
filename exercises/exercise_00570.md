## Exercise
Write a function that calculates the frequency of each word in a given string.

## Reference Solution
```python
import re
from collections import Counter

def calculate_word_frequency(input_string):
    """
    Calculate the frequency of each word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where the keys are the words and the values are their frequencies.
    """
    # Convert the string to lowercase
    input_string = input_string.lower()
    
    # Remove punctuation
    input_string = re.sub(r'[^\w\s]', '', input_string)
    
    # Split the string into words
    words = input_string.split()
    
    # Count the frequency of each word
    frequency = Counter(words)
    
    return dict(frequency)

# Example usage
def main():
    input_string = "This is a test string. This string is just a test."
    frequency = calculate_word_frequency(input_string)
    print(frequency)

if __name__ == "__main__":
    main()
```
