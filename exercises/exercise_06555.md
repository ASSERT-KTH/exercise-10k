## Exercise
Create a function that counts the number of words in a given string.

## Reference Solution
```python
def count_words(input_string: str) -> int:
    """
    Counts the number of words in a given string.

    Args:
        input_string (str): The input string to count words from.

    Returns:
        int: The number of words in the input string.
    """
    # Remove leading and trailing whitespace
    input_string = input_string.strip()
    
    # Split the string into words
    words = input_string.split()
    
    # Return the number of words
    return len(words)

# Example usage
def main():
    input_str = "Hello World, this is a test string."
    word_count = count_words(input_str)
    print(f"The string '{input_str}' contains {word_count} words.")

if __name__ == "__main__":
    main()
```
