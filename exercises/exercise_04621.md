## Exercise
Write a program that takes an array and an index as input, and returns the element at the specified index.

## Reference Solution

// Author: [Your Name]
// Created: [Current Date]
// Purpose: Returns the element at the specified index in an array

#include <iostream>
#include <vector>

int printArray(std::vector<int> arr, int index) {
    if (index < arr.size()) {
        return arr[index];
    } else {
        return -1; // or throw an exception
    }
}

int main() {
    std::vector<int> arr = {1, 2, 3, 4, 5};
    int index = 2;

    int result = printArray(arr, index);
    if (result != -1) {
        std::cout << "Element at index " << index << ": " << result << std::endl;
    } else {
        std::cout << "Index out of bounds" << std::endl;
    }

    return 0;
}
