## Exercise
Write a program that draws different shapes on a canvas, using conditional statements to determine the shape type

## Reference Solution
```python
import tkinter as tk

def draw_shape(canvas, shape_type, x, y, size):
    """
    Draws a shape on the canvas based on the shape type.
    
    Args:
        canvas (tkinter.Canvas): The canvas to draw on.
        shape_type (str): The type of shape to draw. Can be 'circle', 'square', 'triangle', or 'rectangle'.
        x (int): The x-coordinate of the shape.
        y (int): The y-coordinate of the shape.
        size (int): The size of the shape.
    """
    if shape_type == 'circle':
        # Draw a circle
        canvas.create_oval(x, y, x + size, y + size)
    elif shape_type == 'square':
        # Draw a square
        canvas.create_rectangle(x, y, x + size, y + size)
    elif shape_type == 'triangle':
        # Draw a triangle
        canvas.create_polygon(x, y + size, x + size / 2, y, x + size, y + size)
    elif shape_type == 'rectangle':
        # Draw a rectangle
        canvas.create_rectangle(x, y, x + size * 2, y + size)
    else:
        print("Invalid shape type")

def main():
    # Create a new Tkinter window
    window = tk.Tk()
    window.title("Shape Drawer")
    
    # Create a new canvas
    canvas = tk.Canvas(window, width=800, height=600)
    canvas.pack()
    
    # Draw shapes
    draw_shape(canvas, 'circle', 50, 50, 100)
    draw_shape(canvas, 'square', 200, 50, 100)
    draw_shape(canvas, 'triangle', 350, 50, 100)
    draw_shape(canvas, 'rectangle', 50, 200, 100)
    
    # Start the Tkinter event loop
    window.mainloop()

if __name__ == "__main__":
    main()
```
