## Exercise
Write a program that takes a text as input and encrypts it using a Caesar cipher.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts the input text using a Caesar cipher with the given shift.

    Args:
        text (str): The input text to be encrypted.
        shift (int): The number of positions to shift the letters.

    Returns:
        str: The encrypted text.
    """
    result = ""

    for char in text:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            result += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            result += char

    return result

def main():
    text = input("Enter the text to encrypt: ")
    shift = int(input("Enter the shift value: "))

    encrypted_text = caesar_cipher(text, shift)
    print("Encrypted text:", encrypted_text)

if __name__ == "__main__":
    main()
```
