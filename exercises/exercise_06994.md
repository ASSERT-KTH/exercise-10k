## Exercise
Write a program that takes an amount in one currency as input and returns its equivalent in another currency.

## Reference Solution
```python
from forex_python.converter import CurrencyRates

def convert_currency(amount, from_currency, to_currency):
    c = CurrencyRates()
    result = c.convert(from_currency, to_currency, amount)
    return result

def main():
    amount = float(input("Enter the amount: "))
    from_currency = input("Enter the currency to convert from (e.g., USD, EUR, JPY): ").upper()
    to_currency = input("Enter the currency to convert to (e.g., USD, EUR, JPY): ").upper()

    try:
        result = convert_currency(amount, from_currency, to_currency)
        print(f"{amount} {from_currency} is equal to {result} {to_currency}")
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()
```
