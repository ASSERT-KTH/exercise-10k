## Exercise
Write a program that takes a list of numbers as input and returns the first duplicate value.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Find the first duplicate value in a list of numbers.

    Args:
        nums (list): A list of numbers.

    Returns:
        int or None: The first duplicate value, or None if no duplicates are found.
    """
    seen = set()
    for num in nums:
        if num in seen:
            return num
        seen.add(num)
    return None

# Example usage
def main():
    nums = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(nums)
    if duplicate is not None:
        print(f"The first duplicate value is: {duplicate}")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
