## Exercise
Write a program that splits a bill among a group of people.

## Reference Solution
```python
def calculate_split_bill(total_bill, tax_rate, tip_rate, people):
    """
    Calculates the split bill among a group of people.

    Args:
    - total_bill (float): The total bill before tax and tip.
    - tax_rate (float): The tax rate as a decimal (e.g., 8% = 0.08).
    - tip_rate (float): The tip rate as a decimal (e.g., 15% = 0.15).
    - people (int): The number of people to split the bill.

    Returns:
    - float: The amount each person owes.
    """
    # Calculate the total bill with tax and tip
    total_bill_with_tax = total_bill * (1 + tax_rate)
    total_bill_with_tax_and_tip = total_bill_with_tax * (1 + tip_rate)

    # Calculate the split bill
    split_bill = total_bill_with_tax_and_tip / people

    return split_bill


def main():
    print("Bill Splitter Program")

    # Get the total bill from the user
    while True:
        try:
            total_bill = float(input("Enter the total bill: $"))
            if total_bill < 0:
                print("Total bill cannot be negative.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tax rate from the user
    while True:
        try:
            tax_rate = float(input("Enter the tax rate (in %): "))
            if tax_rate < 0:
                print("Tax rate cannot be negative.")
                continue
            tax_rate /= 100  # Convert tax rate to decimal
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip rate from the user
    while True:
        try:
            tip_rate = float(input("Enter the tip rate (in %): "))
            if tip_rate < 0:
                print("Tip rate cannot be negative.")
                continue
            tip_rate /= 100  # Convert tip rate to decimal
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the number of people from the user
    while True:
        try:
            people = int(input("Enter the number of people: "))
            if people <= 0:
                print("Number of people must be greater than 0.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a valid integer.")

    # Calculate and print the split bill
    split_bill = calculate_split_bill(total_bill, tax_rate, tip_rate, people)
    print(f"Each person owes: ${split_bill:.2f}")


if __name__ == "__main__":
    main()
```
