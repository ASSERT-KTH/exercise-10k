## Exercise
Write a program to calculate the monthly mortgage payment given the loan amount, interest rate, and number of years.

## Reference Solution
```python
def calculate_mortgage_payment(loan_amount, interest_rate, years):
    """
    Calculate the monthly mortgage payment.

    Args:
        loan_amount (float): The loan amount.
        interest_rate (float): The annual interest rate as a decimal.
        years (int): The number of years.

    Returns:
        float: The monthly mortgage payment.
    """
    # Convert annual interest rate to monthly interest rate
    monthly_interest_rate = interest_rate / 12

    # Calculate the number of months
    months = years * 12

    # Calculate the monthly mortgage payment
    monthly_payment = (loan_amount * monthly_interest_rate * (1 + monthly_interest_rate) ** months) / ((1 + monthly_interest_rate) ** months - 1)

    return monthly_payment


def main():
    # Get user input
    loan_amount = float(input("Enter the loan amount: "))
    interest_rate = float(input("Enter the annual interest rate (in percent): ")) / 100
    years = int(input("Enter the number of years: "))

    # Calculate the monthly mortgage payment
    monthly_payment = calculate_mortgage_payment(loan_amount, interest_rate, years)

    # Print the result
    print(f"The monthly mortgage payment is: ${monthly_payment:.2f}")


if __name__ == "__main__":
    main()
```
