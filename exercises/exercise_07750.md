## Exercise
Create a program that generates a looping pattern of numbers based on user input. The pattern should be generated using a while loop or a for loop. The user should input the starting number, ending number, and the increment value. The program should then print out the pattern.

## Reference Solution
```python
def generate_pattern():
    # Get user input
    start_num = int(input("Enter the starting number: "))
    end_num = int(input("Enter the ending number: "))
    increment = int(input("Enter the increment value: "))

    # Check if the increment is positive or negative
    if increment > 0:
        # Generate the pattern using a for loop
        print("Pattern:")
        for i in range(start_num, end_num + 1, increment):
            print(i)
    elif increment < 0:
        # Generate the pattern using a for loop
        print("Pattern:")
        for i in range(start_num, end_num - 1, increment):
            print(i)
    else:
        print("Error: Increment cannot be zero.")

def generate_pattern_while():
    # Get user input
    start_num = int(input("Enter the starting number: "))
    end_num = int(input("Enter the ending number: "))
    increment = int(input("Enter the increment value: "))

    # Check if the increment is positive or negative
    if increment > 0:
        # Generate the pattern using a while loop
        print("Pattern:")
        i = start_num
        while i <= end_num:
            print(i)
            i += increment
    elif increment < 0:
        # Generate the pattern using a while loop
        print("Pattern:")
        i = start_num
        while i >= end_num:
            print(i)
            i += increment
    else:
        print("Error: Increment cannot be zero.")

def main():
    print("Pattern Generator")
    print("1. Generate pattern using for loop")
    print("2. Generate pattern using while loop")
    choice = int(input("Enter your choice: "))
    
    if choice == 1:
        generate_pattern()
    elif choice == 2:
        generate_pattern_while()
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
