## Exercise
Create a function that takes a string as input and returns the longest substring without repeating characters

## Reference Solution
```python
def longest_substring_without_repeating_chars(input_str):
    """
    Returns the longest substring without repeating characters.

    Args:
        input_str (str): The input string.

    Returns:
        str: The longest substring without repeating characters.
    """
    if not input_str:
        return ""

    left = 0
    max_length = 0
    max_substring = ""
    char_set = set()

    for right in range(len(input_str)):
        while input_str[right] in char_set:
            char_set.remove(input_str[left])
            left += 1

        char_set.add(input_str[right])

        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = input_str[left:right + 1]

    return max_substring


# Example usage
if __name__ == "__main__":

    print(longest_substring_without_repeating_chars("bbbbb"))  # Output: "b"
    print(longest_substring_without_repeating_chars("pwwkew"))  # Output: "wke"
    print(longest_substring_without_repeating_chars(""))  # Output: ""
```
