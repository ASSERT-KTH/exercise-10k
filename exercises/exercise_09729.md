## Exercise
Write a program that takes two points as input and returns the distance between them.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.

    Args:
    point1 (tuple): The coordinates of the first point as a tuple (x, y).
    point2 (tuple): The coordinates of the second point as a tuple (x, y).

    Returns:
    float: The Euclidean distance between the two points.
    """

    # Unpack the coordinates from the input tuples
    x1, y1 = point1
    x2, y2 = point2

    # Calculate the difference in x and y coordinates
    dx = x2 - x1
    dy = y2 - y1

    # Calculate the Euclidean distance using the formula sqrt(dx^2 + dy^2)
    distance = math.sqrt(dx ** 2 + dy ** 2)

    return distance

# Example usage
if __name__ == "__main__":
    point1 = (1, 2)  # coordinates of the first point
    point2 = (4, 6)  # coordinates of the second point

    distance = calculate_distance(point1, point2)
    print(f"The distance between {point1} and {point2} is {distance:.2f}")
```
