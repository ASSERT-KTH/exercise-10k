## Exercise
Write a program that asks the user for a subject, verb, and object, and prints out a sentence using those words

## Reference Solution
```python
def get_user_input(prompt):
    """Get user input for a given prompt"""
    while True:
        user_input = input(prompt)
        if user_input:
            return user_input
        print("Please enter a value.")

def generate_sentence(subject, verb, object):
    """Generate a sentence using the given subject, verb, and object"""
    return f"The {subject} will {verb} the {object}."

def main():
    # Get user input for subject, verb, and object
    subject = get_user_input("Enter a subject (e.g. dog, cat, car): ")
    verb = get_user_input("Enter a verb (e.g. run, jump, drive): ")
    object = get_user_input("Enter an object (e.g. ball, tree, road): ")

    # Generate and print the sentence
    sentence = generate_sentence(subject, verb, object)
    print(sentence)

if __name__ == "__main__":
    main()
```
