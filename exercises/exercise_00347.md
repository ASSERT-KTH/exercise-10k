## Exercise
Write a program to reduce an array to a single value using a reduction operation.

## Reference Solution
```python
import numpy as np

def reduce_array(array, operation):
    """
    Reduces an array to a single value using a reduction operation.

    Args:
        array (list): The input array.
        operation (str): The reduction operation to apply. Can be 'sum', 'product', 'min', or 'max'.

    Returns:
        The reduced value.
    """
    if operation == 'sum':
        return np.sum(array)
    elif operation == 'product':
        return np.prod(array)
    elif operation == 'min':
        return np.min(array)
    elif operation == 'max':
        return np.max(array)
    else:
        raise ValueError("Invalid operation. Must be 'sum', 'product', 'min', or 'max'.")

# Example usage
array = [1, 2, 3, 4, 5]
operations = ['sum', 'product', 'min', 'max']

for operation in operations:
    result = reduce_array(array, operation)
    print(f"{operation.capitalize()}: {result}")
```
