## Exercise
Write a program that takes input from the user with buffering and prints out the results.

## Reference Solution
```python
# Define a class for the input buffer
class InputBuffer:
    def __init__(self):
        self.buffer = []

    def add_input(self, user_input):
        self.buffer.append(user_input)

    def print_buffer(self):
        for i, input_str in enumerate(self.buffer):
            print(f"Input {i+1}: {input_str}")


def main():
    # Create an instance of the input buffer
    buffer = InputBuffer()

    while True:
        # Get user input
        user_input = input("Please enter some text (or 'quit' to stop): ")

        # Check if the user wants to quit
        if user_input.lower() == 'quit':
            break

        # Add the input to the buffer
        buffer.add_input(user_input)

    # Print out the buffer
    print("\nYou entered the following text:")
    buffer.print_buffer()


if __name__ == "__main__":
    main()
```
