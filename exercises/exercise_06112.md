## Exercise
Write a program that takes a list of integers as input, and returns the index of the first occurrence of a specified value in the list.

## Reference Solution
```python
#include <iostream>
#include <vector>

int findIndex(const std::vector<int>& arr, int val) {
    for (auto it = arr.begin(); it != arr.end(); ++it) {
        if (*it == val) {
            return std::distance(arr.begin(), it);
        }
    }
    return -1;
}

int main() {
    std::vector<int> arr = {0, 1, 2, 3, 4, 5};
    int val = 2;

    int index = findIndex(arr, val);
    std::cout << "Index of " << val << ": " << index << std::endl;

    return 0;
}
```
