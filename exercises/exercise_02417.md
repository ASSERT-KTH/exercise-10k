## Exercise
Write a function that uses abstract classes to define an interface. Then, call this function with an input value and print the result.

## Reference Solution
```python
#include <iostream>

// Define an abstract class
class AbstractClass {
public:
    // Define an abstract method
    virtual int do_something(int input_value) = 0;
};

// Define a concrete class that implements the abstract class
class ConcreteClass : public AbstractClass {
public:
    int do_something(int input_value) override {
        return input_value * 2;
    }
};

// Define a function that uses the abstract class as an interface
int use_abstract_class(AbstractClass& abstract_class_instance, int input_value) {
    return abstract_class_instance.do_something(input_value);
}

int main() {
    // Create an instance of the concrete class
    ConcreteClass concrete_class_instance;

    // Call the function with the concrete class instance and an input value
    int result = use_abstract_class(concrete_class_instance, 10);

    // Print the result
    std::cout << result << std::endl;  // Output: 20

    return 0;
}
```
