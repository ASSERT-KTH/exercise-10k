## Exercise
Create a program that takes two numbers as input and returns their integer division.

## Reference Solution
```python
def divide_numbers(dividend: int, divisor: int) -> int:
    """
    This function takes two numbers as input and returns their integer division.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        int: The integer division of the two numbers.

    Raises:
        ZeroDivisionError: If the divisor is zero.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend // divisor


def main():
    # Example usage:
    try:
        num1 = int(input("Enter the first number: "))
        num2 = int(input("Enter the second number: "))
        result = divide_numbers(num1, num2)
        print(f"The integer division of {num1} and {num2} is: {result}")
    except ZeroDivisionError as e:
        print(e)
    except ValueError:
        print("Invalid input. Please enter an integer.")


if __name__ == "__main__":
    main()
```
