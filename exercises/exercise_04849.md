## Exercise
Write a program that takes an array as input, and returns its version with duplicates removed.

## Reference Solution
```python
def remove_duplicates(array):
    """
    Returns a new array with duplicates removed.

    Args:
        array (list): The input array.

    Returns:
        list: A new array with duplicates removed.
    """
    return list(set(array))

# Example usage
if __name__ == "__main__":
    array = [1, 2, 2, 3, 4, 4, 5, 6, 6]
    print("Original array:", array)
    print("Array without duplicates:", remove_duplicates(array))
```

```python
def remove_duplicates_ordered(array):
    """
    Returns a new array with duplicates removed, preserving the original order.

    Args:
        array (list): The input array.

    Returns:
        list: A new array with duplicates removed.
    """
    seen = set()
    return [x for x in array if not (x in seen or seen.add(x))]

# Example usage
if __name__ == "__main__":
    array = [1, 2, 2, 3, 4, 4, 5, 6, 6]
    print("Original array:", array)
    print("Array without duplicates (ordered):", remove_duplicates_ordered(array))
```

```python
import numpy as np

def remove_duplicates_numpy(array):
    """
    Returns a new array with duplicates removed.

    Args:
        array (numpy.ndarray): The input array.

    Returns:
        numpy.ndarray: A new array with duplicates removed.
    """
    return np.unique(array)

# Example usage
if __name__ == "__main__":
    array = np.array([1, 2, 2, 3, 4, 4, 5, 6, 6])
    print("Original array:", array)
    print("Array without duplicates (NumPy):", remove_duplicates_numpy(array))
```
