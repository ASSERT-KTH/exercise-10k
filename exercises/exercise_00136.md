## Exercise
Create a program that simplifies a given fraction by dividing both the numerator and denominator by their greatest common divisor.

## Reference Solution
```python
import math

def simplify_fraction(numerator, denominator):
    """
    Simplify a fraction by dividing both the numerator and denominator by their greatest common divisor.

    Args:
        numerator (int): The numerator of the fraction.
        denominator (int): The denominator of the fraction.

    Returns:
        tuple: A tuple containing the simplified numerator and denominator.

    Raises:
        ValueError: If the denominator is zero.
    """

    # Check if the denominator is zero to avoid division by zero error
    if denominator == 0:
        raise ValueError("Denominator cannot be zero.")

    # Calculate the greatest common divisor of the numerator and denominator
    gcd = math.gcd(numerator, denominator)

    # Simplify the fraction by dividing both the numerator and denominator by their greatest common divisor
    simplified_numerator = numerator // gcd
    simplified_denominator = denominator // gcd

    return simplified_numerator, simplified_denominator


def main():
    # Test the function
    numerator = 12
    denominator = 18
    simplified_numerator, simplified_denominator = simplify_fraction(numerator, denominator)
    print(f"The simplified fraction of {numerator}/{denominator} is {simplified_numerator}/{simplified_denominator}")


if __name__ == "__main__":
    main()
```
