## Exercise
Write a program to generate the Fibonacci sequence up to the nth term.

## Reference Solution
```python
def generate_fibonacci(n):
    """
    Generate the Fibonacci sequence up to the nth term.

    Args:
        n (int): The number of terms in the sequence.

    Returns:
        list: A list of integers representing the Fibonacci sequence.
    """
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]

    sequence = [0, 1]
    while len(sequence) < n:
        sequence.append(sequence[-1] + sequence[-2])

    return sequence

def main():
    n = int(input("Enter the number of terms: "))
    fibonacci_sequence = generate_fibonacci(n)
    print(f"Fibonacci sequence up to {n} terms: {fibonacci_sequence}")

if __name__ == "__main__":
    main()
```
