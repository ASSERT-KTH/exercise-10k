## Exercise
Given a dictionary with names of cities as keys and their populations as values, write a program to convert the dictionary to a list of tuples and print the list.

## Reference Solution
```python
def dict_to_list_of_tuples(city_populations: dict) -> list:
    """
    Converts a dictionary with city names as keys and their populations as values to a list of tuples.

    Args:
        city_populations (dict): A dictionary with city names as keys and their populations as values.

    Returns:
        list: A list of tuples, where each tuple contains a city name and its population.
    """
    # Use the items() method to get a view object that displays a list of a dictionaryâ€™s key-value tuple pairs
    list_of_tuples = list(city_populations.items())
    
    return list_of_tuples


def main():
    # Example dictionary with city names as keys and their populations as values
    city_populations = {
        "New York": 8405837,
        "Los Angeles": 3990456,
        "Chicago": 2698459,
        "Houston": 2311793,
        "Phoenix": 1609474
    }

    # Convert the dictionary to a list of tuples
    list_of_tuples = dict_to_list_of_tuples(city_populations)

    # Print the list of tuples
    print("List of tuples:")
    for i, (city, population) in enumerate(list_of_tuples):
        print(f"{i+1}. {city}: {population}")


if __name__ == "__main__":
    main()
```
