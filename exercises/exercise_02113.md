## Exercise
Write a function to update a dictionary with the key-value pairs from another dictionary

## Reference Solution
```python
def update_dictionary(dict1, dict2):
    """
    Updates dict1 with key-value pairs from dict2.

    Args:
        dict1 (dict): The dictionary to be updated.
        dict2 (dict): The dictionary with key-value pairs to update dict1.

    Returns:
        dict: The updated dictionary.
    """
    # Use the built-in update method of dictionaries
    dict1.update(dict2)
    return dict1

# Example usage
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2}
    dict2 = {"b": 3, "c": 4}

    print("Before update:")
    print("dict1:", dict1)
    print("dict2:", dict2)

    updated_dict = update_dictionary(dict1, dict2)

    print("\nAfter update:")
    print("dict1 (updated):", updated_dict)
```
