## Exercise
Write a program to demonstrate the precedence of basic operators, including arithmetic, comparison, and logical operators.

## Reference Solution
```python
# Arithmetic operators
print("Arithmetic Operators:")
print("2 + 3 * 4 =", 2 + 3 * 4)  # Expected output: 14
print("2 + 3 / 4 =", 2 + 3 / 4)  # Expected output: 2.75
print("2 + 3 ** 4 =", 2 + 3 ** 4)  # Expected output: 83

# Comparison operators
print("\nComparison Operators:")
print("2 > 3 and 4 < 5 =", 2 > 3 and 4 < 5)  # Expected output: False
print("2 > 3 or 4 < 5 =", 2 > 3 or 4 < 5)  # Expected output: True

# Logical operators
print("\nLogical Operators:")
print("True and False =", True and False)  # Expected output: False
print("True or False =", True or False)  # Expected output: True
print("not True =", not True)  # Expected output: False

# Precedence of logical operators
print("\nPrecedence of Logical Operators:")

print("(True and False) or True =", (True and False) or True)  # Expected output: True
print("True and (False or True) =", True and (False or True))  # Expected output: True

# Precedence of comparison operators
print("\nPrecedence of Comparison Operators:")
print("2 > 3 and 4 < 5 =", 2 > 3 and 4 < 5)  # Expected output: False
print("(2 > 3) and (4 < 5) =", (2 > 3) and (4 < 5))  # Expected output: False
print("2 > (3 and 4) < 5 =", 2 > (3 and 4) < 5)  # Expected output: True

# Precedence of arithmetic operators
print("\nPrecedence of Arithmetic Operators:")
print("2 + 3 * 4 =", 2 + 3 * 4)  # Expected output: 14
print("(2 + 3) * 4 =", (2 + 3) * 4)  # Expected output: 20
print("2 + (3 * 4) =", 2 + (3 * 4))  # Expected output: 14
```
