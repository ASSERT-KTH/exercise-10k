## Exercise
Write a program that calculates the sum of all even numbers from 1 to n using a for loop or a while loop. The program should prompt the user to enter a positive integer n.

## Reference Solution
```python
def sum_even_numbers(n):
    """
    Calculate the sum of all even numbers from 1 to n.
    
    Args:
        n (int): The upper limit of the range.
    
    Returns:
        int: The sum of all even numbers from 1 to n.
    """
    total = 0
    for i in range(1, n + 1):
        if i % 2 == 0:
            total += i
    return total

def main():
    # Prompt the user to enter a positive integer n
    while True:
        try:
            n = int(input("Enter a positive integer: "))
            if n <= 0:
                print("Please enter a positive integer.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a positive integer.")

    # Calculate and print the sum of even numbers
    sum_even = sum_even_numbers(n)
    print(f"The sum of all even numbers from 1 to {n} is: {sum_even}")

if __name__ == "__main__":
    main()
```
