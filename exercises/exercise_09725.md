## Exercise
Write a program that takes two octal numbers as input and returns their sum.

## Reference Solution
```python
def octal_to_decimal(octal):
    """
    Converts an octal number to decimal.

    Args:
        octal (str): The octal number to convert.

    Returns:
        int: The decimal equivalent of the octal number.
    """
    return int(octal, 8)

def decimal_to_octal(decimal):
    """
    Converts a decimal number to octal.

    Args:
        decimal (int): The decimal number to convert.

    Returns:
        str: The octal equivalent of the decimal number.
    """
    return oct(decimal)[2:]

def add_octal_numbers(octal1, octal2):
    """
    Adds two octal numbers.

    Args:
        octal1 (str): The first octal number.
        octal2 (str): The second octal number.

    Returns:
        str: The sum of the two octal numbers in octal.
    """
    decimal_sum = octal_to_decimal(octal1) + octal_to_decimal(octal2)
    return decimal_to_octal(decimal_sum)

def main():
    # Get the two octal numbers from the user
    octal1 = input("Enter the first octal number: ")
    octal2 = input("Enter the second octal number: ")

    # Add the two octal numbers
    sum_octal = add_octal_numbers(octal1, octal2)

    # Print the result
    print(f"The sum of {octal1} and {octal2} is {sum_octal}")

if __name__ == "__main__":
    main()
```
