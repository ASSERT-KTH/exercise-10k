## Reference Solution

# Define variables with different data types
var1 = 10  # integer
var2 = "10"  # string

# Compare variables using equality
if var1 == int(var2):
    print("Variables are equal")
else:
    print("Variables are not equal")

# Compare variables using inequality
if var1 != float(var2):
    print("Variables are not equal")
else:
    print("Variables are equal")

# Compare variables using greater than
if var1 > int(var2):
    print("var1 is greater than var2")
else:
    print("var1 is not greater than var2")
