## Exercise
Write a program that uses a while loop to solve the 0/1 Knapsack problem, given a set of items and a knapsack capacity.

## Reference Solution
```python
def knapsack(items, capacity):
    # Create a 2D array to store the maximum value for each subproblem
    dp = [[0 for _ in range(capacity + 1)] for _ in range(len(items) + 1)]

    row = 1
    while row <= len(items):
        col = 1
        while col <= capacity:
            if items[row - 1][1] > col:
                dp[row][col] = dp[row - 1][col]
            else:
                dp[row][col] = max(dp[row - 1][col], dp[row - 1][col - items[row - 1][1]] + items[row - 1][0])
            col += 1
        row += 1

    # Reconstruct the solution
    solution = []
    row, col = len(items), capacity
    while row > 0 and col > 0:
        if dp[row][col] != dp[row - 1][col]:
            solution.append(items[row - 1])
            col -= items[row - 1][1]
        row -= 1

    return dp[-1][-1], solution[::-1]

# Example usage
items = [(60, 10), (100, 20), (120, 30)]  # (value, weight)
capacity = 50
max_value, solution = knapsack(items, capacity)
print(f"Maximum value: {max_value}")
print(f"Solution: {solution}")
```
