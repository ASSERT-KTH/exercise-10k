## Exercise
Write a program that takes the coefficients of a linear equation as input and prints its solution.

## Reference Solution
```python
def solve_linear_equation(a, b, c):
    """
    Solves a linear equation of the form ax + b = c.
    
    Args:
        a (float): The coefficient of x.
        b (float): The constant term on the left side of the equation.
        c (float): The constant term on the right side of the equation.
    
    Returns:
        float: The solution to the equation.
    """
    if a == 0:
        if b == c:
            return "Infinite solutions"
        else:
            return "No solution"
    else:
        return (c - b) / a

def main():
    # Get the coefficients from the user
    a = float(input("Enter the coefficient of x: "))
    b = float(input("Enter the constant term on the left side of the equation: "))
    c = float(input("Enter the constant term on the right side of the equation: "))

    # Solve and print the solution
    solution = solve_linear_equation(a, b, c)
    print("The solution to the equation is:", solution)

if __name__ == "__main__":
    main()
```
