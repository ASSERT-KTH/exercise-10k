## Exercise
Write a program to calculate the salary of an employee given their basic salary, allowances, and deductions

## Reference Solution
```python
class Employee:
    def __init__(self, name, basic_salary):
        """
        Initialize an Employee object.

        Args:
            name (str): Employee name
            basic_salary (float): Basic salary
        """
        self.name = name
        self.basic_salary = basic_salary
        self.allowances = {}
        self.deductions = {}

    def add_allowance(self, name, amount):
        """
        Add an allowance to the employee's salary.

        Args:
            name (str): Allowance name
            amount (float): Allowance amount
        """
        self.allowances[name] = amount

    def add_deduction(self, name, amount):
        """
        Add a deduction to the employee's salary.

        Args:
            name (str): Deduction name
            amount (float): Deduction amount
        """
        self.deductions[name] = amount

    def calculate_salary(self):
        """
        Calculate the employee's salary.

        Returns:
            float: Employee's salary
        """
        total_allowances = sum(self.allowances.values())
        total_deductions = sum(self.deductions.values())
        salary = self.basic_salary + total_allowances - total_deductions
        return salary


def main():
    # Create an Employee object
    employee = Employee("John Doe", 50000)

    # Add allowances
    employee.add_allowance("Housing Allowance", 10000)
    employee.add_allowance("Transportation Allowance", 5000)

    # Add deductions
    employee.add_deduction("Tax Deduction", 5000)
    employee.add_deduction("Insurance Deduction", 2000)

    # Calculate and print the employee's salary
    salary = employee.calculate_salary()
    print(f"Employee Name: {employee.name}")
    print(f"Basic Salary: {employee.basic_salary}")
    print("Allowances:")
    for name, amount in employee.allowances.items():
        print(f"  - {name}: {amount}")
    print("Deductions:")
    for name, amount in employee.deductions.items():
        print(f"  - {name}: {amount}")
    print(f"Total Salary: {salary}")


if __name__ == "__main__":
    main()
```
