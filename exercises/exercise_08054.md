## Exercise
Write a program that uses an if/else statement to manage personal finances with budgeting based on user input.

## Reference Solution
```python
def get_user_input():
    """Get user input for income, expenses, and savings goals."""
    income = float(input("Enter your monthly income: $"))
    expenses = float(input("Enter your monthly expenses: $"))
    savings_goal = float(input("Enter your monthly savings goal: $"))
    return income, expenses, savings_goal

def calculate_budget(income, expenses, savings_goal):
    """Calculate budget based on user input."""
    if expenses > income:
        print("Error: Expenses cannot be greater than income.")
        return None
    else:
        remaining_income = income - expenses
        if remaining_income < savings_goal:
            print("Warning: You may not be able to meet your savings goal.")
        budget = {
            "income": income,
            "expenses": expenses,
            "savings": min(remaining_income, savings_goal),
            "remaining": remaining_income - min(remaining_income, savings_goal)
        }
        return budget

def print_budget(budget):
    """Print budget plan."""
    if budget is not None:
        print("\nBudget Plan:")
        print(f"Income: ${budget['income']:.2f}")
        print(f"Expenses: ${budget['expenses']:.2f}")
        print(f"Savings: ${budget['savings']:.2f}")
        print(f"Remaining: ${budget['remaining']:.2f}")

def main():
    income, expenses, savings_goal = get_user_input()
    budget = calculate_budget(income, expenses, savings_goal)
    print_budget(budget)

if __name__ == "__main__":
    main()
```
