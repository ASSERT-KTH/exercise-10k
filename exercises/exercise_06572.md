## Exercise
Write a function that reduces a dictionary to a single value

## Reference Solution
```python
from functools import reduce

def reduce_dict(dictionary, reduction_func):
    """
    Reduces a dictionary to a single value by applying a reduction function.

    Args:
    - dictionary (dict): The input dictionary.
    - reduction_func (function): A function that takes two arguments and returns a single value.

    Returns:
    - The result of applying the reduction function to the dictionary values.
    """
    # Get the dictionary values
    values = dictionary.values()

    # Check if the dictionary is empty
    if not values:
        return None

    # Apply the reduction function to the dictionary values
    return reduce(reduction_func, values)

# Example usage:
if __name__ == "__main__":
    # Define a sample dictionary
    sample_dict = {"a": 1, "b": 2, "c": 3}

    # Define a reduction function (e.g., sum)
    def sum_func(x, y):
        return x + y

    # Reduce the dictionary using the sum function
    result = reduce_dict(sample_dict, sum_func)
    print("Result:", result)  # Output: 6

    # Define another reduction function (e.g., product)
    def product_func(x, y):
        return x * y

    # Reduce the dictionary using the product function
    result = reduce_dict(sample_dict, product_func)
    print("Result:", result)  # Output: 6
```
