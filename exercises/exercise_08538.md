## Exercise
Create a program to calculate the surface area of a cylinder given its radius and height.

## Reference Solution
```python
import math

def calculate_cylinder_surface_area(radius, height):
    """
    Calculates the surface area of a cylinder given its radius and height.

    Args:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.

    Returns:
        float: The surface area of the cylinder.
    """
    if radius < 0 or height < 0:
        raise ValueError("Radius and height must be non-negative.")

    surface_area = 2 * math.pi * radius * (radius + height)
    return surface_area

def main():
    radius = float(input("Enter the radius of the cylinder: "))
    height = float(input("Enter the height of the cylinder: "))

    try:
        surface_area = calculate_cylinder_surface_area(radius, height)
        print(f"The surface area of the cylinder is: {surface_area:.2f}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
