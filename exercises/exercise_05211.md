## Exercise
Create a program that calculates the simple interest amount based on the principal, rate, and time provided by the user.

## Reference Solution
```python
def calculate_simple_interest(principal, rate, time):
    """
    Calculates the simple interest amount.

    Args:
        principal (float): The initial amount of money.
        rate (float): The interest rate (in decimal form, e.g., 4% = 0.04).
        time (float): The time period (in years).

    Returns:
        float: The simple interest amount.
    """
    interest = principal * rate * time
    return interest

def main():
    print("Simple Interest Calculator")
    print("---------------------------")

    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = float(input("Enter the time period (in years): "))

    # Calculate simple interest
    interest = calculate_simple_interest(principal, rate, time)

    # Calculate total amount
    total_amount = principal + interest

    # Display results
    print("\nResults:")
    print(f"Principal: ${principal:.2f}")
    print(f"Interest Rate: {rate*100:.2f}%")
    print(f"Time: {time:.2f} years")
    print(f"Simple Interest: ${interest:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
