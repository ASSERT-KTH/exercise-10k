## Exercise
Write a function that finds the cartesian product between two dictionaries

## Reference Solution
```python
import itertools

def cartesian_product(dict1, dict2):
    """
    Calculate the cartesian product of two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        list: A list of tuples, where each tuple contains a key-value pair from dict1 and a key-value pair from dict2.
    """
    # Get the key-value pairs from each dictionary
    items1 = dict1.items()
    items2 = dict2.items()

    # Calculate the cartesian product
    product = list(itertools.product(items1, items2))

    return product

# Example usage:
dict1 = {'a': 1, 'b': 2}
dict2 = {'x': 10, 'y': 20}

result = cartesian_product(dict1, dict2)
for pair in result:
    print(pair)
```
