## Exercise
Write a program that reads and writes strings of data

## Reference Solution
```python
def read_string(filename):
    """
    Reads a string from a file.

    Args:
        filename (str): The name of the file to read from.

    Returns:
        str: The string read from the file.
    """
    try:
        with open(filename, 'r') as file:
            return file.read()
    except FileNotFoundError:
        print(f"File '{filename}' not found.")
        return None


def write_string(filename, data):
    """
    Writes a string to a file.

    Args:
        filename (str): The name of the file to write to.
        data (str): The string to write to the file.
    """
    with open(filename, 'w') as file:
        file.write(data)


def main():
    filename = input("Enter the filename: ")
    action = input("Do you want to read or write? (r/w): ")

    if action.lower() == 'r':
        print(read_string(filename))
    elif action.lower() == 'w':
        data = input("Enter the string to write: ")
        write_string(filename, data)
        print(f"String written to {filename} successfully.")
    else:
        print("Invalid action. Please enter 'r' or 'w'.")


if __name__ == "__main__":
    main()
```
