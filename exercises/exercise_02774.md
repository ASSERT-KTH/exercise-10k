## Exercise
Write a program that takes a text as input and prints out statistics about the text (e.g. word count, character count, most frequent word).

## Reference Solution
```python
import re
from collections import Counter

def calculate_text_statistics(text):
    """
    Calculate statistics about the text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary containing the text statistics.
    """
    # Remove punctuation and convert to lowercase
    text = re.sub(r'[^\w\s]', '', text).lower()

    # Split the text into words
    words = text.split()

    # Calculate word count
    word_count = len(words)

    # Calculate character count
    character_count = len(text)

    # Calculate most frequent word
    word_frequencies = Counter(words)
    most_frequent_word = word_frequencies.most_common(1)[0][0]

    # Create a dictionary to store the statistics
    statistics = {
        'word_count': word_count,
        'character_count': character_count,
        'most_frequent_word': most_frequent_word
    }

    return statistics

def print_text_statistics(statistics):
    """
    Print the text statistics.

    Args:
        statistics (dict): A dictionary containing the text statistics.
    """
    print("Text Statistics:")
    print("----------------")
    print(f"Word Count: {statistics['word_count']}")
    print(f"Character Count: {statistics['character_count']}")
    print(f"Most Frequent Word: {statistics['most_frequent_word']}")

def main():
    text = input("Enter a text: ")
    statistics = calculate_text_statistics(text)
    print_text_statistics(statistics)

if __name__ == "__main__":
    main()
```
