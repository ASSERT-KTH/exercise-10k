## Exercise
Create a function that uses distributed computing to compute the sum of a list of numbers in parallel across multiple machines

## Reference Solution
```python
import dask
from dask.distributed import Client, LocalCluster
import numpy as np

def distributed_sum(numbers, machines):
    """
    Compute the sum of a list of numbers in parallel across multiple machines.

    Parameters:
    numbers (list): List of numbers to sum
    machines (list): List of machine hostnames or IP addresses

    Returns:
    int: Sum of the numbers
    """
    # Create a local cluster
    with LocalCluster() as cluster:
        # Create a client with the cluster
        with Client(cluster) as client:
            # Connect to remote machines
            client.cluster.scale(len(machines))
            futures = client.scatter(numbers)

            # Split the list into chunks for parallel processing
            chunk_size = len(numbers) // len(machines)
            chunks = [numbers[i:i + chunk_size] for i in range(0, len(numbers), chunk_size)]

            # Compute the sum of each chunk in parallel
            results = []
            for chunk in chunks:
                result = dask.delayed(sum)(chunk)
                results.append(result)

            # Compute the results
            results = dask.compute(*results)

            # Sum the results
            total_sum = sum(results)

    return total_sum

# Example usage
if __name__ == "__main__":
    numbers = np.random.randint(0, 100, size=1000).tolist()
    machines = ["machine1", "machine2", "machine3"]  # replace with your machine hostnames or IP addresses
    result = distributed_sum(numbers, machines)
    print("Sum:", result)
```
