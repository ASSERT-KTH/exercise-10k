## Exercise
Write a program that encrypts a message using the Caesar cipher algorithm.

## Reference Solution
```python
def caesar_cipher(message, shift, mode):
    """
    Encrypts or decrypts a message using the Caesar cipher algorithm.

    Args:
    message (str): The message to be encrypted or decrypted.
    shift (int): The number of positions each letter in the message should be shifted.
    mode (str): 'e' for encryption or 'd' for decryption.

    Returns:
    str: The encrypted or decrypted message.
    """

    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in message:
        if char.isalpha():
            # Find the position of the character in the alphabet
            index = alphabet.index(char.lower())

            # Shift the position based on the mode
            if mode == 'e':
                new_index = (index + shift) % 26
            elif mode == 'd':
                new_index = (index - shift) % 26

            # Preserve the original case of the character
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            # Non-alphabet characters are not shifted
            result += char

    return result


def main():
    while True:
        print("\nCaesar Cipher Program")
        print("1. Encrypt a message")
        print("2. Decrypt a message")
        print("3. Quit")

        choice = input("Choose an option: ")

        if choice == '1':
            message = input("Enter the message to encrypt: ")
            shift = int(input("Enter the shift value: "))
            encrypted_message = caesar_cipher(message, shift, 'e')
            print(f"Encrypted message: {encrypted_message}")

        elif choice == '2':
            message = input("Enter the message to decrypt: ")
            shift = int(input("Enter the shift value: "))
            decrypted_message = caesar_cipher(message, shift, 'd')
            print(f"Decrypted message: {decrypted_message}")

        elif choice == '3':
            break

        else:
            print("Invalid option. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
