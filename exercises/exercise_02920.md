## Exercise
Create a program to find the least frequent character in a given string.

## Reference Solution
```python
from collections import Counter

def least_frequent_char(s):
    """
    Finds the least frequent character in a given string.

    Args:
        s (str): The input string.

    Returns:
        str: The least frequent character.
    """
    # Count the frequency of each character
    char_count = Counter(s)

    # Find the minimum frequency
    min_freq = min(char_count.values())

    # Find all characters with the minimum frequency
    least_frequent_chars = [char for char, freq in char_count.items() if freq == min_freq]

    # Return the first least frequent character
    return least_frequent_chars[0]

def main():
    # Example usage
    s = "hello world"
    print(f"The least frequent character in '{s}' is: {least_frequent_char(s)}")

if __name__ == "__main__":
    main()
```
