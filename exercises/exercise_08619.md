## Exercise
Write a program that takes two strings as input and returns the shortest common supersequence.

def shortest_common_supersequence(str1, str2):
    """
    Returns the shortest common supersequence of two input strings.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        str: The shortest common supersequence of str1 and str2.
    """
    m = len(str1)
    n = len(str2)

    # Create a 2D array to store the lengths of the longest common subsequences
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # Fill the dp array in a bottom-up manner
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if str1[i - 1] == str2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # Reconstruct the SCS from the dp array
    result = []
    i, j = m, n
    while i > 0 and j > 0:
        if str1[i - 1] == str2[j - 1]:
            result.append(str1[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            result.append(str1[i - 1])
            i -= 1
        else:
            result.append(str2[j - 1])
            j -= 1

    # Add any remaining characters from str1 or str2
    while i > 0:
        result.append(str1[i - 1])
        i -= 1
    while j > 0:
        result.append(str2[j - 1])
        j -= 1

    # Return the SCS in the correct order
    return "".join(reversed(result))


# Example usage
if __name__ == "__main__":
    str1 = "ABCBDAB"
    str2 = "BDCABA"
    print("Shortest Common Supersequence:", shortest_common_supersequence(str1, str2))
