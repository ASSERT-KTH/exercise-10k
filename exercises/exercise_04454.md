## Exercise
Write a program that calculates a student's grade based on their scores on assignments and exams. The user should be able to input the scores and the program should then print out the grade.

## Reference Solution
```python
def calculate_grade():
    # Get the number of assignments and exams
    num_assignments = int(input("Enter the number of assignments: "))
    num_exams = int(input("Enter the number of exams: "))

    # Initialize total scores for assignments and exams
    total_assignment_score = 0
    total_exam_score = 0

    # Get scores for assignments
    print("\nEnter scores for assignments (out of 100):")
    for i in range(num_assignments):
        score = float(input(f"Assignment {i+1} score: "))
        total_assignment_score += score

    # Get scores for exams
    print("\nEnter scores for exams (out of 100):")
    for i in range(num_exams):
        score = float(input(f"Exam {i+1} score: "))
        total_exam_score += score

    # Calculate average scores for assignments and exams
    average_assignment_score = total_assignment_score / num_assignments
    average_exam_score = total_exam_score / num_exams

    # Calculate final grade (assuming assignments are 40% of the grade and exams are 60%)
    final_grade = (average_assignment_score * 0.4) + (average_exam_score * 0.6)

    # Print the final grade
    print(f"\nYour final grade is: {final_grade:.2f}%")

    # Determine the letter grade
    if final_grade >= 90:
        letter_grade = "A"
    elif final_grade >= 80:
        letter_grade = "B"
    elif final_grade >= 70:
        letter_grade = "C"
    elif final_grade >= 60:
        letter_grade = "D"
    else:
        letter_grade = "F"

    # Print the letter grade
    print(f"Your letter grade is: {letter_grade}")

calculate_grade()
```
