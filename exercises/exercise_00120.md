## Exercise
Write a program to check if an array is sorted in ascending order.

## Reference Solution
```python
def is_sorted(arr):
    """
    Check if an array is sorted in ascending order.

    Args:
        arr (list): The input array.

    Returns:
        bool: True if the array is sorted, False otherwise.
    """
    return arr == sorted(arr)

def is_sorted_efficient(arr):
    """
    Check if an array is sorted in ascending order (efficient version).

    Args:
        arr (list): The input array.

    Returns:
        bool: True if the array is sorted, False otherwise.
    """
    return all(arr[i] <= arr[i + 1] for i in range(len(arr) - 1))

# Example usage
if __name__ == "__main__":
    arr1 = [1, 2, 3, 4, 5]
    arr2 = [5, 4, 3, 2, 1]

    print(f"Is {arr1} sorted? {is_sorted(arr1)}")  # Expected output: True
    print(f"Is {arr1} sorted (efficient)? {is_sorted_efficient(arr1)}")  # Expected output: True
    print(f"Is {arr2} sorted? {is_sorted(arr2)}")  # Expected output: False
    print(f"Is {arr2} sorted (efficient)? {is_sorted_efficient(arr2)}")  # Expected output: False
```
