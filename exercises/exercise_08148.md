## Exercise
Create a program that validates a URL using if/else statements

## Reference Solution
```python
import re

def validate_url(url):
    """
    Validate a URL using if/else statements.

    Args:
    url (str): The URL to be validated.

    Returns:
    bool: True if the URL is valid, False otherwise.
    """

    # Check if the URL is empty
    if not url:
        return False

    # Check if the URL starts with http or https
    if not (url.startswith("http://") or url.startswith("https://")):
        return False

    # Check if the URL has a valid domain
    domain_pattern = r"^https?:\/\/[a-zA-Z0-9\-\.]+\.[a-zA-Z]{2,}$"
    if not re.match(domain_pattern, url):
        return False

    # Check if the URL has a valid path
    path_pattern = r"^https?:\/\/[a-zA-Z0-9\-\.]+\.[a-zA-Z]{2,}([\/][a-zA-Z0-9\-\.\/]*)?$"
    if not re.match(path_pattern, url):
        return False

    # If all checks pass, the URL is valid
    return True


def main():
    # Test the validate_url function
    urls = [
        "http://www.example.com",
        "https://example.com",
        "ftp://example.com",
        "http://example",
        "http://example.com/path/to/resource"
    ]

    for url in urls:
        print(f"URL: {url}, Valid: {validate_url(url)}")


if __name__ == "__main__":
    main()
```
