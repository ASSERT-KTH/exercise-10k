## Exercise
Write a function to find the symmetric difference between two dictionaries.

## Reference Solution
```python
def symmetric_diff(dict1, dict2):
    """
    Calculate the symmetric difference between two dictionaries.

    Args:
    dict1 (dict): The first dictionary.
    dict2 (dict): The second dictionary.

    Returns:
    dict: A dictionary containing key-value pairs that are in exactly one of the input dictionaries.
    """
    # Create a new dictionary to store the symmetric difference
    symmetric_diff_dict = {}

    # Iterate over the items in the first dictionary
    for key, value in dict1.items():
        # If the key is not in the second dictionary, add it to the symmetric difference
        if key not in dict2:
            symmetric_diff_dict[key] = value

    # Iterate over the items in the second dictionary
    for key, value in dict2.items():
        # If the key is not in the first dictionary, add it to the symmetric difference
        if key not in dict1:
            symmetric_diff_dict[key] = value
        # If the key is in both dictionaries but has different values, add it to the symmetric difference
        elif key in dict1 and dict1[key] != value:
            symmetric_diff_dict[key] = value

    return symmetric_diff_dict

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 20, 'c': 3, 'd': 4}

print(symmetric_diff(dict1, dict2))
```
