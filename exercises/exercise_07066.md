## Exercise
Write a program that takes a input as input and handles any errors that may occur.

## Reference Solution
```python
def get_user_input():
    """
    This function takes user input and handles potential errors.
    """
    while True:
        try:
            # Take user input
            user_input = input("Please enter something: ")
            
            # Try to convert the input to an integer (if needed)
            try:
                user_input = int(user_input)
            except ValueError:
                pass  # If conversion fails, keep the input as a string
            
            # Return the user input
            return user_input
        
        except KeyboardInterrupt:
            print("\nError: Input interrupted by the user.")
            continue
        except Exception as e:
            print(f"An error occurred: {e}")

def main():
    user_input = get_user_input()
    print("You entered:", user_input)

if __name__ == "__main__":
    main()
```
