## Exercise
Create a function that takes a string and a number as arguments and returns the string repeated the specified number of times

## Reference Solution
```python
def repeat_string(input_string: str, repeat_count: int) -> str:
    """
    Repeats a string the specified number of times.

    Args:
        input_string (str): The string to be repeated.
        repeat_count (int): The number of times to repeat the string.

    Returns:
        str: The repeated string.
    """
    if not isinstance(input_string, str):
        raise TypeError("Input must be a string.")
    if not isinstance(repeat_count, int):
        raise TypeError("Repeat count must be an integer.")
    if repeat_count < 0:
        raise ValueError("Repeat count must be a non-negative integer.")

    return input_string * repeat_count

# Example usage
def main():
    input_string = "Hello, World! "
    repeat_count = 3
    print(repeat_string(input_string, repeat_count))

if __name__ == "__main__":
    main()
```
