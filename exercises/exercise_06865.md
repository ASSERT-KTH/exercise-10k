## Exercise
Write a program that takes in three numbers and performs addition, subtraction, multiplication, and division operations on them. Then, it should display the results of each operation. Consider handles for division by zero errors.

## Reference Solution
```python
# Define a function to perform operations
def perform_operations(num1, num2, num3):
    # Perform addition
    add_result = num1 + num2 + num3
    print(f"Addition: {num1} + {num2} + {num3} = {add_result}")

    # Perform subtraction
    sub_result1 = num1 - num2 - num3
    sub_result2 = num2 - num1 - num3
    sub_result3 = num3 - num1 - num2
    print(f"Subtraction: {num1} - {num2} - {num3} = {sub_result1}")
    print(f"Subtraction: {num2} - {num1} - {num3} = {sub_result2}")
    print(f"Subtraction: {num3} - {num1} - {num2} = {sub_result3}")

    # Perform multiplication
    mul_result = num1 * num2 * num3
    print(f"Multiplication: {num1} * {num2} * {num3} = {mul_result}")

    # Perform division
    try:
        div_result1 = num1 / num2 / num3
        div_result2 = num2 / num1 / num3
        div_result3 = num3 / num1 / num2
        print(f"Division: {num1} / {num2} / {num3} = {div_result1}")
        print(f"Division: {num2} / {num1} / {num3} = {div_result2}")
        print(f"Division: {num3} / {num1} / {num2} = {div_result3}")
    except ZeroDivisionError:
        print("Error: Division by zero is not allowed.")

# Take input from the user
def main():
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    num3 = float(input("Enter the third number: "))

    perform_operations(num1, num2, num3)

if __name__ == "__main__":
    main()
```
