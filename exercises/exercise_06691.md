## Exercise
Create a program that takes two numbers as input and returns the percentage increase from the first to the second.

## Reference Solution
```python
def calculate_percentage_increase(initial_value, final_value):
    """
    Calculate the percentage increase from the initial value to the final value.

    Args:
        initial_value (float): The initial value.
        final_value (float): The final value.

    Returns:
        float: The percentage increase.
    """
    if initial_value == 0:
        raise ZeroDivisionError("Initial value cannot be zero")

    percentage_increase = ((final_value - initial_value) / initial_value) * 100
    return percentage_increase


def main():
    # Get user input
    initial_value = float(input("Enter the initial value: "))
    final_value = float(input("Enter the final value: "))

    try:
        # Calculate the percentage increase
        percentage_increase = calculate_percentage_increase(initial_value, final_value)
        print(f"The percentage increase is: {percentage_increase:.2f}%")
    except ZeroDivisionError as e:
        print(e)


if __name__ == "__main__":
    main()
```
