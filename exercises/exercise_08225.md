## Exercise
Write a program that recognizes a pattern in a given sequence using if/else statements

## Reference Solution
```python
def recognize_pattern(sequence):
    """
    Recognize a pattern in a given sequence of numbers.

    Args:
        sequence (list): A list of numbers.

    Returns:
        str: The type of pattern recognized.
    """
    if len(sequence) < 2:
        return "Insufficient data to recognize a pattern"

    # Check for arithmetic sequence
    diff = sequence[1] - sequence[0]
    is_arithmetic = all(sequence[i + 1] - sequence[i] == diff for i in range(1, len(sequence) - 1))

    if is_arithmetic:
        return "Arithmetic sequence"

    # Check for geometric sequence
    ratio = sequence[1] / sequence[0]
    is_geometric = all(sequence[i + 1] / sequence[i] == ratio for i in range(1, len(sequence) - 1))

    if is_geometric:
        return "Geometric sequence"

    # If no pattern is recognized, return a message
    return "No recognizable pattern found"

# Example usage
if __name__ == "__main__":
    sequences = [
        [1, 3, 5, 7, 9],
        [1, 2, 4, 8, 16],
        [1, 2, 3, 5, 7],
        [2, 4, 6, 8, 10]
    ]

    for sequence in sequences:
        print(f"Sequence: {sequence}")
        print(f"Pattern: {recognize_pattern(sequence)}")
        print()
```
