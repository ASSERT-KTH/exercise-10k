## Exercise
Develop a program to find the shortest word in a given string.

## Reference Solution
```python
def find_shortest_word(input_string):
    """
    Find the shortest word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The shortest word in the input string.
    """
    # Split the input string into words
    words = input_string.split()

    # Initialize the shortest word and its length
    shortest_word = words[0]
    shortest_length = len(shortest_word)

    # Iterate over each word in the list
    for word in words:
        # Remove punctuation from the word
        word = ''.join(e for e in word if e.isalnum())

        # Check if the current word is shorter than the shortest word
        if len(word) < shortest_length:
            shortest_word = word
            shortest_length = len(word)

    return shortest_word


def main():
    # Test the function
    input_string = "This is a test string with multiple words"
    print("Input String:", input_string)
    print("Shortest Word:", find_shortest_word(input_string))


if __name__ == "__main__":
    main()
```
