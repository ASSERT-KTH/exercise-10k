## Exercise
Write a program that takes a list and an index as input and modifies the element at the specified index.

## Reference Solution
```python
def modify_element(input_list, index, new_value):
    """
    Modifies the element at the specified index in the list.

    Args:
    input_list (list): The input list.
    index (int): The index of the element to modify.
    new_value (any): The new value for the element.

    Returns:
    list: The modified list.
    """
    # Check if the index is within the valid range
    if index < 0 or index >= len(input_list):
        raise IndexError("Index out of range")

    # Modify the element at the specified index
    input_list[index] = new_value

    return input_list


# Example usage
def main():
    input_list = [1, 2, 3, 4, 5]
    index = 2
    new_value = 10

    print("Original list:", input_list)
    modified_list = modify_element(input_list, index, new_value)
    print("Modified list:", modified_list)


if __name__ == "__main__":
    main()
```
