## Exercise
Create a program to calculate the grade of a student based on their score using if/else statements

## Reference Solution
```python
def calculate_grade(score):
    """
    Calculate the grade of a student based on their score.

    Args:
        score (float): The student's score.

    Returns:
        str: The student's grade.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the student's score from the user
    score = float(input("Enter the student's score (0-100): "))

    # Validate the score
    while score < 0 or score > 100:
        print("Invalid score. Please enter a score between 0 and 100.")
        score = float(input("Enter the student's score (0-100): "))

    # Calculate and display the student's grade
    grade = calculate_grade(score)
    print(f"The student's grade is: {grade}")

if __name__ == "__main__":
    main()
```
