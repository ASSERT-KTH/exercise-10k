## Exercise
Write a program that implements a graph algorithm to solve a given problem. The problem should be: find the shortest path between two nodes in a weighted graph.

## Reference Solution
```python
import sys
import heapq

class Graph:
    def __init__(self, num_nodes):
        self.num_nodes = num_nodes
        self.adj_list = [[] for _ in range(num_nodes)]

    def add_edge(self, node1, node2, weight):
        self.adj_list[node1].append((node2, weight))
        self.adj_list[node2].append((node1, weight))  # Comment this line for directed graph

    def dijkstra(self, start_node):
        distances = [sys.maxsize] * self.num_nodes
        distances[start_node] = 0
        pq = [(0, start_node)]

        while pq:
            current_distance, current_node = heapq.heappop(pq)

            if current_distance > distances[current_node]:
                continue

            for neighbor, weight in self.adj_list[current_node]:
                distance = current_distance + weight

                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    heapq.heappush(pq, (distance, neighbor))

        return distances

    def shortest_path(self, start_node, end_node):
        distances = self.dijkstra(start_node)
        return distances[end_node]

# Example usage
if __name__ == "__main__":
    num_nodes = 6
    edges = [
        (0, 1, 4),
        (0, 2, 2),
        (1, 3, 5),
        (2, 1, 1),
        (2, 3, 8),
        (2, 4, 10),
        (3, 4, 2),
        (3, 5, 6),
        (4, 5, 3)
    ]

    graph = Graph(num_nodes)
    for node1, node2, weight in edges:
        graph.add_edge(node1, node2, weight)

    start_node = 0
    end_node = 5
    shortest_distance = graph.shortest_path(start_node, end_node)
    print(f"Shortest distance from node {start_node} to node {end_node}: {shortest_distance}")
```
