## Exercise
Write a program that converts a given value from one data type to another

## Reference Solution
```python
def convert_data_type(value, current_type, new_type):
    """
    Converts a given value from one data type to another.

    Args:
        value (any): The value to be converted.
        current_type (str): The current data type of the value.
        new_type (str): The desired data type of the value.

    Returns:
        any: The converted value.
    """

    # Define a dictionary to map data types to their corresponding conversion functions
    conversion_functions = {
        'int': int,
        'float': float,
        'str': str,
        'bool': bool
    }

    # Check if the current and new data types are supported
    if current_type not in conversion_functions or new_type not in conversion_functions:
        raise ValueError("Unsupported data type")

    # Try to convert the value to the new data type
    try:
        # If the current type is 'bool' and the new type is not 'bool', we need to convert the bool to an integer first
        if current_type == 'bool' and new_type != 'bool':
            value = int(value)

        # Convert the value to the new data type
        converted_value = conversion_functions[new_type](value)

        return converted_value

    except ValueError:
        raise ValueError("Invalid value for conversion")

def main():
    # Example usage:
    value = "123"
    current_type = "str"
    new_type = "int"

    try:
        converted_value = convert_data_type(value, current_type, new_type)
        print(f"Converted {value} from {current_type} to {new_type}: {converted_value}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
