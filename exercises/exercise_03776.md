## Exercise
Group a dictionary by a common key.

## Reference Solution
```python
from collections import defaultdict

def group_by_key(data, key):
    """
    Group a dictionary by a common key.

    Args:
    - data (dict): The dictionary to group.
    - key (str): The key to group by.

    Returns:
    - dict: A dictionary with the grouped data.
    """
    grouped_data = defaultdict(list)
    for item in data:
        grouped_data[data[item][key]].append(item)
    return dict(grouped_data)

# Example usage
data = {
    "item1": {"category": "A", "price": 10.99},
    "item2": {"category": "B", "price": 5.99},
    "item3": {"category": "A", "price": 7.99},
    "item4": {"category": "B", "price": 3.99},
}

grouped_data = group_by_key(data, "category")
print(grouped_data)
```
