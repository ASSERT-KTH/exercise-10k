## Exercise
Create a program that allows users to add, remove, and mark tasks as completed using conditional statements.

## Reference Solution
```python
# Task Manager Program

# Initialize an empty dictionary to store tasks
tasks = {}

def display_tasks():
    """Display all tasks"""
    print("\nTask List:")
    for task, status in tasks.items():
        status_str = "Completed" if status else "Not Completed"
        print(f"{task}: {status_str}")

def add_task():
    """Add a new task"""
    task_name = input("Enter task name: ")
    if task_name not in tasks:
        tasks[task_name] = False
        print(f"Task '{task_name}' added successfully!")
    else:
        print(f"Task '{task_name}' already exists!")

def remove_task():
    """Remove a task"""
    task_name = input("Enter task name: ")
    if task_name in tasks:
        del tasks[task_name]
        print(f"Task '{task_name}' removed successfully!")
    else:
        print(f"Task '{task_name}' does not exist!")

def mark_task_completed():
    """Mark a task as completed"""
    task_name = input("Enter task name: ")
    if task_name in tasks:
        tasks[task_name] = True
        print(f"Task '{task_name}' marked as completed!")
    else:
        print(f"Task '{task_name}' does not exist!")

def main():
    while True:
        print("\nTask Manager Menu:")
        print("1. Display Tasks")
        print("2. Add Task")
        print("3. Remove Task")
        print("4. Mark Task as Completed")
        print("5. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            display_tasks()
        elif choice == "2":
            add_task()
        elif choice == "3":
            remove_task()
        elif choice == "4":
            mark_task_completed()
        elif choice == "5":
            print("Exiting the program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
