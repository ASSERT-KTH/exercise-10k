## Exercise
Read an array of numbers from the user and print the minimum of the array

## Reference Solution
```python
def read_array_from_user():
    """Reads an array of numbers from the user."""
    user_input = input("Enter an array of numbers (space-separated): ")
    try:
        array = [float(num) for num in user_input.split()]
        return array
    except ValueError:
        print("Invalid input. Please enter numbers only.")
        return read_array_from_user()

def find_minimum(array):
    """Finds the minimum value in the given array."""
    if not array:
        return None
    return min(array)

def main():
    array = read_array_from_user()
    minimum = find_minimum(array)
    if minimum is not None:
        print("Minimum value in the array:", minimum)

if __name__ == "__main__":
    main()
```
