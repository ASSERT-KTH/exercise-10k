## Exercise
Given a dictionary {'name': 'Natalie', 'age': 20, 'city': 'Sunshine Coast'}, write a program to use the popitem() method to remove the last key-value pair that was added to the dictionary.

## Reference Solution
```python
def remove_last_item(dictionary):
    """
    Removes the last key-value pair that was added to the dictionary.

    Args:
        dictionary (dict): The dictionary from which to remove the last item.

    Returns:
        dict: The dictionary with the last item removed.
    """
    # Check if the dictionary is empty
    if not dictionary:
        print("Dictionary is empty.")
        return dictionary

    # Use popitem() to remove the last key-value pair
    removed_item = dictionary.popitem()

    print(f"Removed item: {removed_item}")
    return dictionary


def main():
    # Create a dictionary
    person = {'name': 'Natalie', 'age': 20, 'city': 'Sunshine Coast'}

    print("Original dictionary:")
    print(person)

    # Remove the last key-value pair
    updated_person = remove_last_item(person)

    print("\nUpdated dictionary:")
    print(updated_person)


if __name__ == "__main__":
    main()
```
