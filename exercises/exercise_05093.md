## Exercise
Write a program that asks the user a series of survey questions and prints out the results.

## Reference Solution
```python
def survey():
    print("Welcome to our survey!")
    
    # Initialize an empty dictionary to store the answers
    answers = {}
    
    # Ask the user the survey questions
    answers['name'] = input("What is your name? ")
    answers['age'] = input("How old are you? ")
    answers['favorite_color'] = input("What is your favorite color? ")
    answers['favorite_food'] = input("What is your favorite food? ")
    answers['favorite_activity'] = input("What is your favorite activity? ")
    
    # Print out the results
    print("\nSurvey Results:")
    for question, answer in answers.items():
        print(f"{question.capitalize().replace('_', ' ')}: {answer}")

def main():
    survey()
    again = input("\nWould you like to take the survey again? (yes/no): ")
    while again.lower() == 'yes':
        survey()
        again = input("\nWould you like to take the survey again? (yes/no): ")

if __name__ == "__main__":
    main()
```
