## Exercise
Write a program that generates a random password that meets the following requirements: at least 8 characters long, contains at least one uppercase letter, one lowercase letter, and one digit.

## Reference Solution
```python
import random
import string

def generate_password(length=8):
    """
    Generates a random password of the specified length.
    The password will contain at least one uppercase letter, one lowercase letter, and one digit.
    """
    # Define character sets
    uppercase = string.ascii_uppercase
    lowercase = string.ascii_lowercase
    digits = string.digits

    # Ensure password has at least one character from each set
    password = [random.choice(uppercase), random.choice(lowercase), random.choice(digits)]

    # Fill the rest of the password with random characters
    all_characters = uppercase + lowercase + digits
    for _ in range(length - 3):
        password.append(random.choice(all_characters))

    # Shuffle the password to avoid the first three characters always being in the same character set
    random.shuffle(password)

    # Join the password list into a single string
    return ''.join(password)

def main():
    # Generate and print a password of the default length (8)
    print("Generated Password:", generate_password())

    # Generate and print a password of a custom length
    custom_length = 12
    print(f"Generated Password ({custom_length} characters):", generate_password(custom_length))

if __name__ == "__main__":
    main()
```
