## Exercise
Write a program to calculate the correlation coefficient of two sets of numbers.

import numpy as np

def calculate_correlation_coefficient(x, y):
    """
    Calculate the correlation coefficient of two sets of numbers.

    Args:
        x (list): The first set of numbers.
        y (list): The second set of numbers.

    Returns:
        float: The correlation coefficient.
    """
    # Calculate the means of x and y
    mean_x = np.mean(x)
    mean_y = np.mean(y)

    # Calculate the deviations from the means
    deviations_x = [xi - mean_x for xi in x]
    deviations_y = [yi - mean_y for yi in y]

    # Calculate the covariance and variances
    covariance = sum(xi * yi for xi, yi in zip(deviations_x, deviations_y)) / (len(x) - 1)
    variance_x = sum(xi ** 2 for xi in deviations_x) / (len(x) - 1)
    variance_y = sum(yi ** 2 for yi in deviations_y) / (len(y) - 1)

    # Calculate the correlation coefficient
    correlation_coefficient = covariance / np.sqrt(variance_x * variance_y)

    return correlation_coefficient

# Example usage
x = [1, 2, 3, 4, 5]
y = [2, 3, 5, 7, 11]

correlation_coefficient = calculate_correlation_coefficient(x, y)
print("Correlation Coefficient:", correlation_coefficient)
