## Exercise
Write a function that takes two sorted lists as input and returns a single sorted list containing all elements from both lists.

## Reference Solution
```python
def merge_sorted_lists(list1, list2):
    new_list = []
    list1_index = 0
    list2_index = 0

    while list1_index < len(list1) and list2_index < len(list2):
        if list1[list1_index] < list2[list2_index]:
            new_list.append(list1[list1_index])
            list1_index += 1
        else:
            new_list.append(list2[list2_index])
            list2_index += 1

    while list1_index < len(list1):
        new_list.append(list1[list1_index])
        list1_index += 1

    while list2_index < len(list2):
        new_list.append(list2[list2_index])
        list2_index += 1

    return new_list

# Example usage:
list1 = [1, 3, 5, 7, 9]
list2 = [2, 4, 6, 8, 10]
result = merge_sorted_lists(list1, list2)
print(result)
```
