## Exercise
Write a program that finds the key of a given value in a dictionary

## Reference Solution
```python
def find_key(dictionary, value):
    """
    Find the key of a given value in a dictionary.

    Args:
        dictionary (dict): The dictionary to search.
        value: The value to find.

    Returns:
        The key associated with the value, or None if not found.
    """
    for key, val in dictionary.items():
        if val == value:
            return key
    return None

# Example usage:
def main():
    dictionary = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }

    value = "John"
    key = find_key(dictionary, value)

    if key is not None:
        print(f"The key for '{value}' is: {key}")
    else:
        print(f"No key found for '{value}'")

if __name__ == "__main__":
    main()
```
