## Exercise
Create a program that asks the user for the current time and then prints out the schedule for the next bus based on the following rules: If the current time is between 7:00 and 9:00, the next bus is in 15 minutes; if the current time is between 9:00 and 16:00, the next bus is in 30 minutes; if the current time is between 16:00 and 18:00, the next bus is in 15 minutes; if the current time is after 18:00, there are no more buses

## Reference Solution
```python
from datetime import datetime

def get_next_bus_time(current_time):
    """
    Calculate the next bus time based on the current time.
    
    Args:
    current_time (datetime): The current time.
    
    Returns:
    str: The next bus time or a message indicating that there are no more buses.
    """
    hour = current_time.hour
    minute = current_time.minute
    
    if 7 <= hour < 9 or 16 <= hour < 18:
        next_bus_minute = minute + 15
        next_bus_hour = hour + (next_bus_minute // 60)
        next_bus_minute %= 60
        return f"{next_bus_hour:02d}:{next_bus_minute:02d}"
    elif 9 <= hour < 16:
        next_bus_minute = minute + 30
        next_bus_hour = hour + (next_bus_minute // 60)
        next_bus_minute %= 60
        return f"{next_bus_hour:02d}:{next_bus_minute:02d}"
    else:
        return "There are no more buses today."

def main():
    # Get the current time from the user
    while True:
        user_input = input("Please enter the current time (HH:MM): ")
        try:
            current_time = datetime.strptime(user_input, "%H:%M")
            break
        except ValueError:
            print("Invalid time format. Please use HH:MM.")
    
    # Calculate and print the next bus time
    next_bus_time = get_next_bus_time(current_time)
    print(f"The next bus is at {next_bus_time}.")

if __name__ == "__main__":
    main()
```
