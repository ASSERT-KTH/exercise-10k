## Exercise
Given a list of integers and a pivot element, partition the array such that all elements less than the pivot come before all elements greater than the pivot

## Reference Solution

def partition_array(nums, pivot):
    """
    Partition the array such that all elements less than the pivot come before all elements greater than the pivot.

    Args:
    - nums (list): A list of integers.
    - pivot (int): The pivot element.

    Returns:
    - list: The partitioned array.
    """
    # Initialize two pointers
    left = 0
    right = 0

    # Iterate over the array
    while right < len(nums):
        # If the current element is less than the pivot, swap it with the left pointer
        if nums[right] < pivot:
            nums[left], nums[right] = nums[right], nums[left]
            left += 1
        right += 1

    return nums

# Example usage
nums = [7, 9, 4, 2, 5, 8, 1, 6, 3]
pivot = 5
print("Original array:", nums)
print("Pivot:", pivot)
print("Partitioned array:", partition_array(nums, pivot))
