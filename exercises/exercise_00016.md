## Exercise
Write a program to use regular expressions to match and extract patterns in strings.

## Reference Solution
```python
import re

def extract_pattern(input_string, pattern):
    """
    Extracts the first occurrence of the given pattern in the input string.

    Args:
        input_string (str): The string to search for the pattern.
        pattern (str): The regular expression pattern to match.

    Returns:
        str: The matched pattern, or None if no match is found.
    """
    match = re.search(pattern, input_string)
    if match:
        return match.group()
    else:
        return None

def extract_all_patterns(input_string, pattern):
    """
    Extracts all occurrences of the given pattern in the input string.

    Args:
        input_string (str): The string to search for the pattern.
        pattern (str): The regular expression pattern to match.

    Returns:
        list: A list of all matched patterns.
    """
    return re.findall(pattern, input_string)

def main():
    # Example usage:
    input_string = "Hello, my email is john.doe@example.com and my phone number is 123-456-7890."
    email_pattern = r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b"
    phone_pattern = r"\d{3}-\d{3}-\d{4}"

    email_match = extract_pattern(input_string, email_pattern)
    phone_match = extract_pattern(input_string, phone_pattern)

    print(f"Email match: {email_match}")
    print(f"Phone match: {phone_match}")

    # Extract all occurrences of the pattern
    all_emails = extract_all_patterns(input_string, email_pattern)
    all_phones = extract_all_patterns(input_string, phone_pattern)

    print(f"All emails: {all_emails}")
    print(f"All phones: {all_phones}")

if __name__ == "__main__":
    main()
```
