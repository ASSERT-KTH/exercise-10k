## Exercise
Write a function that sorts a dictionary by keys or values

## Reference Solution
```python
def sort_dict(input_dict, sort_by='keys', reverse=False):
    """
    Sorts a dictionary by its keys or values.

    Args:
    - input_dict (dict): The dictionary to be sorted.
    - sort_by (str): 'keys' or 'values'. Default is 'keys'.
    - reverse (bool): Whether to sort in descending order. Default is False.

    Returns:
    - dict: The sorted dictionary.
    """
    if sort_by == 'keys':
        return dict(sorted(input_dict.items()))
    elif sort_by == 'values':
        return dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))
    else:
        raise ValueError("Invalid sort_by parameter. It should be 'keys' or 'values'.")

# Example usage:
if __name__ == "__main__":
    my_dict = {'c': 3, 'b': 2, 'a': 1, 'd': 4}
    
    # Sort by keys
    sorted_by_keys = sort_dict(my_dict)
    print("Sorted by keys:", sorted_by_keys)
    
    # Sort by values in ascending order
    sorted_by_values_asc = sort_dict(my_dict, sort_by='values')
    print("Sorted by values (asc):", sorted_by_values_asc)
    
    # Sort by values in descending order
    sorted_by_values_desc = sort_dict(my_dict, sort_by='values', reverse=True)
    print("Sorted by values (desc):", sorted_by_values_desc)
```
