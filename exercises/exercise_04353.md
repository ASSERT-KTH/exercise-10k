## Exercise
Write a program that calculates the discount on a given price based on a simple discount system (e.g., 10% off for students, 5% off for seniors)

## Reference Solution
```python
class DiscountCalculator:
    def __init__(self):
        # Define the discount percentages
        self.discounts = {
            "student": 0.10,
            "senior": 0.05,
        }

    def calculate_discount(self, price, discount_type):
        """
        Calculate the discount based on the given price and discount type.

        Args:
        - price (float): The original price.
        - discount_type (str): The type of discount (e.g., "student", "senior").

        Returns:
        - discount (float): The calculated discount amount.
        """
        if discount_type in self.discounts:
            discount_percentage = self.discounts[discount_type]
            discount = price * discount_percentage
            return discount
        else:
            raise ValueError(f"Invalid discount type: {discount_type}")

    def calculate_price_after_discount(self, price, discount_type):
        """
        Calculate the price after applying the discount.

        Args:
        - price (float): The original price.
        - discount_type (str): The type of discount (e.g., "student", "senior").

        Returns:
        - price_after_discount (float): The price after applying the discount.
        """
        discount = self.calculate_discount(price, discount_type)
        price_after_discount = price - discount
        return price_after_discount


def main():
    calculator = DiscountCalculator()

    # Test with a student discount
    price = 100.0
    discount_type = "student"
    discount = calculator.calculate_discount(price, discount_type)
    price_after_discount = calculator.calculate_price_after_discount(price, discount_type)

    print(f"Original Price: ${price:.2f}")
    print(f"Discount Type: {discount_type.capitalize()}")
    print(f"Discount: ${discount:.2f}")
    print(f"Price after Discount: ${price_after_discount:.2f}")

    # Test with a senior discount
    price = 200.0
    discount_type = "senior"
    discount = calculator.calculate_discount(price, discount_type)
    price_after_discount = calculator.calculate_price_after_discount(price, discount_type)

    print(f"\nOriginal Price: ${price:.2f}")
    print(f"Discount Type: {discount_type.capitalize()}")
    print(f"Discount: ${discount:.2f}")
    print(f"Price after Discount: ${price_after_discount:.2f}")


if __name__ == "__main__":
    main()
```
