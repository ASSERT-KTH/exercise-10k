## Exercise
Write a function that takes another function as input and adds a logging feature to it

## Reference Solution
```python
import logging
import functools

# Set up logging configuration
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def add_logging(func):
    """
    A decorator function that adds logging to the input function.
    
    Args:
    func: The function to be decorated with logging.

    Returns:
    A wrapper function that logs the input and output of the original function.
    """
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # Log the input to the function
        logging.info(f"Calling {func.__name__} with arguments: args={args}, kwargs={kwargs}")
        
        # Call the original function
        result = func(*args, **kwargs)
        
        # Log the output of the function
        logging.info(f"{func.__name__} returned: {result}")
        
        return result
    
    return wrapper

# Example usage
@add_logging
def example_function(a, b):
    """
    An example function that adds two numbers.
    
    Args:
    a (int): The first number.
    b (int): The second number.

    Returns:
    int: The sum of a and b.
    """
    return a + b

# Test the decorated function
if __name__ == "__main__":
    result = example_function(2, 3)
    print("Result:", result)
```
