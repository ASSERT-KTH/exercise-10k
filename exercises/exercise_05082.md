## Exercise
Create a program that takes two numbers and an operator as input and performs the corresponding mathematical operation.

## Reference Solution
```python
def math_operation(num1, num2, operator):
    """
    Perform a mathematical operation based on the operator.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The mathematical operator. Can be '+', '-', '*', '/'.

    Returns:
        float: The result of the mathematical operation.

    Raises:
        ValueError: If the operator is not one of '+', '-', '*', '/'.
        ZeroDivisionError: If the operator is '/' and num2 is 0.
    """

    # Check if the operator is valid
    if operator not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operator. Must be one of '+', '-', '*', '/'.")

    # Perform the mathematical operation
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


def main():
    # Get the numbers and operator from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operator = input("Enter the operator (+, -, *, /): ")

    try:
        # Perform the mathematical operation
        result = math_operation(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except ValueError as e:
        print(e)
    except ZeroDivisionError as e:
        print(e)


if __name__ == "__main__":
    main()
```
