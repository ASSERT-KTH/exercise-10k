## Exercise
Create a function that takes in a password and checks if it meets certain requirements (e.g. length, contains uppercase and lowercase letters, contains numbers).

## Reference Solution
```python
import re

def validate_password(password: str, min_length: int = 8) -> bool:
    """
    Validate a password based on certain requirements.

    Args:
    - password (str): The password to be validated.
    - min_length (int): The minimum required length of the password. Defaults to 8.

    Returns:
    - bool: True if the password is valid, False otherwise.
    """
    # Check if password meets the minimum length requirement
    if len(password) < min_length:
        return False
    
    # Check if password contains uppercase and lowercase letters
    if not re.search(r"[A-Z]", password) or not re.search(r"[a-z]", password):
        return False
    
    # Check if password contains numbers
    if not re.search(r"\d", password):
        return False
    
    return True

# Example usage:
if __name__ == "__main__":
    passwords = [
        "short",
        "onlylowercase",
        "ONLYUPPERCASE",
        "no_numbers",
        "ValidP@ssw0rd"
    ]

    for password in passwords:
        if validate_password(password):
            print(f"{password} is a valid password.")
        else:
            print(f"{password} is not a valid password.")
```
