## Exercise
Write a program that generates a random story based on user input.

## Reference Solution
```python
import random

def get_user_input():
    """Get user input"""
    story_type = input("Enter the type of story (e.g. adventure, romance, science fiction): ")
    main_character_name = input("Enter the main character's name: ")
    main_character_profession = input("Enter the main character's profession: ")
    random_object = input("Enter a random object: ")
    random_location = input("Enter a random location: ")

    return story_type, main_character_name, main_character_profession, random_object, random_location

def generate_story(story_type, main_character_name, main_character_profession, random_object, random_location):
    """Generate a random story"""
    story = ""

    if story_type == "adventure":
        story += f"{main_character_name}, a {main_character_profession}, was on a quest to find the legendary {random_object} in the ancient city of {random_location}. "
        story += "As they navigated through the ruins, they encountered fierce creatures and overcame treacherous obstacles. "
        story += f"Finally, they found the {random_object} and returned home as heroes, with the {random_object} in hand."
    elif story_type == "romance":
        story += f"{main_character_name}, a {main_character_profession}, was searching for love in the bustling city of {random_location}. "
        story += f"One day, they stumbled upon a mysterious {random_object} that led them to their soulmate. "
        story += "As they fell deeply in love, they realized that the {random_object} was the key to their happiness."
    elif story_type == "science fiction":
        story += f"{main_character_name}, a {main_character_profession}, was on a mission to explore the distant planet of {random_location}. "
        story += f"As they navigated through the alien landscape, they discovered a strange {random_object} that held the secrets of the universe. "
        story += "As they unraveled the mysteries of the {random_object}, they realized that it was the key to saving humanity from extinction."

    return story

def main():
    story_type, main_character_name, main_character_profession, random_object, random_location = get_user_input()
    story = generate_story(story_type, main_character_name, main_character_profession, random_object, random_location)

    print("Your random story:")
    print(story)

if __name__ == "__main__":
    main()
```
