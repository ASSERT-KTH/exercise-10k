## Exercise
Create a function that implements a tree traversal algorithm to traverse a binary tree. The function should take a tree as an argument and return a list of the node values in the order they were visited.

## Reference Solution
```python
class Node:
    """Represents a node in the binary tree."""
    
    def __init__(self, value):
        # Initialize the node with a value and two children (left and right)
        self.value = value
        self.left = None
        self.right = None

def inorder_traversal(tree):
    """
    Performs an in-order traversal of a binary tree and returns a list of node values in the order they were visited.

    Args:
        tree (Node): The root node of the binary tree.

    Returns:
        list: A list of node values in the order they were visited.
    """
    
    # Initialize an empty list to store the node values
    result = []
    
    # Define a helper function to perform the recursive traversal
    def traverse(node):
        # If the current node is None, return immediately
        if node is None:
            return
        
        # Recursively traverse the left subtree
        traverse(node.left)
        
        # Append the value of the current node to the result list
        result.append(node.value)
        
        # Recursively traverse the right subtree
        traverse(node.right)
    
    # Start the traversal from the root node
    traverse(tree)
    
    # Return the list of node values
    return result

# Example usage:
# Create a sample binary tree
root = Node(1)
root.left = Node(2)
root.right = Node(3)
root.left.left = Node(4)
root.left.right = Node(5)

# Perform the in-order traversal and print the result
print(inorder_traversal(root))  # Output: [4, 2, 5, 1, 3]
```
