## Exercise
Write a program that simulates a queue of people waiting in line.

## Reference Solution
```python
class Person:
    """Represents a person in the queue."""
    def __init__(self, name):
        self.name = name

class Queue:
    """Represents a queue of people."""
    def __init__(self):
        self.queue = []

    def add_person(self, person):
        """Adds a person to the end of the queue."""
        self.queue.append(person)
        print(f"{person.name} has been added to the queue.")

    def remove_person(self):
        """Removes the person at the front of the queue."""
        if self.is_empty():
            print("The queue is empty.")
        else:
            person = self.queue.pop(0)
            print(f"{person.name} has been removed from the queue.")

    def display_queue(self):
        """Displays the current state of the queue."""
        if self.is_empty():
            print("The queue is empty.")
        else:
            print("The current queue is:")
            for i, person in enumerate(self.queue):
                print(f"{i+1}. {person.name}")

    def is_empty(self):
        """Returns True if the queue is empty, False otherwise."""
        return len(self.queue) == 0

def main():
    queue = Queue()

    while True:
        print("\nQueue Simulator Menu:")
        print("1. Add person to queue")
        print("2. Remove person from queue")
        print("3. Display queue")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter the person's name: ")
            person = Person(name)
            queue.add_person(person)
        elif choice == "2":
            queue.remove_person()
        elif choice == "3":
            queue.display_queue()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
