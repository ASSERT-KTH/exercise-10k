## Exercise
Write a program that simulates a to-do list. The program should take in the task and output the updated task list based on the following conditions: if the user adds a task, add the task to the list, if the user removes a task, remove the task from the list, if the user checks the task list, output the current task list.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        """Add a task to the to-do list."""
        self.tasks.append(task)
        print(f"Task '{task}' added to the list.")

    def remove_task(self, task):
        """Remove a task from the to-do list."""
        if task in self.tasks:
            self.tasks.remove(task)
            print(f"Task '{task}' removed from the list.")
        else:
            print(f"Task '{task}' not found in the list.")

    def check_list(self):
        """Print the current to-do list."""
        if not self.tasks:
            print("The to-do list is empty.")
        else:
            print("Current to-do list:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")


def main():
    todo = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add task")
        print("2. Remove task")
        print("3. Check list")
        print("4. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            task = input("Enter a task: ")
            todo.add_task(task)
        elif choice == "2":
            task = input("Enter a task to remove: ")
            todo.remove_task(task)
        elif choice == "3":
            todo.check_list()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
