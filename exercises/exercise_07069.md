Write a program that takes an encrypted string as input and decrypts it using a simple decryption algorithm.

def decrypt_string(encrypted_string, shift):
    """
    Decrypts an encrypted string using Caesar Cipher decryption algorithm.

    Args:
        encrypted_string (str): The encrypted string to be decrypted.
        shift (int): The number of places each letter in the encrypted string should be shifted back.

    Returns:
        str: The decrypted string.
    """

    # Initialize an empty string to store the decrypted string
    decrypted_string = ""

    # Iterate over each character in the encrypted string
    for char in encrypted_string:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Calculate the ASCII value of the decrypted character
            # The ASCII value of 'A' is 65, so we subtract 65 to get the position of the letter in the alphabet
            # We then subtract the shift value and use the modulo operator to ensure the result is within the range of 0 to 25
            # Finally, we add 65 to get the ASCII value of the decrypted character
            decrypted_char = chr((ord(char) - 65 - shift) % 26 + 65)
            # Append the decrypted character to the decrypted string
            decrypted_string += decrypted_char
        # Check if the character is a lowercase letter
        elif char.islower():
            # Calculate the ASCII value of the decrypted character
            # The ASCII value of 'a' is 97, so we subtract 97 to get the position of the letter in the alphabet
            # We then subtract the shift value and use the modulo operator to ensure the result is within the range of 0 to 25
            # Finally, we add 97 to get the ASCII value of the decrypted character
            decrypted_char = chr((ord(char) - 97 - shift) % 26 + 97)
            # Append the decrypted character to the decrypted string
            decrypted_string += decrypted_char
        # If the character is not a letter, append it to the decrypted string as is
        else:
            decrypted_string += char

    # Return the decrypted string
    return decrypted_string


# Example usage:
encrypted_string = "Khoor Zruog"
shift = 3
decrypted_string = decrypt_string(encrypted_string, shift)
print(f"Encrypted String: {encrypted_string}")
print(f"Shift: {shift}")
print(f"Decrypted String: {decrypted_string}")
