## Exercise
Write a function that finds the minimum element in a list of lists.

## Reference Solution
```python
def find_min_element(list_of_lists):
    """
    Finds the minimum element in a list of lists.

    Args:
    list_of_lists (list): A list containing lists of integers.

    Returns:
    int: The minimum element in the list of lists.

    Raises:
    ValueError: If the input list is empty or any of the sublists are empty.
    """

    # Check if the input list is empty
    if not list_of_lists:
        raise ValueError("Input list is empty")

    # Flatten the list of lists
    flat_list = [item for sublist in list_of_lists for item in sublist]

    # Check if any of the sublists are empty
    if not flat_list:
        raise ValueError("At least one sublist is empty")

    # Return the minimum element
    return min(flat_list)


# Example usage:
if __name__ == "__main__":
    list_of_lists = [[5, 2, 9], [1, 7, 3], [6, 8, 4]]
    try:
        min_element = find_min_element(list_of_lists)
        print(f"Minimum element: {min_element}")
    except ValueError as e:
        print(f"Error: {e}")
```
