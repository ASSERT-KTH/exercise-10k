## Exercise
Write a program that generates a list of valid variable names based on a given set of rules (e.g. starts with a letter, only contains letters and digits).

## Reference Solution
```python
import itertools
import string

def generate_variable_names(length, prefix=None):
    """
    Generate a list of valid variable names.

    Args:
        length (int): The length of the variable names.
        prefix (str): An optional prefix for the variable names.

    Returns:
        list: A list of valid variable names.
    """

    # Create a string of all lowercase letters and digits
    all_chars = string.ascii_letters + string.digits

    # Initialize an empty list to store the variable names
    variable_names = []

    # If a prefix is provided, use it as the starting point
    if prefix:
        # Check if the prefix is a valid variable name
        if not prefix[0].isalpha():
            raise ValueError("Prefix must start with a letter")

        # Generate all possible variable names with the given prefix
        for chars in itertools.product(all_chars, repeat=length - len(prefix)):
            variable_name = prefix + ''.join(chars)
            variable_names.append(variable_name)

    else:
        # Generate all possible variable names
        for first_char in string.ascii_letters:
            for chars in itertools.product(all_chars, repeat=length - 1):
                variable_name = first_char + ''.join(chars)
                variable_names.append(variable_name)

    return variable_names

# Test the function
length = 3
variable_names = generate_variable_names(length)
print(f"Generated {len(variable_names)} variable names of length {length}:")
print(variable_names[:10])  # Print the first 10 variable names

prefix = "abc"
variable_names_with_prefix = generate_variable_names(length, prefix)
print(f"\nGenerated {len(variable_names_with_prefix)} variable names of length {length} with prefix '{prefix}':")
print(variable_names_with_prefix[:10])  # Print the first 10 variable names with prefix
```
