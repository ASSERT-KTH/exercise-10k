## Exercise
Write a program to visualize data using a data visualization library.

## Reference Solution
```python
import matplotlib.pyplot as plt
import numpy as np

# Line Graph
def plot_line_graph():
    x = np.linspace(0, 10, 100)
    y = np.sin(x)
    plt.plot(x, y)
    plt.title('Line Graph')
    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.show()

# Bar Chart
def plot_bar_chart():
    x = np.arange(5)
    y = np.array([10, 15, 7, 12, 20])
    plt.bar(x, y)
    plt.title('Bar Chart')
    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.show()

# Scatter Plot
def plot_scatter_plot():
    x = np.random.rand(50)
    y = np.random.rand(50)
    plt.scatter(x, y)
    plt.title('Scatter Plot')
    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.show()

# Main Function
def main():
    print("1. Line Graph")
    print("2. Bar Chart")
    print("3. Scatter Plot")
    choice = int(input("Enter your choice: "))
    
    if choice == 1:
        plot_line_graph()
    elif choice == 2:
        plot_bar_chart()
    elif choice == 3:
        plot_scatter_plot()
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
