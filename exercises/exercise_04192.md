## Exercise
Write a function that generates a leaderboard for a game, ranking players by their score.

## Reference Solution
```python
def generate_leaderboard(player_scores):
    """
    Generates a leaderboard for a game, ranking players by their score.

    Args:
    player_scores (dict): A dictionary where keys are player names and values are their scores.

    Returns:
    list: A list of tuples containing the player name and score, sorted in descending order by score.
    """
    # Sort the player scores in descending order
    sorted_scores = sorted(player_scores.items(), key=lambda x: x[1], reverse=True)

    # Return the sorted list of player scores
    return sorted_scores


def print_leaderboard(leaderboard):
    """
    Prints the leaderboard in a formatted manner.

    Args:
    leaderboard (list): A list of tuples containing the player name and score.
    """
    # Print the leaderboard header
    print("Leaderboard:")
    print("Rank | Player Name | Score")

    # Print each player's rank, name, and score
    for i, (player, score) in enumerate(leaderboard, start=1):
        print(f"{i:4} | {player:12} | {score:5}")


# Example usage
if __name__ == "__main__":
    player_scores = {
        "John": 100,
        "Alice": 120,
        "Bob": 90,
        "Charlie": 110
    }

    leaderboard = generate_leaderboard(player_scores)
    print_leaderboard(leaderboard)
```
