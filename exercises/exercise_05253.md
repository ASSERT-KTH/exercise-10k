## Exercise
Write a program to round a given number to the nearest integer or to a specified number of decimal places.

## Reference Solution
```python
def round_to_nearest_integer(num):
    """
    Rounds a given number to the nearest integer.

    Args:
        num (float): The number to be rounded.

    Returns:
        int: The rounded number.
    """
    return round(num)


def round_to_decimal_places(num, decimal_places):
    """
    Rounds a given number to a specified number of decimal places.

    Args:
        num (float): The number to be rounded.
        decimal_places (int): The number of decimal places to round to.

    Returns:
        float: The rounded number.
    """
    return round(num, decimal_places)


def main():
    # Example usage:
    num = 12.3456

    # Round to nearest integer
    rounded_to_nearest_integer = round_to_nearest_integer(num)
    print(f"Rounded to nearest integer: {rounded_to_nearest_integer}")

    # Round to 2 decimal places
    decimal_places = 2
    rounded_to_decimal_places = round_to_decimal_places(num, decimal_places)
    print(f"Rounded to {decimal_places} decimal places: {rounded_to_decimal_places}")


if __name__ == "__main__":
    main()
```
