## Exercise
Write a program that asks the user for a recipe and then prints out the ingredients and instructions.

## Reference Solution
```python
class Recipe:
    def __init__(self, name):
        self.name = name
        self.ingredients = []
        self.instructions = []

    def add_ingredient(self, ingredient, quantity):
        self.ingredients.append((ingredient, quantity))

    def add_instruction(self, instruction):
        self.instructions.append(instruction)

    def print_recipe(self):
        print(f"**{self.name}**")
        print("\n**Ingredients:**")
        for i, (ingredient, quantity) in enumerate(self.ingredients, start=1):
            print(f"{i}. {ingredient} - {quantity}")
        print("\n**Instructions:**")
        for i, instruction in enumerate(self.instructions, start=1):
            print(f"{i}. {instruction}")


def main():
    print("Welcome to the Recipe Book!")
    recipe_name = input("Please enter the name of your recipe: ")

    recipe = Recipe(recipe_name)

    num_ingredients = int(input("How many ingredients does your recipe have? "))

    for i in range(num_ingredients):
        ingredient = input(f"Enter ingredient {i + 1}: ")
        quantity = input(f"Enter quantity for {ingredient}: ")
        recipe.add_ingredient(ingredient, quantity)

    num_instructions = int(input("How many instructions does your recipe have? "))

    for i in range(num_instructions):
        instruction = input(f"Enter instruction {i + 1}: ")
        recipe.add_instruction(instruction)

    recipe.print_recipe()


if __name__ == "__main__":
    main()
```
