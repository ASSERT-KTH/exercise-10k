## Exercise
Generate the power set of a given list of integers

## Reference Solution
```python
def power_set(nums):
    """
    Generate the power set of a given list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        list: A list of lists, where each sublist is a subset of the input list.
    """
    # Base case: if the list is empty, return a list containing an empty list
    if not nums:
        return [[]]

    # Recursive case: get the power set of the rest of the list
    rest_power_set = power_set(nums[1:])

    # Add the current number to each subset in the power set of the rest of the list
    new_subsets = [[nums[0]] + subset for subset in rest_power_set]

    # Return the power set of the rest of the list plus the new subsets
    return rest_power_set + new_subsets


def main():
    # Example usage
    nums = [1, 2, 3]
    power_set_result = power_set(nums)
    print("Power Set:")
    for i, subset in enumerate(power_set_result):
        print(f"Subset {i+1}: {subset}")


if __name__ == "__main__":
    main()
```
