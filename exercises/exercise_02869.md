## Exercise
Write a function to split a given string into a list of words

## Reference Solution
```python
def split_string_into_words(input_string):
    """
    Splits a given string into a list of words.

    Args:
        input_string (str): The string to be split into words.

    Returns:
        list: A list of words from the input string.
    """
    # Remove leading and trailing whitespace
    input_string = input_string.strip()
    
    # Split the string into a list of words
    words = input_string.split()
    
    return words

# Example usage:
input_str = "Hello World, this is a test string."
print(split_string_into_words(input_str))

import re

def split_string_into_words(input_string):
    """
    Splits a given string into a list of words, removing punctuation.

    Args:
        input_string (str): The string to be split into words.

    Returns:
        list: A list of words from the input string.
    """
    # Remove leading and trailing whitespace
    input_string = input_string.strip()
    
    # Remove punctuation
    input_string = re.sub(r'[^\w\s]', '', input_string)
    
    # Split the string into a list of words
    words = input_string.split()
    
    return words

# Example usage:
input_str = "Hello World, this is a test string."
print(split_string_into_words(input_str))
```
