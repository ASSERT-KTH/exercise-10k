## Exercise
Create a function that uses the multiprocessing module to run tasks in parallel.

## Reference Solution
```python
import multiprocessing
import time
from functools import partial

def task(name, delay):
    """
    Simulate a task with a given delay.

    Args:
        name (str): Task name.
        delay (int): Delay in seconds.
    """
    print(f"Task '{name}' started")
    time.sleep(delay)
    print(f"Task '{name}' finished")

def run_tasks_in_parallel(tasks):
    """
    Run tasks in parallel using multiprocessing.

    Args:
        tasks (list): List of tasks. Each task is a tuple of (name, delay).
    """
    # Create a pool of worker processes
    with multiprocessing.Pool() as pool:
        # Use partial to fix the function and pass a list of arguments
        partial_task = partial(task)
        # Map the tasks to the pool
        pool.starmap(partial_task, tasks)

if __name__ == "__main__":
    # Define tasks
    tasks = [
        ("Task 1", 2),
        ("Task 2", 3),
        ("Task 3", 1),
        ("Task 4", 4),
    ]

    # Run tasks in parallel
    start_time = time.time()
    run_tasks_in_parallel(tasks)
    end_time = time.time()

    print(f"Total execution time: {end_time - start_time} seconds")
```
