## Exercise
Write a program that scrambles a word, then asks the user to unscramble it.

## Reference Solution
```python
import random

# List of words to choose from
words = ["apple", "banana", "cherry", "date", "elderberry"]

def scramble_word(word):
    """Scramble a word by shuffling its letters."""
    word_list = list(word)
    random.shuffle(word_list)
    return "".join(word_list)

def play_game():
    """Play the word scramble game."""
    # Choose a random word from the list
    word = random.choice(words)
    scrambled_word = scramble_word(word)

    print("Unscramble the letters to form a valid English word:")
    print(scrambled_word)

    # Ask the user for their answer
    answer = input("Enter your answer: ")

    # Check if the answer is correct
    if answer.lower() == word:
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {word}.")

def main():
    """Main function to start the game."""
    print("Welcome to the Word Scramble Game!")
    play_again = "y"
    while play_again.lower() == "y":
        play_game()
        play_again = input("Would you like to play again? (y/n): ")

if __name__ == "__main__":
    main()
```
