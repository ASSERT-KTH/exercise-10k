## Exercise
Write a function that checks if a given list is sorted in ascending order.

## Reference Solution

def is_sorted(lst):
    """
    Checks if a given list is sorted in ascending order.

    Args:
        lst (list): The list to be checked.

    Returns:
        bool: True if the list is sorted in ascending order, False otherwise.
    """

    # If the list is empty or contains only one element, it's considered sorted
    if len(lst) <= 1:
        return True

    # Initialize a variable to store the previous element in the list
    # We start with the first element in the list
    previous_element = lst[0]

    # Iterate over the list starting from the second element
    for current_element in lst[1:]:
        # If the current element is less than the previous element, the list is not sorted
        if current_element < previous_element:
            return False
        
        # Update the previous element for the next iteration
        previous_element = current_element

    # If we've iterated over the entire list and haven't found any unsorted pairs, the list is sorted
    return True
