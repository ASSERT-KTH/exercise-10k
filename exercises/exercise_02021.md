## Exercise
Create a program that determines the winner of a rock, paper, scissors game between two players. The program should take two inputs: the choice of player 1 and the choice of player 2.

## Reference Solution
```python
def determine_winner(player1_choice, player2_choice):
    """
    This function determines the winner of a rock, paper, scissors game.
    
    Args:
        player1_choice (str): The choice of player 1. It can be 'rock', 'paper', or 'scissors'.
        player2_choice (str): The choice of player 2. It can be 'rock', 'paper', or 'scissors'.
    
    Returns:
        str: The result of the game. It can be 'Player 1 wins', 'Player 2 wins', or 'It\'s a tie'.
    """
    
    # Define the winning conditions
    winning_conditions = {
        'rock': 'scissors',
        'paper': 'rock',
        'scissors': 'paper'
    }
    
    # Check if the inputs are valid
    if player1_choice not in winning_conditions or player2_choice not in winning_conditions:
        return "Invalid input. Please enter 'rock', 'paper', or 'scissors'."
    
    # Check if it's a tie
    if player1_choice == player2_choice:
        return "It's a tie."
    
    # Check if player 1 wins
    if winning_conditions[player1_choice] == player2_choice:
        return "Player 1 wins."
    
    # If none of the above conditions are met, player 2 wins
    return "Player 2 wins."


def main():
    # Get the choices of the players
    player1_choice = input("Enter the choice of player 1 (rock, paper, or scissors): ")
    player2_choice = input("Enter the choice of player 2 (rock, paper, or scissors): ")
    
    # Determine the winner
    result = determine_winner(player1_choice.lower(), player2_choice.lower())
    
    # Print the result
    print(result)


if __name__ == "__main__":
    main()
```
