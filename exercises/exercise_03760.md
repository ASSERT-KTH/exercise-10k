## Exercise
Create a program to generate all permutations of a given list of elements using a basic algorithm

## Reference Solution
```python
def generate_permutations(elements):
    """
    Generate all permutations of a given list of elements.

    Args:
        elements (list): The list of elements.

    Returns:
        list: A list of all permutations.
    """
    # Base case: if the list is empty or contains only one element
    if len(elements) <= 1:
        return [elements]

    permutations = []
    for i in range(len(elements)):
        # Choose the current element as the first element of the permutation
        current_element = elements[i]
        # Get the remaining elements
        remaining_elements = elements[:i] + elements[i+1:]
        # Generate all permutations of the remaining elements
        for permutation in generate_permutations(remaining_elements):
            # Add the current element to the permutation
            permutations.append([current_element] + permutation)

    return permutations

def main():
    elements = [1, 2, 3]
    permutations = generate_permutations(elements)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"{i+1}: {permutation}")

if __name__ == "__main__":
    main()
```
