## Exercise
Write a program to perform basic arithmetic operations using variables.

## Reference Solution
```python
# Define a function to perform arithmetic operations
def perform_arithmetic_operations(num1, num2):
    """
    Perform basic arithmetic operations on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.

    Returns:
        dict: A dictionary containing the results of the arithmetic operations.
    """
    # Perform arithmetic operations
    add_result = num1 + num2
    subtract_result = num1 - num2
    multiply_result = num1 * num2
    divide_result = num1 / num2 if num2 != 0 else "Error: Division by zero"

    # Store the results in a dictionary
    results = {
        "Addition": add_result,
        "Subtraction": subtract_result,
        "Multiplication": multiply_result,
        "Division": divide_result
    }

    return results

# Main function
def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    # Perform arithmetic operations
    results = perform_arithmetic_operations(num1, num2)

    # Print the results
    print("Arithmetic Operation Results:")
    for operation, result in results.items():
        print(f"{operation}: {result}")

# Run the main function
if __name__ == "__main__":
    main()
```
