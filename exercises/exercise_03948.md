## Exercise
Write a program to simulate a game where two players take turns rolling a dice. The first player to roll a 6 wins.

## Reference Solution
```python
import random

def roll_dice():
    """Simulate a dice roll and return the result."""
    return random.randint(1, 6)

def game():
    """Simulate the game and return the winner."""
    player_turn = "Player 1"
    while True:
        input(f"{player_turn}, press Enter to roll the dice...")
        roll = roll_dice()
        print(f"{player_turn} rolled a {roll}")
        if roll == 6:
            return player_turn
        player_turn = "Player 2" if player_turn == "Player 1" else "Player 1"

def main():
    """Run the game and print the winner."""
    winner = game()
    print(f"{winner} wins!")

if __name__ == "__main__":
    main()
```
