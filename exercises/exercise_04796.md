## Exercise
Write a program that checks if two words are anagrams of each other.

## Reference Solution
```python
def are_anagrams(word1, word2):
    """
    Checks if two words are anagrams of each other.

    Args:
        word1 (str): The first word.
        word2 (str): The second word.

    Returns:
        bool: True if the words are anagrams, False otherwise.
    """
    # Convert both words to lowercase and remove any whitespace
    word1 = word1.replace(" ", "").lower()
    word2 = word2.replace(" ", "").lower()

    # Check if the sorted letters of both words are equal
    return sorted(word1) == sorted(word2)


def main():
    # Example usage
    word1 = "listen"
    word2 = "silent"
    print(f"Are '{word1}' and '{word2}' anagrams? {are_anagrams(word1, word2)}")

    word1 = "hello"
    word2 = "world"
    print(f"Are '{word1}' and '{word2}' anagrams? {are_anagrams(word1, word2)}")


if __name__ == "__main__":
    main()
```
