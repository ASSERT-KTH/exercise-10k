## Exercise
Write a program that reverses the order of elements in a given list.

## Reference Solution
```python
def reverse_list(input_list):
    """
    Reverses the order of elements in a given list.

    Args:
        input_list (list): The list to be reversed.

    Returns:
        list: The reversed list.
    """
    return input_list[::-1]

# Example usage
if __name__ == "__main__":
    # Create a sample list
    sample_list = [1, 2, 3, 4, 5]
    print("Original List:", sample_list)

    # Reverse the list
    reversed_list = reverse_list(sample_list)
    print("Reversed List:", reversed_list)
```

```python
def reverse_list_in_place(input_list):
    """
    Reverses the order of elements in a given list in-place.

    Args:
        input_list (list): The list to be reversed.
    """
    input_list.reverse()

# Example usage
if __name__ == "__main__":
    # Create a sample list
    sample_list = [1, 2, 3, 4, 5]
    print("Original List:", sample_list)

    # Reverse the list in-place
    reverse_list_in_place(sample_list)
    print("Reversed List:", sample_list)
```
