## Exercise
Write a program that checks the data type of a variable and prints an error message if it's not the expected type.

## Reference Solution
```python
def check_type(a):
    if type(a) == int:
        print(f"{a} is an integer")
    elif type(a) == str:
        print(f"{a} is a string")
    elif type(a) == list:
        print(f"{a} is a list")
    elif type(a) == dict:
        print(f"{a} is a dictionary")
    elif type(a) == tuple:
        print(f"{a} is a tuple")
    elif type(a) == set:
        print(f"{a} is a set")
    elif type(a) == complex:
        print(f"{a} is a complex number")
    elif type(a) == float:
        print(f"{a} is a float")
    elif type(a) == bool:
        print(f"{a} is a boolean")
    else:
        print(f"{a} is of type {type(a)}")

# Test the function
check_type(5)
check_type("hello")
check_type([1, 2, 3])
check_type({"name": "John", "age": 30})
check_type((1, 2, 3))
check_type({1, 2, 3})
check_type(3+4j)
check_type(3.14)
check_type(True)
check_type(None)
```
