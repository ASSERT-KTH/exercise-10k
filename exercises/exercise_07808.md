## Exercise
Write a program that formats a phone number in a standard way (e.g. (123) 456-7890).

## Reference Solution
```python
def format_phone_number(phone_number: str) -> str:
    """
    Formats a phone number in a standard way (e.g., (123) 456-7890).

    Args:
        phone_number (str): The phone number to be formatted.

    Returns:
        str: The formatted phone number.
    """
    # Remove non-numeric characters from the phone number
    digits = ''.join(filter(str.isdigit, phone_number))

    # Check if the phone number has exactly 10 digits
    if len(digits) != 10:
        raise ValueError("Invalid phone number. It should have exactly 10 digits.")

    # Format the phone number
    formatted_phone_number = f"({digits[:3]}) {digits[3:6]}-{digits[6:]}"

    return formatted_phone_number


def main():
    # Example usage
    phone_number = "(123) 456-7890"
    print(f"Original phone number: {phone_number}")
    print(f"Formatted phone number: {format_phone_number(phone_number)}")

    # Test with different inputs
    print("\nTesting with different inputs:")
    test_cases = [
        "(123) 456-7890",
        "123-456-7890",
        "1234567890",
        "123.456.7890",
        "(123)456-7890",
        "1234567"  # Invalid phone number
    ]

    for phone_number in test_cases:
        try:
            print(f"Original phone number: {phone_number}")
            print(f"Formatted phone number: {format_phone_number(phone_number)}")
        except ValueError as e:
            print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
