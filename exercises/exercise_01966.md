## Exercise
Create a function that takes two functions as arguments and returns a new function that is the composition of the two input functions.

# We start by defining a function that takes two functions as arguments.
def compose_functions(func1, func2):
    """
    This function takes two functions as arguments and returns a new function 
    that is the composition of the two input functions.

    Args:
        func1 (function): The first function to be composed.
        func2 (function): The second function to be composed.

    Returns:
        function: A new function that is the composition of func1 and func2.
    """

    # We define a new function inside our compose_functions function. This new function will be returned by compose_functions.
    def composed_function(*args, **kwargs):
        
        # First, we apply func2 to the input arguments and keyword arguments.
        result_of_func2 = func2(*args, **kwargs)
        
        # Then, we apply func1 to the result of func2.
        result_of_func1 = func1(result_of_func2)
        
        # Finally, we return the result of func1.
        return result_of_func1
    
    # We return the composed_function.
    return composed_function

# Example usage:
def add_one(x):
    return x + 1

def square(x):
    return x ** 2

composed = compose_functions(add_one, square)
print(composed(5))  # Output: 26
