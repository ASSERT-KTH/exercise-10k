## Exercise
Given two strings, calculate the edit distance between them, which is the minimum number of operations (insertions, deletions, substitutions) needed to transform one string into the other.

## Reference Solution

def edit_distance(str1, str2):
    """
    Calculate the edit distance between two strings.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        int: The minimum number of operations needed to transform str1 into str2.
    """

    # Initialize the matrix with zeros
    dp = [[0 for _ in range(len(str2) + 1)] for _ in range(len(str1) + 1)]

    # Initialize the first row and column
    for i in range(len(str1) + 1):
        dp[i][0] = i
    for j in range(len(str2) + 1):
        dp[0][j] = j

    # Fill in the rest of the matrix
    for i in range(1, len(str1) + 1):
        for j in range(1, len(str2) + 1):
            # Check if the current characters match
            if str1[i - 1] == str2[j - 1]:
                # If they match, no operation is needed
                dp[i][j] = dp[i - 1][j - 1]
            else:
                # If they don't match, consider insertion, deletion, and substitution
                dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1])

    # The edit distance is stored in the bottom-right corner of the matrix
    return dp[len(str1)][len(str2)]

# Example usage
str1 = "kitten"
str2 = "sitting"
print("Edit distance:", edit_distance(str1, str2))
