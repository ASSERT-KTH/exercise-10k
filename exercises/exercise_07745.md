## Exercise
Create a program that uses a dictionary to store student names as keys and their corresponding grades as values. The program should allow users to add students, update grades, and calculate the average grade of all students.

## Reference Solution
```python
# Define a dictionary to store student grades
student_grades = {}

# Function to add a student
def add_student(name, grade):
    student_grades[name] = grade

# Function to update a student's grade
def update_grade(name, new_grade):
    if name in student_grades:
        student_grades[name] = new_grade
    else:
        print("Student not found.")

# Function to calculate the average grade
def calculate_average():
    total = sum(student_grades.values())
    average = total / len(student_grades)
    return average

# Main function
def main():
    while True:
        print("\n1. Add student")
        print("2. Update grade")
        print("3. Calculate average grade")
        print("4. Exit")
        
        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter student's name: ")
            grade = float(input("Enter student's grade: "))
            add_student(name, grade)
        elif choice == "2":
            name = input("Enter student's name: ")
            new_grade = float(input("Enter new grade: "))
            update_grade(name, new_grade)
        elif choice == "3":
            average = calculate_average()
            print("The average grade is: ", average)
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
