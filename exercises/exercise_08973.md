## Exercise
Write a function that takes a string and a character as input and returns the index of the character in the string.

## Reference Solution
```python
def find_char_index(input_str, char):
    """
    Find the index of a character in a string.

    Args:
        input_str (str): The input string.
        char (str): The character to find.

    Returns:
        int: The index of the character if found, -1 otherwise.
    """
    try:
        # Use the index method to find the index of the character
        index = input_str.index(char)
        return index
    except ValueError:
        # If the character is not found, return -1
        return -1

# Example usage:
def main():
    input_str = "Hello, World!"
    char = "W"
    index = find_char_index(input_str, char)
    if index != -1:
        print(f"The character '{char}' is found at index {index}.")
    else:
        print(f"The character '{char}' is not found in the string.")

if __name__ == "__main__":
    main()
```
