## Exercise
Write a program to compress and decompress data using a simple compression algorithm.

## Reference Solution
```python
def compress(data):
    """
    Compress data using Run-Length Encoding (RLE) algorithm.

    Args:
        data (str): The input data to be compressed.

    Returns:
        str: The compressed data.
    """
    if not data:
        return ""

    compressed = []
    count = 1

    for i in range(1, len(data)):
        if data[i] == data[i - 1]:
            count += 1
        else:
            compressed.append(str(count) + data[i - 1])
            count = 1

    compressed.append(str(count) + data[-1])
    return "".join(compressed)


def decompress(compressed_data):
    """
    Decompress data using Run-Length Encoding (RLE) algorithm.

    Args:
        compressed_data (str): The compressed data to be decompressed.

    Returns:
        str: The decompressed data.
    """
    if not compressed_data:
        return ""

    decompressed = []
    count = ""

    for char in compressed_data:
        if char.isdigit():
            count += char
        else:
            decompressed.append(char * int(count))
            count = ""

    return "".join(decompressed)


# Example usage
if __name__ == "__main__":
    data = "AAABBBCCCDDDD"
    print("Original Data:", data)

    compressed_data = compress(data)
    print("Compressed Data:", compressed_data)

    decompressed_data = decompress(compressed_data)
    print("Decompressed Data:", decompressed_data)
```
