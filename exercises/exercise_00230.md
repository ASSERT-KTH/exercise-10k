## Exercise
Write a program to convert a given string to uppercase or lowercase.

## Reference Solution
```python
def convert_case(input_string, case):
    """
    Converts the input string to the specified case.

    Args:
        input_string (str): The string to be converted.
        case (str): The case to convert to. Can be 'upper' or 'lower'.

    Returns:
        str: The converted string.
    """
    if case == 'upper':
        return input_string.upper()
    elif case == 'lower':
        return input_string.lower()
    else:
        raise ValueError("Invalid case. Please choose 'upper' or 'lower'.")

def main():
    input_string = input("Enter a string: ")
    print("Choose an option:")
    print("1. Convert to uppercase")
    print("2. Convert to lowercase")
    choice = input("Enter your choice (1/2): ")

    if choice == '1':
        case = 'upper'
    elif choice == '2':
        case = 'lower'
    else:
        print("Invalid choice. Exiting program.")
        return

    try:
        converted_string = convert_case(input_string, case)
        print(f"Converted string: {converted_string}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
