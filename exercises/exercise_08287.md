## Exercise
Write a program that solves a linear equation.

## Reference Solution
```python
def solve_linear_equation(a, b, c):
    """
    Solves a linear equation of the form ax + b = c.

    Args:
        a (float): Coefficient of x.
        b (float): Constant term.
        c (float): Right-hand side of the equation.

    Returns:
        float: Solution to the equation.
    """
    if a == 0:
        if b == c:
            return "Infinite solutions"
        else:
            return "No solution"
    else:
        return (c - b) / a

def main():
    # Get user input
    a = float(input("Enter coefficient of x (a): "))
    b = float(input("Enter constant term (b): "))
    c = float(input("Enter right-hand side of the equation (c): "))

    # Solve the equation
    solution = solve_linear_equation(a, b, c)

    # Print the solution
    if isinstance(solution, str):
        print(solution)
    else:
        print(f"The solution to the equation is x = {solution}")

if __name__ == "__main__":
    main()
```
