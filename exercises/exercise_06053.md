## Exercise
Create a program that allows users to track their expenses and income using conditional statements, providing budgeting recommendations.

## Reference Solution
```python
class ExpenseTracker:
    def __init__(self):
        self.income = 0
        self.expenses = {}

    def set_income(self):
        while True:
            try:
                self.income = float(input("Enter your monthly income: $"))
                if self.income <= 0:
                    print("Income must be greater than zero.")
                else:
                    break
            except ValueError:
                print("Invalid input. Please enter a number.")

    def add_expense(self):
        category = input("Enter expense category (e.g., rent, groceries, entertainment): ")
        while True:
            try:
                amount = float(input(f"Enter {category} expense amount: $"))
                if amount <= 0:
                    print("Expense amount must be greater than zero.")
                else:
                    self.expenses[category] = amount
                    break
            except ValueError:
                print("Invalid input. Please enter a number.")

    def view_expenses(self):
        print("\nExpense Summary:")
        for category, amount in self.expenses.items():
            print(f"{category}: ${amount:.2f}")

    def calculate_total_expenses(self):
        return sum(self.expenses.values())

    def provide_budget_recommendations(self):
        total_expenses = self.calculate_total_expenses()
        if total_expenses > self.income:
            print("\nWarning: Your total expenses exceed your income.")
            print("Consider reducing expenses or increasing income.")
        elif total_expenses / self.income > 0.7:
            print("\nWarning: Your expenses are high. Consider reducing expenses.")
        else:
            print("\nYou are within a healthy budget range.")

    def run(self):
        self.set_income()
        while True:
            print("\nOptions:")
            print("1. Add expense")
            print("2. View expenses")
            print("3. Calculate total expenses")
            print("4. Provide budget recommendations")
            print("5. Quit")
            choice = input("Choose an option: ")
            if choice == "1":
                self.add_expense()
            elif choice == "2":
                self.view_expenses()
            elif choice == "3":
                print(f"\nTotal expenses: ${self.calculate_total_expenses():.2f}")
            elif choice == "4":
                self.provide_budget_recommendations()
            elif choice == "5":
                break
            else:
                print("Invalid option. Please choose again.")

if __name__ == "__main__":
    tracker = ExpenseTracker()
    tracker.run()
```
