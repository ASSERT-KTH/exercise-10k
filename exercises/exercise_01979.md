## Exercise
Create a function that implements a graph search algorithm to find the shortest path between two nodes in a graph. The function should take a graph and two nodes as arguments and return the shortest path.

from collections import deque

def shortest_path(graph, node1, node2):
    # Check if the graph is empty
    if not graph:
        return None
    
    # Check if the nodes are in the graph
    if node1 not in graph or node2 not in graph:
        return None
    
    # Create a queue and enqueue the starting node
    queue = deque([[node1]])
    
    # Create a set to store visited nodes
    visited = set()
    
    # While the queue is not empty
    while queue:
        # Dequeue the first path from the queue
        path = queue.popleft()
        
        # Get the last node from the path
        node = path[-1]
        
        # If the node is not visited
        if node not in visited:
            # Mark the node as visited
            visited.add(node)
            
            # If the node is the destination node, return the path
            if node == node2:
                return path
            
            # For each neighbor of the node that has not been visited
            for neighbor in graph[node]:
                if neighbor not in visited:
                    # Create a new path by appending the neighbor to the current path
                    new_path = list(path)
                    new_path.append(neighbor)
                    
                    # Enqueue the new path
                    queue.append(new_path)
    
    # If there is no path between the nodes, return None
    return None

# Example usage:
graph = {
    'A': ['B', 'C'],
    'B': ['A', 'D', 'E'],
    'C': ['A', 'F'],
    'D': ['B'],
    'E': ['B', 'F'],
    'F': ['C', 'E'],
}

print(shortest_path(graph, 'A', 'F'))  # Output: ['A', 'C', 'F']
