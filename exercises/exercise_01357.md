## Exercise
Write a program to demonstrate the use of timing and sleep functions

## Reference Solution
```python
import time

def main():
    # Get the current time
    current_time = time.time()
    print(f"Current time: {time.ctime(current_time)}")

    # Specify the sleep duration in seconds
    sleep_duration = 5

    # Pause the program for the specified duration
    print(f"Sleeping for {sleep_duration} seconds...")
    time.sleep(sleep_duration)

    # Get the current time after sleeping
    current_time = time.time()
    print(f"Current time after sleeping: {time.ctime(current_time)}")

    # Measure the execution time of a block of code
    start_time = time.time()
    for i in range(10000000):
        pass
    end_time = time.time()
    execution_time = end_time - start_time
    print(f"Execution time: {execution_time:.2f} seconds")

if __name__ == "__main__":
    main()
```
