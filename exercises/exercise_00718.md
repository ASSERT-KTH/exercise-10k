## Exercise
Write a function to apply a function to all values in a dictionary and return a single value

## Reference Solution
```python
from functools import reduce
from operator import add

def apply_function_to_dict_values(dictionary, func, reduction_func=add):
    """
    Applies a given function to all values in a dictionary and returns a single value.
    
    Args:
        dictionary (dict): The input dictionary.
        func (function): The function to apply to each value in the dictionary.
        reduction_func (function, optional): The function to use for reducing the results. Defaults to add.
    
    Returns:
        The result of applying the function to all values and reducing them.
    """
    
    # Apply the function to each value in the dictionary
    results = [func(value) for value in dictionary.values()]
    
    # Reduce the results to a single value
    if len(results) == 0:
        return None  # or any other default value for empty dictionary
    elif len(results) == 1:
        return results[0]
    else:
        return reduce(reduction_func, results)

# Example usage:
if __name__ == "__main__":
    # Define a dictionary
    example_dict = {'a': 1, 'b': 2, 'c': 3}
    
    # Define a simple function to square a number
    def square(x):
        return x ** 2
    
    # Apply the square function to the dictionary values and sum the results
    result = apply_function_to_dict_values(example_dict, square)
    print("Sum of squares:", result)
    
    # Apply the square function to the dictionary values and multiply the results
    from operator import mul
    result = apply_function_to_dict_values(example_dict, square, reduction_func=mul)
    print("Product of squares:", result)
```
