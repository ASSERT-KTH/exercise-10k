## Exercise
Write a program that takes the principal, rate, and time as input and returns the compound interest.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time):
    """
    Calculate compound interest.

    Args:
        principal (float): The initial amount of money.
        rate (float): The interest rate (in decimal form, e.g., 4% = 0.04).
        time (int): The time period (in years).

    Returns:
        float: The compound interest.
    """
    compound_interest = principal * (1 + rate) ** time - principal
    return compound_interest

def main():
    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))

    # Calculate compound interest
    interest = calculate_compound_interest(principal, rate, time)

    # Print result
    print(f"The compound interest is: ${interest:.2f}")

if __name__ == "__main__":
    main()
```
