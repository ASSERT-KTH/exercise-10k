## Exercise
Create a function to find the uncommon characters between two given strings.

## Reference Solution
```python
def find_uncommon_chars(str1, str2):
    """
    Find the uncommon characters between two given strings.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        set: A set of uncommon characters.
    """
    # Convert the strings to sets of characters
    set1 = set(str1)
    set2 = set(str2)

    # Use set difference to find the uncommon characters
    uncommon_chars = set1.symmetric_difference(set2)

    return uncommon_chars

def main():
    # Example usage
    str1 = "abcde"
    str2 = "acefg"
    uncommon_chars = find_uncommon_chars(str1, str2)
    print("Uncommon characters:", uncommon_chars)

if __name__ == "__main__":
    main()
```
