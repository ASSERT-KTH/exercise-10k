## Exercise
Write a function that uses currying to transform a function that takes multiple arguments into a sequence of functions each taking one argument.

## Reference Solution

def curry(func):
    """
    This function uses currying to transform a function that takes multiple arguments into a sequence of functions each taking one argument.

    Args:
        func (function): The function to be curried.

    Returns:
        function: A curried version of the input function.
    """

    # Check if the input is a function
    if not callable(func):
        raise TypeError("Input must be a function")

    # Get the number of arguments the function takes
    num_args = func.__code__.co_argcount

    # Define a helper function to perform the currying
    def helper(*args, **kwargs):
        # If the number of arguments passed is equal to the number of arguments the function takes,
        # we can call the function and return the result
        if len(args) + len(kwargs) == num_args:
            return func(*args, **kwargs)
        
        # If the number of arguments passed is less than the number of arguments the function takes,
        # we return a new function that takes the remaining arguments
        else:
            # Define a new function that takes the remaining arguments
            def new_func(*new_args, **new_kwargs):
                return helper(*(args + new_args), **dict(kwargs, **new_kwargs))
            
            # Return the new function
            return new_func

    # Return the helper function
    return helper


# Example usage:
def add(a, b, c):
    return a + b + c

curried_add = curry(add)

print(curried_add(1)(2)(3))  # Output: 6
