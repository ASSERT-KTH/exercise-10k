## Exercise
Write a program that plays Rock, Paper, Scissors with the user, keeps track of the score, and announces the winner after a set number of rounds.

## Reference Solution
```python
# Importing the random module
import random

# Defining the choices
choices = ['rock', 'paper', 'scissors']

# Defining the beats dictionary
beats = {
    'rock': 'scissors',
    'paper': 'rock',
    'scissors': 'paper'
}

# Defining the score variables
user_score = 0
computer_score = 0

# Defining the rounds variable
rounds = 5

# Defining the play_game function
def play_game():
    global user_score, computer_score
    print("Welcome to Rock, Paper, Scissors!")
    for i in range(rounds):
        user_choice = input("Enter your choice (rock, paper, or scissors): ").lower()
        computer_choice = random.choice(choices)
        print(f"Computer chose {computer_choice}")
        if beats[user_choice] == computer_choice:
            user_score += 1
            print("You win this round!")
        elif beats[computer_choice] == user_choice:
            computer_score += 1
            print("Computer wins this round!")
        else:
            print("It's a tie!")
        print(f"Score: You {user_score}, Computer {computer_score}\n")
    print("Game over!")
    if user_score > computer_score:
        print("You win!")
    elif computer_score > user_score:
        print("Computer wins!")
    else:
        print("It's a tie game!")

# Calling the play_game function
play_game()
```
