## Exercise
Develop a program that calculates statistical measures (mean, median, mode, standard deviation) for a given dataset.

## Reference Solution
```python
import statistics
import math

def calculate_statistical_measures(data):
    """
    Calculate statistical measures (mean, median, mode, standard deviation) for a given dataset.

    Args:
        data (list): A list of numbers.

    Returns:
        dict: A dictionary containing the calculated statistical measures.
    """
    # Check if the input data is a list
    if not isinstance(data, list):
        raise ValueError("Input data must be a list")

    # Check if the list is empty
    if len(data) == 0:
        raise ValueError("Input data cannot be empty")

    # Check if all elements in the list are numbers
    if not all(isinstance(x, (int, float)) for x in data):
        raise ValueError("All elements in the input data must be numbers")

    # Calculate the mean
    mean = statistics.mean(data)

    # Calculate the median
    median = statistics.median(data)

    # Calculate the mode
    try:
        mode = statistics.mode(data)
    except statistics.StatisticsError:
        mode = "No unique mode found"

    # Calculate the standard deviation
    if len(data) > 1:
        std_dev = statistics.stdev(data)
    else:
        std_dev = 0

    # Return the calculated statistical measures
    return {
        "mean": mean,
        "median": median,
        "mode": mode,
        "standard_deviation": std_dev
    }

def main():
    # Example usage
    data = [1, 2, 3, 4, 5]
    statistical_measures = calculate_statistical_measures(data)

    print("Statistical Measures:")
    print("----------------------")
    print(f"Mean: {statistical_measures['mean']}")
    print(f"Median: {statistical_measures['median']}")
    print(f"Mode: {statistical_measures['mode']}")
    print(f"Standard Deviation: {statistical_measures['standard_deviation']}")

if __name__ == "__main__":
    main()
```
