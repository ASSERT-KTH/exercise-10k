## Exercise
Create a program that takes a list of numbers as input and outputs the mean, median, and mode.

## Reference Solution
```python
import statistics
from collections import Counter

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and mode of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the mean, median, and mode.
    """
    mean = statistics.mean(numbers)
    median = statistics.median(numbers)
    mode = statistics.mode(numbers)

    return mean, median, mode

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 4, 5, 5, 5]
    mean, median, mode = calculate_statistics(numbers)

    print("Mean:", mean)
    print("Median:", median)
    print("Mode:", mode)

    # Alternative way to calculate mode when there are multiple modes
    counter = Counter(numbers)
    modes = [num for num, count in counter.items() if count == max(counter.values())]
    print("Modes (alternative method):", modes)

if __name__ == "__main__":
    main()
```
