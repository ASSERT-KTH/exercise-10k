Write a program that takes an object as input and processes it, then outputs the result.

## Reference Solution

def process_object(obj):
    """
    This function takes an object as input, processes it, and returns the result.
    
    The processing step in this example is simply to print the type and value of the object.
    In a real-world application, you would replace this with your actual processing logic.
    
    Args:
        obj: The object to be processed.
    
    Returns:
        The processed object.
    """
    print(f"Object type: {type(obj)}")
    print(f"Object value: {obj}")
    return obj

def main():
    # Test the function with different types of objects
    objects = [1, "hello", 3.14, True, None, [1, 2, 3], {"a": 1, "b": 2}]
    
    for obj in objects:
        print(f"Processing object: {obj}")
        result = process_object(obj)
        print(f"Result: {result}\n")

if __name__ == "__main__":
    main()
