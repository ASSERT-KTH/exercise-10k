## Exercise
Write a program to calculate an employee's salary based on their hourly wage and the number of hours worked, using a for loop to handle multiple employees.

## Reference Solution
```python
# Function to calculate an employee's salary
def calculate_salary(hourly_wage, hours_worked):
    return hourly_wage * hours_worked

# Main function
def main():
    # List to store employee information
    employees = []

    # Ask the user for the number of employees
    num_employees = int(input("Enter the number of employees: "))

    # Loop through each employee and ask for their hourly wage and hours worked
    for i in range(num_employees):
        print(f"\nEnter information for Employee {i + 1}:")
        hourly_wage = float(input("Enter hourly wage: $"))
        hours_worked = float(input("Enter hours worked: "))

        # Calculate the employee's salary
        salary = calculate_salary(hourly_wage, hours_worked)

        # Store the employee's information and salary
        employees.append({
            "hourly_wage": hourly_wage,
            "hours_worked": hours_worked,
            "salary": salary
        })

    # Print out the salary for each employee
    print("\nEmployee Salaries:")
    for i, employee in enumerate(employees):
        print(f"Employee {i + 1}:")
        print(f"Hourly Wage: ${employee['hourly_wage']:.2f}")
        print(f"Hours Worked: {employee['hours_worked']:.2f} hours")
        print(f"Salary: ${employee['salary']:.2f}\n")

# Call the main function
if __name__ == "__main__":
    main()
```
