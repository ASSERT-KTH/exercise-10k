## Exercise
Create a program that determines the number of days in a given month, using conditional statements to account for leap years and month lengths

## Reference Solution
```python
def is_leap_year(year):
    """
    Check if a year is a leap year.
    
    Args:
    year (int): The year to check.
    
    Returns:
    bool: True if the year is a leap year, False otherwise.
    """
    return year % 4 == 0 and (year % 100 != 0 or year % 400 == 0)


def days_in_month(month, year):
    """
    Calculate the number of days in a given month and year.
    
    Args:
    month (int): The month (1-12) to calculate days for.
    year (int): The year to consider for leap year checks.
    
    Returns:
    int: The number of days in the given month and year.
    """
    if month in [1, 3, 5, 7, 8, 10, 12]:  # Months with 31 days
        return 31
    elif month == 2:  # February
        if is_leap_year(year):
            return 29  # Leap year, February has 29 days
        else:
            return 28  # Not a leap year, February has 28 days
    else:  # Months with 30 days
        return 30


def main():
    year = int(input("Enter the year: "))
    month = int(input("Enter the month (1-12): "))
    
    if 1 <= month <= 12:
        days = days_in_month(month, year)
        print(f"There are {days} days in month {month} of the year {year}.")
    else:
        print("Invalid month. Please enter a number between 1 and 12.")


if __name__ == "__main__":
    main()
```
