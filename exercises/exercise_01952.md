## Exercise
Write a program that takes a list as input and returns the maximum value in the list.

## Reference Solution
```python
def find_max_value(lst):
    """
    This function finds the maximum value in a list.

    Args:
        lst (list): A list of numbers.

    Returns:
        max_value: The maximum value in the list.
    """
    if not lst:
        return None  # Return None if the list is empty
    return max(lst)

# Example usage:
numbers = [4, 2, 9, 6, 5, 1, 8, 3, 7]
max_value = find_max_value(numbers)
print("Maximum value in the list:", max_value)
```

def find_max_value_manual(lst):
    """
    This function manually finds the maximum value in a list.

    Args:
        lst (list): A list of numbers.

    Returns:
        max_value: The maximum value in the list.
    """
    if not lst:
        return None  # Return None if the list is empty
    max_value = lst[0]  # Initialize max_value with the first element of the list
    for num in lst[1:]:  # Iterate through the rest of the list
        if num > max_value:  # If a number is greater than max_value, update max_value
            max_value = num
    return max_value

# Example usage:
numbers = [4, 2, 9, 6, 5, 1, 8, 3, 7]
max_value = find_max_value_manual(numbers)
print("Maximum value in the list (manual):", max_value)
