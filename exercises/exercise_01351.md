## Exercise
Write a program to convert a given value from one data type to another (e.g., int to float, str to int, etc.)

## Reference Solution
```python
def convert_data_type(value, from_type, to_type):
    """
    Converts a given value from one data type to another.

    Args:
        value (any): The value to be converted.
        from_type (str): The current data type of the value.
        to_type (str): The desired data type of the value.

    Returns:
        any: The converted value.

    Raises:
        ValueError: If the conversion is not supported or fails.
    """
    conversions = {
        ('int', 'float'): float,
        ('str', 'int'): int,
        ('str', 'float'): float,
        ('float', 'int'): int,
        ('int', 'str'): str,
        ('float', 'str'): str
    }

    if (from_type, to_type) not in conversions:
        raise ValueError(f"Conversion from {from_type} to {to_type} is not supported")

    try:
        return conversions[(from_type, to_type)](value)
    except ValueError:
        raise ValueError(f"Failed to convert {value} from {from_type} to {to_type}")

def main():
    # Example usage
    print(convert_data_type(10, 'int', 'float'))  # Output: 10.0
    print(convert_data_type('10', 'str', 'int'))  # Output: 10
    print(convert_data_type('10.5', 'str', 'float'))  # Output: 10.5
    print(convert_data_type(10.5, 'float', 'int'))  # Output: 10
    print(convert_data_type(10, 'int', 'str'))  # Output: '10'
    print(convert_data_type(10.5, 'float', 'str'))  # Output: '10.5'

if __name__ == "__main__":
    main()
```
