## Exercise
Write a program that uses a for loop to check if two given strings are anagrams of each other.

## Reference Solution
```python
def are_anagrams(str1, str2):
    """
    Check if two strings are anagrams of each other.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        bool: True if the strings are anagrams, False otherwise.
    """
    # Remove any white spaces and convert to lower case
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # If the lengths of the strings are not equal, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Create a copy of the second string
    str2_copy = list(str2)

    # Iterate over each character in the first string
    for char in str1:
        # Check if the character exists in the copy of the second string
        if char in str2_copy:
            # Remove the character from the copy of the second string
            str2_copy.remove(char)
        else:
            # If the character does not exist, the strings are not anagrams
            return False

    # If we have checked all characters and removed them from the copy of the second string,
    # the strings are anagrams
    return True


# Example usage
if __name__ == "__main__":
    str1 = "Listen"
    str2 = "Silent"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")

    str1 = "Hello"
    str2 = "World"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")
```
