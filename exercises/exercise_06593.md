## Exercise
Write a program to find the variance of all numbers in an array.

## Reference Solution
```python
import numpy as np

def calculate_variance(array):
    """
    Calculate the variance of all numbers in an array.

    Parameters:
    array (list): A list of numbers.

    Returns:
    float: The variance of the numbers in the array.
    """
    # Calculate the mean of the array
    mean = np.mean(array)
    
    # Calculate the squared differences from the mean
    squared_diffs = [(x - mean) ** 2 for x in array]
    
    # Calculate the variance
    variance = np.mean(squared_diffs)
    
    return variance

def main():
    # Example usage
    array = [1, 2, 3, 4, 5]
    variance = calculate_variance(array)
    print("Variance:", variance)

if __name__ == "__main__":
    main()
```
