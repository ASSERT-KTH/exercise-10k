## Exercise
Create a program that calculates the grade of a student given their scores in different subjects.

## Reference Solution
```python
# Define a function to calculate the grade
def calculate_grade(scores):
    """
    Calculate the grade of a student based on their scores in different subjects.

    Args:
        scores (dict): A dictionary where keys are subject names and values are scores.

    Returns:
        str: The grade of the student.
    """
    # Initialize total score and total maximum score
    total_score = 0
    total_max_score = 0

    # Iterate over each subject and its score
    for subject, score in scores.items():
        # Assume each subject has a maximum score of 100
        total_score += score
        total_max_score += 100

    # Calculate the average score
    average_score = (total_score / total_max_score) * 100

    # Determine the grade based on the average score
    if average_score >= 90:
        grade = "A"
    elif average_score >= 80:
        grade = "B"
    elif average_score >= 70:
        grade = "C"
    elif average_score >= 60:
        grade = "D"
    else:
        grade = "F"

    return grade


# Define a function to get the scores from the user
def get_scores():
    """
    Get the scores of a student in different subjects from the user.

    Returns:
        dict: A dictionary where keys are subject names and values are scores.
    """
    scores = {}

    # Get the number of subjects
    num_subjects = int(input("Enter the number of subjects: "))

    # Get the scores for each subject
    for i in range(num_subjects):
        subject = input(f"Enter the name of subject {i + 1}: ")
        score = float(input(f"Enter the score for {subject}: "))

        # Validate the score
        while score < 0 or score > 100:
            print("Invalid score. Score must be between 0 and 100.")
            score = float(input(f"Enter the score for {subject}: "))

        scores[subject] = score

    return scores


# Main function
def main():
    print("Student Grade Calculator")
    print("------------------------")

    # Get the scores from the user
    scores = get_scores()

    # Calculate the grade
    grade = calculate_grade(scores)

    # Print the grade
    print(f"\nGrade: {grade}")


if __name__ == "__main__":
    main()
```
