## Exercise
Write a program that sorts an array in ascending or descending order.

## Reference Solution
```python
def sort_array(array, order):
    """
    Sorts the input array in the specified order.

    Args:
        array (list): The input array to be sorted.
        order (str): The order of sorting. It can be 'asc' for ascending or 'desc' for descending.

    Returns:
        list: The sorted array.
    """
    if order == 'asc':
        return sorted(array)
    elif order == 'desc':
        return sorted(array, reverse=True)
    else:
        raise ValueError("Invalid order. It can be 'asc' or 'desc'.")

def main():
    array = [64, 34, 25, 12, 22, 11, 90]
    print("Original array:", array)

    asc_array = sort_array(array, 'asc')
    print("Sorted array in ascending order:", asc_array)

    desc_array = sort_array(array, 'desc')
    print("Sorted array in descending order:", desc_array)

if __name__ == "__main__":
    main()
```
