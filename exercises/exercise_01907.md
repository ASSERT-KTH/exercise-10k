## Exercise
Write a program to calculate the compound interest given the principal amount, interest rate, time period, and number of times interest applied per time period.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, n):
    """
    Calculate compound interest.

    Args:
        principal (float): Initial amount of money.
        rate (float): Interest rate (in decimal).
        time (int): Time period (in years).
        n (int): Number of times interest applied per time period.

    Returns:
        float: Final amount after compound interest.
    """
    return principal * (1 + rate / n) ** (n * time)


def main():
    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))
    n = int(input("Enter the number of times interest applied per time period: "))

    # Calculate compound interest
    final_amount = calculate_compound_interest(principal, rate, time, n)

    # Print result
    print(f"Principal: ${principal:.2f}")
    print(f"Interest Rate: {rate * 100:.2f}%")
    print(f"Time Period: {time} years")
    print(f"Number of times interest applied per time period: {n}")
    print(f"Final Amount: ${final_amount:.2f}")
    print(f"Compound Interest: ${final_amount - principal:.2f}")


if __name__ == "__main__":
    main()
```
