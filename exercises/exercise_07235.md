## Exercise
Write a program that takes two numbers as input and performs addition, subtraction, multiplication, and division operations. The program should then output the results of these operations.

#include <iostream>
using namespace std;

int main() {
    float a, b;
    cout << "Enter first number: ";
    cin >> a;
    cout << "Enter second number: ";
    cin >> b;

    float sum = a + b;
    float diff = a - b;
    float product = a * b;
    float div = a / b;

    cout << "Sum: " << sum << endl;
    cout << "Difference: " << diff << endl;
    cout << "Product: " << product << endl;
    cout << "Division: " << div << endl;

    return 0;
}
