## Exercise
Given two sequences of characters, align them to maximize the number of matching characters. The function should take in two sequences of characters as input and return the aligned sequences.

def sequence_alignment(seq1, seq2):
    """
    Align two sequences of characters to maximize the number of matching characters.

    Args:
        seq1 (str): The first sequence of characters.
        seq2 (str): The second sequence of characters.

    Returns:
        tuple: The aligned sequences.
    """

    # Initialize a 2D matrix to store the maximum number of matches
    dp = [[0] * (len(seq2) + 1) for _ in range(len(seq1) + 1)]

    # Build the 2D matrix using dynamic programming
    for i in range(1, len(seq1) + 1):
        for j in range(1, len(seq2) + 1):
            if seq1[i - 1] == seq2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # Initialize variables to store the aligned sequences
    aligned_seq1 = []
    aligned_seq2 = []

    # Initialize pointers to the bottom-right cell of the 2D matrix
    i, j = len(seq1), len(seq2)

    # Backtrack through the 2D matrix to construct the aligned sequences
    while i > 0 and j > 0:
        if seq1[i - 1] == seq2[j - 1]:
            aligned_seq1.append(seq1[i - 1])
            aligned_seq2.append(seq2[j - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            aligned_seq1.append(seq1[i - 1])
            aligned_seq2.append('-')
            i -= 1
        else:
            aligned_seq1.append('-')
            aligned_seq2.append(seq2[j - 1])
            j -= 1

    # Add any remaining characters from the first sequence
    while i > 0:
        aligned_seq1.append(seq1[i - 1])
        aligned_seq2.append('-')
        i -= 1

    # Add any remaining characters from the second sequence
    while j > 0:
        aligned_seq1.append('-')
        aligned_seq2.append(seq2[j - 1])
        j -= 1

    # Reverse the aligned sequences
    aligned_seq1 = aligned_seq1[::-1]
    aligned_seq2 = aligned_seq2[::-1]

    # Join the aligned sequences into strings
    aligned_seq1 = ''.join(aligned_seq1)
    aligned_seq2 = ''.join(aligned_seq2)

    return aligned_seq1, aligned_seq2

# Example usage
seq1 = "kitten"
seq2 = "sitting"
aligned_seq1, aligned_seq2 = sequence_alignment(seq1, seq2)
print("Aligned Sequence 1:", aligned_seq1)
print("Aligned Sequence 2:", aligned_seq2)
