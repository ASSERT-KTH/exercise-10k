## Exercise
Write a program that converts a numerical grade to a letter grade. The program uses if-else statements to determine the letter grade based on the numerical grade.

## Reference Solution
```python
#include <iostream>

using namespace std;

int main() {
    int grade;
    char letter;
    cout << "Enter your grade: ";
    cin >> grade;

    if (grade >= 90) {
        letter = 'A';
    } else if (grade >= 80 && grade < 90) {
        letter = 'B';
    } else if (grade >= 70 && grade < 80) {
        letter = 'C';
    } else if (grade >= 60 && grade < 70) {
        letter = 'D';
    } else {
        letter = 'F';
    }

    cout << "Your letter grade is: " << letter << endl;

    return 0;
}
```
