## Exercise
Write a program to find the shortest path between two nodes in a given graph.

import sys
import heapq

def dijkstra(graph, start, end):
    """
    Dijkstra's algorithm to find the shortest path between two nodes.

    Args:
    - graph (dict): Adjacency list representation of the graph.
    - start (node): Starting node.
    - end (node): Ending node.

    Returns:
    - shortest_path (list): Shortest path from start to end.
    - shortest_distance (int): Shortest distance from start to end.
    """
    # Initialize distances and previous nodes
    distances = {node: sys.maxsize for node in graph}
    distances[start] = 0
    previous_nodes = {node: None for node in graph}

    # Priority queue to store nodes to be processed
    priority_queue = [(0, start)]

    while priority_queue:
        # Extract the node with the minimum distance
        current_distance, current_node = heapq.heappop(priority_queue)

        # If the current node is the end node, we've found the shortest path
        if current_node == end:
            break

        # Skip if the current distance is greater than the already known distance
        if current_distance > distances[current_node]:
            continue

        # Iterate over the neighbors of the current node
        for neighbor, weight in graph[current_node].items():
            # Calculate the distance to the neighbor
            distance = current_distance + weight

            # If a shorter path to the neighbor is found, update the distances and previous nodes
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                previous_nodes[neighbor] = current_node
                heapq.heappush(priority_queue, (distance, neighbor))

    # Build the shortest path
    shortest_path = []
    current_node = end
    while current_node is not None:
        shortest_path.append(current_node)
        current_node = previous_nodes[current_node]
    shortest_path.reverse()

    return shortest_path, distances[end]

# Example usage
if __name__ == "__main__":
    # Define the graph as an adjacency list
    graph = {
        'A': {'B': 1, 'C': 4},
        'B': {'A': 1, 'C': 2, 'D': 5},
        'C': {'A': 4, 'B': 2, 'D': 1},
        'D': {'B': 5, 'C': 1}
    }

    start_node = 'A'
    end_node = 'D'

    shortest_path, shortest_distance = dijkstra(graph, start_node, end_node)

    print(f"Shortest path from {start_node} to {end_node}: {shortest_path}")
    print(f"Shortest distance from {start_node} to {end_node}: {shortest_distance}")
