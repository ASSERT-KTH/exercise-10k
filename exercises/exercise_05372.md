## Exercise
Write a program to group a list of elements based on a common attribute.

## Reference Solution
```python
from itertools import groupby
from operator import itemgetter

class Element:
    def __init__(self, name, category):
        self.name = name
        self.category = category

    def __repr__(self):
        return f"Element('{self.name}', '{self.category}')"

# List of elements
elements = [
    Element('Apple', 'Fruit'),
    Element('Banana', 'Fruit'),
    Element('Carrot', 'Vegetable'),
    Element('Potato', 'Vegetable'),
    Element('Grapes', 'Fruit'),
    Element('Onion', 'Vegetable'),
]

# Group elements by category
def group_elements(elements):
    # Sort elements by category
    elements.sort(key=lambda x: x.category)
    
    # Group elements by category
    grouped_elements = {}
    for category, group in groupby(elements, key=lambda x: x.category):
        grouped_elements[category] = list(group)
    
    return grouped_elements

# Print grouped elements
def print_grouped_elements(grouped_elements):
    for category, elements in grouped_elements.items():
        print(f"Category: {category}")
        for element in elements:
            print(f"- {element.name}")
        print()

# Main function
def main():
    grouped_elements = group_elements(elements)
    print_grouped_elements(grouped_elements)

if __name__ == "__main__":
    main()
```
