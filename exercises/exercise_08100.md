## Exercise
Write a program that takes a string and a character as input and returns the string with the character removed from the start and end of the string.

use std::io;

fn main() {
    let mut str = String::new();
    let mut char = String::new();

    io::stdin().read_line(&mut str).expect("Failed to read line");
    io::stdin().read_line(&mut char).expect("Failed to read line");

    let str = str.trim();
    let char = char.trim();

    let str_len = str.len();
    let char = char.chars().next().unwrap();

    let mut new_str = String::new();

    for c in str.chars() {
        if c != char {
            new_str.push(c);
        }
    }

    let new_str = new_str.trim_matches(char);

    println!("{}", new_str);
}
