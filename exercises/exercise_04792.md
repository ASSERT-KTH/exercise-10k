## Exercise
Write a program that allows the user to add, remove, and view items in a to-do list.

## Reference Solution
```python
# Initialize an empty list to store to-do items
to_do_list = []

def add_item():
    """Add an item to the to-do list"""
    item = input("Enter an item to add to your to-do list: ")
    to_do_list.append(item)
    print(f"Item '{item}' added to your to-do list.")

def remove_item():
    """Remove an item from the to-do list"""
    if not to_do_list:
        print("Your to-do list is empty. There are no items to remove.")
        return

    print("Your current to-do list:")
    for i, item in enumerate(to_do_list):
        print(f"{i + 1}. {item}")

    try:
        index = int(input("Enter the number of the item to remove: ")) - 1
        if index < 0 or index >= len(to_do_list):
            print("Invalid item number. Please try again.")
        else:
            item = to_do_list.pop(index)
            print(f"Item '{item}' removed from your to-do list.")
    except ValueError:
        print("Invalid input. Please enter a number.")

def view_items():
    """View all items in the to-do list"""
    if not to_do_list:
        print("Your to-do list is empty.")
    else:
        print("Your current to-do list:")
        for i, item in enumerate(to_do_list):
            print(f"{i + 1}. {item}")

def main():
    while True:
        print("\nTo-Do List Menu:")
        print("1. Add an item")
        print("2. Remove an item")
        print("3. View all items")
        print("4. Quit")

        try:
            choice = int(input("Enter your choice: "))
            if choice == 1:
                add_item()
            elif choice == 2:
                remove_item()
            elif choice == 3:
                view_items()
            elif choice == 4:
                print("Goodbye!")
                break
            else:
                print("Invalid choice. Please try again.")
        except ValueError:
            print("Invalid input. Please enter a number.")

if __name__ == "__main__":
    main()
```
