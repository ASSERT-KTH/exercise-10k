## Exercise
Write a program that converts currencies given the exchange rate.

## Reference Solution
```python
def convert_currency(amount, exchange_rate, from_currency, to_currency):
    """
    Converts the given amount of money from one currency to another.

    Args:
        amount (float): The amount of money to convert.
        exchange_rate (float): The exchange rate between the two currencies.
        from_currency (str): The currency to convert from.
        to_currency (str): The currency to convert to.

    Returns:
        float: The converted amount of money.
    """
    converted_amount = amount * exchange_rate
    return converted_amount

def main():
    print("Currency Converter Program")
    print("---------------------------")

    amount = float(input("Enter the amount of money to convert: "))
    exchange_rate = float(input("Enter the exchange rate (1 {} = ? {}): ".format(input("Enter the currency to convert from (e.g. USD, EUR, JPY): "), input("Enter the currency to convert to (e.g. USD, EUR, JPY): "))))
    from_currency = input("Enter the currency to convert from (e.g. USD, EUR, JPY): ")
    to_currency = input("Enter the currency to convert to (e.g. USD, EUR, JPY): ")

    converted_amount = convert_currency(amount, exchange_rate, from_currency, to_currency)

    print("{} {} = {} {}".format(amount, from_currency, converted_amount, to_currency))

if __name__ == "__main__":
    main()
```
