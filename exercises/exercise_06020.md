## Exercise
Write a program that simulates a chatbot that responds to basic user queries

## Reference Solution
```python
import datetime
import random

# Define a dictionary to store chatbot responses
responses = {
    "greet": ["Hello!", "Hi!", "Hey!"],
    "info": ["I'm a basic chatbot. I can tell you the time, perform simple arithmetic operations, and more!", "I'm here to help. What do you need?"],
    "default": ["Sorry, I didn't understand that. Can you please rephrase?", "I'm not sure what you mean. Can you please try again?"]
}

# Define a function to perform arithmetic operations
def perform_arithmetic(operation, num1, num2):
    if operation == "add":
        return num1 + num2
    elif operation == "subtract":
        return num1 - num2
    elif operation == "multiply":
        return num1 * num2
    elif operation == "divide":
        if num2 != 0:
            return num1 / num2
        else:
            return "Error: Division by zero is not allowed."
    else:
        return "Error: Invalid operation."

# Define a function to get the current time
def get_current_time():
    return datetime.datetime.now().strftime("%H:%M:%S")

# Define a function to respond to user queries
def respond_to_query(query):
    if query.lower() == "hello" or query.lower() == "hi":
        return random.choice(responses["greet"])
    elif query.lower() == "what is your name" or query.lower() == "who are you":
        return random.choice(responses["info"])
    elif query.lower() == "what is the time":
        return "The current time is " + get_current_time()
    elif query.lower().startswith("add") or query.lower().startswith("subtract") or query.lower().startswith("multiply") or query.lower().startswith("divide"):
        parts = query.split()
        operation = parts[0].lower()
        num1 = float(parts[1])
        num2 = float(parts[2])
        return str(perform_arithmetic(operation, num1, num2))
    elif query.lower() == "exit":
        return "Goodbye!"
    else:
        return random.choice(responses["default"])

# Main program loop
def main():
    print("Welcome to the chatbot simulator!")
    while True:
        query = input("You: ")
        response = respond_to_query(query)
        print("Chatbot: " + response)
        if query.lower() == "exit":
            break

# Run the main program
if __name__ == "__main__":
    main()
```
