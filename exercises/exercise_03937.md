## Exercise
Write a program that takes two numbers as input and returns the percent decrease from the first number to the second number.

## Reference Solution
```python
def calculate_percent_decrease(original_value, new_value):
    """
    Calculate the percent decrease from the original value to the new value.

    Args:
        original_value (float): The original value.
        new_value (float): The new value.

    Returns:
        float: The percent decrease.
    """
    if original_value == 0:
        raise ZeroDivisionError("Original value cannot be zero")
    percent_decrease = ((original_value - new_value) / original_value) * 100
    return percent_decrease

def main():
    # Get user input
    original_value = float(input("Enter the original value: "))
    new_value = float(input("Enter the new value: "))

    # Calculate percent decrease
    try:
        percent_decrease = calculate_percent_decrease(original_value, new_value)
        print(f"The percent decrease is: {percent_decrease:.2f}%")
    except ZeroDivisionError as e:
        print(e)

if __name__ == "__main__":
    main()
```
