## Exercise
Create a program that takes the principal amount, interest rate, and time as input, and returns the compound interest.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, n=1):
    """
    Calculate compound interest.

    Args:
        principal (float): The initial amount of money.
        rate (float): The interest rate (in decimal).
        time (float): The time the money is invested for (in years).
        n (int, optional): The number of times the interest is compounded per year. Defaults to 1.

    Returns:
        float: The compound interest.
    """
    return principal * ((1 + rate/n) ** (n*time)) - principal

def main():
    # Get user input
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = float(input("Enter the time (in years): "))
    n = int(input("Enter the number of times the interest is compounded per year (default=1): ") or 1)

    # Calculate compound interest
    interest = calculate_compound_interest(principal, rate, time, n)

    # Print the result
    print(f"The compound interest is: {interest:.2f}")

if __name__ == "__main__":
    main()
```
