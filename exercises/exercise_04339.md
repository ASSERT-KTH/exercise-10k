## Exercise
Write a program to rotate an array by 90 degrees.

## Reference Solution
```python
import numpy as np

def rotate_matrix(matrix):
    """
    Rotate a 2D array by 90 degrees clockwise.
    
    Args:
        matrix (list or numpy.ndarray): A 2D array of size n x n.
    
    Returns:
        numpy.ndarray: The rotated matrix.
    """
    # Convert the input to a numpy array
    matrix = np.array(matrix)
    
    # Get the size of the matrix
    n = matrix.shape[0]
    
    # Check if the matrix is square
    if matrix.shape[0] != matrix.shape[1]:
        raise ValueError("The input matrix must be square")
    
    # Rotate the matrix by 90 degrees clockwise
    rotated_matrix = np.rot90(matrix, -1)
    
    return rotated_matrix

def main():
    # Define a sample 2D array
    matrix = [
        [1, 2, 3],
        [4, 5, 6],
        [7, 8, 9]
    ]
    
    # Rotate the matrix
    rotated_matrix = rotate_matrix(matrix)
    
    # Print the original and rotated matrices
    print("Original Matrix:")
    print(np.array(matrix))
    print("\nRotated Matrix:")
    print(rotated_matrix)

if __name__ == "__main__":
    main()
```
