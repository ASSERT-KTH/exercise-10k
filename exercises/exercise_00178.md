## Exercise
Write a program to understand the concept of infinite and finite numbers, and how to check for them.

## Reference Solution
```python
import math

def check_infinity(num):
    """
    Checks if a number is infinite or finite.
    
    Args:
        num (float): The number to check.
    
    Returns:
        str: "Infinite" if the number is infinite, "Finite" otherwise.
    """
    if math.isinf(num):
        return "Infinite"
    else:
        return "Finite"

def main():
    # Example usage:
    print("Checking for infinity:")
    
    # Check for positive infinity
    num1 = float('inf')
    print(f"Is {num1} infinite? {check_infinity(num1)}")
    
    # Check for negative infinity
    num2 = float('-inf')
    print(f"Is {num2} infinite? {check_infinity(num2)}")
    
    # Check for finite number
    num3 = 10.0
    print(f"Is {num3} infinite? {check_infinity(num3)}")
    
    # Check for NaN (Not a Number)
    num4 = float('nan')
    print(f"Is {num4} infinite? {check_infinity(num4)}")

if __name__ == "__main__":
    main()
```
