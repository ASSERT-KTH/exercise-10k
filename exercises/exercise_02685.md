## Exercise
Write a program that takes two numbers and an operator as input and performs the specified operation.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    Perform arithmetic operation based on the input operator.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use for the operation.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not one of '+', '-', '*', '/'.
    """

    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return num1 / num2
    else:
        raise ValueError("Invalid operator. Supported operators are '+', '-', '*', '/'.")


def main():
    num1 = float(input("Enter the first number: "))
    operator = input("Enter the operator (+, -, *, /): ")
    num2 = float(input("Enter the second number: "))

    try:
        result = calculate(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
