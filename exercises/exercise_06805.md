## Exercise
Create a program that takes in a mathematical expression as a string, evaluates it, and then checks if the result matches the expected output. The expression will only contain integer values and the four basic arithmetic operations (+, -, *, /). If the result does not match the expected output, the program should provide a step-by-step breakdown of how it evaluated the expression.

## Reference Solution

# Define the precedence of operators
precedence = {
    '+': 1,
    '-': 1,
    '*': 2,
    '/': 2
}

def apply_operator(operators, values):
    """
    Apply the operator to the top two values in the values stack.
    
    Args:
    operators (list): A list of operators.
    values (list): A list of values.
    
    Returns:
    None
    """
    op = operators.pop()
    right = values.pop()
    left = values.pop()
    
    if op == '+':
        values.append(left + right)
    elif op == '-':
        values.append(left - right)
    elif op == '*':
        values.append(left * right)
    elif op == '/':
        values.append(left / right)

def evaluate_expression(expression):
    """
    Evaluate the mathematical expression.
    
    Args:
    expression (str): A string representing the mathematical expression.
    
    Returns:
    list: A list containing the result and the step-by-step breakdown.
    """
    # Split the expression into tokens (operators and operands)
    tokens = []
    current_token = ''
    for char in expression:
        if char.isdigit():
            current_token += char
        else:
            if current_token:
                tokens.append(int(current_token))
                current_token = ''
            tokens.append(char)
    if current_token:
        tokens.append(int(current_token))
    
    # Initialize the operators and values stacks
    operators = []
    values = []
    steps = []
    
    # Iterate over the tokens
    for token in tokens:
        if isinstance(token, int):
            # If the token is an operand, push it to the values stack
            values.append(token)
            steps.append(f"Push {token} to the values stack: {values}")
        else:
            # If the token is an operator, pop operators with higher or equal precedence from the operators stack
            while operators and precedence[operators[-1]] >= precedence[token]:
                apply_operator(operators, values)
                steps.append(f"Apply operator: {values}")
            # Push the token to the operators stack
            operators.append(token)
            steps.append(f"Push {token} to the operators stack: {operators}")
    
    # Apply any remaining operators
    while operators:
        apply_operator(operators, values)
        steps.append(f"Apply operator: {values}")
    
    # The final result is the only element left in the values stack
    result = values[0]
    
    return result, steps

def check_result(expression, expected_output):
    """
    Check if the result of the expression matches the expected output.
    
    Args:
    expression (str): A string representing the mathematical expression.
    expected_output (float): The expected output of the expression.
    
    Returns:
    None
    """
    result, steps = evaluate_expression(expression)
    
    if result == expected_output:
        print(f"The result of the expression '{expression}' is {result}, which matches the expected output.")
    else:
        print(f"The result of the expression '{expression}' is {result}, which does not match the expected output {expected_output}.")
        print("Step-by-step breakdown:")
        for step in steps:
            print(step)

# Test the program
expression = "2+3*4"
expected_output = 14.0
check_result(expression, expected_output)
