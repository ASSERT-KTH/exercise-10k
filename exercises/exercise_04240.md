## Exercise
Write a program to calculate the area of a triangle.

## Reference Solution
```python
import math

def calculate_area_using_sides(a, b, c):
    """
    Calculate the area of a triangle using Heron's formula.

    Args:
        a (float): Length of side a.
        b (float): Length of side b.
        c (float): Length of side c.

    Returns:
        float: Area of the triangle.
    """
    s = (a + b + c) / 2
    area = math.sqrt(s * (s - a) * (s - b) * (s - c))
    return area

def calculate_area_using_base_and_height(base, height):
    """
    Calculate the area of a triangle using its base and height.

    Args:
        base (float): Length of the base.
        height (float): Height of the triangle.

    Returns:
        float: Area of the triangle.
    """
    area = 0.5 * base * height
    return area

def main():
    print("Triangle Area Calculator")
    print("1. Calculate area using sides")
    print("2. Calculate area using base and height")
    
    choice = input("Enter your choice (1/2): ")
    
    if choice == "1":
        a = float(input("Enter length of side a: "))
        b = float(input("Enter length of side b: "))
        c = float(input("Enter length of side c: "))
        area = calculate_area_using_sides(a, b, c)
        print(f"The area of the triangle is: {area:.2f}")
    elif choice == "2":
        base = float(input("Enter length of the base: "))
        height = float(input("Enter height of the triangle: "))
        area = calculate_area_using_base_and_height(base, height)
        print(f"The area of the triangle is: {area:.2f}")
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
