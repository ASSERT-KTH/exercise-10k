## Exercise
Create a program that takes in a student's score and checks if they passed or failed using conditional statements.

## Reference Solution
```python
def check_score(score):
    """
    Checks if a student passed or failed based on their score.

    Args:
        score (float): The student's score.

    Returns:
        str: A message indicating whether the student passed or failed.
    """
    if score >= 60:
        return "Passed"
    else:
        return "Failed"

def main():
    # Get the student's score from the user
    score = float(input("Enter the student's score (out of 100): "))

    # Check if the score is valid (between 0 and 100)
    if score < 0 or score > 100:
        print("Invalid score. Please enter a score between 0 and 100.")
    else:
        # Check if the student passed or failed
        result = check_score(score)
        print(f"The student {result} with a score of {score}.")

if __name__ == "__main__":
    main()
```
