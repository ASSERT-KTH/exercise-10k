## Exercise
Read a positive integer from the user and check if it is an Armstrong number

## Reference Solution
```python
def is_armstrong(num):
    """
    Check if a number is an Armstrong number.
    
    Args:
    num (int): The number to check.
    
    Returns:
    bool: True if the number is an Armstrong number, False otherwise.
    """
    # Convert the number to a string to easily get the number of digits
    num_str = str(num)
    num_digits = len(num_str)
    
    # Calculate the sum of the cubes of the digits
    sum_cubes = sum(int(digit) ** num_digits for digit in num_str)
    
    # Check if the sum of the cubes is equal to the original number
    return sum_cubes == num

def main():
    # Read a positive integer from the user
    while True:
        try:
            num = int(input("Enter a positive integer: "))
            if num > 0:
                break
            else:
                print("Please enter a positive integer.")
        except ValueError:
            print("Invalid input. Please enter an integer.")
    
    # Check if the number is an Armstrong number
    if is_armstrong(num):
        print(f"{num} is an Armstrong number.")
    else:
        print(f"{num} is not an Armstrong number.")

if __name__ == "__main__":
    main()
```
