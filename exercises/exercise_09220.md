## Exercise
Create a program that uses functions to perform basic arithmetic operations such as addition, subtraction, multiplication, and division. The program should take two numbers and an operation as input and output the result. You can assume that the input will always be valid.

## Reference Solution
```python
#include <iostream>
using namespace std;

int add(int num1, int num2) {
    return num1 + num2;
}

int subtract(int num1, int num2) {
    return num1 - num2;
}

int multiply(int num1, int num2) {
    return num1 * num2;
}

int divide(int num1, int num2) {
    return num1 / num2;
}

int main() {
    int num1, num2, addResult, subtractResult, multiplyResult, divideResult;
    char operation;

    cout << "Enter the first number: ";
    cin >> num1;

    cout << "Enter the second number: ";
    cin >> num2;

    cout << "Enter the operation (+, -, *, /): ";
    cin >> operation;

    switch (operation) {
        case '+':
            addResult = add(num1, num2);
            cout << "Result: " << addResult << endl;
            break;
        case '-':
            subtractResult = subtract(num1, num2);
            cout << "Result: " << subtractResult << endl;
            break;
        case '*':
            multiplyResult = multiply(num1, num2);
            cout << "Result: " << multiplyResult << endl;
            break;
        case '/':
            divideResult = divide(num1, num2);
            cout << "Result: " << divideResult << endl;
            break;
        default:
            cout << "Invalid operation. Please try again." << endl;
            break;
    }

    return 0;
}
```
