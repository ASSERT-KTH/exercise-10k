## Exercise
Write a function to find the missing number in an array of consecutive integers.

## Reference Solution

def find_missing_number(nums):
    """
    Find the missing number in an array of consecutive integers.

    Args:
        nums (list): A list of consecutive integers with one number missing.

    Returns:
        int: The missing number in the sequence.
    """
    n = len(nums) + 1  # The total number of elements in the sequence
    start = min(nums)  # The starting number of the sequence
    end = max(nums)  # The ending number of the sequence

    # Calculate the sum of the complete sequence using the formula for the sum of an arithmetic series
    total_sum = (n * (start + end)) // 2

    # Calculate the sum of the given sequence
    given_sum = sum(nums)

    # The missing number is the difference between the total sum and the given sum
    missing_number = total_sum - given_sum

    return missing_number

# Example usage
if __name__ == "__main__":
    nums = [1, 2, 3, 5]
    missing_number = find_missing_number(nums)
    print(f"The missing number is: {missing_number}")
