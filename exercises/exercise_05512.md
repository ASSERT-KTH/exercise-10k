Write a program to solve the 0/1 knapsack problem using dynamic programming.

def knapsack(W, wt, val, n):
    """
    Solves the 0/1 knapsack problem using dynamic programming.
    
    Parameters:
    W (int): Maximum capacity of the knapsack.
    wt (list): Weights of the items.
    val (list): Values of the items.
    n (int): Number of items.
    
    Returns:
    int: Maximum value that can be put in the knapsack.
    """
    # Create a 2D array to store the maximum value at each subproblem
    K = [[0 for w in range(W + 1)] for i in range(n + 1)]

    # Fill the 2D array using dynamic programming
    for i in range(n + 1):
        for w in range(W + 1):
            if i == 0 or w == 0:
                K[i][w] = 0
            elif wt[i - 1] <= w:
                K[i][w] = max(val[i - 1] + K[i - 1][w - wt[i - 1]], K[i - 1][w])
            else:
                K[i][w] = K[i - 1][w]

    # The maximum value is stored in the bottom-right cell of the 2D array
    return K[n][W]


def print_knapsack_items(W, wt, val, n):
    """

    Parameters:

    val (list): Values of the items.
    n (int): Number of items.
    """
    # Create a 2D array to store the maximum value at each subproblem
    K = [[0 for w in range(W + 1)] for i in range(n + 1)]

    # Fill the 2D array using dynamic programming
    for i in range(n + 1):
        for w in range(W + 1):
            if i == 0 or w == 0:
                K[i][w] = 0
            elif wt[i - 1] <= w:
                K[i][w] = max(val[i - 1] + K[i - 1][w - wt[i - 1]], K[i - 1][w])
            else:
                K[i][w] = K[i - 1][w]

    # Print the items included in the optimal solution
    w = W
    for i in range(n, 0, -1):
        if K[i][w] != K[i - 1][w]:
            print(f"Item {i}: weight = {wt[i - 1]}, value = {val[i - 1]}")
            w -= wt[i - 1]
