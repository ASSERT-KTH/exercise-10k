## Exercise
Create a program to count the occurrences of vowels in a given string.

## Reference Solution
```python
def count_vowels(input_string):
    """
    Counts the occurrences of vowels in a given string.

    Args:
        input_string (str): The input string to count vowels from.

    Returns:
        dict: A dictionary with vowel characters as keys and their counts as values.
    """
    vowels = 'aeiouAEIOU'
    vowel_counts = {vowel: 0 for vowel in 'aeiouAEIOU'}

    for char in input_string:
        if char in vowels:
            vowel_counts[char] += 1

    return vowel_counts


def main():
    input_string = input("Enter a string: ")
    vowel_counts = count_vowels(input_string)

    print("Vowel Counts:")
    for vowel, count in vowel_counts.items():
        if count > 0:
            print(f"{vowel}: {count}")


if __name__ == "__main__":
    main()
```
