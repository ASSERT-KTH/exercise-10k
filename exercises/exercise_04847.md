## Exercise
Write a program that takes an array as input, and returns the frequency of each element.

## Reference Solution
```python
def element_frequency(arr):
    """
    Returns a dictionary with the frequency of each element in the array.

    Args:
        arr (list): The input array.

    Returns:
        dict: A dictionary where the keys are the unique elements in the array
            and the values are their corresponding frequencies.
    """
    frequency = {}
    for element in arr:
        if element in frequency:
            frequency[element] += 1
        else:
            frequency[element] = 1
    return frequency

def main():
    # Example usage:
    arr = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
    frequency = element_frequency(arr)
    print("Element Frequency:")
    for element, count in frequency.items():
        print(f"{element}: {count}")

if __name__ == "__main__":
    main()
```
