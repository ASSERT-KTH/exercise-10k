## Exercise
Write a program that multiplies two matrices.

```python
import numpy as np

def multiply_matrices(mat1, mat2):
    """
    Multiply two matrices.

    Args:
    mat1 (list of lists): The first matrix.
    mat2 (list of lists): The second matrix.

    Returns:
    list of lists: The product of the two matrices.

    Raises:
    ValueError: If the number of columns in the first matrix does not match the number of rows in the second matrix.
    """

    # Get the dimensions of the matrices
    rows_mat1 = len(mat1)
    cols_mat1 = len(mat1[0])
    rows_mat2 = len(mat2)
    cols_mat2 = len(mat2[0])

    # Check if the matrices can be multiplied
    if cols_mat1 != rows_mat2:
        raise ValueError("The number of columns in the first matrix must match the number of rows in the second matrix.")

    # Create a result matrix filled with zeros
    result = [[0 for _ in range(cols_mat2)] for _ in range(rows_mat1)]

    # Perform the matrix multiplication
    for i in range(rows_mat1):
        for j in range(cols_mat2):
            for k in range(cols_mat1):
                result[i][j] += mat1[i][k] * mat2[k][j]

    return result

def print_matrix(matrix):
    """
    Print a matrix.

    Args:
    matrix (list of lists): The matrix to print.
    """
    for row in matrix:
        print(row)

def main():
    # Example usage
    mat1 = [[1, 2, 3], [4, 5, 6]]
    mat2 = [[7, 8], [9, 10], [11, 12]]

    print("Matrix 1:")
    print_matrix(mat1)
    print("\nMatrix 2:")
    print_matrix(mat2)

    try:
        result = multiply_matrices(mat1, mat2)
        print("\nResult:")
        print_matrix(result)
    except ValueError as e:
        print("\nError:", e)

if __name__ == "__main__":
    main()
```
