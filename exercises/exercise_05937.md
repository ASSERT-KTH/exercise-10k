## Exercise
Write a program that takes a string as input and finds the longest word in the string.

## Reference Solution
```python
def find_longest_word(input_string):
    """
    Finds the longest word in a given string.

    Args:
        input_string (str): The input string to search for the longest word.

    Returns:
        str: The longest word in the input string.
    """
    # Split the input string into words
    words = input_string.split()

    # Initialize the longest word and its length
    longest_word = max(words, key=len)

    return longest_word

def main():
    # Example usage
    input_string = "This is a sample string with multiple words"
    longest_word = find_longest_word(input_string)
    print(f"The longest word is: {longest_word}")

if __name__ == "__main__":
    main()
```
