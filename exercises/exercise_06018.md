## Exercise
Write a program that simulates a conversation between two people

## Reference Solution
```python
import random

# Define a dictionary of responses for each person
responses = {
    'alice': {
        'hello': ['Hi, how are you?', 'Hello! What\'s up?'],
        'how are you': ['I\'m good, thanks. How about you?', 'I\'m doing well, thanks.'],
        'what\'s up': ['Not much, just hanging out.', 'Just got back from a walk.'],
        'default': ['I didn\'t understand that.', 'Can you repeat that?']
    },
    'bob': {
        'hello': ['Hey, what\'s up?', 'Hi! How\'s it going?'],
        'how are you': ['I\'m good, thanks. Just got back from work.', 'I\'m doing alright, thanks.'],
        'what\'s up': ['Just got back from the gym.', 'Not much, just watching TV.'],
        'default': ['I didn\'t catch that.', 'Can you repeat that?']
    }
}

# Define a function to get a response from a person
def get_response(person, message):
    message = message.lower()
    for key in responses[person]:
        if key in message:
            return random.choice(responses[person][key])
    return random.choice(responses[person]['default'])

# Define a function to simulate a conversation
def simulate_conversation():
    print("Conversation Simulator")
    print("------------------------")
    alice_turn = True
    while True:
        if alice_turn:
            print("Alice's turn:")
            message = input("Enter a message: ")
            print("Bob:", get_response('bob', message))
        else:
            print("Bob's turn:")
            message = input("Enter a message: ")
            print("Alice:", get_response('alice', message))
        alice_turn = not alice_turn

# Run the conversation simulator
if __name__ == "__main__":
    simulate_conversation()
```
