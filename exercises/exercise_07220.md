## Exercise
Create a program that demonstrates the use of nested loops using a for loop.

## Reference Solution
```python
# Define a function to print a matrix
def print_matrix(rows, cols):
    """
    Prints a matrix with rows and cols as input.
    """
    for i in range(rows):
        for j in range(cols):
            print(f"({i}, {j})", end=" ")
        print()

# Define a function to print a multiplication table
def print_multiplication_table(size):
    """
    Prints a multiplication table of size x size.
    """
    for i in range(1, size + 1):
        for j in range(1, size + 1):
            print(f"{i} x {j} = {i * j}", end="   ")
        print()

# Main function
def main():
    print("Matrix Example:")
    print_matrix(3, 3)
    print("\nMultiplication Table Example:")
    print_multiplication_table(5)

if __name__ == "__main__":
    main()
```
