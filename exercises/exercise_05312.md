## Exercise
Write a function that inherits behavior from another function

## Reference Solution
```python
def inherit_behavior(original_function):
    def decorator(new_function):
        def wrapper(*args, **kwargs):
            # Call the original function
            original_result = original_function(*args, **kwargs)
            # Call the new function
            new_result = new_function(*args, **kwargs)
            # Combine the results
            return original_result, new_result
        return wrapper
    return decorator

# Original function
def greet(name):
    return f"Hello, {name}!"

# New function that "inherits" behavior from the original function
@inherit_behavior(greet)
def greet_with_message(name, message):
    return f"{message} {name}!"

# Usage
result = greet_with_message("John", "Hi")
print(result)  # Output: ('Hello, John!', 'Hi John!')
```

class Function:
    def __init__(self, func):
        self.func = func

    def __call__(self, *args, **kwargs):
        return self.func(*args, **kwargs)

class InheritedFunction(Function):
    def __init__(self, original_func, new_func):
        self.original_func = original_func
        self.new_func = new_func

    def __call__(self, *args, **kwargs):
        original_result = self.original_func(*args, **kwargs)
        new_result = self.new_func(*args, **kwargs)
        return original_result, new_result

# Original function
def greet(name):
    return f"Hello, {name}!"

# New function that "inherits" behavior from the original function
def greet_with_message(name, message):
    return f"{message} {name}!"

# Usage
inherited_func = InheritedFunction(greet, lambda name, message: f"{message} {name}!")
result = inherited_func("John", "Hi")
print(result)  # Output: ('Hello, John!', 'Hi John!')
