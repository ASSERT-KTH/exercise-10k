## Exercise
Create a program that uses a for loop to scramble the letters of a given string.

## Reference Solution
```python
import random

def scramble_string(input_str):
    """
    Scramble the letters of a given string.
    
    Args:
        input_str (str): The input string to scramble.
    
    Returns:
        str: The scrambled string.
    """
    # Convert the string to a list of characters
    char_list = list(input_str)
    
    # Scramble the list of characters
    for i in range(len(char_list)):
        # Generate a random index to swap with
        j = random.randint(0, len(char_list) - 1)
        
        # Swap the characters at indices i and j
        char_list[i], char_list[j] = char_list[j], char_list[i]
    
    # Join the scrambled list of characters back into a string
    scrambled_str = "".join(char_list)
    
    return scrambled_str

# Example usage:
def main():
    input_str = "Hello, World!"
    print(f"Original string: {input_str}")
    scrambled_str = scramble_string(input_str)
    print(f"Scrambled string: {scrambled_str}")

if __name__ == "__main__":
    main()
```
