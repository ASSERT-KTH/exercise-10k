## Exercise
Write a program to generate all permutations of a given list.

## Reference Solution
```python
def generate_permutations(lst):
    """
    Generate all permutations of a given list.

    Args:
        lst (list): The input list.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    # Base case: if the list is empty or contains only one element
    if len(lst) <= 1:
        return [lst]

    permutations = []
    for i, elem in enumerate(lst):
        # Choose the current element as the first element of the permutation
        remaining_list = lst[:i] + lst[i+1:]
        for p in generate_permutations(remaining_list):
            permutations.append([elem] + p)

    return permutations

def main():
    # Example usage
    input_list = [1, 2, 3]
    permutations = generate_permutations(input_list)
    print("Permutations of", input_list, ":")
    for i, p in enumerate(permutations):
        print(f"Permutation {i+1}: {p}")

if __name__ == "__main__":
    main()
```
