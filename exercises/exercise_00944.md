## Exercise
Write a program that lets the user guess a random number between 1 and 10 using a while loop.

## Reference Solution
```python
import random

def guess_number():
    # Generate a random number between 1 and 10
    number_to_guess = random.randint(1, 10)
    guess = None
    attempts = 0

    # Continue guessing until the correct number is guessed
    while guess != number_to_guess:
        # Ask the user for their guess
        try:
            guess = int(input("Guess a number between 1 and 10: "))
        except ValueError:
            print("Invalid input. Please enter a number.")
            continue

        # Check if the guess is in the correct range
        if guess < 1 or guess > 10:
            print("Please guess a number between 1 and 10.")
            continue

        # Increment the number of attempts
        attempts += 1

        # Check if the guess is correct
        if guess < number_to_guess:
            print("Too low! Try again.")
        elif guess > number_to_guess:
            print("Too high! Try again.")

    # Print a success message with the number of attempts
    print(f"Congratulations! You guessed the number {number_to_guess} in {attempts} attempts.")

# Example usage
if __name__ == "__main__":
    guess_number()
```
