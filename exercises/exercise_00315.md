## Exercise
Write a function that rotates the elements of a given array to the left or right by a specified number of positions.

## Reference Solution
```python
def rotate_array(arr, num_positions, direction):
    """
    Rotate the elements of a given array to the left or right by a specified number of positions.

    Args:
        arr (list): The input array.
        num_positions (int): The number of positions to rotate the array.
        direction (str): The direction of rotation ('left' or 'right').

    Returns:
        list: The rotated array.
    """
    num_positions = num_positions % len(arr)  # Handle cases where num_positions is greater than len(arr)
    if direction == 'left':
        return arr[num_positions:] + arr[:num_positions]
    elif direction == 'right':
        return arr[-num_positions:] + arr[:-num_positions]
    else:
        raise ValueError("Invalid direction. Please specify 'left' or 'right'.")

# Example usage:
if __name__ == "__main__":
    arr = [1, 2, 3, 4, 5]
    num_positions = 2
    direction = 'left'
    print(f"Original array: {arr}")
    print(f"Rotated array ({direction} by {num_positions} positions): {rotate_array(arr, num_positions, direction)}")

    direction = 'right'
    print(f"Rotated array ({direction} by {num_positions} positions): {rotate_array(arr, num_positions, direction)}")
```
