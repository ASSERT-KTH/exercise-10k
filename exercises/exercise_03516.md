## Exercise
Write a program that declares and operates on lists.

## Reference Solution
```python
# List Operations Program

# Declare a list
def declare_list():
    """Declares a list with initial values."""
    my_list = [1, 2, 3, 4, 5]
    return my_list

# Append an element to the list
def append_element(my_list, element):
    """Appends an element to the end of the list."""
    my_list.append(element)
    return my_list

# Insert an element at a specific index
def insert_element(my_list, index, element):
    """Inserts an element at the specified index."""
    my_list.insert(index, element)
    return my_list

# Remove the first occurrence of an element
def remove_element(my_list, element):
    """Removes the first occurrence of the specified element."""
    if element in my_list:
        my_list.remove(element)
    return my_list

# Print the list
def print_list(my_list):
    """Prints the list."""
    print(my_list)

# Main function
def main():
    my_list = declare_list()
    print("Initial List:")
    print_list(my_list)

    # Append an element
    my_list = append_element(my_list, 6)
    print("\nList after appending 6:")
    print_list(my_list)

    # Insert an element
    my_list = insert_element(my_list, 2, 10)
    print("\nList after inserting 10 at index 2:")
    print_list(my_list)

    # Remove an element
    my_list = remove_element(my_list, 4)
    print("\nList after removing 4:")
    print_list(my_list)

if __name__ == "__main__":
    main()
```
