## Exercise
Create a program that takes in a day number and outputs the corresponding day of the week using conditional statements.

## Reference Solution
```python
def find_day(day_number):
    """
    This function takes in a day number and returns the corresponding day of the week.

    Parameters:
    day_number (int): The day number (1-7) where 1 is Sunday and 7 is Saturday.

    Returns:
    str: The day of the week corresponding to the input day number.
    """
    
    # Check if the input day number is within the valid range (1-7)
    if day_number < 1 or day_number > 7:
        return "Invalid day number. Please enter a number between 1 and 7."
    
    # Use conditional statements to determine the day of the week
    if day_number == 1:
        return "Sunday"
    elif day_number == 2:
        return "Monday"
    elif day_number == 3:
        return "Tuesday"
    elif day_number == 4:
        return "Wednesday"
    elif day_number == 5:
        return "Thursday"
    elif day_number == 6:
        return "Friday"
    else:
        return "Saturday"
```

```python
def find_day(day_number):
    """
    This function takes in a day number and returns the corresponding day of the week.

    Parameters:
    day_number (int): The day number (1-7) where 1 is Sunday and 7 is Saturday.

    Returns:
    str: The day of the week corresponding to the input day number.
    """
    
    # Define a list of days of the week
    days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"]
    
    # Check if the input day number is within the valid range (1-7)
    if day_number < 1 or day_number > 7:
        return "Invalid day number. Please enter a number between 1 and 7."
    
    # Return the day of the week corresponding to the input day number
    return days[day_number - 1]

# Test the function
print(find_day(1))  # Sunday
```
