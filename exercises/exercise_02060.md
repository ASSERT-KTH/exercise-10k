## Exercise
Write a function to reduce a dictionary to a single value (e.g., sum, average).

## Reference Solution
```python
import statistics

def reduce_dictionary(dictionary, method):
    """
    Reduces a dictionary to a single value based on the specified method.

    Args:
        dictionary (dict): The input dictionary.
        method (str): The reduction method. Supported methods are 'sum', 'average', 'min', 'max', and 'median'.

    Returns:
        The reduced value.

    Raises:
        ValueError: If the method is not supported.
    """

    # Check if the dictionary is empty
    if not dictionary:
        raise ValueError("Dictionary is empty")

    # Extract values from the dictionary
    values = dictionary.values()

    # Apply the reduction method
    if method == 'sum':
        return sum(values)
    elif method == 'average':
        return sum(values) / len(values)
    elif method == 'min':
        return min(values)
    elif method == 'max':
        return max(values)
    elif method == 'median':
        return statistics.median(values)
    else:
        raise ValueError("Unsupported method. Supported methods are 'sum', 'average', 'min', 'max', and 'median'.")

# Example usage
dictionary = {'a': 1, 'b': 2, 'c': 3, 'd': 4, 'e': 5}
print(reduce_dictionary(dictionary, 'sum'))  # Output: 15
print(reduce_dictionary(dictionary, 'average'))  # Output: 3.0
print(reduce_dictionary(dictionary, 'min'))  # Output: 1
print(reduce_dictionary(dictionary, 'max'))  # Output: 5
print(reduce_dictionary(dictionary, 'median'))  # Output: 3
```
