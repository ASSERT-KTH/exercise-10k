## Exercise
Create a program that simulates the stock market, using conditional statements to determine the stock prices

## Reference Solution
```python
import random

class StockMarket:
    def __init__(self):
        self.apple_price = 100.0
        self.google_price = 1500.0
        self.microsoft_price = 200.0
        self.market_open = True

    def simulate_day(self):
        event = random.randint(1, 5)

        if event == 1:  # Economic downturn
            self.apple_price *= 0.9
            self.google_price *= 0.9
            self.microsoft_price *= 0.9
            print("Economic downturn: Stock prices decreased by 10%")
        elif event == 2:  # Tech boom
            self.apple_price *= 1.1
            self.google_price *= 1.1
            self.microsoft_price *= 1.1
            print("Tech boom: Stock prices increased by 10%")
        elif event == 3:  # Apple innovation
            self.apple_price *= 1.2
            print("Apple innovation: Apple stock price increased by 20%")
        elif event == 4:  # Google scandal
            self.google_price *= 0.8
            print("Google scandal: Google stock price decreased by 20%")
        elif event == 5:  # Microsoft expansion
            self.microsoft_price *= 1.1
            print("Microsoft expansion: Microsoft stock price increased by 10%")

        print(f"Apple stock price: ${self.apple_price:.2f}")
        print(f"Google stock price: ${self.google_price:.2f}")
        print(f"Microsoft stock price: ${self.microsoft_price:.2f}")

    def buy_stock(self, company, amount):
        if not self.market_open:
            print("Market is closed. Cannot buy stock.")
            return

        if company == "Apple":
            if self.apple_price * amount > 10000:
                print("Insufficient funds to buy Apple stock.")
                return
            print(f"Bought {amount} shares of Apple stock for ${self.apple_price * amount:.2f}")
        elif company == "Google":
            if self.google_price * amount > 10000:
                print("Insufficient funds to buy Google stock.")
                return
            print(f"Bought {amount} shares of Google stock for ${self.google_price * amount:.2f}")
        elif company == "Microsoft":
            if self.microsoft_price * amount > 10000:
                print("Insufficient funds to buy Microsoft stock.")
                return
            print(f"Bought {amount} shares of Microsoft stock for ${self.microsoft_price * amount:.2f}")

    def sell_stock(self, company, amount):
        if not self.market_open:
            print("Market is closed. Cannot sell stock.")
            return

        if company == "Apple":
            print(f"Sold {amount} shares of Apple stock for ${self.apple_price * amount:.2f}")
        elif company == "Google":
            print(f"Sold {amount} shares of Google stock for ${self.google_price * amount:.2f}")
        elif company == "Microsoft":
            print(f"Sold {amount} shares of Microsoft stock for ${self.microsoft_price * amount:.2f}")

def main():
    market = StockMarket()
    while True:
        print("\n1. Simulate day")
        print("2. Buy stock")
        print("3. Sell stock")
        print("4. Exit")
        choice = input("Enter choice: ")

        if choice == "1":
            market.simulate_day()
        elif choice == "2":
            company = input("Enter company (Apple, Google, Microsoft): ")
            amount = float(input("Enter amount of shares to buy: "))
            market.buy_stock(company, amount)
        elif choice == "3":
            company = input("Enter company (Apple, Google, Microsoft): ")
            amount = float(input("Enter amount of shares to sell: "))
            market.sell_stock(company, amount)
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
