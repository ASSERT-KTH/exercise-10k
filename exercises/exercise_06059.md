## Exercise
Create a program that simulates a restaurant menu using conditional statements, allowing users to order food and drinks.

## Reference Solution
```python
# Define a dictionary to store menu items and prices
menu = {
    "food": {
        "burger": 10.99,
        "pizza": 12.99,
        "salad": 8.99
    },
    "drinks": {
        "soda": 2.99,
        "juice": 3.99,
        "water": 1.99
    }
}

def display_menu():
    """Display the restaurant menu"""
    print("Restaurant Menu:")
    print("-----------------")
    print("Food:")
    for item, price in menu["food"].items():
        print(f"{item.capitalize()}: ${price:.2f}")
    print("Drinks:")
    for item, price in menu["drinks"].items():
        print(f"{item.capitalize()}: ${price:.2f}")

def order_food():
    """Allow user to order food"""
    print("Order Food:")
    print("-----------")
    for item, price in menu["food"].items():
        print(f"{item.capitalize()}: ${price:.2f}")
    choice = input("Enter the name of the food item: ").lower()
    if choice in menu["food"]:
        return choice, menu["food"][choice]
    else:
        print("Invalid choice. Please try again.")
        return order_food()

def order_drink():
    """Allow user to order a drink"""
    print("Order Drink:")
    print("------------")
    for item, price in menu["drinks"].items():
        print(f"{item.capitalize()}: ${price:.2f}")
    choice = input("Enter the name of the drink item: ").lower()
    if choice in menu["drinks"]:
        return choice, menu["drinks"][choice]
    else:
        print("Invalid choice. Please try again.")
        return order_drink()

def main():
    """Main program loop"""
    order = []
    while True:
        print("\nOptions:")
        print("1. Display Menu")
        print("2. Order Food")
        print("3. Order Drink")
        print("4. View Order")
        print("5. Checkout")
        choice = input("Enter your choice: ")
        if choice == "1":
            display_menu()
        elif choice == "2":
            food, price = order_food()
            order.append((food, price))
            print(f"Added {food} to your order.")
        elif choice == "3":
            drink, price = order_drink()
            order.append((drink, price))
            print(f"Added {drink} to your order.")
        elif choice == "4":
            print("\nYour Order:")
            for item, price in order:
                print(f"{item.capitalize()}: ${price:.2f}")
        elif choice == "5":
            print("\nCheckout:")
            total = sum(price for _, price in order)
            print(f"Total: ${total:.2f}")
            print("Thank you for dining with us!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
