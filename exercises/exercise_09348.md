// Demonstrate Temporal Dead Zone (TDZ) in JavaScript

// Example 1: Accessing a variable before its declaration
try {
  console.log(x); // This will throw a ReferenceError
  let x = 10;
} catch (error) {
  console.error("Error:", error.message);
}

// Example 2: Accessing a variable before its declaration in a block scope
try {
  if (true) {
    console.log(y); // This will throw a ReferenceError
    let y = 20;
  }
} catch (error) {
  console.error("Error:", error.message);
}

// Example 3: Variable is accessible after its declaration
let z = 30;
console.log(z); // This will print 30

// Example 4: Variable is not accessible in the TDZ
try {
  if (true) {
    console.log(w); // This will throw a ReferenceError
    const w = 40;
  }
} catch (error) {
  console.error("Error:", error.message);
}
