## Exercise
Write a program that allows the user to create and manage a to-do list. The user should be able to add tasks, remove tasks, and view the list.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self):
        """Add a task to the to-do list."""
        task = input("Enter a task: ")
        self.tasks.append(task)
        print(f"Task '{task}' added to the list.")

    def remove_task(self):
        """Remove a task from the to-do list."""
        if not self.tasks:
            print("The list is empty.")
            return

        print("Current tasks:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")

        while True:
            try:
                task_number = int(input("Enter the task number to remove: "))
                if 1 <= task_number <= len(self.tasks):
                    task = self.tasks.pop(task_number - 1)
                    print(f"Task '{task}' removed from the list.")
                    break
                else:
                    print("Invalid task number. Please try again.")
            except ValueError:
                print("Invalid input. Please enter a task number.")

    def view_list(self):
        """Print the current to-do list."""
        if not self.tasks:
            print("The list is empty.")
        else:
            print("Current tasks:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")


def main():
    todo = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add a task")
        print("2. Remove a task")
        print("3. View the list")
        print("4. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            todo.add_task()
        elif choice == "2":
            todo.remove_task()
        elif choice == "3":
            todo.view_list()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please try again.")


if __name__ == "__main__":
    main()
```
