## Exercise
Write a function to find the difference between two dictionaries.

def dict_diff(dict1, dict2):
    """
    Calculate the difference between two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A dictionary containing the keys and values that are unique to either dictionary.
    """
    diff = {}
    all_keys = set(dict1.keys()) | set(dict2.keys())

    for key in all_keys:
        if key not in dict1:
            diff[key] = dict2[key]
        elif key not in dict2:
            diff[key] = dict1[key]
        elif dict1[key] != dict2[key]:
            diff[key] = {'dict1': dict1[key], 'dict2': dict2[key]}

    return diff

# Example usage

    dict1 = {'a': 1, 'b': 2, 'c': 3}
    dict2 = {'b': 2, 'c': 4, 'd': 5}

    print(dict_diff(dict1, dict2))
