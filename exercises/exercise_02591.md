## Exercise
Write a program to find the second smallest number in a given list of numbers.

## Reference Solution
```python
def find_second_smallest(numbers):
    """
    Finds the second smallest number in a given list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The second smallest number in the list. If the list has less than two unique elements, returns None.
    """
    # Remove duplicates and sort the list
    unique_numbers = sorted(set(numbers))
    
    # Check if the list has at least two unique elements
    if len(unique_numbers) < 2:
        return None
    
    # Return the second smallest number
    return unique_numbers[1]

# Example usage:
numbers = [5, 2, 8, 12, 3, 2, 5]
second_smallest = find_second_smallest(numbers)

if second_smallest is not None:
    print(f"The second smallest number is: {second_smallest}")
else:
    print("The list has less than two unique elements.")
```
