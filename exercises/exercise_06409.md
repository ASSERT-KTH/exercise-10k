## Exercise
Write a program that takes two numbers and an operator as input and performs the specified operation. Use if/else statements to determine the operation.

## Reference Solution
```python
def calculate(num1, num2, operator):
    """
    This function performs the specified operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use for the operation. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not one of the four basic arithmetic operators.
        ZeroDivisionError: If the operator is '/' and the second number is zero.
    """

    if operator == '+':
        # Perform addition
        result = num1 + num2
    elif operator == '-':
        # Perform subtraction
        result = num1 - num2
    elif operator == '*':
        # Perform multiplication
        result = num1 * num2
    elif operator == '/':
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        # Perform division
        result = num1 / num2
    else:
        # Raise an error for invalid operators
        raise ValueError("Invalid operator. Please use '+', '-', '*', or '/'.")

    return result


def main():
    # Get the numbers and operator from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operator = input("Enter the operator (+, -, *, /): ")

    try:
        # Perform the operation and print the result
        result = calculate(num1, num2, operator)
        print(f"The result of {num1} {operator} {num2} is {result}")
    except (ValueError, ZeroDivisionError) as e:
        # Print any error messages
        print(str(e))


if __name__ == "__main__":
    main()
```
