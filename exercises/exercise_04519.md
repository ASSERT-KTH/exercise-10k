## Exercise
Create a program that simulates a chatbot. The user can ask the chatbot questions and it will respond accordingly.

## Reference Solution
```python
import random

# Define a dictionary of intents and responses
intents = {
    'greeting': {
        'patterns': ['hello', 'hi', 'hey'],
        'responses': ['Hello! How can I help you?', 'Hi! What\'s up?', 'Hey! How are you?']
    },
    'goodbye': {
        'patterns': ['bye', 'see you later', 'goodbye'],
        'responses': ['See you later!', 'Goodbye!', 'Have a great day!']
    },
    'help': {
        'patterns': ['help', 'what can you do', 'how can you help me'],
        'responses': ['I can help you with basic questions and tasks.', 'I can provide information on a variety of topics.', 'I can assist you with simple calculations and conversions.']
    },
    'default': {
        'patterns': [],
        'responses': ['I didn\'t understand that. Can you please rephrase?', 'Sorry, I couldn\'t understand your question.', 'Can you please provide more context or information?']
    }
}

def match_intent(message):
    """Matches the user's message to an intent"""
    for intent, data in intents.items():
        for pattern in data['patterns']:
            if pattern.lower() in message.lower():
                return intent
    return 'default'

def get_response(intent):
    """Gets a response based on the matched intent"""
    return random.choice(intents[intent]['responses'])

def chatbot(message):
    """Simulates a chatbot conversation"""
    intent = match_intent(message)
    response = get_response(intent)
    return response

def main():
    print("Welcome to the chatbot simulator!")
    while True:
        user_input = input("You: ")
        if user_input.lower() == 'quit':
            break
        response = chatbot(user_input)
        print("Chatbot: ", response)

if __name__ == "__main__":
    main()
```
