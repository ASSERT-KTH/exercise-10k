## Exercise
Write a function to find the maximum value in a dictionary

## Reference Solution
```python
def find_max_value(dictionary):
    """
    Find the maximum value in a dictionary.

    Args:
        dictionary (dict): The input dictionary.

    Returns:
        max_value: The maximum value in the dictionary.
        key: The key corresponding to the maximum value.

    Raises:
        ValueError: If the dictionary is empty.
    """
    if not dictionary:
        raise ValueError("Dictionary is empty")

    max_value = max(dictionary.values())
    key = [k for k, v in dictionary.items() if v == max_value][0]

    return max_value, key

# Example usage:
if __name__ == "__main__":
    my_dict = {"a": 1, "b": 2, "c": 3, "d": 4}
    max_val, key = find_max_value(my_dict)
    print(f"Maximum value: {max_val}")
    print(f"Key corresponding to maximum value: {key}")
```
