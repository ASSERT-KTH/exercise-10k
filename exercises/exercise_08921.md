## Exercise
Write a program to find the first missing positive integer in a given list of integers.

## Reference Solution
```python
def first_missing_positive(nums):
    """
    Find the first missing positive integer in a given list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The first missing positive integer.
    """
    if not nums:
        return 1

    # Create a set of positive integers in the list
    num_set = set(num for num in nums if num > 0)

    # Initialize the first missing positive integer to 1
    missing_positive = 1

    # Iterate from 1 to find the first missing positive integer
    while missing_positive in num_set:
        missing_positive += 1

    return missing_positive

# Example usage
def main():
    print(first_missing_positive([3, 4, -1, 1]))  # Output: 2
    print(first_missing_positive([1, 2, 0]))  # Output: 3
    print(first_missing_positive([-1, -2, -3]))  # Output: 1
    print(first_missing_positive([]))  # Output: 1

if __name__ == "__main__":
    main()
```
