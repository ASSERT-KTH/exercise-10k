## Exercise
Write a program to calculate the volume of a torus given its inner and outer radii.

## Reference Solution
```python
import math

def calculate_torus_volume(inner_radius, outer_radius):
    """
    Calculate the volume of a torus given its inner and outer radii.

    Args:
        inner_radius (float): The inner radius of the torus.
        outer_radius (float): The outer radius of the torus.

    Returns:
        float: The volume of the torus.
    """
    if inner_radius >= outer_radius:
        raise ValueError("Inner radius must be less than outer radius")

    major_radius = (outer_radius + inner_radius) / 2
    minor_radius = (outer_radius - inner_radius) / 2

    volume = 2 * math.pi**2 * major_radius * minor_radius**2
    return volume

def main():
    inner_radius = float(input("Enter the inner radius of the torus: "))
    outer_radius = float(input("Enter the outer radius of the torus: "))

    try:
        volume = calculate_torus_volume(inner_radius, outer_radius)
        print(f"The volume of the torus is: {volume:.2f} cubic units")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
