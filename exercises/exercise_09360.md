## Exercise
Write a program that demonstrates the use of the string data type.

## Reference Solution
```python
# Define a function to demonstrate string operations
def demonstrate_string_operations():
    # Define two strings
    str1 = "Hello"
    str2 = "World"

    # Concatenate strings
    print("Concatenation: " + str1 + " " + str2)

    # Extract substring
    print("Substring (0-4): " + str1[0:5])

    # String length
    print("Length of str1: " + str(len(str1)))

    # String upper and lower case
    print("Upper case: " + str1.upper())
    print("Lower case: " + str1.lower())

    # String split
    str3 = "apple,banana,cherry"
    fruits = str3.split(",")
    print("Fruits: ")
    for fruit in fruits:
        print(fruit)

    # String join
    fruits = ["apple", "banana", "cherry"]
    str4 = ",".join(fruits)
    print("Joined fruits: " + str4)

# Define a main function
def main():
    demonstrate_string_operations()

# Run the main function
if __name__ == "__main__":
    main()
```
