## Exercise
Write a program that takes the coordinates of two points as input and returns the distance between them.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the distance between two points.

    Args:
        point1 (tuple): The coordinates of the first point as a tuple (x, y).
        point2 (tuple): The coordinates of the second point as a tuple (x, y).

    Returns:
        float: The distance between the two points.
    """
    x1, y1 = point1
    x2, y2 = point2

    # Calculate the difference between the x and y coordinates of the two points
    dx = x2 - x1
    dy = y2 - y1

    # Use the Euclidean distance formula to calculate the distance
    distance = math.sqrt(dx ** 2 + dy ** 2)

    return distance

def main():
    # Get the coordinates of the two points from the user
    x1 = float(input("Enter the x-coordinate of the first point: "))
    y1 = float(input("Enter the y-coordinate of the first point: "))
    x2 = float(input("Enter the x-coordinate of the second point: "))
    y2 = float(input("Enter the y-coordinate of the second point: "))

    # Calculate and print the distance between the two points
    point1 = (x1, y1)
    point2 = (x2, y2)
    distance = calculate_distance(point1, point2)
    print(f"The distance between the two points is: {distance}")

if __name__ == "__main__":
    main()
```
