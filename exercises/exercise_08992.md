## Exercise
Write a function that takes two numbers and an operator as arguments and returns the result of the operation.

## Reference Solution
```python
def math_operation(num1, num2, operator):
    """
    This function performs basic arithmetic operations.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The operator to use. Supports '+', '-', '*', '/'.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not supported.
        ZeroDivisionError: If division by zero is attempted.
    """

    # Define a dictionary to map operators to their corresponding functions
    operators = {
        '+': lambda x, y: x + y,
        '-': lambda x, y: x - y,
        '*': lambda x, y: x * y,
        '/': lambda x, y: x / y if y != 0 else float('inf')
    }

    # Check if the operator is supported
    if operator not in operators:
        raise ValueError("Unsupported operator. Supported operators are '+', '-', '*', '/'.")

    # Perform the operation
    try:
        result = operators[operator](num1, num2)
        if result == float('inf'):
            raise ZeroDivisionError("Cannot divide by zero.")
        return result
    except Exception as e:
        raise e


# Example usage
if __name__ == "__main__":
    print(math_operation(10, 2, '+'))  # Output: 12
    print(math_operation(10, 2, '-'))  # Output: 8
    print(math_operation(10, 2, '*'))  # Output: 20
    print(math_operation(10, 2, '/'))  # Output: 5.0
    try:
        print(math_operation(10, 0, '/'))  # Raises ZeroDivisionError
    except ZeroDivisionError as e:
        print(e)
    try:
        print(math_operation(10, 2, '%'))  # Raises ValueError
    except ValueError as e:
        print(e)
```
