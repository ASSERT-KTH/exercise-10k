## Exercise
Create a program to calculate the median given a list of numbers.

## Reference Solution
```python
def calculate_median(numbers):
    """
    Calculate the median of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The median of the list.
    """
    # Sort the list in ascending order
    numbers.sort()

    # Get the length of the list
    n = len(numbers)

    # If the list has an odd number of elements, the median is the middle value
    if n % 2 == 1:
        median = numbers[n // 2]
    # If the list has an even number of elements, the median is the average of the two middle values
    else:
        median = (numbers[n // 2 - 1] + numbers[n // 2]) / 2

    return median


def main():
    # Example usage
    numbers = [1, 3, 5, 7, 9]
    median = calculate_median(numbers)
    print(f"The median of {numbers} is {median}")

    numbers = [1, 2, 3, 4, 5, 6]
    median = calculate_median(numbers)
    print(f"The median of {numbers} is {median}")


if __name__ == "__main__":
    main()
```
