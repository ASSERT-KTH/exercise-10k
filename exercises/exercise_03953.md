## Exercise
Write a program to simulate a chatbot that responds to basic user queries.

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary with intents and responses
intents = {
    'greeting': {
        'patterns': ['hello', 'hi', 'hey'],
        'responses': ['Hello! How can I assist you today?', 'Hi! What\'s on your mind?', 'Hey! How are you doing today?']
    },
    'goodbye': {
        'patterns': ['bye', 'goodbye', 'see you later'],
        'responses': ['Goodbye! Have a great day!', 'See you later! It was nice chatting with you.', 'Bye! Feel free to come back anytime.']
    },
    'help': {
        'patterns': ['help', 'what can you do', 'how can you assist me'],
        'responses': ['I can assist you with basic queries, provide information, and answer questions.', 'I can help you with a variety of topics, including but not limited to, general knowledge, definitions, and calculations.', 'I\'m here to help with any questions or concerns you may have.']
    },
    'thanks': {
        'patterns': ['thanks', 'thank you', 'appreciate it'],
        'responses': ['You\'re welcome! It was my pleasure to assist you.', 'No problem! Is there anything else I can help you with?', 'I\'m glad I could help. Feel free to ask me anything else.']
    }
}

# Define a function to match user input with patterns
def match_intent(message):
    for intent, values in intents.items():
        for pattern in values['patterns']:
            if pattern.lower() in message.lower():
                return intent
    return None

# Define a function to generate a response based on the matched intent
def generate_response(intent):
    if intent is not None:
        return random.choice(intents[intent]['responses'])
    else:
        return 'I didn\'t quite understand that. Could you please rephrase?'

# Define the main function to simulate the chatbot
def chatbot():
    print('Welcome to the chatbot! Type "quit" to exit.')
    while True:
        message = input('User: ')
        if message.lower() == 'quit':
            break
        intent = match_intent(message)
        response = generate_response(intent)
        print('Chatbot: ', response)

# Run the chatbot
if __name__ == '__main__':
    chatbot()
```
