## Exercise
Write a program that takes an array of integers and two indices as input, and returns a new array containing the elements between the two indices.

## Reference Solution
```python
#include <iostream>
#include <vector>

using namespace std;

vector<int> between(vector<int> arr, int start, int end) {
    vector<int> newarr;
    for (int i = start; i < end; i++) {
        newarr.push_back(arr[i]);
    }
    return newarr;
}

int main() {
    vector<int> arr = {1, 2, 3, 4, 5};
    vector<int> result = between(arr, 1, 3);
    for (int i = 0; i < result.size(); i++) {
        cout << result[i] << " ";
    }
    return 0;
}
```
