## Exercise
Write a program to display a weather forecast for the next 5 days, using a for loop to handle each day's forecast.

## Reference Solution
```python
# Define a dictionary with weather forecasts for the next 5 days
weather_forecast = {
    "Monday": {"high": 75, "low": 50, "condition": "Sunny"},
    "Tuesday": {"high": 80, "low": 55, "condition": "Partly Cloudy"},
    "Wednesday": {"high": 85, "low": 60, "condition": "Mostly Sunny"},
    "Thursday": {"high": 70, "low": 45, "condition": "Rainy"},
    "Friday": {"high": 65, "low": 40, "condition": "Windy"}
}

# Define a list of days of the week
days_of_week = list(weather_forecast.keys())

# Use a for loop to display each day's forecast
for i in range(5):
    day = days_of_week[i]
    forecast = weather_forecast[day]
    print(f"Weather forecast for {day}:")
    print(f"High temperature: {forecast['high']}°F")
    print(f"Low temperature: {forecast['low']}°F")
    print(f"Condition: {forecast['condition']}\n")
```

```python
import requests
import json

# Set your API key and location
api_key = "YOUR_API_KEY"
location = "London,uk"

# Set the API endpoint and parameters
endpoint = f"http://api.openweathermap.org/data/2.5/forecast?q={location}&appid={api_key}&units=imperial"

# Make the API request
response = requests.get(endpoint)

# Check if the request was successful
if response.status_code == 200:
    # Parse the JSON response
    data = json.loads(response.text)

    # Extract the forecast data
    forecast_data = data["list"]

    # Use a for loop to display each day's forecast
    for i in range(0, len(forecast_data), 8):  # OpenWeatherMap API returns 8 forecasts per day
        forecast = forecast_data[i]
        print(f"Weather forecast for {forecast['dt_txt']}:")
        print(f"High temperature: {forecast['main']['temp_max']}°F")
        print(f"Low temperature: {forecast['main']['temp_min']}°F")
        print(f"Condition: {forecast['weather'][0]['description']}\n")
else:
    print("Failed to retrieve weather forecast")
```
