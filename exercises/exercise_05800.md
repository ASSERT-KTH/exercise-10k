## Exercise
Write a function that finds the maximum contiguous subarray of a given list of numbers.

## Reference Solution
```python
def max_contiguous_subarray(numbers):
    """
    Finds the maximum contiguous subarray of a given list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the maximum sum and the corresponding subarray.
    """
    if not numbers:
        return 0, []

    max_sum = float('-inf')
    current_sum = 0
    start = 0
    max_start = 0
    max_end = 0

    for i, num in enumerate(numbers):
        if current_sum <= 0:
            current_sum = num
            start = i
        else:
            current_sum += num

        if current_sum > max_sum:
            max_sum = current_sum
            max_start = start
            max_end = i

    return max_sum, numbers[max_start:max_end+1]

# Example usage
numbers = [-2, -3, 4, -1, -2, 1, 5, -3]
max_sum, max_subarray = max_contiguous_subarray(numbers)
print(f"Maximum sum: {max_sum}")
print(f"Maximum contiguous subarray: {max_subarray}")
```
