## Exercise
Create a program that takes a dictionary where keys are student names and values are lists of their grades. Calculate the average grade for each student, then return a new dictionary with the student names as keys and their average grades as values.

## Reference Solution
```python
def calculate_average_grades(student_grades):
    """
    Calculate the average grade for each student.

    Args:
        student_grades (dict): A dictionary where keys are student names and values are lists of their grades.

    Returns:
        dict: A dictionary with student names as keys and their average grades as values.
    """

    # Initialize an empty dictionary to store the average grades
    average_grades = {}

    # Iterate over each student in the student_grades dictionary
    for student, grades in student_grades.items():
        # Check if the list of grades is not empty
        if grades:
            # Calculate the average grade for the current student
            average_grade = sum(grades) / len(grades)
            # Store the average grade in the average_grades dictionary
            average_grades[student] = average_grade
        else:
            # If the list of grades is empty, store None as the average grade
            average_grades[student] = None

    # Return the dictionary with the average grades
    return average_grades


# Example usage:
if __name__ == "__main__":
    student_grades = {
        "Alice": [90, 85, 95],
        "Bob": [80, 70, 90],
        "Charlie": [95, 90, 85],
        "David": []
    }

    average_grades = calculate_average_grades(student_grades)
    print("Average Grades:")
    for student, average_grade in average_grades.items():
        if average_grade is not None:
            print(f"{student}: {average_grade:.2f}")
        else:
            print(f"{student}: No grades available")
```
