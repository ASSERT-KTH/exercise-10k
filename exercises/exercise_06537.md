## Exercise
Create a function that returns a list of genres from a list of movies.

def extract_genres(movies):
    """
    Extracts a list of unique genres from a list of movies.

    Args:
        movies (list): A list of dictionaries, where each dictionary contains
            information about a movie, including its title and genres.

    Returns:
        list: A list of unique genres.

    Example:
        movies = [
            {"title": "Movie 1", "genres": ["Action", "Thriller"]},
            {"title": "Movie 2", "genres": ["Comedy", "Drama"]},
            {"title": "Movie 3", "genres": ["Action", "Romance"]}
        ]
        print(extract_genres(movies))  # Output: ["Action", "Thriller", "Comedy", "Drama", "Romance"]
    """
    genres = set()
    for movie in movies:
        genres.update(movie.get("genres", []))
    return list(genres)

# Example usage
movies = [
    {"title": "Movie 1", "genres": ["Action", "Thriller"]},
    {"title": "Movie 2", "genres": ["Comedy", "Drama"]},
    {"title": "Movie 3", "genres": ["Action", "Romance"]}
]

print(extract_genres(movies))
