## Exercise
Create a function that finds the cartesian product of two lists.

## Reference Solution
```python
import itertools

def cartesian_product(list1, list2):
    """
    Calculate the cartesian product of two lists.

    Args:
        list1 (list): The first list.
        list2 (list): The second list.

    Returns:
        list: A list of tuples, where each tuple is a pair of elements from list1 and list2.
    """
    return list(itertools.product(list1, list2))

# Example usage
list1 = [1, 2, 3]
list2 = ['a', 'b', 'c']
result = cartesian_product(list1, list2)

print(result)
```

def cartesian_product_manual(list1, list2):
    """
    Calculate the cartesian product of two lists manually.

    Args:
        list1 (list): The first list.
        list2 (list): The second list.

    Returns:
        list: A list of tuples, where each tuple is a pair of elements from list1 and list2.
    """
    result = []
    for elem1 in list1:
        for elem2 in list2:
            result.append((elem1, elem2))
    return result

# Example usage
list1 = [1, 2, 3]
list2 = ['a', 'b', 'c']
result = cartesian_product_manual(list1, list2)

print(result)
