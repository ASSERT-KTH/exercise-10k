## Exercise
Create a function to generate all permutations of a given string.

## Reference Solution
```python
from typing import List

def generate_permutations(s: str) -> List[str]:
    """
    Generates all permutations of a given string.
    
    Args:
    s (str): The input string.
    
    Returns:
    List[str]: A list of all permutations of the input string.
    """
    # Base case: if the string has only one character, return it
    if len(s) == 1:
        return [s]
    
    permutations = []
    for i, char in enumerate(s):
        # Choose the current character as the first character
        remaining_string = s[:i] + s[i+1:]
        
        # Generate all permutations of the remaining string
        for perm in generate_permutations(remaining_string):
            permutations.append(char + perm)
    
    return permutations

# Example usage
def main():
    input_string = "abc"
    permutations = generate_permutations(input_string)
    print(f"Permutations of '{input_string}': {permutations}")

if __name__ == "__main__":
    main()
```
