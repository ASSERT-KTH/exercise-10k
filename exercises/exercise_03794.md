## Exercise
Write a function to solve the N-Queens problem and return all possible configurations of queens on a chessboard.

## Reference Solution

def solve_n_queens(n):
    """
    Solves the N-Queens problem and returns all possible configurations of queens on a chessboard.

    Args:
    n (int): The size of the chessboard (number of rows and columns).

    Returns:
    list: A list of all possible configurations, where each configuration is a list of strings representing the chessboard.
    """

    def is_safe(board, row, col):
        """
        Checks if a queen can be placed at the given position on the board.

        Args:
        board (list): The current state of the board.
        row (int): The row to check.
        col (int): The column to check.

        Returns:
        bool: True if the queen can be placed, False otherwise.
        """
        for i in range(row):
            if board[i][col] == 'Q':
                return False
            if col - (row - i) >= 0 and board[i][col - (row - i)] == 'Q':
                return False
            if col + (row - i) < n and board[i][col + (row - i)] == 'Q':
                return False
        return True

    def backtrack(board, row):
        """
        Recursively tries to place queens on the board.

        Args:
        board (list): The current state of the board.
        row (int): The current row to place a queen.

        Yields:
        list: A solution configuration.
        """
        if row == n:
            yield [''.join(row) for row in board]
        else:
            for col in range(n):
                if is_safe(board, row, col):
                    board[row][col] = 'Q'
                    yield from backtrack(board, row + 1)
                    board[row][col] = '.'

    # Initialize the board and start the backtracking process
    board = [['.' for _ in range(n)] for _ in range(n)]
    return list(backtrack(board, 0))

# Example usage
n = 4
solutions = solve_n_queens(n)
for i, solution in enumerate(solutions):
    print(f"Solution {i+1}:")
    for row in solution:
        print(row)
    print()
