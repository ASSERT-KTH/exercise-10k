## Exercise
Create a program that allows the user to add, remove, and view items in a to-do list, taking the user's input as commands.

## Reference Solution
```python
# Define a class for the to-do list
class ToDoList:
    def __init__(self):
        self.tasks = []

    # Method to add a task to the list
    def add_task(self, task):
        self.tasks.append(task)
        print(f"Task '{task}' added to the list.")

    # Method to remove a task from the list
    def remove_task(self, task_number):
        try:
            task_number = int(task_number)
            if task_number <= len(self.tasks):
                task = self.tasks.pop(task_number - 1)
                print(f"Task '{task}' removed from the list.")
            else:
                print("Invalid task number.")
        except ValueError:
            print("Invalid task number.")

    # Method to view all tasks in the list
    def view_tasks(self):
        if not self.tasks:
            print("No tasks in the list.")
        else:
            print("Tasks:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")


def main():
    to_do_list = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add task")
        print("2. Remove task")
        print("3. View tasks")
        print("4. Quit")

        option = input("Enter your choice: ")

        if option == "1":
            task = input("Enter the task: ")
            to_do_list.add_task(task)
        elif option == "2":
            to_do_list.view_tasks()
            task_number = input("Enter the task number to remove: ")
            to_do_list.remove_task(task_number)
        elif option == "3":
            to_do_list.view_tasks()
        elif option == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please try again.")


if __name__ == "__main__":
    main()
```
