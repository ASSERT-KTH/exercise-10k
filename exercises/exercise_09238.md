## Exercise
Write a program that accesses and manipulates elements of an array using indexing.

## Reference Solution
```python
def main():
    # Define an array
    array = [1, 2, 3, 4, 5]

    # Print the original array
    print("Original Array:", array)

    # Accessing elements using indexing
    print("\nAccessing Elements:")
    print("Element at index 0:", array[0])
    print("Element at index 2:", array[2])

    # Manipulating elements using indexing
    print("\nManipulating Elements:")
    array[0] = 10
    array[2] = 30
    print("Modified Array:", array)

    # Accessing elements using negative indexing
    print("\nAccessing Elements using Negative Indexing:")
    print("Last Element:", array[-1])
    print("Second Last Element:", array[-2])

    # Slicing the array
    print("\nSlicing the Array:")
    print("First three elements:", array[:3])
    print("Last three elements:", array[-3:])

if __name__ == "__main__":
    main()
```
