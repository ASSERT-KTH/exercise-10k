## Exercise
Write a program that takes two strings as input and returns the minimum window substring that contains all characters of the second string.

from collections import Counter

def min_window(s1, s2):
    """
    Returns the minimum window substring of s1 that contains all characters of s2.

    Args:
    s1 (str): The string to search for the minimum window.
    s2 (str): The string containing the characters to search for.

    Returns:
    str: The minimum window substring of s1 that contains all characters of s2. If no such window exists, returns an empty string.
    """

    # Base case: If s2 is longer than s1, there can't be a valid window
    if len(s2) > len(s1):
        return ""

    # Create a counter for the characters in s2
    s2_count = Counter(s2)

    # Initialize the minimum window size to infinity
    min_window_size = float('inf')
    min_window = ""

    # Initialize the left and right pointers of the sliding window
    left = 0
    right = 0

    # Initialize a counter for the characters in the current window
    window_count = {}

    # Initialize the number of characters in the window that match s2
    matched_chars = 0

    # Expand the window to the right
    while right < len(s1):
        # Add the character at the right pointer to the window counter
        char = s1[right]
        window_count[char] = window_count.get(char, 0) + 1

        # If the character is in s2 and its count in the window is less than or equal to its count in s2, increment the matched chars count
        if char in s2_count and window_count[char] <= s2_count[char]:
            matched_chars += 1

        # While the window contains all characters of s2, try to shrink the window from the left
        while matched_chars == len(s2):
            # Update the minimum window if the current window is smaller
            if right - left + 1 < min_window_size:
                min_window_size = right - left + 1
                min_window = s1[left:right + 1]

            # Remove the character at the left pointer from the window counter
            char = s1[left]
            window_count[char] -= 1

            # If the character is in s2 and its count in the window is less than its count in s2, decrement the matched chars count
            if char in s2_count and window_count[char] < s2_count[char]:
                matched_chars -= 1

            # Move the left pointer to the right
            left += 1

        # Move the right pointer to the right
        right += 1

    return min_window


# Example usage:
s1 = "ADOBECODEBANC"
s2 = "ABC"
print(min_window(s1, s2))  # Output: "BANC"
