## Exercise
Write a program that takes an array and a partition size as input, and returns a list of sub-arrays, each with the specified partition size.

## Reference Solution
```python
export default function partition<T>(array: T[], partitionSize: number): T[][] {
  const partitions: T[][] = [];

  if (partitionSize <= 0) {
    return partitions;
  }

  for (let i = 0; i < array.length; i += partitionSize) {
    partitions.push(array.slice(i, i + partitionSize));
  }

  return partitions;
}
```
