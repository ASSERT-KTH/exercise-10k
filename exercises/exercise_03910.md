## Exercise
Create a program that calculates the total cost of a meal including a tip given the cost of the meal and the tip percentage.

## Reference Solution
```python
def calculate_total_cost(meal_cost, tip_percentage):
    """
    Calculate the total cost of a meal, including a tip.

    Args:
        meal_cost (float): The cost of the meal.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The total cost of the meal, including the tip.
    """
    # Calculate the tip amount
    tip_amount = (tip_percentage / 100) * meal_cost

    # Calculate the total cost
    total_cost = meal_cost + tip_amount

    return total_cost


def main():
    # Get the meal cost from the user
    meal_cost = float(input("Enter the cost of the meal: $"))

    # Get the tip percentage from the user
    tip_percentage = float(input("Enter the tip percentage (%): "))

    # Calculate the total cost
    total_cost = calculate_total_cost(meal_cost, tip_percentage)

    # Print the result
    print(f"Meal Cost: ${meal_cost:.2f}")
    print(f"Tip ({tip_percentage}%): ${((tip_percentage / 100) * meal_cost):.2f}")
    print(f"Total Cost: ${total_cost:.2f}")


if __name__ == "__main__":
    main()
```
