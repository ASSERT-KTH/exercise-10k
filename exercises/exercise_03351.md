## Exercise
Write a program that uses a for loop to manage a list of bank accounts and perform transactions (deposit, withdraw, check balance).

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_number, account_holder, balance=0):
        self.account_number = account_number
        self.account_holder = account_holder
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        print(f"Deposited ${amount:.2f} into account {self.account_number}. New balance: ${self.balance:.2f}")

    def withdraw(self, amount):
        if amount > self.balance:
            print("Insufficient funds.")
        else:
            self.balance -= amount
            print(f"Withdrew ${amount:.2f} from account {self.account_number}. New balance: ${self.balance:.2f}")

    def check_balance(self):
        print(f"Account {self.account_number} balance: ${self.balance:.2f}")


def main():
    # Create a list of bank accounts
    accounts = [
        BankAccount("12345", "John Doe", 1000),
        BankAccount("67890", "Jane Smith", 500),
        BankAccount("11111", "Bob Johnson", 2000)
    ]

    # Manage accounts using a for loop
    for account in accounts:
        print(f"\nAccount Number: {account.account_number}, Account Holder: {account.account_holder}")
        account.check_balance()

        # Perform transactions
        while True:
            print("\n1. Deposit")
            print("2. Withdraw")
            print("3. Check Balance")
            print("4. Next Account")
            choice = input("Enter your choice: ")

            if choice == "1":
                amount = float(input("Enter amount to deposit: "))
                account.deposit(amount)
            elif choice == "2":
                amount = float(input("Enter amount to withdraw: "))
                account.withdraw(amount)
            elif choice == "3":
                account.check_balance()
            elif choice == "4":
                break
            else:
                print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
