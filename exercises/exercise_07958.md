## Exercise
Write a program to implement a basic graph using an adjacency list or adjacency matrix.

## Reference Solution
```python
class GraphAdjacencyList:
    def __init__(self):
        """
        Initialize an empty graph.
        """
        self.adjacency_list = {}

    def add_vertex(self, vertex):
        """
        Add a vertex to the graph.

        Args:
            vertex (any): The vertex to be added.
        """
        if vertex not in self.adjacency_list:
            self.adjacency_list[vertex] = []

    def add_edge(self, vertex1, vertex2):
        """
        Add an edge between two vertices.

        Args:
            vertex1 (any): The first vertex.
            vertex2 (any): The second vertex.
        """
        if vertex1 in self.adjacency_list and vertex2 in self.adjacency_list:
            self.adjacency_list[vertex1].append(vertex2)
            self.adjacency_list[vertex2].append(vertex1)

    def print_graph(self):
        """
        Print the adjacency list of the graph.
        """
        for vertex in self.adjacency_list:
            print(vertex, "->", self.adjacency_list[vertex])


# Example usage
if __name__ == "__main__":
    graph = GraphAdjacencyList()
    graph.add_vertex(1)
    graph.add_vertex(2)
    graph.add_vertex(3)
    graph.add_edge(1, 2)
    graph.add_edge(2, 3)
    graph.add_edge(3, 1)
    graph.print_graph()
```

```python
class GraphAdjacencyMatrix:
    def __init__(self, num_vertices):
        """
        Initialize an empty graph with a specified number of vertices.

        Args:
            num_vertices (int): The number of vertices in the graph.
        """
        self.num_vertices = num_vertices
        self.adjacency_matrix = [[0] * num_vertices for _ in range(num_vertices)]

    def add_edge(self, vertex1, vertex2):
        """
        Add an edge between two vertices.

        Args:
            vertex1 (int): The index of the first vertex.
            vertex2 (int): The index of the second vertex.
        """
        if 0 <= vertex1 < self.num_vertices and 0 <= vertex2 < self.num_vertices:
            self.adjacency_matrix[vertex1][vertex2] = 1
            self.adjacency_matrix[vertex2][vertex1] = 1

    def print_graph(self):
        """
        Print the adjacency matrix of the graph.
        """
        for row in self.adjacency_matrix:
            print(row)


# Example usage
if __name__ == "__main__":
    graph = GraphAdjacencyMatrix(4)
    graph.add_edge(0, 1)
    graph.add_edge(1, 2)
    graph.add_edge(2, 3)
    graph.add_edge(3, 0)
    graph.print_graph()
```
