## Exercise
Write a program that interacts with the user, asking for input and providing feedback based on the input provided.

## Reference Solution
```python
def get_user_input():
    """Get user input and provide feedback."""
    print("Welcome to our interactive program!")
    
    # Ask for user's name
    name = input("Please enter your name: ")
    print(f"Hello, {name}!")
    
    # Ask for user's age
    while True:
        try:
            age = int(input("Please enter your age: "))
            if age < 0:
                print("Age cannot be negative. Please enter a valid age.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid age.")
    
    # Ask for user's favorite color
    favorite_color = input("Please enter your favorite color: ")
    print(f"Your favorite color is {favorite_color}.")
    
    # Provide feedback based on age
    if age < 18:
        print(f"{name}, you are still young and have your whole life ahead of you!")
    elif age < 65:
        print(f"{name}, you are in the prime of your life! Enjoy every moment.")
    else:
        print(f"{name}, you have gained a lot of experience and wisdom. Keep sharing it with others.")

def main():
    """Main function."""
    get_user_input()
    print("Thank you for interacting with our program!")

if __name__ == "__main__":
    main()
```
