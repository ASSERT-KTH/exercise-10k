## Exercise
Write a program that takes the distance traveled and the amount of fuel used as input and outputs the fuel efficiency.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel):
    """
    Calculate the fuel efficiency.

    Args:
        distance (float): The distance traveled in miles.
        fuel (float): The amount of fuel used in gallons.

    Returns:
        float: The fuel efficiency in miles per gallon.
    """
    if fuel <= 0:
        raise ValueError("Fuel used must be greater than zero.")
    if distance < 0:
        raise ValueError("Distance traveled cannot be negative.")

    fuel_efficiency = distance / fuel
    return fuel_efficiency

def main():
    # Get user input
    distance = float(input("Enter the distance traveled in miles: "))
    fuel = float(input("Enter the amount of fuel used in gallons: "))

    try:
        # Calculate and display the fuel efficiency
        fuel_efficiency = calculate_fuel_efficiency(distance, fuel)
        print(f"The fuel efficiency is {fuel_efficiency:.2f} miles per gallon.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
