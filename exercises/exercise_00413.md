## Exercise
Write a program to find all permutations of an array.

## Reference Solution
```python
def get_permutations(array):
    """
    Generate all permutations of a given array.

    Args:
    array (list): The input array.

    Returns:
    list: A list of lists, where each sublist is a permutation of the input array.
    """
    # Base case: if the array has only one element, return it as the only permutation
    if len(array) == 1:
        return [array]

    permutations = []
    for i in range(len(array)):
        # Choose the current element as the first element of the permutation
        current_element = array[i]
        # Get the remaining elements
        remaining_elements = array[:i] + array[i + 1:]
        # Generate permutations of the remaining elements
        for permutation in get_permutations(remaining_elements):
            # Add the current element to the beginning of each permutation
            permutations.append([current_element] + permutation)

    return permutations

def main():
    # Example usage:
    array = [1, 2, 3]
    permutations = get_permutations(array)
    print("Permutations of the array:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i + 1}: {permutation}")

if __name__ == "__main__":
    main()
```
