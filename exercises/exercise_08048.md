## Exercise
Write a program that uses an if/else statement to manage a to-do list with priorities based on user input.

## Reference Solution
```python
# Initialize an empty dictionary to store tasks
tasks = {}

def add_task():
    task_name = input("Enter task name: ")
    priority = input("Enter task priority (High/Medium/Low): ")
    
    # Validate priority input
    while priority not in ["High", "Medium", "Low"]:
        print("Invalid priority. Please enter High, Medium, or Low.")
        priority = input("Enter task priority (High/Medium/Low): ")
        
    # Add task to dictionary
    tasks[task_name] = priority
    print(f"Task '{task_name}' added with priority '{priority}'")

def view_tasks():
    if not tasks:
        print("No tasks in the list.")
    else:
        print("Tasks:")
        for task, priority in tasks.items():
            print(f"{task}: {priority}")

def delete_task():
    task_name = input("Enter task name to delete: ")
    
    # Check if task exists
    if task_name in tasks:
        del tasks[task_name]
        print(f"Task '{task_name}' deleted.")
    else:
        print(f"Task '{task_name}' not found.")

def main():
    while True:
        print("\nTo-Do List Menu:")
        print("1. Add task")
        print("2. View tasks")
        print("3. Delete task")
        print("4. Exit")
        
        choice = input("Enter your choice: ")
        
        # Use if/else statement to manage user input
        if choice == "1":
            add_task()
        elif choice == "2":
            view_tasks()
        elif choice == "3":
            delete_task()
        elif choice == "4":
            print("Exiting program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
