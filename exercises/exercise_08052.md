## Exercise
Write a program that uses an if/else statement to play a simple game with levels based on user input.

## Reference Solution
```python
# Game with levels based on user input

def game():
    # Initialize the user's level
    level = 1

    while True:
        print(f"\nYou are currently on level {level}.")

        # Ask the user for input
        user_input = input("Do you want to proceed to the next level? (yes/no): ")

        # Check the user's input
        if user_input.lower() == "yes":
            # If the user wants to proceed, increment the level
            level += 1

            # Check the level and provide a challenge
            if level == 2:
                print("Level 2: Solve a simple math problem.")
                math_problem = input("What is 2 + 2? ")
                if math_problem == "4":
                    print("Correct! You can proceed to the next level.")
                else:
                    print("Incorrect. Game over.")
                    break
            elif level == 3:
                print("Level 3: Guess a random number between 1 and 10.")
                guess = int(input("Enter your guess: "))
                if guess == 7:
                    print("Correct! You can proceed to the next level.")
                else:
                    print("Incorrect. Game over.")
                    break
            elif level == 4:
                print("Level 4: Answer a trivia question.")
                trivia_question = input("What is the capital of France? ")
                if trivia_question.lower() == "paris":
                    print("Correct! You can proceed to the next level.")
                else:
                    print("Incorrect. Game over.")
                    break
            elif level == 5:
                print("Level 5: Congratulations, you have reached the final level!")
                print("You win! Well done!")
                break
        elif user_input.lower() == "no":
            print("Game over. Thanks for playing!")
            break
        else:
            print("Invalid input. Please enter 'yes' or 'no'.")

# Run the game
game()
```
