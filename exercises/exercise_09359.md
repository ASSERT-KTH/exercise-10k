## Exercise
Write a program that demonstrates the use of the number data type.

## Reference Solution
```python
# Import the cmath module for complex numbers
import cmath

def main():
    # Integer example
    integer_num = 10
    print(f"Integer: {integer_num}")
    print(f"Integer type: {type(integer_num)}")

    # Floating-point example
    float_num = 10.5
    print(f"\nFloating-point: {float_num}")
    print(f"Floating-point type: {type(float_num)}")

    # Complex number example
    complex_num = 10 + 5j
    print(f"\nComplex number: {complex_num}")
    print(f"Complex number type: {type(complex_num)}")

    # Arithmetic operations
    print("\nArithmetic Operations:")
    print(f"10 + 5 = {10 + 5}")
    print(f"10 - 5 = {10 - 5}")
    print(f"10 * 5 = {10 * 5}")
    print(f"10 / 5 = {10 / 5}")

    # Complex number arithmetic
    print("\nComplex Number Arithmetic:")
    print(f"(10 + 5j) + (2 + 3j) = {(10 + 5j) + (2 + 3j)}")
    print(f"(10 + 5j) - (2 + 3j) = {(10 + 5j) - (2 + 3j)}")
    print(f"(10 + 5j) * (2 + 3j) = {(10 + 5j) * (2 + 3j)}")
    print(f"(10 + 5j) / (2 + 3j) = {(10 + 5j) / (2 + 3j)}")

    # Square root calculation using cmath
    print("\nSquare Root Calculation:")
    print(f"Square root of -1: {cmath.sqrt(-1)}")

if __name__ == "__main__":
    main()
```
