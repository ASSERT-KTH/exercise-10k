## Exercise
Write a program that checks if the user's input is a valid integer

## Reference Solution
```python
def is_valid_integer(input_str):
    """
    Checks if the input string is a valid integer.

    Args:
        input_str (str): The user's input.

    Returns:
        bool: True if the input is a valid integer, False otherwise.
    """
    try:
        int(input_str)
        return True
    except ValueError:
        return False

def main():
    # Get user input
    user_input = input("Please enter an integer: ")

    # Check if the input is a valid integer
    if is_valid_integer(user_input):
        print(f"'{user_input}' is a valid integer.")
    else:
        print(f"'{user_input}' is not a valid integer.")

if __name__ == "__main__":
    main()
```
