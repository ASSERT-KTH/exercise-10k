## Exercise
Write a program that generates a random quote and prints it out.

## Reference Solution
```python
import random

# List of quotes
quotes = [
    {"quote": "Believe you can and you're halfway there.", "author": "Theodore Roosevelt"},
    {"quote": "It does not matter how slowly you go as long as you do not stop.", "author": "Confucius"},
    {"quote": "Success is not final, failure is not fatal: It is the courage to continue that counts.", "author": "Winston Churchill"},
    {"quote": "Don't watch the clock; do what it does. Keep going.", "author": "Sam Levenson"},
    {"quote": "You miss 100% of the shots you don't take.", "author": "Wayne Gretzky"}
]

def generate_random_quote(quotes):
    """Returns a random quote from the list of quotes"""
    random_quote = random.choice(quotes)
    return random_quote

def print_quote(quote):
    """Prints the quote and its author"""
    print(f"'{quote['quote']}' - {quote['author']}")

def main():
    random_quote = generate_random_quote(quotes)
    print_quote(random_quote)

if __name__ == "__main__":
    main()
```
