## Exercise
Develop a program that maintains a student's grade book. The program should allow users to add grades for assignments, and print the average grade for the student.

## Reference Solution
```python
class StudentGradeBook:
    def __init__(self, student_name):
        self.student_name = student_name
        self.grades = []

    def add_grade(self, assignment_name, grade):
        """Add a grade for an assignment"""
        self.grades.append((assignment_name, grade))

    def calculate_average_grade(self):
        """Calculate the average grade"""
        if not self.grades:
            return 0
        total_grade = sum(grade for _, grade in self.grades)
        return total_grade / len(self.grades)

    def print_grades(self):
        """Print all grades"""
        print(f"Grades for {self.student_name}:")
        for assignment_name, grade in self.grades:
            print(f"{assignment_name}: {grade}")

    def print_average_grade(self):
        """Print the average grade"""
        average_grade = self.calculate_average_grade()
        print(f"Average grade for {self.student_name}: {average_grade:.2f}")


def main():
    student_name = input("Enter the student's name: ")
    grade_book = StudentGradeBook(student_name)

    while True:
        print("\nOptions:")
        print("1. Add grade")
        print("2. Print grades")
        print("3. Print average grade")
        print("4. Exit")

        option = input("Choose an option: ")

        if option == "1":
            assignment_name = input("Enter the assignment name: ")
            grade = float(input("Enter the grade: "))
            grade_book.add_grade(assignment_name, grade)
        elif option == "2":
            grade_book.print_grades()
        elif option == "3":
            grade_book.print_average_grade()
        elif option == "4":
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
