## Exercise
Convert a numeric value from one data type to another

## Reference Solution
```python
def convert_numeric_types():
    # Integer to Float
    int_value = 5
    float_value = float(int_value)
    print(f"Integer {int_value} converted to Float: {float_value}")

    # Float to Integer
    float_value = 3.14
    int_value = int(float_value)
    print(f"Float {float_value} converted to Integer: {int_value}")

    # Integer to Double
    int_value = 5
    double_value = float(int_value)
    print(f"Integer {int_value} converted to Double: {double_value}")

    # Double to Float
    double_value = 3.14159265359
    float_value = float(double_value)
    print(f"Double {double_value} converted to Float: {float_value}")

    # Long to Integer
    long_value = 1234567890
    int_value = int(long_value)
    print(f"Long {long_value} converted to Integer: {int_value}")

convert_numeric_types()
```
