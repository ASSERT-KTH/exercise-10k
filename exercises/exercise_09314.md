## Exercise
Create a function that takes a string as input and returns the string converted to zigzag pattern.

def convert_to_zigzag(s: str, num_rows: int) -> str:
    """
    Converts a string to a zigzag pattern.

    Args:
    s (str): The input string.
    num_rows (int): The number of rows in the zigzag pattern.

    Returns:
    str: The string converted to the zigzag pattern.
    """

    # Handle edge cases
    if num_rows == 1 or num_rows >= len(s):
        return s

    # Initialize a list to store the characters in each row
    rows = [''] * num_rows

    # Initialize variables to keep track of the current row and direction
    row, step = 0, 1

    # Iterate over each character in the string
    for char in s:
        # Add the character to the current row
        rows[row] += char

        # Update the current row and direction
        if row == 0:
            step = 1
        elif row == num_rows - 1:
            step = -1
        row += step

    # Join the characters in each row to form the zigzag pattern
    return ''.join(rows)

if __name__ == "__main__":

    num_rows = 3
    print(convert_to_zigzag(s, num_rows))
