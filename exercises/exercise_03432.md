## Exercise
Write a function that takes a string and an integer as input and returns the cyclically rotated string

## Reference Solution
```python
def cyclic_rotate(input_str: str, num_positions: int) -> str:
    """
    Rotate a string cyclically by a specified number of positions.

    Args:
        input_str (str): The string to be rotated.
        num_positions (int): The number of positions to rotate the string.

    Returns:
        str: The rotated string.
    """
    num_positions = num_positions % len(input_str)  # Handle cases where num_positions is greater than the string length
    return input_str[-num_positions:] + input_str[:-num_positions]

# Example usage
def main():
    input_str = "abcdef"
    num_positions = 2
    rotated_str = cyclic_rotate(input_str, num_positions)
    print(f"Rotating '{input_str}' by {num_positions} positions: '{rotated_str}'")

if __name__ == "__main__":
    main()
```
