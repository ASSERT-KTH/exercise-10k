## Exercise
Write a program that takes two numbers as input and outputs their greatest common divisor.

## Reference Solution
```python
import math

def calculate_gcd(a, b):
    """
    Calculate the greatest common divisor of two numbers using the Euclidean algorithm.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The greatest common divisor of a and b.
    """
    while b != 0:
        a, b = b, a % b
    return abs(a)

def main():
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))

    gcd = calculate_gcd(num1, num2)

    print(f"The greatest common divisor of {num1} and {num2} is {gcd}.")

if __name__ == "__main__":
    main()
```
