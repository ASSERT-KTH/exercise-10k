## Exercise
Create a program that converts currency from one unit to another (USD to EUR, GBP to USD).

## Reference Solution
```python
from forex_python.converter import CurrencyRates, CurrencyCodes

def convert_currency(from_currency, to_currency, amount):
    """
    Converts currency from one unit to another.

    Args:
        from_currency (str): The currency to convert from (e.g. 'USD')
        to_currency (str): The currency to convert to (e.g. 'EUR')
        amount (float): The amount to convert

    Returns:
        float: The converted amount
    """
    c = CurrencyRates()
    result = c.convert(from_currency, to_currency, amount)
    return result

def get_currency_symbol(currency_code):
    """
    Gets the symbol for a given currency code.

    Args:
        currency_code (str): The currency code (e.g. 'USD')

    Returns:
        str: The currency symbol (e.g. '$')
    """
    cc = CurrencyCodes()
    symbol = cc.get_symbol(currency_code)
    return symbol

def main():
    print("Currency Converter")
    print("------------------")

    from_currency = input("Enter the currency to convert from (e.g. USD): ")
    to_currency = input("Enter the currency to convert to (e.g. EUR): ")
    amount = float(input("Enter the amount to convert: "))

    result = convert_currency(from_currency, to_currency, amount)
    from_symbol = get_currency_symbol(from_currency)
    to_symbol = get_currency_symbol(to_currency)

    print(f"{amount} {from_symbol} ({from_currency}) is equal to {result} {to_symbol} ({to_currency})")

if __name__ == "__main__":
    main()
```
