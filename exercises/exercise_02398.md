## Exercise
Write a program that generates the Sierpinski triangle using a while loop.

## Reference Solution
```python
import matplotlib.pyplot as plt
import numpy as np

def draw_triangle(points, color, ax):
    x = [points[0][0], points[1][0], points[2][0], points[0][0]]
    y = [points[0][1], points[1][1], points[2][1], points[0][1]]
    ax.plot(x, y, color=color)

def sierpinski(points, degree, ax):
    colormap = plt.cm.get_cmap("jet")
    draw_triangle(points, colormap(degree), ax)
    if degree > 0:
        p1, p2, p3 = points
        mid12 = ( (p1[0]+p2[0])/2, (p1[1]+p2[1])/2)
        mid23 = ( (p2[0]+p3[0])/2, (p2[1]+p3[1])/2)
        mid31 = ( (p3[0]+p1[0])/2, (p3[1]+p1[1])/2)
        sierpinski([p1, mid12, mid31], degree-1, ax)
        sierpinski([p2, mid23, mid12], degree-1, ax)
        sierpinski([p3, mid31, mid23], degree-1, ax)

def main():
    fig, ax = plt.subplots()
    my_points = [[-1, -1], [1, -1], [0, 1]]
    sierpinski(my_points, 6, ax)
    ax.set_aspect('equal')
    plt.axis('off')
    plt.show()

if __name__ == "__main__":
    main()
```
