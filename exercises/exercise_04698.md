## Exercise
Create a function that checks the strength of a given password by looking for a mix of uppercase and lowercase letters, numbers, and special characters.

## Reference Solution
```python
import re

def check_password_strength(password: str) -> dict:
    """
    Checks the strength of a given password.

    Args:
        password (str): The password to check.

    Returns:
        dict: A dictionary containing the results of the password strength check.
    """

    # Initialize the results dictionary
    results = {
        "has_uppercase": False,
        "has_lowercase": False,
        "has_numbers": False,
        "has_special_chars": False,
        "meets_length_requirement": False,
        "strong_password": False
    }

    # Check for uppercase letters
    if re.search("[A-Z]", password):
        results["has_uppercase"] = True

    # Check for lowercase letters
    if re.search("[a-z]", password):
        results["has_lowercase"] = True

    # Check for numbers
    if re.search("[0-9]", password):
        results["has_numbers"] = True

    # Check for special characters
    if re.search("[^A-Za-z0-9]", password):
        results["has_special_chars"] = True

    # Check if password meets the length requirement (minimum 8 characters)
    if len(password) >= 8:
        results["meets_length_requirement"] = True

    # Check if password is strong (meets all requirements)
    if (results["has_uppercase"] and
        results["has_lowercase"] and
        results["has_numbers"] and
        results["has_special_chars"] and
        results["meets_length_requirement"]):
        results["strong_password"] = True

    return results


# Example usage
def main():
    password = "P@ssw0rd"
    results = check_password_strength(password)

    print("Password Strength Check Results:")
    for key, value in results.items():
        print(f"{key}: {value}")


if __name__ == "__main__":
    main()
```
