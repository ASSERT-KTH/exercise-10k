## Exercise
Write a program to demonstrate the principles of object-oriented programming (OOP) using classes and objects.

## Reference Solution
```python
# Define a class called "Vehicle"
class Vehicle:
    def __init__(self, brand, model, year):
        """
        Initialize a Vehicle object.
        
        Args:
            brand (str): The brand of the vehicle.
            model (str): The model of the vehicle.
            year (int): The year of the vehicle.
        """
        self.brand = brand
        self.model = model
        self.year = year
        self.mileage = 0

    def drive(self, miles):
        """
        Increase the mileage of the vehicle.
        
        Args:
            miles (int): The number of miles to drive.
        """
        self.mileage += miles

    def describe_vehicle(self):
        """
        Print a description of the vehicle.
        """
        print(f"This vehicle is a {self.year} {self.brand} {self.model} with {self.mileage} miles.")


# Define a subclass called "Car" that inherits from "Vehicle"
class Car(Vehicle):
    def __init__(self, brand, model, year, doors):
        """
        Initialize a Car object.
        
        Args:
            brand (str): The brand of the car.
            model (str): The model of the car.
            year (int): The year of the car.
            doors (int): The number of doors on the car.
        """
        super().__init__(brand, model, year)
        self.doors = doors

    def describe_vehicle(self):
        """
        Print a description of the car.
        """
        super().describe_vehicle()
        print(f"It has {self.doors} doors.")


# Define a subclass called "Truck" that inherits from "Vehicle"
class Truck(Vehicle):
    def __init__(self, brand, model, year, capacity):
        """
        Initialize a Truck object.
        
        Args:
            brand (str): The brand of the truck.
            model (str): The model of the truck.
            year (int): The year of the truck.
            capacity (int): The capacity of the truck.
        """
        super().__init__(brand, model, year)
        self.capacity = capacity

    def describe_vehicle(self):
        """
        Print a description of the truck.
        """
        super().describe_vehicle()
        print(f"It has a capacity of {self.capacity} tons.")


# Create objects and demonstrate OOP principles
def main():
    # Create a Vehicle object
    vehicle = Vehicle("Toyota", "Corolla", 2015)
    vehicle.describe_vehicle()
    vehicle.drive(100)
    vehicle.describe_vehicle()

    print()

    # Create a Car object
    car = Car("Honda", "Civic", 2020, 4)
    car.describe_vehicle()
    car.drive(50)
    car.describe_vehicle()

    print()

    # Create a Truck object
    truck = Truck("Ford", "F-150", 2018, 2)
    truck.describe_vehicle()
    truck.drive(200)
    truck.describe_vehicle()


if __name__ == "__main__":
    main()
```
