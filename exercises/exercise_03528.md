## Exercise
Write a program that searches for a character in a given string using a while loop.

## Reference Solution
```python
def search_character(string, char):
    """
    Searches for a character in a given string using a while loop.

    Args:
        string (str): The input string to search in.
        char (str): The character to search for.

    Returns:
        int: The index of the character if found, -1 otherwise.
    """
    index = 0
    while index < len(string):
        if string[index] == char:
            return index
        index += 1
    return -1


def main():
    string = input("Enter a string: ")
    char = input("Enter a character to search for: ")

    result = search_character(string, char)

    if result != -1:
        print(f"The character '{char}' is found at index {result}.")
    else:
        print(f"The character '{char}' is not found in the string.")


if __name__ == "__main__":
    main()
```
