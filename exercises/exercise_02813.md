## Exercise
Write a program to ask the user for two colors and print out the resulting color when they are mixed.

## Reference Solution
```python
# Define a dictionary to map color combinations to their resulting colors
color_mixing = {
    ("red", "blue"): "purple",
    ("blue", "red"): "purple",
    ("red", "yellow"): "orange",
    ("yellow", "red"): "orange",
    ("blue", "yellow"): "green",
    ("yellow", "blue"): "green",
}

def get_color(prompt):
    """Get a color from the user."""
    while True:
        color = input(prompt).lower()
        if color in ["red", "blue", "yellow"]:
            return color
        print("Invalid color. Please enter 'red', 'blue', or 'yellow'.")

def mix_colors(color1, color2):
    """Mix two colors and return the resulting color."""
    return color_mixing.get((color1, color2), "unknown")

def main():
    # Get two colors from the user
    color1 = get_color("Enter the first color (red, blue, or yellow): ")
    color2 = get_color("Enter the second color (red, blue, or yellow): ")

    # Mix the colors and print the resulting color
    resulting_color = mix_colors(color1, color2)
    print(f"When you mix {color1} and {color2}, you get {resulting_color}.")

if __name__ == "__main__":
    main()
```
