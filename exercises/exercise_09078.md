## Exercise
Create a program that asks the user a series of survey questions and prints out the results using if statements

## Reference Solution
```python
# Define the survey questions and answers
def survey():
    print("Welcome to our survey! Please answer the following questions:")
    
    # Question 1: How old are you?
    age = int(input("How old are you? "))
    
    # Question 2: What is your favorite color?
    color = input("What is your favorite color? ")
    
    # Question 3: Do you like reading books?
    reading = input("Do you like reading books? (yes/no) ")
    
    # Question 4: What is your favorite hobby?
    hobby = input("What is your favorite hobby? ")
    
    # Print out the results
    print("\nHere are your survey results:")
    print(f"Age: {age}")
    print(f"Favorite color: {color}")
    
    # Use if statement to determine if the user likes reading books
    if reading.lower() == "yes":
        print("You enjoy reading books.")
    elif reading.lower() == "no":
        print("You do not enjoy reading books.")
    else:
        print("Invalid response.")
        
    # Use if statement to determine the user's favorite hobby
    if hobby.lower() == "sports":
        print("You enjoy playing sports.")
    elif hobby.lower() == "music":
        print("You enjoy listening to music.")
    elif hobby.lower() == "hiking":
        print("You enjoy hiking outdoors.")
    else:
        print(f"Your favorite hobby is {hobby}.")
        
    # Use if statement to determine the user's age group
    if age < 18:
        print("You are a minor.")
    elif age < 65:
        print("You are an adult.")
    else:
        print("You are a senior.")

# Call the survey function
survey()
```
