Write a function that takes two strings as input and returns True if the strings are equal and False otherwise

def are_strings_equal(str1: str, str2: str) -> bool:
    """
    Checks if two input strings are equal.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        bool: True if the strings are equal, False otherwise.
    """
    return str1 == str2

# Example usage:
def main():
    print(are_strings_equal("hello", "hello"))  # Expected output: True
    print(are_strings_equal("hello", "world"))  # Expected output: False

if __name__ == "__main__":
    main()
