## Exercise
Write a program that creates a text-based adventure game where the user can navigate through a virtual world and interact with objects and characters.

## Reference Solution
```python
# Text-Based Adventure Game

# Game Description
print("Welcome to the land of Eldoria, a mystical realm of wonder and danger.")
print("You are a brave adventurer seeking fortune and glory in these untamed lands.")
print("Your journey begins in the town of Willowdale, where rumors of a legendary treasure have drawn you and many other adventurers.")

# Game Data
rooms = {
    'town_square': {
        'description': 'You are in the town square. There is a shop to the north and a forest to the east.',
        'items': ['sword', 'shield'],
        'exits': {'north': 'shop', 'east': 'forest'}
    },
    'shop': {
        'description': 'You are in the shop. There is a merchant who will sell you items.',
        'items': ['health_potion', 'armor'],
        'exits': {'south': 'town_square'}
    },
    'forest': {
        'description': 'You are in the forest. There is a clearing to the north and a cave to the west.',
        'items': ['berries', 'wood'],
        'exits': {'north': 'clearing', 'west': 'cave'}
    },
    'clearing': {
        'description': 'You are in the clearing. There is a treasure chest in the center of the room.',
        'items': ['treasure_chest'],
        'exits': {'south': 'forest'}
    },
    'cave': {
        'description': 'You are in the cave. There is a dragon guarding the exit.',
        'items': [],
        'exits': {'east': 'forest'}
    }
}

player_room = 'town_square'
player_inventory = []

# Game Loop
while True:
    # Print current room description
    print("\n" + rooms[player_room]['description'])

    # Print available items
    if rooms[player_room]['items']:
        print("You see the following items: " + ', '.join(rooms[player_room]['items']))

    # Get player input
    action = input("\nWhat do you want to do? (type 'go' to move, 'take' to pick up an item, 'inventory' to view your inventory, 'quit' to exit the game) ").lower()

    # Handle player input
    if action == 'quit':
        print("Thanks for playing!")
        break
    elif action == 'inventory':
        if player_inventory:
            print("You are carrying: " + ', '.join(player_inventory))
        else:
            print("You are not carrying anything.")
    elif action.startswith('go '):
        direction = action.split(' ')[1]
        if direction in rooms[player_room]['exits']:
            player_room = rooms[player_room]['exits'][direction]
        else:
            print("You cannot go that way.")
    elif action.startswith('take '):
        item = action.split(' ')[1]
        if item in rooms[player_room]['items']:
            player_inventory.append(item)
            rooms[player_room]['items'].remove(item)
            print("You took the " + item + ".")
        else:
            print("There is no " + item + " here.")
    else:
        print("Invalid command. Please try again.")
```
