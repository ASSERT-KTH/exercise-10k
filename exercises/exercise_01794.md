Write a function to find the union of two dictionaries and return a new dictionary containing all key-value pairs from both dictionaries.

def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A new dictionary containing all key-value pairs from both input dictionaries.
    """
    # Use dictionary unpacking to merge the two dictionaries
    merged_dict = {**dict1, **dict2}
    
    return merged_dict

# Example usage

dict2 = {'c': 3, 'd': 4}
