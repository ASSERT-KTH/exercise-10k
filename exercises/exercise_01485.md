## Exercise
Write a program that demonstrates the precision of floating-point numbers in programming.

## Reference Solution
```python
def floating_point_precision_demo():
    """
    Demonstrates the precision of floating-point numbers.
    """
    # Initialize two floating-point numbers
    num1 = 0.1
    num2 = 0.2

    # Calculate the sum of the two numbers
    sum_result = num1 + num2

    # Print the result
    print(f"Sum: {sum_result}")

    # Check if the sum is equal to the expected result (0.3)
    print(f"Is sum equal to 0.3? {sum_result == 0.3}")

    # Initialize two floating-point numbers with a large difference in magnitude
    num3 = 1e16
    num4 = 1.0

    # Calculate the sum of the two numbers
    sum_result_large = num3 + num4

    # Print the result
    print(f"Sum (large numbers): {sum_result_large}")

    # Check if the sum is equal to the expected result (1e16)
    print(f"Is sum (large numbers) equal to 1e16? {sum_result_large == 1e16}")

if __name__ == "__main__":
    floating_point_precision_demo()
```
