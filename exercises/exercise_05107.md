Work with date and time data types to perform calculations and format output.

```python
from datetime import datetime, timedelta
```

```python
# Create a date and time object
now = datetime.now()

# Create a date object
date_only = datetime(2022, 12, 25)
print(date_only)

# Create a time object
time_only = datetime(2022, 1, 1, 12, 30, 0)
print(time_only)
```

```python
# Calculate the difference between two dates
date1 = datetime(2022, 1, 1)
date2 = datetime(2022, 1, 15)
difference = date2 - date1
print(difference)

# Calculate the difference between two times
time1 = datetime(2022, 1, 1, 12, 0, 0)
time2 = datetime(2022, 1, 1, 13, 30, 0)
difference = time2 - time1
print(difference)
```

```python
# Format date and time output
now = datetime.now()
print(now.strftime("%Y-%m-%d %H:%M:%S"))

# Format date output
date_only = datetime(2022, 12, 25)
print(date_only.strftime("%B %d, %Y"))

# Format time output
time_only = datetime(2022, 1, 1, 12, 30, 0)
print(time_only.strftime("%I:%M %p"))
```

```python
# Add a time interval to a date and time object
now = datetime.now()
future_date = now + timedelta(days=30)
print(future_date)

# Subtract a time interval from a date and time object
now = datetime.now()
past_date = now - timedelta(days=30)
print(past_date)
```

```python
from datetime import datetime, timedelta

def main():
    # Create a date and time object
    now = datetime.now()
    print("Current Date and Time: ", now)

    # Calculate the difference between two dates
    date1 = datetime(2022, 1, 1)
    date2 = datetime(2022, 1, 15)
    difference = date2 - date1
    print("Difference between dates: ", difference)

    # Format date and time output
    print("Formatted Date and Time: ", now.strftime("%Y-%m-%d %H:%M:%S"))

    # Add a time interval to a date and time object
    future_date = now + timedelta(days=30)
    print("Future Date: ", future_date)

if __name__ == "__main__":
    main()
```
