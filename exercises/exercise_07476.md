## Exercise
Write a program that performs a given set operation on two sets

## Reference Solution
```python
def perform_set_operation(set1, set2, operation):
    """
    Perform a set operation on two sets.

    Args:
        set1 (set): The first set.
        set2 (set): The second set.
        operation (str): The set operation to perform. Can be 'union', 'intersection', 'difference', or 'symmetric_difference'.

    Returns:
        set: The result of the set operation.

    Raises:
        ValueError: If the operation is not one of the supported operations.
    """
    if operation == 'union':
        return set1.union(set2)
    elif operation == 'intersection':
        return set1.intersection(set2)
    elif operation == 'difference':
        return set1.difference(set2)
    elif operation == 'symmetric_difference':
        return set1.symmetric_difference(set2)
    else:
        raise ValueError("Unsupported operation. Supported operations are 'union', 'intersection', 'difference', and 'symmetric_difference'.")

def main():
    # Example usage
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}

    operations = {
        'union': 'Union',
        'intersection': 'Intersection',
        'difference': 'Difference',
        'symmetric_difference': 'Symmetric Difference'
    }

    for operation, name in operations.items():
        result = perform_set_operation(set1, set2, operation)
        print(f"{name}: {result}")

if __name__ == "__main__":
    main()
```
