## Exercise
Create a program that takes the side length of a cube as input and returns its surface area.

## Reference Solution
```python
def cube_surface_area(side_length):
    """
    Calculate the surface area of a cube given its side length.

    Args:
        side_length (float): The length of one side of the cube.

    Returns:
        float: The surface area of the cube.
    """
    if side_length <= 0:
        raise ValueError("Side length must be a positive number")

    # A cube has 6 faces, each with an area of side_length^2
    surface_area = 6 * (side_length ** 2)
    return surface_area

# Example usage:
def main():
    side_length = float(input("Enter the side length of the cube: "))
    try:
        surface_area = cube_surface_area(side_length)
        print(f"The surface area of the cube is {surface_area} square units.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
