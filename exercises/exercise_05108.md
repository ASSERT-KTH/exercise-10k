## Exercise
Define and use an enumeration data type to represent a set of named values.

## Reference Solution
```python
from enum import Enum

class DayOfWeek(Enum):
    MONDAY = 1
    TUESDAY = 2
    WEDNESDAY = 3
    THURSDAY = 4
    FRIDAY = 5
    SATURDAY = 6
    SUNDAY = 7

# Using the enumeration
day = DayOfWeek.MONDAY
print(day)  # Output: DayOfWeek.MONDAY
print(day.name)  # Output: MONDAY
print(day.value)  # Output: 1

# Iterating over the enumeration
for day in DayOfWeek:
    print(f"{day.name}: {day.value}")
```

from enum import Enum

class StatusCode(Enum):
    SUCCESS = 200
    NOT_FOUND = 404
    INTERNAL_SERVER_ERROR = 500

def handle_request(status_code):
    if status_code == StatusCode.SUCCESS:
        print("Request successful")
    elif status_code == StatusCode.NOT_FOUND:
        print("Resource not found")
    elif status_code == StatusCode.INTERNAL_SERVER_ERROR:
        print("Internal server error")

# Using the function
handle_request(StatusCode.SUCCESS)  # Output: Request successful
handle_request(StatusCode.NOT_FOUND)  # Output: Resource not found
