## Exercise
Create a program that takes a dictionary containing student information, including names, grades, and test scores. The program should calculate the average test score for each grade level and store the results in a new dictionary. The program should then print out the dictionary with the average test scores.

## Reference Solution
```python
def calculate_average_test_scores(student_info):
    """
    Calculate the average test score for each grade level.

    Args:
        student_info (dict): A dictionary containing student information.
            Each key is a student's name, and each value is another dictionary
            containing the student's grade and test score.

    Returns:
        dict: A dictionary where each key is a grade level and each value is
            the average test score for that grade level.
    """

    # Initialize an empty dictionary to store the sum of test scores and the count of students for each grade level
    grade_info = {}

    # Iterate over each student in the student_info dictionary
    for student, info in student_info.items():
        # Get the grade and test score of the current student
        grade = info['grade']
        test_score = info['test_score']

        # If the grade is not already in the grade_info dictionary, add it with a list to store the sum of test scores and the count of students
        if grade not in grade_info:
            # The list will store the sum of test scores at index 0 and the count of students at index 1
            grade_info[grade] = [0, 0]

        # Add the test score to the sum of test scores for the current grade and increment the count of students
        grade_info[grade][0] += test_score
        grade_info[grade][1] += 1

    # Initialize an empty dictionary to store the average test scores for each grade level
    average_test_scores = {}

    # Iterate over each grade level in the grade_info dictionary
    for grade, info in grade_info.items():
        # Calculate the average test score by dividing the sum of test scores by the count of students
        average_test_score = info[0] / info[1]

        # Store the average test score in the average_test_scores dictionary
        average_test_scores[grade] = average_test_score

    # Return the average_test_scores dictionary
    return average_test_scores


def main():
    # Example usage:
    student_info = {
        'Alice': {'grade': 9, 'test_score': 85},
        'Bob': {'grade': 9, 'test_score': 90},
        'Charlie': {'grade': 10, 'test_score': 78},
        'David': {'grade': 10, 'test_score': 92},
        'Eve': {'grade': 11, 'test_score': 88},
        'Frank': {'grade': 11, 'test_score': 76}
    }

    average_test_scores = calculate_average_test_scores(student_info)

    # Print the average_test_scores dictionary
    print("Average Test Scores:")
    for grade, average_test_score in average_test_scores.items():
        print(f"Grade {grade}: {average_test_score}")


if __name__ == "__main__":
    main()
```
