## Exercise
Create a program that plans a trip. The user can input their destination, departure date, and return date, and the program displays the trip details.

## Reference Solution
```python
class TripPlanner:
    def __init__(self):
        self.trips = {}

    def plan_trip(self):
        """Plan a trip by inputting destination, departure date, and return date."""
        destination = input("Enter your destination: ")
        departure_date = input("Enter your departure date (YYYY-MM-DD): ")
        return_date = input("Enter your return date (YYYY-MM-DD): ")

        trip_details = {
            "Destination": destination,
            "Departure Date": departure_date,
            "Return Date": return_date
        }

        self.trips[len(self.trips) + 1] = trip_details

        print("\nTrip Details:")
        for key, value in trip_details.items():
            print(f"{key}: {value}")

    def view_trips(self):
        """View all planned trips."""
        if not self.trips:
            print("No trips planned.")
        else:
            for trip_id, trip_details in self.trips.items():
                print(f"\nTrip {trip_id} Details:")
                for key, value in trip_details.items():
                    print(f"{key}: {value}")


def main():
    trip_planner = TripPlanner()

    while True:
        print("\nTrip Planner Menu:")
        print("1. Plan a trip")
        print("2. View all trips")
        print("3. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            trip_planner.plan_trip()
        elif choice == "2":
            trip_planner.view_trips()
        elif choice == "3":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
