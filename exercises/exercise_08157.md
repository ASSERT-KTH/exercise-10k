## Exercise
Create an array of 5 integers and search for a specific element.

## Reference Solution
```python
def search_element(array, target):
    """
    Search for a specific element in the array.

    Args:
        array (list): The list of integers to search in.
        target (int): The element to search for.

    Returns:
        int: The index of the target element if found, -1 otherwise.
    """
    try:
        # Use the index method to find the index of the target element
        index = array.index(target)
        return index
    except ValueError:
        # If the target element is not found, return -1
        return -1


# Create an array of 5 integers
array = [10, 20, 30, 40, 50]

# Search for a specific element
target = 30
index = search_element(array, target)

if index != -1:
    print(f"Element {target} found at index {index}.")
else:
    print(f"Element {target} not found in the array.")
```
