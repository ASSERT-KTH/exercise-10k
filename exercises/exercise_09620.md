## Exercise
Write a program that generates a random trivia question based on user input

```python
import random

# Dictionary to store questions, options, and answers
trivia_questions = {
    "History": {
        "What was the name of the first president of the United States?": {
            "A": "George Washington",
            "B": "Thomas Jefferson",
            "C": "Abraham Lincoln",
            "D": "Franklin D. Roosevelt",
            "Answer": "A"
        },
        "Which ancient civilization built the Great Pyramid of Giza?": {
            "A": "Egyptians",
            "B": "Greeks",
            "C": "Romans",
            "D": "Mesopotamians",
            "Answer": "A"
        }
    },
    "Science": {
        "What is the largest planet in our solar system?": {
            "A": "Earth",
            "B": "Saturn",
            "C": "Jupiter",
            "D": "Uranus",
            "Answer": "C"
        },
        "What is the process by which plants make their own food?": {
            "A": "Respiration",
            "B": "Photosynthesis",
            "C": "Decomposition",
            "D": "Fermentation",
            "Answer": "B"
        }
    },
    "Sports": {
        "Which team has won the most Super Bowl titles in the NFL?": {
            "A": "New England Patriots",
            "B": "Dallas Cowboys",
            "C": "Pittsburgh Steelers",
            "D": "Green Bay Packers",
            "Answer": "A"
        },
        "Who is the all-time leading scorer in the NBA?": {
            "A": "Michael Jordan",
            "B": "Kareem Abdul-Jabbar",
            "C": "LeBron James",
            "D": "Kevin Durant",
            "Answer": "B"
        }
    }
}

def generate_trivia_question(category):
    """Generate a random trivia question based on the user's category choice"""
    questions = list(trivia_questions[category].keys())
    question = random.choice(questions)
    options = trivia_questions[category][question]
    return question, options

def main():
    print("Welcome to the Trivia Question Generator!")
    print("Choose a category:")
    print("1. History")
    print("2. Science")
    print("3. Sports")
    
    choice = input("Enter the number of your chosen category: ")
    
    if choice == "1":
        category = "History"
    elif choice == "2":
        category = "Science"
    elif choice == "3":
        category = "Sports"
    else:
        print("Invalid choice. Please try again.")
        return
    
    question, options = generate_trivia_question(category)
    print("\nHere's your trivia question:")
    print(question)
    for option, value in options.items():
        if option != "Answer":
            print(f"{option}: {value}")
    
    answer = input("\nEnter your answer (A, B, C, or D): ")
    if answer.upper() == options["Answer"]:
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {options['Answer']}.")

if __name__ == "__main__":
    main()
```
