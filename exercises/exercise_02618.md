## Exercise
Write a program that finds all keys in a dictionary that have unique values

## Reference Solution
```python
def find_unique_value_keys(dictionary):
    """
    Finds all keys in a dictionary that have unique values.

    Args:
        dictionary (dict): The input dictionary.

    Returns:
        dict: A dictionary containing keys with unique values.
    """
    # Create a dictionary to store the count of each value
    value_count = {}
    for key, value in dictionary.items():
        if value in value_count:
            value_count[value] += 1
        else:
            value_count[value] = 1

    # Create a dictionary to store keys with unique values
    unique_value_keys = {}
    for key, value in dictionary.items():
        if value_count[value] == 1:
            unique_value_keys[key] = value

    return unique_value_keys


# Example usage
if __name__ == "__main__":
    dictionary = {
        "key1": "value1",
        "key2": "value2",
        "key3": "value1",
        "key4": "value3",
        "key5": "value2",
    }

    unique_keys = find_unique_value_keys(dictionary)
    print("Keys with unique values:")
    for key, value in unique_keys.items():
        print(f"{key}: {value}")
```
