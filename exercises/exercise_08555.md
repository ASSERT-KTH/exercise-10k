## Exercise
Create a dictionary with 3 key-value pairs and check if a value exists.

## Reference Solution
```python
def create_and_check_dict():
    # Create a dictionary with 3 key-value pairs
    my_dict = {
        "key1": "value1",
        "key2": "value2",
        "key3": "value3"
    }

    # Value to check for
    value_to_check = "value2"

    # Check if the value exists
    if value_to_check in my_dict.values():
        print(f"The value '{value_to_check}' exists in the dictionary.")
    else:
        print(f"The value '{value_to_check}' does not exist in the dictionary.")

# Example usage
if __name__ == "__main__":
    create_and_check_dict()
```
