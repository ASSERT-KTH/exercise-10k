## Exercise
Develop a program to play Rock, Paper, Scissors using conditional statements to determine the winner

## Reference Solution
```python
import random

def get_player_choice(player_name):
    """Get the player's choice"""
    while True:
        choice = input(f"{player_name}, please enter your choice (rock/paper/scissors): ")
        if choice.lower() in ["rock", "paper", "scissors"]:
            return choice.lower()
        print("Invalid choice. Please try again.")

def get_computer_choice():
    """Get the computer's random choice"""
    choices = ["rock", "paper", "scissors"]
    return random.choice(choices)

def determine_winner(player1_choice, player2_choice):
    """Determine the winner based on the players' choices"""
    if player1_choice == player2_choice:
        return "It's a tie!"
    elif (player1_choice == "rock" and player2_choice == "scissors") or \
         (player1_choice == "paper" and player2_choice == "rock") or \
         (player1_choice == "scissors" and player2_choice == "paper"):
        return "Player 1 wins!"
    else:
        return "Player 2 wins!"

def play_game():
    """Play the Rock, Paper, Scissors game"""
    player1_name = input("Enter Player 1 name: ")
    player2_name = "Computer"

    player1_choice = get_player_choice(player1_name)
    player2_choice = get_computer_choice()

    print(f"\n{player1_name} chose: {player1_choice}")
    print(f"{player2_name} chose: {player2_choice}")

    winner = determine_winner(player1_choice, player2_choice)
    print(f"\n{winner}")

def main():
    play_game()

if __name__ == "__main__":
    main()
```
