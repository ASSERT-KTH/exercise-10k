## Exercise
Write a program that calculates the time it takes to travel a certain distance given the speed.

## Reference Solution
```python
def calculate_travel_time(distance, speed):
    """
    Calculate the time it takes to travel a certain distance given the speed.

    Args:
    distance (float): The distance to travel (in kilometers or miles).
    speed (float): The speed of travel (in kilometers per hour or miles per hour).

    Returns:
    float: The time it takes to travel the distance (in hours).
    """
    if speed <= 0:
        raise ValueError("Speed must be greater than zero.")
    if distance < 0:
        raise ValueError("Distance cannot be negative.")
    travel_time = distance / speed
    return travel_time

def main():
    # Get user input
    distance = float(input("Enter the distance to travel (in kilometers or miles): "))
    speed = float(input("Enter the speed of travel (in kilometers per hour or miles per hour): "))

    # Calculate travel time
    try:
        travel_time = calculate_travel_time(distance, speed)
    except ValueError as e:
        print(e)
        return

    # Print result
    print(f"The time it takes to travel {distance} kilometers/miles at a speed of {speed} kilometers/miles per hour is {travel_time:.2f} hours.")

if __name__ == "__main__":
    main()
```
