## Exercise
Write a program to calculate the square root of a number using a while loop.

## Reference Solution
```python
def calculate_square_root(number, precision=0.00001):
    """
    Calculate the square root of a number using a while loop.

    Args:
        number (float): The number to calculate the square root of.
        precision (float): The desired precision of the result. Defaults to 0.00001.

    Returns:
        float: The approximate square root of the number.
    """
    if number < 0:
        raise ValueError("Cannot calculate square root of negative number")

    # Initial guess
    guess = number / 2.0

    while True:
        # Calculate a better guess
        better_guess = (guess + number / guess) / 2.0

        # Check if the guess is good enough
        if abs(guess - better_guess) < precision:
            return better_guess

        # Update the guess
        guess = better_guess

# Example usage
if __name__ == "__main__":
    number = 9.0
    square_root = calculate_square_root(number)
    print(f"The square root of {number} is approximately {square_root}")
```
