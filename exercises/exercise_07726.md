## Exercise
Group a list of strings by their anagrams.

## Reference Solution
```python
from typing import List

def group_anagrams(strs: List[str]) -> List[List[str]]:
    """
    Group a list of strings by their anagrams.

    Args:
    strs (List[str]): A list of strings.

    Returns:
    List[List[str]]: A list of lists, where each sublist contains strings that are anagrams of each other.
    """
    anagrams = {}
    for s in strs:
        # Sort the characters in the string to create a key for the dictionary
        key = "".join(sorted(s))
        # If the key is not in the dictionary, add it with an empty list as its value
        if key not in anagrams:
            anagrams[key] = []
        # Add the string to the list of anagrams for the given key
        anagrams[key].append(s)
    # Return the list of anagrams
    return list(anagrams.values())

# Example usage:
strs = ["eat", "tea", "tan", "ate", "nat", "bat"]
print(group_anagrams(strs))
```
