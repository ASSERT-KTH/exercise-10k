## Exercise
Create a function that calculates a student's grade based on their scores on several assignments.

## Reference Solution
```python
def calculate_grade(assignment_scores, assignment_weights):
    """
    Calculate a student's grade based on their scores on several assignments.

    Args:
        assignment_scores (dict): A dictionary of assignment names and scores.
        assignment_weights (dict): A dictionary of assignment names and weights.

    Returns:
        float: The student's grade as a percentage.
    """
    # Initialize total score and total weight
    total_score = 0
    total_weight = 0

    # Iterate over each assignment
    for assignment, score in assignment_scores.items():
        # Get the weight of the current assignment
        weight = assignment_weights.get(assignment)

        # If the assignment has a weight, add the weighted score to the total score
        if weight is not None:
            total_score += score * weight
            total_weight += weight

    # Calculate the grade as a percentage
    if total_weight == 0:
        return 0
    else:
        return (total_score / total_weight) * 100


# Example usage
if __name__ == "__main__":
    assignment_scores = {
        "Assignment 1": 90,
        "Assignment 2": 80,
        "Assignment 3": 95
    }

    assignment_weights = {
        "Assignment 1": 0.3,
        "Assignment 2": 0.2,
        "Assignment 3": 0.5
    }

    grade = calculate_grade(assignment_scores, assignment_weights)
    print(f"Student's grade: {grade:.2f}%")
```
