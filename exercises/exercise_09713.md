## Exercise
Write a program that takes two numbers as input and returns the quotient and remainder of their division.

## Reference Solution
```python
def divide_numbers(dividend, divisor):
    """
    This function takes two numbers as input and returns the quotient and remainder of their division.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        tuple: A tuple containing the quotient and remainder of the division.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    
    quotient = dividend // divisor
    remainder = dividend % divisor
    
    return quotient, remainder

def main():
    # Get user input
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))

    try:
        # Call the divide_numbers function
        quotient, remainder = divide_numbers(dividend, divisor)
        
        # Print the results
        print(f"The quotient of {dividend} divided by {divisor} is {quotient}")
        print(f"The remainder of {dividend} divided by {divisor} is {remainder}")
    except ZeroDivisionError as e:
        print(str(e))

if __name__ == "__main__":
    main()
```
