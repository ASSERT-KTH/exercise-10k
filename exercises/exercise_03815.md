## Exercise
Write a program to implement a binary search algorithm using a while loop.

## Reference Solution
```python
def binary_search(arr, target):
    """
    Perform a binary search on a sorted array.

    Args:
        arr (list): The sorted array to search.
        target (int): The target value to find.

    Returns:
        int: The index of the target value if found, -1 otherwise.
    """
    left = 0
    right = len(arr) - 1

    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1


# Example usage
arr = [2, 5, 8, 12, 16, 23, 38, 56, 72, 91]
target = 23

result = binary_search(arr, target)

if result != -1:
    print(f"Element {target} found at index {result}")
else:
    print(f"Element {target} not found in the array")
```
