## Exercise
Write a program that generates random phrases using a while loop.

## Reference Solution
```python
import random

# Define lists of subjects, verbs, and objects
subjects = ["cat", "dog", "bird", "fish", "turtle"]
verbs = ["ran", "jumped", "swam", "flew", "crawled"]
objects = ["around the house", "across the street", "in the park", "through the garden", "over the bridge"]

def generate_phrase():
    """Generate a random phrase"""
    subject = random.choice(subjects)
    verb = random.choice(verbs)
    object = random.choice(objects)
    return f"The {subject} {verb} {object}."

def main():
    """Main function"""
    print("Random Phrase Generator")
    print("------------------------")
    
    # Initialize a variable to control the loop
    generate_more = "y"
    
    while generate_more.lower() == "y":
        # Generate and print a random phrase
        print(generate_phrase())
        
        # Ask the user if they want to generate another phrase
        generate_more = input("Generate another phrase? (y/n): ")
        
        # Check if the user wants to exit
        if generate_more.lower() != "y" and generate_more.lower() != "n":
            print("Invalid input. Exiting program.")
            break

if __name__ == "__main__":
    main()
```
