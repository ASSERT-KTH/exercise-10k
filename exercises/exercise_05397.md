## Exercise
Create a function that checks if a given password meets certain requirements (e.g. length, characters).

## Reference Solution
```python
import re

def validate_password(password: str) -> bool:
    """
    Validate a password based on the following requirements:
    - Minimum length of 8 characters
    - At least one uppercase letter
    - At least one lowercase letter
    - At least one digit
    - At least one special character

    Args:
        password (str): The password to be validated

    Returns:
        bool: True if the password is valid, False otherwise
    """
    # Define the requirements
    requirements = {
        "min_length": 8,
        "uppercase": r"[A-Z]",
        "lowercase": r"[a-z]",
        "digit": r"\d",
        "special_char": r"[^A-Za-z0-9]"
    }

    # Check each requirement
    if len(password) < requirements["min_length"]:
        print("Password must be at least 8 characters long")
        return False

    if not re.search(requirements["uppercase"], password):
        print("Password must contain at least one uppercase letter")
        return False

    if not re.search(requirements["lowercase"], password):
        print("Password must contain at least one lowercase letter")
        return False

    if not re.search(requirements["digit"], password):
        print("Password must contain at least one digit")
        return False

    if not re.search(requirements["special_char"], password):
        print("Password must contain at least one special character")
        return False

    # If all requirements are met, the password is valid
    print("Password is valid")
    return True

# Example usage
if __name__ == "__main__":
    password = "P@ssw0rd"
    validate_password(password)
```
