## Exercise
Write a program that determines the day of the week based on the day number (1-7) using conditional statements.

## Reference Solution
```python
#include <iostream>

int main() {
    int dayNumber;

    std::cout << "Enter the day number: ";
    std::cin >> dayNumber;

    if (dayNumber == 1) {
        std::cout << "Monday" << std::endl;
    } else if (dayNumber == 2) {
        std::cout << "Tuesday" << std::endl;
    } else if (dayNumber == 3) {
        std::cout << "Wednesday" << std::endl;
    } else if (dayNumber == 4) {
        std::cout << "Thursday" << std::endl;
    } else if (dayNumber == 5) {
        std::cout << "Friday" << std::endl;
    } else if (dayNumber == 6) {
        std::cout << "Saturday" << std::endl;
    } else if (dayNumber == 7) {
        std::cout << "Sunday" << std::endl;
    } else {
        std::cout << "Invalid day number. Please enter a number between 1 and 7." << std::endl;
    }

    return 0;
}
```
