## Exercise
Create a function that takes two strings as arguments, 'rock', 'paper', or 'scissors', and returns the winner of the game

## Reference Solution

def rock_paper_scissors(player1, player2):
    """
    This function determines the winner of a rock paper scissors game.

    Args:
        player1 (str): The choice of the first player. Can be 'rock', 'paper', or 'scissors'.
        player2 (str): The choice of the second player. Can be 'rock', 'paper', or 'scissors'.

    Returns:
        str: The winner of the game. Can be 'player1', 'player2', or 'tie'.
    """

    # Define the winning conditions
    winning_conditions = {
        'rock': 'scissors',
        'paper': 'rock',
        'scissors': 'paper'
    }

    # Check if the inputs are valid
    if player1 not in winning_conditions or player2 not in winning_conditions:
        raise ValueError("Invalid input. Please enter 'rock', 'paper', or 'scissors'.")

    # Check if it's a tie
    if player1 == player2:
        return 'tie'

    # Determine the winner
    if winning_conditions[player1] == player2:
        return 'player1'
    else:
        return 'player2'

# Example usage:
print(rock_paper_scissors('rock', 'scissors'))  # Output: player1
print(rock_paper_scissors('paper', 'rock'))  # Output: player1
print(rock_paper_scissors('scissors', 'paper'))  # Output: player1
print(rock_paper_scissors('rock', 'rock'))  # Output: tie
