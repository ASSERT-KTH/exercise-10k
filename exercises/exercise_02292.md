## Exercise
Develop a program that allows the user to add items to a shopping list, remove items from the list, and view all items in the list.

## Reference Solution
```python
class ShoppingList:
    def __init__(self):
        self.shopping_list = []

    def add_item(self, item):
        """Add an item to the shopping list."""
        self.shopping_list.append(item)
        print(f"Item '{item}' added to the shopping list.")

    def remove_item(self, item):
        """Remove an item from the shopping list."""
        if item in self.shopping_list:
            self.shopping_list.remove(item)
            print(f"Item '{item}' removed from the shopping list.")
        else:
            print(f"Item '{item}' not found in the shopping list.")

    def view_list(self):
        """View all items in the shopping list."""
        if self.shopping_list:
            print("Shopping List:")
            for i, item in enumerate(self.shopping_list, start=1):
                print(f"{i}. {item}")
        else:
            print("Shopping list is empty.")

    def run(self):
        while True:
            print("\nShopping List Menu:")
            print("1. Add item to list")
            print("2. Remove item from list")
            print("3. View shopping list")
            print("4. Exit")
            
            choice = input("Enter your choice: ")

            if choice == "1":
                item = input("Enter item to add: ")
                self.add_item(item)
            elif choice == "2":
                item = input("Enter item to remove: ")
                self.remove_item(item)
            elif choice == "3":
                self.view_list()
            elif choice == "4":
                print("Exiting shopping list program.")
                break
            else:
                print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    shopping_list = ShoppingList()
    shopping_list.run()
```
