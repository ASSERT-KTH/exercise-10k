## Exercise
Write a program that takes in a time in one format and converts it to another format using conditional statements.

## Reference Solution
```python
def convert_time_format(time_str, input_format, output_format):
    """
    Converts a time string from one format to another.

    Args:
        time_str (str): The time string to convert.
        input_format (str): The format of the input time string.
        output_format (str): The desired output format.

    Returns:
        str: The converted time string.
    """

    # Parse the input time string
    if input_format == "12-hour":
        time_parts = time_str.split()
        hour, minute = map(int, time_parts[0].split(":"))
        am_pm = time_parts[1]
        if am_pm == "PM" and hour != 12:
            hour += 12
        elif am_pm == "AM" and hour == 12:
            hour = 0
    elif input_format == "24-hour":
        hour, minute = map(int, time_str.split(":"))
    elif input_format == "military":
        hour = int(time_str[:2])
        minute = int(time_str[2:])

    # Convert to the desired output format
    if output_format == "12-hour":
        if hour == 0:
            hour = 12
            am_pm = "AM"
        elif hour == 12:
            am_pm = "PM"
        elif hour > 12:
            hour -= 12
            am_pm = "PM"
        else:
            am_pm = "AM"
        return f"{hour:02d}:{minute:02d} {am_pm}"
    elif output_format == "24-hour":
        return f"{hour:02d}:{minute:02d}"
    elif output_format == "military":
        return f"{hour:02d}{minute:02d}"

def main():
    time_str = input("Enter a time string: ")
    input_format = input("Enter the input format (12-hour, 24-hour, military): ")
    output_format = input("Enter the desired output format (12-hour, 24-hour, military): ")

    converted_time = convert_time_format(time_str, input_format, output_format)
    print(f"Converted time: {converted_time}")

if __name__ == "__main__":
    main()
```
