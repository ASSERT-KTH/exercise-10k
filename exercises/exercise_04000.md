## Exercise
Create a program that finds the minimum window substring that contains all characters of another string.

```python
from collections import Counter

def min_window(s: str, t: str) -> str:
    """
    This function finds the minimum window substring that contains all characters of another string.
    
    Args:
        s (str): The string in which we need to find the minimum window substring.
        t (str): The string that contains all the characters that the minimum window substring should have.
        
    Returns:
        str: The minimum window substring if it exists, otherwise an empty string.
    """
    
    # Base case: If string t is longer than string s, return an empty string
    if len(t) > len(s):
        return ""
    
    # Create a counter for string t
    t_count = Counter(t)
    
    # Initialize variables to keep track of the minimum window substring
    required_chars = len(t_count)
    formed_chars = 0
    
    # Initialize variables for the window boundaries
    window_counts = {}
    min_window = (0, float('inf'))
    
    # Initialize the window boundaries
    left = 0
    
    # Traverse the string s
    for right in range(len(s)):
        # Add the character at the right boundary to the window counts
        character = s[right]
        window_counts[character] = window_counts.get(character, 0) + 1
        
        # If the added character is in t and its frequency in the window is equal to its frequency in t,
        # increment the formed_chars count
        if character in t_count and window_counts[character] == t_count[character]:
            formed_chars += 1
        
        # While the window contains all characters of t and the left boundary is not at the beginning of the string,
        # try to shrink the window from the left
        while left <= right and formed_chars == required_chars:
            # Update the minimum window substring if the current window is smaller
            if right - left + 1 < min_window[1] - min_window[0]:
                min_window = (left, right + 1)
            
            # Remove the character at the left boundary from the window counts
            character = s[left]
            window_counts[character] -= 1
            
            # If the removed character is in t and its frequency in the window is less than its frequency in t,
            # decrement the formed_chars count
            if character in t_count and window_counts[character] < t_count[character]:
                formed_chars -= 1
            
            # Move the left boundary to the right
            left += 1
    
    # Return the minimum window substring if it exists, otherwise an empty string
    return "" if min_window[1] == float('inf') else s[min_window[0]:min_window[1]]

# Example usage:
s = "ADOBECODEBANC"
t = "ABC"
print(min_window(s, t))  # Output: "BANC"
```
