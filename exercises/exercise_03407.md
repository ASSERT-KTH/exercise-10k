## Exercise
Create a program to add a new key-value pair to a given dictionary.

## Reference Solution
```python
def add_key_value_pair(dictionary, key, value):
    """
    Adds a new key-value pair to a given dictionary.

    Args:
        dictionary (dict): The dictionary to be updated.
        key (any): The key to be added.
        value (any): The value to be added.

    Returns:
        dict: The updated dictionary.
    """
    dictionary[key] = value
    return dictionary

# Example usage
def main():
    # Create a sample dictionary
    sample_dict = {"name": "John", "age": 30}

    # Print the original dictionary
    print("Original Dictionary:")
    print(sample_dict)

    # Add a new key-value pair
    new_key = "city"
    new_value = "New York"
    updated_dict = add_key_value_pair(sample_dict, new_key, new_value)

    # Print the updated dictionary
    print("\nUpdated Dictionary:")
    print(updated_dict)

if __name__ == "__main__":
    main()
```
