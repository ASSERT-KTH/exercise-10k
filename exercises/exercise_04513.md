## Exercise
Create a program that plays a quiz game against the user. The user has to answer questions and the program keeps track of the score.

## Reference Solution
```python
# Import the random module to shuffle questions
import random

# Define a dictionary to store questions, answers, and options
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Who painted the Mona Lisa?": {
        "A": "Leonardo da Vinci",
        "B": "Michelangelo",
        "C": "Raphael",
        "D": "Caravaggio",
        "answer": "A"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "answer": "B"
    },
    "What is the smallest country in the world?": {
        "A": "Vatican City",
        "B": "Monaco",
        "C": "Nauru",
        "D": "Tuvalu",
        "answer": "A"
    }
}

def play_quiz():
    # Initialize score
    score = 0
    
    # Shuffle questions
    question_list = list(questions.items())
    random.shuffle(question_list)
    
    # Iterate through questions
    for question, options in question_list:
        print(f"\nQuestion: {question}")
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        
        # Get user's answer
        user_answer = input("\nEnter your answer (A, B, C, D): ")
        
        # Check if user's answer is correct
        if user_answer.upper() == options["answer"]:
            print("Correct!")
            score += 1
        else:
            print(f"Incorrect. The correct answer is {options['answer']}.")
    
    # Print final score
    print(f"\nQuiz finished. Your final score is {score} out of {len(questions)}.")

# Run the quiz
play_quiz()
```
