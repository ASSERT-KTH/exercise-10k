## Exercise
Write a program that reads and writes data in a structured format

## Reference Solution
```python
import json

def read_data(file_path):
    """
    Reads data from a JSON file.

    Args:
        file_path (str): Path to the JSON file.

    Returns:
        dict: The data read from the file.
    """
    try:
        with open(file_path, 'r') as file:
            data = json.load(file)
            return data
    except FileNotFoundError:
        print(f"File not found: {file_path}")
        return None
    except json.JSONDecodeError:
        print(f"Invalid JSON format: {file_path}")
        return None

def write_data(file_path, data):
    """
    Writes data to a JSON file.

    Args:
        file_path (str): Path to the JSON file.
        data (dict): The data to be written.
    """
    try:
        with open(file_path, 'w') as file:
            json.dump(data, file, indent=4)
    except Exception as e:
        print(f"Error writing data: {e}")

# Example usage
if __name__ == "__main__":
    file_path = "data.json"
    data = {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }

    # Write data to file
    write_data(file_path, data)

    # Read data from file
    read_data_from_file = read_data(file_path)
    print(read_data_from_file)
```
