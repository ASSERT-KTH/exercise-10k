## Exercise
Create a program that manages an event calendar. The user can add or remove events from the calendar, and the program displays the upcoming events.

## Reference Solution
```python
import datetime

class Event:
    """Represents an event with a title, date, and description."""

    def __init__(self, title, date, description):
        self.title = title
        self.date = datetime.datetime.strptime(date, "%Y-%m-%d").date()
        self.description = description

    def __str__(self):
        return f"{self.title} - {self.date} - {self.description}"


class EventCalendar:
    """Manages a list of events."""

    def __init__(self):
        self.events = []

    def add_event(self, title, date, description):
        """Adds an event to the calendar."""
        event = Event(title, date, description)
        self.events.append(event)
        self.events.sort(key=lambda x: x.date)

    def remove_event(self, title):
        """Removes an event from the calendar."""
        self.events = [event for event in self.events if event.title != title]

    def display_upcoming_events(self):
        """Displays upcoming events."""
        today = datetime.date.today()
        upcoming_events = [event for event in self.events if event.date >= today]
        if upcoming_events:
            print("Upcoming Events:")
            for event in upcoming_events:
                print(event)
        else:
            print("No upcoming events.")


def main():
    calendar = EventCalendar()

    while True:
        print("\nEvent Calendar Menu:")
        print("1. Add Event")
        print("2. Remove Event")
        print("3. Display Upcoming Events")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            title = input("Enter event title: ")
            date = input("Enter event date (YYYY-MM-DD): ")
            description = input("Enter event description: ")
            calendar.add_event(title, date, description)
        elif choice == "2":
            title = input("Enter event title to remove: ")
            calendar.remove_event(title)
        elif choice == "3":
            calendar.display_upcoming_events()
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
