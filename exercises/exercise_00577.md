## Exercise
Write a function that finds the common characters between two given strings.

## Reference Solution
```python
def common_characters(str1, str2):
    """
    Finds the common characters between two given strings.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        list: A list of common characters.
    """
    # Convert both strings to lower case to ignore case sensitivity
    str1 = str1.lower()
    str2 = str2.lower()

    # Create a dictionary to store the count of each character in str1
    str1_count = {}
    for char in str1:
        if char in str1_count:
            str1_count[char] += 1
        else:
            str1_count[char] = 1

    # Create a dictionary to store the count of each character in str2
    str2_count = {}
    for char in str2:
        if char in str2_count:
            str2_count[char] += 1
        else:
            str2_count[char] = 1

    # Find the common characters
    common_chars = []
    for char in str1_count:
        if char in str2_count:
            common_chars.extend([char] * min(str1_count[char], str2_count[char]))

    return common_chars

# Example usage
str1 = "Hello"
str2 = "World"
print(common_characters(str1, str2))
```
