## Exercise
Write a program to find the index of the maximum element in a list using a for loop

## Reference Solution
```python
def find_max_index(lst):
    """
    Finds the index of the maximum element in a list.

    Args:
        lst (list): A list of numbers.

    Returns:
        int: The index of the maximum element.

    Raises:
        ValueError: If the list is empty.
    """
    if not lst:
        raise ValueError("List is empty")

    max_index = 0
    max_value = lst[0]

    for i in range(1, len(lst)):
        if lst[i] > max_value:
            max_value = lst[i]
            max_index = i

    return max_index


# Example usage
def main():
    numbers = [4, 2, 9, 6, 5, 1, 8, 3, 7]
    print("List:", numbers)
    max_index = find_max_index(numbers)
    print("Index of the maximum element:", max_index)
    print("Maximum element:", numbers[max_index])


if __name__ == "__main__":
    main()
```
