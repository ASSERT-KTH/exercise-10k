## Exercise
Write a function that finds the symmetric difference between two dictionaries

def symmetric_difference(dict1, dict2):
    """
    Returns a dictionary that contains all the keys that are in exactly one of the two input dictionaries.

    :param dict1: The first dictionary.
    :param dict2: The second dictionary.
    :return: A dictionary containing the symmetric difference.
    """
    # Find keys that are in dict1 but not in dict2
    dict1_unique_keys = {key: dict1[key] for key in dict1 if key not in dict2}
    
    # Find keys that are in dict2 but not in dict1
    dict2_unique_keys = {key: dict2[key] for key in dict2 if key not in dict1}
    
    # Combine the two dictionaries
    symmetric_diff = {**dict1_unique_keys, **dict2_unique_keys}
    
    return symmetric_diff

# Example usage:

def symmetric_difference(dict1, dict2):
    """
    Returns a dictionary that contains all the keys that are in exactly one of the two input dictionaries.

    :param dict1: The first dictionary.
    :param dict2: The second dictionary.
    :return: A dictionary containing the symmetric difference.
    """
    symmetric_diff_keys = (set(dict1.keys()) - set(dict2.keys())) | (set(dict2.keys()) - set(dict1.keys()))
    symmetric_diff = {key: dict1.get(key, dict2.get(key)) for key in symmetric_diff_keys}
    
    return symmetric_diff

# Example usage:
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 2, 'c': 4, 'd': 5}

result = symmetric_difference(dict1, dict2)
