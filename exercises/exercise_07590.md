## Exercise
Write a program that checks if a given variable is a dictionary.

## Reference Solution
```python
def is_dict(var):
    """
    Checks if a given variable is a dictionary.

    Args:
        var: The variable to check.

    Returns:
        bool: True if the variable is a dictionary, False otherwise.
    """
    return isinstance(var, dict)

# Example usage
if __name__ == "__main__":
    var1 = {"key": "value"}
    var2 = "not a dictionary"
    var3 = [1, 2, 3]

    print(f"Is var1 a dictionary? {is_dict(var1)}")  # Expected output: True
    print(f"Is var2 a dictionary? {is_dict(var2)}")  # Expected output: False
    print(f"Is var3 a dictionary? {is_dict(var3)}")  # Expected output: False
```
