## Exercise
Create a program that transforms a given list of variables from one data type to another (e.g., integers to floats).

def convert_data_type(data, input_type, output_type):
    """
    Converts a list of variables from one data type to another.

    Args:
        data (list): The list of variables to be converted.
        input_type (str): The current data type of the variables.
        output_type (str): The desired data type of the variables.

    Returns:
        list: The list of variables converted to the desired data type.

    Raises:
        ValueError: If the input_type or output_type is not supported.
    """

    # Define the supported data types
    supported_types = ['int', 'float', 'str']

    # Check if the input_type and output_type are supported
    if input_type not in supported_types or output_type not in supported_types:
        raise ValueError("Unsupported data type. Supported types are: int, float, str")

    # Initialize an empty list to store the converted data
    converted_data = []

    # Convert the data type based on the input_type and output_type
    if input_type == 'int' and output_type == 'float':
        converted_data = [float(var) for var in data]
    elif input_type == 'int' and output_type == 'str':
        converted_data = [str(var) for var in data]
    elif input_type == 'float' and output_type == 'int':
        converted_data = [int(var) for var in data]
    elif input_type == 'float' and output_type == 'str':
        converted_data = [str(var) for var in data]
    elif input_type == 'str' and output_type == 'int':
        converted_data = [int(var) for var in data]
    elif input_type == 'str' and output_type == 'float':
        converted_data = [float(var) for var in data]
    else:
        # If the input_type and output_type are the same, return the original data
        converted_data = data

    return converted_data
