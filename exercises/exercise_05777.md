## Exercise
Write a program that keeps track of the score in a game. The program should allow the user to add points to their score.

## Reference Solution
```python
class GameScoreTracker:
    def __init__(self):
        self.score = 0

    def add_points(self, points):
        """Add points to the current score."""
        self.score += points

    def get_score(self):
        """Get the current score."""
        return self.score

    def display_score(self):
        """Display the current score."""
        print(f"Current Score: {self.score}")


def main():
    tracker = GameScoreTracker()

    while True:
        print("\nGame Score Tracker Menu:")
        print("1. Add Points to Score")
        print("2. Display Current Score")
        print("3. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            points = int(input("Enter points to add: "))
            tracker.add_points(points)
        elif choice == "2":
            tracker.display_score()
        elif choice == "3":
            print("Exiting game. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
