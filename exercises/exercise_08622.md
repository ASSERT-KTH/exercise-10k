## Exercise
Write a program that takes a string as input and returns True if the string is a valid IP address, False otherwise.

## Reference Solution

import re

def is_valid_ip(ip):
    """
    Checks if the given string is a valid IP address.

    Args:
        ip (str): The IP address to check.

    Returns:
        bool: True if the IP address is valid, False otherwise.
    """
    # Regular expression pattern for a valid IP address
    pattern = r"^((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$"

    # Check if the IP address matches the pattern
    if re.match(pattern, ip):
        return True
    else:
        return False

# Test the function
print(is_valid_ip("192.168.1.1"))  # True
print(is_valid_ip("255.255.255.255"))  # True
print(is_valid_ip("192.168.1"))  # False
print(is_valid_ip("192.168.1.1.1"))  # False
print(is_valid_ip("192.168.1.256"))  # False

def is_valid_ip(ip):

    Args:

    Returns:
        bool: True if the IP address is valid, False otherwise.
    """
    try:
        ipaddress.ip_address(ip)
        return True
    except ValueError:
        return False
