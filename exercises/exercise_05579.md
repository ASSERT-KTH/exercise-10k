## Exercise
Create a program that takes an image as input and applies a filter or effect to it.

## Reference Solution
```python
from PIL import Image, ImageEnhance, ImageFilter
import os

def apply_filter(image_path, filter_type):
    """
    Applies a filter to the image.

    Args:
        image_path (str): Path to the image file.
        filter_type (str): Type of filter to apply. Can be one of the following:
            - 'grayscale'
            - 'blur'
            - 'sharpen'
            - 'brightness'
            - 'contrast'

    Returns:
        Image: The filtered image.
    """
    image = Image.open(image_path)

    if filter_type == 'grayscale':
        # Convert image to grayscale
        image = image.convert('L')
    elif filter_type == 'blur':
        # Apply blur filter
        image = image.filter(ImageFilter.GaussianBlur(radius=5))
    elif filter_type == 'sharpen':
        # Apply sharpen filter
        enhancer = ImageEnhance.Sharpness(image)
        image = enhancer.enhance(1.5)
    elif filter_type == 'brightness':
        # Adjust brightness
        enhancer = ImageEnhance.Brightness(image)
        image = enhancer.enhance(1.5)
    elif filter_type == 'contrast':
        # Adjust contrast
        enhancer = ImageEnhance.Contrast(image)
        image = enhancer.enhance(1.5)
    else:
        print("Invalid filter type. Please choose from the following: 'grayscale', 'blur', 'sharpen', 'brightness', 'contrast'")

    return image

def main():
    image_path = input("Enter the path to the image file: ")
    filter_type = input("Enter the type of filter to apply (grayscale, blur, sharpen, brightness, contrast): ")

    filtered_image = apply_filter(image_path, filter_type)

    # Save the filtered image
    output_path = os.path.splitext(image_path)[0] + '_filtered' + os.path.splitext(image_path)[1]
    filtered_image.save(output_path)

    print("Filtered image saved to:", output_path)

if __name__ == "__main__":
    main()
```
