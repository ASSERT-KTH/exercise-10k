## Exercise
Write a program that takes a 2D array as input and returns the transposed array.

## Reference Solution
```python
import numpy as np

def transpose_array(array):
    """
    This function takes a 2D array as input and returns the transposed array.

    Args:
        array (numpy.ndarray): A 2D array.

    Returns:
        numpy.ndarray: The transposed array.
    """
    # Check if the input is a numpy array
    if not isinstance(array, np.ndarray):
        raise ValueError("Input must be a numpy array")

    # Check if the array is 2D
    if len(array.shape) != 2:
        raise ValueError("Input array must be 2D")

    # Use numpy's transpose function to transpose the array
    transposed_array = np.transpose(array)

    return transposed_array

# Example usage:
if __name__ == "__main__":
    # Create a 2D array
    array = np.array([[1, 2, 3], [4, 5, 6]])

    # Print the original array
    print("Original array:")
    print(array)

    # Transpose the array
    transposed_array = transpose_array(array)

    # Print the transposed array
    print("\nTransposed array:")
    print(transposed_array)
```
