## Exercise
Write a program that takes a string as input and returns a hexagon pattern of the string

## Reference Solution
```python
def hexagon_pattern(input_str, size):
    """
    Generate a hexagon pattern from the input string.

    Args:
        input_str (str): The input string.
        size (int): The size of the hexagon.

    Returns:
        str: The hexagon pattern as a string.
    """
    result = []
    input_str = input_str * (size * 2 + 1)

    # Generate the top half of the hexagon
    for i in range(size):
        row = []
        for j in range(size - i - 1):
            row.append(" ")
        for k in range(2 * i + 1):
            row.append(input_str[k])
        result.append("".join(row))

    # Generate the bottom half of the hexagon
    for i in range(size - 2, -1, -1):
        row = []
        for j in range(size - i - 1):
            row.append(" ")
        for k in range(2 * i + 1):
            row.append(input_str[k])
        result.append("".join(row))

    return "\n".join(result)


# Example usage
def main():
    input_str = "ABC"
    size = 5
    print(hexagon_pattern(input_str, size))


if __name__ == "__main__":
    main()
```
