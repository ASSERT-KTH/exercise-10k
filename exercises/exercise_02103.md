## Exercise
Write a function to update the value associated with a given key in a dictionary

## Reference Solution
```python
def update_dict_value(dictionary, key, new_value):
    """
    Updates the value associated with a given key in a dictionary.

    Args:
    - dictionary (dict): The dictionary to be updated.
    - key: The key whose value is to be updated.
    - new_value: The new value to be associated with the key.

    Returns:
    - None

    Raises:
    - KeyError: If the key does not exist in the dictionary.
    """
    if key in dictionary:
        dictionary[key] = new_value
    else:
        raise KeyError(f"The key '{key}' does not exist in the dictionary.")

# Example usage
if __name__ == "__main__":
    example_dict = {"name": "John", "age": 30}
    print("Original dictionary:", example_dict)

    try:
        update_dict_value(example_dict, "age", 31)
        print("Updated dictionary:", example_dict)
    except KeyError as e:
        print(e)

    try:
        update_dict_value(example_dict, " occupation", "Software Engineer")
    except KeyError as e:
        print(e)
```
