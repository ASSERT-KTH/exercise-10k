## Exercise
Write a program that allows the user to add and remove items from a to-do list.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self):
        task = input("Enter a task to add: ")
        self.tasks.append(task)
        print(f"Task '{task}' added successfully!")

    def remove_task(self):
        if not self.tasks:
            print("No tasks in the list.")
            return

        print("Current tasks:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")

        task_number = input("Enter the task number to remove: ")
        try:
            task_number = int(task_number)
            if task_number < 1 or task_number > len(self.tasks):
                print("Invalid task number.")
                return
            task = self.tasks.pop(task_number - 1)
            print(f"Task '{task}' removed successfully!")
        except ValueError:
            print("Invalid input. Please enter a number.")

    def view_tasks(self):
        if not self.tasks:
            print("No tasks in the list.")
            return

        print("Current tasks:")
        for i, task in enumerate(self.tasks, start=1):
            print(f"{i}. {task}")


def main():
    todo_list = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add task")
        print("2. Remove task")
        print("3. View tasks")
        print("4. Quit")

        option = input("Choose an option: ")

        if option == "1":
            todo_list.add_task()
        elif option == "2":
            todo_list.remove_task()
        elif option == "3":
            todo_list.view_tasks()
        elif option == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
