## Exercise
Create a program that keeps track of the top scores in a game and prints out the leaderboard.

## Reference Solution
```python
class Leaderboard:
    def __init__(self):
        self.scores = {}

    def add_score(self, name, score):
        """Add a score to the leaderboard."""
        if name in self.scores:
            if score > self.scores[name]:
                self.scores[name] = score
        else:
            self.scores[name] = score

    def view_leaderboard(self):
        """Print out the current leaderboard."""
        sorted_scores = sorted(self.scores.items(), key=lambda x: x[1], reverse=True)
        print("Leaderboard:")
        for i, (name, score) in enumerate(sorted_scores):
            print(f"{i+1}. {name}: {score}")

    def reset_leaderboard(self):
        """Reset the leaderboard."""
        self.scores = {}


def main():
    leaderboard = Leaderboard()
    while True:
        print("\nOptions:")
        print("1. Add score")
        print("2. View leaderboard")
        print("3. Reset leaderboard")
        print("4. Quit")
        choice = input("Choose an option: ")
        if choice == "1":
            name = input("Enter player name: ")
            score = int(input("Enter player score: "))
            leaderboard.add_score(name, score)
        elif choice == "2":
            leaderboard.view_leaderboard()
        elif choice == "3":
            leaderboard.reset_leaderboard()
            print("Leaderboard has been reset.")
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
