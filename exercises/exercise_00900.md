## Exercise
Create a program that asks the user a series of trivia questions and keeps track of their score.

## Reference Solution
```python
# Import the random module to shuffle questions
import random

# Define a dictionary of trivia questions, answers, and categories
trivia_questions = {
    "What is the capital of France?": {"A": "Paris", "B": "London", "C": "Berlin", "D": "Rome", "correct": "A"},
    "Which planet is known as the 'Red Planet'?": {"A": "Earth", "B": "Mars", "C": "Jupiter", "D": "Saturn", "correct": "B"},
    "Who painted the famous painting 'The Starry Night'?": {"A": "Leonardo da Vinci", "B": "Vincent van Gogh", "C": "Claude Monet", "D": "Pablo Picasso", "correct": "B"},
    "What is the largest mammal on Earth?": {"A": "Blue whale", "B": "Fin whale", "C": "Humpback whale", "D": "Sperm whale", "correct": "A"},
    "Which author wrote the famous novel 'To Kill a Mockingbird'?": {"A": "F. Scott Fitzgerald", "B": "Harper Lee", "C": "Jane Austen", "D": "J.K. Rowling", "correct": "B"},
}

def ask_question(question, answers):
    """Ask the user a trivia question and return their response"""
    print(question)
    for option, answer in answers.items():
        if option != "correct":
            print(f"{option}: {answer}")
    response = input("Enter the letter of your answer: ")
    return response

def check_answer(response, correct_answer):
    """Check the user's response against the correct answer and return True if correct, False otherwise"""
    return response.upper() == correct_answer

def main():
    # Initialize the score to 0
    score = 0
    # Shuffle the questions to randomize the order
    questions = list(trivia_questions.items())
    random.shuffle(questions)
    # Ask each question and check the user's response
    for question, answers in questions:
        response = ask_question(question, answers)
        if check_answer(response, answers["correct"]):
            print("Correct!")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer was {answers['correct']}.")
        print(f"Your current score is {score} out of {len(questions)}\n")
    # Display the final score
    print(f"Game over! Your final score is {score} out of {len(questions)}")

if __name__ == "__main__":
    main()
```
