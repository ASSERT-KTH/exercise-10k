## Exercise
Write a program to find the most frequent character in a given string

## Reference Solution
```python
def most_frequent_char(s):
    """
    Finds the most frequent character in a given string.

    Args:
        s (str): The input string.

    Returns:
        tuple: A tuple containing the most frequent character and its frequency.
    """
    # Create a dictionary to store character frequencies
    char_freq = {}

    # Iterate over each character in the string
    for char in s:
        # If the character is already in the dictionary, increment its frequency
        if char in char_freq:
            char_freq[char] += 1
        # Otherwise, add the character to the dictionary with a frequency of 1
        else:
            char_freq[char] = 1

    # Find the character with the maximum frequency
    max_freq_char = max(char_freq, key=char_freq.get)

    # Return the most frequent character and its frequency
    return max_freq_char, char_freq[max_freq_char]


def main():
    # Example usage
    s = "Hello, World!"
    most_frequent, frequency = most_frequent_char(s)
    print(f"The most frequent character in '{s}' is '{most_frequent}' with a frequency of {frequency}.")


if __name__ == "__main__":
    main()
```
