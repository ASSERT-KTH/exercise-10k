## Exercise
Create a program that analyzes a given text and provides statistics (e.g. word count, character count), taking the user's input as the text to analyze.

## Reference Solution
```python
import re
from collections import Counter

def analyze_text(text):
    """
    Analyzes the given text and returns statistics.
    
    Args:
    text (str): The text to analyze.
    
    Returns:
    dict: A dictionary containing the text statistics.
    """
    
    # Remove leading and trailing whitespace
    text = text.strip()
    
    # Calculate word count
    words = re.findall(r'\b\w+\b', text.lower())
    word_count = len(words)
    
    # Calculate character count
    character_count = len(text)
    
    # Calculate sentence count
    sentence_count = len(re.findall(r'[.!?]', text))
    
    # Calculate average word length
    average_word_length = sum(len(word) for word in words) / word_count if word_count > 0 else 0
    
    # Calculate most common words
    most_common_words = Counter(words).most_common(5)
    
    # Calculate most common word length
    word_lengths = [len(word) for word in words]
    most_common_word_length = Counter(word_lengths).most_common(1)[0][0]
    
    # Create a dictionary to store the statistics
    statistics = {
        "word_count": word_count,
        "character_count": character_count,
        "sentence_count": sentence_count,
        "average_word_length": average_word_length,
        "most_common_words": most_common_words,
        "most_common_word_length": most_common_word_length
    }
    
    return statistics

def main():
    # Get the text from the user
    text = input("Please enter the text to analyze: ")
    
    # Analyze the text
    statistics = analyze_text(text)
    
    # Print the statistics
    print("\nText Statistics:")
    print("----------------")
    print(f"Word Count: {statistics['word_count']}")
    print(f"Character Count: {statistics['character_count']}")
    print(f"Sentence Count: {statistics['sentence_count']}")
    print(f"Average Word Length: {statistics['average_word_length']:.2f}")
    print("Most Common Words:")
    for word, count in statistics['most_common_words']:
        print(f"  {word}: {count}")
    print(f"Most Common Word Length: {statistics['most_common_word_length']}")

if __name__ == "__main__":
    main()
```
