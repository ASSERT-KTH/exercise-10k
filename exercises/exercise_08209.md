## Exercise
Write a program that scrambles words and asks the user to unscramble them using a while loop.

## Reference Solution
```python
import random

# List of words to scramble
words = ["apple", "banana", "cherry", "date", "elderberry"]

def scramble_word(word):
    """Scramble a word by shuffling its characters"""
    word_list = list(word)
    random.shuffle(word_list)
    return "".join(word_list)

def main():
    print("Welcome to the Word Scramble Game!")
    print("Unscramble the words to win. Type 'quit' to exit.")

    while True:
        word = random.choice(words)
        scrambled_word = scramble_word(word)
        print(f"Unscramble the word: {scrambled_word}")

        answer = input("Your answer: ")

        if answer.lower() == "quit":
            print("Thanks for playing!")
            break
        elif answer.lower() == word:
            print("Correct!")
        else:
            print(f"Sorry, that's incorrect. The correct answer was {word}.")

if __name__ == "__main__":
    main()
```
