## Exercise
Create a program that simulates rolling two dice and calculates the average and standard deviation of the sums.

## Reference Solution
```python
import random
import math

def roll_dice():
    """Simulate rolling two dice and return the sum."""
    dice1 = random.randint(1, 6)
    dice2 = random.randint(1, 6)
    return dice1 + dice2

def calculate_average(sums):
    """Calculate the average of the sums."""
    return sum(sums) / len(sums)

def calculate_standard_deviation(sums, average):
    """Calculate the standard deviation of the sums."""
    variance = sum((x - average) ** 2 for x in sums) / len(sums)
    return math.sqrt(variance)

def main():
    num_rolls = 10000  # Number of times to roll the dice
    sums = [roll_dice() for _ in range(num_rolls)]  # Simulate rolling the dice

    average = calculate_average(sums)
    std_dev = calculate_standard_deviation(sums, average)

    print(f"Average sum: {average:.2f}")
    print(f"Standard deviation: {std_dev:.2f}")

if __name__ == "__main__":
    main()
```
