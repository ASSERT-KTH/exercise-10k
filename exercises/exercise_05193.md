## Exercise
Write a program that asks the user for the score of a cricket match and then prints out the score in the format 'team_name score/overs' (e.g. 'Australia 100/2').

## Reference Solution
```python
def get_team_name(team_number):
    """Get the team name from the user."""
    team_name = input(f"Enter the name of team {team_number}: ")
    return team_name

def get_score():
    """Get the score from the user."""
    while True:
        try:
            score = int(input("Enter the score: "))
            overs = int(input("Enter the overs: "))
            if overs <= 0:
                print("Overs must be greater than 0.")
                continue
            return score, overs
        except ValueError:
            print("Invalid input. Please enter a valid score and overs.")

def print_score(team_name, score, overs):
    """Print the score in the format 'team_name score/overs'."""
    print(f"{team_name} {score}/{overs}")

def main():
    team_name = get_team_name(1)
    score, overs = get_score()
    print_score(team_name, score, overs)

if __name__ == "__main__":
    main()
```
