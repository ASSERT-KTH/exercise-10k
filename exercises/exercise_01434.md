## Exercise
Create a function that takes a list and two indices as input and copies the elements from the first index to the second index, replacing the elements at the second index and onwards.

## Reference Solution

def copy_and_replace(input_list, start_index, end_index):
    """
    This function takes a list and two indices as input, copies the elements from the start index to the end index,
    and replaces the elements at the end index and onwards in the list.

    Args:
        input_list (list): The input list.
        start_index (int): The index where the copying starts.
        end_index (int): The index where the copying ends and replacement starts.

    Returns:
        list: The modified list.
    """

    # Check if the input list is None or not a list
    if not isinstance(input_list, list):
        raise TypeError("Input must be a list.")

    # Check if the start and end indices are integers
    if not isinstance(start_index, int) or not isinstance(end_index, int):
        raise TypeError("Indices must be integers.")

    # Check if the start index is less than or equal to the end index
    if start_index > end_index:
        raise ValueError("Start index must be less than or equal to the end index.")

    # Check if the start and end indices are within the bounds of the list
    if start_index < 0 or end_index < 0 or start_index >= len(input_list) or end_index > len(input_list):
        raise IndexError("Indices out of range.")

    # Slice the input list from the start index to the end index + 1 to get the elements to be copied
    elements_to_copy = input_list[start_index:end_index + 1]

    # Replace the elements at the end index and onwards with the copied elements
    # We use list slicing to replace the elements in the input list
    # The end index is not included in the slice, so we use end_index + 1 if end_index < len(input_list) - 1
    input_list[end_index:] = elements_to_copy

    # Return the modified list
    return input_list

end_index = 5
