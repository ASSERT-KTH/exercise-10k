## Exercise
Write a program that takes the radius of a circle as input and outputs its area and circumference.

## Reference Solution
```python
import math

def calculate_circle_properties(radius):
    """
    Calculate the area and circumference of a circle given its radius.
    
    Parameters:
    radius (float): The radius of the circle.
    
    Returns:
    tuple: A tuple containing the area and circumference of the circle.
    """
    area = math.pi * (radius ** 2)
    circumference = 2 * math.pi * radius
    
    return area, circumference

def main():
    # Get the radius from the user
    while True:
        try:
            radius = float(input("Enter the radius of the circle: "))
            if radius <= 0:
                print("Radius must be positive.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a number.")
    
    # Calculate the area and circumference
    area, circumference = calculate_circle_properties(radius)
    
    # Print the results
    print(f"Area: {area:.2f}")
    print(f"Circumference: {circumference:.2f}")

if __name__ == "__main__":
    main()
```
