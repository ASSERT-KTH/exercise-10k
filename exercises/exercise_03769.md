## Exercise
Reduce a dictionary to a single value.

## Reference Solution
```python
def sum_dict_values(dictionary):
    """Returns the sum of all dictionary values."""
    return sum(dictionary.values())

# Example usage:
my_dict = {'a': 1, 'b': 2, 'c': 3}
result = sum_dict_values(my_dict)
print(result)  # Output: 6
```

```python
def average_dict_values(dictionary):
    """Returns the average of all dictionary values."""
    return sum(dictionary.values()) / len(dictionary)

# Example usage:
my_dict = {'a': 1, 'b': 2, 'c': 3}
result = average_dict_values(my_dict)
print(result)  # Output: 2.0
```

```python
def concat_dict_values(dictionary):
    """Returns a concatenated string of all dictionary values."""
    return ''.join(str(value) for value in dictionary.values())

# Example usage:
my_dict = {'a': 'hello', 'b': 'world', 'c': '!'}
result = concat_dict_values(my_dict)
print(result)  # Output: 'helloworld!'
```

import functools
import operator

    """Applies a custom reduction function to dictionary values."""
