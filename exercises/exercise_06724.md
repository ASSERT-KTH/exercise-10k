## Exercise
Write a program that reads and writes records of data

## Reference Solution
```python
import json
import os

class Record:
    def __init__(self, id, name, email):
        self.id = id
        self.name = name
        self.email = email

class RecordManager:
    def __init__(self, filename):
        self.filename = filename
        self.records = self.load_records()

    def load_records(self):
        if os.path.exists(self.filename):
            with open(self.filename, 'r') as file:
                data = json.load(file)
                return [Record(**record) for record in data]
        else:
            return []

    def save_records(self):
        data = [{'id': record.id, 'name': record.name, 'email': record.email} for record in self.records]
        with open(self.filename, 'w') as file:
            json.dump(data, file, indent=4)

    def add_record(self, id, name, email):
        self.records.append(Record(id, name, email))
        self.save_records()

    def get_record(self, id):
        for record in self.records:
            if record.id == id:
                return record
        return None

    def update_record(self, id, name=None, email=None):
        record = self.get_record(id)
        if record:
            if name:
                record.name = name
            if email:
                record.email = email
            self.save_records()

    def delete_record(self, id):
        record = self.get_record(id)
        if record:
            self.records.remove(record)
            self.save_records()

def main():
    filename = 'records.json'
    manager = RecordManager(filename)

    while True:
        print("1. Add Record")
        print("2. Get Record")
        print("3. Update Record")
        print("4. Delete Record")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == '1':
            id = input("Enter ID: ")
            name = input("Enter Name: ")
            email = input("Enter Email: ")
            manager.add_record(id, name, email)
        elif choice == '2':
            id = input("Enter ID: ")
            record = manager.get_record(id)
            if record:
                print(f"Name: {record.name}, Email: {record.email}")
            else:
                print("Record not found")
        elif choice == '3':
            id = input("Enter ID: ")
            name = input("Enter new Name (press enter to skip): ")
            email = input("Enter new Email (press enter to skip): ")
            manager.update_record(id, name or None, email or None)
        elif choice == '4':
            id = input("Enter ID: ")
            manager.delete_record(id)
        elif choice == '5':
            break
        else:
            print("Invalid option")

if __name__ == '__main__':
    main()
```
