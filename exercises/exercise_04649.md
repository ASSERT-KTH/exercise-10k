## Exercise
Create a program that converts a time from 12-hour format to 24-hour format or from 24-hour format to 12-hour format. The user can choose the conversion type, and the program performs the conversion.

## Reference Solution
```python
def convert_to_24_hour(time_str):
    """
    Converts time from 12-hour format to 24-hour format.

    Args:
        time_str (str): The time string in 12-hour format (e.g., "08:30 PM").

    Returns:
        str: The time string in 24-hour format (e.g., "20:30").
    """
    components = time_str.split()
    time = components[0].split(":")
    hours = int(time[0])
    minutes = time[1]
    am_pm = components[1]

    if am_pm == "AM":
        if hours == 12:
            hours = 0
    else:
        if hours != 12:
            hours += 12

    return f"{hours:02d}:{minutes}"


def convert_to_12_hour(time_str):
    """
    Converts time from 24-hour format to 12-hour format.

    Args:
        time_str (str): The time string in 24-hour format (e.g., "20:30").

    Returns:
        str: The time string in 12-hour format (e.g., "08:30 PM").
    """
    time = time_str.split(":")
    hours = int(time[0])
    minutes = time[1]

    am_pm = "AM" if hours < 12 else "PM"
    hours_12 = hours % 12
    if hours_12 == 0:
        hours_12 = 12

    return f"{hours_12:02d}:{minutes} {am_pm}"


def main():
    print("Time Conversion Program")
    print("1. Convert 12-hour to 24-hour")
    print("2. Convert 24-hour to 12-hour")
    choice = input("Choose the conversion type (1/2): ")

    if choice == "1":
        time_str = input("Enter time in 12-hour format (e.g., 08:30 PM): ")
        print(f"Time in 24-hour format: {convert_to_24_hour(time_str)}")
    elif choice == "2":
        time_str = input("Enter time in 24-hour format (e.g., 20:30): ")
        print(f"Time in 12-hour format: {convert_to_12_hour(time_str)}")
    else:
        print("Invalid choice. Please choose 1 or 2.")


if __name__ == "__main__":
    main()
```
