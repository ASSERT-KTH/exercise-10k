## Exercise
Write a program that checks if a string matches a certain pattern and prints a message accordingly.

## Reference Solution
```python
import re

def check_pattern(input_string, pattern):
    """
    Checks if the input string matches the given pattern.
    
    Args:
        input_string (str): The string to check.
        pattern (str): The pattern to match against.
        
    Returns:
        bool: True if the string matches the pattern, False otherwise.
    """
    
    # Use the re.match function to check if the string matches the pattern
    match = re.match(pattern, input_string)
    
    # If the string matches the pattern, print a success message
    if match:
        print(f"The string '{input_string}' matches the pattern '{pattern}'.")
        return True
    else:
        # If the string does not match the pattern, print a failure message
        print(f"The string '{input_string}' does not match the pattern '{pattern}'.")
        return False

# Example usage
input_string = "Hello, world!"
pattern = "^Hello, world!$"
check_pattern(input_string, pattern)

input_string = "Goodbye, world!"
pattern = "^Hello, world!$"
check_pattern(input_string, pattern)
```
