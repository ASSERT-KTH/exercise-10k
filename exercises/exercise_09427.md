## Exercise
Create a function that takes a list of numbers and returns their product using the reduce function

## Reference Solution
```python
from functools import reduce

def product_of_numbers(numbers):
    """
    Calculate the product of a list of numbers using the reduce function.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int or float: The product of the numbers in the list.
    """
    # Check if the input is a list
    if not isinstance(numbers, list):
        raise TypeError("Input must be a list of numbers.")

    # Check if all elements in the list are numbers
    if not all(isinstance(num, (int, float)) for num in numbers):
        raise ValueError("All elements in the list must be numbers.")

    # Use the reduce function to calculate the product
    try:
        product = reduce(lambda x, y: x * y, numbers)
    except TypeError:
        raise ValueError("The list must contain at least one number.")

    return product

# Example usage
if __name__ == "__main__":
    numbers = [1, 2, 3, 4, 5]
    result = product_of_numbers(numbers)
    print(f"The product of {numbers} is {result}.")
```
