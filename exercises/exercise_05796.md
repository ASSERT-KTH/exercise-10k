## Exercise
Write a function that generates a list of random numbers with a specified length and range.

## Reference Solution
```python
import random

def generate_random_numbers(length: int, min_value: int, max_value: int) -> list:
    """
    Generates a list of random numbers with a specified length and range.

    Args:
        length (int): The number of random numbers to generate.
        min_value (int): The minimum value of the random numbers.
        max_value (int): The maximum value of the random numbers.

    Returns:
        list: A list of random numbers with the specified length and range.
    """
    if not isinstance(length, int) or length <= 0:
        raise ValueError("Length must be a positive integer.")
    if not isinstance(min_value, int) or not isinstance(max_value, int):
        raise ValueError("Min value and max value must be integers.")
    if min_value >= max_value:
        raise ValueError("Min value must be less than max value.")

    random_numbers = [random.randint(min_value, max_value) for _ in range(length)]
    return random_numbers

# Example usage
if __name__ == "__main__":
    length = 10
    min_value = 1
    max_value = 100
    random_numbers = generate_random_numbers(length, min_value, max_value)
    print(f"Random numbers: {random_numbers}")
```
