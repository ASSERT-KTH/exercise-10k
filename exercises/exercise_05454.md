## Exercise
Write a program that takes a phone number as input and prints out the formatted phone number

## Reference Solution
```python
def format_phone_number(phone_number):
    """
    Formats a phone number into the standard (XXX) XXX-XXXX format.

    Args:
        phone_number (str): The phone number to format.

    Returns:
        str: The formatted phone number.
    """
    # Remove any non-digit characters from the phone number
    digits = ''.join(filter(str.isdigit, phone_number))

    # Check if the phone number has 10 digits
    if len(digits) != 10:
        raise ValueError("Invalid phone number. Please enter a 10-digit phone number.")

    # Format the phone number
    formatted_phone_number = f"({digits[:3]}) {digits[3:6]}-{digits[6:]}"

    return formatted_phone_number


def main():
    # Get the phone number from the user
    phone_number = input("Please enter a phone number: ")

    try:
        # Format and print the phone number
        formatted_phone_number = format_phone_number(phone_number)
        print(f"Formatted phone number: {formatted_phone_number}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
