## Exercise
Write a program that estimates the total cost of items based on their prices and quantities.

## Reference Solution
```python
class Item:
    """Class representing an item with price and quantity."""

    def __init__(self, name, price, quantity):
        """
        Initialize an item.

        Args:
            name (str): Name of the item.
            price (float): Price of the item.
            quantity (int): Quantity of the item.
        """
        self.name = name
        self.price = price
        self.quantity = quantity

    def calculate_cost(self):
        """
        Calculate the total cost of the item.

        Returns:
            float: Total cost of the item.
        """
        return self.price * self.quantity


class ShoppingCart:
    """Class representing a shopping cart with items."""

    def __init__(self):
        """Initialize an empty shopping cart."""
        self.items = []

    def add_item(self, item):
        """
        Add an item to the shopping cart.

        Args:
            item (Item): Item to be added.
        """
        self.items.append(item)

    def estimate_total_cost(self):
        """
        Estimate the total cost of all items in the shopping cart.

        Returns:
            float: Total cost of all items.
        """
        return sum(item.calculate_cost() for item in self.items)


def main():
    # Create a shopping cart
    cart = ShoppingCart()

    # Create items
    item1 = Item("Apple", 1.00, 5)
    item2 = Item("Banana", 0.50, 10)
    item3 = Item("Orange", 1.50, 3)

    # Add items to the shopping cart
    cart.add_item(item1)
    cart.add_item(item2)
    cart.add_item(item3)

    # Estimate the total cost
    total_cost = cart.estimate_total_cost()

    # Print the total cost
    print(f"Estimated total cost: ${total_cost:.2f}")


if __name__ == "__main__":
    main()
```
