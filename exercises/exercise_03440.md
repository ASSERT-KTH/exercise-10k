## Exercise
Write a function that takes a run-length encoded string as input and returns the decoded string

## Reference Solution
```python
def run_length_decode(encoded_str):
    """
    Decodes a run-length encoded string.

    Args:
        encoded_str (str): The run-length encoded string.

    Returns:
        str: The decoded string.
    """
    decoded_str = ""
    i = 0

    while i < len(encoded_str):
        # Get the count
        count = ""
        while i < len(encoded_str) and encoded_str[i].isdigit():
            count += encoded_str[i]
            i += 1

        # Get the character
        char = encoded_str[i]
        i += 1

        # Add the character to the decoded string the specified number of times
        decoded_str += char * int(count)

    return decoded_str


# Example usage:
if __name__ == "__main__":
    encoded_str = "3a2b1c"
    decoded_str = run_length_decode(encoded_str)
    print(f"Encoded string: {encoded_str}")
    print(f"Decoded string: {decoded_str}")
```
