## Exercise
Write a program that calculates a vehicle's fuel efficiency given its distance traveled and fuel consumed.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumed, unit='kmpl'):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
        distance (float): The distance traveled by the vehicle.
        fuel_consumed (float): The amount of fuel consumed by the vehicle.
        unit (str, optional): The unit of fuel efficiency. Defaults to 'kmpl'.

    Returns:
        float: The fuel efficiency of the vehicle.
    """

    # Check if the distance or fuel consumed is zero
    if distance == 0 or fuel_consumed == 0:
        return 0

    # Calculate the fuel efficiency
    fuel_efficiency = distance / fuel_consumed

    # Convert the fuel efficiency to the desired unit
    if unit == 'kmpl':
        fuel_efficiency = round(fuel_efficiency, 2)  # in km/l
    elif unit == 'mpg':
        fuel_efficiency = round(fuel_efficiency * 2.352, 2)  # in miles/gallon
    elif unit == 'l/100km':
        fuel_efficiency = round(100 / fuel_efficiency, 2)  # in l/100km
    else:
        raise ValueError("Invalid unit. Supported units are 'kmpl', 'mpg', and 'l/100km'.")

    return fuel_efficiency


def main():
    # Example usage:
    distance = float(input("Enter the distance traveled (in km): "))
    fuel_consumed = float(input("Enter the amount of fuel consumed (in liters): "))

    print(f"The fuel efficiency of the vehicle is {calculate_fuel_efficiency(distance, fuel_consumed)} kmpl.")

    # Ask the user if they want to convert the fuel efficiency to a different unit
    convert_unit = input("Do you want to convert the fuel efficiency to a different unit? (yes/no): ")
    if convert_unit.lower() == 'yes':
        unit = input("Enter the unit you want to convert to (kmpl, mpg, or l/100km): ")
        print(f"The fuel efficiency of the vehicle is {calculate_fuel_efficiency(distance, fuel_consumed, unit)} {unit}.")


if __name__ == "__main__":
    main()
```
