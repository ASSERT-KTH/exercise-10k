## Exercise
Write a program to solve the vertex cover problem using a greedy algorithm.

## Reference Solution

import networkx as nx

def greedy_vertex_cover(graph):
    """
    Computes a vertex cover for the given graph using a greedy algorithm.

    Args:
        graph (networkx.Graph): The input graph.

    Returns:
        list: A list of vertices that form a vertex cover for the graph.
    """
    vertex_cover = []
    edges = list(graph.edges)

    while edges:
        # Find the vertex with the highest degree
        max_degree_vertex = max(graph.nodes, key=lambda node: graph.degree(node))

        # Add the vertex to the vertex cover
        vertex_cover.append(max_degree_vertex)

        # Remove all edges incident to the selected vertex
        edges_to_remove = [edge for edge in edges if max_degree_vertex in edge]
        edges = [edge for edge in edges if edge not in edges_to_remove]

        # Remove the selected vertex from the graph
        graph.remove_node(max_degree_vertex)

    return vertex_cover

# Example usage:
if __name__ == "__main__":
    # Create a sample graph
    graph = nx.Graph()
    graph.add_edges_from([(1, 2), (1, 3), (2, 3), (2, 4), (3, 4), (3, 5)])

    # Compute a vertex cover using the greedy algorithm
    vertex_cover = greedy_vertex_cover(graph)

    print("Vertex Cover:", vertex_cover)
