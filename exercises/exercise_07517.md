## Exercise
Create an Anagram game using variables and string manipulation.

## Reference Solution
```python
import random

# List of words for the game
words = ["apple", "banana", "cherry", "date", "elderberry"]

def generate_anagram(word):
    """
    Generates an anagram of the given word.
    
    Args:
    word (str): The word to generate an anagram for.
    
    Returns:
    str: The anagrammed word.
    """
    # Convert the word to a list of characters
    char_list = list(word)
    
    # Shuffle the list of characters
    random.shuffle(char_list)
    
    # Join the shuffled characters back into a string
    anagram = "".join(char_list)
    
    return anagram

def play_anagram_game():
    """
    Plays the anagram game.
    """
    # Choose a random word from the list
    word = random.choice(words)
    
    # Generate an anagram of the word
    anagram = generate_anagram(word)
    
    # Ask the player to unscramble the word
    print(f"Unscramble the letters to form a valid English word: {anagram}")
    answer = input("Enter your answer: ")
    
    # Check if the answer is correct
    if answer.lower() == word:
        print("Correct! You unscrambled the word successfully.")
    else:
        print(f"Sorry, the correct answer is {word}.")

def main():
    print("Welcome to the Anagram Game!")
    play_again = "y"
    while play_again.lower() == "y":
        play_anagram_game()
        play_again = input("Would you like to play again? (y/n): ")

if __name__ == "__main__":
    main()
```
