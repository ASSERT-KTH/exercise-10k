## Exercise
Write a program that reads input from a database and processes it.

## Reference Solution
```python
import sqlite3
import pandas as pd

def read_from_database(db_name, table_name):
    """
    Reads data from a SQLite database.

    Args:
        db_name (str): Name of the database file.
        table_name (str): Name of the table to read from.

    Returns:
        pd.DataFrame: A DataFrame containing the data from the table.
    """
    conn = sqlite3.connect(db_name)
    query = f"SELECT * FROM {table_name}"
    df = pd.read_sql_query(query, conn)
    conn.close()
    return df

def process_data(df):
    """
    Processes the data by calculating the mean and median of each column.

    Args:
        df (pd.DataFrame): The DataFrame containing the data.

    Returns:
        dict: A dictionary containing the mean and median of each column.
    """
    results = {}
    for col in df.columns:
        mean = df[col].mean()
        median = df[col].median()
        results[col] = {"mean": mean, "median": median}
    return results

def main():
    db_name = "example.db"
    table_name = "data"
    df = read_from_database(db_name, table_name)
    results = process_data(df)
    print("Results:")
    for col, stats in results.items():
        print(f"Column: {col}")
        print(f"Mean: {stats['mean']}")
        print(f"Median: {stats['median']}\n")

if __name__ == "__main__":
    main()
```
