## Exercise
Write a function that takes a string as input and returns True if it contains any duplicate characters

## Reference Solution
```python
def has_duplicate_characters(input_string: str) -> bool:
    """
    Checks if a string contains any duplicate characters.

    Args:
        input_string (str): The input string to check.

    Returns:
        bool: True if the string contains duplicate characters, False otherwise.
    """
    seen = set()
    for char in input_string:
        if char in seen:
            return True
        seen.add(char)
    return False

# Example usage
if __name__ == "__main__":
    print(has_duplicate_characters("hello"))  # True
    print(has_duplicate_characters("abcde"))  # False
```
