## Exercise
Create a program to calculate the total cost of an item after applying a tax rate.

## Reference Solution
```python
# Function to calculate total cost after tax
def calculate_total_cost(item_cost, tax_rate):
    """
    Calculate the total cost of an item after applying a tax rate.

    Args:
        item_cost (float): The cost of the item.
        tax_rate (float): The tax rate as a percentage.

    Returns:
        float: The total cost of the item after tax.
    """
    # Convert tax rate from percentage to decimal
    tax_rate_decimal = tax_rate / 100

    # Calculate tax amount
    tax_amount = item_cost * tax_rate_decimal

    # Calculate total cost
    total_cost = item_cost + tax_amount

    return total_cost


# Main function
def main():
    # Get item cost from user
    item_cost = float(input("Enter the cost of the item: "))

    # Get tax rate from user
    tax_rate = float(input("Enter the tax rate (in %): "))

    # Calculate total cost
    total_cost = calculate_total_cost(item_cost, tax_rate)

    # Print total cost
    print(f"The total cost of the item after tax is: ${total_cost:.2f}")


if __name__ == "__main__":
    main()
```
