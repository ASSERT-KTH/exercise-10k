## Exercise
Create a program that generates a geometric sequence up to a given term.

## Reference Solution
```python
def geometric_sequence(a, r, n):
    """
    Generates a geometric sequence up to the nth term.

    Args:
        a (float): The first term of the sequence.
        r (float): The common ratio of the sequence.
        n (int): The number of terms to generate.

    Returns:
        list: A list of the first n terms of the geometric sequence.
    """
    sequence = [a]
    for i in range(1, n):
        sequence.append(sequence[i-1] * r)
    return sequence

def main():
    # Example usage:
    a = 2  # First term
    r = 3  # Common ratio
    n = 10  # Number of terms

    sequence = geometric_sequence(a, r, n)
    print("Geometric Sequence:")
    print(sequence)

if __name__ == "__main__":
    main()
```
