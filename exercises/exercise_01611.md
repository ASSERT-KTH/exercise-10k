## Exercise
Write a program that takes a temperature in Celsius as input and outputs its equivalent temperature in Fahrenheit.

## Reference Solution
```python
# Function to convert Celsius to Fahrenheit
def celsius_to_fahrenheit(celsius):
    """
    This function converts temperature from Celsius to Fahrenheit.

    Parameters:
    celsius (float): Temperature in Celsius.

    Returns:
    float: Temperature in Fahrenheit.
    """
    if celsius < -273.15:
        raise ValueError("Temperature cannot be lower than absolute zero.")
    fahrenheit = (celsius * 9/5) + 32
    return fahrenheit


def main():
    # Get user input
    while True:
        try:
            celsius = float(input("Enter temperature in Celsius: "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Convert Celsius to Fahrenheit
    fahrenheit = celsius_to_fahrenheit(celsius)

    # Print the result
    print(f"{celsius}째C is equal to {fahrenheit}째F")


if __name__ == "__main__":
    main()
```
