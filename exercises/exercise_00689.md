## Exercise
Write a program to round a given number to the nearest integer.

def round_to_nearest_integer(num):
    """
    Rounds a given number to the nearest integer.

    Args:
        num (float): The number to be rounded.

    Returns:
        int: The rounded number.
    """
    return round(num)


# Example usage:
if __name__ == "__main__":
    num = float(input("Enter a number: "))
    rounded_num = round_to_nearest_integer(num)
    print(f"The number {num} rounded to the nearest integer is {rounded_num}")
