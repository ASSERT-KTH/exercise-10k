## Exercise
Write a function that takes a dictionary and returns a new dictionary with the key-value pairs sorted by key or value

## Reference Solution
```python
def sort_dict(input_dict, sort_by='key', reverse=False):
    """
    Sorts a dictionary by key or value.

    Args:
        input_dict (dict): The input dictionary.
        sort_by (str, optional): The sorting parameter. Defaults to 'key'.
        reverse (bool, optional): Whether to sort in reverse order. Defaults to False.

    Returns:
        dict: A new dictionary with the sorted key-value pairs.
    """
    if sort_by not in ['key', 'value']:
        raise ValueError("Invalid sorting parameter. It should be either 'key' or 'value'.")

    if sort_by == 'key':
        return dict(sorted(input_dict.items(), key=lambda item: item[0], reverse=reverse))
    else:
        return dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))

# Example usage:
if __name__ == "__main__":
    input_dict = {'b': 2, 'a': 1, 'c': 3}
    sorted_by_key = sort_dict(input_dict, sort_by='key')
    sorted_by_value = sort_dict(input_dict, sort_by='value')
    sorted_by_key_reverse = sort_dict(input_dict, sort_by='key', reverse=True)
    sorted_by_value_reverse = sort_dict(input_dict, sort_by='value', reverse=True)

    print("Original Dictionary:", input_dict)
    print("Sorted by Key:", sorted_by_key)
    print("Sorted by Value:", sorted_by_value)
    print("Sorted by Key in Reverse Order:", sorted_by_key_reverse)
    print("Sorted by Value in Reverse Order:", sorted_by_value_reverse)
```
