## Exercise
Write a program to swap the values of two variables.

## Reference Solution

def swap_variables(a, b):
    """
    Swap the values of two variables.

    Args:
        a (any): The first variable.
        b (any): The second variable.

    Returns:
        tuple: A tuple containing the swapped values.
    """
    # Use tuple packing and unpacking to swap the values
    a, b = b, a
    return a, b

# Example usage
if __name__ == "__main__":
    var1 = 5
    var2 = 10

    print("Before swap:")
    print(f"var1 = {var1}")
    print(f"var2 = {var2}")

    var1, var2 = swap_variables(var1, var2)

    print("After swap:")
    print(f"var1 = {var1}")
    print(f"var2 = {var2}")
