## Exercise
Write a program that transforms input data in some way

## Reference Solution
```python
def double_numbers(numbers):
    """Double each number in the list"""
    return [n * 2 for n in numbers]

def square_numbers(numbers):
    """Square each number in the list"""
    return [n ** 2 for n in numbers]

def increase_numbers(numbers, amount):
    """Increase each number in the list by a specified amount"""
    return [n + amount for n in numbers]

def main():
    numbers = [1, 2, 3, 4, 5]
    print("Original numbers:", numbers)

    doubled_numbers = double_numbers(numbers)
    print("Doubled numbers:", doubled_numbers)

    squared_numbers = square_numbers(numbers)
    print("Squared numbers:", squared_numbers)

    increased_numbers = increase_numbers(numbers, 5)
    print("Increased numbers:", increased_numbers)

if __name__ == "__main__":
    main()
```
