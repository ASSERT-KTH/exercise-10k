## Exercise
Write a program to initialize variables with user input.

## Reference Solution
```python
# user_input_program.py

def get_user_input():
    """
    Get user input to initialize variables.
    """
    # Get string input
    name = input("Please enter your name: ")

    # Get integer input
    age = int(input("Please enter your age: "))

    # Get float input
    height = float(input("Please enter your height (in meters): "))

    # Get boolean input
    is_student = input("Are you a student? (yes/no): ")
    is_student = is_student.lower() == "yes"

    return name, age, height, is_student


def main():
    """
    Main function to execute the program.
    """
    name, age, height, is_student = get_user_input()

    print("\nUser Input Summary:")
    print(f"Name: {name}")
    print(f"Age: {age} years")
    print(f"Height: {height} meters")
    print(f"Is Student: {is_student}")


if __name__ == "__main__":
    main()
```
