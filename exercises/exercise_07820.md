## Exercise
Write a program that finds the shortest word in a given string.

## Reference Solution
```python
def find_shortest_word(input_string):
    """
    Finds the shortest word in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The shortest word in the input string.
    """
    # Split the input string into words
    words = input_string.split()

    # Initialize the shortest word with the first word
    shortest_word = words[0]

    # Iterate over each word in the list
    for word in words:
        # Remove punctuation from the word
        word_no_punct = ''.join(e for e in word if e.isalnum()).lower()
        
        # Update the shortest word if the current word is shorter
        if len(word_no_punct) < len(''.join(e for e in shortest_word if e.isalnum()).lower()):
            shortest_word = word

    return shortest_word

def main():
    # Example usage
    input_string = "This is a test string with multiple words"
    shortest_word = find_shortest_word(input_string)
    print(f"The shortest word is: {shortest_word}")

if __name__ == "__main__":
    main()
```
