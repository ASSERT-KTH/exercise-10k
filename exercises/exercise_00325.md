## Exercise
Write a program to demonstrate the concept of variable scope, including global and local variables.

## Reference Solution
```python
# Global variable
x = 10

def outer_function():
    # Local variable
    y = 20

    def inner_function():
        # Nonlocal variable
        nonlocal y
        y = 30
        z = 40  # Local variable

        print("Inside inner_function:")
        print(f"x: {x} (global)")
        print(f"y: {y} (nonlocal)")
        print(f"z: {z} (local)")

    inner_function()
    print("\nInside outer_function:")
    print(f"x: {x} (global)")
    print(f"y: {y} (local)")
    # print(f"z: {z}")  # Error: z is not defined in outer_function

outer_function()
print("\nGlobal scope:")
print(f"x: {x} (global)")
# print(f"y: {y}")  # Error: y is not defined in global scope
# print(f"z: {z}")  # Error: z is not defined in global scope

# Modifying global variable
def modify_global_variable():
    global x
    x = 50

modify_global_variable()
print("\nAfter modifying global variable:")
print(f"x: {x} (global)")
```
