## Exercise
Write a program that takes two numbers and an operator as input and prints the result of the operation to the console.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int num1, num2;
    char ch;

    cout << "Enter first number: ";
    cin >> num1;
    cout << "Enter operator (+, -, *, /): ";
    cin >> ch;
    cout << "Enter second number: ";
    cin >> num2;

    switch(ch) {
        case '+':
            cout << "Result: " << num1 + num2 << endl;
            break;
        case '-':
            cout << "Result: " << num1 - num2 << endl;
            break;
        case '*':
            cout << "Result: " << num1 * num2 << endl;
            break;
        case '/':
            cout << "Result: " << (float)num1 / num2 << endl;
            break;
        default:
            cout << "Invalid operator" << endl;
            break;
    }

    return 0;
}
```
