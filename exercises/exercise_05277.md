## Exercise
Develop a program to split a bill among a group of people using conditional statements

## Reference Solution
```python
def calculate_split_bill(total_bill, num_people, tip_percentage=0):
    """
    Calculate the amount each person needs to pay when splitting a bill.

    Args:
        total_bill (float): The total bill amount.
        num_people (int): The number of people splitting the bill.
        tip_percentage (float, optional): The tip percentage. Defaults to 0.

    Returns:
        float: The amount each person needs to pay.
    """
    if num_people <= 0:
        raise ValueError("Number of people must be greater than 0")
    if total_bill < 0:
        raise ValueError("Total bill amount cannot be negative")
    if tip_percentage < 0 or tip_percentage > 100:
        raise ValueError("Tip percentage must be between 0 and 100")

    # Calculate the total amount including tip
    if tip_percentage > 0:
        total_amount = total_bill + (total_bill * (tip_percentage / 100))
    else:
        total_amount = total_bill

    # Calculate the amount each person needs to pay
    split_amount = total_amount / num_people

    return split_amount

def main():
    print("Bill Splitter Program")
    print("---------------------")

    # Get user input
    total_bill = float(input("Enter the total bill amount: $"))
    num_people = int(input("Enter the number of people splitting the bill: "))
    tip_percentage = float(input("Enter the tip percentage (0 for no tip): "))

    # Calculate and display the split amount
    try:
        split_amount = calculate_split_bill(total_bill, num_people, tip_percentage)
        print(f"Each person needs to pay: ${split_amount:.2f}")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
