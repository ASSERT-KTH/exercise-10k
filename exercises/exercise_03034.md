## Exercise
Write a program that takes a string as input and returns true if it is a valid password, and false otherwise.

## Reference Solution
```python
import re

def is_valid_password(password: str) -> bool:
    """
    Checks if a given string is a valid password.

    Args:
        password (str): The password to check.

    Returns:
        bool: True if the password is valid, False otherwise.
    """

    # Define the regular expression pattern for a valid password
    pattern = r"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*#?&])[A-Za-z\d@$!%*#?&]{8,}$"

    # Check if the password matches the pattern
    return bool(re.match(pattern, password))

# Example usage
if __name__ == "__main__":
    passwords = [
        "InvalidPassword",
        "ValidP@ssw0rd",
        "Short1@",
        "NoDigits",
        "NoSpecialChars123"
    ]

    for password in passwords:
        print(f"Password: {password}, Valid: {is_valid_password(password)}")
```
