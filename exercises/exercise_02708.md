## Exercise
Write a program to implement a simple text editor that can perform operations like insert, delete, and replace.

## Reference Solution
```python
class TextEditor:
    def __init__(self):
        self.text = ""

    def insert(self, text_to_insert, position):
        """
        Inserts the given text at the specified position.

        Args:
            text_to_insert (str): The text to be inserted.
            position (int): The position where the text will be inserted.

        Raises:
            ValueError: If the position is out of range.
        """
        if position < 0 or position > len(self.text):
            raise ValueError("Position out of range")
        self.text = self.text[:position] + text_to_insert + self.text[position:]

    def delete(self, start_position, end_position):
        """
        Deletes the text between the start and end positions.

        Args:
            start_position (int): The start position of the text to be deleted.
            end_position (int): The end position of the text to be deleted.

        Raises:
            ValueError: If the start or end position is out of range.
        """
        if start_position < 0 or start_position > len(self.text):
            raise ValueError("Start position out of range")
        if end_position < 0 or end_position > len(self.text):
            raise ValueError("End position out of range")
        if start_position > end_position:
            raise ValueError("Start position cannot be greater than end position")
        self.text = self.text[:start_position] + self.text[end_position:]

    def replace(self, old_text, new_text):
        """
        Replaces all occurrences of the old text with the new text.

        Args:
            old_text (str): The text to be replaced.
            new_text (str): The text to replace with.
        """
        self.text = self.text.replace(old_text, new_text)

    def display_text(self):
        """
        Displays the current text.
        """
        print(self.text)


def main():
    editor = TextEditor()

    while True:
        print("\nText Editor Menu:")
        print("1. Insert text")
        print("2. Delete text")
        print("3. Replace text")
        print("4. Display text")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            text_to_insert = input("Enter the text to insert: ")
            position = int(input("Enter the position to insert: "))
            editor.insert(text_to_insert, position)
        elif choice == "2":
            start_position = int(input("Enter the start position to delete: "))
            end_position = int(input("Enter the end position to delete: "))
            editor.delete(start_position, end_position)
        elif choice == "3":
            old_text = input("Enter the text to replace: ")
            new_text = input("Enter the new text: ")
            editor.replace(old_text, new_text)
        elif choice == "4":
            editor.display_text()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
