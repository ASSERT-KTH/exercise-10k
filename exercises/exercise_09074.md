## Exercise
Create a program that allows users to chat with each other using if statements to handle different messages

## Reference Solution
```python
# Define a class for the chat program
class ChatProgram:
    def __init__(self):
        self.user1_name = input("Enter User 1 name: ")
        self.user2_name = input("Enter User 2 name: ")
        self.chat_log = []

    # Method to display the chat log
    def display_chat_log(self):
        print("\nChat Log:")
        for i, message in enumerate(self.chat_log):
            print(f"{i+1}. {message}")

    # Method to handle user messages
    def handle_message(self, user_name, message):
        if message.lower() == "hello":
            self.chat_log.append(f"{user_name}: Hello!")
            print(f"{user_name}: Hello!")
        elif message.lower() == "how are you":
            self.chat_log.append(f"{user_name}: I'm good, thanks!")
            print(f"{user_name}: I'm good, thanks!")
        elif message.lower() == "what's your name":
            self.chat_log.append(f"{user_name}: My name is {user_name}")
            print(f"{user_name}: My name is {user_name}")
        elif message.lower() == "exit":
            self.chat_log.append(f"{user_name}: Exiting the chat")
            print(f"{user_name}: Exiting the chat")
            return False
        else:
            self.chat_log.append(f"{user_name}: {message}")
            print(f"{user_name}: {message}")
        return True

    # Method to start the chat
    def start_chat(self):
        print(f"\nWelcome to the chat, {self.user1_name} and {self.user2_name}!")
        current_user = self.user1_name
        while True:
            print(f"\n{current_user}'s turn:")
            message = input("Enter your message: ")
            if not self.handle_message(current_user, message):
                break
            # Switch the current user
            if current_user == self.user1_name:
                current_user = self.user2_name
            else:
                current_user = self.user1_name
        self.display_chat_log()

# Create an instance of the chat program and start the chat
if __name__ == "__main__":
    chat_program = ChatProgram()
    chat_program.start_chat()
```
