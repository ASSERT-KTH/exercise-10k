## Exercise
Create a program that sends a weather alert using if/else statements

## Reference Solution
```python
def send_weather_alert(temperature, precipitation, wind_speed):
    """
    Sends a weather alert based on the current weather conditions.

    Args:
        temperature (float): The current temperature in degrees Celsius.
        precipitation (float): The current precipitation in millimeters.
        wind_speed (float): The current wind speed in kilometers per hour.

    Returns:
        str: The weather alert message.
    """
    alert_message = ""

    if temperature < 0:
        alert_message += "Freezing temperatures warning: "
    elif temperature > 30:
        alert_message += "Heat wave warning: "

    if precipitation > 50:
        if alert_message:
            alert_message += " Heavy precipitation warning. "
        else:
            alert_message += "Heavy precipitation warning: "
    elif precipitation > 10:
        if alert_message:
            alert_message += " Precipitation warning. "
        else:
            alert_message += "Precipitation warning: "

    if wind_speed > 50:
        if alert_message:
            alert_message += " Strong wind warning. "
        else:
            alert_message += "Strong wind warning: "

    if alert_message:
        return f"Weather Alert: {alert_message}"
    else:
        return "No weather alerts."

# Example usage
if __name__ == "__main__":
    temperature = 25
    precipitation = 20
    wind_speed = 60

    alert = send_weather_alert(temperature, precipitation, wind_speed)
    print(alert)
```
