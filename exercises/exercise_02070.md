## Exercise
Write a program that generates all permutations of a given list of numbers.

## Reference Solution
```python
import itertools

def generate_permutations(numbers):
    """
    Generate all permutations of a given list of numbers.
    
    Args:
    numbers (list): A list of numbers.
    
    Returns:
    list: A list of lists, where each sublist is a permutation of the input list.
    """
    return list(itertools.permutations(numbers))

def main():
    numbers = [1, 2, 3]
    permutations = generate_permutations(numbers)
    
    print("Input List:", numbers)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i+1}: {permutation}")

if __name__ == "__main__":
    main()
```

```python
def generate_permutations(numbers):
    if len(numbers) == 1:
        return [numbers]
    
    permutations = []
    for i, num in enumerate(numbers):
        remaining_numbers = numbers[:i] + numbers[i+1:]
        for permutation in generate_permutations(remaining_numbers):
            permutations.append([num] + permutation)
    
    return permutations

def main():
    numbers = [1, 2, 3]
    permutations = generate_permutations(numbers)
    
    print("Input List:", numbers)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i+1}: {permutation}")

if __name__ == "__main__":
    main()
```
