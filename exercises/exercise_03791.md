## Exercise
Create a function that removes duplicates from a sorted array in-place.

## Reference Solution
```python
def remove_duplicates(sorted_array):
    """
    Removes duplicates from a sorted array in-place.

    Args:
        sorted_array (list): A sorted list of elements.

    Returns:
        int: The length of the array after removing duplicates.
    """
    # Check if the array is empty or contains only one element
    if len(sorted_array) < 2:
        return len(sorted_array)

    # Initialize two pointers
    i = 0  # The slow pointer
    j = 1  # The fast pointer

    # Traverse the array
    while j < len(sorted_array):
        # If the current element is not equal to the previous one
        if sorted_array[j] != sorted_array[i]:
            # Move the slow pointer one step forward
            i += 1
            # Replace the next element with the current one
            sorted_array[i] = sorted_array[j]
        # Move the fast pointer one step forward
        j += 1

    # Remove the duplicate elements
    del sorted_array[i + 1:]

    return len(sorted_array)

# Example usage:
sorted_array = [1, 1, 2, 3, 3, 3, 4, 5, 5]
print("Original array:", sorted_array)
new_length = remove_duplicates(sorted_array)
print("Array after removing duplicates:", sorted_array)
print("New length:", new_length)
```
