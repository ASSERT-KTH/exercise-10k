## Exercise
Write a program that reads and writes streams of data

## Reference Solution
```python
import io

class StreamReaderWriter:
    def __init__(self):
        pass

    def write_stream(self, data):
        """Write data to a stream"""
        stream = io.BytesIO()
        stream.write(data.encode('utf-8'))
        return stream

    def read_stream(self, stream):
        """Read data from a stream"""
        return stream.read().decode('utf-8')

def main():
    reader_writer = StreamReaderWriter()

    # Write data to a stream
    data = "Hello, World!"
    stream = reader_writer.write_stream(data)
    print(f"Data written to stream: {data}")

    # Read data from a stream
    read_data = reader_writer.read_stream(stream)
    print(f"Data read from stream: {read_data}")

if __name__ == "__main__":
    main()
```
