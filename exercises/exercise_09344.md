## Exercise
Write a program to demonstrate the concept of garbage collection.

## Reference Solution
```python
import gc
import os
import psutil

# Get the current process
process = psutil.Process(os.getpid())

class GarbageCollectionExample:
    def __init__(self):
        self.data = [i for i in range(1000000)]

    def __del__(self):
        print("Object has been deleted")

def main():
    # Get the initial memory usage
    mem_before = process.memory_info().rss / (1024 * 1024)
    print(f"Initial Memory Usage: {mem_before} MB")

    # Create an instance of the class
    obj = GarbageCollectionExample()

    # Get the memory usage after creating the object
    mem_after = process.memory_info().rss / (1024 * 1024)
    print(f"Memory Usage after creating object: {mem_after} MB")

    # Delete the object
    del obj

    # Force garbage collection
    gc.collect()

    # Get the memory usage after deleting the object and forcing garbage collection
    mem_after_gc = process.memory_info().rss / (1024 * 1024)
    print(f"Memory Usage after deleting object and forcing GC: {mem_after_gc} MB")

if __name__ == "__main__":
    main()
```
