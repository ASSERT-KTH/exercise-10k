## Exercise
Write a function that takes a dictionary and detects if it contains a cycle

## Reference Solution

def has_cycle(dictionary):
    """
    Detects if a dictionary contains a cycle.

    Args:
    dictionary (dict): The dictionary to check for cycles.

    Returns:
    bool: True if the dictionary contains a cycle, False otherwise.
    """
    visited = set()
    current_path = set()

    def dfs(key):
        # If the key is already in the current path, a cycle is detected
        if key in current_path:
            return True

        # If the key has already been visited and is not in the current path, skip it
        if key in visited:
            return False

        # Mark the key as visited and add it to the current path
        visited.add(key)
        current_path.add(key)

        # If the value is a dictionary, recursively check its keys
        if isinstance(dictionary.get(key), dict):
            for k in dictionary.get(key, {}):
                if dfs(k):
                    return True

        # Remove the key from the current path after visiting all its neighbors
        current_path.remove(key)

        return False

    # Check all keys in the dictionary
    for key in dictionary:
        if dfs(key):
            return True

    return False


# Example usage
if __name__ == "__main__":
    # Dictionary with a cycle
    cyclic_dict = {
        'a': {'b': 'c'},
        'b': {'c': 'a'}
    }
    print(has_cycle(cyclic_dict))  # Output: True

    # Dictionary without a cycle
    acyclic_dict = {
        'a': {'b': 'c'},
        'b': {'d': 'e'}
    }
    print(has_cycle(acyclic_dict))  # Output: False
