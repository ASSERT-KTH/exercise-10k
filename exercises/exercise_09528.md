## Exercise
Write a program that takes an array of numbers as input and returns the maximum number in the array. The program should iterate through the array and compare each number to find the maximum.

## Reference Solution
```python
#include <iostream>

using namespace std;

int maxOfArray(int arr[], int size) {
    int max = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    int max = maxOfArray(arr, size);
    cout << "Max element in array: " << max << endl;
    return 0;
}
```
