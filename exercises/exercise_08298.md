## Exercise
Create a program that calculates the exponential growth of a population.

## Reference Solution
```python
def exponential_growth(initial_population, growth_rate, time_period):
    """
    Calculate the exponential growth of a population.

    Args:
        initial_population (float): The initial population size.
        growth_rate (float): The growth rate of the population (in percent).
        time_period (int): The time period over which the growth occurs (in years).

    Returns:
        float: The final population size after the specified time period.
    """
    growth_rate = growth_rate / 100  # convert growth rate from percent to decimal
    final_population = initial_population * (1 + growth_rate) ** time_period
    return final_population

def main():
    print("Exponential Growth Calculator")
    print("--------------------------------")

    initial_population = float(input("Enter the initial population size: "))
    growth_rate = float(input("Enter the growth rate (in percent): "))
    time_period = int(input("Enter the time period (in years): "))

    final_population = exponential_growth(initial_population, growth_rate, time_period)

    print(f"The final population size after {time_period} years is: {final_population:.2f}")

if __name__ == "__main__":
    main()
```
