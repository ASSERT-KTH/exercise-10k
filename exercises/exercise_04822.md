## Exercise
Write a program to convert a string to an integer and vice versa, and print out the converted values.

## Reference Solution
```python
def convert_string_to_int(string):
    try:
        integer = int(string)
        print(f"String '{string}' converted to integer: {integer}")
        return integer
    except ValueError:
        print(f"Error: Unable to convert '{string}' to integer.")
        return None

def convert_int_to_string(integer):
    string = str(integer)
    print(f"Integer {integer} converted to string: '{string}'")
    return string

def main():
    # Test string to integer conversion
    string = "123"
    integer = convert_string_to_int(string)

    # Test integer to string conversion
    if integer is not None:
        convert_int_to_string(integer)

    # Test with a string that can't be converted to integer
    convert_string_to_int("hello")

    # Test integer to string conversion
    convert_int_to_string(456)

if __name__ == "__main__":
    main()
```
