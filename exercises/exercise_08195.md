## Exercise
Write a program that generates ASCII art using a nested for loop.

## Reference Solution
```python
def generate_ascii_art(n):
    """
    Generates ASCII art of a square filled with asterisks.

    Args:
    n (int): The size of the square.

    Returns:
    None
    """

    # Loop through each row of the square
    for i in range(n):
        # Loop through each column of the square
        for j in range(n):
            # Print an asterisk for each position in the square
            print("*", end=" ")
        # Move to the next line after printing each row
        print()

def generate_ascii_art_hollow_square(n):
    """
    Generates ASCII art of a hollow square filled with asterisks.

    Args:
    n (int): The size of the square.

    Returns:
    None
    """

    # Loop through each row of the square
    for i in range(n):
        # Loop through each column of the square
        for j in range(n):
            # Print an asterisk for the border of the square
            if i == 0 or i == n - 1 or j == 0 or j == n - 1:
                print("*", end=" ")
            # Print a space for the inside of the square
            else:
                print(" ", end=" ")
        # Move to the next line after printing each row
        print()

def generate_ascii_art_cross(n):
    """
    Generates ASCII art of a cross filled with asterisks.

    Args:
    n (int): The size of the cross.

    Returns:
    None
    """

    # Loop through each row of the cross
    for i in range(n):
        # Loop through each column of the cross
        for j in range(n):
            # Print an asterisk for the middle row and column
            if i == n // 2 or j == n // 2:
                print("*", end=" ")
            # Print a space for the rest of the cross
            else:
                print(" ", end=" ")
        # Move to the next line after printing each row
        print()

def main():
    n = 10
    print("Solid Square:")
    generate_ascii_art(n)
    print("\nHollow Square:")
    generate_ascii_art_hollow_square(n)
    print("\nCross:")
    generate_ascii_art_cross(n)

if __name__ == "__main__":
    main()
```
