## Exercise
Create a function that creates a pipeline of functions, applying each function in sequence to the input data.

## Reference Solution
```python
def create_pipeline(functions):
    """
    Creates a pipeline of functions, applying each function in sequence to the input data.

    Args:
        functions (list): A list of functions to be applied in sequence.

    Returns:
        function: A function that applies each function in the pipeline in sequence.
    """

    # Define a new function that will apply the functions in the pipeline
    def pipeline(data):
        
        # Initialize the result with the input data
        result = data
        
        # Iterate over each function in the pipeline
        for func in functions:
            
            # Apply the current function to the result and update the result
            result = func(result)
        
        # Return the final result after applying all functions in the pipeline
        return result
    
    # Return the pipeline function
    return pipeline


# Example usage:

# Define some example functions
def double(x):
    return x * 2

def add_one(x):
    return x + 1

def square(x):
    return x ** 2

# Create a pipeline with the example functions
pipeline = create_pipeline([double, add_one, square])

# Apply the pipeline to some input data
result = pipeline(5)

print(result)  # Output: 121
```
