## Exercise
Create a program that takes an input string and converts it to different data types (integer, float, boolean) based on the user's choice. The program should handle invalid inputs and provide meaningful error messages.

## Reference Solution
```python
#include <iostream>
#include <string>
#include <stdexcept>

void convertToInteger(const std::string& input) {
    try {
        int convertedValue = std::stoi(input);
        std::cout << "The integer value of '" << input << "' is: " << convertedValue << std::endl;
    } catch (const std::invalid_argument& e) {
        std::cerr << "Error: Invalid argument. Unable to convert '" << input << "' to an integer." << std::endl;
    } catch (const std::out_of_range& e) {
        std::cerr << "Error: Out of range. The absolute value of '" << input << "' is too large to be represented as an integer." << std::endl;
    }
}

void convertToFloat(const std::string& input) {
    try {
        float convertedValue = std::stof(input);
        std::cout << "The float value of '" << input << "' is: " << convertedValue << std::endl;
    } catch (const std::invalid_argument& e) {
        std::cerr << "Error: Invalid argument. Unable to convert '" << input << "' to a float." << std::endl;
    } catch (const std::out_of_range& e) {
        std::cerr << "Error: Out of range. The absolute value of '" << input << "' is too large to be represented as a float." << std::endl;
    }
}

void convertToBoolean(const std::string& input) {
    if (input == "true") {
        std::cout << "The boolean value of '" << input << "' is: true" << std::endl;
    } else if (input == "false") {
        std::cout << "The boolean value of '" << input << "' is: false" << std::endl;
    } else {
        std::cerr << "Error: Invalid boolean value. '" << input << "' is not a valid boolean value." << std::endl;
    }
}

int main() {
    std::string input;
    std::cout << "Enter a string: ";
    std::cin >> input;

    std::cout << "Enter a data type (1 for integer, 2 for float, 3 for boolean): ";
    int choice;
    std::cin >> choice;

    switch (choice) {
        case 1:
            convertToInteger(input);
            break;
        case 2:
            convertToFloat(input);
            break;
        case 3:
            convertToBoolean(input);
            break;
        default:
            std::cerr << "Error: Invalid choice. Please choose a valid data type." << std::endl;
    }

    return 0;
}
```
