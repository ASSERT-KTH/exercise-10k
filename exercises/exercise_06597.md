## Exercise
Write a program to find the most frequent character in a string.

## Reference Solution
```python
from collections import Counter

def most_frequent_char(s):
    """
    Find the most frequent character in a string.

    Args:
        s (str): Input string.

    Returns:
        tuple: (most_frequent_char, frequency)
    """
    # Remove whitespace and convert to lowercase
    s = s.replace(" ", "").lower()

    # Count character frequencies
    char_counts = Counter(s)

    # Find the most frequent character
    most_frequent_char = max(char_counts, key=char_counts.get)

    # Get its frequency
    frequency = char_counts[most_frequent_char]

    return most_frequent_char, frequency

def main():
    # Example usage
    input_str = "Hello, World!"
    most_frequent, frequency = most_frequent_char(input_str)
    print(f"Most frequent character: '{most_frequent}' (appears {frequency} times)")

if __name__ == "__main__":
    main()
```
