## Exercise
Write a program that demonstrates the concept of variable scope (global and local variables).

## Reference Solution
```python
# Global variable
global_variable = 10

def local_scope_example():
    """
    Demonstrates local variable scope.
    """
    # Local variable
    local_variable = 20
    print("Inside local_scope_example:")
    print(f"Global variable: {global_variable}")
    print(f"Local variable: {local_variable}")

def global_scope_example():
    """
    Demonstrates global variable scope.
    """
    global global_variable
    global_variable = 30
    print("Inside global_scope_example:")
    print(f"Global variable: {global_variable}")

def nonlocal_scope_example():
    """
    Demonstrates nonlocal variable scope.
    """
    def nested_function():
        nonlocal local_variable
        local_variable = 40
        print("Inside nested_function:")
        print(f"Local variable: {local_variable}")

    local_variable = 50
    print("Before calling nested_function:")
    print(f"Local variable: {local_variable}")
    nested_function()
    print("After calling nested_function:")
    print(f"Local variable: {local_variable}")

def main():
    print("Before calling local_scope_example:")
    print(f"Global variable: {global_variable}")

    local_scope_example()

    print("After calling local_scope_example:")
    print(f"Global variable: {global_variable}")

    global_scope_example()

    print("After calling global_scope_example:")
    print(f"Global variable: {global_variable}")

    nonlocal_scope_example()

if __name__ == "__main__":
    main()
```
