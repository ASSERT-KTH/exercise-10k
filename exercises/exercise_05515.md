## Exercise
Write a program to find the strongly connected components of a graph using Tarjan's algorithm.

## Reference Solution
```python
from collections import defaultdict

class Tarjan:
    def __init__(self, graph):
        """
        Initialize the Tarjan algorithm with a graph.

        Args:
        graph (dict): A dictionary representing the graph, where each key is a node and its corresponding value is a list of its neighbors.
        """
        self.graph = graph
        self.index = 0
        self.stack = []
        self.lowlinks = {}
        self.indexes = {}
        self.result = []

    def strong_connect(self, node):
        """
        Recursive function to find strongly connected components.

        Args:
        node: The current node being processed.
        """
        self.indexes[node] = self.index
        self.lowlinks[node] = self.index
        self.index += 1
        self.stack.append(node)

        for successor in self.graph[node]:
            if successor not in self.lowlinks:
                self.strong_connect(successor)
                self.lowlinks[node] = min(self.lowlinks[node], self.lowlinks[successor])
            elif successor in self.stack:
                self.lowlinks[node] = min(self.lowlinks[node], self.indexes[successor])

        if self.lowlinks[node] == self.indexes[node]:
            connected_component = []
            while True:
                successor = self.stack.pop()
                connected_component.append(successor)
                if successor == node:
                    break
            self.result.append(connected_component)

    def find_scc(self):
        """
        Find strongly connected components in the graph.
        """
        for node in self.graph:
            if node not in self.lowlinks:
                self.strong_connect(node)
        return self.result


# Example usage
if __name__ == "__main__":
    graph = {
        'A': ['B'],
        'B': ['C', 'D'],
        'C': ['A', 'D'],
        'D': ['E'],
        'E': []
    }

    tarjan = Tarjan(graph)
    scc = tarjan.find_scc()
    print("Strongly Connected Components:")
    for i, component in enumerate(scc):
        print(f"Component {i+1}: {component}")
```
