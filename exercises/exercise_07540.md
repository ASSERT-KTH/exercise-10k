## Exercise
Create a program that takes a string input from the user, checks if it contains only alphabets, and then outputs the string with all vowels removed. If the input contains non-alphabet characters, the program should output an error message.

## Reference Solution
```python
#include <stdio.h>
#include <ctype.h>

int main() {
    char str[100];

    printf("Enter a string: ");
    scanf("%s", str);

    int i, j;
    for (i = 0, j = 0; str[i] != '\0'; i++) {
        if (isalpha(str[i])) {
            str[j++] = str[i];
        } else {
            printf("Error: Input string contains non-alphabet characters.\n");
            return 1;
        }
    }
    str[j] = '\0';

    for (i = 0; i < j; i++) {
        if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u' ||
            str[i] == 'A' || str[i] == 'E' || str[i] == 'I' || str[i] == 'O' || str[i] == 'U') {
            for (int k = i; k < j - 1; k++) {
                str[k] = str[k + 1];
            }
            j--;
            i--;
        }
    }

    printf("String without vowels: ");
    for (i = 0; i < j; i++) {
        printf("%c", str[i]);
    }
    printf("\n");

    return 0;
}
```
