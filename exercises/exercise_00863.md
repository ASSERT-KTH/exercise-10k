## Exercise
Create a program that allows two players to play Rock, Paper, Scissors. The program should determine the winner based on the game's rules and output the result.

## Reference Solution
```python
import random

def game():
    while True:
        user1 = input("Player 1, enter a choice (rock, paper, scissors): ").lower()
        user2 = input("Player 2, enter a choice (rock, paper, scissors): ").lower()

        while user1 not in ["rock", "paper", "scissors"]:
            user1 = input("Invalid input. Player 1, enter a choice (rock, paper, scissors): ").lower()
        while user2 not in ["rock", "paper", "scissors"]:
            user2 = input("Invalid input. Player 2, enter a choice (rock, paper, scissors): ").lower()

        print(f"\nPlayer 1 chose {user1}, Player 2 chose {user2}.\n")

        if user1 == user2:
            print(f"Both players selected {user1}. It's a tie!")
        elif user1 == "rock":
            if user2 == "scissors":
                print("Rock smashes scissors! Player 1 wins!")
            else:
                print("Paper covers rock! Player 2 wins.")
        elif user1 == "paper":
            if user2 == "rock":
                print("Paper covers rock! Player 1 wins.")
            else:
                print("Scissors cuts paper! Player 2 wins.")
        elif user1 == "scissors":
            if user2 == "paper":
                print("Scissors cuts paper! Player 1 wins.")
            else:
                print("Rock smashes scissors! Player 2 wins.")

        play_again = input("Play again? (yes/no): ").lower()
        while play_again not in ["yes", "no"]:
            play_again = input("Invalid input. Play again? (yes/no): ").lower()
        if play_again != "yes":
            break

if __name__ == "__main__":
    game()
```
