## Exercise
Create a program that takes two numbers and an arithmetic operation as input and returns the result. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
# Define a function to perform arithmetic operations
def arithmetic_operation(num1, num2, operation):
    if operation == "addition":
        return num1 + num2
    elif operation == "subtraction":
        return num1 - num2
    elif operation == "multiplication":
        return num1 * num2
    elif operation == "division":
        if num2 != 0:
            return num1 / num2
        else:
            return "Error: Division by zero is not allowed"
    else:
        return "Error: Invalid operation. Please choose from addition, subtraction, multiplication, or division"


# Main function
def main():
    # Take user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (addition, subtraction, multiplication, division): ")

    # Perform the operation and print the result
    result = arithmetic_operation(num1, num2, operation)
    print(f"The result of {operation} is: {result}")


# Run the main function
if __name__ == "__main__":
    main()
```
