## Exercise
Write a program that registers a student for a course using if/else statements

## Reference Solution
```python
class Student:
    def __init__(self, name, age, courses_taken):
        self.name = name
        self.age = age
        self.courses_taken = courses_taken

class Course:
    def __init__(self, name, prerequisites):
        self.name = name
        self.prerequisites = prerequisites

def register_student(student, course):
    # Check if student is eligible to register for the course
    if student.age >= 18:
        # Check if student has taken all prerequisites
        if all(prerequisite in student.courses_taken for prerequisite in course.prerequisites):
            print(f"{student.name} has been successfully registered for {course.name}.")
            return True
        else:
            print(f"{student.name} has not taken all prerequisites for {course.name}.")
            return False
    else:
        print(f"{student.name} is not eligible to register for {course.name} due to age restrictions.")
        return False

# Example usage:
student1 = Student("John Doe", 20, ["Math 101", "Science 102"])
course1 = Course("Computer Science 103", ["Math 101", "Science 102"])

student2 = Student("Jane Doe", 17, ["Math 101"])
course2 = Course("Computer Science 103", ["Math 101", "Science 102"])

register_student(student1, course1)
register_student(student2, course2)
```
