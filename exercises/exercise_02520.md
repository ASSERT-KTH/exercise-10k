## Exercise
Create a program that takes a linear equation as input, and returns the solution.

## Reference Solution
```python
def solve_linear_equation(a, b, c):
    """
    Solves a linear equation of the form ax + b = c for x.

    Args:
        a (float): The coefficient of x.
        b (float): The constant term on the left side of the equation.
        c (float): The constant term on the right side of the equation.

    Returns:
        float: The solution for x.

    Raises:
        ZeroDivisionError: If a is zero.
    """
    if a == 0:
        if b == c:
            return "Infinite solutions"
        else:
            return "No solution"
    else:
        return (c - b) / a

def main():
    print("Linear Equation Solver")
    print("------------------------")
    a = float(input("Enter the coefficient of x (a): "))
    b = float(input("Enter the constant term on the left side of the equation (b): "))
    c = float(input("Enter the constant term on the right side of the equation (c): "))

    solution = solve_linear_equation(a, b, c)
    print(f"The solution for x is: {solution}")

if __name__ == "__main__":
    main()
```
