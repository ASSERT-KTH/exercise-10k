## Exercise
Write a program that takes two numbers as input and returns their modulus

## Reference Solution
```python
def calculate_modulus(dividend: int, divisor: int) -> int:
    """
    Calculate the modulus of two numbers.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        int: The modulus of the two numbers.

    Raises:
        ZeroDivisionError: If the divisor is zero.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend % divisor

def main():
    # Get user input
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))

    try:
        # Calculate and print the modulus
        modulus = calculate_modulus(dividend, divisor)
        print(f"The modulus of {dividend} and {divisor} is: {modulus}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
