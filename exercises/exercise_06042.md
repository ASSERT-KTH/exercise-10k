## Exercise
Write a program that takes in a percentage score and converts it to a letter grade using conditional statements.

## Reference Solution
```python
def convert_to_letter_grade(score):
    """
    Converts a percentage score to a letter grade.

    Args:
        score (float): The percentage score.

    Returns:
        str: The corresponding letter grade.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the percentage score from the user
    score = float(input("Enter your percentage score: "))

    # Validate the score
    if score < 0 or score > 100:
        print("Invalid score. Please enter a score between 0 and 100.")
    else:
        # Convert the score to a letter grade
        letter_grade = convert_to_letter_grade(score)
        print(f"Your letter grade is: {letter_grade}")

if __name__ == "__main__":
    main()
```
