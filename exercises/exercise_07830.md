## Exercise
Write a program that takes an array of integers as input and returns the maximum value in the array. The program should use a basic algorithm to iterate through the array and find the maximum value.

## Reference Solution
```python
#include <iostream>
using namespace std;

int findMax(int arr[], int n) {
    int max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int n = sizeof(arr) / sizeof(arr[0]);
    int max = findMax(arr, n);
    cout << "Maximum value: " << max << endl;
    return 0;
}
```
