## Exercise
Write a function to find the next greater element for each element in an array.

def next_greater_element(arr):
    """
    Find the next greater element for each element in an array.

    Args:
    arr (list): The input array.

    Returns:
    list: A list of next greater elements. If no greater element is found, -1 is returned.
    """
    stack = []
    result = [-1] * len(arr)

    for i in range(len(arr)):
        # While the stack is not empty and the current element is greater than the top of the stack
        while stack and arr[i] > arr[stack[-1]]:
            # Pop the top of the stack and update the result
            idx = stack.pop()
            result[idx] = arr[i]
        # Push the current index to the stack
        stack.append(i)

    return result

# Example usage:
arr = [4, 5, 2, 10]
print(next_greater_element(arr))  # Output: [5, 10, 10, -1]
