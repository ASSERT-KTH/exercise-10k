## Exercise
Given two arrays of integers, write a program to compare them and determine if they are equal.

## Reference Solution
```python
def compare_arrays(arr1, arr2):
    """
    Compare two arrays of integers and determine if they are equal.

    Args:
        arr1 (list): The first array of integers.
        arr2 (list): The second array of integers.

    Returns:
        bool: True if the arrays are equal, False otherwise.
    """
    # Check if the lengths of the arrays are equal
    if len(arr1) != len(arr2):
        return False

    # Compare the elements of the arrays
    for i in range(len(arr1)):
        if arr1[i] != arr2[i]:
            return False

    # If no differences are found, the arrays are equal
    return True


def main():
    # Example usage
    arr1 = [1, 2, 3, 4, 5]
    arr2 = [1, 2, 3, 4, 5]
    arr3 = [1, 2, 3, 4, 6]

    print(compare_arrays(arr1, arr2))  # Output: True
    print(compare_arrays(arr1, arr3))  # Output: False


if __name__ == "__main__":
    main()
```
