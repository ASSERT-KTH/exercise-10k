## Exercise
Write a program that reads an XML file and prints out the file's contents.

## Reference Solution
```python
import xml.etree.ElementTree as ET

def parse_xml_file(file_path):
    """
    Parse an XML file and print its contents.

    Args:
        file_path (str): Path to the XML file
    """
    try:
        # Parse the XML file
        tree = ET.parse(file_path)
        root = tree.getroot()

        # Print the XML contents
        print(ET.tostring(root, encoding='unicode'))

    except FileNotFoundError:
        print(f"File not found: {file_path}")
    except ET.ParseError as e:
        print(f"Error parsing XML file: {e}")

def main():
    file_path = "example.xml"  # Replace with your XML file path
    parse_xml_file(file_path)

if __name__ == "__main__":
    main()
```
