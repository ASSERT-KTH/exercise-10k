## Exercise
Declare two variables and perform bitwise operations (AND, OR, XOR, NOT) on them.

## Reference Solution
```python
def bitwise_operations():
    # Declare two variables
    var1 = 5   # Binary representation: 101
    var2 = 3   # Binary representation: 011

    # Perform bitwise operations
    and_result = var1 & var2  # Bitwise AND
    or_result = var1 | var2   # Bitwise OR
    xor_result = var1 ^ var2  # Bitwise XOR
    not_result1 = ~var1       # Bitwise NOT on var1
    not_result2 = ~var2       # Bitwise NOT on var2

    # Print the results
    print(f"Variable 1 (var1): {var1} (Binary: {bin(var1)})")
    print(f"Variable 2 (var2): {var2} (Binary: {bin(var2)})")
    print(f"Bitwise AND (var1 & var2): {and_result} (Binary: {bin(and_result)})")
    print(f"Bitwise OR (var1 | var2): {or_result} (Binary: {bin(or_result)})")
    print(f"Bitwise XOR (var1 ^ var2): {xor_result} (Binary: {bin(xor_result)})")
    print(f"Bitwise NOT (~var1): {not_result1} (Binary: {bin(not_result1)})")
    print(f"Bitwise NOT (~var2): {not_result2} (Binary: {bin(not_result2)})")

# Example usage
bitwise_operations()
```
