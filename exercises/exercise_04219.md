## Exercise
Calculate the quartiles (Q1, Q2, Q3) of an array.

## Reference Solution
```python
import numpy as np

def calculate_quartiles(array):
    """
    Calculate the quartiles (Q1, Q2, Q3) of an array.

    Parameters:
    array (list): The input array.

    Returns:
    tuple: A tuple containing Q1, Q2, and Q3.
    """
    # Sort the array
    sorted_array = sorted(array)

    # Find the median (Q2)
    q2 = np.median(sorted_array)

    # Find Q1 (first quartile)
    lower_half = sorted_array[:len(sorted_array)//2]
    q1 = np.median(lower_half)

    # Find Q3 (third quartile)
    upper_half = sorted_array[len(sorted_array)//2 + len(sorted_array)%2:]
    q3 = np.median(upper_half)

    return q1, q2, q3

# Example usage
array = [1, 3, 5, 7, 9, 11, 13, 15]
q1, q2, q3 = calculate_quartiles(array)
print(f"Q1: {q1}, Q2: {q2}, Q3: {q3}")
```
