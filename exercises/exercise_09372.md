## Exercise
Write a program that demonstrates string manipulation techniques.

## Reference Solution
```python
def demonstrate_string_manipulation():
    # Define a sample string
    sample_string = "Hello, World!"

    # Concatenation
    print("**Concatenation**")
    concatenated_string = sample_string + " This is a test."
    print(f"Original string: {sample_string}")
    print(f"Concatenated string: {concatenated_string}")

    # Substring extraction
    print("\n**Substring Extraction**")
    substring = sample_string[0:5]
    print(f"Original string: {sample_string}")
    print(f"Substring (0-5): {substring}")

    # String replacement
    print("\n**String Replacement**")
    replaced_string = sample_string.replace("World", "Universe")
    print(f"Original string: {sample_string}")
    print(f"Replaced string: {replaced_string}")

    # Case conversion
    print("\n**Case Conversion**")
    upper_case_string = sample_string.upper()
    lower_case_string = sample_string.lower()
    print(f"Original string: {sample_string}")
    print(f"Upper case string: {upper_case_string}")
    print(f"Lower case string: {lower_case_string}")

    # String splitting
    print("\n**String Splitting**")
    split_string = sample_string.split(", ")
    print(f"Original string: {sample_string}")
    print(f"Split string: {split_string}")

    # String joining
    print("\n**String Joining**")
    join_string = "-".join(["Hello", "World", "This", "is", "a", "test"])
    print(f"Original list: ['Hello', 'World', 'This', 'is', 'a', 'test']")
    print(f"Joined string: {join_string}")


if __name__ == "__main__":
    demonstrate_string_manipulation()
```
