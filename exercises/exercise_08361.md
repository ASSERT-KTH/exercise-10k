def add_numbers(num1, num2):
    """
    This function takes two numbers as input and returns their sum.
    
    Parameters:
    num1 (float): The first number
    num2 (float): The second number
    
    Returns:
    float: The sum of num1 and num2
    """
    return num1 + num2
