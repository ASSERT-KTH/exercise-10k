## Exercise
Create a program that uses a for loop to check if a string is a palindrome.

## Reference Solution
```python
def is_palindrome(s: str) -> bool:
    """
    Checks if a string is a palindrome.

    Args:
    s (str): The input string.

    Returns:
    bool: True if the string is a palindrome, False otherwise.
    """
    s = ''.join(c for c in s if c.isalnum()).lower()  # Remove non-alphanumeric characters and convert to lowercase
    left, right = 0, len(s) - 1  # Initialize two pointers, one at the start and one at the end of the string

    for _ in range(len(s) // 2):  # Loop through the first half of the string
        if s[left] != s[right]:  # If the characters at the current positions do not match
            return False  # Return False, the string is not a palindrome
        left += 1  # Move the left pointer to the right
        right -= 1  # Move the right pointer to the left

    return True  # If the loop completes without finding any mismatches, the string is a palindrome


def main():
    # Example usage:
    strings = ["radar", "hello", "A man, a plan, a canal: Panama"]

    for s in strings:
        result = is_palindrome(s)
        print(f"'{s}' is a palindrome: {result}")


if __name__ == "__main__":
    main()
```
