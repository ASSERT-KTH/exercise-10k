## Exercise
Create a program that determines a phone plan based on the number of minutes and data used. The program should use if/else statements to determine the phone plan.

## Reference Solution
```python
def determine_phone_plan(minutes, data):
    """
    Determine a phone plan based on the number of minutes and data used.

    Args:
        minutes (int): The number of minutes used.
        data (int): The amount of data used in GB.

    Returns:
        str: The determined phone plan.
    """

    # Define the phone plans
    plans = {
        "Basic": {"minutes": 0, "data": 0, "minutes_limit": 500, "data_limit": 2},
        "Standard": {"minutes": 0, "data": 0, "minutes_limit": 1000, "data_limit": 5},
        "Premium": {"minutes": 0, "data": 0, "minutes_limit": 2000, "data_limit": 10},
        "Unlimited": {"minutes": 0, "data": 0, "minutes_limit": float("inf"), "data_limit": float("inf")},
    }

    # Check if the user qualifies for the Unlimited plan
    if minutes > plans["Premium"]["minutes_limit"] or data > plans["Premium"]["data_limit"]:
        return "Unlimited"

    # Check if the user qualifies for the Premium plan
    elif minutes > plans["Standard"]["minutes_limit"] or data > plans["Standard"]["data_limit"]:
        return "Premium"

    # Check if the user qualifies for the Standard plan
    elif minutes > plans["Basic"]["minutes_limit"] or data > plans["Basic"]["data_limit"]:
        return "Standard"

    # If none of the above conditions are met, the user qualifies for the Basic plan
    else:
        return "Basic"


def main():
    # Get the number of minutes and data used from the user
    minutes = int(input("Enter the number of minutes used: "))
    data = int(input("Enter the amount of data used in GB: "))

    # Determine the phone plan
    plan = determine_phone_plan(minutes, data)

    # Print the determined phone plan
    print(f"Based on your usage, you qualify for the {plan} plan.")


if __name__ == "__main__":
    main()
```
