## Exercise
Write a program that acts as a simple chatbot and responds to basic user queries using if statements

## Reference Solution
```python
# Import the required libraries
import datetime

# Define a function to get the current time
def get_current_time():
    return datetime.datetime.now().strftime("%H:%M")

# Define a function to get the current date
def get_current_date():
    return datetime.datetime.now().strftime("%Y-%m-%d")

# Define the main function for the chatbot
def chatbot():
    print("Welcome to the simple chatbot!")
    print("Type 'exit' to quit the chatbot.")

    while True:
        user_input = input("User: ")

        # Check if the user wants to exit the chatbot
        if user_input.lower() == "exit":
            print("Goodbye!")
            break

        # Check if the user is asking for the current time
        elif "time" in user_input.lower():
            print(f"Current time: {get_current_time()}")

        # Check if the user is asking for the current date
        elif "date" in user_input.lower():
            print(f"Current date: {get_current_date()}")

        # Check if the user is asking for their name
        elif "name" in user_input.lower():
            print("I don't know your name. Please tell me!")

        # Check if the user is telling their name
        elif "my name is" in user_input.lower():
            name = user_input.split("is ")[1]
            print(f"Nice to meet you, {name}!")

        # Check if the user is asking for help
        elif "help" in user_input.lower():
            print("Available commands:")
            print("  - time: Get the current time")
            print("  - date: Get the current date")
            print("  - name: Tell me your name")

        # If none of the above conditions are met, respond with a default message
        else:
            print("I didn't understand that. Please try again!")

# Run the chatbot
if __name__ == "__main__":
    chatbot()
```
