## Exercise
Write a program to calculate the interest earned on a deposit

## Reference Solution
```python
def calculate_interest(principal, rate, time):
    """
    Calculate the interest earned on a deposit.

    Args:
        principal (float): The initial deposit amount.
        rate (float): The annual interest rate (in decimal form, e.g., 4% = 0.04).
        time (int): The time period in years.

    Returns:
        float: The interest earned.
    """
    interest = principal * rate * time
    return interest

def main():
    print("Deposit Interest Calculator")
    print("---------------------------")

    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))

    # Calculate and display interest
    interest = calculate_interest(principal, rate, time)
    print(f"Interest earned: ${interest:.2f}")
    print(f"Total amount after {time} years: ${principal + interest:.2f}")

if __name__ == "__main__":
    main()
```
