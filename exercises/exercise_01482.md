## Exercise
Write a program that takes a variable as input and converts it to a different data type.

## Reference Solution
```python
def convert_data_type(input_variable, output_type):
    """
    Converts the input variable to the specified output type.

    Args:
        input_variable: The variable to be converted.
        output_type (type): The desired output type.

    Returns:
        The converted variable.
    """

    try:
        # Attempt to convert the variable to the specified type
        converted_variable = output_type(input_variable)
        return converted_variable
    except ValueError:
        print("Invalid conversion. Please ensure the input variable can be converted to the specified type.")
        return None
    except TypeError:
        print("Invalid conversion. Please ensure the output type is a valid Python type.")
        return None


# Example usage
if __name__ == "__main__":
    input_variable = "123.45"
    output_type = float

    converted_variable = convert_data_type(input_variable, output_type)

    if converted_variable is not None:
        print(f"Converted {input_variable} to {output_type.__name__}: {converted_variable}")
```
