## Exercise
Create a program that takes in a shape and calculates its perimeter using conditional statements.

## Reference Solution
```python
import math

def calculate_perimeter(shape, **kwargs):
    """
    Calculates the perimeter of a given shape.

    Args:
        shape (str): The type of shape (rectangle, square, circle, triangle)
        **kwargs: Additional arguments depending on the shape
            - rectangle: length, width
            - square: side
            - circle: radius
            - triangle: side1, side2, side3

    Returns:
        float: The perimeter of the shape
    """
    if shape == "rectangle":
        length = kwargs["length"]
        width = kwargs["width"]
        perimeter = 2 * (length + width)
    elif shape == "square":
        side = kwargs["side"]
        perimeter = 4 * side
    elif shape == "circle":
        radius = kwargs["radius"]
        perimeter = 2 * math.pi * radius
    elif shape == "triangle":
        side1 = kwargs["side1"]
        side2 = kwargs["side2"]
        side3 = kwargs["side3"]
        perimeter = side1 + side2 + side3
    else:
        raise ValueError("Invalid shape")

    return perimeter

def main():
    print("Shape Perimeter Calculator")
    print("----------------------------")

    shape = input("Enter the shape (rectangle, square, circle, triangle): ")

    if shape == "rectangle":
        length = float(input("Enter the length: "))
        width = float(input("Enter the width: "))
        perimeter = calculate_perimeter(shape, length=length, width=width)
    elif shape == "square":
        side = float(input("Enter the side: "))
        perimeter = calculate_perimeter(shape, side=side)
    elif shape == "circle":
        radius = float(input("Enter the radius: "))
        perimeter = calculate_perimeter(shape, radius=radius)
    elif shape == "triangle":
        side1 = float(input("Enter the first side: "))
        side2 = float(input("Enter the second side: "))
        side3 = float(input("Enter the third side: "))
        perimeter = calculate_perimeter(shape, side1=side1, side2=side2, side3=side3)
    else:
        print("Invalid shape")
        return

    print(f"The perimeter of the {shape} is: {perimeter:.2f}")

if __name__ == "__main__":
    main()
```
