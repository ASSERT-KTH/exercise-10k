## Exercise
Create a program that reads a JSON file from the user, processes the data, and prints out the results.

## Reference Solution
```python
import json

def process_json_data(file_path):
    """
    Reads a JSON file and processes the data.

    Args:
        file_path (str): Path to the JSON file

    Returns:
        dict: Processed JSON data
    """
    try:
        with open(file_path, 'r') as file:
            data = json.load(file)
            return data
    except FileNotFoundError:
        print("File not found. Please check the file path.")
        return None
    except json.JSONDecodeError:
        print("Invalid JSON format. Please check the file contents.")
        return None

def print_json_data(data):
    """
    Prints the processed JSON data.

    Args:
        data (dict): Processed JSON data
    """
    if data is not None:
        print("Processed JSON Data:")
        print(json.dumps(data, indent=4))

def main():
    file_path = input("Enter the JSON file path: ")
    data = process_json_data(file_path)
    print_json_data(data)

if __name__ == "__main__":
    main()
```
