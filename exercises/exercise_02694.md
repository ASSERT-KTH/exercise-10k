## Exercise
Write a program that takes a password as input and determines its strength based on the following conditions: the password must be at least 8 characters long, contain at least one uppercase letter, one lowercase letter, and one number.

## Reference Solution
```python
import re

def check_password_strength(password):
    """
    Checks the strength of a given password.

    Args:
        password (str): The password to check.

    Returns:
        bool: True if the password is strong, False otherwise.
    """
    # Define the conditions for a strong password
    conditions = {
        "length": len(password) >= 8,
        "uppercase": any(char.isupper() for char in password),
        "lowercase": any(char.islower() for char in password),
        "number": any(char.isdigit() for char in password),
    }

    # Check if all conditions are met
    if all(conditions.values()):
        return True
    else:
        return False

def get_password_feedback(password):
    """
    Provides feedback on the strength of a given password.

    Args:
        password (str): The password to check.

    Returns:
        str: A message indicating the strength of the password and any missing conditions.
    """
    # Define the conditions for a strong password
    conditions = {
        "length": len(password) >= 8,
        "uppercase": any(char.isupper() for char in password),
        "lowercase": any(char.islower() for char in password),
        "number": any(char.isdigit() for char in password),
    }

    # Initialize the feedback message
    feedback = "Password strength: "

    # Check if all conditions are met
    if all(conditions.values()):
        feedback += "Strong"
    else:
        feedback += "Weak"
        missing_conditions = [key for key, value in conditions.items() if not value]
        if "length" in missing_conditions:
            feedback += "\n* Password must be at least 8 characters long."
        if "uppercase" in missing_conditions:
            feedback += "\n* Password must contain at least one uppercase letter."
        if "lowercase" in missing_conditions:
            feedback += "\n* Password must contain at least one lowercase letter."
        if "number" in missing_conditions:
            feedback += "\n* Password must contain at least one number."

    return feedback

def main():
    # Get the password from the user
    password = input("Enter a password: ")

    # Check the password strength
    if check_password_strength(password):
        print("Password is strong.")
    else:
        print(get_password_feedback(password))

if __name__ == "__main__":
    main()
```
