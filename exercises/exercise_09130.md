## Exercise
Create a program that checks the data type of user input and validates if it matches the expected data type. The program should ask the user for their input and the expected data type. It should then print out whether the input is valid or not.

## Reference Solution
```python
def get_user_input():
    """Get user input and expected data type from the user."""
    user_input = input("Please enter your input: ")
    expected_data_type = input("Please enter the expected data type (int, float, str, bool): ")
    return user_input, expected_data_type

def validate_input(user_input, expected_data_type):
    """Validate the user input against the expected data type."""
    try:
        # Attempt to convert the user input to the expected data type
        if expected_data_type == "int":
            int(user_input)
        elif expected_data_type == "float":
            float(user_input)
        elif expected_data_type == "str":
            str(user_input)
        elif expected_data_type == "bool":
            if user_input.lower() == "true":
                bool_val = True
            elif user_input.lower() == "false":
                bool_val = False
            else:
                raise ValueError
        else:
            raise ValueError("Invalid expected data type")

        # If the conversion is successful, the input is valid
        return True

    except ValueError:
        # If a ValueError is raised, the input is invalid
        return False

def main():
    user_input, expected_data_type = get_user_input()
    is_valid = validate_input(user_input, expected_data_type)

    if is_valid:
        print(f"The input '{user_input}' is valid for the expected data type '{expected_data_type}'")
    else:
        print(f"The input '{user_input}' is not valid for the expected data type '{expected_data_type}'")

if __name__ == "__main__":
    main()
```
