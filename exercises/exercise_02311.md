## Exercise
Write a program that generates a scrambled version of a word and asks the user to unscramble it.

## Reference Solution
```python
import random

# List of words to choose from
word_list = ["apple", "banana", "cherry", "date", "elderberry"]

def scramble_word(word):
    """
    Scramble the letters of a word.
    
    Args:
    word (str): The word to scramble.
    
    Returns:
    str: The scrambled word.
    """
    word_list = list(word)
    random.shuffle(word_list)
    return "".join(word_list)

def play_game():
    """
    Play the word scramble game.
    """
    # Choose a random word from the list
    word = random.choice(word_list)
    
    # Scramble the word
    scrambled_word = scramble_word(word)
    
    # Ask the user to unscramble the word
    print("Unscramble the word: ", scrambled_word)
    answer = input("Enter your answer: ")
    
    # Check if the answer is correct
    if answer.lower() == word:
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {word}.")

def main():
    """
    Main function.
    """
    print("Welcome to the Word Scramble Game!")
    play_again = "yes"
    
    while play_again.lower() == "yes":
        play_game()
        play_again = input("Would you like to play again? (yes/no): ")
    
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
```
