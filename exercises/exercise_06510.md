## Exercise
Write a program to perform basic tuple operations (indexing, slicing, concatenation).

## Reference Solution
```python
def main():
    # Create a tuple
    my_tuple = (1, 2, 3, 4, 5)
    print("Original Tuple:", my_tuple)

    # Indexing
    print("\nIndexing:")
    print("Element at index 0:", my_tuple[0])
    print("Element at index -1 (last element):", my_tuple[-1])

    # Slicing
    print("\nSlicing:")
    print("Elements from index 1 to 3:", my_tuple[1:4])
    print("Elements from start to index 3:", my_tuple[:4])
    print("Elements from index 2 to end:", my_tuple[2:])
    print("All elements:", my_tuple[:])

    # Concatenation
    print("\nConcatenation:")
    another_tuple = (6, 7, 8)
    print("Concatenated Tuple:", my_tuple + another_tuple)

    # Repetition
    print("\nRepetition:")
    repeated_tuple = my_tuple * 2
    print("Repeated Tuple:", repeated_tuple)

if __name__ == "__main__":
    main()
```
