## Exercise
Create a program that simulates the Game of Life, a cellular automaton devised by the British mathematician John Horton Conway.

## Reference Solution
```python
import pygame
import numpy as np

# Window dimensions
WIDTH, HEIGHT = 800, 600

# Cell size
CELL_SIZE = 10

# Colors
ALIVE_COLOR = (255, 255, 255)
DEAD_COLOR = (0, 0, 0)

class GameOfLife:
    def __init__(self, width, height, cell_size):
        self.width = width
        self.height = height
        self.cell_size = cell_size
        self.grid = np.random.choice([0, 1], size=(height // cell_size, width // cell_size))

    def count_neighbors(self, x, y):
        """Count the number of alive neighbors for a cell"""
        count = 0
        for i in range(-1, 2):
            for j in range(-1, 2):
                col = (x + i + self.grid.shape[1]) % self.grid.shape[1]
                row = (y + j + self.grid.shape[0]) % self.grid.shape[0]
                count += self.grid[row, col]
        count -= self.grid[y, x]
        return count

    def next_generation(self):
        """Compute the next generation of the game"""
        new_grid = np.copy(self.grid)
        for y in range(self.grid.shape[0]):
            for x in range(self.grid.shape[1]):
                alive_neighbors = self.count_neighbors(x, y)
                if self.grid[y, x] == 1 and (alive_neighbors < 2 or alive_neighbors > 3):
                    new_grid[y, x] = 0
                elif self.grid[y, x] == 0 and alive_neighbors == 3:
                    new_grid[y, x] = 1
        self.grid = new_grid

def main():
    pygame.init()
    window = pygame.display.set_mode((WIDTH, HEIGHT))
    clock = pygame.time.Clock()

    game = GameOfLife(WIDTH, HEIGHT, CELL_SIZE)

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

        window.fill(DEAD_COLOR)
        for y in range(game.grid.shape[0]):
            for x in range(game.grid.shape[1]):
                if game.grid[y, x] == 1:
                    pygame.draw.rect(window, ALIVE_COLOR, (x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE))

        pygame.display.update()
        game.next_generation()
        clock.tick(60)

    pygame.quit()

if __name__ == "__main__":
    main()
```
