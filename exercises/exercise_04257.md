## Exercise
Develop a program that manages a restaurant menu system. The program should take the menu item, price, and order type as input, and print the order details.

## Reference Solution
```python
class MenuItem:
    def __init__(self, name, price):
        """
        Initialize a menu item with name and price.

        Args:
            name (str): Menu item name.
            price (float): Menu item price.
        """
        self.name = name
        self.price = price

class Order:
    def __init__(self, menu_item, order_type, quantity=1):
        """
        Initialize an order with menu item, order type, and quantity.

        Args:
            menu_item (MenuItem): Menu item object.
            order_type (str): Order type (e.g., "dine-in", "take-out", "delivery").
            quantity (int, optional): Order quantity. Defaults to 1.
        """
        self.menu_item = menu_item
        self.order_type = order_type
        self.quantity = quantity

    def calculate_total(self):
        """
        Calculate the total cost of the order.

        Returns:
            float: Total cost of the order.
        """
        return self.menu_item.price * self.quantity

def main():
    # Create menu items
    menu_items = [
        MenuItem("Burger", 10.99),
        MenuItem("Pizza", 14.99),
        MenuItem("Salad", 8.99)
    ]

    # Print menu
    print("Menu:")
    for i, item in enumerate(menu_items):
        print(f"{i+1}. {item.name}: ${item.price:.2f}")

    # Get user input
    menu_item_index = int(input("Enter the menu item number: ")) - 1
    order_type = input("Enter the order type (dine-in, take-out, delivery): ")
    quantity = int(input("Enter the quantity: "))

    # Create order
    menu_item = menu_items[menu_item_index]
    order = Order(menu_item, order_type, quantity)

    # Print order details
    print("\nOrder Details:")
    print(f"Menu Item: {order.menu_item.name}")
    print(f"Order Type: {order.order_type}")
    print(f"Quantity: {order.quantity}")
    print(f"Total: ${order.calculate_total():.2f}")

if __name__ == "__main__":
    main()
```
