## Exercise
Write a program that reads a binary file and prints out the file's contents.

## Reference Solution
```python
def read_binary_file(file_path):
    """
    Reads a binary file and prints out its contents.

    Args:
        file_path (str): The path to the binary file.
    """
    try:
        with open(file_path, 'rb') as file:
            # Read the file contents
            contents = file.read()
            
            # Print the file contents in hexadecimal format
            print("Binary File Contents (Hexadecimal):")
            for i, byte in enumerate(contents):
                print(f"{byte:02x}", end=' ')
                if (i + 1) % 16 == 0:
                    print()
            print()
            
            # Print the file contents in ASCII format (if possible)
            print("Binary File Contents (ASCII):")
            try:
                ascii_contents = contents.decode('ascii')
                print(ascii_contents)
            except UnicodeDecodeError:
                print("The file contains non-ASCII characters and cannot be printed in ASCII format.")
    except FileNotFoundError:
        print(f"File not found: {file_path}")

# Example usage
if __name__ == "__main__":
    file_path = "example.bin"  # Replace with your binary file path
    read_binary_file(file_path)
```
