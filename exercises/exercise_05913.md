## Exercise
Write a function that takes a string and a character as input and returns the number of occurrences of the character in the string

## Reference Solution
```python
def count_char_occurrences(input_string: str, char: str) -> int:
    """
    Counts the number of occurrences of a character in a string.

    Args:
        input_string (str): The input string to search in.
        char (str): The character to search for.

    Returns:
        int: The number of occurrences of the character in the string.
    """
    if not isinstance(input_string, str) or not isinstance(char, str):
        raise TypeError("Both input_string and char must be strings.")
    if len(char) != 1:
        raise ValueError("Char must be a single character.")

    return input_string.count(char)

# Example usage
def main():
    input_string = "Hello, World!"
    char = "l"
    occurrences = count_char_occurrences(input_string, char)
    print(f"The character '{char}' occurs {occurrences} times in the string '{input_string}'.")

if __name__ == "__main__":
    main()
```
