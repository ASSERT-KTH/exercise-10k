def calculate_median(numbers):
    """
    Calculate the median of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The median of the input list.
    """
    # Sort the list of numbers
    numbers.sort()

    # Get the length of the list
    n = len(numbers)

    # If the length of the list is odd, the median is the middle number
    if n % 2 != 0:
        return numbers[n // 2]
    # If the length of the list is even, the median is the average of the two middle numbers
    else:
        return (numbers[n // 2 - 1] + numbers[n // 2]) / 2

# Example usage:
numbers = [1, 3, 5, 2, 4]
print("Median:", calculate_median(numbers))
