## Exercise
Write a program that takes two numbers as input and returns their division and remainder

## Reference Solution
```python
def divide_and_remainder(dividend: int, divisor: int) -> tuple:
    """
    Divide two numbers and return the quotient and remainder.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        tuple: A tuple containing the quotient and remainder.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")

    quotient = dividend // divisor
    remainder = dividend % divisor

    return quotient, remainder

def main():
    # Get user input
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))

    # Calculate and display the result
    try:
        quotient, remainder = divide_and_remainder(dividend, divisor)
        print(f"{dividend} divided by {divisor} is {quotient} with a remainder of {remainder}")
    except ZeroDivisionError as e:
        print(e)

if __name__ == "__main__":
    main()
```
