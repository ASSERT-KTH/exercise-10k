## Exercise
Write a program to find the minimum value in an array of integers

## Reference Solution
```python
def find_min_value(array):
    """
    Finds the minimum value in an array of integers.

    Args:
        array (list): A list of integers.

    Returns:
        int: The minimum value in the array.

    Raises:
        ValueError: If the input array is empty.
    """
    if not array:
        raise ValueError("Input array is empty")

    return min(array)


def main():
    # Example usage
    array = [5, 2, 8, 12, 3]
    try:
        min_value = find_min_value(array)
        print(f"Minimum value in the array: {min_value}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
