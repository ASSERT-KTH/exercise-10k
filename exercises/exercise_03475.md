## Exercise
Create a program that tells a story with user-inputted characters, location, and objects.

## Reference Solution
```python
def get_user_input():
    """Get user input for characters, location, and objects"""
    characters = input("Enter the names of the characters (separated by commas): ")
    characters = [char.strip() for char in characters.split(",")]
    
    location = input("Enter the location: ")
    
    objects = input("Enter the objects (separated by commas): ")
    objects = [obj.strip() for obj in objects.split(",")]
    
    return characters, location, objects

def create_story(characters, location, objects):
    """Create a story based on user input"""
    story = f"In the land of {location}, there lived {', '.join(characters[:-1])} and {characters[-1]}. "
    story += f"They stumbled upon a hidden treasure chest filled with {', '.join(objects[:-1])} and {objects[-1]}. "
    story += f"As they explored the chest, they discovered that the {objects[0]} was magical and granted them wishes. "
    story += f"{characters[0]} wished for infinite {objects[1]}, {characters[1]} wished for a {objects[2]} that could take them anywhere, "
    story += f"and {characters[-1]} wished for the power to make the {location} a utopia. "
    story += f"And so, they lived happily ever after in their new paradise."
    
    return story

def main():
    characters, location, objects = get_user_input()
    story = create_story(characters, location, objects)
    print("\nHere is your story:")
    print(story)

if __name__ == "__main__":
    main()
```
