## Exercise
Write a program that converts between different units of measurement (e.g. length, temperature), taking the user's input as the value and unit to convert.

def length_conversion():
    """Length conversion menu"""
    print("Length Conversion Menu:")
    print("1. Kilometers to Miles")
    print("2. Miles to Kilometers")
    print("3. Meters to Feet")
    print("4. Feet to Meters")
    print("5. Centimeters to Inches")
    print("6. Inches to Centimeters")

    choice = input("Enter your choice (1-6): ")

    if choice in ["1", "2", "3", "4", "5", "6"]:
        value = float(input("Enter the value to convert: "))

        if choice == "1":
            # Kilometers to Miles
            result = value * 0.621371
            print(f"{value} kilometers is equal to {result} miles")
        elif choice == "2":
            # Miles to Kilometers
            result = value * 1.60934
            print(f"{value} miles is equal to {result} kilometers")
        elif choice == "3":
            # Meters to Feet
            result = value * 3.28084
            print(f"{value} meters is equal to {result} feet")
        elif choice == "4":
            # Feet to Meters
            result = value * 0.3048
            print(f"{value} feet is equal to {result} meters")
        elif choice == "5":
            # Centimeters to Inches
            result = value * 0.393701
            print(f"{value} centimeters is equal to {result} inches")
        elif choice == "6":
            # Inches to Centimeters
            result = value * 2.54
            print(f"{value} inches is equal to {result} centimeters")
    else:
        print("Invalid choice. Please try again.")

def temperature_conversion():
    """Temperature conversion menu"""
    print("Temperature Conversion Menu:")
    print("1. Celsius to Fahrenheit")
    print("2. Fahrenheit to Celsius")
    print("3. Kelvin to Celsius")
    print("4. Celsius to Kelvin")
    print("5. Fahrenheit to Kelvin")
    print("6. Kelvin to Fahrenheit")

    choice = input("Enter your choice (1-6): ")

    if choice in ["1", "2", "3", "4", "5", "6"]:
        value = float(input("Enter the value to convert: "))

        if choice == "1":
            # Celsius to Fahrenheit
            result = (value * 9/5) + 32
            print(f"{value} degrees Celsius is equal to {result} degrees Fahrenheit")
        elif choice == "2":
            # Fahrenheit to Celsius
            result = (value - 32) * 5/9
            print(f"{value} degrees Fahrenheit is equal to {result} degrees Celsius")
        elif choice == "3":
            # Kelvin to Celsius
            result = value - 273.15
            print(f"{value} Kelvin is equal to {result} degrees Celsius")
        elif choice == "4":
            # Celsius to Kelvin
            result = value + 273.15
            print(f"{value} degrees Celsius is equal to {result} Kelvin")
        elif choice == "5":
            # Fahrenheit to Kelvin
            result = (value - 32) * 5/9 + 273.15
            print(f"{value} degrees Fahrenheit is equal to {result} Kelvin")
        elif choice == "6":
            # Kelvin to Fahrenheit
            result = (value - 273.15) * 9/5 + 32
            print(f"{value} Kelvin is equal to {result} degrees Fahrenheit")
    else:
        print("Invalid choice. Please try again.")

def main():
    """Main program loop"""
    while True:
        print("Unit Conversion Program")
        print("1. Length Conversion")
        print("2. Temperature Conversion")
        print("3. Quit")

        choice = input("Enter your choice (1-3): ")

        if choice == "1":
            length_conversion()
        elif choice == "2":
            temperature_conversion()
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
