## Exercise
Given an array of characters, write a program to print the character at a specific index.

## Reference Solution
```python
def print_character_at_index(char_array, index):
    """
    Prints the character at a specific index in a given array of characters.

    Args:
        char_array (list): The array of characters.
        index (int): The index of the character to print.

    Returns:
        None
    """
    try:
        # Check if the index is within the bounds of the array
        if 0 <= index < len(char_array):
            # Print the character at the specified index
            print(f"The character at index {index} is: {char_array[index]}")
        else:
            # Print an error message if the index is out of bounds
            print("Error: Index is out of bounds.")
    except TypeError:
        # Print an error message if the input is not a list or the index is not an integer
        print("Error: Invalid input type.")

def main():
    # Example usage
    char_array = ['a', 'b', 'c', 'd', 'e']
    index = 2
    print_character_at_index(char_array, index)

if __name__ == "__main__":
    main()
```
