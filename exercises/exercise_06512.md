## Exercise
Write a program to perform basic set operations (union, intersection, difference).

## Reference Solution
```python
class SetOperations:
    def __init__(self, set1, set2):
        """
        Initialize the SetOperations class with two sets.

        Args:
            set1 (set): The first set.
            set2 (set): The second set.
        """
        self.set1 = set1
        self.set2 = set2

    def union(self):
        """
        Calculate the union of the two sets.

        Returns:
            set: The union of the two sets.
        """
        return self.set1.union(self.set2)

    def intersection(self):
        """
        Calculate the intersection of the two sets.

        Returns:
            set: The intersection of the two sets.
        """
        return self.set1.intersection(self.set2)

    def difference(self, set1_first=True):
        """
        Calculate the difference between the two sets.

        Args:
            set1_first (bool, optional): Whether to calculate set1 - set2 (True) or set2 - set1 (False). Defaults to True.

        Returns:
            set: The difference between the two sets.
        """
        if set1_first:
            return self.set1.difference(self.set2)
        else:
            return self.set2.difference(self.set1)


def main():
    # Create two sets
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}

    # Create a SetOperations object
    set_ops = SetOperations(set1, set2)

    # Perform set operations
    print("Set 1:", set1)
    print("Set 2:", set2)
    print("Union:", set_ops.union())
    print("Intersection:", set_ops.intersection())
    print("Difference (Set 1 - Set 2):", set_ops.difference())
    print("Difference (Set 2 - Set 1):", set_ops.difference(set1_first=False))


if __name__ == "__main__":
    main()
```
