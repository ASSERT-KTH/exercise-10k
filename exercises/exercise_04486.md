## Exercise
Create a function that allows the user to add, remove, and view items on a to-do list.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        """Initialize an empty to-do list."""
        self.tasks = []

    def add_task(self, task):
        """Add a task to the to-do list."""
        self.tasks.append(task)
        print(f"Task '{task}' added successfully.")

    def remove_task(self, task_number):
        """Remove a task from the to-do list."""
        try:
            task_number = int(task_number)
            if task_number <= 0:
                print("Invalid task number. Please enter a positive integer.")
            elif task_number > len(self.tasks):
                print("Task not found.")
            else:
                task = self.tasks.pop(task_number - 1)
                print(f"Task '{task}' removed successfully.")
        except ValueError:
            print("Invalid task number. Please enter a positive integer.")

    def view_tasks(self):
        """Display all tasks in the to-do list."""
        if not self.tasks:
            print("No tasks available.")
        else:
            print("Your To-Do List:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")


def main():
    """Main function to interact with the to-do list."""
    todo = ToDoList()

    while True:
        print("\nTo-Do List Menu:")
        print("1. Add Task")
        print("2. Remove Task")
        print("3. View Tasks")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            task = input("Enter a task: ")
            todo.add_task(task)
        elif choice == "2":
            if not todo.tasks:
                print("No tasks available to remove.")
            else:
                todo.view_tasks()
                task_number = input("Enter the task number to remove: ")
                todo.remove_task(task_number)
        elif choice == "3":
            todo.view_tasks()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
