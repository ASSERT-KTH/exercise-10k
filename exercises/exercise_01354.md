## Exercise
Write a program to demonstrate the use of built-in functions (e.g., len, range, etc.)

## Reference Solution
```python
def main():
    # len() function
    print("## len() Function")
    my_string = "Hello, World!"
    print(f"Length of '{my_string}': {len(my_string)}")

    # range() function
    print("\n## range() Function")
    for i in range(5):
        print(i)

    # enumerate() function
    print("\n## enumerate() Function")
    fruits = ["Apple", "Banana", "Cherry"]
    for i, fruit in enumerate(fruits):
        print(f"{i}: {fruit}")

    # zip() function
    print("\n## zip() Function")
    names = ["John", "Alice", "Bob"]
    ages = [25, 30, 35]
    for name, age in zip(names, ages):
        print(f"{name}: {age}")

    # all() and any() functions
    print("\n## all() and any() Functions")
    numbers = [1, 2, 3, 4, 5]
    print(f"All numbers greater than 0: {all(num > 0 for num in numbers)}")
    print(f"Any number greater than 3: {any(num > 3 for num in numbers)}")

    # sum() function
    print("\n## sum() Function")
    numbers = [1, 2, 3, 4, 5]
    print(f"Sum of numbers: {sum(numbers)}")

if __name__ == "__main__":
    main()
```
