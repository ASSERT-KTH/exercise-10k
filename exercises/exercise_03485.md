## Exercise
Write a program that takes two numbers as input and returns their remainder.

## Reference Solution
```python
def calculate_remainder(dividend, divisor):
    """
    Calculate the remainder of two numbers.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        int: The remainder of the division.

    Raises:
        ZeroDivisionError: If the divisor is zero.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend % divisor


def main():
    # Get user input
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))

    try:
        # Calculate and print the remainder
        remainder = calculate_remainder(dividend, divisor)
        print(f"The remainder of {dividend} divided by {divisor} is {remainder}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
