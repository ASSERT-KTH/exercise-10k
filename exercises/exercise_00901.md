## Exercise
Create a program that checks the data type of a given variable and prints out whether it is an integer, float, or string.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a given variable.

    Args:
        var (any): The variable to check.

    Returns:
        None
    """
    if isinstance(var, int):
        print(f"The variable is an integer: {var}")
    elif isinstance(var, float):
        print(f"The variable is a float: {var}")
    elif isinstance(var, str):
        print(f"The variable is a string: {var}")
    else:
        print(f"The variable is of type {type(var).__name__}: {var}")


def main():
    # Test cases
    test_cases = [
        10,  # integer
        3.14,  # float
        "Hello, World!",  # string
        [1, 2, 3],  # list
        {"a": 1, "b": 2},  # dictionary
        True,  # boolean
    ]

    for var in test_cases:
        check_data_type(var)


if __name__ == "__main__":
    main()
```
