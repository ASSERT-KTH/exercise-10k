## Exercise
Write a program to create a game leaderboard that displays the top scores and allows users to add new scores.

## Reference Solution
```python
# Game Leaderboard Program

class Leaderboard:
    def __init__(self):
        self.leaderboard = {}

    def add_score(self, name, score):
        """Add a new score to the leaderboard."""
        if name in self.leaderboard:
            if score > self.leaderboard[name]:
                self.leaderboard[name] = score
                print(f"Updated {name}'s score to {score}.")
            else:
                print(f"{name} already has a higher score of {self.leaderboard[name]}.")
        else:
            self.leaderboard[name] = score
            print(f"Added {name} with a score of {score}.")

    def display_leaderboard(self, num_scores=10):
        """Display the top scores on the leaderboard."""
        sorted_leaderboard = sorted(self.leaderboard.items(), key=lambda x: x[1], reverse=True)
        print("\nLeaderboard:")
        for i, (name, score) in enumerate(sorted_leaderboard[:num_scores]):
            print(f"{i+1}. {name} - {score}")

def main():
    leaderboard = Leaderboard()

    while True:
        print("\nOptions:")
        print("1. Add score")
        print("2. Display leaderboard")
        print("3. Quit")
        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter player name: ")
            score = int(input("Enter player score: "))
            leaderboard.add_score(name, score)
        elif choice == "2":
            leaderboard.display_leaderboard()
        elif choice == "3":
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
