## Exercise
Write a program that takes a list and a rotation amount as input, and returns the rotated list.

## Reference Solution
```python
def rotate_list(input_list, rotation_amount):
    """
    Rotate a list by a specified amount.

    Args:
    input_list (list): The list to be rotated.
    rotation_amount (int): The amount of rotation.

    Returns:
    list: The rotated list.
    """
    # Handle case where rotation amount is greater than list length
    rotation_amount = rotation_amount % len(input_list)
    
    # Split the list into two parts at the rotation point
    first_part = input_list[-rotation_amount:]
    second_part = input_list[:-rotation_amount]
    
    # Combine the two parts in reverse order
    rotated_list = first_part + second_part
    
    return rotated_list

# Example usage
def main():
    input_list = [1, 2, 3, 4, 5]
    rotation_amount = 2
    
    rotated_list = rotate_list(input_list, rotation_amount)
    print("Original List:", input_list)
    print("Rotated List:", rotated_list)

if __name__ == "__main__":
    main()
```
