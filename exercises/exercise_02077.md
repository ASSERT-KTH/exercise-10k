## Exercise
Write a program that finds the closest pair of points in a list of 2D points.

## Reference Solution
```python
import math

def distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.
    
    Args:
    point1 (tuple): The first point.
    point2 (tuple): The second point.
    
    Returns:
    float: The Euclidean distance between the two points.
    """
    return math.sqrt((point1[0] - point2[0]) ** 2 + (point1[1] - point2[1]) ** 2)

def closest_pair(points):
    """
    Find the closest pair of points in a list of 2D points.
    
    Args:
    points (list): A list of 2D points.
    
    Returns:
    tuple: The closest pair of points and their distance.
    """
    def closest_pair_recursive(points_sorted_x, points_sorted_y):
        if len(points_sorted_x) <= 3:
            min_distance = float('inf')
            closest_points = None
            for i in range(len(points_sorted_x)):
                for j in range(i + 1, len(points_sorted_x)):
                    dist = distance(points_sorted_x[i], points_sorted_x[j])
                    if dist < min_distance:
                        min_distance = dist
                        closest_points = (points_sorted_x[i], points_sorted_x[j])
            return closest_points, min_distance
        
        mid = len(points_sorted_x) // 2
        mid_point = points_sorted_x[mid]
        
        left_points_sorted_x = points_sorted_x[:mid]
        right_points_sorted_x = points_sorted_x[mid:]
        
        left_points_sorted_y = [point for point in points_sorted_y if point[0] < mid_point[0]]
        right_points_sorted_y = [point for point in points_sorted_y if point[0] >= mid_point[0]]
        
        left_closest_points, left_min_distance = closest_pair_recursive(left_points_sorted_x, left_points_sorted_y)
        right_closest_points, right_min_distance = closest_pair_recursive(right_points_sorted_x, right_points_sorted_y)
        
        min_distance = min(left_min_distance, right_min_distance)
        closest_points = left_closest_points if left_min_distance < right_min_distance else right_closest_points
        
        # Find points in the middle strip
        middle_strip_points = [point for point in points_sorted_y if abs(point[0] - mid_point[0]) < min_distance]
        
        for i in range(len(middle_strip_points)):
            for j in range(i + 1, min(len(middle_strip_points), i + 7)):
                dist = distance(middle_strip_points[i], middle_strip_points[j])
                if dist < min_distance:
                    min_distance = dist
                    closest_points = (middle_strip_points[i], middle_strip_points[j])
        
        return closest_points, min_distance
    
    points_sorted_x = sorted(points)
    points_sorted_y = sorted(points, key=lambda point: point[1])
    
    return closest_pair_recursive(points_sorted_x, points_sorted_y)

# Example usage
points = [(2, 3), (12, 30), (40, 50), (5, 1), (12, 10), (3, 4)]
closest_points, min_distance = closest_pair(points)
print(f"Closest points: {closest_points}")
print(f"Minimum distance: {min_distance}")
```
