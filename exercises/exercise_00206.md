## Exercise
Write a function that simulates a game of Rock, Paper, Scissors and determines the winner.

## Reference Solution
```python
import random

def rock_paper_scissors(player1_choice, player2_choice):
    """
    Simulates a game of Rock, Paper, Scissors and determines the winner.

    Args:
        player1_choice (str): Player 1's choice (Rock, Paper, or Scissors)
        player2_choice (str): Player 2's choice (Rock, Paper, or Scissors)

    Returns:
        str: The winner of the game (Player 1, Player 2, or Tie)
    """
    # Define the winning conditions
    winning_conditions = {
        "Rock": "Scissors",
        "Scissors": "Paper",
        "Paper": "Rock"
    }

    # Check for a tie
    if player1_choice == player2_choice:
        return "Tie"

    # Check if player 1 wins
    if winning_conditions[player1_choice] == player2_choice:
        return "Player 1"

    # Check if player 2 wins
    if winning_conditions[player2_choice] == player1_choice:
        return "Player 2"

def play_game():
    """
    Plays a game of Rock, Paper, Scissors between two players.
    """
    choices = ["Rock", "Paper", "Scissors"]
    player1_choice = input("Enter Player 1's choice (Rock, Paper, or Scissors): ")
    player2_choice = input("Enter Player 2's choice (Rock, Paper, or Scissors): ")

    # Validate player inputs
    while player1_choice not in choices:
        player1_choice = input("Invalid input. Enter Player 1's choice (Rock, Paper, or Scissors): ")
    while player2_choice not in choices:
        player2_choice = input("Invalid input. Enter Player 2's choice (Rock, Paper, or Scissors): ")

    # Determine the winner
    winner = rock_paper_scissors(player1_choice, player2_choice)

    # Print the result
    print(f"Player 1 chose: {player1_choice}")
    print(f"Player 2 chose: {player2_choice}")
    print(f"The winner is: {winner}")

def play_against_computer():
    """
    Plays a game of Rock, Paper, Scissors against the computer.
    """
    choices = ["Rock", "Paper", "Scissors"]
    player_choice = input("Enter your choice (Rock, Paper, or Scissors): ")
    computer_choice = random.choice(choices)

    # Validate player input
    while player_choice not in choices:
        player_choice = input("Invalid input. Enter your choice (Rock, Paper, or Scissors): ")

    # Determine the winner
    winner = rock_paper_scissors(player_choice, computer_choice)

    # Print the result
    print(f"You chose: {player_choice}")
    print(f"The computer chose: {computer_choice}")
    print(f"The winner is: {winner}")

# Example usage
if __name__ == "__main__":
    print("1. Play against another player")
    print("2. Play against the computer")
    choice = input("Enter your choice (1 or 2): ")
    if choice == "1":
        play_game()
    elif choice == "2":
        play_against_computer()
    else:
        print("Invalid input. Please try again.")
```
