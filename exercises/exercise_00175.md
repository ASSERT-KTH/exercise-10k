## Exercise
Write a program to demonstrate explicit type conversion, where the data type of a variable is manually converted.

## Reference Solution
```python
# Explicit Type Conversion in Python

def main():
    # Integer to Float
    print("## Integer to Float")
    integer_value = 10
    print(f"Original Value: {integer_value} (Type: {type(integer_value).__name__})")
    float_value = float(integer_value)
    print(f"Converted Value: {float_value} (Type: {type(float_value).__name__})")

    # Float to Integer
    print("\n## Float to Integer")
    float_value = 10.5
    print(f"Original Value: {float_value} (Type: {type(float_value).__name__})")
    integer_value = int(float_value)
    print(f"Converted Value: {integer_value} (Type: {type(integer_value).__name__})")

    # String to Integer
    print("\n## String to Integer")
    string_value = "10"
    print(f"Original Value: {string_value} (Type: {type(string_value).__name__})")
    integer_value = int(string_value)
    print(f"Converted Value: {integer_value} (Type: {type(integer_value).__name__})")

    # Integer to String
    print("\n## Integer to String")
    integer_value = 10
    print(f"Original Value: {integer_value} (Type: {type(integer_value).__name__})")
    string_value = str(integer_value)
    print(f"Converted Value: {string_value} (Type: {type(string_value).__name__})")

    # String to Float
    print("\n## String to Float")
    string_value = "10.5"
    print(f"Original Value: {string_value} (Type: {type(string_value).__name__})")
    float_value = float(string_value)
    print(f"Converted Value: {float_value} (Type: {type(float_value).__name__})")

    # Float to String
    print("\n## Float to String")
    float_value = 10.5
    print(f"Original Value: {float_value} (Type: {type(float_value).__name__})")
    string_value = str(float_value)
    print(f"Converted Value: {string_value} (Type: {type(string_value).__name__})")

if __name__ == "__main__":
    main()
```
