## Exercise
Write a program that demonstrates good variable naming conventions.

## Reference Solution
```python
# Import necessary libraries
import datetime

def calculate_total_cost(subtotal, tax_rate, discount):
    """
    Calculate the total cost of an order.

    Args:
        subtotal (float): The subtotal of the order.
        tax_rate (float): The tax rate as a decimal.
        discount (float): The discount amount.

    Returns:
        float: The total cost of the order.
    """
    # Calculate the tax amount
    tax_amount = subtotal * tax_rate

    # Calculate the total cost
    total_cost = subtotal + tax_amount - discount

    return total_cost

def main():
    # Define variables with descriptive names
    order_subtotal = 100.0
    tax_rate_percentage = 0.08
    customer_discount = 10.0

    # Calculate the total cost
    order_total_cost = calculate_total_cost(order_subtotal, tax_rate_percentage, customer_discount)

    # Print the result
    print("Order Subtotal: $", order_subtotal)
    print("Tax Rate: ", tax_rate_percentage * 100, "%")
    print("Discount: $", customer_discount)
    print("Total Cost: $", order_total_cost)

    # Get the current date and time
    current_date_time = datetime.datetime.now()
    print("Current Date and Time: ", current_date_time)

if __name__ == "__main__":
    main()
```
