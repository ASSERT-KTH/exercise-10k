## Exercise
Create a function that uses collaboration to work with others and share code

## Reference Solution

import git
import os

def collaborate_on_code(repo_url, username, filename, file_content):
    """
    Clone a Git repository, add a new file, and push the changes to the remote repository.

    Args:
        repo_url (str): URL of the Git repository
        username (str): Your GitHub username
        filename (str): Name of the new file to add
        file_content (str): Content of the new file

    Returns:
        bool: True if the collaboration was successful, False otherwise
    """
    try:
        # Clone the repository
        repo = git.Repo.clone_from(repo_url, 'collaboration')

        # Create a new file
        with open(os.path.join('collaboration', filename), 'w') as f:
            f.write(file_content)

        # Add the new file to the repository
        repo.git.add(filename)

        # Commit the changes
        repo.git.commit('-m', f'Added {filename} by {username}')

        # Push the changes to the remote repository
        repo.git.push('origin', 'master')

        return True

    except git.exc.GitError as e:
        print(f'Error: {e}')
        return False

# Example usage
if __name__ == '__main__':
    repo_url = 'https://github.com/your-username/your-repo.git'
    username = 'your-username'
    filename = 'example.txt'
    file_content = 'Hello, world!'

    collaboration_successful = collaborate_on_code(repo_url, username, filename, file_content)
    print(f'Collaboration successful: {collaboration_successful}')
