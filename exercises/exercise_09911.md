## Exercise
Write a program that encodes and decodes a given text using a simple substitution cipher.

## Reference Solution
```python
def encode(text, shift):
    encoded_text = ""
    for char in text:
        if char.isalpha():
            ascii_offset = 97 if char.islower() else 65
            encoded_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encoded_text += encoded_char
        else:
            encoded_text += char
    return encoded_text

def decode(text, shift):
    decoded_text = ""
    for char in text:
        if char.isalpha():
            ascii_offset = 97 if char.islower() else 65
            decoded_char = chr((ord(char) - ascii_offset - shift) % 26 + ascii_offset)
            decoded_text += decoded_char
        else:
            decoded_text += char
    return decoded_text

# Testing the functions
text = "Hello, World!"
shift = 3
encoded_text = encode(text, shift)
print(f"Encoded text: {encoded_text}")

decoded_text = decode(encoded_text, shift)
print(f"Decoded text: {decoded_text}")
```
