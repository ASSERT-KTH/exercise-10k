## Exercise
Create a program that takes a paragraph of text as input, counts the frequency of each word, and prints the words along with their frequencies in descending order. Ignore case sensitivity, punctuation, and consider 'word' and 'words' as different words.

## Reference Solution
```python
import re
from collections import Counter

def count_word_frequencies(paragraph):
    # Remove punctuation and convert to lower case
    words = re.findall(r'\b\w+\b', paragraph.lower())
    
    # Count word frequencies
    word_freq = Counter(words)
    
    # Print words and their frequencies in descending order
    for word, freq in word_freq.most_common():
        print(f"{word}: {freq}")

# Test the function
paragraph = "This is a test paragraph. It contains multiple sentences, and some of them are quite long. This is just a test."
count_word_frequencies(paragraph)
```

```python
import re
from collections import Counter

def count_word_frequencies(paragraph):
    # Remove punctuation
    words = re.findall(r'\b\w+\b', paragraph)
    
    # Count word frequencies, ignoring case sensitivity
    word_freq = Counter(word.lower() for word in words)
    
    # Create a dictionary to store the original case of each word
    original_case = {}
    for word in words:
        if word.lower() not in original_case:
            original_case[word.lower()] = word
    
    # Print words and their frequencies in descending order
    for word, freq in word_freq.most_common():
        print(f"{original_case[word]}: {freq}")

# Test the function
paragraph = "This is a test paragraph. It contains multiple sentences, and some of them are quite long. This is just a test."
count_word_frequencies(paragraph)
```
