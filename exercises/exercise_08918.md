## Exercise
Write a program to check if a given string is a valid anagram of another string.

## Reference Solution
```python
def is_anagram(str1, str2):
    """
    Checks if str1 is an anagram of str2.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        bool: True if str1 is an anagram of str2, False otherwise.
    """
    # Remove any whitespace and convert to lowercase
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # If the lengths of the strings are not equal, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Sort the characters in each string and compare
    return sorted(str1) == sorted(str2)


def main():
    # Example usage
    str1 = "Listen"
    str2 = "Silent"
    print(f"'{str1}' is an anagram of '{str2}': {is_anagram(str1, str2)}")

    str1 = "Hello"
    str2 = "World"
    print(f"'{str1}' is an anagram of '{str2}': {is_anagram(str1, str2)}")


if __name__ == "__main__":
    main()
```
