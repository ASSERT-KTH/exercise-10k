## Exercise
Write a program that analyzes a given text and provides statistics such as word count, character count, and average word length.

## Reference Solution
```python
import re

def analyze_text(text):
    """
    Analyze the given text and return statistics.

    Args:
        text (str): The text to be analyzed.

    Returns:
        dict: A dictionary containing the word count, character count, and average word length.
    """
    # Remove leading and trailing whitespace
    text = text.strip()

    # Calculate character count
    character_count = len(text)

    # Split text into words
    words = re.findall(r'\b\w+\b', text)

    # Calculate word count
    word_count = len(words)

    # Calculate average word length
    if word_count > 0:
        average_word_length = sum(len(word) for word in words) / word_count
    else:
        average_word_length = 0

    # Return statistics
    return {
        "word_count": word_count,
        "character_count": character_count,
        "average_word_length": average_word_length
    }

def main():
    # Example usage
    text = "This is an example text for analysis."
    statistics = analyze_text(text)

    print("Text Statistics:")
    print("----------------")
    print(f"Word Count: {statistics['word_count']}")
    print(f"Character Count: {statistics['character_count']}")
    print(f"Average Word Length: {statistics['average_word_length']:.2f}")

if __name__ == "__main__":
    main()
```
