## Exercise
Write a program to encrypt a variable's value using a simple cipher.

## Reference Solution

def caesar_cipherEncrypt(text, shift):
    """
    Encrypts the input text using a Caesar cipher with the given shift.

    Args:
        text (str): The input text to be encrypted.
        shift (int): The number of positions to shift the alphabet.

    Returns:
        str: The encrypted text.
    """
    result = ""

    for char in text:
        # Check if character is uppercase
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - 65 + shift) % 26 + 65)

        # Check if character is lowercase
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - 97 + shift) % 26 + 97)

        # If the character is not a letter, add it as is
        else:
            result += char

    return result


def main():
    # Example usage
    variable_value = "Hello, World!"
    shift = 3

    encrypted_value = caesar_cipherEncrypt(variable_value, shift)
    print(f"Original value: {variable_value}")
    print(f"Encrypted value: {encrypted_value}")


if __name__ == "__main__":
    main()
