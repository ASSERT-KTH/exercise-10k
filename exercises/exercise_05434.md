## Exercise
Write a program that manages bank account information. If the user chooses to deposit money, add the amount to the balance. If the user chooses to withdraw money, subtract the amount from the balance.

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_number, balance=0.0):
        """
        Initialize a BankAccount object.

        Args:
            account_number (str): The account number.
            balance (float, optional): The initial balance. Defaults to 0.0.
        """
        self.account_number = account_number
        self.balance = balance

    def deposit(self, amount):
        """
        Deposit money into the account.

        Args:
            amount (float): The amount to deposit.

        Raises:
            ValueError: If the amount is negative.
        """
        if amount < 0:
            raise ValueError("Deposit amount cannot be negative")
        self.balance += amount

    def withdraw(self, amount):
        """
        Withdraw money from the account.

        Args:
            amount (float): The amount to withdraw.

        Raises:
            ValueError: If the amount is negative or exceeds the balance.
        """
        if amount < 0:
            raise ValueError("Withdrawal amount cannot be negative")
        if amount > self.balance:
            raise ValueError("Insufficient funds")
        self.balance -= amount

    def get_balance(self):
        """
        Get the current balance.

        Returns:
            float: The current balance.
        """
        return self.balance


def main():
    # Create a new bank account
    account_number = input("Enter your account number: ")
    account = BankAccount(account_number)

    while True:
        print("\nBank Account Manager")
        print("1. Deposit money")
        print("2. Withdraw money")
        print("3. Check balance")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            try:
                amount = float(input("Enter the deposit amount: "))
                account.deposit(amount)
                print(f"Deposit successful. New balance: {account.get_balance()}")
            except ValueError as e:
                print(e)
        elif choice == "2":
            try:
                amount = float(input("Enter the withdrawal amount: "))
                account.withdraw(amount)
                print(f"Withdrawal successful. New balance: {account.get_balance()}")
            except ValueError as e:
                print(e)
        elif choice == "3":
            print(f"Current balance: {account.get_balance()}")
        elif choice == "4":
            print("Exiting program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
