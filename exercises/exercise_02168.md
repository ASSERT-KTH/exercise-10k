## Exercise
Ask the user for two numbers and print their greatest common divisor.

## Reference Solution
```python
import math

def calculate_gcd(a, b):
    """
    Calculate the greatest common divisor (GCD) of two numbers using the Euclidean algorithm.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The GCD of the two numbers.
    """
    while b != 0:
        a, b = b, a % b
    return abs(a)

def main():
    # Ask the user for two numbers
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))

    # Calculate and print the GCD
    gcd = calculate_gcd(num1, num2)
    print(f"The GCD of {num1} and {num2} is: {gcd}")

    # Alternatively, use the math.gcd function (available in Python 3.5 and later)
    gcd_builtin = math.gcd(num1, num2)
    print(f"The GCD of {num1} and {num2} (using math.gcd) is: {gcd_builtin}")

if __name__ == "__main__":
    main()
```
