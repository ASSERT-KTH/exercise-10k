## Exercise
Write a program that calculates the price after a discount is applied given the original price and the discount percentage.

## Reference Solution
```python
def calculate_discounted_price(original_price, discount_percentage):
    """
    Calculate the price after a discount is applied.

    Args:
        original_price (float): The original price of the item.
        discount_percentage (float): The discount percentage.

    Returns:
        float: The price after the discount is applied.
    """
    if original_price < 0 or discount_percentage < 0:
        raise ValueError("Original price and discount percentage must be non-negative.")

    if discount_percentage > 100:
        raise ValueError("Discount percentage cannot exceed 100%.")

    discount_amount = (original_price / 100) * discount_percentage
    discounted_price = original_price - discount_amount

    return discounted_price


def main():
    # Example usage
    original_price = 100.0
    discount_percentage = 20.0

    try:
        discounted_price = calculate_discounted_price(original_price, discount_percentage)
        print(f"Original Price: ${original_price:.2f}")
        print(f"Discount Percentage: {discount_percentage}%")
        print(f"Discounted Price: ${discounted_price:.2f}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
