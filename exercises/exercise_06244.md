## Exercise
Write a function that takes a dictionary and a key as input and deletes the key-value pair from the dictionary if it exists

## Reference Solution
```python
def delete_key_value_pair(dictionary, key):
    """
    Deletes a key-value pair from a dictionary if the key exists.

    Args:
        dictionary (dict): The dictionary from which to delete the key-value pair.
        key: The key to delete.

    Returns:
        dict: The updated dictionary.
    """
    if key in dictionary:
        del dictionary[key]
    return dictionary

# Example usage:
my_dict = {'a': 1, 'b': 2, 'c': 3}
print("Original dictionary:", my_dict)

updated_dict = delete_key_value_pair(my_dict, 'b')
print("Updated dictionary:", updated_dict)
```

def delete_key_value_pair(dictionary, key):
    """
    Deletes a key-value pair from a dictionary if the key exists and returns a new dictionary.

    Args:
        dictionary (dict): The dictionary from which to delete the key-value pair.
        key: The key to delete.

    Returns:
        dict: A new dictionary with the key-value pair deleted.
    """
    new_dict = dictionary.copy()
    if key in new_dict:
        del new_dict[key]
    return new_dict

# Example usage:
my_dict = {'a': 1, 'b': 2, 'c': 3}
print("Original dictionary:", my_dict)

updated_dict = delete_key_value_pair(my_dict, 'b')
print("Original dictionary (unchanged):", my_dict)
print("Updated dictionary:", updated_dict)
