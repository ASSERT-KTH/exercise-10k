## Exercise
Write a function that takes a string and a width as input and returns the justified text.

## Reference Solution

def justify_text(text, width):
    """
    Justify the given text to the specified width.

    Args:
        text (str): The input text.
        width (int): The desired width of the justified text.

    Returns:
        str: The justified text.
    """
    words = text.split()
    justified_lines = []
    current_line = []

    for word in words:
        # Check if adding the word to the current line would exceed the width
        if sum(len(w) for w in current_line) + len(current_line) + len(word) > width:
            # Justify the current line and add it to the list of justified lines
            justified_lines.append(justify_line(current_line, width))
            current_line = [word]
        else:
            current_line.append(word)

    # Justify the last line and add it to the list of justified lines
    justified_lines.append(justify_last_line(current_line, width))

    return '\n'.join(justified_lines)


def justify_line(words, width):
    """
    Justify a single line of text.

    Args:
        words (list): The list of words in the line.
        width (int): The desired width of the justified text.

    Returns:
        str: The justified line.
    """
    total_width = sum(len(w) for w in words)
    num_spaces = width - total_width
    num_gaps = len(words) - 1

    if num_gaps == 0:
        return words[0] + ' ' * num_spaces

    base_spaces = num_spaces // num_gaps
    extra_spaces = num_spaces % num_gaps

    justified_line = ''
    for i, word in enumerate(words):
        justified_line += word
        if i < len(words) - 1:
            justified_line += ' ' * (base_spaces + (1 if i < extra_spaces else 0))

    return justified_line


def justify_last_line(words, width):
    """
    Justify the last line of text.

    Args:
        words (list): The list of words in the line.
        width (int): The desired width of the justified text.

    Returns:
        str: The justified line.
    """
    justified_line = ' '.join(words)
    return justified_line + ' ' * (width - len(justified_line))


# Example usage
text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
width = 50
print(justify_text(text, width))
