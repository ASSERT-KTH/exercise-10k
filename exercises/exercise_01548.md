## Exercise
Given an array of strings, write a function to replace a specified element with a new value.

## Reference Solution
```python
def replace_element(array, old_value, new_value):
    """
    Replaces a specified element with a new value in an array of strings.

    Args:
        array (list): The input array of strings.
        old_value (str): The element to be replaced.
        new_value (str): The new value to replace the old element.

    Returns:
        list: The modified array with the old element replaced.
    """
    return [new_value if x == old_value else x for x in array]

# Example usage
if __name__ == "__main__":
    array = ["apple", "banana", "cherry", "banana"]
    old_value = "banana"
    new_value = "mango"

    print("Original array:", array)
    print("Old value:", old_value)
    print("New value:", new_value)

    modified_array = replace_element(array, old_value, new_value)
    print("Modified array:", modified_array)
```
