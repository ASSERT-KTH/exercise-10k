## Exercise
Write a program that simplifies a given boolean expression

import re

def simplify_expression(expression):
    """
    Simplify a boolean expression using the rules of Boolean algebra.

    Args:
        expression (str): The boolean expression to simplify.

    Returns:
        str: The simplified boolean expression.
    """

    # Define the rules of Boolean algebra
    rules = {
        r'NOT NOT (.*)': r'\1',  # Double negation
        r'(.*) AND (.*) AND (.*)': r'\1 AND \2',  # Idempotent law
        r'(.*) OR (.*) OR (.*)': r'\1 OR \2',  # Idempotent law
        r'(.*) AND TRUE': r'\1',  # Identity law
        r'(.*) OR FALSE': r'\1',  # Identity law
        r'(.*) AND FALSE': r'FALSE',  # Annihilation law
        r'(.*) OR TRUE': r'TRUE',  # Annihilation law
        r'NOT TRUE': r'FALSE',  # Complement law
        r'NOT FALSE': r'TRUE',  # Complement law
    }

    # Apply the rules of Boolean algebra
    for pattern, replacement in rules.items():
        while re.search(pattern, expression):
            expression = re.sub(pattern, replacement, expression)

    return expression

# Example usage
if __name__ == '__main__':
    expression = "NOT NOT A AND B OR FALSE"
    print(f"Original expression: {expression}")
    simplified_expression = simplify_expression(expression)
    print(f"Simplified expression: {simplified_expression}")
