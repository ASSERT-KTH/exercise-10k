## Exercise
Write a program that takes the initial position, final position, and time as input and returns the velocity.

## Reference Solution
```python
def calculate_velocity(initial_position, final_position, time):
    """
    Calculate the velocity of an object.

    Args:
        initial_position (float): The initial position of the object.
        final_position (float): The final position of the object.
        time (float): The time over which the object moved.

    Returns:
        float: The velocity of the object.
    """
    if time == 0:
        raise ValueError("Time cannot be zero.")

    velocity = (final_position - initial_position) / time
    return velocity


def main():
    # Get user input
    initial_position = float(input("Enter the initial position: "))
    final_position = float(input("Enter the final position: "))
    time = float(input("Enter the time: "))

    # Calculate velocity
    try:
        velocity = calculate_velocity(initial_position, final_position, time)
        print(f"The velocity is: {velocity} units per second")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
