## Exercise
Create a function to validate a password based on the following rules: it must be at least 8 characters long, contain at least one uppercase letter, one lowercase letter, and one digit.

## Reference Solution
```python
import re

def validate_password(password: str) -> bool:
    """
    Validate a password based on the following rules:
    - At least 8 characters long
    - Contains at least one uppercase letter
    - Contains at least one lowercase letter
    - Contains at least one digit

    Args:
        password (str): The password to be validated.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the regular expression pattern
    pattern = r"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).{8,}$"
    
    # Use the regular expression to match the password
    if re.match(pattern, password):
        return True
    else:
        return False

# Example usage
def main():
    passwords = [
        "short",  # Too short
        "longenough",  # No uppercase letter, no digit
        "LongEnough",  # No digit
        "LongEnough1",  # Valid
    ]

    for password in passwords:
        print(f"Password: {password}, Valid: {validate_password(password)}")

if __name__ == "__main__":
    main()
```
