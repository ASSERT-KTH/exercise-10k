## Exercise
Create a program that takes a date as input and prints out whether it is valid or not. A valid date should be in the format MM/DD/YYYY.

## Reference Solution
```python
import re
from datetime import datetime

def is_valid_date(date_string):
    # Regular expression to match the MM/DD/YYYY format
    pattern = re.compile(r'^(\d{2})/(\d{2})/(\d{4})$')

    # Check if the date string matches the pattern
    match = pattern.match(date_string)
    if match:
        month, day, year = map(int, match.groups())
        try:
            # Try to create a datetime object with the given date
            datetime(year, month, day)
            return True
        except ValueError:
            # If the date is invalid (e.g., February 30), this will raise a ValueError
            return False
    else:
        return False

def main():
    date_string = input("Enter a date (MM/DD/YYYY): ")
    if is_valid_date(date_string):
        print(f"'{date_string}' is a valid date.")
    else:
        print(f"'{date_string}' is not a valid date.")

if __name__ == "__main__":
    main()
```
