## Exercise
Write a program that takes a few words as input and generates a short story

## Reference Solution
```python
import random

# Define a dictionary of sentence structures
sentence_structures = {
    "start": ["In a world where {adjective} {noun} roamed free,",
              "Once upon a time, in a land of {adjective} {noun},",
              "It was a {adjective} day in the city of {noun}."],
    "middle": ["{adjective} {noun} appeared out of nowhere,",
               "Suddenly, a {adjective} {noun} came into view,",
               "As {pronoun} walked through the {noun}, {adjective} {noun} caught {pronoun} eye."],
    "end": ["And so, {pronoun} lived happily ever after with {adjective} {noun}.",
            "In the end, {pronoun} realized that {adjective} {noun} was the key to {noun}.",
            "From that day on, {pronoun} never forgot the {adjective} {noun}."]
}

# Define a dictionary of word substitutions
word_substitutions = {
    "adjective": ["happy", "sad", "angry", "excited", "curious"],
    "noun": ["dog", "cat", "house", "car", "tree"],
    "pronoun": ["he", "she", "it", "they", "we"]
}

def generate_short_story(words):
    # Split the input words into a list
    words_list = words.split()

    # Select a random sentence structure for the start, middle, and end of the story
    start_sentence = random.choice(sentence_structures["start"])
    middle_sentence = random.choice(sentence_structures["middle"])
    end_sentence = random.choice(sentence_structures["end"])

    # Substitute the input words into the sentence structures
    for word in words_list:
        if word in word_substitutions["adjective"]:
            start_sentence = start_sentence.replace("{adjective}", word, 1)
            middle_sentence = middle_sentence.replace("{adjective}", word, 1)
            end_sentence = end_sentence.replace("{adjective}", word, 1)
        elif word in word_substitutions["noun"]:
            start_sentence = start_sentence.replace("{noun}", word, 1)
            middle_sentence = middle_sentence.replace("{noun}", word, 1)
            end_sentence = end_sentence.replace("{noun}", word, 1)
        elif word in word_substitutions["pronoun"]:
            start_sentence = start_sentence.replace("{pronoun}", word, 1)
            middle_sentence = middle_sentence.replace("{pronoun}", word, 1)
            end_sentence = end_sentence.replace("{pronoun}", word, 1)

    # Fill in any remaining blanks with random word substitutions
    for sentence in [start_sentence, middle_sentence, end_sentence]:
        while "{adjective}" in sentence:
            sentence = sentence.replace("{adjective}", random.choice(word_substitutions["adjective"]), 1)
        while "{noun}" in sentence:
            sentence = sentence.replace("{noun}", random.choice(word_substitutions["noun"]), 1)
        while "{pronoun}" in sentence:
            sentence = sentence.replace("{pronoun}", random.choice(word_substitutions["pronoun"]), 1)

    # Return the generated short story
    return f"{start_sentence} {middle_sentence} {end_sentence}"

# Example usage
words = input("Enter a few words: ")
print(generate_short_story(words))
```
