## Exercise
Write a program that takes a message as input and encrypts it using a Vigenere cipher.

def vigenere_cipher(message, keyword):
    """
    Encrypts a message using a Vigenere cipher.

    Args:
        message (str): The message to be encrypted.
        keyword (str): The keyword used for encryption.

    Returns:
        str: The encrypted message.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    encrypted_message = ''

    # Convert the message and keyword to lowercase
    message = message.lower()
    keyword = keyword.lower()

    # Initialize the keyword index
    keyword_index = 0

    # Iterate over the characters in the message
    for char in message:
        if char.isalpha():
            # Find the position of the character in the alphabet
            char_position = alphabet.index(char)

            # Find the position of the keyword character in the alphabet
            keyword_char_position = alphabet.index(keyword[keyword_index % len(keyword)])

            # Calculate the new position of the character
            new_position = (char_position + keyword_char_position) % 26

            # Append the encrypted character to the encrypted message
            encrypted_message += alphabet[new_position]

            # Increment the keyword index
            keyword_index += 1
        else:
            # If the character is not a letter, append it as is
            encrypted_message += char

    return encrypted_message

def main():
    message = input("Enter the message to be encrypted: ")
    keyword = input("Enter the keyword: ")

    encrypted_message = vigenere_cipher(message, keyword)

    print("Encrypted message:", encrypted_message)

if __name__ == "__main__":
    main()
