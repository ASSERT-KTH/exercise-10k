## Exercise
Write a program that performs dictionary operations (insert, delete, search, etc.) and prints the result.

## Reference Solution
```python
class DictionaryOperations:
    def __init__(self):
        # Initialize an empty dictionary
        self.dictionary = {}

    def insert(self, key, value):
        """Insert a key-value pair into the dictionary"""
        self.dictionary[key] = value
        print(f"Inserted key-value pair: {key} = {value}")

    def delete(self, key):
        """Delete a key-value pair from the dictionary"""
        if key in self.dictionary:
            del self.dictionary[key]
            print(f"Deleted key-value pair: {key}")
        else:
            print(f"Key '{key}' not found in dictionary")

    def search(self, key):
        """Search for a key in the dictionary and print its value"""
        if key in self.dictionary:
            print(f"Value for key '{key}': {self.dictionary[key]}")
        else:
            print(f"Key '{key}' not found in dictionary")

    def update(self, key, value):
        """Update the value for a given key in the dictionary"""
        if key in self.dictionary:
            self.dictionary[key] = value
            print(f"Updated key-value pair: {key} = {value}")
        else:
            print(f"Key '{key}' not found in dictionary")

    def display(self):
        """Print all key-value pairs in the dictionary"""
        print("Dictionary:")
        for key, value in self.dictionary.items():
            print(f"{key} = {value}")


def main():
    dictionary_ops = DictionaryOperations()

    while True:
        print("\nDictionary Operations Menu:")
        print("1. Insert")
        print("2. Delete")
        print("3. Search")
        print("4. Update")
        print("5. Display")
        print("6. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            key = input("Enter key: ")
            value = input("Enter value: ")
            dictionary_ops.insert(key, value)
        elif choice == "2":
            key = input("Enter key to delete: ")
            dictionary_ops.delete(key)
        elif choice == "3":
            key = input("Enter key to search: ")
            dictionary_ops.search(key)
        elif choice == "4":
            key = input("Enter key to update: ")
            value = input("Enter new value: ")
            dictionary_ops.update(key, value)
        elif choice == "5":
            dictionary_ops.display()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
