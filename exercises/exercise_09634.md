## Exercise
Create a function that uses lazy evaluation to delay the evaluation of an expression until its value is actually needed.

## Reference Solution

class LazyEvaluation:
    """
    A class that uses lazy evaluation to delay the evaluation of an expression until its value is actually needed.
    
    Attributes:
    expression (function): The expression to be evaluated.
    
    Methods:
    get_value: Returns the value of the expression.
    """

    # Initialize the class with an expression
    def __init__(self, expression):
        # Store the expression as a function
        self.expression = expression
        # Initialize a flag to check if the expression has been evaluated
        self.evaluated = False
        # Initialize a variable to store the result of the expression
        self.result = None

    # Define a method to get the value of the expression
    def get_value(self):
        # If the expression has not been evaluated, evaluate it and store the result
        if not self.evaluated:
            # Evaluate the expression
            self.result = self.expression()
            # Set the flag to True to indicate that the expression has been evaluated
            self.evaluated = True
        # Return the result of the expression
        return self.result


# Example usage:
def add(a, b):
    # Simulate a time-consuming operation
    import time
    time.sleep(2)
    # Return the sum of a and b
    return a + b

# Create a LazyEvaluation object with the expression
lazy_add = LazyEvaluation(lambda: add(2, 3))

# The expression is not evaluated yet
print("Before getting the value")
# Get the value of the expression
print(lazy_add.get_value())
# The expression is now evaluated
print("After getting the value")

# Get the value of the expression again
print(lazy_add.get_value())
