## Exercise
Handle null and undefined variables

## Reference Solution
```python
// IF Statement
let variable;
if (variable !== null && variable !== undefined) {
    console.log("Variable is not null or undefined");
} else {
    console.log("Variable is null or undefined");
}

// Optional Chaining
let obj = { a: { b: { c: 'value' } } };
console.log(obj?.a?.b?.c); // Output: "value"
console.log(obj?.a?.b?.d); // Output: undefined

// Nullish Coalescing
let variable2 = null;
console.log(variable2 ?? 'default value'); // Output: "default value"
```
