## Exercise
Create a program that performs arithmetic operations modulo a given number.

## Reference Solution
```python
def add_mod(a, b, mod):
    """
    Adds two numbers modulo a given number.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulo number.

    Returns:
        int: The result of the addition modulo the given number.
    """
    return (a + b) % mod


def subtract_mod(a, b, mod):
    """
    Subtracts two numbers modulo a given number.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulo number.

    Returns:
        int: The result of the subtraction modulo the given number.
    """
    return (a - b) % mod


def multiply_mod(a, b, mod):
    """
    Multiplies two numbers modulo a given number.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulo number.

    Returns:
        int: The result of the multiplication modulo the given number.
    """
    return (a * b) % mod


def divide_mod(a, b, mod):
    """
    Divides two numbers modulo a given number.

    Args:
        a (int): The first number.
        b (int): The second number.
        mod (int): The modulo number.

    Returns:
        int: The result of the division modulo the given number.

    Raises:
        ZeroDivisionError: If the second number is zero.
    """
    if b == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return (a * pow(b, mod-2, mod)) % mod


def main():
    """
    The main function.

    Asks the user for the operation and the numbers, then performs the operation and prints the result.
    """
    print("Arithmetic Operations Modulo a Given Number")
    print("-----------------------------------------")
    print("1. Addition")
    print("2. Subtraction")
    print("3. Multiplication")
    print("4. Division")
    
    choice = input("Enter the number of your chosen operation: ")
    
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))
    mod = int(input("Enter the modulo number: "))
    
    if choice == "1":
        print(f"{num1} + {num2} mod {mod} = {add_mod(num1, num2, mod)}")
    elif choice == "2":
        print(f"{num1} - {num2} mod {mod} = {subtract_mod(num1, num2, mod)}")
    elif choice == "3":
        print(f"{num1} * {num2} mod {mod} = {multiply_mod(num1, num2, mod)}")
    elif choice == "4":
        try:
            print(f"{num1} / {num2} mod {mod} = {divide_mod(num1, num2, mod)}")
        except ZeroDivisionError as e:
            print(e)
    else:
        print("Invalid choice")


if __name__ == "__main__":
    main()
```
