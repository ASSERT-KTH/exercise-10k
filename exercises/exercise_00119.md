## Exercise
Write a program to find the subarray with a given sum.

def subarray_with_given_sum(array, target_sum):
    """
    Finds the subarray with the given sum.

    Args:
        array (list): The input array of integers.
        target_sum (int): The target sum.

    Returns:
        list: The subarray with the given sum, or None if not found.
    """
    # Initialize the window boundaries
    left = 0
    current_sum = 0

    # Iterate over the array
    for right in range(len(array)):
        # Add the current element to the window sum
        current_sum += array[right]

        # Shrink the window from the left if the sum exceeds the target
        while current_sum > target_sum and left <= right:
            current_sum -= array[left]
            left += 1

        # Check if the current window sum matches the target sum
        if current_sum == target_sum:
            # Return the subarray with the given sum
            return array[left:right + 1]

    # Return None if no subarray with the given sum is found
    return None


# Example usage
if __name__ == "__main__":
    array = [1, 4, 20, 3, 10, 5]
    target_sum = 33
    result = subarray_with_given_sum(array, target_sum)

    if result:
        print("Subarray with given sum:", result)
    else:
        print("No subarray with given sum found")
