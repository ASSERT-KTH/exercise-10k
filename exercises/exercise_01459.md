## Exercise
Write a program that takes the start and end times as input and outputs the time elapsed.

## Reference Solution
```python
from datetime import datetime

def calculate_time_elapsed(start_time, end_time):
    """
    Calculate the time elapsed between two given times.

    Args:
        start_time (str): The start time in the format "HH:MM:SS".
        end_time (str): The end time in the format "HH:MM:SS".

    Returns:
        tuple: A tuple containing the time elapsed in hours, minutes, and seconds.
    """
    # Parse the input times
    start_time_obj = datetime.strptime(start_time, "%H:%M:%S")
    end_time_obj = datetime.strptime(end_time, "%H:%M:%S")

    # Calculate the time elapsed
    time_elapsed = end_time_obj - start_time_obj

    # Extract the hours, minutes, and seconds from the time elapsed
    hours = time_elapsed.seconds // 3600
    minutes = (time_elapsed.seconds // 60) % 60
    seconds = time_elapsed.seconds % 60

    return hours, minutes, seconds

def main():
    # Get the start and end times from the user
    start_time = input("Enter the start time (HH:MM:SS): ")
    end_time = input("Enter the end time (HH:MM:SS): ")

    # Calculate the time elapsed
    hours, minutes, seconds = calculate_time_elapsed(start_time, end_time)

    # Print the time elapsed
    print(f"Time elapsed: {hours} hours, {minutes} minutes, {seconds} seconds")

if __name__ == "__main__":
    main()
```
