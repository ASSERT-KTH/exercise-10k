## Exercise
Write a program that takes two fractions as input and returns their sum.

## Reference Solution

def add_fractions(frac1, frac2):
    # Split the fractions into numerators and denominators
    num1, denom1 = map(int, frac1.split('/'))
    num2, denom2 = map(int, frac2.split('/'))

    # Calculate the sum of the fractions
    lcm = (denom1 * denom2) // gcd(denom1, denom2)
    sum_num = (lcm // denom1) * num1 + (lcm // denom2) * num2

    # Simplify the sum
    g = gcd(sum_num, lcm)
    simplified_num = sum_num // g
    simplified_denom = lcm // g

    return f"{simplified_num}/{simplified_denom}"

def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

# Test the function
frac1 = input("Enter the first fraction (e.g., 1/2): ")
frac2 = input("Enter the second fraction (e.g., 1/3): ")

result = add_fractions(frac1, frac2)
print(f"The sum of {frac1} and {frac2} is {result}.")
