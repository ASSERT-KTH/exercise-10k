## Exercise
Write a program that calculates the monthly payment for a mortgage given the principal, rate, and term.

## Reference Solution
```python
def calculate_mortgage_payment(principal, rate, term):
    """
    Calculate the monthly payment for a mortgage.

    Args:
        principal (float): The initial amount borrowed.
        rate (float): The annual interest rate as a decimal.
        term (int): The number of years to repay the mortgage.

    Returns:
        float: The monthly payment amount.
    """
    monthly_rate = rate / 12
    number_of_payments = term * 12
    monthly_payment = principal * monthly_rate * (1 + monthly_rate) ** number_of_payments / ((1 + monthly_rate) ** number_of_payments - 1)
    return monthly_payment

def main():
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    term = int(input("Enter the term of the mortgage (in years): "))

    monthly_payment = calculate_mortgage_payment(principal, rate, term)
    print(f"The monthly payment is: ${monthly_payment:.2f}")

if __name__ == "__main__":
    main()
```
