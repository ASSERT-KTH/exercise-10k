## Exercise
Write a program to calculate the distance between two points given their coordinates.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the distance between two points.

    Args:
        point1 (list): Coordinates of the first point.
        point2 (list): Coordinates of the second point.

    Returns:
        float: Distance between the two points.
    """
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of coordinates")

    squared_diffs = [(a - b) ** 2 for a, b in zip(point1, point2)]
    return math.sqrt(sum(squared_diffs))

def main():
    # Example usage
    point1_2d = [1, 2]
    point2_2d = [4, 6]
    print("Distance between 2D points:", calculate_distance(point1_2d, point2_2d))

    point1_3d = [1, 2, 3]
    point2_3d = [4, 5, 6]
    print("Distance between 3D points:", calculate_distance(point1_3d, point2_3d))

if __name__ == "__main__":
    main()
```
