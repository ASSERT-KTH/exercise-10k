## Exercise
Create a program that takes a word as input from the user, scrambles the letters, and asks the user to unscramble the word.

## Reference Solution
```python
import random

def scramble_word(word):
    """
    Scrambles the letters of the input word.
    
    Args:
        word (str): The word to scramble.
    
    Returns:
        str: The scrambled word.
    """
    word_list = list(word)
    random.shuffle(word_list)
    return ''.join(word_list)

def check_answer(scrambled_word, original_word, user_answer):
    """
    Checks if the user's answer matches the original word.
    
    Args:
        scrambled_word (str): The scrambled word.
        original_word (str): The original word.
        user_answer (str): The user's answer.
    
    Returns:
        bool: True if the user's answer is correct, False otherwise.
    """
    return user_answer.lower() == original_word.lower()

def main():
    # Get the word from the user
    word = input("Enter a word: ")
    
    # Scramble the word
    scrambled_word = scramble_word(word)
    
    # Print the scrambled word
    print(f"Unscramble the letters: {scrambled_word}")
    
    # Get the user's answer
    user_answer = input("Enter your answer: ")
    
    # Check if the user's answer is correct
    if check_answer(scrambled_word, word, user_answer):
        print("Correct! The unscrambled word is indeed", word)
    else:
        print(f"Sorry, that's incorrect. The correct answer is {word}.")

if __name__ == "__main__":
    main()
```
