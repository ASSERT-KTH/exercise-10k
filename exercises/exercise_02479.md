## Exercise
Create a function that searches for a given substring within a larger string.

## Reference Solution
```python
def find_substring(larger_string, substring):
    """
    Searches for a given substring within a larger string.

    Args:
    larger_string (str): The string to search in.
    substring (str): The substring to search for.

    Returns:
    int: The index of the first occurrence of the substring if found, -1 otherwise.
    """
    try:
        # Use the str.find() method to find the substring
        index = larger_string.find(substring)
        return index
    except Exception as e:
        # Handle any exceptions that may occur
        print(f"An error occurred: {e}")
        return -1

# Example usage
def main():
    larger_string = "Hello, World!"
    substring = "World"
    result = find_substring(larger_string, substring)
    
    if result != -1:
        print(f"The substring '{substring}' was found at index {result}.")
    else:
        print(f"The substring '{substring}' was not found.")

if __name__ == "__main__":
    main()
```
