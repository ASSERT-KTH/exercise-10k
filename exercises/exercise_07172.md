## Exercise
Write a program that decrypts a string by shifting each character back by a fixed amount

## Reference Solution
```python
def caesar_decrypt(text: str, shift: int) -> str:
    """
    Decrypts a string by shifting each character back by a fixed amount.

    Args:
    - text (str): The encrypted text to be decrypted.
    - shift (int): The amount of shift to apply to each character.

    Returns:
    - str: The decrypted text.
    """
    decrypted_text = ""
    for char in text:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            decrypted_char = chr((ord(char) - ascii_offset - shift) % 26 + ascii_offset)
            decrypted_text += decrypted_char
        else:
            decrypted_text += char
    return decrypted_text

def main():
    text = input("Enter the encrypted text: ")
    shift = int(input("Enter the shift amount: "))
    decrypted_text = caesar_decrypt(text, shift)
    print("Decrypted text:", decrypted_text)

if __name__ == "__main__":
    main()
```
