## Exercise
Create a program that traverses a binary tree and prints the values of the nodes in a given order (inorder, preorder, or postorder).

## Reference Solution
```python
class Node:
    """Represents a node in a binary tree."""
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None


class BinaryTree:
    """Represents a binary tree with methods to traverse it."""
    def __init__(self):
        self.root = None

    def insert(self, value):
        """Inserts a new node with the given value into the tree."""
        if not self.root:
            self.root = Node(value)
        else:
            self._insert(self.root, value)

    def _insert(self, node, value):
        """Helper method to insert a new node into the tree."""
        if value < node.value:
            if node.left:
                self._insert(node.left, value)
            else:
                node.left = Node(value)
        else:
            if node.right:
                self._insert(node.right, value)
            else:
                node.right = Node(value)

    def inorder(self, node):
        """Prints the values of the nodes in the tree in inorder."""
        if node:
            self.inorder(node.left)
            print(node.value, end=" ")
            self.inorder(node.right)

    def preorder(self, node):
        """Prints the values of the nodes in the tree in preorder."""
        if node:
            print(node.value, end=" ")
            self.preorder(node.left)
            self.preorder(node.right)

    def postorder(self, node):
        """Prints the values of the nodes in the tree in postorder."""
        if node:
            self.postorder(node.left)
            self.postorder(node.right)
            print(node.value, end=" ")


def main():
    tree = BinaryTree()
    tree.insert(5)
    tree.insert(3)
    tree.insert(7)
    tree.insert(2)
    tree.insert(4)
    tree.insert(6)
    tree.insert(8)

    print("Inorder traversal: ")
    tree.inorder(tree.root)
    print("\nPreorder traversal: ")
    tree.preorder(tree.root)
    print("\nPostorder traversal: ")
    tree.postorder(tree.root)


if __name__ == "__main__":
    main()
```
