## Exercise
Write a program that generates the Julia set using a while loop.

## Reference Solution
```python
import numpy as np
import matplotlib.pyplot as plt

def julia_set(xmin,xmax,ymin,ymax,width,height,max_iter,c):
    r1 = np.linspace(xmin, xmax, width)
    r2 = np.linspace(ymin, ymax, height)
    return (r1,r2,np.array([[julia(c, complex(r, i),max_iter) for r in r1] for i in r2]))

def julia(c, z, max_iter):
    for n in range(max_iter):
        if abs(z) > 2:
            return n
        z = z*z + c
    return max_iter

def draw_julia_set(xmin,xmax,ymin,ymax,width,height,max_iter,c):
    d = julia_set(xmin,xmax,ymin,ymax,width,height,max_iter,c)
    plt.imshow(d[2], extent=(xmin, xmax, ymin, ymax))
    plt.show()

# Example usage:
draw_julia_set(-1.5,1.5,-1.5,1.5,1000,1000,256,-0.4+0.6j)
```
