## Exercise
Create a program that generates a random password based on user input. The program should ask for the password length, the character set (e.g., 'letters', 'numbers', 'special characters'), and then print out the generated password.

## Reference Solution
```python
import random
import string

def generate_password(length, char_set):
    """
    Generates a random password of the given length using the specified character set.
    
    Parameters:
    length (int): The length of the password to generate.
    char_set (str): The character set to use (e.g., 'letters', 'numbers', 'special characters', 'all').
    
    Returns:
    str: The generated password.
    """
    if char_set == 'letters':
        characters = string.ascii_letters
    elif char_set == 'numbers':
        characters = string.digits
    elif char_set == 'special characters':
        characters = string.punctuation
    elif char_set == 'all':
        characters = string.ascii_letters + string.digits + string.punctuation
    else:
        raise ValueError("Invalid character set. Please choose from 'letters', 'numbers', 'special characters', or 'all'.")

    password = ''.join(random.choice(characters) for _ in range(length))
    return password

def main():
    print("Password Generator")
    print("-------------------")

    # Get password length from user
    while True:
        try:
            length = int(input("Enter the password length: "))
            if length < 8:
                print("Password length should be at least 8 characters.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a positive integer.")

    # Get character set from user
    print("Choose a character set:")
    print("1. Letters")
    print("2. Numbers")
    print("3. Special characters")
    print("4. All")
    while True:
        choice = input("Enter your choice (1-4): ")
        if choice == '1':
            char_set = 'letters'
            break
        elif choice == '2':
            char_set = 'numbers'
            break
        elif choice == '3':
            char_set = 'special characters'
            break
        elif choice == '4':
            char_set = 'all'
            break
        else:
            print("Invalid choice. Please enter a number between 1 and 4.")

    # Generate and print password
    password = generate_password(length, char_set)
    print(f"Generated password: {password}")

if __name__ == "__main__":
    main()
```
