## Exercise
Write a program that calculates the mean, median, and standard deviation of an array of numbers.

## Reference Solution
```python
import numpy as np

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and standard deviation of an array of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the mean, median, and standard deviation.
    """
    # Calculate the mean
    mean = np.mean(numbers)

    # Calculate the median
    median = np.median(numbers)

    # Calculate the standard deviation
    std_dev = np.std(numbers)

    return mean, median, std_dev


def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    mean, median, std_dev = calculate_statistics(numbers)

    print("Mean:", mean)
    print("Median:", median)
    print("Standard Deviation:", std_dev)


if __name__ == "__main__":
    main()
```
