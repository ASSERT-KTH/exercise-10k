## Exercise
Create a program that uses a dictionary to store student grades. The dictionary should have student names as keys and a list of grades as values. The program should allow users to add students, add grades for a student, and calculate the average grade for a student.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.student_grades = {}

    def add_student(self, name):
        """
        Add a new student to the dictionary.

        Args:
            name (str): The name of the student.
        """
        if name not in self.student_grades:
            self.student_grades[name] = []
            print(f"Student '{name}' added successfully.")
        else:
            print(f"Student '{name}' already exists.")

    def add_grade(self, name, grade):
        """
        Add a grade for a student.

        Args:
            name (str): The name of the student.
            grade (float): The grade to be added.
        """
        if name in self.student_grades:
            self.student_grades[name].append(grade)
            print(f"Grade {grade} added for student '{name}'.")
        else:
            print(f"Student '{name}' does not exist. Please add the student first.")

    def calculate_average(self, name):
        """
        Calculate the average grade for a student.

        Args:
            name (str): The name of the student.

        Returns:
            float: The average grade for the student.
        """
        if name in self.student_grades:
            if self.student_grades[name]:
                return sum(self.student_grades[name]) / len(self.student_grades[name])
            else:
                print(f"No grades found for student '{name}'.")
                return None
        else:
            print(f"Student '{name}' does not exist. Please add the student first.")
            return None


def main():
    student_grades = StudentGrades()

    while True:
        print("\n1. Add Student")
        print("2. Add Grade")
        print("3. Calculate Average")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter the student's name: ")
            student_grades.add_student(name)
        elif choice == "2":
            name = input("Enter the student's name: ")
            grade = float(input("Enter the grade: "))
            student_grades.add_grade(name, grade)
        elif choice == "3":
            name = input("Enter the student's name: ")
            average = student_grades.calculate_average(name)
            if average is not None:
                print(f"The average grade for student '{name}' is {average:.2f}.")
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
