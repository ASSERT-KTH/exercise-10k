## Exercise
Create a program that uses a while loop to check whether two given strings are anagrams of each other.

## Reference Solution
```python
def are_anagrams(str1, str2):
    """
    Check if two strings are anagrams of each other.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        bool: True if the strings are anagrams, False otherwise.
    """
    # Convert both strings to lowercase and remove any whitespace
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # If the strings are not the same length, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Sort the characters in each string
    sorted_str1 = sorted(str1)
    sorted_str2 = sorted(str2)

    # Use a while loop to compare the sorted characters
    i = 0
    while i < len(sorted_str1):
        if sorted_str1[i] != sorted_str2[i]:
            return False
        i += 1

    # If the loop completes without finding any differences, the strings are anagrams
    return True


def main():
    # Test the function
    str1 = "Listen"
    str2 = "Silent"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")

    str1 = "Hello"
    str2 = "World"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")


if __name__ == "__main__":
    main()
```
