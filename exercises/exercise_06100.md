## Exercise
Write a program that uses a for loop to sort a list of integers using the radix sort algorithm.

```python
# Radix Sort Algorithm Implementation

def radix_sort(arr):
    """
    Sorts a list of integers using the Radix Sort algorithm.

    Args:
        arr (list): The list of integers to be sorted.

    Returns:
        list: The sorted list of integers.
    """
    # Find the maximum number to determine the number of digits
    max_num = max(arr)

    # Initialize the exponent to 1 (for the least significant digit)
    exp = 1

    # Loop through each digit position
    while max_num // exp > 0:
        # Initialize the buckets for each digit value (0-9)
        buckets = [[] for _ in range(10)]

        # Distribute the numbers into the buckets based on the current digit
        for num in arr:
            # Calculate the digit value at the current exponent
            digit = (num // exp) % 10
            # Add the number to the corresponding bucket
            buckets[digit].append(num)

        # Collect the numbers from the buckets back into the array
        arr = []
        for bucket in buckets:
            arr.extend(bucket)

        # Move to the next digit position
        exp *= 10

    return arr

# Example usage:
numbers = [170, 45, 75, 90, 802, 24, 2, 66]
sorted_numbers = radix_sort(numbers)
print("Original list:", numbers)
print("Sorted list:", sorted_numbers)
```
