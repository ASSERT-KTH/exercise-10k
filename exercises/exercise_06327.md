## Exercise
Write a program that takes an array as input, reverses it, and returns the reversed array.

## Reference Solution

#include <stdio.h>
#include <stdlib.h>

// Function to reverse an array
void reverse(int a[], int n) {
    int temp;
    int i;
    for (i = 0; i < n / 2; i++) {
        temp = a[i];
        a[i] = a[n - 1 - i];
        a[n - 1 - i] = temp;
    }
}

int main() {
    int n = 5;
    int a[n] = {1, 2, 3, 4, 5};

    printf("Original array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");

    reverse(a, n);

    printf("Reversed array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n");

    return 0;
}
