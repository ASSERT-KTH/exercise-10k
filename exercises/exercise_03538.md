## Exercise
Develop a program that generates Pascal's triangle up to n using a while loop.

## Reference Solution
```python
def generate_pascals_triangle(n):
    """
    Generates Pascal's triangle up to n rows.

    Args:
        n (int): The number of rows to generate.

    Returns:
        list: A list of lists, where each sublist represents a row in the triangle.
    """
    triangle = [[1]]
    row = 1
    while row < n:
        # Initialize the new row with 1
        new_row = [1]
        # Calculate the middle elements of the new row
        for i in range(1, row):
            new_row.append(triangle[row - 1][i - 1] + triangle[row - 1][i])
        # Append the last element (1) to the new row
        new_row.append(1)
        # Add the new row to the triangle
        triangle.append(new_row)
        # Increment the row counter
        row += 1
    return triangle

def print_pascals_triangle(triangle):
    """
    Prints Pascal's triangle in a formatted manner.

    Args:
        triangle (list): A list of lists, where each sublist represents a row in the triangle.
    """
    for row in triangle:
        print(' ' * (len(triangle) - len(row)), end='')
        for num in row:
            print(f'{num} ', end='')
        print()

def main():
    n = int(input("Enter the number of rows: "))
    triangle = generate_pascals_triangle(n)
    print_pascals_triangle(triangle)

if __name__ == "__main__":
    main()
```
