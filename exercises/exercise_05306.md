## Exercise
Write a function that takes two functions as input and returns their composition

## Reference Solution
```python
def compose(f, g):
    """
    Returns the composition of two functions f and g.

    Args:
        f (function): The outer function.
        g (function): The inner function.

    Returns:
        function: The composition of f and g.
    """
    def composition(*args, **kwargs):
        return f(g(*args, **kwargs))
    return composition

# Example usage
def double(x):
    return 2 * x

def add_one(x):
    return x + 1

double_then_add_one = compose(add_one, double)
print(double_then_add_one(5))  # Output: 11
```

```python
def compose(f, g):
    return lambda *args, **kwargs: f(g(*args, **kwargs))

# Example usage
double = lambda x: 2 * x
add_one = lambda x: x + 1

double_then_add_one = compose(add_one, double)
print(double_then_add_one(5))  # Output: 11
```

```python
import functools

def compose(*funcs):
    return functools.reduce(lambda f, g: lambda *args, **kwargs: f(g(*args, **kwargs)), funcs)

# Example usage
double = lambda x: 2 * x
add_one = lambda x: x + 1

double_then_add_one = compose(add_one, double)
print(double_then_add_one(5))  # Output: 11
```
