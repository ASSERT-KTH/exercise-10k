## Exercise
Create a function to unpivot a dictionary from a wide format to a long format (e.g., unpivot a dictionary).

## Reference Solution

def unpivot_dict(wide_dict, id_key):
    """
    Unpivots a dictionary from a wide format to a long format.

    Args:
        wide_dict (dict): The dictionary in wide format.
        id_key (str): The key of the column that will be used as the identifier in the long format.

    Returns:
        dict: A list of dictionaries, where each dictionary represents a row in the long format.
    """

    # Initialize an empty list to store the unpivoted dictionaries
    long_format = []

    # Iterate over the items in the wide dictionary
    for key, value in wide_dict.items():
        # If the key is the id key, we don't need to unpivot this column
        if key == id_key:
            # So, we just add it to the long format as is
            for i, v in enumerate(value):
                # Create a new dictionary for each value in the id column
                long_dict = {id_key: v}
                long_format.append(long_dict)
        else:
            # If the key is not the id key, we need to unpivot this column
            for i, v in enumerate(value):
                # Find the corresponding dictionary in the long format
                long_dict = long_format[i]
                # Add the value to the dictionary with the column name as the key
                long_dict[key] = v

    return long_format


# Example usage:
wide_dict = {
    'id': ['A', 'B', 'C'],
    'column1': [1, 2, 3],
    'column2': [4, 5, 6],
    'column3': [7, 8, 9]
}

unpivoted_dict = unpivot_dict(wide_dict, 'id')
print(unpivoted_dict)
