## Exercise
Write a program that takes two arrays as input, and returns a boolean indicating whether the first array is a superset of the second.

## Reference Solution
```python
#include <iostream>
#include <vector>

bool isSuperset(std::vector<int> arr1, std::vector<int> arr2) {
    for (int i = 0; i < arr2.size(); i++) {
        bool flag = false;
        for (int j = 0; j < arr1.size(); j++) {
            if (arr1[j] == arr2[i]) {
                flag = true;
                break;
            }
        }
        if (!flag) {
            return false;
        }
    }
    return true;
}

int main() {
    std::vector<int> arr1 = {1, 2, 3, 4, 5};
    std::vector<int> arr2 = {1, 2, 3};
    std::cout << std::boolalpha << isSuperset(arr1, arr2) << std::endl;
    return 0;
}
```
