## Exercise
Create a program that simulates the effects of climate change, using conditional statements to determine the climate models

## Reference Solution

import random

class ClimateModel:
    def __init__(self, name, temperature_increase, sea_level_rise, extreme_weather_events):
        self.name = name
        self.temperature_increase = temperature_increase
        self.sea_level_rise = sea_level_rise
        self.extreme_weather_events = extreme_weather_events

    def simulate(self):
        print(f"Simulating {self.name} climate model:")
        print(f"Temperature increase: {self.temperature_increase}Â°C")
        print(f"Sea level rise: {self.sea_level_rise} meters")
        print(f"Extreme weather events: {self.extreme_weather_events} times more frequent")

def determine_climate_model(co2_levels, deforestation_rate, population_growth):
    if co2_levels > 450 and deforestation_rate > 10 and population_growth > 1.5:
        return ClimateModel("Catastrophic", 4, 2, 5)
    elif co2_levels > 350 and deforestation_rate > 5 and population_growth > 1:
        return ClimateModel("Severe", 2.5, 1.5, 3)
    elif co2_levels > 250 and deforestation_rate > 2 and population_growth > 0.5:
        return ClimateModel("Moderate", 1.5, 1, 2)
    else:
        return ClimateModel("Mild", 0.5, 0.5, 1)

def main():
    co2_levels = float(input("Enter CO2 levels (ppm): "))
    deforestation_rate = float(input("Enter deforestation rate (%): "))
    population_growth = float(input("Enter population growth rate (%): "))

    climate_model = determine_climate_model(co2_levels, deforestation_rate, population_growth)
    climate_model.simulate()

if __name__ == "__main__":
    main()
