## Exercise
Write a program that takes user-provided data and serializes it to a file

## Reference Solution
```python
import json

def get_user_data():
    """Get user-provided data."""
    data = {}
    num_fields = int(input("Enter the number of fields: "))

    for i in range(num_fields):
        key = input(f"Enter key {i+1}: ")
        value = input(f"Enter value for {key}: ")
        data[key] = value

    return data

def serialize_data(data, filename):
    """Serialize data to a file."""
    try:
        with open(filename, 'w') as file:
            json.dump(data, file, indent=4)
        print(f"Data serialized to {filename} successfully.")
    except Exception as e:
        print(f"Error serializing data: {e}")

def main():
    data = get_user_data()
    filename = input("Enter the filename to save the data: ")
    if not filename.endswith('.json'):
        filename += '.json'
    serialize_data(data, filename)

if __name__ == "__main__":
    main()
```
