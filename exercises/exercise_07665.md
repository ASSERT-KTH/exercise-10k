## Exercise
Create a function to check if a given number is an automorphic number.

## Reference Solution
```python
def is_automorphic(n: int) -> bool:
    """
    Checks if a given number is an automorphic number.

    Args:
    n (int): The number to check.

    Returns:
    bool: True if the number is automorphic, False otherwise.
    """
    square = n ** 2
    return str(square).endswith(str(n))

# Example usage
def main():
    numbers_to_check = [76, 25, 37]
    for num in numbers_to_check:
        if is_automorphic(num):
            print(f"{num} is an automorphic number.")
        else:
            print(f"{num} is not an automorphic number.")

if __name__ == "__main__":
    main()
```
