## Exercise
Write a program that takes a string as input and returns a list of duplicate words in the string.

## Reference Solution
```python
from collections import Counter
import re

def find_duplicate_words(input_string):
    """
    Find and return duplicate words in a given string.

    Args:
        input_string (str): The input string to search for duplicates.

    Returns:
        list: A list of duplicate words.
    """
    # Convert to lowercase and remove punctuation
    cleaned_string = re.sub(r'[^\w\s]', '', input_string.lower())

    # Split the string into words
    words = cleaned_string.split()

    # Count the occurrences of each word
    word_counts = Counter(words)

    # Filter out words that appear only once
    duplicate_words = [word for word, count in word_counts.items() if count > 1]

    return duplicate_words

def main():
    # Example usage
    input_string = "This is a test string with duplicate words like this and this"
    duplicate_words = find_duplicate_words(input_string)
    print("Duplicate words:", duplicate_words)

if __name__ == "__main__":
    main()
```
