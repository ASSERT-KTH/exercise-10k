## Exercise
Write a program to perform topological sorting on a directed acyclic graph (DAG)

## Reference Solution
```python
from collections import defaultdict, deque

def topological_sorting(graph):
    """
    Performs topological sorting on a given DAG.

    Args:
    graph (dict): Adjacency list representation of the DAG.

    Returns:
    list: Topologically sorted list of vertices.
    """
    # Initialize in-degree dictionary
    in_degree = {vertex: 0 for vertex in graph}
    
    # Calculate in-degrees of all vertices
    for vertex in graph:
        for neighbor in graph[vertex]:
            in_degree[neighbor] += 1
    
    # Initialize queue with vertices having in-degree 0
    queue = deque([vertex for vertex in in_degree if in_degree[vertex] == 0])
    
    # Initialize result list
    result = []
    
    # Perform topological sorting
    while queue:
        vertex = queue.popleft()
        result.append(vertex)
        
        # Decrease in-degrees of neighboring vertices
        for neighbor in graph[vertex]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)
    
    # Check for cycles
    if len(result) != len(graph):
        raise ValueError("Graph contains a cycle")
    
    return result

# Example usage
if __name__ == "__main__":
    # Define a sample DAG
    graph = {
        'A': ['B', 'C'],
        'B': ['D', 'E'],
        'C': ['F'],
        'D': [],
        'E': ['F'],
        'F': []
    }
    
    # Perform topological sorting
    try:
        sorted_vertices = topological_sorting(graph)
        print("Topologically sorted vertices:", sorted_vertices)
    except ValueError as e:
        print(e)
```
