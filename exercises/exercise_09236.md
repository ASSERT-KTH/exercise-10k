## Exercise
Write a program that manipulates a string (concatenation, substring, etc.) and prints the result.

## Reference Solution
```python
def main():
    # Define the original string
    original_string = "Hello, World!"

    # Print the original string
    print("Original String: ", original_string)

    # Concatenate the string with another string
    concatenated_string = original_string + " This is a test."
    print("Concatenated String: ", concatenated_string)

    # Get a substring from the original string
    substring = original_string[0:5]
    print("Substring (0-5): ", substring)

    # Get the length of the original string
    length = len(original_string)
    print("Length of Original String: ", length)

    # Convert the original string to uppercase
    uppercase_string = original_string.upper()
    print("Uppercase String: ", uppercase_string)

    # Convert the original string to lowercase
    lowercase_string = original_string.lower()
    print("Lowercase String: ", lowercase_string)

    # Replace a substring in the original string
    replaced_string = original_string.replace("World", "Earth")
    print("Replaced String: ", replaced_string)

    # Split the original string into a list of words
    words = original_string.split(", ")
    print("Split String: ", words)

if __name__ == "__main__":
    main()
```
