## Exercise
Ask the user for a sentence and print out the number of words, characters, and vowels in it.

## Reference Solution
```python
def count_words(sentence):
    """Returns the number of words in a sentence."""
    return len(sentence.split())

def count_characters(sentence):
    """Returns the number of characters in a sentence."""
    return len(sentence)

def count_vowels(sentence):
    """Returns the number of vowels in a sentence."""
    vowels = 'aeiouAEIOU'
    return sum(1 for char in sentence if char in vowels)

def main():
    # Ask the user for a sentence
    sentence = input("Please enter a sentence: ")

    # Count the number of words, characters, and vowels
    word_count = count_words(sentence)
    character_count = count_characters(sentence)
    vowel_count = count_vowels(sentence)

    # Print the results
    print(f"**Sentence Analysis Results**")
    print(f"* Number of words: {word_count}")
    print(f"* Number of characters: {character_count}")
    print(f"* Number of vowels: {vowel_count}")

if __name__ == "__main__":
    main()
```
