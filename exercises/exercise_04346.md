## Exercise
Write a program that takes two numbers and an operator (+, -, *, /) as input and performs the corresponding operation

## Reference Solution
```python
def calculator(num1, num2, operator):
    """
    Perform arithmetic operation based on the operator.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.
    operator (str): The operator (+, -, *, /).

    Returns:
    float: The result of the operation.
    """
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero")
    else:
        raise ValueError("Invalid operator. Only +, -, *, / are supported")

def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operator = input("Enter the operator (+, -, *, /): ")

    # Perform the operation
    try:
        result = calculator(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ZeroDivisionError, ValueError) as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
