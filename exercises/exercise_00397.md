## Exercise
Write a program that calculates the variance of a set of numbers.

## Reference Solution
```python
def calculate_variance(numbers):
    """
    Calculate the variance of a set of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The variance of the set of numbers.
    """
    # Calculate the mean of the numbers
    mean = sum(numbers) / len(numbers)

    # Calculate the squared differences from the mean
    squared_diffs = [(x - mean) ** 2 for x in numbers]

    # Calculate the variance
    variance = sum(squared_diffs) / len(numbers)

    return variance

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    variance = calculate_variance(numbers)
    print(f"The variance of the numbers is: {variance}")

if __name__ == "__main__":
    main()
```
