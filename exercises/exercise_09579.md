## Exercise
Write a program that takes two numbers as input and performs addition, subtraction, multiplication, and division based on user choice. The program should handle invalid inputs and division by zero.

## Reference Solution
```python
# Importing Modules
from tkinter import *
import tkinter as tk
from tkinter import messagebox

# Defining Functions
def add():
    try:
        num1 = float(num1_entry.get())
        num2 = float(num2_entry.get())
        result = num1 + num2
        result_label.config(text=f"Result: {result:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Invalid input. Please enter numbers only.")

def sub():
    try:
        num1 = float(num1_entry.get())
        num2 = float(num2_entry.get())
        result = num1 - num2
        result_label.config(text=f"Result: {result:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Invalid input. Please enter numbers only.")

def mul():
    try:
        num1 = float(num1_entry.get())
        num2 = float(num2_entry.get())
        result = num1 * num2
        result_label.config(text=f"Result: {result:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Invalid input. Please enter numbers only.")

def div():
    try:
        num1 = float(num1_entry.get())
        num2 = float(num2_entry.get())
        if num2 == 0:
            messagebox.showerror("Error", "Cannot divide by zero.")
        else:
            result = num1 / num2
            result_label.config(text=f"Result: {result:.2f}")
    except ValueError:
        messagebox.showerror("Error", "Invalid input. Please enter numbers only.")

def exit():
    root.destroy()

# Creating the GUI
root = tk.Tk()
root.title("Calculator")

# Creating Frames
frame1 = tk.Frame(root)
frame1.pack(padx=10, pady=10)

frame2 = tk.Frame(root)
frame2.pack(padx=10, pady=10)

frame3 = tk.Frame(root)
frame3.pack(padx=10, pady=10)

# Creating Widgets
num1_label = tk.Label(frame1, text="Number 1:")
num1_label.pack(side=tk.LEFT)

num1_entry = tk.Entry(frame1)
num1_entry.pack(side=tk.LEFT)

num2_label = tk.Label(frame2, text="Number 2:")
num2_label.pack(side=tk.LEFT)

num2_entry = tk.Entry(frame2)
num2_entry.pack(side=tk.LEFT)

add_button = tk.Button(frame3, text="Add", command=add)
add_button.pack(side=tk.LEFT, padx=10)

sub_button = tk.Button(frame3, text="Subtract", command=sub)
sub_button.pack(side=tk.LEFT, padx=10)

mul_button = tk.Button(frame3, text="Multiply", command=mul)
mul_button.pack(side=tk.LEFT, padx=10)

div_button = tk.Button(frame3, text="Divide", command=div)
div_button.pack(side=tk.LEFT, padx=10)

exit_button = tk.Button(frame3, text="Exit", command=exit)
exit_button.pack(side=tk.LEFT, padx=10)

result_label = tk.Label(root, text="Result:")
result_label.pack(padx=10, pady=10)

# Running the GUI
root.mainloop()
```
