## Exercise
Create a program that uses a while loop to test the Collatz conjecture for a given starting number.

## Reference Solution
```python
def collatz_conjecture(starting_number):
    """
    Tests the Collatz Conjecture for a given starting number.

    Args:
        starting_number (int): The starting number for the sequence.

    Returns:
        list: A list of numbers in the sequence.
    """
    if starting_number <= 0:
        raise ValueError("Starting number must be a positive integer")

    sequence = [starting_number]
    while starting_number != 1:
        if starting_number % 2 == 0:
            starting_number = starting_number // 2
        else:
            starting_number = 3 * starting_number + 1
        sequence.append(starting_number)

    return sequence


def main():
    starting_number = int(input("Enter a starting number: "))
    sequence = collatz_conjecture(starting_number)
    print("Sequence:", sequence)
    print("Number of steps:", len(sequence) - 1)


if __name__ == "__main__":
    main()
```
