Create a function that uses the socket module to send and receive UDP packets.

import socket

def send_and_receive_udp_packet(host, port, message):
    """
    Send a UDP packet to a specified host and port, and then wait for a response.

    Args:
    - host (str): The host IP address to send the packet to.
    - port (int): The port number to send the packet to.
    - message (str): The message to be sent in the UDP packet.

    Returns:
    - response (str): The response received from the server.
    """
    # Create a UDP socket
    sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

    # Bind the socket to a local address and port
    # This is optional, but it can be useful if you want to specify a local address and port
    # sock.bind(('localhost', 12345))

    # Send the UDP packet
    sock.sendto(message.encode(), (host, port))

    # Wait for a response from the server
    response, address = sock.recvfrom(1024)

    # Close the socket
    sock.close()

    # Return the response
    return response.decode()

    port = 12345        # Port number to send the packet to
    message = 'Hello, server!'  # Message to be sent in the UDP packet

    response = send_and_receive_udp_packet(host, port, message)
    print(f'Response from server: {response}')

import socket

def create_udp_server(host, port):
    """
    Create a UDP server that listens for incoming UDP packets and responds to them.

    Args:
    - host (str): The host IP address to bind the server to.
    - port (int): The port number to bind the server to.
    """
    # Create a UDP socket
    sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

    # Bind the socket to the specified host and port
    sock.bind((host, port))

    print(f'UDP server listening on {host}:{port}...')

    while True:
        # Wait for an incoming UDP packet
        message, address = sock.recvfrom(1024)
        print(f'Received message from {address}: {message.decode()}')

        # Send a response back to the client
        response = 'Hello, client!'
        sock.sendto(response.encode(), address)

# Example usage:
if __name__ == "__main__":
    host = '127.0.0.1'  # Localhost IP address
    port = 12345        # Port number to bind the server to

    create_udp_server(host, port)
