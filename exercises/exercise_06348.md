```python
def calculate_exponentiation(base, exponent):
    """
    Calculate the result of exponentiation.

    Args:
        base (float): The base number.
        exponent (float): The exponent.

    Returns:
        float: The result of exponentiation.
    """
    return base ** exponent

# Define variables
base1 = 2
exponent1 = 3
base2 = 4
exponent2 = 2

# Perform exponentiation
result1 = calculate_exponentiation(base1, exponent1)
result2 = calculate_exponentiation(base2, exponent2)

# Print the results
print(f"{base1} raised to the power of {exponent1} is {result1}")
```
