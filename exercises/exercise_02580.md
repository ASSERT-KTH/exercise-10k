## Exercise
Write a program that finds the last occurrence of a character in a string using a while loop.

## Reference Solution
```python
def last_occurrence(string, char):
    """
    Finds the last occurrence of a character in a string.

    Args:
        string (str): The input string.
        char (str): The character to find.

    Returns:
        int: The index of the last occurrence of the character. Returns -1 if the character is not found.
    """
    index = -1
    i = len(string) - 1
    while i >= 0:
        if string[i] == char:
            index = i
            break
        i -= 1
    return index

# Example usage
def main():
    string = "Hello, World!"
    char = "l"
    last_index = last_occurrence(string, char)
    if last_index != -1:
        print(f"The last occurrence of '{char}' in '{string}' is at index {last_index}.")
    else:
        print(f"'{char}' is not found in '{string}'.")

if __name__ == "__main__":
    main()
```
