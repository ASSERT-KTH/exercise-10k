## Exercise
Write a program that takes a string as input and returns the count of consonants in the string.

## Reference Solution
```python
def count_consonants(input_string):
    """
    Returns the count of consonants in the input string.

    Args:
        input_string (str): The input string to count consonants from.

    Returns:
        int: The count of consonants in the input string.
    """
    # Define the set of vowels
    vowels = set('aeiouAEIOU')

    # Initialize the consonant count
    consonant_count = 0

    # Iterate over each character in the input string
    for char in input_string:
        # Check if the character is an alphabet letter and not a vowel
        if char.isalpha() and char not in vowels:
            # Increment the consonant count
            consonant_count += 1

    # Return the consonant count
    return consonant_count


# Example usage
def main():
    input_string = input("Enter a string: ")
    consonant_count = count_consonants(input_string)
    print(f"The string '{input_string}' contains {consonant_count} consonants.")


if __name__ == "__main__":
    main()
```
