Unpivot a dictionary from columns to rows.

def unpivot_dict(data_dict):
    """
    Unpivots a dictionary from columns to rows.
    
    :param data_dict: A dictionary where keys are column names and values are lists of column values.
    :return: A list of dictionaries, where each dictionary represents a row with column names as keys.
    """
    # Get the column names
    columns = list(data_dict.keys())
    
    # Find the maximum length of values among all columns to handle uneven columns
    max_length = max(len(values) for values in data_dict.values())
    
    # Initialize an empty list to store the unpivoted rows
    unpivoted_rows = []
    
    # Iterate over the range of the maximum length
    for i in range(max_length):
        row = {}
        # For each column, try to get the value at the current index
        for column in columns:
            # If the list of values for this column is long enough, add the value to the row
            if i < len(data_dict[column]):
                row[column] = data_dict[column][i]
            # If not, you might want to add a default value (e.g., None) depending on your needs
            else:
                row[column] = None
        # Add the row to the list of unpivoted rows
        unpivoted_rows.append(row)
    
    return unpivoted_rows

# Example usage
data_dict = {
    'Name': ['John', 'Alice', 'Bob'],
    'Age': [25, 30, 35, 40],  # Demonstrating uneven columns
    'City': ['New York', 'Los Angeles', 'Chicago']
}

unpivoted = unpivot_dict(data_dict)
for row in unpivoted:
    print(row)
