## Exercise
Create a function that implements a basic graph data structure.

## Reference Solution
```python
class Graph:
    def __init__(self):
        """
        Initializes an empty graph.
        """
        self.vertices = {}

    def add_vertex(self, value):
        """
        Adds a new vertex to the graph.

        Args:
        value (any): The value of the vertex.
        """
        if value not in self.vertices:
            self.vertices[value] = []

    def add_edge(self, vertex1, vertex2):
        """
        Adds a new edge between two vertices.

        Args:
        vertex1 (any): The first vertex.
        vertex2 (any): The second vertex.
        """
        if vertex1 in self.vertices and vertex2 in self.vertices:
            self.vertices[vertex1].append(vertex2)
            self.vertices[vertex2].append(vertex1)  # For undirected graph

    def display_graph(self):
        """
        Displays the graph.
        """
        for vertex in self.vertices:
            print(vertex, "->", self.vertices[vertex])

    def remove_vertex(self, value):
        """
        Removes a vertex from the graph.

        Args:
        value (any): The value of the vertex to be removed.
        """
        if value in self.vertices:
            del self.vertices[value]
            for vertex in self.vertices:
                if value in self.vertices[vertex]:
                    self.vertices[vertex].remove(value)

    def remove_edge(self, vertex1, vertex2):
        """
        Removes an edge between two vertices.

        Args:
        vertex1 (any): The first vertex.
        vertex2 (any): The second vertex.
        """
        if vertex1 in self.vertices and vertex2 in self.vertices:
            if vertex2 in self.vertices[vertex1]:
                self.vertices[vertex1].remove(vertex2)
            if vertex1 in self.vertices[vertex2]:
                self.vertices[vertex2].remove(vertex1)


# Example usage:
if __name__ == "__main__":
    graph = Graph()
    graph.add_vertex(1)
    graph.add_vertex(2)
    graph.add_vertex(3)
    graph.add_edge(1, 2)
    graph.add_edge(2, 3)
    graph.add_edge(3, 1)
    print("Graph:")
    graph.display_graph()
    graph.remove_edge(1, 2)
    print("After removing edge 1-2:")
    graph.display_graph()
    graph.remove_vertex(3)
    print("After removing vertex 3:")
    graph.display_graph()
```
