## Exercise
Write a program that sets an alarm for a specific time. The program should take the current time and the alarm time as input, and print 'Wake Up' if the current time matches the alarm time.

## Reference Solution
```python
from datetime import datetime

def set_alarm(alarm_time):
    """
    Sets an alarm for a specific time.

    Args:
        alarm_time (str): The time at which the alarm should go off, in the format 'HH:MM:SS'.

    Returns:
        None
    """
    while True:
        # Get the current time
        current_time = datetime.now().strftime('%H:%M:%S')

        # Check if the current time matches the alarm time
        if current_time == alarm_time:
            print('Wake Up')
            break

def main():
    # Get the current time
    current_time = datetime.now().strftime('%H:%M:%S')
    print(f'Current Time: {current_time}')

    # Get the alarm time from the user
    alarm_time = input('Enter the alarm time in the format HH:MM:SS: ')

    # Set the alarm
    set_alarm(alarm_time)

if __name__ == '__main__':
    main()
```
