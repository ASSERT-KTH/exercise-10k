## Exercise
Write a program that checks if a given string contains a given substring.

## Reference Solution
```python
def check_substring(main_string, sub_string):
    """
    Checks if the main string contains the sub string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        bool: True if the main string contains the sub string, False otherwise.
    """
    return sub_string in main_string

def main():
    # Example usage
    main_string = input("Enter the main string: ")
    sub_string = input("Enter the substring to search for: ")

    result = check_substring(main_string, sub_string)

    if result:
        print(f"The substring '{sub_string}' is found in the main string.")
    else:
        print(f"The substring '{sub_string}' is not found in the main string.")

if __name__ == "__main__":
    main()
```
