## Exercise
Write a program that deserializes a JSON string to a dictionary.

## Reference Solution
```python
import json

def deserialize_json(json_string):
    """
    Deserializes a JSON string to a dictionary.

    Args:
        json_string (str): The JSON string to deserialize.

    Returns:
        dict: The deserialized dictionary.
    """
    try:
        # Attempt to deserialize the JSON string
        return json.loads(json_string)
    except json.JSONDecodeError as e:
        # If deserialization fails, raise an error
        raise ValueError(f"Failed to deserialize JSON string: {e}")

# Example usage:
if __name__ == "__main__":
    # Define a sample JSON string
    json_string = '{"name": "John", "age": 30, "city": "New York"}'

    # Deserialize the JSON string
    try:
        deserialized_dict = deserialize_json(json_string)
        print("Deserialized Dictionary:")
        print(deserialized_dict)
    except ValueError as e:
        print(f"Error: {e}")
```
