## Exercise
Given two strings, find the minimum number of operations to transform the first string into the second string

## Reference Solution

def levenshtein_distance(str1, str2):
    """
    Calculate the minimum number of operations to transform str1 into str2.
    
    Args:
        str1 (str): The original string.
        str2 (str): The target string.
    
    Returns:
        int: The minimum number of operations.
    """
    m, n = len(str1), len(str2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    
    # Initialize the first row and column
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    
    # Fill in the rest of the matrix
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if str1[i - 1] == str2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1])
    
    return dp[m][n]

# Example usage
str1 = "kitten"
str2 = "sitting"
min_operations = levenshtein_distance(str1, str2)
print(f"Minimum operations to transform '{str1}' into '{str2}': {min_operations}")
