## Exercise
Build a simple chatbot that responds to basic user queries like 'hello', 'how are you?', and 'goodbye'.

## Reference Solution
```python
class ChatBot:
    def __init__(self, name):
        self.name = name
        self.responses = {
            'hello': f'Hello! I\'m {name}.',
            'hi': f'Hello! I\'m {name}.',
            'hey': f'Hello! I\'m {name}.',
            'how are you': 'I\'m doing great, thanks for asking!',
            'what\'s up': 'Not much, just chatting with you!',
            'goodbye': 'See you later!',
            'bye': 'See you later!',
            'default': 'Sorry, I didn\'t understand that.'
        }

    def respond(self, message):
        message = message.lower()
        for keyword in self.responses:
            if keyword in message:
                return self.responses[keyword]
        return self.responses['default']

def main():
    chatbot = ChatBot('Llama')
    print('Welcome to the chatbot! Type "goodbye" to exit.')
    while True:
        user_input = input('You: ')
        if user_input.lower() == 'goodbye':
            print('Chatbot: ', chatbot.respond(user_input))
            break
        print('Chatbot: ', chatbot.respond(user_input))

if __name__ == '__main__':
    main()
```
