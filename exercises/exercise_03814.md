## Exercise
Create a program that uses a while loop to convert a given decimal number to its hexadecimal equivalent.

## Reference Solution
```python
def decimal_to_hexadecimal(decimal_number):
    """
    Converts a decimal number to its hexadecimal equivalent.

    Args:
        decimal_number (int): The decimal number to be converted.

    Returns:
        str: The hexadecimal equivalent of the decimal number.
    """
    hexadecimal = ""
    while decimal_number > 0:
        remainder = decimal_number % 16
        if remainder < 10:
            hexadecimal = str(remainder) + hexadecimal
        else:
            hexadecimal = chr(65 + remainder - 10) + hexadecimal
        decimal_number = decimal_number // 16
    return hexadecimal


def main():
    decimal_number = int(input("Enter a decimal number: "))
    hexadecimal = decimal_to_hexadecimal(decimal_number)
    print(f"The hexadecimal equivalent of {decimal_number} is {hexadecimal}")


if __name__ == "__main__":
    main()
```
