## Exercise
Create a program that implements a cookbook app using a for loop.

## Reference Solution
```python
class Recipe:
    def __init__(self, name, ingredients, instructions):
        self.name = name
        self.ingredients = ingredients
        self.instructions = instructions

class Cookbook:
    def __init__(self):
        self.recipes = []

    def add_recipe(self, name, ingredients, instructions):
        new_recipe = Recipe(name, ingredients, instructions)
        self.recipes.append(new_recipe)

    def delete_recipe(self, name):
        for recipe in self.recipes:
            if recipe.name == name:
                self.recipes.remove(recipe)
                print(f"Recipe '{name}' deleted successfully.")
                return
        print(f"Recipe '{name}' not found.")

    def view_recipes(self):
        if not self.recipes:
            print("No recipes available.")
        else:
            for i, recipe in enumerate(self.recipes, start=1):
                print(f"\nRecipe {i}: {recipe.name}")
                print("Ingredients:")
                for ingredient in recipe.ingredients:
                    print(f"- {ingredient}")
                print("Instructions:")
                for instruction in recipe.instructions:
                    print(f"- {instruction}")

def main():
    cookbook = Cookbook()

    # Add some sample recipes
    cookbook.add_recipe(
        "Grilled Chicken",
        ["chicken breast", "olive oil", "salt", "pepper"],
        ["Preheat grill to medium-high heat.", "Season chicken with salt and pepper.", "Grill chicken for 5-6 minutes per side."]
    )
    cookbook.add_recipe(
        "Pasta with Tomato Sauce",
        ["pasta", "tomato sauce", "parmesan cheese"],
        ["Boil pasta according to package instructions.", "Heat tomato sauce in a pan.", "Combine pasta and sauce, top with parmesan cheese."]
    )

    while True:
        print("\nCookbook App Menu:")
        print("1. View Recipes")
        print("2. Add Recipe")
        print("3. Delete Recipe")
        print("4. Quit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            cookbook.view_recipes()
        elif choice == "2":
            name = input("Enter recipe name: ")
            num_ingredients = int(input("Enter number of ingredients: "))
            ingredients = []
            for i in range(num_ingredients):
                ingredient = input(f"Enter ingredient {i+1}: ")
                ingredients.append(ingredient)
            num_instructions = int(input("Enter number of instructions: "))
            instructions = []
            for i in range(num_instructions):
                instruction = input(f"Enter instruction {i+1}: ")
                instructions.append(instruction)
            cookbook.add_recipe(name, ingredients, instructions)
        elif choice == "3":
            name = input("Enter recipe name to delete: ")
            cookbook.delete_recipe(name)
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
