## Exercise
Write a program that stores and retrieves recipes using conditional statements, allowing users to search for recipes based on ingredients and cooking time.

## Reference Solution
```python
# Define a class for Recipe
class Recipe:
    def __init__(self, name, ingredients, cooking_time):
        self.name = name
        self.ingredients = ingredients
        self.cooking_time = cooking_time

# Create a list to store recipes
recipes = []

# Function to add a recipe
def add_recipe():
    name = input("Enter the name of the recipe: ")
    ingredients = input("Enter the ingredients (separated by comma): ").split(',')
    cooking_time = int(input("Enter the cooking time in minutes: "))
    recipe = Recipe(name, ingredients, cooking_time)
    recipes.append(recipe)
    print("Recipe added successfully.")

# Function to search recipes based on ingredients
def search_by_ingredients():
    ingredient = input("Enter an ingredient to search: ")
    found_recipes = [recipe for recipe in recipes if ingredient.strip() in [i.strip() for i in recipe.ingredients]]
    if found_recipes:
        for recipe in found_recipes:
            print(f"Recipe: {recipe.name}, Ingredients: {', '.join(recipe.ingredients)}, Cooking Time: {recipe.cooking_time} minutes")
    else:
        print("No recipes found with the given ingredient.")

# Function to search recipes based on cooking time
def search_by_cooking_time():
    cooking_time = int(input("Enter the maximum cooking time in minutes: "))
    found_recipes = [recipe for recipe in recipes if recipe.cooking_time <= cooking_time]
    if found_recipes:
        for recipe in found_recipes:
            print(f"Recipe: {recipe.name}, Ingredients: {', '.join(recipe.ingredients)}, Cooking Time: {recipe.cooking_time} minutes")
    else:
        print("No recipes found with the given cooking time.")

# Function to display all recipes
def display_recipes():
    if recipes:
        for recipe in recipes:
            print(f"Recipe: {recipe.name}, Ingredients: {', '.join(recipe.ingredients)}, Cooking Time: {recipe.cooking_time} minutes")
    else:
        print("No recipes available.")

# Main function
def main():
    while True:
        print("\nRecipe Manager")
        print("1. Add a recipe")
        print("2. Search recipes by ingredient")
        print("3. Search recipes by cooking time")
        print("4. Display all recipes")
        print("5. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            add_recipe()
        elif choice == "2":
            search_by_ingredients()
        elif choice == "3":
            search_by_cooking_time()
        elif choice == "4":
            display_recipes()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

# Run the main function
if __name__ == "__main__":
    main()
```
