## Exercise
Ask the user for two numbers and print their division result with remainder.

## Reference Solution
```python
def divide_with_remainder(dividend, divisor):
    """
    Performs division and returns the quotient and remainder.
    
    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.
    
    Returns:
        tuple: A tuple containing the quotient and remainder.
    """
    quotient = dividend // divisor
    remainder = dividend % divisor
    return quotient, remainder

def main():
    # Ask the user for two numbers
    dividend = int(input("Enter the dividend (number being divided): "))
    divisor = int(input("Enter the divisor (number by which we are dividing): "))

    # Check for division by zero
    if divisor == 0:
        print("Error: Division by zero is not allowed.")
        return

    # Perform division and print the result
    quotient, remainder = divide_with_remainder(dividend, divisor)
    print(f"{dividend} divided by {divisor} is {quotient} with a remainder of {remainder}.")

if __name__ == "__main__":
    main()
```
