## Exercise
Create a program to extract a substring from a given string.

## Reference Solution
```python
def extract_substring(input_string, start_index, end_index):
    """
    Extracts a substring from the input string.

    Args:
        input_string (str): The original string.
        start_index (int): The starting index of the substring (inclusive).
        end_index (int): The ending index of the substring (exclusive).

    Returns:
        str: The extracted substring.

    Raises:
        ValueError: If the start or end index is out of range.
    """
    if start_index < 0 or end_index > len(input_string):
        raise ValueError("Start or end index is out of range")

    return input_string[start_index:end_index]

def main():
    input_string = "Hello, World!"
    start_index = 7
    end_index = 12

    try:
        substring = extract_substring(input_string, start_index, end_index)
        print(f"Extracted substring: {substring}")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
