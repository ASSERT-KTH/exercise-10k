def calculate_median(array):
    """
    Calculate the median of an array.
    
    Args:
    array (list): A list of numbers.
    
    Returns:
    float: The median of the array.
    """
    # Sort the array
    array.sort()
    
    # Get the length of the array
    length = len(array)
    
    # If the length is even
    if length % 2 == 0:
        # Calculate the median as the average of the two middle numbers
        median = (array[length // 2 - 1] + array[length // 2]) / 2
    else:
        # Calculate the median as the middle number
        median = array[length // 2]
    
    return median

# Example usage:
numbers = [1, 3, 5, 7, 9]
print(calculate_median(numbers))
