## Exercise
Write a program to analyze a piece of text and calculate statistics such as word count, sentence count, and average word length.

## Reference Solution
```python
import re

def calculate_word_count(text):
    """Calculates the word count of the text."""
    words = text.split()
    return len(words)

def calculate_sentence_count(text):
    """Calculates the sentence count of the text."""
    sentences = re.split(r'[.!?]', text)
    # Remove empty strings from the list
    sentences = [sentence for sentence in sentences if sentence.strip()]
    return len(sentences)

def calculate_average_word_length(text):
    """Calculates the average word length of the text."""
    words = text.split()
    total_length = sum(len(word) for word in words)
    return total_length / len(words)

def analyze_text(text):
    """Analyzes the text and returns statistics."""
    word_count = calculate_word_count(text)
    sentence_count = calculate_sentence_count(text)
    average_word_length = calculate_average_word_length(text)
    return {
        "word_count": word_count,
        "sentence_count": sentence_count,
        "average_word_length": average_word_length
    }

def main():
    text = input("Enter a piece of text: ")
    statistics = analyze_text(text)
    print("Text Statistics:")
    print(f"Word Count: {statistics['word_count']}")
    print(f"Sentence Count: {statistics['sentence_count']}")
    print(f"Average Word Length: {statistics['average_word_length']:.2f}")

if __name__ == "__main__":
    main()
```
