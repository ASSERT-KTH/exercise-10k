## Exercise
Create a program that takes a user's message as input and generates a response.

## Reference Solution
```python
import random

# Dictionary of responses to common greetings and phrases
responses = {
    "hello": ["Hi, how are you?", "Hello! How can I assist you today?", "Hey, what's up?"],
    "hi": ["Hello! How can I help you?", "Hi, how's it going?", "Hey, what's new?"],
    "how are you": ["I'm doing well, thanks for asking!", "I'm good, thanks!", "I'm just a program, I don't have feelings, but thanks for asking!"],
    "what's up": ["Not much, just waiting for your next message!", "Just processing some code, what about you?", "Just hanging out, what's new with you?"],
    "default": ["I didn't quite understand that, could you please rephrase?", "Sorry, I'm not sure what you mean by that.", "I'm not sure how to respond to that, could you please provide more context?"]
}

def generate_response(message):
    """
    Generate a response to the user's message.

    Args:
        message (str): The user's message.

    Returns:
        str: A response to the user's message.
    """
    message = message.lower()

    for key in responses:
        if key in message:
            return random.choice(responses[key])

    return random.choice(responses["default"])

def main():
    print("Welcome to the message response generator!")
    print("Type 'quit' to exit the program.")

    while True:
        message = input("You: ")
        if message.lower() == "quit":
            break
        print("Program: ", generate_response(message))

if __name__ == "__main__":
    main()
```
