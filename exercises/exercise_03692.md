## Exercise
Write a program to calculate the score of a quiz given the number of correct answers and the total number of questions.

## Reference Solution
```python
def calculate_quiz_score(correct_answers, total_questions):
    """
    Calculate the score of a quiz.

    Args:
        correct_answers (int): The number of correct answers.
        total_questions (int): The total number of questions.

    Returns:
        float: The quiz score as a percentage.
    """
    if total_questions <= 0:
        raise ValueError("Total questions must be greater than 0")

    score = (correct_answers / total_questions) * 100
    return score

def main():
    # Get user input
    correct_answers = int(input("Enter the number of correct answers: "))
    total_questions = int(input("Enter the total number of questions: "))

    # Calculate and display the quiz score
    try:
        score = calculate_quiz_score(correct_answers, total_questions)
        print(f"Your quiz score is: {score:.2f}%")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
