## Exercise
Write a function that takes a function as input and returns a curried version of it

## Reference Solution
```python
def curry(func):
    """
    Returns a curried version of the input function.
    
    Args:
        func: The input function to be curried.
    
    Returns:
        A curried version of the input function.
    """
    def curried(*args, **kwargs):
        if len(args) + len(kwargs) >= func.__code__.co_argcount:
            return func(*args, **kwargs)
        return lambda *x, **y: curried(*(args + x), **dict(kwargs, **y))
    return curried

# Example usage:

def add(a, b, c):
    return a + b + c

curried_add = curry(add)

print(curried_add(1)(2)(3))  # Output: 6
print(curried_add(1, 2)(3))   # Output: 6
print(curried_add(1)(2, 3))   # Output: 6
print(curried_add(1, 2, 3))  # Output: 6
```
