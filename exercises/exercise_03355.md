    int n = 10; // grid size
    int grid[n][n]; // declare grid

    // initialize grid with random values
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            grid[i][j] = rand() % 2;
        }
    }

    // game loop
    int gen = 0;
    while (gen < p) {

        grid = next_gen(grid, n);
