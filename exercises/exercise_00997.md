## Exercise
Write a program that converts time from one unit to another (e.g., seconds to minutes, minutes to hours). The program uses if-else statements to determine the conversion factor.

## Reference Solution
```python
// Importing Libraries
#include <iostream>
using namespace std;

// Function Declaration
void convert();

// Main Function
int main() {
    int choice;
    cout << "Time Converter Program" << endl;
    cout << "1. Seconds to Minutes" << endl;
    cout << "2. Minutes to Hours" << endl;
    cout << "3. Hours to Days" << endl;
    cout << "Enter your choice: ";
    cin >> choice;
    convert();
    return 0;
}

// convert Function
void convert() {
    int value;
    cout << "Enter a value: ";
    cin >> value;
    if (value < 0) {
        cout << "Invalid input. Please enter a non-negative value." << endl;
    } else {
        int choice;
        cout << "Enter your choice: ";
        cin >> choice;
        if (choice == 1) {
            cout << value << " seconds is equal to " << value / 60 << " minutes." << endl;
        } else if (choice == 2) {
            cout << value << " minutes is equal to " << value / 60 << " hours." << endl;
        } else if (choice == 3) {
            cout << value << " hours is equal to " << value / 24 << " days." << endl;
        } else {
            cout << "Invalid choice. Please try again." << endl;
        }
    }
}
```
