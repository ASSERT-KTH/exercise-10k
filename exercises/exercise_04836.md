## Exercise
Write a program to demonstrate the lifetime of a variable, including when it is created and when it is destroyed.

## Reference Solution
```python
class Variable:
    def __init__(self, name):
        self.name = name
        print(f"Variable {self.name} created")

    def __del__(self):
        print(f"Variable {self.name} destroyed")


def create_variable(name):
    var = Variable(name)
    print(f"Variable {var.name} is in scope")
    return var


def main():
    var1 = create_variable("var1")
    print("Exiting create_variable function")

    print("\nCreating var2")
    var2 = Variable("var2")
    print("Variable var2 is in scope")

    # delete var2
    del var2
    print("Variable var2 deleted")

    # delete var1
    del var1
    print("Variable var1 deleted")


if __name__ == "__main__":
    main()
```
