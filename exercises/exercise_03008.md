## Exercise
Write a program to replace a substring within a given string with another string.

## Reference Solution
```python
def replace_substring(original_string, substring_to_replace, replacement_string):
    """
    Replaces a substring within a given string with another string.

    Args:
        original_string (str): The original string.
        substring_to_replace (str): The substring to be replaced.
        replacement_string (str): The replacement string.

    Returns:
        str: The modified string with the substring replaced.
    """
    return original_string.replace(substring_to_replace, replacement_string)

def main():
    # Example usage
    original_string = "Hello, world! Hello again!"
    substring_to_replace = "Hello"
    replacement_string = "Hi"

    modified_string = replace_substring(original_string, substring_to_replace, replacement_string)
    print("Original String:", original_string)
    print("Substring to Replace:", substring_to_replace)
    print("Replacement String:", replacement_string)
    print("Modified String:", modified_string)

if __name__ == "__main__":
    main()
```
