## Exercise
Write a program that counts the number of vowels in a string using a for loop.

## Reference Solution
```python
def count_vowels(input_string):
    """
    Counts the number of vowels in a given string.

    Args:
        input_string (str): The input string to count vowels from.

    Returns:
        int: The total count of vowels in the input string.
    """
    vowels = 'aeiou'
    count = 0

    # Convert the input string to lowercase for case-insensitive comparison
    input_string = input_string.lower()

    # Iterate over each character in the input string
    for char in input_string:
        # Check if the character is a vowel
        if char in vowels:
            # Increment the vowel count if it is
            count += 1

    return count


# Example usage:
def main():
    input_str = "Hello World"
    vowel_count = count_vowels(input_str)
    print(f"The string '{input_str}' contains {vowel_count} vowels.")


if __name__ == "__main__":
    main()
```
