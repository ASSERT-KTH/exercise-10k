## Exercise
Create a program that asks the user a series of survey questions and prints out the results.

## Reference Solution
```python
# survey_program.py

class Survey:
    def __init__(self):
        self.questions = {
            "What is your age?": "",
            "What is your occupation?": "",
            "How satisfied are you with our service? (Scale: 1-5)": "",
            "Would you recommend our service to a friend? (Yes/No)": "",
            "What could we improve?": ""
        }

    def ask_questions(self):
        for question, answer in self.questions.items():
            print(question)
            self.questions[question] = input("Your answer: ")

    def print_results(self):
        print("\nSurvey Results:")
        for question, answer in self.questions.items():
            print(f"{question}: {answer}")


def main():
    survey = Survey()
    print("Welcome to our survey! Please answer the following questions.")
    survey.ask_questions()
    survey.print_results()


if __name__ == "__main__":
    main()
```
