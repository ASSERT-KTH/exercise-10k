import re

def evaluate_expression(expression):
    # Define allowed operators and their corresponding functions
    operators = {
        'and': lambda x, y: x and y,
        'or': lambda x, y: x or y,
        'not': lambda x: not x,
        '==': lambda x, y: x == y,
        '!=': lambda x, y: x != y,
    }

    # Define allowed operands
    operands = {
        'True': True,
        'False': False,
    }

    # Parse the expression
    match = re.match(r'(\w+)\s*(==|!=|and|or)?\s*(\w+)?', expression)
    if match:
        operand1 = match.group(1)
        operator = match.group(2)
        operand2 = match.group(3)

        # Check if the operands and operator are allowed
        if operand1 in operands and (operator in operators or operator is None) and (operand2 is None or operand2 in operands):
            # Evaluate the expression
            if operator is None:
                return operands[operand1]
            elif operator == 'not':
                return operators[operator](operands[operand1])
            else:
                return operators[operator](operands[operand1], operands[operand2])
        else:
            return "Invalid expression"
    else:
        return "Invalid expression"

# Example usage:
print(evaluate_expression("True and False"))  # Output: False
print(evaluate_expression("True or False"))  # Output: True
print(evaluate_expression("not True"))  # Output: False
print(evaluate_expression("True == True"))  # Output: True
print(evaluate_expression("True != True"))  # Output: False
