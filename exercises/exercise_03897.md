## Exercise
Develop a program that takes an image file as input and applies basic image processing operations like resizing and cropping.

## Reference Solution
```python
from PIL import Image
import os

def resize_image(image_path, output_path, size):
    """
    Resize an image to a specified size.

    Args:
        image_path (str): Path to the input image file.
        output_path (str): Path to the output image file.
        size (tuple): New size of the image (width, height).
    """
    image = Image.open(image_path)
    image = image.resize(size)
    image.save(output_path)

def crop_image(image_path, output_path, box):
    """
    Crop an image to a specified region.

    Args:
        image_path (str): Path to the input image file.
        output_path (str): Path to the output image file.
        box (tuple): Region to crop (left, upper, right, lower).
    """
    image = Image.open(image_path)
    image = image.crop(box)
    image.save(output_path)

def main():
    image_path = input("Enter the path to the input image file: ")
    output_path = input("Enter the path to the output image file: ")

    print("Choose an operation:")
    print("1. Resize")
    print("2. Crop")
    operation = input("Enter your choice (1/2): ")

    if operation == "1":
        width = int(input("Enter the new width: "))
        height = int(input("Enter the new height: "))
        resize_image(image_path, output_path, (width, height))
    elif operation == "2":
        left = int(input("Enter the left coordinate: "))
        upper = int(input("Enter the upper coordinate: "))
        right = int(input("Enter the right coordinate: "))
        lower = int(input("Enter the lower coordinate: "))
        crop_image(image_path, output_path, (left, upper, right, lower))
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
