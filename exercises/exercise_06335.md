## Exercise
Write a program that takes an array and a number of positions as input, rotates the array by the specified number of positions, and returns the rotated array.

## Reference Solution
```python
#include <stdio.h>

int* rotate(int* nums, int numsSize, int k, int* returnSize);

int main() {
    int nums[] = {1, 2, 3, 4, 5, 6, 7};
    int numsSize = sizeof(nums) / sizeof(nums[0]);
    int k = 3;

    int* rotated = rotate(nums, numsSize, k, &numsSize);

    printf("Rotated array: ");
    for (int i = 0; i < numsSize; i++) {
        printf("%d ", rotated[i]);
    }
    printf("\n");

    return 0;
}

int* rotate(int* nums, int numsSize, int k, int* returnSize) {
    int* result = (int*)malloc(numsSize * sizeof(int));
    *returnSize = numsSize;

    k = k % numsSize;

    for (int i = 0; i < k; i++) {
        result[i] = nums[numsSize - k + i];
    }

    for (int i = k; i < numsSize; i++) {
        result[i] = nums[i - k];
    }

    return result;
}
```
