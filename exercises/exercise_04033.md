## Exercise
Implement a function to find all pairs of elements in an array that sum to a given target

## Reference Solution

def find_pairs(nums, target):
    """
    Find all pairs of elements in an array that sum to a given target.

    Args:
        nums (list): A list of integers.
        target (int): The target sum.

    Returns:
        list: A list of tuples, where each tuple contains a pair of elements that sum to the target.
    """
    nums.sort()  # Sort the array to use the two-pointer technique
    pairs = []  # Initialize an empty list to store the pairs
    left, right = 0, len(nums) - 1  # Initialize two pointers

    while left < right:
        current_sum = nums[left] + nums[right]
        if current_sum == target:
            pairs.append((nums[left], nums[right]))  # Add the pair to the list
            left += 1  # Move the left pointer to the right
            right -= 1  # Move the right pointer to the left
        elif current_sum < target:
            left += 1  # Move the left pointer to the right to increase the sum
        else:
            right -= 1  # Move the right pointer to the left to decrease the sum

    return pairs

# Example usage:
nums = [1, 2, 3, 4, 5, 6]
target = 7
print(find_pairs(nums, target))  # Output: [(1, 6), (2, 5), (3, 4)]
