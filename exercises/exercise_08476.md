## Exercise
Write a program to calculate the compound interest given the principal amount, interest rate, and number of years

## Reference Solution
```python
def calculate_compound_interest(principal, interest_rate, years, compounding_frequency=1):
    """
    Calculate the compound interest.

    Args:
    - principal (float): The principal amount.
    - interest_rate (float): The annual interest rate (in decimal form).
    - years (int): The number of years.
    - compounding_frequency (int): The number of times that interest is compounded per year. Defaults to 1.

    Returns:
    - float: The amount of money accumulated after the specified number of years, including interest.
    """
    amount = principal * (1 + interest_rate / compounding_frequency) ** (compounding_frequency * years)
    return amount

def main():
    principal = float(input("Enter the principal amount: "))
    interest_rate = float(input("Enter the annual interest rate (in percent): ")) / 100
    years = int(input("Enter the number of years: "))
    compounding_frequency = int(input("Enter the compounding frequency (e.g., 1 for annually, 12 for monthly): "))

    amount = calculate_compound_interest(principal, interest_rate, years, compounding_frequency)
    interest = amount - principal

    print(f"Principal Amount: ${principal:.2f}")
    print(f"Interest Rate: {interest_rate * 100:.2f}%")
    print(f"Number of Years: {years}")
    print(f"Compounding Frequency: {compounding_frequency} times per year")
    print(f"Total Amount: ${amount:.2f}")
    print(f"Interest Earned: ${interest:.2f}")

if __name__ == "__main__":
    main()
```
