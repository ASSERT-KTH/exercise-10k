## Exercise
Create a program that takes the coefficients of a quadratic equation as input and returns the solutions.

## Reference Solution
```python
import math
import cmath

def solve_quadratic(a, b, c):
    """
    Solves a quadratic equation of the form ax^2 + bx + c = 0.

    Args:
        a (float): The coefficient of the squared term.
        b (float): The coefficient of the linear term.
        c (float): The constant term.

    Returns:
        tuple: The solutions to the quadratic equation.
    """
    # Calculate the discriminant
    d = b**2 - 4*a*c

    # Check if the discriminant is positive, negative, or zero
    if d > 0:
        # Two distinct real solutions
        solution1 = (-b + math.sqrt(d)) / (2*a)
        solution2 = (-b - math.sqrt(d)) / (2*a)
        return solution1, solution2
    elif d == 0:
        # One repeated real solution
        solution = -b / (2*a)
        return solution, solution
    else:
        # Two complex solutions
        solution1 = (-b + cmath.sqrt(d)) / (2*a)
        solution2 = (-b - cmath.sqrt(d)) / (2*a)
        return solution1, solution2

def main():
    # Get the coefficients from the user
    a = float(input("Enter the coefficient of the squared term (a): "))
    b = float(input("Enter the coefficient of the linear term (b): "))
    c = float(input("Enter the constant term (c): "))

    # Solve the quadratic equation
    solution1, solution2 = solve_quadratic(a, b, c)

    # Print the solutions
    if solution1 == solution2:
        print(f"The solution is x = {solution1}")
    elif solution1.imag == 0 and solution2.imag == 0:
        print(f"The solutions are x = {solution1} and x = {solution2}")
    else:
        print(f"The solutions are x = {solution1} and x = {solution2}")

if __name__ == "__main__":
    main()
```
