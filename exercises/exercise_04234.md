## Exercise
Write a program to convert a given amount of money from one currency to another.

## Reference Solution
```python
# Import required libraries
import requests

def get_exchange_rate(base_currency, target_currency):
    """
    Get the current exchange rate for the given currencies.
    
    Args:
    base_currency (str): The base currency.
    target_currency (str): The target currency.
    
    Returns:
    float: The current exchange rate.
    """
    url = f"https://api.exchangerate-api.com/v4/latest/{base_currency}"
    response = requests.get(url)
    data = response.json()
    exchange_rate = data["rates"][target_currency]
    return exchange_rate

def convert_currency(amount, base_currency, target_currency):
    """
    Convert the given amount of money from one currency to another.
    
    Args:
    amount (float): The amount of money to convert.
    base_currency (str): The base currency.
    target_currency (str): The target currency.
    
    Returns:
    float: The converted amount.
    """
    exchange_rate = get_exchange_rate(base_currency, target_currency)
    converted_amount = amount * exchange_rate
    return converted_amount

def main():
    # Get user input
    amount = float(input("Enter the amount of money to convert: "))
    base_currency = input("Enter the base currency (e.g. USD, EUR, JPY): ").upper()
    target_currency = input("Enter the target currency (e.g. USD, EUR, JPY): ").upper()
    
    # Convert currency
    converted_amount = convert_currency(amount, base_currency, target_currency)
    
    # Print the result
    print(f"{amount} {base_currency} is equal to {converted_amount:.2f} {target_currency}")

if __name__ == "__main__":
    main()
```
