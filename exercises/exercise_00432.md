## Exercise
Write a program that plays Rock, Paper, Scissors with the user.

## Reference Solution
```python
import random

def get_computer_choice():
    """Returns the computer's random choice"""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def determine_winner(user_choice, computer_choice):
    """Determines the winner based on the user's and computer's choices"""
    if user_choice == computer_choice:
        return 'tie'
    if (user_choice == 'rock' and computer_choice == 'scissors') or \
       (user_choice == 'scissors' and computer_choice == 'paper') or \
       (user_choice == 'paper' and computer_choice == 'rock'):
        return 'user'
    return 'computer'

def play_round():
    """Plays a single round of Rock, Paper, Scissors"""
    user_choice = input("Enter your choice (rock, paper, scissors): ").lower()
    while user_choice not in ['rock', 'paper', 'scissors']:
        user_choice = input("Invalid input. Please enter rock, paper, or scissors: ").lower()
    computer_choice = get_computer_choice()
    print(f"\nComputer chose: {computer_choice}")
    winner = determine_winner(user_choice, computer_choice)
    if winner == 'tie':
        print(f"Both players selected {user_choice}. It's a tie!")
    elif winner == 'user':
        print(f"{user_choice.capitalize()} beats {computer_choice}! You win!")
    else:
        print(f"{computer_choice.capitalize()} beats {user_choice}! Computer wins!")

def play_game():
    """Plays multiple rounds of Rock, Paper, Scissors"""
    rounds = int(input("How many rounds would you like to play? "))
    user_score = 0
    computer_score = 0
    for i in range(rounds):
        print(f"\nRound {i+1}:")
        play_round()
        winner = determine_winner(input("Enter your choice (rock, paper, scissors): ").lower(), get_computer_choice())
        if winner == 'user':
            user_score += 1
        elif winner == 'computer':
            computer_score += 1
    print(f"\nFinal Score - You: {user_score}, Computer: {computer_score}")
    if user_score > computer_score:
        print("You win the game! Congratulations!")
    elif user_score < computer_score:
        print("Computer wins the game! Better luck next time!")
    else:
        print("It's a tie game!")

def main():
    print("Welcome to Rock, Paper, Scissors!")
    while True:
        print("\n1. Play a single round")
        print("2. Play multiple rounds")
        print("3. Quit")
        choice = input("Enter your choice: ")
        if choice == '1':
            play_round()
        elif choice == '2':
            play_game()
        elif choice == '3':
            print("Goodbye!")
            break
        else:
            print("Invalid input. Please try again.")

if __name__ == "__main__":
    main()
```
