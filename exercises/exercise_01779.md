## Exercise
Write a function to check if a given graph is connected.

from typing import Dict, List

def is_graph_connected(graph: Dict[int, List[int]]) -> bool:
    """
    Checks if a given graph is connected.

    Args:
    graph: A dictionary representing the graph, where each key is a node and its corresponding value is a list of its neighbors.

    Returns:
    bool: True if the graph is connected, False otherwise.
    """

    # Check if the graph is empty
    if not graph:
        return True  # An empty graph is considered connected

    # Start DFS from an arbitrary node
    start_node = next(iter(graph))

    # Initialize a set to keep track of visited nodes
    visited = set()

    # Perform DFS
    def dfs(node: int) -> None:
        visited.add(node)
        for neighbor in graph[node]:
            if neighbor not in visited:
                dfs(neighbor)

    # Start DFS from the start node
    dfs(start_node)

    # Check if all nodes have been visited
    return len(visited) == len(graph)


# Example usage:
if __name__ == "__main__":
    # Define a connected graph
    connected_graph = {
        0: [1, 2],
        1: [0, 2],
        2: [0, 1]
    }

    # Define a disconnected graph
    disconnected_graph = {
        0: [1],
        1: [0],
        2: [3],
        3: [2]
    }

    print(is_graph_connected(connected_graph))  # Expected output: True
    print(is_graph_connected(disconnected_graph))  # Expected output: False
