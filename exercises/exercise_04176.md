Write a program that takes a value and unit as input and converts it to another unit (e.g. inches to centimeters).

def convert_units(value, from_unit, to_unit):
    """
    Converts a value from one unit to another.

    Args:
        value (float): The value to convert.
        from_unit (str): The unit of the input value.
        to_unit (str): The unit to convert to.

    Returns:
        float: The converted value.
    """
    # Define conversion factors
    conversion_factors = {
        'inches': {'centimeters': 2.54, 'feet': 1/12, 'yards': 1/36, 'miles': 1/63360, 'kilometers': 2.54/100000},
        'centimeters': {'inches': 1/2.54, 'feet': 1/30.48, 'yards': 1/91.44, 'miles': 1/160934, 'kilometers': 1/100000},
        'feet': {'inches': 12, 'centimeters': 30.48, 'yards': 1/3, 'miles': 1/5280, 'kilometers': 30.48/1000},
        'yards': {'inches': 36, 'centimeters': 91.44, 'feet': 3, 'miles': 1/1760, 'kilometers': 91.44/1000},
        'miles': {'inches': 63360, 'centimeters': 160934, 'feet': 5280, 'yards': 1760, 'kilometers': 1.60934},
        'kilometers': {'inches': 100000/2.54, 'centimeters': 100000, 'feet': 1000/3.048, 'yards': 1000/0.9144, 'miles': 1/1.60934}
    }

    # Check if units are valid
    if from_unit not in conversion_factors or to_unit not in conversion_factors:
        raise ValueError("Invalid unit")

    # Convert value
    if from_unit == to_unit:
        return value
    else:
        return value * conversion_factors[from_unit][to_unit]

def main():
    print("Unit Conversion Program")
    print("1. Inches")
    print("2. Centimeters")
    print("3. Feet")
    print("4. Yards")
    print("5. Miles")
    print("6. Kilometers")

    # Get input from user
    value = float(input("Enter a value: "))
    from_unit_index = int(input("Enter the number of the unit to convert from: "))
    to_unit_index = int(input("Enter the number of the unit to convert to: "))

    # Map unit indices to unit names
    units = ['inches', 'centimeters', 'feet', 'yards', 'miles', 'kilometers']
    from_unit = units[from_unit_index - 1]
    to_unit = units[to_unit_index - 1]

    # Convert units
    try:
        result = convert_units(value, from_unit, to_unit)
        print(f"{value} {from_unit} is equal to {result} {to_unit}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
