## Exercise
Create a function that displays a menu with options and takes the user's choice as input.

## Reference Solution
```python
def display_menu():
    """Displays a menu with options and takes the user's choice as input."""
    
    # Define the menu options
    menu_options = {
        "1": "Option 1",
        "2": "Option 2",
        "3": "Option 3",
        "4": "Exit"
    }
    
    # Display the menu
    print("Menu:")
    for key, value in menu_options.items():
        print(f"{key}. {value}")
    
    # Get the user's choice
    while True:
        choice = input("Enter your choice (1-4): ")
        
        # Validate the user's choice
        if choice in menu_options:
            if choice == "4":
                print("Exiting the program.")
                break
            else:
                print(f"You have chosen {menu_options[choice]}")
        else:
            print("Invalid choice. Please choose a valid option.")

# Example usage
display_menu()
```
