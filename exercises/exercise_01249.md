## Exercise
Write a function that demonstrates the concept of variable scope.

## Reference Solution
```python
# Global variable
global_variable = "I am global"

def outer_function():
    """
    Outer function demonstrating variable scope.
    """
    # Enclosed variable
    enclosed_variable = "I am enclosed"

    def inner_function():
        """
        Inner function demonstrating variable scope.
        """
        # Local variable
        local_variable = "I am local"
        
        print("Inside inner_function():")
        print(f"Global variable: {global_variable}")
        print(f"Enclosed variable: {enclosed_variable}")
        print(f"Local variable: {local_variable}")

    inner_function()

    # Try to access the local variable
    try:
        print(f"Local variable inside outer_function(): {local_variable}")
    except NameError:
        print("Local variable is not accessible here")

# Call the outer function
outer_function()

# Try to access the enclosed and local variables
try:
    print(f"Enclosed variable outside functions: {enclosed_variable}")
except NameError:
    print("Enclosed variable is not accessible here")

try:
    print(f"Local variable outside functions: {local_variable}")
except NameError:
    print("Local variable is not accessible here")

print(f"Global variable outside functions: {global_variable}")
```
