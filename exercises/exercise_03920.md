## Exercise
Create a program that calculates the final grade given a list of scores and their respective weights.

## Reference Solution
```python
def calculate_final_grade(scores, weights):
    """
    Calculate the final grade given a list of scores and their respective weights.

    Args:
        scores (list): A list of scores.
        weights (list): A list of weights corresponding to each score.

    Returns:
        float: The final grade.
    """
    if len(scores) != len(weights):
        raise ValueError("The number of scores and weights must be equal.")

    total_score = 0
    total_weight = 0

    for score, weight in zip(scores, weights):
        total_score += score * weight
        total_weight += weight

    if total_weight == 0:
        raise ValueError("The total weight cannot be zero.")

    final_grade = total_score / total_weight
    return final_grade


def main():
    # Example usage
    scores = [90, 80, 70, 95]
    weights = [0.2, 0.3, 0.1, 0.4]

    try:
        final_grade = calculate_final_grade(scores, weights)
        print(f"Final Grade: {final_grade:.2f}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
