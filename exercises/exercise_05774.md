## Exercise
Create a program that checks if a given password meets certain conditions (e.g., length, uppercase letters, numbers).

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validates a password based on the following conditions:
    - Length: at least 8 characters
    - Uppercase letters: at least 1
    - Numbers: at least 1
    - Special characters: at least 1

    Args:
        password (str): The password to validate

    Returns:
        bool: True if the password is valid, False otherwise
    """
    conditions = {
        "length": len(password) >= 8,
        "uppercase": re.search(r"[A-Z]", password) is not None,
        "numbers": re.search(r"[0-9]", password) is not None,
        "special_chars": re.search(r"[^A-Za-z0-9]", password) is not None,
    }

    for condition, met in conditions.items():
        if not met:
            print(f"Password does not meet the '{condition}' condition.")
            return False

    print("Password is valid.")
    return True

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
