## Exercise
Write a program that takes two numbers and an operator as input, calculates the result of the operation, and prints it out.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int x, y;
    char op;

    cout << "Enter two integers and an operator (+, -, *, /): ";
    cin >> x >> op >> y;

    switch (op) {
        case '+':
            cout << x << " + " << y << " = " << x + y << endl;
            break;
        case '-':
            cout << x << " - " << y << " = " << x - y << endl;
            break;
        case '*':
            cout << x << " * " << y << " = " << x * y << endl;
            break;
        case '/':
            if (y != 0) {
                cout << x << " / " << y << " = " << x / y << endl;
            } else {
                cout << "Error: Division by zero!" << endl;
            }
            break;
        default:
            cout << "Error: Invalid operator!" << endl;
            break;
    }

    return 0;
}
```
