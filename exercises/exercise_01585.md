## Exercise
Write a program to take a password as input from the user and check if it meets certain criteria (e.g. length, contains uppercase and lowercase letters, contains numbers).

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on certain criteria.

    Args:
    password (str): The password to be validated.

    Returns:
    bool: True if the password is valid, False otherwise.
    """
    # Check length
    if len(password) < 8:
        print("Password should be at least 8 characters long.")
        return False

    # Check for uppercase letter
    if not re.search("[A-Z]", password):
        print("Password should have at least one uppercase letter.")
        return False

    # Check for lowercase letter
    if not re.search("[a-z]", password):
        print("Password should have at least one lowercase letter.")
        return False

    # Check for number
    if not re.search("[0-9]", password):
        print("Password should have at least one number.")
        return False

    print("Password is valid.")
    return True

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
