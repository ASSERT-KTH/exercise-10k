## Exercise
Create a function that implements a Caesar cipher, shifting each letter in a string by a given number of positions.

## Reference Solution
```python
def caesar_cipher(text: str, shift: int) -> str:
    """
    Encrypts a given string using a Caesar cipher with a specified shift.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The number of positions to shift each letter.

    Returns:
        str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        # Check if character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        # If the character is not a letter, leave it unchanged
        else:
            result += char

    return result

# Example usage:
def main():
    text = "Hello, World!"
    shift = 3
    encrypted = caesar_cipher(text, shift)
    print(f"Text: {text}")
    print(f"Shift: {shift}")
    print(f"Encrypted: {encrypted}")

if __name__ == "__main__":
    main()
```
