## Exercise
Write a program that takes a positive integer as input and checks if it is a prime number. A prime number is a number that is divisible only by 1 and itself.

## Reference Solution
```python
def is_prime(n):
    """
    Checks if a given positive integer is a prime number.

    Args:
        n (int): A positive integer.

    Returns:
        bool: True if the number is prime, False otherwise.
    """

    # Check if the input is a positive integer
    if not isinstance(n, int) or n <= 0:
        raise ValueError("Input must be a positive integer.")

    # Numbers less than 2 are not prime
    if n < 2:
        return False

    # 2 is the only even prime number
    if n == 2:
        return True

    # All other even numbers are not prime
    if n % 2 == 0:
        return False

    # Check for divisibility up to the square root of n
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False

    # If no divisors are found, the number is prime
    return True


# Example usage
def main():
    num = int(input("Enter a positive integer: "))
    if is_prime(num):
        print(f"{num} is a prime number.")
    else:
        print(f"{num} is not a prime number.")


if __name__ == "__main__":
    main()
```
