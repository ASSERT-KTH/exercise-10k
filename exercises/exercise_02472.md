## Exercise
Write a function that formats a given phone number into a standard format.

## Reference Solution
```python
def format_phone_number(phone_number: str) -> str:
    """
    Formats a given phone number into a standard format of (XXX) XXX-XXXX.

    Args:
    phone_number (str): The phone number to be formatted.

    Returns:
    str: The formatted phone number.
    """
    # Remove all non-digit characters from the phone number
    digits_only = ''.join(filter(str.isdigit, phone_number))

    # Check if the phone number has 10 digits
    if len(digits_only) != 10:
        raise ValueError("Invalid phone number. Please enter a 10-digit phone number.")

    # Format the phone number
    formatted_phone_number = f"({digits_only[:3]}) {digits_only[3:6]}-{digits_only[6:]}"

    return formatted_phone_number

# Example usage:
if __name__ == "__main__":
    phone_number = "(123) 456-7890"
    formatted_number = format_phone_number(phone_number)
    print(f"Formatted phone number: {formatted_number}")
```
