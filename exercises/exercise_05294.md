## Exercise
Write a program that solves the Tower of Hanoi puzzle, which involves moving a stack of disks from one rod to another, subject to certain rules. Use a while loop to recursively move the disks.

#include <stdio.h>

void hanoi(int n, int from, int to, int aux) {
    if (n == 1) {
        printf("Move disk 1 from rod %d to rod %d\n", from, to);
        return;
    }

    hanoi(n - 1, from, aux, to);
    printf("Move disk %d from rod %d to rod %d\n", n, from, to);
    hanoi(n - 1, aux, to, from);
}

int main() {
    int n;
    scanf("%d", &n);
    hanoi(n, 1, 3, 2);
    return 0;
}
