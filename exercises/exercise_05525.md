## Exercise
Write a program to perform concatenation, substring, and string comparison operations on two strings.

## Reference Solution
```python
def main():
    # Define two strings
    str1 = input("Enter first string: ")
    str2 = input("Enter second string: ")

    # Perform concatenation
    concatenated_str = str1 + str2
    print(f"Concatenation of '{str1}' and '{str2}': {concatenated_str}")

    # Perform substring operation
    start_index = int(input("Enter the start index for substring: "))
    end_index = int(input("Enter the end index for substring: "))
    substring = str1[start_index:end_index]
    print(f"Substring from '{str1}' from index {start_index} to {end_index}: {substring}")

    # Perform string comparison
    if str1 == str2:
        print(f"'{str1}' is equal to '{str2}'")
    elif str1 > str2:
        print(f"'{str1}' is greater than '{str2}'")
    else:
        print(f"'{str1}' is less than '{str2}'")

if __name__ == "__main__":
    main()
```
