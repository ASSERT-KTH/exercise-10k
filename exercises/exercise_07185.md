## Exercise
Write a program that performs basic matrix operations such as addition and multiplication

## Reference Solution
```python
import numpy as np

class MatrixOperations:
    def __init__(self, matrix1, matrix2):
        """
        Initialize the MatrixOperations class with two matrices.

        Args:
            matrix1 (list): The first matrix.
            matrix2 (list): The second matrix.
        """
        self.matrix1 = np.array(matrix1)
        self.matrix2 = np.array(matrix2)

    def add_matrices(self):
        """
        Add two matrices element-wise.

        Returns:
            np.ndarray: The sum of the two matrices.
        """
        try:
            return self.matrix1 + self.matrix2
        except ValueError:
            print("Matrices must have the same dimensions for addition.")
            return None

    def multiply_matrices(self):
        """
        Multiply two matrices.

        Returns:
            np.ndarray: The product of the two matrices.
        """
        try:
            return np.matmul(self.matrix1, self.matrix2)
        except ValueError:
            print("The number of columns in the first matrix must be equal to the number of rows in the second matrix for multiplication.")
            return None

def main():
    # Example usage:
    matrix1 = [[1, 2], [3, 4]]
    matrix2 = [[5, 6], [7, 8]]

    operations = MatrixOperations(matrix1, matrix2)

    print("Matrix 1:")
    print(operations.matrix1)
    print("\nMatrix 2:")
    print(operations.matrix2)

    print("\nAdding matrices:")
    print(operations.add_matrices())

    print("\nMultiplying matrices:")
    print(operations.multiply_matrices())

if __name__ == "__main__":
    main()
```
