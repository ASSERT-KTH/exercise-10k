## Exercise
Write a program to perform basic boolean operations (AND, OR, NOT).

## Reference Solution
```python
def boolean_and(a, b):
    """
    Performs a boolean AND operation.

    Args:
        a (bool): The first boolean value.
        b (bool): The second boolean value.

    Returns:
        bool: The result of the AND operation.
    """
    return a and b

def boolean_or(a, b):
    """
    Performs a boolean OR operation.

    Args:
        a (bool): The first boolean value.
        b (bool): The second boolean value.

    Returns:
        bool: The result of the OR operation.
    """
    return a or b

def boolean_not(a):
    """
    Performs a boolean NOT operation.

    Args:
        a (bool): The boolean value to negate.

    Returns:
        bool: The result of the NOT operation.
    """
    return not a

def main():
    print("Boolean Operations Program")
    print("---------------------------")

    # AND operation
    a = True
    b = False
    print(f"{a} AND {b} = {boolean_and(a, b)}")

    # OR operation
    a = False
    b = True
    print(f"{a} OR {b} = {boolean_or(a, b)}")

    # NOT operation
    a = True
    print(f"NOT {a} = {boolean_not(a)}")

if __name__ == "__main__":
    main()
```
