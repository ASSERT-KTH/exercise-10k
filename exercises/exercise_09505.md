## Exercise
Write a program that takes an input string and converts it into different data types (integer, float, boolean) based on the user's choice. The program should handle invalid inputs and provide appropriate error messages.

## Reference Solution
```python
def convert_to_int(input_str):
    """Converts the input string to an integer."""
    try:
        return int(input_str)
    except ValueError:
        print("Error: Invalid input. Please enter a valid integer.")
        return None

def convert_to_float(input_str):
    """Converts the input string to a float."""
    try:
        return float(input_str)
    except ValueError:
        print("Error: Invalid input. Please enter a valid float.")
        return None

def convert_to_bool(input_str):
    """Converts the input string to a boolean."""
    if input_str.lower() == "true":
        return True
    elif input_str.lower() == "false":
        return False
    else:
        print("Error: Invalid input. Please enter 'true' or 'false'.")
        return None

def main():
    print("Data Type Converter")
    print("--------------------")

    input_str = input("Enter a string: ")
    print("Choose a data type to convert to:")
    print("1. Integer")
    print("2. Float")
    print("3. Boolean")

    choice = input("Enter your choice (1/2/3): ")

    if choice == "1":
        result = convert_to_int(input_str)
        if result is not None:
            print(f"The integer value is: {result}")
    elif choice == "2":
        result = convert_to_float(input_str)
        if result is not None:
            print(f"The float value is: {result}")
    elif choice == "3":
        result = convert_to_bool(input_str)
        if result is not None:
            print(f"The boolean value is: {result}")
    else:
        print("Error: Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
