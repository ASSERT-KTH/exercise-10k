## Exercise
Write a program that asks the user a series of questions and keeps track of the score.

## Reference Solution
```python
# Define the questions, options, and answers
questions = [
    {
        "question": "What is the capital of France?",
        "options": ["Paris", "London", "Berlin", "Madrid"],
        "answer": 0
    },
    {
        "question": "What is the largest planet in our solar system?",
        "options": ["Earth", "Saturn", "Jupiter", "Uranus"],
        "answer": 2
    },
    {
        "question": "Who painted the Mona Lisa?",
        "options": ["Leonardo da Vinci", "Michelangelo", "Raphael", "Caravaggio"],
        "answer": 0
    }
]

def quiz(questions):
    score = 0
    for i, question in enumerate(questions):
        print(f"Question {i+1}: {question['question']}")
        for j, option in enumerate(question['options']):
            print(f"{j+1}. {option}")
        answer = int(input("Enter the number of your answer: ")) - 1
        if answer == question['answer']:
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, the correct answer was {question['options'][question['answer']]}.\n")
    print(f"Quiz finished. Your final score is {score} out of {len(questions)}.")

quiz(questions)
```
