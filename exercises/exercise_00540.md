## Exercise
Implement a function to convert a string into a zigzag pattern and then read off the characters in row order.

## Reference Solution
```python
def zigzag_conversion(s: str, num_rows: int) -> str:
    """
    Converts a string into a zigzag pattern and reads off the characters in row order.

    Args:
    s (str): The input string.
    num_rows (int): The number of rows in the zigzag pattern.

    Returns:
    str: The string read off in row order.
    """
    if num_rows == 1 or num_rows >= len(s):
        return s

    # Initialize a list of strings, each representing a row in the zigzag pattern
    rows = [''] * num_rows

    # Initialize variables to keep track of the current row and direction
    row, step = 0, 1

    # Iterate over each character in the input string
    for char in s:
        # Append the character to the current row
        rows[row] += char

        # If we are at the first or last row, change the direction
        if row == 0:
            step = 1
        elif row == num_rows - 1:
            step = -1

        # Move to the next row
        row += step

    # Join the rows together to form the final string
    return ''.join(rows)

if __name__ == "__main__":
    s = "PAYPALISHIRING"
    num_rows = 3
    print(zigzag_conversion(s, num_rows))  # Output: "PAHNAPLSIIGYIR"
```
