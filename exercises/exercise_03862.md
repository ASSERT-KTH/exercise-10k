## Exercise
Given a dictionary with names of cities as keys and their populations as values, write a program to access and print the population of a specific city.

## Reference Solution
```python
# city_populations.py

def print_city_population(city_populations, city_name):
    """
    Prints the population of a specific city.

    Args:
        city_populations (dict): Dictionary with city names as keys and populations as values.
        city_name (str): Name of the city to access.

    Returns:
        None
    """
    if city_name in city_populations:
        print(f"The population of {city_name} is: {city_populations[city_name]}")
    else:
        print(f"No population data found for {city_name}")


def main():
    # Create a dictionary with city names and populations
    city_populations = {
        "New York": 8405837,
        "Los Angeles": 3990456,
        "Chicago": 2720595,
        "Houston": 2313392,
        "Phoenix": 1732125
    }

    # Specify the city to access
    city_name = "New York"

    # Print the population of the specified city
    print_city_population(city_populations, city_name)


if __name__ == "__main__":
    main()
```
