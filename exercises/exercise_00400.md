## Exercise
Write a program that calculates the median of a set of numbers.

## Reference Solution
```python
def calculate_median(numbers):
    """
    Calculate the median of a set of numbers.
    
    Parameters:
    numbers (list): A list of numbers.
    
    Returns:
    float: The median of the numbers.
    """
    # Sort the list of numbers
    numbers.sort()
    
    # Find the length of the list
    n = len(numbers)
    
    # If the list has an odd number of elements, the median is the middle element
    if n % 2 == 1:
        median = numbers[n // 2]
    # If the list has an even number of elements, the median is the average of the two middle elements
    else:
        median = (numbers[n // 2 - 1] + numbers[n // 2]) / 2
    
    return median

def main():
    numbers = [12, 4, 56, 7, 23, 11, 89, 34]
    median = calculate_median(numbers)
    print(f"The median of the numbers is: {median}")

if __name__ == "__main__":
    main()
```
