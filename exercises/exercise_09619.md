## Exercise
Write a program that simulates a simple store, allowing users to add, remove, and view products

## Reference Solution
```python
class Product:
    def __init__(self, product_id, name, price):
        """
        Initialize a Product object.

        Args:
            product_id (int): Unique ID for the product.
            name (str): Name of the product.
            price (float): Price of the product.
        """
        self.product_id = product_id
        self.name = name
        self.price = price

class Store:
    def __init__(self):
        """
        Initialize a Store object with an empty dictionary of products.
        """
        self.products = {}

    def add_product(self, product_id, name, price):
        """
        Add a product to the store.

        Args:
            product_id (int): Unique ID for the product.
            name (str): Name of the product.
            price (float): Price of the product.
        """
        if product_id in self.products:
            print("Product with ID {} already exists.".format(product_id))
        else:
            self.products[product_id] = Product(product_id, name, price)
            print("Product with ID {} added successfully.".format(product_id))

    def remove_product(self, product_id):
        """
        Remove a product from the store.

        Args:
            product_id (int): Unique ID of the product to remove.
        """
        if product_id in self.products:
            del self.products[product_id]
            print("Product with ID {} removed successfully.".format(product_id))
        else:
            print("Product with ID {} does not exist.".format(product_id))

    def view_products(self):
        """
        Print all products in the store.
        """
        if not self.products:
            print("No products in the store.")
        else:
            for product in self.products.values():
                print("ID: {}, Name: {}, Price: {}".format(product.product_id, product.name, product.price))

def main():
    store = Store()

    while True:
        print("\nStore Simulator Menu:")
        print("1. Add Product")
        print("2. Remove Product")
        print("3. View Products")
        print("4. Quit")

        choice = input("Enter your choice: ")

        if choice == "1":
            product_id = int(input("Enter product ID: "))
            name = input("Enter product name: ")
            price = float(input("Enter product price: "))
            store.add_product(product_id, name, price)
        elif choice == "2":
            product_id = int(input("Enter product ID to remove: "))
            store.remove_product(product_id)
        elif choice == "3":
            store.view_products()
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
