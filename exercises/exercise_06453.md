## Exercise
Write a program that calculates the perimeter of a triangle given the lengths of its sides.

## Reference Solution
```python
def calculate_perimeter(a, b, c):
    """
    Calculates the perimeter of a triangle given the lengths of its sides.

    Args:
        a (float): Length of side a
        b (float): Length of side b
        c (float): Length of side c

    Returns:
        float: Perimeter of the triangle
    """
    if a + b <= c or a + c <= b or b + c <= a:
        raise ValueError("Invalid triangle sides. The sum of any two sides must be greater than the third side.")
    return a + b + c

def main():
    # Example usage
    a = float(input("Enter the length of side a: "))
    b = float(input("Enter the length of side b: "))
    c = float(input("Enter the length of side c: "))

    try:
        perimeter = calculate_perimeter(a, b, c)
        print(f"The perimeter of the triangle is: {perimeter:.2f}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
