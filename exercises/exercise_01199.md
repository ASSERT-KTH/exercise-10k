## Exercise
Write a program that calculates the exam grade based on the number of correct answers and the total number of questions.

## Reference Solution
```python
def calculate_grade(correct_answers, total_questions):
    """
    Calculate the exam grade based on the number of correct answers and the total number of questions.

    Args:
        correct_answers (int): The number of correct answers.
        total_questions (int): The total number of questions.

    Returns:
        float: The exam grade as a percentage.
    """
    if total_questions == 0:
        raise ValueError("Total number of questions cannot be zero")

    grade = (correct_answers / total_questions) * 100
    return grade


def main():
    # Get the number of correct answers and total questions from the user
    correct_answers = int(input("Enter the number of correct answers: "))
    total_questions = int(input("Enter the total number of questions: "))

    # Calculate the exam grade
    grade = calculate_grade(correct_answers, total_questions)

    # Print the exam grade
    print(f"Your exam grade is: {grade:.2f}%")

    # Determine the letter grade based on the percentage
    if grade >= 90:
        print("Your letter grade is: A")
    elif grade >= 80:
        print("Your letter grade is: B")
    elif grade >= 70:
        print("Your letter grade is: C")
    elif grade >= 60:
        print("Your letter grade is: D")
    else:
        print("Your letter grade is: F")


if __name__ == "__main__":
    main()
```
