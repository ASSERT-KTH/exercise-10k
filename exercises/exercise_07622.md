## Exercise
Create a program that uses a for loop to generate a haiku poem with a given number of syllables.

## Reference Solution
```python
import random

# List of words with their corresponding syllable counts
words = {
    "sun": 1,
    "moon": 1,
    "sky": 1,
    "river": 2,
    "mountain": 2,
    "flower": 2,
    "beautiful": 3,
    "serene": 2,
    "peaceful": 2,
    "nature": 2,
    "calm": 1,
    "gentle": 2,
    "breeze": 1,
    "sunset": 2,
    "dawn": 1,
    "night": 1,
    "stars": 1,
    "twinkling": 2,
    "shining": 2,
}

def generate_haiku(syllable_counts):
    """
    Generate a haiku poem with the given syllable counts.

    Args:
    syllable_counts (list): A list of syllable counts for each line of the haiku.

    Returns:
    str: The generated haiku poem.
    """
    haiku = []
    for syllable_count in syllable_counts:
        line = []
        remaining_syllables = syllable_count
        while remaining_syllables > 0:
            word = random.choice(list(words.keys()))
            if words[word] <= remaining_syllables:
                line.append(word)
                remaining_syllables -= words[word]
        haiku.append(" ".join(line))
    return "\n".join(haiku)

# Example usage
syllable_counts = [5, 7, 5]
print(generate_haiku(syllable_counts))
```
