## Exercise
Create two dictionaries with 3 key-value pairs each and merge them into a single dictionary.

## Reference Solution
```python
# Create two dictionaries with 3 key-value pairs each
dict1 = {"a": 1, "b": 2, "c": 3}
dict2 = {"d": 4, "e": 5, "f": 6}

# Method 1: Using the update() method
def merge_dicts_update(dict1, dict2):
    merged_dict = dict1.copy()  # Create a copy to avoid modifying the original dictionary
    merged_dict.update(dict2)
    return merged_dict

# Method 2: Using the {**dict1, **dict2} syntax (Python 3.5+)
def merge_dicts_syntax(dict1, dict2):
    return {**dict1, **dict2}

# Example usage
if __name__ == "__main__":
    print("Dictionary 1:", dict1)
    print("Dictionary 2:", dict2)

    merged_dict_update = merge_dicts_update(dict1, dict2)
    print("Merged Dictionary (update() method):", merged_dict_update)

    merged_dict_syntax = merge_dicts_syntax(dict1, dict2)
    print("Merged Dictionary ({**dict1, **dict2} syntax):", merged_dict_syntax)
```
