## Exercise
Write a program that solves a system of linear simultaneous equations.

## Reference Solution
```python
import numpy as np

def solve_linear_equations(coefficients, constants):
    """
    Solves a system of linear simultaneous equations.

    Args:
    coefficients (list of lists): Coefficients of the variables.
    constants (list): Constants on the right-hand side of the equations.

    Returns:
    solution (list): Values of the variables.
    """
    # Convert the coefficients and constants to NumPy arrays
    coefficients = np.array(coefficients)
    constants = np.array(constants)

    # Use NumPy's linalg.solve function to solve the system of equations
    solution = np.linalg.solve(coefficients, constants)

    return solution

# Example usage
if __name__ == "__main__":
    # Define the coefficients of the variables
    coefficients = [[3, 1], [1, 2]]

    # Define the constants on the right-hand side of the equations
    constants = [9, 8]

    # Solve the system of linear equations
    solution = solve_linear_equations(coefficients, constants)

    # Print the solution
    print("Solution:")
    for i, value in enumerate(solution):
        print(f"x{i+1} = {value}")
```
