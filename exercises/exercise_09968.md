## Exercise
Write a program that replaces a substring with another string.

## Reference Solution
```python
def replace_substring(input_string, old_substring, new_substring):
    """
    Replaces a specified substring with another string in a given input string.

    Args:
        input_string (str): The original string.
        old_substring (str): The substring to be replaced.
        new_substring (str): The substring to replace with.

    Returns:
        str: The modified string with the substring replaced.
    """
    return input_string.replace(old_substring, new_substring)

def main():
    # Example usage
    input_string = "Hello, world! This is a test string."
    old_substring = "world"
    new_substring = "Python"

    print("Original String:")
    print(input_string)

    modified_string = replace_substring(input_string, old_substring, new_substring)

    print("\nModified String:")
    print(modified_string)

if __name__ == "__main__":
    main()
```
