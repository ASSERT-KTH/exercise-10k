## Exercise
Write a program to perform basic set operations (union, intersection, difference, symmetric difference) on two given sets

## Reference Solution
```python
def set_operations(set1, set2):
    """
    Perform basic set operations on two given sets.

    Args:
        set1 (set): The first set.
        set2 (set): The second set.

    Returns:
        dict: A dictionary containing the results of the set operations.
    """
    operations = {
        "union": set1.union(set2),
        "intersection": set1.intersection(set2),
        "difference": set1.difference(set2),
        "symmetric_difference": set1.symmetric_difference(set2)
    }
    return operations

def main():
    # Example usage
    set1 = {1, 2, 3, 4, 5}
    set2 = {4, 5, 6, 7, 8}

    operations = set_operations(set1, set2)

    print("Set 1:", set1)
    print("Set 2:", set2)
    print("Union:", operations["union"])
    print("Intersection:", operations["intersection"])
    print("Difference (Set 1 - Set 2):", operations["difference"])
    print("Symmetric Difference:", operations["symmetric_difference"])

if __name__ == "__main__":
    main()
```
