## Exercise
Write a program to create a phonebook app where users can add, remove, and search contacts.

## Reference Solution
```python
class Contact:
    def __init__(self, name, phone_number):
        self.name = name
        self.phone_number = phone_number

class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone_number):
        if name in self.contacts:
            print(f"Contact with name '{name}' already exists.")
        else:
            new_contact = Contact(name, phone_number)
            self.contacts[name] = new_contact
            print(f"Contact '{name}' added successfully.")

    def remove_contact(self, name):
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact '{name}' removed successfully.")
        else:
            print(f"Contact with name '{name}' does not exist.")

    def search_contact(self, name):
        if name in self.contacts:
            contact = self.contacts[name]
            print(f"Name: {contact.name}, Phone Number: {contact.phone_number}")
        else:
            print(f"Contact with name '{name}' does not exist.")

    def display_contacts(self):
        if not self.contacts:
            print("Phonebook is empty.")
        else:
            for contact in self.contacts.values():
                print(f"Name: {contact.name}, Phone Number: {contact.phone_number}")

def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook App")
        print("1. Add Contact")
        print("2. Remove Contact")
        print("3. Search Contact")
        print("4. Display Contacts")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter contact name: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter contact name: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Exiting phonebook app.")
            break
        else:
            print("Invalid option. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
