## Exercise
Create a program that uses nested for loops to multiply two matrices.

## Reference Solution
```python
def multiply_matrices(mat1, mat2):
    """
    Multiply two matrices using nested for loops.

    Args:
        mat1 (list): The first matrix.
        mat2 (list): The second matrix.

    Returns:
        list: The product of the two matrices.

    Raises:
        ValueError: If the matrices cannot be multiplied.
    """
    # Get the dimensions of the matrices
    rows_mat1 = len(mat1)
    cols_mat1 = len(mat1[0])
    rows_mat2 = len(mat2)
    cols_mat2 = len(mat2[0])

    # Check if the matrices can be multiplied
    if cols_mat1 != rows_mat2:
        raise ValueError("Matrices cannot be multiplied")

    # Create a result matrix filled with zeros
    result = [[0 for _ in range(cols_mat2)] for _ in range(rows_mat1)]

    # Perform the matrix multiplication
    for i in range(rows_mat1):
        for j in range(cols_mat2):
            for k in range(cols_mat1):
                result[i][j] += mat1[i][k] * mat2[k][j]

    return result


def print_matrix(matrix):
    """
    Print a matrix in a readable format.

    Args:
        matrix (list): The matrix to print.
    """
    for row in matrix:
        print(row)


def main():
    # Define two example matrices
    mat1 = [[1, 2, 3], [4, 5, 6]]
    mat2 = [[7, 8], [9, 10], [11, 12]]

    print("Matrix 1:")
    print_matrix(mat1)
    print("\nMatrix 2:")
    print_matrix(mat2)

    try:
        result = multiply_matrices(mat1, mat2)
        print("\nResult:")
        print_matrix(result)
    except ValueError as e:
        print("\nError:", e)


if __name__ == "__main__":
    main()
```
