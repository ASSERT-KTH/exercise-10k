## Exercise
Write a function to find the maximum area that can be trapped between two lines.

## Reference Solution

import math

def max_area_between_lines(m1, b1, m2, b2):
    """
    Calculate the maximum area that can be trapped between two lines.

    Args:
    m1 (float): Slope of the first line
    b1 (float): Y-intercept of the first line
    m2 (float): Slope of the second line
    b2 (float): Y-intercept of the second line

    Returns:
    float: The maximum area that can be trapped between the two lines
    """
    # Check if the lines are parallel
    if m1 == m2:
        # If the lines are parallel, the maximum area is infinity
        return float('inf')

    # Calculate the x-coordinate of the intersection point
    x = (b2 - b1) / (m1 - m2)

    # Calculate the y-coordinate of the intersection point
    y = m1 * x + b1

    # Calculate the height of the triangle formed by the intersection point and the two lines
    height = abs(b2 - b1) / math.sqrt(1 + (m1 - m2) ** 2)

    # The base of the triangle is the distance between the two lines at the intersection point
    base = 1  # The base is arbitrary, as it does not affect the maximum area

    # Calculate the maximum area
    max_area = 0.5 * base * height

    return max_area

# Example usage:
m1 = 1  # Slope of the first line
b1 = 2  # Y-intercept of the first line
m2 = -1  # Slope of the second line
b2 = 3  # Y-intercept of the second line

max_area = max_area_between_lines(m1, b1, m2, b2)
print("The maximum area that can be trapped between the two lines is:", max_area)
