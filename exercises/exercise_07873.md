## Exercise
Write a program that plays a guessing game with a limited number of attempts using conditional statements.

## Reference Solution
```python
# Import the random module
import random

def guessing_game():
    # Define the number of attempts
    attempts = 5

    # Generate a random number between 1 and 100
    number_to_guess = random.randint(1, 100)

    print("Welcome to the guessing game!")
    print(f"You have {attempts} attempts to guess the number.")

    # Loop until the user guesses the number or runs out of attempts
    for i in range(attempts):
        # Ask the user for their guess
        user_guess = input("Guess a number between 1 and 100: ")

        # Check if the user's guess is a valid number
        if not user_guess.isdigit():
            print("Invalid input! Please enter a number.")
            continue

        # Convert the user's guess to an integer
        user_guess = int(user_guess)

        # Check if the user's guess is in the correct range
        if user_guess < 1 or user_guess > 100:
            print("Invalid input! Please enter a number between 1 and 100.")
            continue

        # Check if the user's guess is correct
        if user_guess == number_to_guess:
            print(f"Congratulations! You guessed the number in {i+1} attempts.")
            return

        # If the user's guess is not correct, give them a hint
        if user_guess < number_to_guess:
            print("Too low! Try a higher number.")
        else:
            print("Too high! Try a lower number.")

        # Print the remaining attempts
        print(f"You have {attempts-i-1} attempts left.")

    # If the user runs out of attempts, they lose
    print(f"Sorry, you didn't guess the number. The number was {number_to_guess}.")

# Call the function to start the game
guessing_game()
```
