## Exercise
Create a program that takes in a string input representing a data type (e.g., 'int', 'float', 'str') and outputs the corresponding data type in a different programming language (e.g., Java, C++, Python). The program should handle at least 5 different data types and 3 different programming languages.

## Reference Solution
```python
import java.util.HashMap;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        System.out.println(convertDataType("int", "C#"));  // Output: int
        System.out.println(convertDataType("float", "Java"));  // Output: float
        System.out.println(convertDataType("str", "VB.NET"));  // Output: String
    }

    public static String convertDataType(String inputType, String outputLanguage) {
        Map<String, Map<String, String>> dataTypeConversions = new HashMap<>();
        Map<String, String> csharpTypes = new HashMap<>();
        csharpTypes.put("int", "int");
        csharpTypes.put("float", "float");
        csharpTypes.put("str", "string");
        csharpTypes.put("bool", "bool");
        csharpTypes.put("double", "double");
        dataTypeConversions.put("C#", csharpTypes);

        Map<String, String> javaTypes = new HashMap<>();
        javaTypes.put("int", "int");
        javaTypes.put("float", "float");
        javaTypes.put("str", "String");
        javaTypes.put("bool", "boolean");
        javaTypes.put("double", "double");
        dataTypeConversions.put("Java", javaTypes);

        Map<String, String> vbnetTypes = new HashMap<>();
        vbnetTypes.put("int", "Integer");
        vbnetTypes.put("float", "Single");
        vbnetTypes.put("str", "String");
        vbnetTypes.put("bool", "Boolean");
        vbnetTypes.put("double", "Double");
        dataTypeConversions.put("VB.NET", vbnetTypes);

        if (dataTypeConversions.containsKey(outputLanguage)) {
            if (dataTypeConversions.get(outputLanguage).containsKey(inputType)) {
                return dataTypeConversions.get(outputLanguage).get(inputType);
            } else {
                return "Invalid input data type";
            }
        } else {
            return "Invalid output language";
        }
    }
}
```
