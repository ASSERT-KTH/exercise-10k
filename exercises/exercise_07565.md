## Exercise
Create a function that uses a decorator to modify its behavior or add additional functionality.

## Reference Solution
```python
import time
from functools import wraps

# Define a decorator function
def timer_decorator(func):
    """
    A decorator function that measures the execution time of a given function.
    
    Args:
        func: The function to be decorated.
    
    Returns:
        A wrapper function that includes the execution time measurement.
    """
    @wraps(func)
    def wrapper(*args, **kwargs):
        # Record the start time
        start_time = time.time()
        
        # Call the original function
        result = func(*args, **kwargs)
        
        # Record the end time
        end_time = time.time()
        
        # Calculate the execution time
        execution_time = end_time - start_time
        
        # Print the execution time
        print(f"Function '{func.__name__}' executed in {execution_time:.4f} seconds.")
        
        # Return the result of the original function
        return result
    
    return wrapper

# Apply the decorator to a function
@timer_decorator
def example_function():
    """
    An example function that simulates some work.
    """
    time.sleep(1)  # Simulate some work
    print("Example function executed.")

# Test the decorated function
def main():
    example_function()

if __name__ == "__main__":
    main()
```
