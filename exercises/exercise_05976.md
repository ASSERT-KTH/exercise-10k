## Exercise
Write a program that takes a list as input, and returns all possible permutations of the input list.

## Reference Solution
```python
import itertools

def get_permutations(input_list):
    """
    Returns all possible permutations of the input list.

    Args:
        input_list (list): The input list for which to generate permutations.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    return list(itertools.permutations(input_list))

def main():
    # Example usage:
    input_list = [1, 2, 3]
    permutations = get_permutations(input_list)

    print("Input List:", input_list)
    print("Permutations:")
    for i, permutation in enumerate(permutations):
        print(f"Permutation {i+1}: {permutation}")

if __name__ == "__main__":
    main()
```

```python
def get_permutations_recursive(input_list):
    """
    Returns all possible permutations of the input list using recursion.

    Args:
        input_list (list): The input list for which to generate permutations.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    if len(input_list) == 1:
        return [input_list]

    permutations = []
    for i, element in enumerate(input_list):
        remaining_list = input_list[:i] + input_list[i+1:]
        for permutation in get_permutations_recursive(remaining_list):
            permutations.append([element] + permutation)

    return permutations
```
