## Exercise
Write a program that takes the coordinates of two points as input, and returns the distance between them.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.

    Args:
        point1 (tuple): Coordinates of the first point (x, y) or (x, y, z)
        point2 (tuple): Coordinates of the second point (x, y) or (x, y, z)

    Returns:
        float: The distance between the two points
    """
    # Check if both points have the same number of coordinates
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of coordinates")

    # Calculate the squared differences between corresponding coordinates
    squared_differences = [(a - b) ** 2 for a, b in zip(point1, point2)]

    # Calculate the Euclidean distance using the Pythagorean theorem
    distance = math.sqrt(sum(squared_differences))

    return distance

# Example usage
if __name__ == "__main__":
    point1 = (1, 2)
    point2 = (4, 6)
    print(f"The distance between {point1} and {point2} is {calculate_distance(point1, point2)}")

    point3 = (1, 2, 3)
    point4 = (4, 5, 6)
    print(f"The distance between {point3} and {point4} is {calculate_distance(point3, point4)}")
```
