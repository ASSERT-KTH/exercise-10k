## Exercise
Write a function that uses coroutines to perform concurrent tasks. Then, call this function with an input value and print the result.

## Reference Solution
```python
import asyncio
import time

# Define an asynchronous function to simulate a time-consuming task
async def task(name, delay):
    """Simulate a time-consuming task."""
    print(f"Task {name} started")
    await asyncio.sleep(delay)  # Pause execution for the specified delay
    print(f"Task {name} finished")
    return f"Task {name} result"

# Define an asynchronous function to perform multiple tasks concurrently
async def perform_concurrent_tasks(tasks):
    """Perform multiple tasks concurrently using coroutines."""
    # Create a list to store the tasks
    task_list = []
    
    # Create tasks and add them to the list
    for task_name, delay in tasks:
        task_list.append(task(task_name, delay))
    
    # Use asyncio.gather to run all tasks concurrently
    results = await asyncio.gather(*task_list)
    
    # Return the results
    return results

# Define the main function to call perform_concurrent_tasks
async def main():
    # Define the input tasks
    tasks = [("A", 2), ("B", 1), ("C", 3)]
    
    # Call the function to perform concurrent tasks
    results = await perform_concurrent_tasks(tasks)
    
    # Print the results
    for result in results:
        print(result)

# Run the main function
if __name__ == "__main__":
    start_time = time.time()
    asyncio.run(main())
    print(f"Total execution time: {time.time() - start_time} seconds")
```
