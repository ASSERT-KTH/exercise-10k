## Exercise
Write a program to convert a given date and time from one timezone to another.

## Reference Solution
```python
from datetime import datetime
import pytz

def convert_timezone(date_string, from_timezone, to_timezone, date_format):
    """
    Convert a date and time from one timezone to another.

    Args:
        date_string (str): The date and time string to convert.
        from_timezone (str): The timezone of the input date and time.
        to_timezone (str): The timezone to convert to.
        date_format (str): The format of the input date and time string.

    Returns:
        str: The converted date and time string.
    """

    # Parse the input date and time string
    date_object = datetime.strptime(date_string, date_format)

    # Localize the date object to the from timezone
    from_tz = pytz.timezone(from_timezone)
    localized_date = from_tz.localize(date_object)

    # Convert the localized date to the to timezone
    to_tz = pytz.timezone(to_timezone)
    converted_date = localized_date.astimezone(to_tz)

    # Return the converted date and time string
    return converted_date.strftime(date_format)


def main():
    # Test the function
    date_string = "2022-01-01 12:00:00"
    from_timezone = "US/Pacific"
    to_timezone = "US/Eastern"
    date_format = "%Y-%m-%d %H:%M:%S"

    converted_date = convert_timezone(date_string, from_timezone, to_timezone, date_format)
    print("Original Date:", date_string)
    print("From Timezone:", from_timezone)
    print("To Timezone:", to_timezone)
    print("Converted Date:", converted_date)


if __name__ == "__main__":
    main()
```
