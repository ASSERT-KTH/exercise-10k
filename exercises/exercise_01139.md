## Exercise
Create a program that allows the user to manage a bank account (e.g. deposit, withdraw, check balance)

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_number, account_name, initial_balance=0.0):
        """
        Initialize a bank account.

        Args:
            account_number (str): Unique account number.
            account_name (str): Account holder's name.
            initial_balance (float, optional): Initial account balance. Defaults to 0.0.
        """
        self.account_number = account_number
        self.account_name = account_name
        self.balance = initial_balance

    def deposit(self, amount):
        """
        Deposit funds into the account.

        Args:
            amount (float): Amount to deposit.

        Returns:
            float: Updated account balance.
        """
        if amount > 0:
            self.balance += amount
            return self.balance
        else:
            print("Invalid deposit amount. Please enter a positive value.")

    def withdraw(self, amount):
        """
        Withdraw funds from the account.

        Args:
            amount (float): Amount to withdraw.

        Returns:
            float: Updated account balance if withdrawal is successful, otherwise None.
        """
        if 0 < amount <= self.balance:
            self.balance -= amount
            return self.balance
        else:
            print("Insufficient funds or invalid withdrawal amount.")

    def check_balance(self):
        """
        Check the current account balance.

        Returns:
            float: Current account balance.
        """
        return self.balance


def main():
    # Create a new bank account
    account_number = input("Enter account number: ")
    account_name = input("Enter account holder's name: ")
    initial_balance = float(input("Enter initial balance (default=0.0): ") or 0.0)

    account = BankAccount(account_number, account_name, initial_balance)

    while True:
        print("\nBank Account Management Menu:")
        print("1. Deposit")
        print("2. Withdraw")
        print("3. Check Balance")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            amount = float(input("Enter deposit amount: "))
            new_balance = account.deposit(amount)
            if new_balance is not None:
                print(f"Deposit successful. New balance: ${new_balance:.2f}")
        elif choice == "2":
            amount = float(input("Enter withdrawal amount: "))
            new_balance = account.withdraw(amount)
            if new_balance is not None:
                print(f"Withdrawal successful. New balance: ${new_balance:.2f}")
        elif choice == "3":
            balance = account.check_balance()
            print(f"Current balance: ${balance:.2f}")
        elif choice == "4":
            print("Exiting program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
