## Exercise
Write a program that takes a string as input and returns a snapshot of the string at a given index

## Reference Solution
```python
def string_snapshot(input_str, index):
    """
    Returns a snapshot of the string at a given index.

    Args:
        input_str (str): The input string.
        index (int): The index at which to take the snapshot.

    Returns:
        tuple: A tuple containing the characters before and after the index.
    """
    if index < 0 or index >= len(input_str):
        raise ValueError("Index out of range")

    before = input_str[:index]
    after = input_str[index + 1:]

    return before, input_str[index], after


def main():
    input_str = input("Enter a string: ")
    index = int(input("Enter an index: "))

    try:
        before, char_at_index, after = string_snapshot(input_str, index)
        print(f"Before index {index}: {before}")
        print(f"Character at index {index}: {char_at_index}")
        print(f"After index {index}: {after}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
