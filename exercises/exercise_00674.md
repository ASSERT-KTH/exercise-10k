## Exercise
Given a dictionary {'name': 'Luke', 'age': 30, 'city': 'Brisbane'}, write a program to use the update() method to add a new key-value pair 'country': 'Australia'.

## Reference Solution
```python
def add_key_value_pair(dictionary, key, value):
    """
    Adds a new key-value pair to the dictionary using the update() method.

    Args:
        dictionary (dict): The dictionary to be updated.
        key (str): The key to be added.
        value (str): The value associated with the key.
    """
    dictionary.update({key: value})
    return dictionary

def main():
    # Given dictionary
    person = {'name': 'Luke', 'age': 30, 'city': 'Brisbane'}
    
    # New key-value pair to be added
    new_key = 'country'
    new_value = 'Australia'
    
    print("Original Dictionary:")
    print(person)
    
    # Add the new key-value pair
    updated_person = add_key_value_pair(person, new_key, new_value)
    
    print("\nUpdated Dictionary:")
    print(updated_person)

if __name__ == "__main__":
    main()
```
