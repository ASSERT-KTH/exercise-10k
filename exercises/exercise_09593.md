## Exercise
Write a program that prints out a table with user-provided data

## Reference Solution
```python
def print_table(rows, cols, data):
    """
    Prints a table with the given number of rows and columns, filled with the provided data.

    Args:
        rows (int): The number of rows in the table.
        cols (int): The number of columns in the table.
        data (list): A 2D list of data to be filled in each cell.
    """
    # Print the top border of the table
    print("+", end="")
    for _ in range(cols):
        print("---+", end="")
    print()

    # Print each row of the table
    for row in data:
        print("|", end="")
        for cell in row:
            print(f" {cell} |", end="")
        print()
        print("+", end="")
        for _ in range(cols):
            print("---+", end="")
        print()

def get_table_data(rows, cols):
    """
    Prompts the user for data to fill in each cell of the table.

    Args:
        rows (int): The number of rows in the table.
        cols (int): The number of columns in the table.

    Returns:
        list: A 2D list of data to be filled in each cell.
    """
    data = []
    for i in range(rows):
        row = []
        for j in range(cols):
            cell_data = input(f"Enter data for row {i+1}, column {j+1}: ")
            row.append(cell_data)
        data.append(row)
    return data

def main():
    # Prompt the user for the number of rows and columns
    rows = int(input("Enter the number of rows: "))
    cols = int(input("Enter the number of columns: "))

    # Get the data to fill in each cell
    data = get_table_data(rows, cols)

    # Print the table
    print_table(rows, cols, data)

if __name__ == "__main__":
    main()
```
