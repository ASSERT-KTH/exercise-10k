## Exercise
Write a program that takes a person's weight and height as input and returns their BMI.

## Reference Solution
```python
def calculate_bmi(weight, height):
    """
    Calculate BMI based on weight and height.

    Args:
        weight (float): Weight in kilograms.
        height (float): Height in meters.

    Returns:
        float: BMI value.
    """
    bmi = weight / (height ** 2)
    return bmi

def classify_bmi(bmi):
    """
    Classify BMI into underweight, normal, overweight, or obese categories.

    Args:
        bmi (float): BMI value.

    Returns:
        str: BMI classification.
    """
    if bmi < 18.5:
        return "Underweight"
    elif bmi < 25:
        return "Normal"
    elif bmi < 30:
        return "Overweight"
    else:
        return "Obese"

def main():
    # Get user input
    weight = float(input("Enter your weight in kilograms: "))
    height = float(input("Enter your height in meters: "))

    # Calculate BMI
    bmi = calculate_bmi(weight, height)

    # Classify BMI
    classification = classify_bmi(bmi)

    # Print results
    print(f"Your BMI is: {bmi:.2f}")
    print(f"Your BMI classification is: {classification}")

if __name__ == "__main__":
    main()
```
