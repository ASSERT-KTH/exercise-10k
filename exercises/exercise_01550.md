## Exercise
Given an array of integers, write a function to calculate and return the minimum, maximum, and average values.

## Reference Solution
```python
def calculate_min_max_average(array):
    """
    Calculate and return the minimum, maximum, and average values from a given array of integers.

    Args:
        array (list): A list of integers.

    Returns:
        dict: A dictionary containing the minimum, maximum, and average values.
    """
    if not array:
        raise ValueError("Array cannot be empty")

    min_value = min(array)
    max_value = max(array)
    average_value = sum(array) / len(array)

    return {
        "minimum": min_value,
        "maximum": max_value,
        "average": average_value
    }

# Example usage
def main():
    array = [1, 2, 3, 4, 5]
    result = calculate_min_max_average(array)
    print("Minimum:", result["minimum"])
    print("Maximum:", result["maximum"])
    print("Average:", result["average"])

if __name__ == "__main__":
    main()
```
