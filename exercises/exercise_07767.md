## Exercise
Create a program that takes two numbers and an arithmetic operation as input and prints the result. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
def calculate(num1, num2, operation):
    """
    Perform arithmetic operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to perform. Can be 'addition', 'subtraction', 'multiplication', 'division', '+', '-', '*', '/'.

    Returns:
        float: The result of the operation.
    """

    # Check if the operation is valid
    if operation not in ['addition', 'subtraction', 'multiplication', 'division', '+', '-', '*', '/']:
        raise ValueError("Invalid operation. Must be 'addition', 'subtraction', 'multiplication', 'division', '+', '-', '*', '/'.")

    # Perform the operation
    if operation in ['addition', '+']:
        return num1 + num2
    elif operation in ['subtraction', '-']:
        return num1 - num2
    elif operation in ['multiplication', '*']:
        return num1 * num2
    elif operation in ['division', '/']:
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return num1 / num2


def main():
    # Get the numbers and operation from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, / or addition, subtraction, multiplication, division): ")

    try:
        # Calculate and print the result
        result = calculate(num1, num2, operation)
        print(f"{num1} {get_operation_symbol(operation)} {num2} = {result}")
    except Exception as e:
        print(str(e))


def get_operation_symbol(operation):
    """
    Get the symbol for the operation.

    Args:
        operation (str): The arithmetic operation.

    Returns:
        str: The symbol for the operation.
    """

    if operation in ['addition', '+']:
        return '+'
    elif operation in ['subtraction', '-']:
        return '-'
    elif operation in ['multiplication', '*']:
        return '*'
    elif operation in ['division', '/']:
        return '/'


if __name__ == "__main__":
    main()
```
