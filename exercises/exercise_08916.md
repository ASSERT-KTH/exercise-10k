## Exercise
Write a program to check if a given string of parentheses is balanced.

## Reference Solution
```python
def is_balanced(s):
    """
    Checks if a given string of parentheses is balanced.

    Args:
        s (str): The input string of parentheses.

    Returns:
        bool: True if the string is balanced, False otherwise.
    """
    stack = []
    mapping = {")": "(", "}": "{", "]": "["}

    for char in s:
        if char in mapping.values():
            stack.append(char)
        elif char in mapping.keys():
            if stack == [] or mapping[char] != stack.pop():
                return False

    return stack == []

# Example usage
if __name__ == "__main__":
    test_cases = [
        ("((()))", True),
        ("(()", False),
        ("({[]})", True),
        ("({[}])", False),
    ]

    for s, expected in test_cases:
        result = is_balanced(s)
        print(f"Input: {s}, Expected: {expected}, Result: {result}")
```
