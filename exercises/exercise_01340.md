## Exercise
Write a function that takes two strings as input and returns the index of the first occurrence of the second string in the first string using the Rabin-Karp algorithm.

## Reference Solution
```python
def rabin_karp(s, t):
    n = len(s)
    m = len(t)
    d = 256  # number of possible characters
    q = 101  # a prime number
    pattern_hash = 0
    text_hash = 0
    h = 1

    # Precompute h value
    for _ in range(m - 1):
        h = (h * d) % q

    # Calculate initial hash values for pattern and text
    for i in range(m):
        pattern_hash = (d * pattern_hash + ord(t[i])) % q
        text_hash = (d * text_hash + ord(s[i])) % q

    # Slide the pattern over the text
    for i in range(n - m + 1):
        if pattern_hash == text_hash:
            match = True
            for j in range(m):
                if s[i + j] != t[j]:
                    match = False
                    break
            if match:
                return i  # pattern found

        # Calculate hash value for next window
        if i < n - m:
            text_hash = (d * (text_hash - ord(s[i]) * h) + ord(s[i + m])) % q
            if text_hash < 0:
                text_hash += q

    return -1  # pattern not found

# Example usage
s = "ABCCDDAEFG"
t = "CDD"
index = rabin_karp(s, t)
if index != -1:
    print(f"Pattern found at index {index}")
else:
    print("Pattern not found")
```
