## Exercise
Create a program that calculates the greatest common factor (GCF) of two given numbers.

## Reference Solution
```python
def calculate_gcf(num1: int, num2: int) -> int:
    """
    Calculate the greatest common factor (GCF) of two numbers using the Euclidean algorithm.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The GCF of num1 and num2.
    """
    while num2:
        num1, num2 = num2, num1 % num2
    return abs(num1)

def main():
    # Example usage
    num1 = 48
    num2 = 18
    gcf = calculate_gcf(num1, num2)
    print(f"The GCF of {num1} and {num2} is {gcf}")

if __name__ == "__main__":
    main()
```
