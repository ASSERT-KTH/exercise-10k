## Exercise
Write a program to create a quiz with multiple-choice questions.

## Reference Solution
```python
# Define a dictionary to store questions, options, and answers
questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Who painted the Mona Lisa?": {
        "A": "Leonardo da Vinci",
        "B": "Michelangelo",
        "C": "Raphael",
        "D": "Caravaggio",
        "answer": "A"
    }
}

def take_quiz(questions):
    """Take the quiz and return the score"""
    score = 0
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        answer = input("Choose your answer (A, B, C, D): ")
        if answer.upper() == options["answer"]:
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, the correct answer is {options['answer']}\n")
    return score

def main():
    """Main function to run the quiz"""
    print("Welcome to the quiz!")
    score = take_quiz(questions)
    print(f"Your final score is {score} out of {len(questions)}")

if __name__ == "__main__":
    main()
```
