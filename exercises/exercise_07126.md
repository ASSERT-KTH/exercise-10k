## Exercise
Write a program that calculates and prints the size of different data types such as integers, floats, and characters. Use the sizeof() function to get the size of each data type.

#include <iostream>
#include <conio.h>

int main() {
    // Printing sizes of data types
    std::cout << "Size of int: " << sizeof(int) << std::endl;
    std::cout << "Size of float: " << sizeof(float) << std::endl;
    std::cout << "Size of char: " << sizeof(char) << std::endl;
    std::cout << "Size of long int: " << sizeof(long int) << std::endl;
    std::cout << "Size of double: " << sizeof(double) << std::endl;
    std::cout << "Size of long double: " << sizeof(long double) << std::endl;
    std::cout << "Size of short int: " << sizeof(short int) << std::endl;

    return 0;
}
