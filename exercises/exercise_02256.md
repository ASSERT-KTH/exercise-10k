## Exercise
Write a program that takes an array of numbers as input and returns the minimum value.

## Reference Solution
```python
def find_min_value(array):
    """
    This function finds the minimum value in a given array.

    Args:
        array (list): A list of numbers.

    Returns:
        int: The minimum value in the array.

    Raises:
        ValueError: If the input array is empty.
    """
    if not array:
        raise ValueError("Input array is empty")

    return min(array)


# Example usage
def main():
    numbers = [12, 45, 7, 23, 56, 89, 34]
    try:
        min_value = find_min_value(numbers)
        print(f"The minimum value in the array is: {min_value}")
    except ValueError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
