## Exercise
Write a program that takes a student's grades as input, and prints their GPA based on the following scale: A (4.0), B (3.0), C (2.0), D (1.0), F (0.0).

## Reference Solution
```python
def calculate_gpa(grades):
    """
    Calculate the GPA based on the given grades.

    Args:
        grades (list): A list of grades (A, B, C, D, F)

    Returns:
        float: The calculated GPA
    """
    gpa_scale = {
        'A': 4.0,
        'B': 3.0,
        'C': 2.0,
        'D': 1.0,
        'F': 0.0
    }

    total_points = 0
    total_courses = 0

    for grade in grades:
        if grade.upper() in gpa_scale:
            total_points += gpa_scale[grade.upper()]
            total_courses += 1
        else:
            print(f"Invalid grade: {grade}. Skipping...")

    if total_courses == 0:
        return 0.0

    return total_points / total_courses


def main():
    num_courses = int(input("Enter the number of courses: "))
    grades = []

    for i in range(num_courses):
        grade = input(f"Enter grade for course {i+1}: ")
        grades.append(grade)

    gpa = calculate_gpa(grades)

    print(f"GPA: {gpa:.2f}")


if __name__ == "__main__":
    main()
```
