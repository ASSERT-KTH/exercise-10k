## Exercise
Given a dictionary {'name': 'Matthew', 'age': 25, 'city': 'Gold Coast'}, write a program to use the pop() method to remove the key-value pair 'age': 25.

## Reference Solution
```python
def remove_key_value_pair(dictionary, key):
    """
    Removes a key-value pair from a dictionary using the pop() method.

    Args:
        dictionary (dict): The dictionary from which to remove the key-value pair.
        key (str): The key of the key-value pair to remove.

    Returns:
        dict: The updated dictionary with the key-value pair removed.
    """
    try:
        # Use the pop() method to remove the key-value pair
        dictionary.pop(key)
        print(f"Key '{key}' removed successfully.")
    except KeyError:
        print(f"Key '{key}' not found in the dictionary.")

    return dictionary


# Example usage
if __name__ == "__main__":
    # Define the dictionary
    person = {'name': 'Matthew', 'age': 25, 'city': 'Gold Coast'}

    # Print the original dictionary
    print("Original Dictionary:")
    print(person)

    # Remove the key-value pair 'age': 25
    updated_person = remove_key_value_pair(person, 'age')

    # Print the updated dictionary
    print("\nUpdated Dictionary:")
    print(updated_person)
```
