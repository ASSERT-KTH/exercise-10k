## Exercise
Write a program that takes a string as input and returns the longest palindromic substring

## Reference Solution
```python
def longest_palindromic_substring(s: str) -> str:
    """
    Returns the longest palindromic substring in the given string.
    
    Args:
        s (str): The input string.
    
    Returns:
        str: The longest palindromic substring.
    """
    def expand_around_center(s: str, left: int, right: int) -> str:
        """
        Expands around the center of the palindrome to find the longest palindromic substring.
        
        Args:
            s (str): The input string.
            left (int): The left index of the center.
            right (int): The right index of the center.
        
        Returns:
            str: The longest palindromic substring centered at the given indices.
        """
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
        return s[left + 1:right]

    longest = ""
    for i in range(len(s)):
        # Odd length palindrome
        odd_length = expand_around_center(s, i, i)
        if len(odd_length) > len(longest):
            longest = odd_length
        
        # Even length palindrome
        even_length = expand_around_center(s, i, i + 1)
        if len(even_length) > len(longest):
            longest = even_length

    return longest


def main():
    # Example usage
    input_string = "babad"
    print(f"Longest palindromic substring in '{input_string}': {longest_palindromic_substring(input_string)}")


if __name__ == "__main__":
    main()
```
