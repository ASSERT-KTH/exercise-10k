## Exercise
Create a function to check if a given list contains any duplicate elements.

## Reference Solution
```python
def has_duplicates(input_list):
    """
    Checks if a given list contains any duplicate elements.

    Args:
    input_list (list): The list to check for duplicates.

    Returns:
    bool: True if the list contains duplicates, False otherwise.
    """
    return len(input_list) != len(set(input_list))

# Example usage:
if __name__ == "__main__":

    # Test case 2: List without duplicates
    list_without_duplicates = [1, 2, 3, 4, 5]
    print(has_duplicates(list_without_duplicates))  # Expected output: False

    # Test case 3: Empty list
    empty_list = []
    print(has_duplicates(empty_list))  # Expected output: False
```
