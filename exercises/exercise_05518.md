## Exercise
Write a program to solve the satisfiability problem using a backtracking algorithm.

## Reference Solution

def sat_solver(clauses):
    """
    Solves the SAT problem using a backtracking algorithm.

    Args:
    clauses (list): A list of clauses, where each clause is a list of literals.

    Returns:
    dict: A dictionary mapping variables to their assigned values, or None if no satisfying assignment exists.
    """
    # Create a dictionary to store the assigned values
    assignment = {}

    # Define a recursive helper function
    def backtrack(clauses, assignment):
        # If all clauses are satisfied, return the assignment
        if all(is_satisfied(clause, assignment) for clause in clauses):
            return assignment

        # Select an unassigned variable
        var = select_unassigned_variable(clauses, assignment)

        # If no unassigned variable exists, return None
        if var is None:
            return None

        # Try assigning true and false to the variable
        for value in [True, False]:
            # Create a copy of the assignment
            new_assignment = assignment.copy()

            # Assign the value to the variable
            new_assignment[var] = value

            # Recursively call the backtrack function
            result = backtrack(clauses, new_assignment)

            # If a satisfying assignment is found, return it
            if result is not None:
                return result

        # If no satisfying assignment is found, return None
        return None

    # Call the backtrack function
    return backtrack(clauses, assignment)


def is_satisfied(clause, assignment):
    """
    Checks if a clause is satisfied by an assignment.

    Args:
    clause (list): A list of literals.
    assignment (dict): A dictionary mapping variables to their assigned values.

    Returns:
    bool: True if the clause is satisfied, False otherwise.
    """
    for literal in clause:
        var, negated = literal
        value = assignment.get(var)

        if value is not None:
            if (not negated and value) or (negated and not value):
                return True

    return False


def select_unassigned_variable(clauses, assignment):
    """
    Selects an unassigned variable from the clauses.

    Args:
    clauses (list): A list of clauses.
    assignment (dict): A dictionary mapping variables to their assigned values.

    Returns:
    str: The selected variable, or None if no unassigned variable exists.
    """
    for clause in clauses:
        for var, _ in clause:
            if var not in assignment:
                return var

    return None


# Example usage
clauses = [
    [("A", False), ("B", True)],
    [("A", True), ("C", False)],
    [("B", False), ("C", True)],
]

assignment = sat_solver(clauses)

if assignment is not None:
    print("Satisfying assignment:")
    for var, value in assignment.items():
        print(f"{var} = {value}")
else:
    print("No satisfying assignment exists.")
