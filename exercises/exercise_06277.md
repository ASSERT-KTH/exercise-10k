## Exercise
Reduce a dictionary to a single value by applying a reduction operation

## Reference Solution
```python
import operator
import statistics

def reduce_dict(dictionary, reduction_operation):
    """
    Reduces a dictionary to a single value by applying a reduction operation.

    Args:
        dictionary (dict): The dictionary to reduce.
        reduction_operation (str): The reduction operation to apply. Can be 'sum', 'mul', 'max', 'min', or 'avg'.

    Returns:
        The reduced value.
    """
    values = dictionary.values()

    if reduction_operation == 'sum':
        return sum(values)
    elif reduction_operation == 'mul':
        result = 1
        for value in values:
            result *= value
        return result
    elif reduction_operation == 'max':
        return max(values)
    elif reduction_operation == 'min':
        return min(values)
    elif reduction_operation == 'avg':
        return statistics.mean(values)
    else:
        raise ValueError("Invalid reduction operation")

# Example usage
dictionary = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

print("Sum:", reduce_dict(dictionary, 'sum'))  # Output: 10
print("Product:", reduce_dict(dictionary, 'mul'))  # Output: 24
print("Maximum:", reduce_dict(dictionary, 'max'))  # Output: 4
print("Minimum:", reduce_dict(dictionary, 'min'))  # Output: 1
print("Average:", reduce_dict(dictionary, 'avg'))  # Output: 2.5
```

```python
import functools
import operator

def reduce_dict(dictionary, reduction_operation):
    """
    Reduces a dictionary to a single value by applying a reduction operation.

    Args:
        dictionary (dict): The dictionary to reduce.
        reduction_operation (str): The reduction operation to apply. Can be 'sum', 'mul', 'max', 'min', or 'avg'.

    Returns:
        The reduced value.
    """
    values = dictionary.values()

    if reduction_operation == 'sum':
        return functools.reduce(operator.add, values)
    elif reduction_operation == 'mul':
        return functools.reduce(operator.mul, values, 1)
    elif reduction_operation == 'max':
        return functools.reduce(max, values)
    elif reduction_operation == 'min':
        return functools.reduce(min, values)
    elif reduction_operation == 'avg':
        return sum(values) / len(values)
    else:
        raise ValueError("Invalid reduction operation")

# Example usage
dictionary = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

print("Maximum:", reduce_dict(dictionary, 'max'))  # Output: 4
```
