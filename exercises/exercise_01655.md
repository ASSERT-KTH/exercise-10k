## Exercise
Perform basic set operations, such as union and intersection, on two sets.

## Reference Solution
```python
def set_operations(set1, set2):
    """
    Performs basic set operations.

    Args:
        set1 (set): The first set.
        set2 (set): The second set.

    Returns:
        tuple: A tuple containing the union and intersection of the two sets.
    """
    # Calculate the union of the two sets
    union = set1.union(set2)
    
    # Calculate the intersection of the two sets
    intersection = set1.intersection(set2)

    return union, intersection


def main():
    # Example usage
    set1 = {1, 2, 3, 4, 5}
    set2 = {4, 5, 6, 7, 8}

    union, intersection = set_operations(set1, set2)

    print("Set 1:", set1)
    print("Set 2:", set2)
    print("Union:", union)
    print("Intersection:", intersection)


if __name__ == "__main__":
    main()
```
