## Exercise
Write a program that takes the distance traveled and amount of fuel consumed as input and outputs the fuel efficiency.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumed):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
    - distance (float): Distance traveled in kilometers.
    - fuel_consumed (float): Amount of fuel consumed in liters.

    Returns:
    - fuel_efficiency (float): Fuel efficiency in kilometers per liter.
    """
    if fuel_consumed <= 0:
        raise ValueError("Fuel consumed must be greater than zero.")

    fuel_efficiency = distance / fuel_consumed
    return fuel_efficiency


def main():
    # Get user input
    distance = float(input("Enter the distance traveled in kilometers: "))
    fuel_consumed = float(input("Enter the amount of fuel consumed in liters: "))

    # Calculate fuel efficiency
    try:
        fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumed)
        print(f"Fuel efficiency: {fuel_efficiency:.2f} km/L")
    except ValueError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
