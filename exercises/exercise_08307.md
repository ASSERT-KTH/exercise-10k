## Exercise
Write a function that takes an array as input and returns the reversed array. The function should not use the built-in reverse method.

## Reference Solution
```python
def reverse_array(arr):
    """
    Reverses the input array without using the built-in reverse method.

    Args:
    arr (list): The input array to be reversed.

    Returns:
    list: The reversed array.
    """
    left = 0
    right = len(arr) - 1

    while left < right:
        # Swap elements at left and right indices
        arr[left], arr[right] = arr[right], arr[left]
        # Move towards the center
        left += 1
        right -= 1

    return arr


# Example usage
def main():
    arr = [1, 2, 3, 4, 5]
    print("Original Array:", arr)
    print("Reversed Array:", reverse_array(arr))


if __name__ == "__main__":
    main()
```
