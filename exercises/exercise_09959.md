## Exercise
Write a program that checks if a given string is a valid variable name.

## Reference Solution
```python
import re

def is_valid_variable_name(name: str) -> bool:
    """
    Checks if a given string is a valid variable name.

    Args:
    name (str): The string to check.

    Returns:
    bool: True if the string is a valid variable name, False otherwise.
    """
    # Regular expression pattern for valid variable names
    pattern = r'^[a-zA-Z_][a-zA-Z0-9_]*$'
    
    # Use the regular expression to match the string
    return bool(re.match(pattern, name))

# Example usage
def main():
    variable_names = ["hello_world", "123abc", "__init__", "variable-name"]
    
    for name in variable_names:
        print(f"Is '{name}' a valid variable name? {is_valid_variable_name(name)}")

if __name__ == "__main__":
    main()
```
