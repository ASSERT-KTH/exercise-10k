## Exercise
Build a program that takes user information like name, age, and email, and stores it in a dictionary for later use.

## Reference Solution
```python
class UserInfoStorage:
    def __init__(self):
        self.users = {}

    def add_user(self, name, age, email):
        """Add a new user to the storage."""
        self.users[email] = {
            "name": name,
            "age": age,
            "email": email
        }
        print(f"User {name} added successfully!")

    def get_user(self, email):
        """Get user information by email."""
        if email in self.users:
            return self.users[email]
        else:
            print(f"User with email {email} not found.")
            return None

    def update_user(self, email, name=None, age=None):
        """Update user information."""
        if email in self.users:
            if name:
                self.users[email]["name"] = name
            if age:
                self.users[email]["age"] = age
            print(f"User {email} updated successfully!")
        else:
            print(f"User with email {email} not found.")

    def delete_user(self, email):
        """Delete a user from the storage."""
        if email in self.users:
            del self.users[email]
            print(f"User {email} deleted successfully!")
        else:
            print(f"User with email {email} not found.")

    def display_users(self):
        """Display all users in the storage."""
        for user in self.users.values():
            print(f"Name: {user['name']}, Age: {user['age']}, Email: {user['email']}")


def main():
    storage = UserInfoStorage()

    while True:
        print("\n1. Add User")
        print("2. Get User")
        print("3. Update User")
        print("4. Delete User")
        print("5. Display Users")
        print("6. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter name: ")
            age = int(input("Enter age: "))
            email = input("Enter email: ")
            storage.add_user(name, age, email)
        elif choice == "2":
            email = input("Enter email: ")
            user = storage.get_user(email)
            if user:
                print(f"Name: {user['name']}, Age: {user['age']}, Email: {user['email']}")
        elif choice == "3":
            email = input("Enter email: ")
            name = input("Enter new name (press enter to skip): ")
            age = input("Enter new age (press enter to skip): ")
            if age:
                age = int(age)
            else:
                age = None
            storage.update_user(email, name or None, age)
        elif choice == "4":
            email = input("Enter email: ")
            storage.delete_user(email)
        elif choice == "5":
            storage.display_users()
        elif choice == "6":
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
