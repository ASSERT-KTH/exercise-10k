## Exercise
Write a program to multiply two matrices.

## Reference Solution
```python
import numpy as np

def multiply_matrices(matrix1, matrix2):
    """
    Multiply two matrices.

    Args:
        matrix1 (list): The first matrix.
        matrix2 (list): The second matrix.

    Returns:
        list: The product of the two matrices.

    Raises:
        ValueError: If the number of columns in the first matrix does not match the number of rows in the second matrix.
    """
    # Get the dimensions of the matrices
    rows1, cols1 = len(matrix1), len(matrix1[0])
    rows2, cols2 = len(matrix2), len(matrix2[0])

    # Check if the matrices can be multiplied
    if cols1 != rows2:
        raise ValueError("The number of columns in the first matrix must match the number of rows in the second matrix.")

    # Create a result matrix filled with zeros
    result = [[0 for _ in range(cols2)] for _ in range(rows1)]

    # Perform the matrix multiplication
    for i in range(rows1):
        for j in range(cols2):
            for k in range(cols1):
                result[i][j] += matrix1[i][k] * matrix2[k][j]

    return result

def multiply_matrices_numpy(matrix1, matrix2):
    """
    Multiply two matrices using NumPy.

    Args:
        matrix1 (list): The first matrix.
        matrix2 (list): The second matrix.

    Returns:
        list: The product of the two matrices.
    """
    # Convert the matrices to NumPy arrays
    array1 = np.array(matrix1)
    array2 = np.array(matrix2)

    # Perform the matrix multiplication using NumPy
    result = np.dot(array1, array2)

    return result.tolist()

# Example usage
matrix1 = [[1, 2, 3], [4, 5, 6]]
matrix2 = [[7, 8], [9, 10], [11, 12]]

print("Matrix 1:")
for row in matrix1:
    print(row)

print("\nMatrix 2:")

print("\nResult (Manual):")
result_manual = multiply_matrices(matrix1, matrix2)
for row in result_manual:
    print(row)

print("\nResult (NumPy):")
result_numpy = multiply_matrices_numpy(matrix1, matrix2)
for row in result_numpy:
    print(row)
```
