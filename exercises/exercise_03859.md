## Exercise
Create a program that uses a for loop to solve the Cutting Stock problem, given a set of lengths of stock and a demand for each length.

## Reference Solution
```python
def cutting_stock(stock_length, demands):
    """
    Solves the Cutting Stock problem using the First-Fit Decreasing Height algorithm.

    Args:
    stock_length (int): The length of the stock.
    demands (dict): A dictionary where the keys are the lengths of stock and the values are the demands for each length.

    Returns:
    A list of lists, where each inner list represents a stock length and the pieces that it is cut into.
    """
    # Sort the demands in decreasing order of length
    sorted_demands = sorted(demands.items(), key=lambda x: x[0], reverse=True)

    # Initialize the result
    result = []

    # Iterate over the demands
    for length, demand in sorted_demands:
        # Initialize the number of stock lengths used for this demand
        num_stock_lengths = 0

        # Iterate over the demand
        for _ in range(demand):
            # Initialize a flag to indicate if the piece is placed
            placed = False

            # Iterate over the result
            for i, stock in enumerate(result):
                # Check if the piece can be placed in the current stock length
                if sum(stock) + length <= stock_length:
                    # Place the piece in the current stock length
                    result[i].append(length)
                    placed = True
                    break

            # If the piece is not placed, create a new stock length
            if not placed:
                result.append([length])
                num_stock_lengths += 1

    return result


# Example usage
stock_length = 10
demands = {3: 2, 5: 1, 2: 1}
result = cutting_stock(stock_length, demands)
for i, stock in enumerate(result):
    print(f"Stock {i+1}: {stock}")
```
