## Exercise
Create a function that groups a list of elements based on a common attribute.

## Reference Solution
```python
from itertools import groupby
from operator import attrgetter

class Element:
    def __init__(self, name, attribute):
        self.name = name
        self.attribute = attribute

def group_elements(elements, attribute_name):
    """
    Group a list of elements based on a common attribute.

    Args:
    - elements (list): A list of Element objects.
    - attribute_name (str): The name of the attribute to group by.

    Returns:
    - dict: A dictionary where the keys are the unique attribute values and the values are lists of corresponding elements.
    """
    # Sort the elements by the specified attribute
    sorted_elements = sorted(elements, key=attrgetter(attribute_name))

    # Group the sorted elements by the specified attribute
    grouped_elements = {}
    for key, group in groupby(sorted_elements, key=attrgetter(attribute_name)):
        grouped_elements[key] = list(group)

    return grouped_elements

# Example usage:
if __name__ == "__main__":
    # Create a list of Element objects
    elements = [
        Element("Element 1", "A"),
        Element("Element 2", "B"),
        Element("Element 3", "A"),
        Element("Element 4", "C"),
        Element("Element 5", "B"),
    ]

    # Group the elements by the 'attribute' attribute
    grouped_elements = group_elements(elements, "attribute")

    # Print the grouped elements
    for attribute_value, elements in grouped_elements.items():
        print(f"Attribute: {attribute_value}")
        for element in elements:
            print(f"  - {element.name}")
```
