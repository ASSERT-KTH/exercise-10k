## Exercise
Create a program that checks whether a given string is a palindrome using conditional statements.

## Reference Solution
```python
def is_palindrome(s):
    """
    Checks whether a given string is a palindrome.

    Args:
        s (str): The input string.

    Returns:
        bool: True if the string is a palindrome, False otherwise.
    """
    s = ''.join(c for c in s if c.isalnum()).lower()  # remove non-alphanumeric characters and convert to lowercase
    return s == s[::-1]  # check if the string is equal to its reverse


def main():
    # example usage:
    strings_to_check = ["radar", "hello", "A man, a plan, a canal: Panama"]

    for s in strings_to_check:
        if is_palindrome(s):
            print(f'"{s}" is a palindrome.')
        else:
            print(f'"{s}" is not a palindrome.')


if __name__ == "__main__":
    main()
```
