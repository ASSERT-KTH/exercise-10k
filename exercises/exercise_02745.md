## Exercise
Write a program that solves a quadratic equation given its coefficients.

## Reference Solution
```python
import math
import cmath

def solve_quadratic(a, b, c):
    """
    Solves a quadratic equation given its coefficients.

    Args:
        a (float): The coefficient of the squared term.
        b (float): The coefficient of the linear term.
        c (float): The constant term.

    Returns:
        tuple: A tuple containing the solutions to the equation.
    """

    # Calculate the discriminant
    d = b**2 - 4*a*c

    # If the discriminant is positive, there are two real solutions
    if d > 0:
        sol1 = (-b - math.sqrt(d)) / (2*a)
        sol2 = (-b + math.sqrt(d)) / (2*a)
        return (sol1, sol2)

    # If the discriminant is zero, there is one real solution
    elif d == 0:
        sol = -b / (2*a)
        return (sol, sol)

    # If the discriminant is negative, there are two complex solutions
    else:
        sol1 = (-b - cmath.sqrt(d)) / (2*a)
        sol2 = (-b + cmath.sqrt(d)) / (2*a)
        return (sol1, sol2)


def main():
    # Example usage:
    a = 1
    b = -3
    c = 2
    solutions = solve_quadratic(a, b, c)

    print(f"The solutions to the equation {a}x^2 + {b}x + {c} = 0 are:")
    print(f"x = {solutions[0]}")
    print(f"x = {solutions[1]}")


if __name__ == "__main__":
    main()
```
