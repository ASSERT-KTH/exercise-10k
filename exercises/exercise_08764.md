## Exercise
Create a function to find the most frequent character in a given string.

## Reference Solution
```python
from collections import Counter
import re

def most_frequent_char(s):
    """
    Find the most frequent character in a given string.

    Args:
        s (str): The input string.

    Returns:
        str: The most frequent character.
    """
    # Remove non-alphanumeric characters and convert to lower case
    cleaned_s = re.sub(r'[^a-zA-Z0-9]', '', s).lower()
    
    # Count the frequency of each character
    char_count = Counter(cleaned_s)
    
    # Find the most common character
    most_common_char = char_count.most_common(1)
    
    # Return the most frequent character
    if most_common_char:
        return most_common_char[0][0]
    else:
        return None

# Example usage
if __name__ == "__main__":
    s = "Hello, World!"
    print(most_frequent_char(s))
```
