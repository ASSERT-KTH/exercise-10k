## Exercise
Find the maximum sum subarray in an array of integers

## Reference Solution
```python
def max_sum_subarray(arr):
    """
    Find the maximum sum subarray in an array of integers.

    Args:
        arr (list): The input array of integers.

    Returns:
        tuple: A tuple containing the maximum sum and the subarray itself.
    """
    max_sum = float('-inf')
    current_sum = 0
    start = 0
    end = 0
    temp_start = 0

    for i in range(len(arr)):
        current_sum += arr[i]

        if current_sum > max_sum:
            max_sum = current_sum
            start = temp_start
            end = i

        if current_sum < 0:
            current_sum = 0
            temp_start = i + 1

    return max_sum, arr[start:end+1]

# Example usage
arr = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
max_sum, subarray = max_sum_subarray(arr)
print("Maximum sum:", max_sum)
print("Subarray:", subarray)
```
