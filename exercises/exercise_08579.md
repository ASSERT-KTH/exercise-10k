## Exercise
Write a function to create a copy of a dictionary

## Reference Solution
```python
def create_dict_copy(original_dict):
    """
    Creates a shallow copy of a given dictionary.

    Args:
        original_dict (dict): The dictionary to be copied.

    Returns:
        dict: A shallow copy of the original dictionary.
    """
    return original_dict.copy()

# Example usage:
if __name__ == "__main__":
    original = {"key1": "value1", "key2": "value2"}
    copied_dict = create_dict_copy(original)

    print("Original Dictionary:", original)
    print("Copied Dictionary:", copied_dict)

    # To demonstrate that the copy is indeed separate
    copied_dict["key3"] = "value3"
    print("After modification:")
    print("Original Dictionary:", original)
    print("Copied Dictionary:", copied_dict)
```
