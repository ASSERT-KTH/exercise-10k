Write a function that takes a dictionary as input and returns a view object that displays a list of all key-value pairs

def get_key_value_pairs(input_dict):
    """
    Returns a view object displaying a list of all key-value pairs in the input dictionary.

    Args:
        input_dict (dict): The dictionary to retrieve key-value pairs from.

    Returns:
        dict_items: A view object that displays a list of all key-value pairs.
    """
    return input_dict.items()

# Example usage
if __name__ == "__main__":
    example_dict = {"apple": 1, "banana": 2, "cherry": 3}
    key_value_pairs = get_key_value_pairs(example_dict)
