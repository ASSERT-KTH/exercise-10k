## Exercise
Create a function that calculates the tip for a meal based on the total cost and a percentage.

## Reference Solution
```python
def calculate_tip(total_cost, tip_percentage):
    """
    Calculate the tip for a meal based on the total cost and a percentage.

    Args:
        total_cost (float): The total cost of the meal.
        tip_percentage (float): The percentage of the total cost to use as the tip.

    Returns:
        float: The calculated tip amount.
    """
    # Validate the input values
    if total_cost < 0:
        raise ValueError("Total cost cannot be negative")
    if tip_percentage < 0 or tip_percentage > 100:
        raise ValueError("Tip percentage must be between 0 and 100")

    # Calculate the tip amount
    tip_amount = (tip_percentage / 100) * total_cost

    return tip_amount


def main():
    # Example usage:
    total_cost = 50.0  # Total cost of the meal
    tip_percentage = 15.0  # Tip percentage
    tip_amount = calculate_tip(total_cost, tip_percentage)

    print(f"Total Cost: ${total_cost:.2f}")
    print(f"Tip Percentage: {tip_percentage}%")
    print(f"Tip Amount: ${tip_amount:.2f}")
    print(f"Total Amount Due: ${total_cost + tip_amount:.2f}")


if __name__ == "__main__":
    main()
```
