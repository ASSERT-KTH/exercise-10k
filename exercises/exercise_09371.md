## Exercise
Write a program that checks for NaN and Infinity values.

## Reference Solution
```python
import math

def check_nan_and_infinity(numbers):
    """
    Checks for NaN and Infinity values in a list of numbers.

    Args:
        numbers (list): A list of numbers to check.

    Returns:
        dict: A dictionary with keys 'nan' and 'infinity' containing the indices of NaN and Infinity values respectively.
    """
    result = {'nan': [], 'infinity': []}

    for i, num in enumerate(numbers):
        if math.isnan(num):
            result['nan'].append(i)
        elif math.isinf(num):
            result['infinity'].append(i)

    return result

def main():
    # Example usage
    numbers = [1, 2, float('nan'), 4, float('inf'), -float('inf'), 7]
    result = check_nan_and_infinity(numbers)

    print("Indices of NaN values:", result['nan'])
    print("Indices of Infinity values:", result['infinity'])

if __name__ == "__main__":
    main()
```
