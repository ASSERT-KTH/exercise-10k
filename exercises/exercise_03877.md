## Exercise
Given two dictionaries with names of products as keys and their prices as values, write a program to find the difference of the two dictionaries and print the resulting dictionary.

## Reference Solution
```python
def dict_difference(dict1, dict2):
    """
    Calculate the difference between two dictionaries.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        dict: A dictionary containing the differences between the two input dictionaries.
    """
    diff_dict = {}

    # Find products in dict1 but not in dict2
    for product, price in dict1.items():
        if product not in dict2:
            diff_dict[product] = price

    # Find products in dict2 but not in dict1
    for product, price in dict2.items():
        if product not in dict1:
            diff_dict[product] = price

    # Find products with different prices
    for product, price in dict1.items():
        if product in dict2 and price != dict2[product]:
            diff_dict[product] = (price, dict2[product])

    return diff_dict


def main():
    # Example dictionaries
    products1 = {
        "Apple": 1.00,
        "Banana": 0.50,
        "Cherry": 2.00
    }

    products2 = {
        "Apple": 1.00,
        "Banana": 0.75,
        "Grapes": 3.00
    }

    # Calculate and print the difference
    difference = dict_difference(products1, products2)
    print("Difference:")
    for product, price in difference.items():
        if isinstance(price, tuple):
            print(f"{product}: {price[0]} (in dict1) vs {price[1]} (in dict2)")
        else:
            print(f"{product}: {price}")


if __name__ == "__main__":
    main()
```
