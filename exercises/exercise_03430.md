## Exercise
Write a function that takes a string as input and returns all permutations of the string

## Reference Solution
```python
import itertools

def get_permutations(s):
    """
    Generate all permutations of a given string.

    Args:
        s (str): The input string.

    Returns:
        list: A list of all permutations of the input string.
    """
    # Use itertools.permutations to generate all permutations
    perms = [''.join(p) for p in itertools.permutations(s)]
    return perms

def main():
    # Example usage
    input_str = "abc"
    perms = get_permutations(input_str)
    print("Permutations of '{}'".format(input_str))
    for perm in perms:
        print(perm)

if __name__ == "__main__":
    main()
```

```python
def get_permutations_recursive(s):
    """
    Generate all permutations of a given string using recursion.

    Args:
        s (str): The input string.

    Returns:
        list: A list of all permutations of the input string.
    """
    # Base case: if the string is empty or has only one character
    if len(s) <= 1:
        return [s]

    # Initialize an empty list to store the permutations
    perms = []

    # Iterate over each character in the string
    for i, char in enumerate(s):
        # Remove the current character from the string
        remaining_str = s[:i] + s[i+1:]

        # Generate permutations of the remaining string
        for perm in get_permutations_recursive(remaining_str):
            # Add the current character to the beginning of each permutation
            perms.append(char + perm)

    return perms

def main_recursive():
    # Example usage
    input_str = "abc"
    perms = get_permutations_recursive(input_str)
    print("Permutations of '{}'".format(input_str))
    for perm in perms:
        print(perm)

if __name__ == "__main__":
    main_recursive()
```
