## Exercise
Write a program that uses a for loop to encrypt a given string using a Caesar cipher with a given shift.

## Reference Solution
```python
def caesar_cipher_encrypt(text, shift):
    """
    Encrypts a given string using a Caesar cipher with a given shift.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The shift value for the Caesar cipher.

    Returns:
        str: The encrypted string.
    """
    encrypted_text = ""
    for char in text:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            encrypted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encrypted_text += encrypted_char
        else:
            encrypted_text += char
    return encrypted_text

def main():
    text = input("Enter a string to encrypt: ")
    shift = int(input("Enter a shift value: "))
    encrypted_text = caesar_cipher_encrypt(text, shift)
    print("Encrypted text:", encrypted_text)

if __name__ == "__main__":
    main()
```
