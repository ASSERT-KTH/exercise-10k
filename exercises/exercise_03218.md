## Exercise
Write a program that takes the principal, rate, and time as input and prints the compound interest.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time):
    """
    Calculate the compound interest.

    Args:
        principal (float): The initial amount of money
        rate (float): The interest rate (in decimal form, e.g., 4% is 0.04)
        time (int): The time period (in years)

    Returns:
        float: The compound interest
    """
    ci = principal * (1 + rate) ** time - principal
    return ci

def main():
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))

    ci = calculate_compound_interest(principal, rate, time)

    print(f"The compound interest is: {ci:.2f}")

if __name__ == "__main__":
    main()
```
