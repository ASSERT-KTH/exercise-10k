## Exercise
Create a program that simulates a phonebook using a dictionary. The program should allow users to add, remove, and search for contacts. The contacts should be stored in a dictionary with the name as the key and the phone number as the value.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        """Initialize an empty phonebook."""
        self.phonebook = {}

    def add_contact(self, name, phone_number):
        """Add a contact to the phonebook.

        Args:
            name (str): The name of the contact.
            phone_number (str): The phone number of the contact.
        """
        if name in self.phonebook:
            print(f"Contact '{name}' already exists. Updating phone number.")
        self.phonebook[name] = phone_number
        print(f"Contact '{name}' added successfully.")

    def remove_contact(self, name):
        """Remove a contact from the phonebook.

        Args:
            name (str): The name of the contact to be removed.
        """
        if name not in self.phonebook:
            print(f"Contact '{name}' does not exist.")
            return
        del self.phonebook[name]
        print(f"Contact '{name}' removed successfully.")

    def search_contact(self, name):
        """Search for a contact in the phonebook.

        Args:
            name (str): The name of the contact to be searched.
        """
        if name not in self.phonebook:
            print(f"Contact '{name}' does not exist.")
            return
        print(f"Contact '{name}' found: {self.phonebook[name]}")

    def display_contacts(self):
        """Display all contacts in the phonebook."""
        if not self.phonebook:
            print("Phonebook is empty.")
            return
        print("Contacts:")
        for name, phone_number in self.phonebook.items():
            print(f"{name}: {phone_number}")


def main():
    phonebook = Phonebook()
    while True:
        print("\nPhonebook Menu:")
        print("1. Add contact")
        print("2. Remove contact")
        print("3. Search contact")
        print("4. Display contacts")
        print("5. Quit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter contact name to remove: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter contact name to search: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
