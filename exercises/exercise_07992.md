## Exercise
Write a program that calculates the difference of two lists of numbers.

## Reference Solution
```python
def calculate_list_difference(list1, list2):
    """
    Calculate the difference between two lists of numbers.

    Args:
        list1 (list): The first list of numbers.
        list2 (list): The second list of numbers.

    Returns:
        list: A list containing the absolute differences between corresponding elements in the two input lists.

    Raises:
        ValueError: If the two input lists have different lengths.
    """
    if len(list1) != len(list2):
        raise ValueError("The two input lists must have the same length")

    return [abs(a - b) for a, b in zip(list1, list2)]


def main():
    # Example usage
    list1 = [1, 2, 3, 4, 5]
    list2 = [4, 5, 6, 7, 8]

    try:
        difference = calculate_list_difference(list1, list2)
        print("List 1:", list1)
        print("List 2:", list2)
        print("Difference:", difference)
    except ValueError as e:
        print("Error:", e)


if __name__ == "__main__":
    main()
```
