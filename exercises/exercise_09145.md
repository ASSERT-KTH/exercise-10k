## Exercise
Create two integer variables and perform basic arithmetic operations (addition, subtraction, multiplication, division) on them, then print out the results.

## Reference Solution

def perform_arithmetic_operations():
    # Define two integer variables
    num1 = 10
    num2 = 3

    # Perform basic arithmetic operations
    addition_result = num1 + num2
    subtraction_result = num1 - num2
    multiplication_result = num1 * num2
    division_result = num1 / num2

    # Print the results
    print(f"Addition: {num1} + {num2} = {addition_result}")
    print(f"Subtraction: {num1} - {num2} = {subtraction_result}")
    print(f"Multiplication: {num1} * {num2} = {multiplication_result}")
    print(f"Division: {num1} / {num2} = {division_result}")

# Example usage:
if __name__ == "__main__":
    perform_arithmetic_operations()
