## Exercise
Write a program that generates the Koch curve using a while loop.

import turtle

def koch_curve(turtle, length, depth):
    """
    Recursively generates the Koch curve.

    Args:
        turtle (Turtle): The turtle object to draw with.
        length (int): The length of the line segment.
        depth (int): The recursion depth.
    """
    if depth == 0:
        # Base case: draw a line segment
        turtle.forward(length)
    else:
        # Recursive case: divide the line segment into 4 parts
        koch_curve(turtle, length / 3, depth - 1)
        turtle.left(60)
        koch_curve(turtle, length / 3, depth - 1)
        turtle.right(120)
        koch_curve(turtle, length / 3, depth - 1)
        turtle.left(60)
        koch_curve(turtle, length / 3, depth - 1)

def draw_koch_snowflake(turtle, length, depth):
    """
    Draws a Koch snowflake.

    Args:
        turtle (Turtle): The turtle object to draw with.
        length (int): The length of the line segment.
        depth (int): The recursion depth.
    """
    for _ in range(3):
        koch_curve(turtle, length, depth)
        turtle.right(120)

def main():
    window = turtle.Screen()
    window.setup(800, 800)

    my_turtle = turtle.Turtle()
    my_turtle.speed(0)

    length = 300
    depth = 4

    draw_koch_snowflake(my_turtle, length, depth)

    window.mainloop()

if __name__ == "__main__":
    main()

import turtle

class KochCurve:
    def __init__(self, turtle, length, depth):
        self.turtle = turtle
        self.length = length
        self.depth = depth
        self.stack = [(0, length, depth)]

    def generate(self):
        while self.stack:
            start, length, depth = self.stack.pop()
            if depth == 0:
                self.turtle.forward(length)
            else:
                segment_length = length / 3
                self.stack.append((start + 2 * segment_length, segment_length, depth - 1))
                self.turtle.left(60)
                self.stack.append((start + segment_length, segment_length, depth - 1))
                self.turtle.right(120)
                self.stack.append((start + segment_length, segment_length, depth - 1))
                self.turtle.left(60)
                self.stack.append((start, segment_length, depth - 1))
                self.turtle.forward(segment_length)

def draw_koch_snowflake(turtle, length, depth):
    for _ in range(3):
        koch_curve = KochCurve(turtle, length, depth)
        koch_curve.generate()
        turtle.right(120)

def main():
    window = turtle.Screen()
    window.setup(800, 800)

    my_turtle = turtle.Turtle()
    my_turtle.speed(0)

    length = 300
    depth = 4

    draw_koch_snowflake(my_turtle, length, depth)

    window.mainloop()

if __name__ == "__main__":
    main()
