## Exercise
Create a program that generates a number sequence (e.g., Fibonacci, arithmetic) based on user input. The program should ask for the sequence type, the starting number, and the length of the sequence, and then print out the generated sequence.

## Reference Solution
```python
def generate_fibonacci_sequence(starting_number, length):
    """
    Generate a Fibonacci sequence.

    Args:
        starting_number (int): The first number in the sequence.
        length (int): The number of terms in the sequence.

    Returns:
        list: A list of integers representing the Fibonacci sequence.
    """
    sequence = [starting_number, starting_number + 1]
    while len(sequence) < length:
        sequence.append(sequence[-1] + sequence[-2])
    return sequence[:length]

def generate_arithmetic_sequence(starting_number, length, common_difference):
    """
    Generate an arithmetic sequence.

    Args:
        starting_number (int): The first number in the sequence.
        length (int): The number of terms in the sequence.
        common_difference (int): The difference between each term.

    Returns:
        list: A list of integers representing the arithmetic sequence.
    """
    sequence = []
    for i in range(length):
        sequence.append(starting_number + i * common_difference)
    return sequence

def main():
    print("Sequence Generator")
    print("------------------")

    sequence_type = input("Enter the sequence type (fibonacci/arithmetic): ")
    starting_number = int(input("Enter the starting number: "))

    if sequence_type.lower() == "fibonacci":
        length = int(input("Enter the length of the sequence: "))
        sequence = generate_fibonacci_sequence(starting_number, length)
    elif sequence_type.lower() == "arithmetic":
        common_difference = int(input("Enter the common difference: "))
        length = int(input("Enter the length of the sequence: "))
        sequence = generate_arithmetic_sequence(starting_number, length, common_difference)
    else:
        print("Invalid sequence type. Please try again.")
        return

    print("Generated sequence:")
    print(sequence)

if __name__ == "__main__":
    main()
```
