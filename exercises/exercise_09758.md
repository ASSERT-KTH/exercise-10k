## Exercise
Write a program to calculate the cost of goods after tax

## Reference Solution
```python
def calculate_cost_after_tax(cost_before_tax, tax_rate):
    """
    Calculates the cost of goods after tax.

    Args:
        cost_before_tax (float): The cost of goods before tax.
        tax_rate (float): The tax rate as a decimal (e.g., 8% = 0.08).

    Returns:
        float: The cost of goods after tax.
    """
    tax_amount = cost_before_tax * tax_rate
    cost_after_tax = cost_before_tax + tax_amount
    return cost_after_tax

def main():
    # Get user input
    cost_before_tax = float(input("Enter the cost of goods before tax: $"))
    tax_rate = float(input("Enter the tax rate as a percentage (e.g., 8 for 8%): "))
    tax_rate = tax_rate / 100  # Convert tax rate to decimal

    # Calculate cost after tax
    cost_after_tax = calculate_cost_after_tax(cost_before_tax, tax_rate)

    # Print result
    print(f"Cost before tax: ${cost_before_tax:.2f}")
    print(f"Tax rate: {tax_rate*100}%")
    print(f"Tax amount: ${cost_before_tax * tax_rate:.2f}")
    print(f"Cost after tax: ${cost_after_tax:.2f}")

if __name__ == "__main__":
    main()
```
