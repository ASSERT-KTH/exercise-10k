## Exercise
Create a program to simulate a coin flip using conditional statements to determine the outcome

## Reference Solution
```python
import random

def coin_flip():
    # Generate a random number (0 or 1) to simulate a coin flip
    flip_result = random.randint(0, 1)
    
    # Use conditional statements to determine the outcome
    if flip_result == 0:
        return "Heads"
    else:
        return "Tails"

def main():
    print("Welcome to the Coin Flip Simulator!")
    play_again = "y"
    
    while play_again.lower() == "y":
        result = coin_flip()
        print(f"The coin landed on: {result}")
        play_again = input("Would you like to flip again? (y/n): ")
    
    print("Thanks for playing!")

if __name__ == "__main__":
    main()
```
