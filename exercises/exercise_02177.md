## Exercise
Ask the user for the total distance traveled and the amount of gas consumed, and print the gas mileage.

## Reference Solution
```python
def calculate_gas_mileage(distance, gas_consumed):
    """
    Calculate gas mileage.

    Args:
        distance (float): Total distance traveled in miles.
        gas_consumed (float): Total amount of gas consumed in gallons.

    Returns:
        float: Gas mileage in miles per gallon.
    """
    if gas_consumed <= 0:
        raise ValueError("Gas consumed must be greater than zero.")

    return distance / gas_consumed


def main():
    # Get user input
    distance = float(input("Enter the total distance traveled in miles: "))
    gas_consumed = float(input("Enter the amount of gas consumed in gallons: "))

    try:
        # Calculate gas mileage
        gas_mileage = calculate_gas_mileage(distance, gas_consumed)
        print(f"Your gas mileage is: {gas_mileage:.2f} miles per gallon")
    except ValueError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
