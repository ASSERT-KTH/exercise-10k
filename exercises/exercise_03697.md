## Exercise
Write a program to calculate the total travel expense given the cost of transportation, accommodation, and food.

## Reference Solution
```python
def calculate_travel_expense(transportation_cost, accommodation_cost, food_cost):
    """
    Calculates the total travel expense.

    Args:
        transportation_cost (float): The cost of transportation.
        accommodation_cost (float): The cost of accommodation.
        food_cost (float): The cost of food.

    Returns:
        float: The total travel expense.
    """
    total_expense = transportation_cost + accommodation_cost + food_cost
    return total_expense


def main():
    print("Travel Expense Calculator")
    print("-------------------------")

    # Get user input
    transportation_cost = float(input("Enter the cost of transportation: $"))
    accommodation_cost = float(input("Enter the cost of accommodation: $"))
    food_cost = float(input("Enter the cost of food: $"))

    # Calculate total travel expense
    total_expense = calculate_travel_expense(transportation_cost, accommodation_cost, food_cost)

    # Print the result
    print(f"Total travel expense: ${total_expense:.2f}")


if __name__ == "__main__":
    main()
```
