## Exercise
Write a function that uses version control to manage its changes and collaborate with others

## Reference Solution
```python
import os
import subprocess

def collaborate_with_git(repo_name, file_name, file_content, remote_url):
    """
    Create a new Git repository, add files, commit changes, and push them to a remote repository.

    Args:
    - repo_name (str): The name of the Git repository.
    - file_name (str): The name of the file to be added to the repository.
    - file_content (str): The content of the file to be added to the repository.
    - remote_url (str): The URL of the remote Git repository.

    Returns:
    - None
    """

    # Create a new directory for the Git repository
    os.mkdir(repo_name)

    # Change into the new directory
    os.chdir(repo_name)

    # Initialize a new Git repository
    subprocess.run(["git", "init"])

    # Create a new file and write the content to it
    with open(file_name, "w") as file:
        file.write(file_content)

    # Add the file to the Git repository
    subprocess.run(["git", "add", file_name])

    # Commit the changes with a meaningful message
    subprocess.run(["git", "commit", "-m", f"Added {file_name} to the repository"])

    # Add the remote repository
    subprocess.run(["git", "remote", "add", "origin", remote_url])

    # Push the changes to the remote repository
    subprocess.run(["git", "push", "-u", "origin", "master"])

# Example usage
if __name__ == "__main__":
    repo_name = "my-repo"
    file_name = "example.txt"
    file_content = "This is an example file."
    remote_url = "https://github.com/username/my-repo.git"

    collaborate_with_git(repo_name, file_name, file_content, remote_url)
```
