## Exercise
Write a program that checks if three given lengths can form a valid triangle.

## Reference Solution
```python
def is_valid_triangle(a: float, b: float, c: float) -> bool:
    """
    Checks if three given lengths can form a valid triangle.

    Args:
        a (float): The length of the first side.
        b (float): The length of the second side.
        c (float): The length of the third side.

    Returns:
        bool: True if the lengths can form a valid triangle, False otherwise.
    """
    # Check if all sides are positive
    if a <= 0 or b <= 0 or c <= 0:
        return False

    # Check if the sum of the lengths of any two sides is greater than the length of the third side
    return (a + b > c) and (a + c > b) and (b + c > a)


def main():
    # Example usage
    a = 3.0
    b = 4.0
    c = 5.0

    if is_valid_triangle(a, b, c):
        print(f"The lengths {a}, {b}, and {c} can form a valid triangle.")
    else:
        print(f"The lengths {a}, {b}, and {c} cannot form a valid triangle.")


if __name__ == "__main__":
    main()
```
