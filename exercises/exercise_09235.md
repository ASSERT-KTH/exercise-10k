## Exercise
Write a program that checks the data type of a variable and prints a message accordingly.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a variable and prints a message accordingly.

    Args:
        var: The variable to check.
    """
    if isinstance(var, int):
        print(f"The variable is of type integer with value {var}.")
    elif isinstance(var, float):
        print(f"The variable is of type float with value {var}.")
    elif isinstance(var, str):
        print(f"The variable is of type string with value '{var}'.")
    elif isinstance(var, list):
        print(f"The variable is of type list with value {var}.")
    elif isinstance(var, dict):
        print(f"The variable is of type dictionary with value {var}.")
    elif isinstance(var, tuple):
        print(f"The variable is of type tuple with value {var}.")
    elif isinstance(var, set):
        print(f"The variable is of type set with value {var}.")
    elif isinstance(var, bool):
        print(f"The variable is of type boolean with value {var}.")
    else:
        print(f"The variable is of type {type(var).__name__} with value {var}.")

# Example usage:
if __name__ == "__main__":
    # Test with different data types
    check_data_type(10)  # integer
    check_data_type(10.5)  # float
    check_data_type("Hello")  # string
    check_data_type([1, 2, 3])  # list
    check_data_type({"a": 1, "b": 2})  # dictionary
    check_data_type((1, 2, 3))  # tuple
    check_data_type({1, 2, 3})  # set
    check_data_type(True)  # boolean
```
