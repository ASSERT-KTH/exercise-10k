## Exercise
Develop a program that calculates the average grade of a student given their scores on a series of assignments.

## Reference Solution
```python
def calculate_average_grade(scores):
    """
    Calculate the average grade given a list of scores.

    Args:
        scores (list): A list of scores.

    Returns:
        float: The average grade.
    """
    total_score = sum(scores)
    average_grade = total_score / len(scores)
    return average_grade

def main():
    # Get the number of assignments from the user
    num_assignments = int(input("Enter the number of assignments: "))

    # Initialize an empty list to store the scores
    scores = []

    # Get the scores for each assignment from the user
    for i in range(num_assignments):
        score = float(input(f"Enter the score for assignment {i + 1}: "))
        scores.append(score)

    # Calculate the average grade
    average_grade = calculate_average_grade(scores)

    # Print the average grade
    print(f"The average grade is: {average_grade:.2f}")

if __name__ == "__main__":
    main()
```
