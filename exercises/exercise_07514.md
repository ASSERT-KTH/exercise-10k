Create a Quiz program using variables and conditional statements.

```python
# Import the random module to shuffle questions
import random

# Define a dictionary with questions, options, and answers
quiz_questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "Answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "Answer": "C"
    },
    "Who painted the Starry Night?": {
        "A": "Leonardo da Vinci",
        "B": "Vincent van Gogh",
        "C": "Pablo Picasso",
        "D": "Claude Monet",
        "Answer": "B"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "Answer": "B"
    },
    "What is the largest mammal on Earth?": {
        "A": "Blue whale",
        "B": "African elephant",
        "C": "Hippopotamus",
        "D": "Giraffe",
        "Answer": "A"
    }
}

def quiz_program():
    # Initialize score
    score = 0

    # Shuffle questions
    questions = list(quiz_questions.items())
    random.shuffle(questions)

    # Iterate over questions
    for question, options in questions:
        print(f"\nQuestion: {question}")
        for option, value in options.items():
            if option != "Answer":
                print(f"{option}: {value}")

        # Get user response
        user_response = input("Enter your response (A, B, C, D): ")

        # Evaluate user response
        if user_response.upper() == options["Answer"]:
            print("Correct!")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer is {options['Answer']}.")

    # Print final score
    print(f"\nQuiz complete! Your final score is {score} out of {len(questions)}.")

# Run the quiz program
if __name__ == "__main__":
    quiz_program()
```
