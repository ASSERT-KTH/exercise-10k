## Exercise
Write a function that checks if a string contains a given substring

## Reference Solution
```python
def check_substring(main_string: str, sub_string: str) -> bool:
    """
    Checks if a substring is present in a main string.

    Args:
    main_string (str): The main string to search in.
    sub_string (str): The substring to search for.

    Returns:
    bool: True if the substring is found, False otherwise.
    """
    return sub_string.lower() in main_string.lower()

# Example usage
if __name__ == "__main__":
    main_str = "Hello, World!"
    sub_str = "World"
    result = check_substring(main_str, sub_str)
    print(f"'{sub_str}' is in '{main_str}': {result}")

    main_str = "Hello, World!"
    sub_str = "Universe"
    result = check_substring(main_str, sub_str)
    print(f"'{sub_str}' is in '{main_str}': {result}")
```
