## Exercise
Given a list of integers, write a program to insert a new element at a specific position.

## Reference Solution
```python
def insert_at_position(lst, element, position):
    """
    Inserts a new element at a specific position in a list.

    Args:
        lst (list): The input list.
        element: The element to be inserted.
        position (int): The position where the element should be inserted.

    Returns:
        list: The modified list with the element inserted.

    Raises:
        IndexError: If the position is out of range.
    """
    if position < 0 or position > len(lst):
        raise IndexError("Position out of range")
    
    return lst[:position] + [element] + lst[position:]

# Example usage
if __name__ == "__main__":
    numbers = [1, 2, 3, 5, 6]
    new_number = 4
    position = 3

    print("Original List:", numbers)
    print("New Element:", new_number)
    print("Position:", position)

    updated_list = insert_at_position(numbers, new_number, position)
    print("Updated List:", updated_list)
```
