## Exercise
Create a function that takes a string and a shift value as input and returns the encrypted string using the Caesar Cipher algorithm.

## Reference Solution
```python
def caesar_cipher_encrypt(text: str, shift: int) -> str:
    """
    Encrypts a string using the Caesar Cipher algorithm.

    Args:
    text (str): The input string to be encrypted.
    shift (int): The shift value for the Caesar Cipher.

    Returns:
    str: The encrypted string.
    """
    encrypted_text = ""
    for char in text:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            encrypted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encrypted_text += encrypted_char
        else:
            encrypted_text += char
    return encrypted_text

# Example usage
def main():
    text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_cipher_encrypt(text, shift)
    print(f"Original Text: {text}")
    print(f"Shift Value: {shift}")
    print(f"Encrypted Text: {encrypted_text}")

if __name__ == "__main__":
    main()
```
