## Exercise
Write a function that merges two dictionaries into one

## Reference Solution

```python
def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:
        dict: A new dictionary containing all key-value pairs from both input dictionaries.
    """
    # Create a copy of the first dictionary to avoid modifying it in place
    merged_dict = dict1.copy()
    
    # Update the copied dictionary with the key-value pairs from the second dictionary
    merged_dict.update(dict2)
    
    return merged_dict

# Example usage:
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2}
    dict2 = {"b": 3, "c": 4}

    merged = merge_dictionaries(dict1, dict2)
    print("Merged Dictionary:", merged)
```

```python
def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:
        dict: A new dictionary containing all key-value pairs from both input dictionaries.
    """

# Example usage:
if __name__ == "__main__":

    dict2 = {"b": 3, "c": 4}
```
