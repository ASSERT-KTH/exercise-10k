## Exercise
Write a program to demonstrate explicit type conversion.

## Reference Solution
```python
# Define a function to demonstrate explicit type conversion
def explicit_type_conversion():
    # Integer to Float
    int_num = 10
    float_num = float(int_num)
    print(f"Integer {int_num} to Float: {float_num}")

    # Float to Integer
    float_num = 10.5
    int_num = int(float_num)
    print(f"Float {float_num} to Integer: {int_num}")

    # String to Integer
    str_num = "10"
    int_num = int(str_num)
    print(f"String '{str_num}' to Integer: {int_num}")

    # String to Float
    str_num = "10.5"
    float_num = float(str_num)
    print(f"String '{str_num}' to Float: {float_num}")

    # Integer to String
    int_num = 10
    str_num = str(int_num)
    print(f"Integer {int_num} to String: {str_num}")

    # Float to String
    float_num = 10.5
    str_num = str(float_num)
    print(f"Float {float_num} to String: {str_num}")

    # Boolean to Integer
    bool_val = True
    int_val = int(bool_val)
    print(f"Boolean {bool_val} to Integer: {int_val}")

    # Boolean to String
    bool_val = False
    str_val = str(bool_val)
    print(f"Boolean {bool_val} to String: {str_val}")

# Call the function to demonstrate explicit type conversion
if __name__ == "__main__":
    explicit_type_conversion()
```
