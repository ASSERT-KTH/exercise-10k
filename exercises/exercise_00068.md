## Exercise
Write a function to implement a Caesar cipher to encrypt and decrypt strings

## Reference Solution

def caesar_cipher(text: str, shift: int, mode: str) -> str:
    """
    Encrypts or decrypts a string using the Caesar cipher.

    Args:
        text (str): The input string to be encrypted or decrypted.
        shift (int): The number of positions to shift the alphabet.
        mode (str): 'encrypt' or 'decrypt' mode.

    Returns:
        str: The encrypted or decrypted string.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in text:
        if char.isalpha():
            # Find the position of the character in the alphabet
            index = alphabet.index(char.lower())

            # Shift the position based on the mode
            if mode == 'encrypt':
                new_index = (index + shift) % 26
            elif mode == 'decrypt':
                new_index = (index - shift) % 26
            else:
                raise ValueError("Invalid mode. Use 'encrypt' or 'decrypt'.")

            # Replace the character with the shifted one, preserving case
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            # Non-alphabetical characters are not shifted
            result += char

    return result

def main():
    # Example usage
    text = "Hello, World!"
    shift = 3

    encrypted = caesar_cipher(text, shift, 'encrypt')
    decrypted = caesar_cipher(encrypted, shift, 'decrypt')

    print(f"Original: {text}")
    print(f"Encrypted: {encrypted}")
    print(f"Decrypted: {decrypted}")

if __name__ == "__main__":
    main()
