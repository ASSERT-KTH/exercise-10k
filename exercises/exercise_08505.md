## Exercise
Create a program that stores student records in a dictionary. Each student has a name, age, and grade. The program should allow the user to add, remove, and update student records. It should also be able to display all student records.

## Reference Solution
```python
class StudentRecordSystem:
    def __init__(self):
        self.student_records = {}

    def add_student(self, name, age, grade):
        """Add a new student record"""
        if name in self.student_records:
            print(f"A student with the name '{name}' already exists.")
            return

        self.student_records[name] = {
            "age": age,
            "grade": grade
        }
        print(f"Student '{name}' added successfully.")

    def remove_student(self, name):
        """Remove a student record"""
        if name not in self.student_records:
            print(f"No student with the name '{name}' found.")
            return

        del self.student_records[name]
        print(f"Student '{name}' removed successfully.")

    def update_student(self, name, age=None, grade=None):
        """Update a student record"""
        if name not in self.student_records:
            print(f"No student with the name '{name}' found.")
            return

        if age:
            self.student_records[name]["age"] = age
        if grade:
            self.student_records[name]["grade"] = grade
        print(f"Student '{name}' updated successfully.")

    def display_students(self):
        """Display all student records"""
        if not self.student_records:
            print("No student records found.")
            return

        print("Student Records:")
        for name, record in self.student_records.items():
            print(f"Name: {name}, Age: {record['age']}, Grade: {record['grade']}")


def main():
    student_system = StudentRecordSystem()

    while True:
        print("\nStudent Record System Menu:")
        print("1. Add Student")
        print("2. Remove Student")
        print("3. Update Student")
        print("4. Display Students")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter student name: ")
            age = int(input("Enter student age: "))
            grade = input("Enter student grade: ")
            student_system.add_student(name, age, grade)
        elif choice == "2":
            name = input("Enter student name: ")
            student_system.remove_student(name)
        elif choice == "3":
            name = input("Enter student name: ")
            age = input("Enter new age (optional): ")
            grade = input("Enter new grade (optional): ")
            age = int(age) if age else None
            student_system.update_student(name, age, grade)
        elif choice == "4":
            student_system.display_students()
        elif choice == "5":
            print("Exiting the system. Goodbye!")
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
