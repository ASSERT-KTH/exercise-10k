## Exercise
Create a program that determines a car's insurance rate based on its make, model, and year. The program should use if/else statements to determine the insurance rate.

## Reference Solution
```python
def calculate_insurance_rate(make, model, year):
    """
    Calculates the insurance rate of a car based on its make, model, and year.

    Args:
        make (str): The make of the car.
        model (str): The model of the car.
        year (int): The year of the car.

    Returns:
        float: The insurance rate of the car.
    """
    insurance_rate = 0

    if make == "Toyota":
        if model == "Corolla":
            if year >= 2015:
                insurance_rate = 1200
            else:
                insurance_rate = 1000
        elif model == "Camry":
            if year >= 2015:
                insurance_rate = 1500
            else:
                insurance_rate = 1200
        else:
            insurance_rate = 1000
    elif make == "Honda":
        if model == "Civic":
            if year >= 2015:
                insurance_rate = 1200
            else:
                insurance_rate = 1000
        elif model == "Accord":
            if year >= 2015:
                insurance_rate = 1500
            else:
                insurance_rate = 1200
        else:
            insurance_rate = 1000
    elif make == "Ford":
        if model == "Fusion":
            if year >= 2015:
                insurance_rate = 1200
            else:
                insurance_rate = 1000
        elif model == "Mustang":
            if year >= 2015:
                insurance_rate = 2000
            else:
                insurance_rate = 1500
        else:
            insurance_rate = 1000
    else:
        insurance_rate = 1500

    return insurance_rate


def main():
    print("Car Insurance Rate Calculator")
    print("--------------------------------")

    make = input("Enter the make of the car: ")
    model = input("Enter the model of the car: ")
    year = int(input("Enter the year of the car: "))

    insurance_rate = calculate_insurance_rate(make, model, year)

    print(f"The insurance rate for your {year} {make} {model} is ${insurance_rate} per year.")


if __name__ == "__main__":
    main()
```
