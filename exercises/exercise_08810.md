## Exercise
Write a program that takes the coefficients of a polynomial as input and returns the sum and product of its roots.

## Reference Solution
```python
use std::io;

fn main() {
    let mut input = String::new();

    io::stdin().read_line(&mut input).expect("Error reading input");

    let coefficients: Vec<i64> = input
        .trim()
        .split_whitespace()
        .map(|x| x.parse().expect("Error parsing input"))
        .collect();

    let sum_of_roots = -coefficients[1] / coefficients[0];
    let product_of_roots = (-1).pow((coefficients.len() - 1) as u32) * coefficients[coefficients.len() - 1] / coefficients[0];

    println!("Sum of roots: {}", sum_of_roots);
    println!("Product of roots: {}", product_of_roots);
}
```
