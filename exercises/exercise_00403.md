## Exercise
Write a program to check if an array is sorted in ascending order.

## Reference Solution
```python
def is_sorted_ascending(arr):
    """
    Check if an array is sorted in ascending order.

    Args:
        arr (list): The input array.

    Returns:
        bool: True if the array is sorted in ascending order, False otherwise.
    """
    return all(arr[i] <= arr[i + 1] for i in range(len(arr) - 1))

# Example usage
if __name__ == "__main__":
    # Test case 1: Sorted array
    arr1 = [1, 2, 3, 4, 5]
    print(f"Array {arr1} is sorted in ascending order: {is_sorted_ascending(arr1)}")

    # Test case 2: Unsorted array
    arr2 = [5, 2, 3, 4, 1]
    print(f"Array {arr2} is sorted in ascending order: {is_sorted_ascending(arr2)}")

    # Test case 3: Equal elements
    arr3 = [1, 1, 1, 1, 1]
    print(f"Array {arr3} is sorted in ascending order: {is_sorted_ascending(arr3)}")

    # Test case 4: Empty array
    arr4 = []
    print(f"Array {arr4} is sorted in ascending order: {is_sorted_ascending(arr4)}")

    # Test case 5: Single element array
    arr5 = [5]
    print(f"Array {arr5} is sorted in ascending order: {is_sorted_ascending(arr5)}")
```
