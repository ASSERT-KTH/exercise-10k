## Exercise
Convert a decimal number to binary or vice versa.

## Reference Solution
```python
def decimal_to_binary(decimal_number: int) -> str:
    """
    Converts a decimal number to binary.

    Args:
    decimal_number (int): The decimal number to convert.

    Returns:
    str: The binary representation of the decimal number.
    """
    return bin(decimal_number)[2:]


def binary_to_decimal(binary_number: str) -> int:
    """
    Converts a binary number to decimal.

    Args:
    binary_number (str): The binary number to convert.

    Returns:
    int: The decimal representation of the binary number.
    """
    return int(binary_number, 2)


# Example usage
if __name__ == "__main__":
    decimal_num = 18
    binary_num = "10010"

    print(f"Decimal to Binary: {decimal_num} -> {decimal_to_binary(decimal_num)}")
    print(f"Binary to Decimal: {binary_num} -> {binary_to_decimal(binary_num)}")
```
