## Exercise
Write a program that responds to user input with a simple message. Use if/else statements to determine the response.

## Reference Solution
```python
def get_user_input():
    """Get user input and strip leading/trailing whitespace."""
    return input("Please enter a message: ").strip().lower()

def respond_to_input(user_input):
    """Determine a response based on user input."""
    if user_input == "hello":
        return "Hello! How are you today?"
    elif user_input == "goodbye":
        return "Goodbye! It was nice talking to you."
    elif user_input == "how are you":
        return "I'm doing well, thanks for asking."
    else:
        return "I didn't quite understand that. Could you please rephrase?"

def main():
    """Main program loop."""
    print("Welcome to the simple message response program.")
    while True:
        user_input = get_user_input()
        if user_input == "quit":
            print("Goodbye!")
            break
        print(respond_to_input(user_input))

if __name__ == "__main__":
    main()
```
