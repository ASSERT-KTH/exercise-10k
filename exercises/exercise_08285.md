## Exercise
Write a program that calculates the mean, median, and mode of a dataset.

## Reference Solution
```python
import statistics
from collections import Counter

def calculate_statistics(data):
    """
    Calculate the mean, median, and mode of a dataset.

    Args:
        data (list): A list of numbers.

    Returns:
        tuple: A tuple containing the mean, median, and mode.
    """
    mean = statistics.mean(data)
    median = statistics.median(data)
    mode = statistics.mode(data)

    return mean, median, mode

def calculate_mode_manual(data):
    """
    Calculate the mode of a dataset manually.

    Args:
        data (list): A list of numbers.

    Returns:
        list: A list of modes.
    """
    counter = Counter(data)
    max_count = max(counter.values())
    modes = [num for num, count in counter.items() if count == max_count]

    return modes

def main():
    # Example usage
    data = [1, 2, 3, 4, 4, 5, 5, 5]
    mean, median, mode = calculate_statistics(data)
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Mode: {mode}")

    # Manual mode calculation
    modes = calculate_mode_manual(data)
    print(f"Modes (manual): {modes}")

if __name__ == "__main__":
    main()
```
