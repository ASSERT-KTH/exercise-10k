## Exercise
Write a program that finds the longest common subsequence of two given strings. The program should use dynamic programming to find the optimal solution.

def longest_common_subsequence(str1, str2):
    """
    This function calculates the longest common subsequence of two given strings.
    
    Args:
    str1 (str): The first string.
    str2 (str): The second string.
    
    Returns:
    str: The longest common subsequence of str1 and str2.
    """
    
    # Initialize a 2D array to store the lengths of common subsequences
    dp = [[0] * (len(str2) + 1) for _ in range(len(str1) + 1)]
    
    # Iterate over the characters in both strings
    for i in range(1, len(str1) + 1):
        for j in range(1, len(str2) + 1):
            # If the current characters match, increase the length of the common subsequence
            if str1[i - 1] == str2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            # Otherwise, take the maximum length from the previous characters
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
    
    # Initialize variables to store the longest common subsequence
    lcs = []
    i, j = len(str1), len(str2)
    
    # Backtrack to construct the longest common subsequence
    while i > 0 and j > 0:
        # If the current characters match, add the character to the LCS
        if str1[i - 1] == str2[j - 1]:
            lcs.append(str1[i - 1])
            i -= 1
            j -= 1
        # Move to the cell with the maximum value
        elif dp[i - 1][j] > dp[i][j - 1]:
            i -= 1
        else:
            j -= 1
    
    # Return the longest common subsequence in the correct order
    return ''.join(reversed(lcs))

# Example usage

    str2 = "GXTXAYB"
    print("Longest Common Subsequence:", longest_common_subsequence(str1, str2))
