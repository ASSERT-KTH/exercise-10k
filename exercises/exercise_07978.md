## Exercise
Write a program that performs a linear search on a list to find a target value.

## Reference Solution
```python
def linear_search(lst, target):
    """
    Performs a linear search on a list to find a target value.

    Args:
        lst (list): The list to search.
        target: The value to search for.

    Returns:
        int: The index of the target value if found, -1 otherwise.
    """
    for i, value in enumerate(lst):
        if value == target:
            return i
    return -1

def main():
    # Example usage
    numbers = [4, 2, 9, 6, 23, 12, 34, 0, 1]
    target_value = 12

    index = linear_search(numbers, target_value)

    if index != -1:
        print(f"Target value {target_value} found at index {index}.")
    else:
        print(f"Target value {target_value} not found in the list.")

if __name__ == "__main__":
    main()
```
