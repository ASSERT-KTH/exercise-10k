## Exercise
Create a program to calculate tax based on the income using conditional statements

## Reference Solution
```python
def calculate_tax(income):
    """
    Calculate tax based on income.

    Args:
        income (float): The income to calculate tax for.

    Returns:
        float: The calculated tax.
    """
    if income <= 0:
        return 0
    elif income <= 10000:
        # 10% tax for income up to $10,000
        return income * 0.10
    elif income <= 20000:
        # 15% tax for income between $10,001 and $20,000
        return (10000 * 0.10) + ((income - 10000) * 0.15)
    elif income <= 50000:
        # 20% tax for income between $20,001 and $50,000
        return (10000 * 0.10) + (10000 * 0.15) + ((income - 20000) * 0.20)
    else:
        # 25% tax for income above $50,000
        return (10000 * 0.10) + (10000 * 0.15) + (30000 * 0.20) + ((income - 50000) * 0.25)

def main():
    income = float(input("Enter your income: $"))
    tax = calculate_tax(income)
    print(f"Your tax is: ${tax:.2f}")

if __name__ == "__main__":
    main()
```
