## Exercise
Write a function that takes in two numbers and returns their least common multiple.

## Reference Solution
```python
def calculate_lcm(num1: int, num2: int) -> int:
    """
    Calculate the least common multiple of two numbers.

    Args:
    num1 (int): The first number.
    num2 (int): The second number.

    Returns:
    int: The least common multiple of num1 and num2.
    """
    # Calculate the greatest common divisor (GCD) of num1 and num2
    def calculate_gcd(a: int, b: int) -> int:
        while b:
            a, b = b, a % b
        return a

    # Calculate the LCM using the formula: LCM(a, b) = |a*b| / GCD(a, b)
    gcd = calculate_gcd(num1, num2)
    lcm = abs(num1 * num2) // gcd
    return lcm

# Example usage
if __name__ == "__main__":
    num1 = 12
    num2 = 15
    lcm = calculate_lcm(num1, num2)
    print(f"The least common multiple of {num1} and {num2} is {lcm}.")
```
