## Exercise
Create a program that takes in two numbers and an operation (+, -, *, /) from the user and performs the specified operation, then prints out the result.

## Reference Solution
```python
def calculator():
    """
    Asks the user for two numbers and an operation, performs the operation, and prints the result.
    """
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, /): ")

    # Perform the operation
    if operation == "+":
        result = num1 + num2
    elif operation == "-":
        result = num1 - num2
    elif operation == "*":
        result = num1 * num2
    elif operation == "/":
        # Handle division by zero
        if num2 == 0:
            print("Error: Division by zero is not allowed.")
            return
        result = num1 / num2
    else:
        print("Error: Invalid operation.")
        return

    # Print the result
    print(f"{num1} {operation} {num2} = {result}")

def main():
    calculator()
    # Ask the user if they want to perform another operation
    while True:
        response = input("Do you want to perform another operation? (yes/no): ")
        if response.lower() == "yes":
            calculator()
        elif response.lower() == "no":
            break
        else:
            print("Invalid response. Please enter 'yes' or 'no'.")

if __name__ == "__main__":
    main()
```
