## Exercise
Create a program that uses a dictionary to store student grades. The dictionary should have student names as keys and a list of grades as values. The program should allow users to add students, add grades for a student, and calculate the average grade for a student.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        """
        Initialize an empty dictionary to store student grades.
        """
        self.grades = {}

    def add_student(self, student_name):
        """
        Add a new student to the dictionary.

        Args:
            student_name (str): The name of the student to add.
        """
        if student_name not in self.grades:
            self.grades[student_name] = []
            print(f"Student '{student_name}' added successfully.")
        else:
            print(f"Student '{student_name}' already exists.")

    def add_grade(self, student_name, grade):
        """
        Add a grade for a specific student.

        Args:
            student_name (str): The name of the student to add a grade for.
            grade (float): The grade to add.
        """
        if student_name in self.grades:
            self.grades[student_name].append(grade)
            print(f"Grade '{grade}' added for student '{student_name}'.")
        else:
            print(f"Student '{student_name}' does not exist. Please add the student first.")

    def calculate_average(self, student_name):
        """
        Calculate the average grade for a specific student.

        Args:
            student_name (str): The name of the student to calculate the average grade for.

        Returns:
            float: The average grade for the student.
        """
        if student_name in self.grades:
            grades = self.grades[student_name]
            if len(grades) == 0:
                print(f"No grades available for student '{student_name}'.")
                return None
            average = sum(grades) / len(grades)
            print(f"The average grade for student '{student_name}' is {average:.2f}.")
            return average
        else:
            print(f"Student '{student_name}' does not exist. Please add the student first.")
            return None

    def display_grades(self):
        """
        Display all student grades.
        """
        for student, grades in self.grades.items():
            print(f"Student: {student}, Grades: {grades}")


def main():
    student_grades = StudentGrades()

    while True:
        print("\nStudent Grades Menu:")
        print("1. Add Student")
        print("2. Add Grade")
        print("3. Calculate Average")
        print("4. Display Grades")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            student_name = input("Enter student name: ")
            student_grades.add_student(student_name)
        elif choice == "2":
            student_name = input("Enter student name: ")
            grade = float(input("Enter grade: "))
            student_grades.add_grade(student_name, grade)
        elif choice == "3":
            student_name = input("Enter student name: ")
            student_grades.calculate_average(student_name)
        elif choice == "4":
            student_grades.display_grades()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
