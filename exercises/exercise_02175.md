## Exercise
Ask the user for the total bill and the tip percentage, and print the total amount to pay.

## Reference Solution
```python
def calculate_total_amount():
    """
    Asks the user for the total bill and the tip percentage, 
    then calculates and prints the total amount to pay.
    """
    # Get the total bill from the user
    while True:
        try:
            total_bill = float(input("Enter the total bill: $"))
            if total_bill < 0:
                print("The total bill cannot be negative.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip percentage from the user
    while True:
        try:
            tip_percentage = float(input("Enter the tip percentage (%): "))
            if tip_percentage < 0:
                print("The tip percentage cannot be negative.")
                continue
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the total amount to pay
    tip_amount = (tip_percentage / 100) * total_bill
    total_amount = total_bill + tip_amount

    # Print the total amount to pay
    print(f"Total bill: ${total_bill:.2f}")
    print(f"Tip ({tip_percentage}%): ${tip_amount:.2f}")
    print(f"Total amount to pay: ${total_amount:.2f}")

if __name__ == "__main__":
    calculate_total_amount()
```
