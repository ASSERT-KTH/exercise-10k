## Exercise
Create a program that generates a random number between 1 and 10, then asks the user to guess the number. After each guess, tell the user if their guess was higher or lower than the actual number.

## Reference Solution
```python
import random

def guess_number():
    # Generate a random number between 1 and 10
    number_to_guess = random.randint(1, 10)
    
    # Initialize the number of tries the player has made
    tries = 0
    
    print("Welcome to the number guessing game!")
    print("I'm thinking of a number between 1 and 10.")
    
    while True:
        # Ask the user for their guess
        user_guess = input("Take a guess: ")
        
        # Check if the user wants to quit
        if user_guess.lower() == "quit":
            print("Thanks for playing!")
            break
        
        # Try to convert the user's guess to an integer
        try:
            user_guess = int(user_guess)
        except ValueError:
            print("That's not a valid number!")
            continue
        
        # Check if the user's guess is in the correct range
        if user_guess < 1 or user_guess > 10:
            print("Please enter a number between 1 and 10.")
            continue
        
        # Increment the number of tries the player has made
        tries += 1
        
        # Check if the user's guess is correct
        if user_guess == number_to_guess:
            print(f"Congratulations! You found the number in {tries} tries.")
            break
        elif user_guess < number_to_guess:
            print("Your guess is too low. Try again!")
        else:
            print("Your guess is too high. Try again!")

if __name__ == "__main__":
    guess_number()
```
