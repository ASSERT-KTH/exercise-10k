## Exercise
Write a program that calculates the compound interest given principal amount, rate of interest, and time period.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, compounding_frequency=1):
    """
    Calculate compound interest.

    Args:
    - principal (float): Initial investment amount
    - rate (float): Annual interest rate (in decimal)
    - time (float): Time period in years
    - compounding_frequency (int): Number of times interest is compounded per year (default=1)

    Returns:
    - float: Compound interest amount
    """
    amount = principal * (1 + rate/compounding_frequency)**(compounding_frequency*time)
    return amount

def main():
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in percent): ")) / 100
    time = float(input("Enter the time period in years: "))
    compounding_frequency = int(input("Enter the compounding frequency per year (default=1): ") or 1)

    amount = calculate_compound_interest(principal, rate, time, compounding_frequency)
    interest = amount - principal

    print(f"Compound Interest: {interest:.2f}")
    print(f"Total Amount: {amount:.2f}")

if __name__ == "__main__":
    main()
```
