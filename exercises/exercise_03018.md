## Exercise
Write a program to check whether a given string ends with a specified suffix.

## Reference Solution
```python
def check_suffix(string, suffix):
    """
    Checks whether a given string ends with a specified suffix.

    Args:
        string (str): The input string to check.
        suffix (str): The suffix to look for.

    Returns:
        bool: True if the string ends with the suffix, False otherwise.
    """
    return string.endswith(suffix)

def main():
    # Example usage
    string = "example.txt"
    suffix = ".txt"
    result = check_suffix(string, suffix)
    print(f"The string '{string}' ends with '{suffix}': {result}")

    string = "example.pdf"
    suffix = ".txt"
    result = check_suffix(string, suffix)
    print(f"The string '{string}' ends with '{suffix}': {result}")

if __name__ == "__main__":
    main()
```
