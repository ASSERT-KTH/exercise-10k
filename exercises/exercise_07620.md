## Exercise
Create a program that uses nested for loops to print a multiplication table of a given size.

## Reference Solution
```python
def print_multiplication_table(size):
    """
    Prints a multiplication table of a given size.

    Args:
        size (int): The size of the multiplication table.
    """

    # Print the header row
    print("   |", end="")  # Print the top-left corner
    for i in range(1, size + 1):
        print(f"{i:4d}", end="")  # Print the column headers
    print()  # Move to the next line
    print("---" + "---" * size)  # Print the separator line

    # Print the multiplication table
    for i in range(1, size + 1):
        print(f"{i:3d}|", end="")  # Print the row header
        for j in range(1, size + 1):
            print(f"{i * j:4d}", end="")  # Print the product
        print()  # Move to the next line


def main():
    # Example usage
    size = 10
    print_multiplication_table(size)


if __name__ == "__main__":
    main()
```
