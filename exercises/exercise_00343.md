def concatenate_arrays(*arrays):
    """
    Concatenates two or more arrays into a single array.

    Args:
        *arrays: Variable number of arrays to be concatenated.

    Returns:
        A single array that is the concatenation of all input arrays.
    """

array1 = [1, 2, 3]

import numpy as np

def concatenate_arrays(*arrays):
    """
    Concatenates two or more arrays into a single array using NumPy.

    Args:
        *arrays: Variable number of arrays to be concatenated.

    Returns:
        A single array that is the concatenation of all input arrays.
    """
    return np.concatenate(arrays)

# Example usage:
array1 = np.array([1, 2, 3])
array2 = np.array([4, 5, 6])
array3 = np.array([7, 8, 9])

concatenated_array = concatenate_arrays(array1, array2, array3)
print(concatenated_array)  # Output: [1 2 3 4 5 6 7 8 9]
