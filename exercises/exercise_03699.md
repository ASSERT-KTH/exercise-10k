## Exercise
Write a program to calculate the total price of a meal given the prices of individual items.

## Reference Solution
```python
# Define a function to calculate the total price of a meal
def calculate_total_price():
    # Initialize the total price to 0
    total_price = 0
    
    # Get the number of items in the meal
    num_items = int(input("Enter the number of items in the meal: "))
    
    # Get the price of each item and add it to the total price
    for i in range(num_items):
        item_name = input(f"Enter the name of item {i+1}: ")
        item_price = float(input(f"Enter the price of {item_name}: $"))
        total_price += item_price
    
    # Calculate the tax (8% in this example)
    tax_rate = 0.08
    tax = total_price * tax_rate
    
    # Calculate the total price including tax
    total_price_with_tax = total_price + tax
    
    # Print the total price and the total price including tax
    print(f"\nTotal price: ${total_price:.2f}")
    print(f"Tax ({tax_rate*100}%): ${tax:.2f}")
    print(f"Total price including tax: ${total_price_with_tax:.2f}")

# Call the function to calculate the total price
calculate_total_price()
```
