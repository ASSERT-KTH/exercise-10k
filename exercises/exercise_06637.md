## Exercise
Write a function that uses testing to verify its correctness

## Reference Solution

```python
# calculator.py

def add_numbers(a, b):
    """
    Calculates the sum of two numbers.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The sum of a and b.
    """
    return a + b
```

```python
# test_calculator.py

import unittest
from calculator import add_numbers

class TestAddNumbersFunction(unittest.TestCase):
    """
    Test cases for the add_numbers function.
    """

    def test_positive_numbers(self):
        """
        Test adding two positive numbers.
        """
        self.assertEqual(add_numbers(5, 7), 12)

    def test_negative_numbers(self):
        """
        Test adding two negative numbers.
        """
        self.assertEqual(add_numbers(-5, -7), -12)

    def test_mixed_sign_numbers(self):
        """
        Test adding a positive and a negative number.
        """
        self.assertEqual(add_numbers(5, -7), -2)

    def test_zero(self):
        """
        Test adding a number and zero.
        """
        self.assertEqual(add_numbers(5, 0), 5)

    def test_non_integer_input(self):
        """
        Test adding a non-integer input.
        """
        with self.assertRaises(TypeError):
            add_numbers(5, "7")

if __name__ == "__main__":
    unittest.main()
```
