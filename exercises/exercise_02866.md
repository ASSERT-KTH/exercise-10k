## Exercise
Create a function to implement a Caesar cipher on a given string

def caesar_cipher(text: str, shift: int, direction: str) -> str:
    """
    Implement a Caesar cipher on a given string.

    Args:
    text (str): The input string to be encrypted/decrypted.
    shift (int): The number of positions to shift the letters.
    direction (str): 'encode' to encrypt, 'decode' to decrypt.

    Returns:
    str: The encrypted/decrypted string.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in text:
        if char.isalpha():
            # Find the position of the character in the alphabet
            position = alphabet.index(char.lower())
            
            # Shift the position based on the direction
            if direction == 'encode':
                new_position = (position + shift) % 26
            elif direction == 'decode':
                new_position = (position - shift) % 26
            
            # Add the shifted character to the result
            if char.isupper():
                result += alphabet[new_position].upper()
            else:
                result += alphabet[new_position]
        else:
            # If the character is not a letter, add it as is
            result += char

    return result


def main():
    # Example usage
    text = "Hello, World!"
    shift = 3

    encoded_text = caesar_cipher(text, shift, 'encode')
    decoded_text = caesar_cipher(encoded_text, shift, 'decode')

    print(f"Original text: {text}")
    print(f"Encoded text: {encoded_text}")
    print(f"Decoded text: {decoded_text}")


if __name__ == "__main__":
    main()
