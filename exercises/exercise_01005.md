## Exercise
Write a program that takes two strings as input and returns whether the strings are anagrams of each other or not.

## Reference Solution
```python
def are_anagrams(str1, str2):
    """
    Returns True if str1 and str2 are anagrams, False otherwise.

    :param str1: The first input string
    :param str2: The second input string
    :return: A boolean indicating whether the strings are anagrams
    """
    # Remove any whitespace and convert to lowercase
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # Check if sorted characters are equal
    return sorted(str1) == sorted(str2)


def main():
    # Example usage
    str1 = "Listen"
    str2 = "Silent"
    result = are_anagrams(str1, str2)
    print(f"Are '{str1}' and '{str2}' anagrams? {result}")

    str1 = "Hello"
    str2 = "World"
    result = are_anagrams(str1, str2)
    print(f"Are '{str1}' and '{str2}' anagrams? {result}")


if __name__ == "__main__":
    main()
```
