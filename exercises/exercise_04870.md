## Exercise
Write a program that uses a for loop to simulate 100 coin tosses and count the number of heads and tails.

## Reference Solution
```python
import random

def coin_toss_simulator(num_tosses):
    """
    Simulates a specified number of coin tosses and returns the count of heads and tails.

    Args:
        num_tosses (int): The number of coin tosses to simulate.

    Returns:
        tuple: A tuple containing the count of heads and tails.
    """
    heads = 0
    tails = 0

    for _ in range(num_tosses):
        # Generate a random number (0 or 1) to represent the coin toss
        toss = random.randint(0, 1)
        
        # If the toss is 0, it's tails; otherwise, it's heads
        if toss == 0:
            tails += 1
        else:
            heads += 1

    return heads, tails

def main():
    num_tosses = 100
    heads, tails = coin_toss_simulator(num_tosses)

    print(f"Simulated {num_tosses} coin tosses.")
    print(f"Heads: {heads}")
    print(f"Tails: {tails}")

if __name__ == "__main__":
    main()
```
