## Exercise
Write a program to demonstrate the concepts of inheritance and polymorphism in OOP.

## Reference Solution
```python
# Define a base class for Employee
class Employee:
    def __init__(self, name, age, salary):
        self.name = name
        self.age = age
        self.salary = salary

    def display_info(self):
        print(f"Name: {self.name}")
        print(f"Age: {self.age}")
        print(f"Salary: {self.salary}")

# Define a subclass for Professor that inherits from Employee
class Professor(Employee):
    def __init__(self, name, age, salary, department):
        super().__init__(name, age, salary)
        self.department = department

    def display_info(self):
        super().display_info()
        print(f"Department: {self.department}")

# Define a subclass for Lecturer that inherits from Employee
class Lecturer(Employee):
    def __init__(self, name, age, salary, course):
        super().__init__(name, age, salary)
        self.course = course

    def display_info(self):
        super().display_info()
        print(f"Course: {self.course}")

# Define a function to demonstrate polymorphism
def display_employee_info(employee):
    employee.display_info()

# Main function
def main():
    professor = Professor("John Doe", 40, 50000, "Computer Science")
    lecturer = Lecturer("Jane Smith", 30, 40000, "Mathematics")

    print("Professor's Information:")
    display_employee_info(professor)
    print()

    print("Lecturer's Information:")
    display_employee_info(lecturer)

if __name__ == "__main__":
    main()
```
