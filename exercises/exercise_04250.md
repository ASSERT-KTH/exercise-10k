## Exercise
Create a program that allows users to rate movies based on their genre. The program should take the movie title, genre, and rating as input, and print the average rating for the genre.

## Reference Solution
```python
class MovieRatingProgram:
    def __init__(self):
        self.genre_ratings = {}

    def add_movie(self):
        """Add a movie rating"""
        title = input("Enter movie title: ")
        genre = input("Enter movie genre: ")
        rating = float(input("Enter movie rating (1-5): "))

        if genre in self.genre_ratings:
            self.genre_ratings[genre].append(rating)
        else:
            self.genre_ratings[genre] = [rating]

        print(f"Movie '{title}' added with rating {rating} for genre '{genre}'")

    def get_average_rating(self):
        """Get the average rating for each genre"""
        for genre, ratings in self.genre_ratings.items():
            average_rating = sum(ratings) / len(ratings)
            print(f"Average rating for '{genre}': {average_rating:.2f}")

    def run(self):
        """Run the program"""
        while True:
            print("\nOptions:")
            print("1. Add a movie rating")
            print("2. Get average ratings")
            print("3. Quit")
            choice = input("Enter your choice: ")

            if choice == "1":
                self.add_movie()
            elif choice == "2":
                self.get_average_rating()
            elif choice == "3":
                break
            else:
                print("Invalid choice. Please try again.")

if __name__ == "__main__":
    program = MovieRatingProgram()
    program.run()
```
