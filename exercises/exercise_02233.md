## Exercise
Write a program that multiplies two matrices represented as 2D arrays.

## Reference Solution
```python
import numpy as np

def multiply_matrices(mat1, mat2):
    # Get the dimensions of the matrices
    rows_mat1 = len(mat1)
    cols_mat1 = len(mat1[0])
    rows_mat2 = len(mat2)
    cols_mat2 = len(mat2[0])

    # Check if the matrices can be multiplied
    if cols_mat1 != rows_mat2:
        print("Matrices cannot be multiplied")
        return None

    # Create a new matrix filled with zeros
    result = [[0 for _ in range(cols_mat2)] for _ in range(rows_mat1)]

    # Multiply the matrices
    for i in range(rows_mat1):
        for j in range(cols_mat2):
            for k in range(cols_mat1):  # or for k in range(rows_mat2)
                result[i][j] += mat1[i][k] * mat2[k][j]

    return result


# Example usage
matrix1 = [[1, 2], [3, 4]]
matrix2 = [[5, 6], [7, 8]]

result = multiply_matrices(matrix1, matrix2)

if result is not None:
    print("Matrix 1:")
    for row in matrix1:
        print(row)
    print("Matrix 2:")
    for row in matrix2:
        print(row)
    print("Result:")
    for row in result:
        print(row)


# Alternatively, you can use numpy's built-in matrix multiplication function
matrix1 = np.array([[1, 2], [3, 4]])
matrix2 = np.array([[5, 6], [7, 8]])

result = np.matmul(matrix1, matrix2)

print("Matrix 1:")
print(matrix1)
print("Matrix 2:")
print(matrix2)
print("Result:")
print(result)
```
