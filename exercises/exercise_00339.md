## Exercise
Write a program to infer the data type of a variable based on its value.

## Reference Solution
```python
def infer_data_type(value):
    """
    Infer the data type of a variable based on its value.

    Args:
        value: The value to infer the data type from.

    Returns:
        The inferred data type of the variable.
    """
    if isinstance(value, bool):
        return "Boolean"
    elif isinstance(value, int):
        return "Integer"
    elif isinstance(value, float):
        return "Float"
    elif isinstance(value, complex):
        return "Complex"
    elif isinstance(value, str):
        return "String"
    elif isinstance(value, list):
        return "List"
    elif isinstance(value, tuple):
        return "Tuple"
    elif isinstance(value, dict):
        return "Dictionary"
    elif isinstance(value, set):
        return "Set"
    else:
        return "Unknown"

# Example usage:
def main():
    values = [
        True,
        10,
        10.5,
        10 + 5j,
        "Hello, World!",
        [1, 2, 3],
        (1, 2, 3),
        {"a": 1, "b": 2},
        {1, 2, 3},
    ]

    for value in values:
        data_type = infer_data_type(value)
        print(f"Value: {value}, Data Type: {data_type}")

if __name__ == "__main__":
    main()
```
