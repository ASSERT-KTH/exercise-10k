## Exercise
Create a program that simulates a chatbot conversation. The user should be able to input messages and the program should then print out responses.

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary with intents and responses
intents = {
    'greeting': {
        'patterns': ['hello', 'hi', 'hey'],
        'responses': ['Hello! How can I assist you?', 'Hi! What brings you here?', 'Hey! How are you doing today?']
    },
    'goodbye': {
        'patterns': ['bye', 'goodbye', 'see you later'],
        'responses': ['Goodbye! It was nice chatting with you.', 'See you later! Have a great day.', 'Bye! Take care.']
    },
    'thanks': {
        'patterns': ['thanks', 'thank you'],
        'responses': ['You\'re welcome!', 'No problem, happy to help.']
    },
    'default': {
        'responses': ['I didn\'t quite understand that. Could you please rephrase?', 'Sorry, I\'m not sure what you mean.']
    }
}

# Function to get the chatbot response
def get_response(message):
    """Get the chatbot response based on the user's message."""
    message = message.lower()
    for intent, values in intents.items():
        if intent != 'default':
            for pattern in values['patterns']:
                if pattern in message:
                    return random.choice(values['responses'])
    return random.choice(intents['default']['responses'])

# Main function to simulate the chatbot conversation
def chatbot_conversation():
    """Simulate a chatbot conversation."""
    print("Welcome to the chatbot conversation simulator!")
    while True:
        user_message = input("You: ")
        if user_message.lower() == 'quit':
            break
        print("Chatbot: ", get_response(user_message))

# Run the chatbot conversation simulator
if __name__ == "__main__":
    chatbot_conversation()
```
