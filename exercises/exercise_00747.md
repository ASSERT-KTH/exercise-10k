## Exercise
Write a program that takes a story with blank spaces as input and asks the user to fill in the blanks with words

## Reference Solution
```python
def story_filler(story):
    """
    Asks the user to fill in the blanks in a story with words.

    Args:
        story (str): The story with blank spaces.

    Returns:
        str: The completed story.
    """
    # Split the story into parts separated by blank spaces
    parts = story.split("_____")

    # Initialize an empty list to store the user's input
    user_input = []

    # Ask the user to fill in each blank space
    for i in range(len(parts) - 1):
        word = input(f"Enter a {get_word_type(i)}: ")
        user_input.append(word)

    # Combine the user's input with the story parts
    completed_story = ""
    for i in range(len(parts) - 1):
        completed_story += parts[i] + user_input[i]
    completed_story += parts[-1]

    return completed_story


def get_word_type(index):
    """
    Returns the type of word to ask the user for based on the index.

    Args:
        index (int): The index of the blank space.

    Returns:
        str: The type of word to ask the user for.
    """
    word_types = ["noun", "verb", "adjective", "adverb", "place"]
    return word_types[index % len(word_types)]


def main():
    # Example story with blank spaces
    story = "Once upon a time, there was a _____ who lived in a _____ and loved to _____ in the _____."

    # Ask the user to fill in the blanks
    completed_story = story_filler(story)

    # Print the completed story
    print("Completed Story:")
    print(completed_story)


if __name__ == "__main__":
    main()
```
