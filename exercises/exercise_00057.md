## Exercise
Find the maximum repeating character in a string

## Reference Solution
```python
def max_repeating_char(s):
    """
    Find the maximum repeating character in a string.

    Args:
    s (str): The input string.

    Returns:
    str: The maximum repeating character.
    """
    if not s:
        return None

    # Create a dictionary to store the frequency of each character
    char_freq = {}

    # Iterate through the string and update the frequency of each character
    for char in s:
        if char in char_freq:
            char_freq[char] += 1
        else:
            char_freq[char] = 1

    # Find the character with the maximum frequency
    max_char = max(char_freq, key=char_freq.get)

    return max_char

# Example usage
s = "aabbbcccc"
max_char = max_repeating_char(s)
print(f"The maximum repeating character is: {max_char}")
```
