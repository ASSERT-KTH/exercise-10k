## Exercise
Write a program that sorts a list of numbers using the Bubble Sort algorithm, which uses a while loop to repeatedly iterate over the list and swap adjacent elements if they are in the wrong order.

fn bubble_sort(arr: &mut Vec<i32>) {
    let n = arr.len();
    let mut i = 0;
    let mut swapped = true;
    while swapped {
        swapped = false;
        i += 1;
        let mut j = 0;
        while j < n - i {
            if arr[j] > arr[j + 1] {
                let t = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = t;
                swapped = true;
            }
            j += 1;
        }
    }
}

fn main() {
    let mut arr = vec![5, 1, 2, 8, 3, 4, 9, 6, 7];
    bubble_sort(&mut arr);
    println!("{:?}", arr);
}
