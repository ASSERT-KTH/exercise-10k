## Exercise
Create a function to map a dictionary to a new dictionary with transformed keys and values.

## Reference Solution
```python
def map_dict(input_dict, key_func, value_func):
    """
    Maps a dictionary to a new dictionary with transformed keys and values.

    Args:
        input_dict (dict): The input dictionary.
        key_func (function): A function that takes a key as input and returns a transformed key.
        value_func (function): A function that takes a value as input and returns a transformed value.

    Returns:
        dict: A new dictionary with transformed keys and values.
    """

    # Initialize an empty dictionary to store the transformed key-value pairs
    transformed_dict = {}

    # Iterate over each key-value pair in the input dictionary
    for key, value in input_dict.items():
        # Apply the key transformation function to the current key
        transformed_key = key_func(key)
        
        # Apply the value transformation function to the current value
        transformed_value = value_func(value)
        
        # Store the transformed key-value pair in the new dictionary
        transformed_dict[transformed_key] = transformed_value
    
    # Return the new dictionary with transformed key-value pairs
    return transformed_dict

# Example usage:
def double_key(key):
    return key * 2

def square_value(value):
    return value ** 2

input_dict = {'a': 1, 'b': 2, 'c': 3}
transformed_dict = map_dict(input_dict, double_key, square_value)
print(transformed_dict)  # Output: {'aa': 1, 'bb': 4, 'cc': 9}
```
