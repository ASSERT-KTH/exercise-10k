## Exercise
Write a program that calculates the compound interest on an investment given the principal amount, interest rate, and time period.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time):
    """
    Calculates the compound interest on an investment.

    Args:
    - principal (float): The initial amount of money
    - rate (float): The interest rate (in decimal form)
    - time (int): The time period of the investment (in years)

    Returns:
    - float: The final amount of money after compound interest
    """
    # Calculate the compound interest using the formula: A = P * (1 + r)^t
    amount = principal * (1 + rate) ** time
    
    # Calculate the compound interest by subtracting the principal
    interest = amount - principal
    
    return amount, interest


def main():
    # Get user input
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in %): ")) / 100  # Convert to decimal
    time = int(input("Enter the time period (in years): "))

    # Calculate compound interest
    amount, interest = calculate_compound_interest(principal, rate, time)

    # Print the result
    print(f"Principal amount: ${principal:.2f}")
    print(f"Interest rate: {rate*100:.2f}%")
    print(f"Time period: {time} years")
    print(f"Final amount: ${amount:.2f}")
    print(f"Compound interest: ${interest:.2f}")


if __name__ == "__main__":
    main()
```
