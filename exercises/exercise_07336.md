## Exercise
Write a program that takes input from the user and writes it to a database.

## Reference Solution
```python
import sqlite3

# Connect to the database
def connect_to_database(db_name):
    """Connect to the SQLite database."""
    try:
        conn = sqlite3.connect(db_name)
        return conn
    except sqlite3.Error as e:
        print(f"Error connecting to database: {e}")
        return None

# Create a table in the database
def create_table(conn, table_name):
    """Create a table in the database."""
    try:
        cursor = conn.cursor()
        cursor.execute(f"""
            CREATE TABLE IF NOT EXISTS {table_name} (
                id INTEGER PRIMARY KEY,
                name TEXT NOT NULL,
                email TEXT NOT NULL
            )
        """)
        conn.commit()
    except sqlite3.Error as e:
        print(f"Error creating table: {e}")

# Insert data into the database
def insert_data(conn, table_name, name, email):
    """Insert data into the database."""
    try:
        cursor = conn.cursor()
        cursor.execute(f"""
            INSERT INTO {table_name} (name, email)
            VALUES (?, ?)
        """, (name, email))
        conn.commit()
    except sqlite3.Error as e:
        print(f"Error inserting data: {e}")

# Main function
def main():
    db_name = "user_database.db"
    table_name = "users"

    # Connect to the database
    conn = connect_to_database(db_name)
    if conn is None:
        return

    # Create a table in the database
    create_table(conn, table_name)

    while True:
        # Get user input
        name = input("Enter your name: ")
        email = input("Enter your email: ")

        # Insert data into the database
        insert_data(conn, table_name, name, email)

        # Ask the user if they want to continue
        response = input("Do you want to continue? (yes/no): ")
        if response.lower() != "yes":
            break

    # Close the database connection
    conn.close()

if __name__ == "__main__":
    main()
```
