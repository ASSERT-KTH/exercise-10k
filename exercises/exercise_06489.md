## Exercise
Write a program that plays a game of rock, paper, scissors with the user. The program should generate a random choice (rock, paper, or scissors) and then ask the user for their choice. After the user enters their choice, print out the result of the game.

## Reference Solution
```python
import random

def get_computer_choice():
    """Generate a random choice for the computer"""
    choices = ["rock", "paper", "scissors"]
    return random.choice(choices)

def get_user_choice():
    """Ask the user for their choice"""
    while True:
        user_choice = input("Enter your choice (rock, paper, or scissors): ").lower()
        if user_choice in ["rock", "paper", "scissors"]:
            return user_choice
        print("Invalid choice. Please try again.")

def determine_winner(computer_choice, user_choice):
    """Determine the winner of the game"""
    if computer_choice == user_choice:
        return "It's a tie!"
    if (computer_choice == "rock" and user_choice == "scissors") or \
       (computer_choice == "scissors" and user_choice == "paper") or \
       (computer_choice == "paper" and user_choice == "rock"):
        return "Computer wins!"
    return "User wins!"

def main():
    """Play a game of rock, paper, scissors"""
    computer_choice = get_computer_choice()
    user_choice = get_user_choice()
    print(f"\nComputer chose: {computer_choice}")
    print(f"User chose: {user_choice}\n")
    print(determine_winner(computer_choice, user_choice))

if __name__ == "__main__":
    main()
```
