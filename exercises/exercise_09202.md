## Exercise
Create a program that takes in two numbers and an arithmetic operation (+, -, *, /) as input, and then prints the result of the operation. The program should handle invalid operations and division by zero.

## Reference Solution
```python
#include <iostream>
#include <stdexcept>

using namespace std;

double calculate(double a, double b, char op) {
    double result;

    switch (op) {
        case '+':
            result = a + b;
            break;
        case '-':
            result = a - b;
            break;
        case '*':
            result = a * b;
            break;
        case '/':
            if (b == 0) {
                throw runtime_error("division by zero");
            }
            result = a / b;
            break;
        default:
            throw runtime_error("invalid operation");
    }

    return result;
}

int main() {
    double num1, num2;
    char op;

    cout << "Enter two numbers and an operation (+, -, *, /): ";
    cin >> num1 >> op >> num2;

    try {
        double result = calculate(num1, num2, op);
        cout << "Result: " << result << endl;
    } catch (const runtime_error& e) {
        cerr << "Error: " << e.what() << endl;
    }

    return 0;
}
```
