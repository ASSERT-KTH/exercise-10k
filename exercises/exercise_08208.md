## Exercise
Write a program that generates rhyming words using a for loop.

## Reference Solution
```python
# Define a dictionary of words and their corresponding rhyming words
rhyming_words = {
    "cat": ["hat", "mat", "rat", "sat"],
    "dog": ["log", "bog", "cog", "fog"],
    "sun": ["fun", "run", "done", "gun"],
}

def generate_rhyming_words(word):
    """
    Generate rhyming words for a given word.

    Args:
    word (str): The word for which to generate rhyming words.

    Returns:
    list: A list of rhyming words.
    """
    rhymes = []
    for key, value in rhyming_words.items():
        if key == word:
            rhymes = value
            break
    return rhymes

def main():
    # Test the function
    for word in rhyming_words.keys():
        print(f"Rhyming words for {word}: {generate_rhyming_words(word)}")

if __name__ == "__main__":
    main()
```

import requests

def generate_rhyming_words(word):
    """
    Generate rhyming words for a given word using the Datamuse API.

    Args:
    word (str): The word for which to generate rhyming words.

    Returns:
    list: A list of rhyming words.
    """
    url = f"https://api.datamuse.com/words?rel_rhy={word}"
    response = requests.get(url)
    data = response.json()
    rhymes = [item["word"] for item in data]
    return rhymes

def main():
    # Test the function
    word = "cat"
    print(f"Rhyming words for {word}: {generate_rhyming_words(word)}")

if __name__ == "__main__":
    main()
