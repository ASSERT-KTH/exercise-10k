## Exercise
Create a program that calculates the greatest common divisor (GCD) of two numbers.

## Reference Solution
```python
import math

def calculate_gcd(num1: int, num2: int) -> int:
    """
    Calculate the greatest common divisor (GCD) of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The GCD of num1 and num2.
    """
    while num2 != 0:
        num1, num2 = num2, num1 % num2
    return abs(num1)

def main():
    num1 = int(input("Enter the first number: "))
    num2 = int(input("Enter the second number: "))

    gcd = calculate_gcd(num1, num2)
    print(f"The GCD of {num1} and {num2} is: {gcd}")

    # Alternative method using math.gcd
    gcd_math = math.gcd(num1, num2)
    print(f"Alternative method using math.gcd: {gcd_math}")

if __name__ == "__main__":
    main()
```
