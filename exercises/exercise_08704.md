## Exercise
Create a simple chatbot that responds to basic user input.

## Reference Solution
```python
import random

# Define a dictionary of intents and responses
intents = {
    "greeting": ["Hello!", "Hi!", "Hey!", "Hi there!"],
    "goodbye": ["Goodbye!", "See you later!", "Bye!", "Until next time!"],
    "thanks": ["You're welcome!", "No problem!", "Anytime!", "Glad to help!"],
    "help": ["How can I assist you?", "What do you need help with?", "I'm here to help!", "What's on your mind?"]
}

# Define a function to process user input
def process_input(input_text):
    input_text = input_text.lower()
    if "hello" in input_text or "hi" in input_text:
        return random.choice(intents["greeting"])
    elif "goodbye" in input_text or "bye" in input_text:
        return random.choice(intents["goodbye"])
    elif "thanks" in input_text or "thank you" in input_text:
        return random.choice(intents["thanks"])
    elif "help" in input_text:
        return random.choice(intents["help"])
    else:
        return "I didn't understand that. Please try again!"

# Define a main function to run the chatbot
def main():
    print("Welcome to the chatbot! Type 'quit' to exit.")
    while True:
        user_input = input("You: ")
        if user_input.lower() == "quit":
            break
        response = process_input(user_input)
        print("Chatbot: ", response)

# Run the chatbot
if __name__ == "__main__":
    main()
```
