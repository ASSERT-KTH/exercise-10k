## Exercise
Write a function that uses the subprocess module to run external commands.

## Reference Solution
```python
import subprocess

def run_external_command(command):
    """
    Run an external command using the subprocess module.

    Args:
        command (str or list): The command to run. Can be a string or a list of strings.

    Returns:
        tuple: A tuple containing the return code, stdout, and stderr of the command.
    """
    try:
        # Run the command
        if isinstance(command, str):
            result = subprocess.run(command, shell=True, capture_output=True, text=True)
        else:
            result = subprocess.run(command, capture_output=True, text=True)

        # Return the return code, stdout, and stderr
        return result.returncode, result.stdout, result.stderr

    except Exception as e:
        print(f"An error occurred: {e}")
        return None

# Example usage
if __name__ == "__main__":
    command = "ls -l"
    return_code, stdout, stderr = run_external_command(command)

    print(f"Return Code: {return_code}")
    print(f"STDOUT:")
    print(stdout)
    print(f"STDERR:")
    print(stderr)

    # Running a command with arguments as a list
    command_list = ["ls", "-l"]
    return_code, stdout, stderr = run_external_command(command_list)

    print(f"Return Code: {return_code}")
    print(f"STDOUT:")
    print(stdout)
    print(f"STDERR:")
    print(stderr)
```
