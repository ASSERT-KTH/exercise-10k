## Exercise
Write a program that declares and operates on tuples.

## Reference Solution
```python
# tuple_operations.py

def main():
    # Declare a tuple
    my_tuple = (1, 2, 3, 4, 5)
    print("Declared Tuple:", my_tuple)

    # Accessing tuple elements
    print("First Element:", my_tuple[0])
    print("Last Element:", my_tuple[-1])

    # Tuple slicing
    print("Slice (1:3):", my_tuple[1:3])

    # Tuple concatenation
    tuple1 = (1, 2, 3)
    tuple2 = (4, 5, 6)
    print("Concatenated Tuple:", tuple1 + tuple2)

    # Tuple repetition
    print("Repeated Tuple:", my_tuple * 2)

    # Tuple membership
    print("Is 3 in the tuple?", 3 in my_tuple)

    # Tuple length
    print("Tuple Length:", len(my_tuple))

    # Tuple unpacking
    a, b, c, d, e = my_tuple
    print("Unpacked Tuple:", a, b, c, d, e)

if __name__ == "__main__":
    main()
```
