## Exercise
Write a program that calculates the sum of all elements in a given array.

## Reference Solution
```python
def calculate_sum(array):
    """
    This function calculates the sum of all elements in a given array.

    Args:
        array (list): A list of numbers.

    Returns:
        int: The sum of all elements in the array.

    Raises:
        TypeError: If the input is not a list or if the list contains non-numeric values.
    """
    # Check if the input is a list
    if not isinstance(array, list):
        raise TypeError("Input must be a list")

    # Initialize the sum to 0
    total_sum = 0

    # Iterate over each element in the array
    for num in array:
        # Check if the element is a numeric value
        if not isinstance(num, (int, float)):
            raise TypeError("Array must only contain numeric values")

        # Add the element to the sum
        total_sum += num

    # Return the sum
    return total_sum


# Example usage
if __name__ == "__main__":
    # Test the function with a sample array
    array = [1, 2, 3, 4, 5]
    print("Array:", array)
    print("Sum:", calculate_sum(array))
```
