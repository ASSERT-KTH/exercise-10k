## Exercise
Write a program that performs basic list operations such as indexing, slicing, and concatenation.

## Reference Solution
```python
def list_operations():
    # Define a list
    my_list = [1, 2, 3, 4, 5]

    # Indexing
    print("Indexing:")
    print("Element at index 0:", my_list[0])
    print("Element at index -1 (last element):", my_list[-1])

    # Slicing
    print("\nSlicing:")
    print("Elements from index 1 to 3:", my_list[1:4])
    print("Elements from index 2 to the end:", my_list[2:])
    print("Elements from the start to index 3:", my_list[:4])
    print("All elements:", my_list[:])

    # Concatenation
    print("\nConcatenation:")
    list1 = [1, 2, 3]
    list2 = [4, 5, 6]
    print("List 1:", list1)
    print("List 2:", list2)
    print("Concatenated list:", list1 + list2)

    # Modifying list elements
    print("\nModifying list elements:")
    my_list[0] = 10
    print("Modified list:", my_list)

    # Appending elements to the list
    print("\nAppending elements to the list:")
    my_list.append(6)
    print("List after appending:", my_list)

    # Inserting elements at a specific position
    print("\nInserting elements at a specific position:")
    my_list.insert(0, 0)
    print("List after insertion:", my_list)

    # Removing elements from the list
    print("\nRemoving elements from the list:")
    my_list.remove(10)
    print("List after removal:", my_list)

    # Sorting the list
    print("\nSorting the list:")
    my_list.sort()
    print("Sorted list:", my_list)


if __name__ == "__main__":
    list_operations()
```
