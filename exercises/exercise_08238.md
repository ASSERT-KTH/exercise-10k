## Exercise
Create a program that rents a car using if/else statements

## Reference Solution
```python
# Define a dictionary to store car rental information
car_rental_info = {
    "Economy": {"age_limit": 21, "experience_limit": 1, "rental_fee": 40},
    "Compact": {"age_limit": 25, "experience_limit": 2, "rental_fee": 50},
    "Midsize": {"age_limit": 28, "experience_limit": 3, "rental_fee": 60},
    "Fullsize": {"age_limit": 30, "experience_limit": 5, "rental_fee": 80},
}

def rent_car(car_type, age, experience):
    """
    Rent a car based on the user's age and driving experience.

    Args:
        car_type (str): The type of car to rent.
        age (int): The user's age.
        experience (int): The user's driving experience in years.

    Returns:
        str: A message indicating whether the car can be rented or not.
    """
    # Check if the car type is valid
    if car_type not in car_rental_info:
        return "Invalid car type. Please choose from: Economy, Compact, Midsize, Fullsize"

    # Check if the user meets the age requirement
    if age < car_rental_info[car_type]["age_limit"]:
        return f"You must be at least {car_rental_info[car_type]['age_limit']} years old to rent a {car_type} car."

    # Check if the user meets the driving experience requirement
    if experience < car_rental_info[car_type]["experience_limit"]:
        return f"You must have at least {car_rental_info[car_type]['experience_limit']} years of driving experience to rent a {car_type} car."

    # If all requirements are met, rent the car
    return f"Congratulations! You can rent a {car_type} car for ${car_rental_info[car_type]['rental_fee']} per day."

def main():
    print("Welcome to our car rental service!")
    car_type = input("Which type of car would you like to rent? (Economy, Compact, Midsize, Fullsize): ")
    age = int(input("How old are you? "))
    experience = int(input("How many years of driving experience do you have? "))

    result = rent_car(car_type, age, experience)
    print(result)

if __name__ == "__main__":
    main()
```
