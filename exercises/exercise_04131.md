## Exercise
Use type hinting to specify the data type of a variable or function parameter

## Reference Solution
```python
# Variable type hinting
name: str = "John Doe"

# Function parameter type hinting
def greet(name: str) -> None:
    print(f"Hello, {name}!")

# Function return type hinting
def add(x: int, y: int) -> int:
    return x + y

# Using type hinting with a list
fruits: list[str] = ["apple", "banana", "orange"]

# Using type hinting with a dictionary
person: dict[str, str] = {"name": "John Doe", "age": "30"}
```

```python
from typing import List, Dict

class Person:
    def __init__(self, name: str, age: int) -> None:
        self.name: str = name
        self.age: int = age

    def greet(self) -> None:
        print(f"Hello, my name is {self.name} and I'm {self.age} years old.")

    def get_info(self) -> Dict[str, str]:
        return {"name": self.name, "age": str(self.age)}

    def add_friend(self, friend: 'Person') -> None:
        print(f"{self.name} is now friends with {friend.name}.")

    def get_friends(self) -> List['Person']:
        # This is a placeholder, in a real scenario you would have a list of friends
        return []

# Create a new person
person: Person = Person("John Doe", 30)

# Greet
person.greet()

# Get info
info: Dict[str, str] = person.get_info()
print(info)

# Add friend
friend: Person = Person("Jane Doe", 25)
person.add_friend(friend)

# Get friends
friends: List[Person] = person.get_friends()
print(friends)
```
