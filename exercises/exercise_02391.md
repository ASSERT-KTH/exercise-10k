## Exercise
Write a program that checks if two given strings are anagrams of each other using a while loop.

## Reference Solution
```python
def are_anagrams(str1, str2):
    # Remove any white spaces and convert to lower case
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # If both strings have different lengths, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Sort both strings and compare them
    sorted_str1 = sorted(str1)
    sorted_str2 = sorted(str2)

    i = 0
    while i < len(sorted_str1):
        if sorted_str1[i] != sorted_str2[i]:
            return False
        i += 1

    return True

# Test the function
print(are_anagrams("Listen", "Silent"))  # Returns: True
print(are_anagrams("Hello", "World"))  # Returns: False
```
