## Exercise
Write a program to calculate the tip percentage

## Reference Solution
```python
def calculate_tip(bill_amount, tip_percentage):
    """
    Calculates the tip amount based on the bill amount and tip percentage.

    Args:
        bill_amount (float): The total bill amount.
        tip_percentage (float): The desired tip percentage.

    Returns:
        float: The calculated tip amount.
    """
    tip_amount = (bill_amount / 100) * tip_percentage
    return tip_amount

def main():
    print("Tip Percentage Calculator")
    print("-------------------------")

    # Get the bill amount from the user
    while True:
        try:
            bill_amount = float(input("Enter the bill amount: $"))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip percentage from the user
    while True:
        try:
            tip_percentage = float(input("Enter the tip percentage (e.g., 15 for 15%): "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the tip amount
    tip_amount = calculate_tip(bill_amount, tip_percentage)

    # Calculate the total amount (bill + tip)
    total_amount = bill_amount + tip_amount

    # Display the results
    print(f"Bill Amount: ${bill_amount:.2f}")
    print(f"Tip Percentage: {tip_percentage}%")
    print(f"Tip Amount: ${tip_amount:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
