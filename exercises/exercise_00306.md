## Exercise
Write a function that merges two lists into a single list.

## Reference Solution
```python
def merge_lists(list1, list2):
    """
    Merge two lists into a single list.

    Args:
        list1 (list): The first list to merge.
        list2 (list): The second list to merge.

    Returns:
        list: A new list containing all elements from both input lists.
    """
    return list1 + list2

# Example usage:
list1 = [1, 2, 3]
list2 = [4, 5, 6]

merged_list = merge_lists(list1, list2)
print(merged_list)  # Output: [1, 2, 3, 4, 5, 6]
```

def merge_lists_extend(list1, list2):
    """
    Merge two lists into a single list using extend().

    Args:
        list1 (list): The first list to merge.
        list2 (list): The second list to merge.

    Returns:
        list: A new list containing all elements from both input lists.
    """
    merged_list = list1.copy()  # Create a copy to avoid modifying the original list
    merged_list.extend(list2)
    return merged_list

def merge_lists_chain(list1, list2):
    """
    Merge two lists into a single list using itertools.chain().

    Args:
        list1 (list): The first list to merge.
        list2 (list): The second list to merge.

    Returns:
        list: A new list containing all elements from both input lists.
    """
    import itertools
    return list(itertools.chain(list1, list2))

# Example usage:
list1 = [1, 2, 3]
list2 = [4, 5, 6]

merged_list_extend = merge_lists_extend(list1, list2)
print(merged_list_extend)  # Output: [1, 2, 3, 4, 5, 6]

merged_list_chain = merge_lists_chain(list1, list2)
print(merged_list_chain)  # Output: [1, 2, 3, 4, 5, 6]
