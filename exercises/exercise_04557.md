## Exercise
Create a program that rotates a given string by a specified number of positions.

## Reference Solution
```python
def rotate_string(input_str, num_positions):
    """
    Rotate a given string by a specified number of positions.

    Args:
        input_str (str): The input string to rotate.
        num_positions (int): The number of positions to rotate the string.

    Returns:
        str: The rotated string.
    """
    num_positions = num_positions % len(input_str)
    return input_str[-num_positions:] + input_str[:-num_positions]


def main():
    # Example usage
    input_str = "Hello, World!"
    num_positions = 5

    rotated_str = rotate_string(input_str, num_positions)
    print(f"Original String: {input_str}")
    print(f"Rotated String ({num_positions} positions): {rotated_str}")


if __name__ == "__main__":
    main()
```
