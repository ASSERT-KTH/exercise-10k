## Exercise
Write a program to declare and use a tuple to store a collection of values.

## Reference Solution
```python
# Declare a tuple with a collection of values
def declare_tuple():
    """Declare a tuple with a collection of values."""
    my_tuple = (1, 2, 3, 4, 5)
    return my_tuple

# Access and print tuple values
def access_tuple_values(my_tuple):
    """Access and print tuple values."""
    print("Tuple Values:")
    for i, value in enumerate(my_tuple):
        print(f"Index {i}: {value}")

# Update tuple values (note: tuples are immutable, so we create a new tuple)
def update_tuple_values(my_tuple):
    """Update tuple values by creating a new tuple."""
    new_tuple = list(my_tuple)  # Convert tuple to list to make it mutable
    new_tuple[0] = 10  # Update the first value
    new_tuple = tuple(new_tuple)  # Convert list back to tuple
    return new_tuple

# Main function
def main():
    my_tuple = declare_tuple()
    print("Original Tuple:", my_tuple)
    access_tuple_values(my_tuple)
    updated_tuple = update_tuple_values(my_tuple)
    print("Updated Tuple:", updated_tuple)

if __name__ == "__main__":
    main()
```
