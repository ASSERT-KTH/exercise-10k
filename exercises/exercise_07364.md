## Exercise
Perform boolean operations such as AND, OR, and NOT on variables and print the results.

## Reference Solution
```python
def main():
    # Define boolean variables
    a = True
    b = False

    # AND operation
    print("AND operation:")
    print(f"a: {a}, b: {b}")
    print(f"a AND b: {a and b}")

    # OR operation
    print("\nOR operation:")
    print(f"a: {a}, b: {b}")
    print(f"a OR b: {a or b}")

    # NOT operation
    print("\nNOT operation:")
    print(f"a: {a}")
    print(f"NOT a: {not a}")

    # Example usage with multiple variables
    c = True
    d = False
    print("\nExample usage with multiple variables:")
    print(f"a: {a}, b: {b}, c: {c}, d: {d}")
    print(f"(a AND b) OR (c AND d): {(a and b) or (c and d)}")
    print(f"(a OR b) AND (c OR d): {(a or b) and (c or d)}")

if __name__ == "__main__":
    main()
```
