## Exercise
Write a program to understand the limits of different data types (integer, float, string) and how to handle overflow or underflow.

## Reference Solution
```python
import sys
import decimal

def test_integer_limits():
    """Test integer limits"""
    max_size = sys.maxsize
    print(f"Maximum integer size: {max_size}")
    min_size = -1 * max_size - 1
    print(f"Minimum integer size: {min_size}")

    # Test integer overflow
    try:
        large_int = max_size + 1
        print(f"Large integer: {large_int}")
    except OverflowError:
        print("Integer overflow occurred")

def test_float_limits():
    """Test float limits"""
    max_value = sys.float_info.max
    print(f"Maximum float value: {max_value}")
    min_value = sys.float_info.min
    print(f"Minimum float value: {min_value}")

    # Test float overflow
    try:
        large_float = max_value * 2
        print(f"Large float: {large_float}")
    except OverflowError:
        print("Float overflow occurred")

def test_string_limits():
    """Test string limits"""
    max_size = sys.maxsize
    print(f"Maximum string size: {max_size} characters")
    min_size = 0
    print(f"Minimum string size: {min_size} characters")

    # Test string overflow
    try:
        large_string = "a" * (max_size + 1)
        print(f"Large string: {large_string}")
    except MemoryError:
        print("String overflow occurred")

def use_arbitrary_precision_arithmetic():
    """Use arbitrary-precision arithmetic"""
    decimal.getcontext().prec = 100
    large_decimal = decimal.Decimal("1.7976931348623157e+308") * 2
    print(f"Large decimal: {large_decimal}")

def use_exception_handling():
    """Use exception handling"""
    try:
        large_int = sys.maxsize + 1
        print(f"Large integer: {large_int}")
    except OverflowError:
        print("Integer overflow occurred")

if __name__ == "__main__":
    test_integer_limits()
    test_float_limits()
    test_string_limits()
    use_arbitrary_precision_arithmetic()
    use_exception_handling()
```
