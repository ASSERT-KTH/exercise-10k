## Exercise
Write a program that asks the user for the amount of money they want to give as change and the denominations of the coins they want to use. Then, calculate the number of coins of each denomination needed to make the change and print it out

## Reference Solution
```python
def calculate_change(amount, denominations):
    """
    Calculate the number of coins of each denomination needed to make the change.

    Args:
    amount (float): The amount of money to give as change.
    denominations (list): A list of coin denominations available.

    Returns:
    dict: A dictionary where the keys are the coin denominations and the values are the number of coins needed.
    """
    # Sort the denominations in descending order
    denominations.sort(reverse=True)

    # Initialize a dictionary to store the number of coins of each denomination
    coins_needed = {denomination: 0 for denomination in denominations}

    # Convert the amount to cents to avoid floating point precision issues
    amount_cents = int(amount * 100)

    # Iterate over the denominations
    for denomination in denominations:
        # Convert the denomination to cents
        denomination_cents = int(denomination * 100)

        # Calculate the number of coins of the current denomination needed
        coins_needed[denomination] = amount_cents // denomination_cents

        # Update the remaining amount
        amount_cents %= denomination_cents

    return coins_needed


def main():
    # Ask the user for the amount of money they want to give as change
    amount = float(input("Enter the amount of money you want to give as change: "))

    # Ask the user for the denominations of the coins they want to use
    num_denominations = int(input("Enter the number of coin denominations you want to use: "))

    # Initialize a list to store the coin denominations
    denominations = []

    # Iterate over the number of denominations
    for i in range(num_denominations):
        # Ask the user for the denomination
        denomination = float(input(f"Enter denomination {i+1} (in dollars): "))

        # Add the denomination to the list
        denominations.append(denomination)

    # Calculate the number of coins of each denomination needed
    coins_needed = calculate_change(amount, denominations)

    # Print the result
    print("Number of coins of each denomination needed:")
    for denomination, num_coins in coins_needed.items():
        print(f"{denomination:.2f} dollars: {num_coins}")


if __name__ == "__main__":
    main()
```
