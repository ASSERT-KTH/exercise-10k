## Exercise
Write a program to reverse the order of elements in a given list.

## Reference Solution

def reverse_list(input_list):
    """
    Reverses the order of elements in a given list.

    Args:
        input_list (list): The list to be reversed.

    Returns:
        list: The reversed list.
    """
    input_list.reverse()
    return input_list

# Example usage:
my_list = [1, 2, 3, 4, 5]
print("Original List:", my_list)

def reverse_list(input_list):
    """
    Reverses the order of elements in a given list.

    Args:
        input_list (list): The list to be reversed.

    Returns:
        list: The reversed list.
    """
    if len(input_list) == 0:
        return []
    else:
        return reverse_list(input_list[1:]) + [input_list[0]]

# Example usage:
my_list = [1, 2, 3, 4, 5]
print("Original List:", my_list)
print("Reversed List:", reverse_list(my_list))
