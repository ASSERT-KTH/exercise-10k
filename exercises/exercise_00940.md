## Exercise
Write a program that creates a simple customer relationship management system using if/else statement

## Reference Solution
```python
class Customer:
    def __init__(self, name, email, phone):
        self.name = name
        self.email = email
        self.phone = phone

class CRM:
    def __init__(self):
        self.customers = {}

    def add_customer(self, name, email, phone):
        if name in self.customers:
            print("Customer already exists.")
        else:
            self.customers[name] = Customer(name, email, phone)
            print("Customer added successfully.")

    def delete_customer(self, name):
        if name in self.customers:
            del self.customers[name]
            print("Customer deleted successfully.")
        else:
            print("Customer does not exist.")

    def update_customer(self, name, email=None, phone=None):
        if name in self.customers:
            if email:
                self.customers[name].email = email
            if phone:
                self.customers[name].phone = phone
            print("Customer updated successfully.")
        else:
            print("Customer does not exist.")

    def view_customer(self, name):
        if name in self.customers:
            customer = self.customers[name]
            print(f"Name: {customer.name}")
            print(f"Email: {customer.email}")
            print(f"Phone: {customer.phone}")
        else:
            print("Customer does not exist.")

    def view_all_customers(self):
        if self.customers:
            for customer in self.customers.values():
                print(f"Name: {customer.name}")
                print(f"Email: {customer.email}")
                print(f"Phone: {customer.phone}")
                print("------------------------")
        else:
            print("No customers in the system.")

def main():
    crm = CRM()

    while True:
        print("1. Add Customer")
        print("2. Delete Customer")
        print("3. Update Customer")
        print("4. View Customer")
        print("5. View All Customers")
        print("6. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            name = input("Enter customer name: ")
            email = input("Enter customer email: ")
            phone = input("Enter customer phone: ")
            crm.add_customer(name, email, phone)
        elif choice == "2":
            name = input("Enter customer name: ")
            crm.delete_customer(name)
        elif choice == "3":
            name = input("Enter customer name: ")
            email = input("Enter new email (press enter to skip): ")
            phone = input("Enter new phone (press enter to skip): ")
            crm.update_customer(name, email or None, phone or None)
        elif choice == "4":
            name = input("Enter customer name: ")
            crm.view_customer(name)
        elif choice == "5":
            crm.view_all_customers()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
