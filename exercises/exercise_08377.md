## Exercise
Write a program that generates a harmonic progression of a given length with a given common difference.

## Reference Solution
```python
def harmonic_progression(length, common_difference, start=1):
    """
    Generates a harmonic progression of a given length with a given common difference.

    Args:
        length (int): The length of the progression.
        common_difference (float): The common difference between consecutive terms.
        start (float, optional): The first term of the progression. Defaults to 1.

    Returns:
        list: A list of terms in the harmonic progression.
    """
    progression = []
    for i in range(length):
        term = 1 / (start + i * common_difference)
        progression.append(term)
    return progression

def main():
    length = int(input("Enter the length of the progression: "))
    common_difference = float(input("Enter the common difference: "))
    start = float(input("Enter the first term (default is 1): ") or 1)
    
    progression = harmonic_progression(length, common_difference, start)
    print("Harmonic Progression:")
    print(progression)

if __name__ == "__main__":
    main()
```
