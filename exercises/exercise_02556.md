## Exercise
Create a decorator to measure the execution time of a given function.

## Reference Solution
```python
import time
from functools import wraps

def measure_execution_time(func):
    """
    A decorator to measure the execution time of a given function.

    Args:
        func: The function to be measured.

    Returns:
        A wrapper function that measures the execution time of the input function.
    """
    @wraps(func)
    def wrapper(*args, **kwargs):
        # Record the start time
        start_time = time.time()
        
        # Execute the function
        result = func(*args, **kwargs)
        
        # Record the end time
        end_time = time.time()
        
        # Calculate the execution time
        execution_time = end_time - start_time
        
        # Print the execution time
        print(f"Function '{func.__name__}' executed in {execution_time:.4f} seconds.")
        
        # Return the result of the function
        return result
    
    return wrapper

# Example usage:
@measure_execution_time
def example_function():
    time.sleep(1)  # Simulate some work
    print("Example function executed.")

if __name__ == "__main__":
    example_function()
```
