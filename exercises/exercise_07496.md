## Exercise
Perform basic list operations (append, insert, remove) and print out the resulting list.

## Reference Solution
```python
def main():
    # Initialize the list
    my_list = [1, 2, 3, 4, 5]

    print("Initial list: ", my_list)

    # Append an element to the list
    my_list.append(6)
    print("List after appending 6: ", my_list)

    # Insert an element at a specific position
    my_list.insert(2, 10)
    print("List after inserting 10 at position 2: ", my_list)

    # Remove the first occurrence of an element
    my_list.remove(4)
    print("List after removing 4: ", my_list)

    # Remove an element at a specific position
    my_list.pop(0)
    print("List after removing the element at position 0: ", my_list)


if __name__ == "__main__":
    main()
```
