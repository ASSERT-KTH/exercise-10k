## Exercise
Write a program that takes multiple inputs from the user and prints out the results.

## Reference Solution
```python
def get_user_input():
    """
    Get multiple inputs from the user.
    """
    # Get user's name
    name = input("Please enter your name: ")

    # Get user's age
    while True:
        try:
            age = int(input("Please enter your age: "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid age.")

    # Get user's favorite color
    color = input("Please enter your favorite color: ")

    # Get user's favorite hobby
    hobby = input("Please enter your favorite hobby: ")

    return name, age, color, hobby

def print_results(name, age, color, hobby):
    """
    Print out the user's input.
    """
    print("\nResults:")
    print(f"Name: {name}")
    print(f"Age: {age}")
    print(f"Favorite Color: {color}")
    print(f"Favorite Hobby: {hobby}")

def main():
    name, age, color, hobby = get_user_input()
    print_results(name, age, color, hobby)

if __name__ == "__main__":
    main()
```
