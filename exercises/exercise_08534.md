## Exercise
Create a program to calculate the change owed to a customer after a purchase.

## Reference Solution
```python
def calculate_change(purchase_amount, payment_amount):
    """
    Calculate the change owed to a customer after a purchase.

    Args:
        purchase_amount (float): The amount of the purchase.
        payment_amount (float): The amount paid by the customer.

    Returns:
        float: The change amount.
    """
    if payment_amount < purchase_amount:
        raise ValueError("Payment amount is less than purchase amount")

    change_amount = payment_amount - purchase_amount
    return change_amount

def main():
    print("Change Calculator Program")
    print("-------------------------")

    # Get the purchase amount from the user
    while True:
        try:
            purchase_amount = float(input("Enter the purchase amount: $"))
            if purchase_amount < 0:
                print("Purchase amount cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the payment amount from the user
    while True:
        try:
            payment_amount = float(input("Enter the payment amount: $"))
            if payment_amount < 0:
                print("Payment amount cannot be negative. Please try again.")
            elif payment_amount < purchase_amount:
                print("Payment amount is less than purchase amount. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the change amount
    change_amount = calculate_change(purchase_amount, payment_amount)

    # Print the result
    print(f"Change: ${change_amount:.2f}")

if __name__ == "__main__":
    main()
```
