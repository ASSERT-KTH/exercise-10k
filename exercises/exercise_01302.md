## Exercise
Given a dictionary, write a function to retrieve the value associated with a given key

def get_value(dictionary, key):
    """
    Retrieves the value associated with a given key from a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key (any): The key to search for.

    Returns:
        any: The value associated with the key if found, None otherwise.
    """
    return dictionary.get(key)

# Example usage:

```python
def get_value(dictionary, key):
    """
    Retrieves the value associated with a given key from a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key (any): The key to search for.

    Returns:
        any: The value associated with the key if found, raises KeyError otherwise.
    """
    if key not in dictionary:
        raise KeyError(f"Key '{key}' not found in dictionary")
    return dictionary[key]

# Example usage:
my_dict = {'name': 'John', 'age': 30, 'city': 'New York'}
key = 'country'
try:
    print(get_value(my_dict, key))
except KeyError as e:
    print(e)  # Output: Key 'country' not found in dictionary
```
