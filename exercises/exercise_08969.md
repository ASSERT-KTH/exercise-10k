## Exercise
Write a function that takes a list of strings and a delimiter as input and returns the joined string.

## Reference Solution
```python
def join_strings(string_list, delimiter):
    """
    Join a list of strings with a specified delimiter.

    Args:
    - string_list (list): A list of strings to be joined.
    - delimiter (str): The delimiter to use when joining the strings.

    Returns:
    - str: The joined string.
    """
    # Check if the input is a list and the delimiter is a string
    if not isinstance(string_list, list) or not all(isinstance(s, str) for s in string_list):
        raise TypeError("Input must be a list of strings.")
    if not isinstance(delimiter, str):
        raise TypeError("Delimiter must be a string.")

    # Use the join method of the delimiter string to join the list of strings
    joined_string = delimiter.join(string_list)

    return joined_string

# Example usage:
if __name__ == "__main__":
    strings_to_join = ["Hello", "world", "this", "is", "a", "test"]
    delimiter_to_use = ", "
    print(join_strings(strings_to_join, delimiter_to_use))
```
