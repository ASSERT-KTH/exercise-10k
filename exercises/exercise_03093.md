## Exercise
Create a function to find the maximum contiguous subarray of a given list of numbers.

## Reference Solution
```python
def max_contiguous_subarray(numbers):
    """
    Find the maximum contiguous subarray of a given list of numbers.

    Args:
    numbers (list): A list of integers.

    Returns:
    list: The maximum contiguous subarray.
    int: The sum of the maximum contiguous subarray.

    Raises:
    ValueError: If the input list is empty.
    """
    if not numbers:
        raise ValueError("Input list cannot be empty")

    max_current = max_global = numbers[0]
    start = end = 0
    temp_start = 0

    for i in range(1, len(numbers)):
        if numbers[i] > max_current + numbers[i]:
            max_current = numbers[i]
            temp_start = i
        else:
            max_current += numbers[i]

        if max_current > max_global:
            max_global = max_current
            start = temp_start
            end = i

    return numbers[start:end+1], max_global


# Example usage:
numbers = [-2, -3, 4, -1, -2, 1, 5, -3]
max_subarray, max_sum = max_contiguous_subarray(numbers)
print(f"Maximum contiguous subarray: {max_subarray}")
print(f"Sum of the maximum contiguous subarray: {max_sum}")
```
