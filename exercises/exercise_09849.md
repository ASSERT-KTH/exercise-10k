## Exercise
Write a program that prints all prime numbers from 1 to a given number (n) using a while loop. The program should ask for user input to get the value of n.

## Reference Solution
```python
def get_user_input():
    """Get user input and validate that it's a positive integer"""
    while True:
        try:
            n = int(input("Enter a positive integer: "))
            if n <= 0:
                print("Please enter a positive integer.")
            else:
                return n
        except ValueError:
            print("That's not a valid integer.")

def is_prime(num):
    """Check if a number is prime"""
    if num < 2:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def print_primes(n):
    """Print all prime numbers from 1 to n"""
    i = 1
    while i <= n:
        if is_prime(i):
            print(i, end=" ")
        i += 1

def main():
    n = get_user_input()
    print("Prime numbers from 1 to", n, "are:")
    print_primes(n)

if __name__ == "__main__":
    main()
```
