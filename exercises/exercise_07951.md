## Exercise
Write a program to perform a binary search on a sorted array of integers.

## Reference Solution
```python
def binary_search(arr, target):
    """
    Perform binary search on a sorted array of integers.

    Args:
    - arr (list): A sorted list of integers.
    - target (int): The target integer to search for.

    Returns:
    - int: The index of the target integer in the array if found, -1 otherwise.
    """
    low, high = 0, len(arr) - 1

    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1

    return -1


# Example usage
if __name__ == "__main__":
    arr = [2, 5, 8, 12, 16, 23, 38, 56, 72, 91]
    target = 23

    result = binary_search(arr, target)

    if result != -1:
        print(f"Target {target} found at index {result}.")
    else:
        print(f"Target {target} not found in the array.")
```
