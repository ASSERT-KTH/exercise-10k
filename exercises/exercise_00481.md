## Exercise
Write a program that takes an array and an index as input, and returns the element at that index in the array. If the index is out of bounds, return an error message.

## Reference Solution
```python
// 1. Import modules
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

// 2. Define the Element function
function Element(arr, index) {
  if (index < arr.length) {
    return arr[index];
  } else {
    return "Invalid index";
  }
}

// 3. Read input from the user
rl.question('Enter an array (comma-separated): ', function(arr) {
  var array = arr.split(',').map(Number);
  rl.question('Enter an index: ', function(index) {
    var idx = parseInt(index);
    console.log(Element(array, idx));
    rl.close();
  });
});
```
