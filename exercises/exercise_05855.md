## Exercise
Write a program that thinks of an animal and then asks the user to guess it. After each guess, give the user a hint (e.g., 'too big', 'too small', 'correct'). The user should keep guessing until they guess the animal correctly

## Reference Solution
```python
import random

# List of animals with their average weights (in kg)
animals = {
    "mouse": 0.02,
    "cat": 4.5,
    "dog": 15,
    "cow": 400,
    "elephant": 5000,
    "giraffe": 830,
    "lion": 190,
    "tiger": 260
}

def get_animal():
    """Return a random animal from the list"""
    return random.choice(list(animals.keys()))

def get_hint(guess, target):
    """Return a hint based on the guess"""
    if animals[guess] < animals[target]:
        return "too small"
    elif animals[guess] > animals[target]:
        return "too big"
    else:
        return "correct"

def game():
    """Play the animal guessing game"""
    target_animal = get_animal()
    print("I'm thinking of an animal. Try to guess it!")
    
    while True:
        guess = input("Enter your guess: ").lower()
        
        if guess not in animals:
            print("Invalid animal. Please try again.")
            continue
        
        hint = get_hint(guess, target_animal)
        print(hint)
        
        if hint == "correct":
            print(f"Congratulations! You guessed the animal: {target_animal}")
            break

if __name__ == "__main__":
    game()
```
