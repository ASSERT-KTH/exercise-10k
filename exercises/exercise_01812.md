Create a program that converts time between different units, using conditional statements to determine the conversion rules

def convert_time(value, from_unit, to_unit):
    """
    Convert time between different units.

    Args:
    value (float): The value to convert.
    from_unit (str): The unit of the value.
    to_unit (str): The unit to convert to.

    Returns:
    float: The converted value.

    Raises:
    ValueError: If the from_unit or to_unit is not supported.
    """

    # Define the conversion factors
    conversion_factors = {
        'second': 1,
        'minute': 60,
        'hour': 3600,
        'day': 86400,
        'week': 604800,
        'month': 2629743,  # Approximate number of seconds in a month
        'year': 31556952  # Approximate number of seconds in a year
    }

    # Check if the from_unit and to_unit are supported
    if from_unit not in conversion_factors or to_unit not in conversion_factors:
        raise ValueError("Unsupported unit")

    # Convert the value to seconds
    if from_unit != 'second':
        value_in_seconds = value * conversion_factors[from_unit]
    else:
        value_in_seconds = value

    # Convert the value from seconds to the to_unit
    if to_unit != 'second':
        converted_value = value_in_seconds / conversion_factors[to_unit]
    else:
        converted_value = value_in_seconds

    return converted_value

        from_unit = 'month'
    elif choice == 7:
        from_unit = 'year'
    else:
        print("Invalid choice")
        return

    # Get the to_unit
    print("Select the unit to convert to:")
    print("1. Second")
    print("2. Minute")
    print("3. Hour")
    print("4. Day")
    print("5. Week")
    print("6. Month")
    print("7. Year")

    choice = int(input("Enter your choice (1-7): "))

    if choice == 1:
        to_unit = 'second'
    elif choice == 2:
        to_unit = 'minute'
    elif choice == 3:
        to_unit = 'hour'
    elif choice == 4:
        to_unit = 'day'
    elif choice == 5:
        to_unit = 'week'
    elif choice == 6:
        to_unit = 'month'
    elif choice == 7:
        to_unit = 'year'
    else:
        print("Invalid choice")
        return

    # Convert the time

        converted_value = convert_time(value, from_unit, to_unit)
        print(f"{value} {from_unit}(s) is equal to {converted_value} {to_unit}(s)")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
