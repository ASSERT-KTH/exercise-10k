## Exercise
Write a program that uses identity operators to check if two variables are the same object.

## Reference Solution
```python
# Define two lists
list1 = [1, 2, 3]
list2 = [1, 2, 3]

# Define two variables referencing the same list
list3 = list1
list4 = list1

def main():
    # Check if list1 and list2 are the same object
    print("Are list1 and list2 the same object? ", list1 is list2)
    
    # Check if list1 and list3 are the same object
    print("Are list1 and list3 the same object? ", list1 is list3)
    
    # Check if list1 and list4 are the same object
    print("Are list1 and list4 the same object? ", list1 is list4)
    
    # Check if list1 is not the same object as list2
    print("Is list1 not the same object as list2? ", list1 is not list2)
    
    # Check if list1 is not the same object as list3
    print("Is list1 not the same object as list3? ", list1 is not list3)

if __name__ == "__main__":
    main()
```
