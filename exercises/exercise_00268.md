## Exercise
Write a program that calculates the distance between two points in a 2D plane given their coordinates.

## Reference Solution
```python
import math

def calculate_distance(x1, y1, x2, y2):
    # Calculate the difference in x and y coordinates
    dx = x2 - x1
    dy = y2 - y1
    
    # Calculate the distance using the Pythagorean theorem
    distance = math.sqrt(dx ** 2 + dy ** 2)
    
    return distance

def main():
    # Get the coordinates of the two points from the user
    x1 = float(input("Enter the x-coordinate of the first point: "))
    y1 = float(input("Enter the y-coordinate of the first point: "))
    x2 = float(input("Enter the x-coordinate of the second point: "))
    y2 = float(input("Enter the y-coordinate of the second point: "))

    # Calculate and print the distance between the two points
    distance = calculate_distance(x1, y1, x2, y2)
    print(f"The distance between the two points is {distance:.2f}")

if __name__ == "__main__":
    main()
```
