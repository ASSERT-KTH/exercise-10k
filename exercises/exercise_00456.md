## Exercise
Write a program that asks the user a series of math questions, then keeps track of the user's score.

## Reference Solution
```python
import random
import operator

# Define the operators and their corresponding functions
operators = {
    '+': operator.add,
    '-': operator.sub,
    '*': operator.mul,
    '/': operator.truediv
}

def math_quiz(num_questions):
    """
    Asks the user a series of math questions and keeps track of their score.

    Args:
        num_questions (int): The number of questions to ask the user.

    Returns:
        None
    """
    score = 0

    for i in range(num_questions):
        # Generate two random numbers between 1 and 10
        num1 = random.randint(1, 10)
        num2 = random.randint(1, 10)

        # Choose a random operator
        op = random.choice(list(operators.keys()))

        # Calculate the correct answer
        correct_answer = operators[op](num1, num2)

        # Ask the user for their answer
        user_answer = float(input(f"What is {num1} {op} {num2}? "))

        # Check if the user's answer is correct
        if abs(user_answer - correct_answer) < 0.01:
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer was {correct_answer:.2f}\n")

    # Print the user's final score
    print(f"Quiz complete! Your final score is {score} out of {num_questions}")

def main():
    num_questions = int(input("How many questions would you like to answer? "))
    math_quiz(num_questions)

if __name__ == "__main__":
    main()
```
