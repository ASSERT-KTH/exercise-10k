## Exercise
Create a program that takes in a variable and determines its data type. The program should be able to identify integers, floats, strings, and booleans. It should then print out the data type of the variable.

## Reference Solution
```python
def determine_data_type(var):
    # Use the type() function to get the data type of the variable
    data_type = type(var)
    
    # Use if-elif-else statements to print out the data type
    if data_type == int:
        print("The data type of the variable is: integer")
    elif data_type == float:
        print("The data type of the variable is: float")
    elif data_type == str:
        print("The data type of the variable is: string")
    elif data_type == bool:
        print("The data type of the variable is: boolean")
    else:
        print("The data type of the variable is not recognized.")

# Example usage:
var1 = 10
var2 = 3.14
var3 = "Hello, World!"
var4 = True

determine_data_type(var1)
determine_data_type(var2)
determine_data_type(var3)
determine_data_type(var4)
```

def determine_data_type(var):
    data_types = {
        int: "integer",
        float: "float",
        str: "string",
        bool: "boolean"
    }
    
    data_type = type(var)
    if data_type in data_types:
        print(f"The data type of the variable is: {data_types[data_type]}")
    else:
        print("The data type of the variable is not recognized.")

# Example usage:
var1 = 10
var2 = 3.14
var3 = "Hello, World!"
var4 = True

determine_data_type(var1)
determine_data_type(var2)
