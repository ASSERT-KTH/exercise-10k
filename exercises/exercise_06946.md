## Exercise
Create a function that uses higher-order function composition to combine multiple functions

## Reference Solution
```python
def compose(*funcs):
    """
    Compose multiple functions together.

    Args:
        *funcs: Variable number of functions to compose.

    Returns:
        A new function that applies each function in sequence.
    """
    def composed(x):
        result = x
        for func in reversed(funcs):
            result = func(result)
        return result
    return composed

# Example usage:
def double(x):
    """Double the input."""
    return x * 2

def add_one(x):
    """Add one to the input."""
    return x + 1

def square(x):
    """Square the input."""
    return x ** 2

# Compose the functions
composed_func = compose(square, add_one, double)

# Test the composed function
print(composed_func(5))  # Output: (5*2 + 1)^2 = 121
```
