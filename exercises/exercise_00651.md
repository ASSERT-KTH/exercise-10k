## Exercise
Write a program that reads a story template with blanks from the user and fills in the blanks with user-provided words.

## Reference Solution
```python
def get_template():
    """Get the story template from the user."""
    template = input("Please enter a story template with blanks (use {} for blanks): ")
    return template

def get_words(num_blanks):
    """Get the words to fill in the blanks from the user."""
    words = []
    for i in range(num_blanks):
        word_type = input(f"Please enter a type of word (e.g. noun, verb, adjective, etc.) for blank {i+1}: ")
        word = input(f"Please enter a {word_type}: ")
        words.append(word)
    return words

def fill_in_blanks(template, words):
    """Fill in the blanks in the template with the given words."""
    story = template.format(*words)
    return story

def main():
    template = get_template()
    num_blanks = template.count("{}")
    words = get_words(num_blanks)
    story = fill_in_blanks(template, words)
    print("Here is the completed story:")
    print(story)

if __name__ == "__main__":
    main()
```
