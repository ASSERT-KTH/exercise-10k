## Exercise
Write a program to allocate and deallocate memory dynamically using pointers.

## Reference Solution
```python
#include <stdio.h>
#include <stdlib.h>

// Function to allocate memory dynamically
void* allocate_memory(size_t size) {
    void* ptr = malloc(size);
    if (ptr == NULL) {
        printf("Memory allocation failed\n");
        exit(1);
    }
    return ptr;
}

// Function to deallocate memory
void deallocate_memory(void* ptr) {
    free(ptr);
}

int main() {
    // Allocate memory for an integer
    int* int_ptr = (int*) allocate_memory(sizeof(int));
    *int_ptr = 10;
    printf("Allocated integer: %d\n", *int_ptr);

    // Allocate memory for a character array
    char* str_ptr = (char*) allocate_memory(10 * sizeof(char));
    strcpy(str_ptr, "Hello");
    printf("Allocated string: %s\n", str_ptr);

    // Deallocate memory
    deallocate_memory(int_ptr);
    deallocate_memory(str_ptr);

    return 0;
}
```
