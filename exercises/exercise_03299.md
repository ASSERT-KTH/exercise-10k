## Exercise
Write a program that plays the Simon Says game with the user using a while loop.

## Reference Solution
```python
import random

def simon_says_game():
    """
    Play the Simon Says game with the user.
    """
    print("Welcome to the Simon Says game!")
    print("I will give you commands, and you have to follow them only if they start with 'Simon says.'")
    print("If I don't say 'Simon says,' you should not follow the command.")
    print("Let's start!")

    score = 0
    while True:
        # Generate a random command
        commands = ["jump", "clap", "dance", "run", "hop"]
        command = random.choice(commands)

        # Randomly decide whether to say "Simon says"
        simon_says = random.choice([True, False])

        # Give the command to the user
        if simon_says:
            print("Simon says: " + command)
        else:
            print(command)

        # Get the user's response
        response = input("What do you do? (type 'follow' to follow the command, or 'ignore' to ignore it): ")

        # Check if the user's response is correct
        if simon_says and response.lower() == "follow":
            print("Correct! You followed the command.")
            score += 1
        elif not simon_says and response.lower() == "ignore":
            print("Correct! You ignored the command.")
            score += 1
        elif simon_says and response.lower() == "ignore":
            print("Incorrect! You should have followed the command.")
            break
        elif not simon_says and response.lower() == "follow":
            print("Incorrect! You should have ignored the command.")
            break

        print("Your current score is: " + str(score))

    print("Game over! Your final score is: " + str(score))

if __name__ == "__main__":
    simon_says_game()
```
