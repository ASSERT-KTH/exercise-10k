## Exercise
Write a program that takes a date as input and determines the day of the week.

## Reference Solution
```python
import datetime

def date_to_day_of_week(year, month, day):
    """
    Returns the day of the week for the given date.

    Args:
        year (int): The year of the date.
        month (int): The month of the date.
        day (int): The day of the date.

    Returns:
        str: The day of the week (e.g. Monday, Tuesday, etc.).
    """
    date = datetime.date(year, month, day)
    return date.strftime("%A")

def main():
    # Get user input
    year = int(input("Enter the year: "))
    month = int(input("Enter the month: "))
    day = int(input("Enter the day: "))

    # Get the day of the week
    day_of_week = date_to_day_of_week(year, month, day)

    # Print the result
    print(f"The day of the week for {year}-{month}-{day} is {day_of_week}.")

if __name__ == "__main__":
    main()
```
