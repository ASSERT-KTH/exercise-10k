## Exercise
Write a program to scramble a word and ask the user to unscramble it.

## Reference Solution
```python
import random

def scramble_word(word):
    """Scramble a word"""
    word_list = list(word)
    random.shuffle(word_list)
    return ''.join(word_list)

def check_answer(scrambled_word, correct_word, user_answer):
    """Check if the user's answer is correct"""
    if user_answer.lower() == correct_word.lower():
        return True
    else:
        return False

def main():
    words = {
        "apple": "A type of fruit",
        "car": "A type of vehicle",
        "house": "A type of building"
    }

    # Choose a random word from the dictionary
    correct_word = random.choice(list(words.keys()))

    # Scramble the word
    scrambled_word = scramble_word(correct_word)

    # Ask the user to unscramble the word
    print(f"Unscramble the word: {scrambled_word}")
    print(f"HINT: {words[correct_word]}")

    user_answer = input("Enter your answer: ")

    # Check if the user's answer is correct
    if check_answer(scrambled_word, correct_word, user_answer):
        print("Correct!")
    else:
        print(f"Sorry, the correct answer is {correct_word}")

if __name__ == "__main__":
    main()
```
