## Exercise
Create a simple quiz that asks the user a series of questions and prints out their score at the end.

## Reference Solution
```python
# Define a dictionary with questions, options, and answers
quiz_questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Who painted the Mona Lisa?": {
        "A": "Leonardo da Vinci",
        "B": "Michelangelo",
        "C": "Raphael",
        "D": "Caravaggio",
        "answer": "A"
    }
}

def run_quiz(questions):
    """Runs the quiz and returns the score"""
    score = 0
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        answer = input("Enter your answer (A, B, C, D): ")
        if answer.upper() == options["answer"]:
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer was {options['answer']}.\n")
    return score

def main():
    """Main function"""
    score = run_quiz(quiz_questions)
    print(f"Quiz finished! Your score is {score} out of {len(quiz_questions)}")

if __name__ == "__main__":
    main()
```
