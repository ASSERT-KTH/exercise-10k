## Exercise
Write a program that simulates a simple chatbot conversation

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary with responses to user inputs
responses = {
    "hello": ["Hi, how are you?", "Hello! What's up?", "Hey, how's it going?"],
    "hi": ["Hello! What's up?", "Hey, how's it going?", "Hi, how are you?"],
    "how are you": ["I'm doing great, thanks!", "I'm good, thanks for asking!", "I'm just a chatbot, I don't have feelings, but thanks for asking!"],
    "what's your name": ["I'm a simple chatbot, I don't have a name.", "You can call me Chatbot.", "My name is Simple Chatbot."],
    "what's the weather like": ["It's sunny today!", "It's cloudy outside.", "The weather is perfect today!"],
    "default": ["I didn't understand that.", "Can you please rephrase?", "I'm not sure what you mean."]
}

# Define a function to get the chatbot's response
def get_response(user_input):
    """Get the chatbot's response to the user's input."""
    user_input = user_input.lower()
    for key in responses:
        if key in user_input:
            return random.choice(responses[key])
    return random.choice(responses["default"])

# Define the main function
def main():
    print("Welcome to the simple chatbot conversation!")
    while True:
        user_input = input("You: ")
        if user_input.lower() == "quit":
            print("Chatbot: Goodbye!")
            break
        print("Chatbot:", get_response(user_input))

# Run the main function
if __name__ == "__main__":
    main()
```
