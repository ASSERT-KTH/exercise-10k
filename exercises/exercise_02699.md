## Exercise
Write a program that takes a time as input and converts it from 12-hour format to 24-hour format or vice versa.

## Reference Solution
```python
def convert_time(time_str, input_format):
    """
    Converts time from 12-hour format to 24-hour format or vice versa.

    Args:
        time_str (str): Time string in either 12-hour or 24-hour format.
        input_format (str): '12' for 12-hour format, '24' for 24-hour format.

    Returns:
        str: Converted time string.
    """
    if input_format == '12':
        # Convert 12-hour format to 24-hour format
        components = time_str.split()
        time_parts = components[0].split(':')
        hours = int(time_parts[0])
        minutes = time_parts[1]
        seconds = time_parts[2]

        if components[1] == 'PM' and hours != 12:
            hours += 12
        elif components[1] == 'AM' and hours == 12:
            hours = 0

        return f'{hours:02d}:{minutes}:{seconds}'
    elif input_format == '24':
        # Convert 24-hour format to 12-hour format
        time_parts = time_str.split(':')
        hours = int(time_parts[0])
        minutes = time_parts[1]
        seconds = time_parts[2]

        if hours == 0:
            return f'12:{minutes}:{seconds} AM'
        elif hours < 12:
            return f'{hours:02d}:{minutes}:{seconds} AM'
        elif hours == 12:
            return f'{hours:02d}:{minutes}:{seconds} PM'
        else:
            return f'{hours - 12:02d}:{minutes}:{seconds} PM'
    else:
        raise ValueError('Invalid input format. Use "12" for 12-hour format or "24" for 24-hour format.')


def main():
    time_str = input("Enter time string (e.g., 08:30:00 or 8:30:00 AM): ")
    input_format = input('Enter input format (12 or 24): ')

    try:
        print(convert_time(time_str, input_format))
    except ValueError as e:
        print(str(e))


if __name__ == '__main__':
    main()
```
