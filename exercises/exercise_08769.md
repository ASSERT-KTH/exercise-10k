## Exercise
Write a function to calculate the percentage of vowels in a given string.

## Reference Solution
```python
def calculate_vowel_percentage(input_string):
    """
    Calculate the percentage of vowels in a given string.

    Args:
        input_string (str): The input string to calculate the vowel percentage for.

    Returns:
        float: The percentage of vowels in the input string.
    """
    # Convert the input string to lowercase to handle case-insensitive vowel counting
    input_string = input_string.lower()
    
    # Initialize a counter for vowels
    vowel_count = 0
    
    # Iterate over each character in the input string
    for char in input_string:
        # Check if the character is a vowel
        if char in 'aeiou':
            # Increment the vowel counter
            vowel_count += 1
    
    # Calculate the percentage of vowels
    if len(input_string) == 0:
        # Handle the case where the input string is empty
        vowel_percentage = 0.0
    else:
        vowel_percentage = (vowel_count / len(input_string)) * 100
    
    return vowel_percentage

# Example usage
def main():
    input_string = "Hello, World!"
    vowel_percentage = calculate_vowel_percentage(input_string)
    print(f"The percentage of vowels in '{input_string}' is {vowel_percentage:.2f}%")

if __name__ == "__main__":
    main()
```
