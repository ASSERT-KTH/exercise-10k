## Exercise
Create a Weather Forecast program using variables and data structures.

## Reference Solution
```python
# Import required libraries
import datetime

# Define a class to store weather data
class WeatherData:
    def __init__(self, city, temperature, humidity, condition):
        self.city = city
        self.temperature = temperature
        self.humidity = humidity
        self.condition = condition

# Define a class to store forecast data
class ForecastData:
    def __init__(self, date, temperature, humidity, condition):
        self.date = date
        self.temperature = temperature
        self.humidity = humidity
        self.condition = condition

# Define a dictionary to store city weather data
city_weather_data = {
    "New York": WeatherData("New York", 20, 60, "Sunny"),
    "London": WeatherData("London", 15, 70, "Cloudy"),
    "Paris": WeatherData("Paris", 18, 65, "Rainy")
}

# Define a list to store forecast data
forecast_data = [
    ForecastData(datetime.date.today() + datetime.timedelta(days=1), 22, 55, "Sunny"),
    ForecastData(datetime.date.today() + datetime.timedelta(days=2), 20, 60, "Cloudy"),
    ForecastData(datetime.date.today() + datetime.timedelta(days=3), 18, 65, "Rainy"),
    ForecastData(datetime.date.today() + datetime.timedelta(days=4), 15, 70, "Windy"),
    ForecastData(datetime.date.today() + datetime.timedelta(days=5), 12, 75, "Snowy")
]

# Function to display current weather conditions
def display_current_weather(city):
    if city in city_weather_data:
        weather_data = city_weather_data[city]
        print(f"Weather in {weather_data.city}:")
        print(f"Temperature: {weather_data.temperature}°C")
        print(f"Humidity: {weather_data.humidity}%")
        print(f"Condition: {weather_data.condition}")
    else:
        print("City not found.")

# Function to display forecast
def display_forecast():
    print("Forecast for the next 5 days:")
    for forecast in forecast_data:
        print(f"Date: {forecast.date}")
        print(f"Temperature: {forecast.temperature}°C")
        print(f"Humidity: {forecast.humidity}%")
        print(f"Condition: {forecast.condition}")
        print()

# Function to search for weather conditions by city
def search_weather_by_city(city):
    if city in city_weather_data:
        display_current_weather(city)
    else:
        print("City not found.")

# Main function
def main():
    print("Weather Forecast Program")
    print("1. Display current weather conditions")
    print("2. Display forecast")
    print("3. Search for weather conditions by city")
    print("4. Exit")
    
    while True:
        choice = input("Enter your choice: ")
        
        if choice == "1":
            city = input("Enter city: ")
            display_current_weather(city)
        elif choice == "2":
            display_forecast()
        elif choice == "3":
            city = input("Enter city: ")
            search_weather_by_city(city)
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

# Run the main function
if __name__ == "__main__":
    main()
```
