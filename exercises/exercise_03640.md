## Exercise
Write a program to find the determinant of a given 2x2 or 3x3 matrix.

## Reference Solution
```python
import numpy as np

def calculate_determinant(matrix):
    """
    Calculate the determinant of a given 2x2 or 3x3 matrix.

    Args:
        matrix (list): A 2x2 or 3x3 matrix represented as a list of lists.

    Returns:
        float: The determinant of the input matrix.
    """
    # Convert the input matrix to a NumPy array
    matrix = np.array(matrix)

    # Check if the matrix is 2x2 or 3x3
    if matrix.shape == (2, 2):
        # Calculate the determinant of a 2x2 matrix
        determinant = matrix[0, 0] * matrix[1, 1] - matrix[0, 1] * matrix[1, 0]
    elif matrix.shape == (3, 3):
        # Calculate the determinant of a 3x3 matrix using the NumPy library
        determinant = np.linalg.det(matrix)
    else:
        raise ValueError("Matrix must be 2x2 or 3x3")

    return determinant

def main():
    # Example usage
    matrix_2x2 = [[1, 2], [3, 4]]
    matrix_3x3 = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]

    print("Determinant of 2x2 matrix:")
    print(calculate_determinant(matrix_2x2))

    print("\nDeterminant of 3x3 matrix:")
    print(calculate_determinant(matrix_3x3))

if __name__ == "__main__":
    main()
```
