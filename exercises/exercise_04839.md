## Exercise
Write a program to demonstrate explicit type conversion, including conversions between numbers and strings.

## Reference Solution
```python
# Explicit Type Conversion

# Number to Number
def number_to_number():
    """Demonstrate conversion between number types."""
    # Float to Integer
    float_num = 10.5
    int_num = int(float_num)
    print(f"Float to Integer: {float_num} -> {int_num}")

    # Integer to Float
    int_num = 10
    float_num = float(int_num)
    print(f"Integer to Float: {int_num} -> {float_num}")

    # Integer to Complex
    int_num = 10
    complex_num = complex(int_num)
    print(f"Integer to Complex: {int_num} -> {complex_num}")

    # Float to Complex
    float_num = 10.5
    complex_num = complex(float_num)
    print(f"Float to Complex: {float_num} -> {complex_num}")


# Number to String
def number_to_string():
    """Demonstrate conversion from number to string."""
    # Integer to String
    int_num = 10
    str_num = str(int_num)
    print(f"Integer to String: {int_num} -> {str_num}")

    # Float to String
    float_num = 10.5
    str_num = str(float_num)
    print(f"Float to String: {float_num} -> {str_num}")

    # Complex to String
    complex_num = 10 + 5j
    str_num = str(complex_num)
    print(f"Complex to String: {complex_num} -> {str_num}")


# String to Number
def string_to_number():
    """Demonstrate conversion from string to number."""
    # String to Integer
    str_num = "10"
    int_num = int(str_num)
    print(f"String to Integer: {str_num} -> {int_num}")

    # String to Float
    str_num = "10.5"
    float_num = float(str_num)
    print(f"String to Float: {str_num} -> {float_num}")

    # String to Complex
    str_num = "10+5j"
    complex_num = complex(str_num)
    print(f"String to Complex: {str_num} -> {complex_num}")


if __name__ == "__main__":
    print("Explicit Type Conversion")
    print("-------------------------")
    number_to_number()
    print("\n")
    number_to_string()
    print("\n")
    string_to_number()
```
