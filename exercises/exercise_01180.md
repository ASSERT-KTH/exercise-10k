## Exercise
Given a string and a width, justify the text to fit the width

def justify_text(text, width):
    """
    Justify the text to fit the given width.

    Args:
        text (str): The input text.
        width (int): The desired width of the justified text.

    Returns:
        list: A list of strings, where each string is a line of the justified text.
    """
    words = text.split()
    result = []
    current_line = []
    current_width = 0

    for word in words:
        if current_width + len(word) + len(current_line) > width:
            # Justify the current line
            result.append(justify_line(current_line, width))
            current_line = [word]
            current_width = len(word)
        else:
            current_line.append(word)
            current_width += len(word)

    # Justify the last line
    result.append(justify_last_line(current_line, width))

    return result


def justify_line(words, width):
    """
    Justify a single line of text.

    Args:
        words (list): A list of words in the line.
        width (int): The desired width of the line.

    Returns:
        str: The justified line of text.
    """
    total_width = sum(len(word) for word in words)
    num_spaces = width - total_width
    num_gaps = len(words) - 1

    if num_gaps == 0:
        return words[0] + ' ' * num_spaces

    base_spaces = num_spaces // num_gaps
    extra_spaces = num_spaces % num_gaps

    result = ''
    for i, word in enumerate(words):
        result += word
        if i < len(words) - 1:
            result += ' ' * (base_spaces + (1 if i < extra_spaces else 0))

    return result


def justify_last_line(words, width):
    """
    Justify the last line of text.

    Args:
        words (list): A list of words in the line.
        width (int): The desired width of the line.

    Returns:
        str: The justified last line of text.
    """
    result = ' '.join(words)
    return result + ' ' * (width - len(result))


# Example usage
text = "the quick brown fox jumps over the lazy dog"
width = 20
justified_text = justify_text(text, width)
for line in justified_text:
    print(line)
