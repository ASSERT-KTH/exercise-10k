## Exercise
Create a program to generate all possible anagrams of a given string. Use a for loop to iterate over the characters in the string and swap them with each other.

## Reference Solution
```python
def generate_anagrams(s):
    """
    Generate all possible anagrams of a given string.

    Args:
    s (str): The input string.

    Returns:
    list: A list of all possible anagrams.
    """
    # Base case: if the string has only one character, return it
    if len(s) == 1:
        return [s]

    # Initialize an empty list to store the anagrams
    anagrams = []

    # Iterate over each character in the string
    for i, char in enumerate(s):
        # Get the remaining characters
        remaining_chars = s[:i] + s[i+1:]

        # Generate anagrams for the remaining characters
        for anagram in generate_anagrams(remaining_chars):
            # Append the current character to each anagram
            anagrams.append(char + anagram)

    # Return the list of anagrams
    return anagrams


def main():
    # Test the function
    input_string = "abc"
    anagrams = generate_anagrams(input_string)

    # Print the anagrams
    print("Anagrams of '{}':".format(input_string))
    for i, anagram in enumerate(anagrams):
        print("{}. {}".format(i+1, anagram))


if __name__ == "__main__":
    main()
```
