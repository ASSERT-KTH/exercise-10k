## Exercise
Write a program that demonstrates array iteration techniques.

## Reference Solution
```python
# Array Iteration Techniques

def main():
    # Define an array
    array = [1, 2, 3, 4, 5]

    # **1. For Loop**
    print("For Loop:")
    for element in array:
        print(element)

    # **2. Enumerate**
    print("\nEnumerate:")
    for index, element in enumerate(array):
        print(f"Index: {index}, Element: {element}")

    # **3. Range-Based For Loop**
    print("\nRange-Based For Loop:")
    for i in range(len(array)):
        print(array[i])

    # **4. While Loop**
    print("\nWhile Loop:")
    i = 0
    while i < len(array):
        print(array[i])
        i += 1

    # **5. List Comprehension**
    print("\nList Comprehension:")
    squared_array = [element ** 2 for element in array]
    print(squared_array)

    # **6. Map Function**
    print("\nMap Function:")
    squared_array = list(map(lambda x: x ** 2, array))
    print(squared_array)

if __name__ == "__main__":
    main()
```
