## Exercise
Write a program that calculates the Grade Point Average (GPA) based on the grades and credits provided by the user.

## Reference Solution
```python
def calculate_gpa():
    """
    Calculates the Grade Point Average (GPA) based on the grades and credits provided by the user.
    """
    print("GPA Calculator Program")
    print("------------------------")

    # Initialize variables
    total_credits = 0
    total_points = 0

    # Get the number of courses from the user
    num_courses = int(input("Enter the number of courses: "))

    # Loop through each course
    for i in range(num_courses):
        print(f"\nCourse {i+1}:")
        grade = input("Enter the grade (A, A-, B+, B, B-, C+, C, C-, D+, D, D-, F): ")
        credits = float(input("Enter the credits: "))

        # Convert grade to GPA points
        grade_points = get_grade_points(grade)

        # Calculate total points and credits
        total_points += grade_points * credits
        total_credits += credits

    # Calculate GPA
    gpa = total_points / total_credits

    # Print GPA
    print(f"\nYour GPA is: {gpa:.2f}")

def get_grade_points(grade):
    """
    Converts a grade to GPA points.
    """
    grade_points = {
        'A': 4.0,
        'A-': 3.7,
        'B+': 3.3,
        'B': 3.0,
        'B-': 2.7,
        'C+': 2.3,
        'C': 2.0,
        'C-': 1.7,
        'D+': 1.3,
        'D': 1.0,
        'D-': 0.7,
        'F': 0.0
    }

    return grade_points.get(grade.upper(), 0.0)

# Main function
if __name__ == "__main__":
    calculate_gpa()
```
