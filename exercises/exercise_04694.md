## Exercise
Create a function that justifies a given text by wrapping it to a specified width and aligning the lines.

## Reference Solution

def justify_text(text, width):
    """
    Justify a given text by wrapping it to a specified width and aligning the lines.

    Args:
        text (str): The input text.
        width (int): The specified width.

    Returns:
        str: The justified text.
    """
    words = text.split()
    justified_text = ""
    current_line = []

    for word in words:
        # Check if the current word can be added to the current line
        if len(' '.join(current_line)) + len(word) + 1 <= width:
            current_line.append(word)
        else:
            # Justify the current line and add it to the justified text
            justified_text += justify_line(current_line, width) + "\n"
            current_line = [word]

    # Justify the last line and add it to the justified text
    justified_text += justify_last_line(current_line, width)

    return justified_text


def justify_line(words, width):
    """
    Justify a line of text by distributing the spaces evenly.

    Args:
        words (list): The list of words in the line.
        width (int): The specified width.

    Returns:
        str: The justified line.
    """
    total_chars = sum(len(word) for word in words)
    total_spaces = width - total_chars
    num_gaps = len(words) - 1

    if num_gaps == 0:
        return words[0] + " " * total_spaces

    spaces_per_gap = total_spaces // num_gaps
    extra_spaces = total_spaces % num_gaps

    justified_line = ""
    for i, word in enumerate(words):
        justified_line += word
        if i < num_gaps:
            justified_line += " " * spaces_per_gap
            if i < extra_spaces:
                justified_line += " "

    return justified_line


def justify_last_line(words, width):
    """
    Justify the last line of text by left-justifying it.

    Args:
        words (list): The list of words in the line.
        width (int): The specified width.

    Returns:
        str: The justified last line.
    """
    last_line = " ".join(words)
    return last_line + " " * (width - len(last_line))


# Example usage
text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
width = 50
print(justify_text(text, width))
