## Exercise
Write a program that takes a list as input, and returns its reverse.

## Reference Solution
```python
def reverse_list(input_list):
    """
    Returns the reverse of the input list.

    Args:
        input_list (list): The list to be reversed.

    Returns:
        list: The reversed list.
    """
    return input_list[::-1]

# Example usage:
if __name__ == "__main__":
    # Test the function with a sample list
    sample_list = [1, 2, 3, 4, 5]
    print("Original List:", sample_list)
    reversed_list = reverse_list(sample_list)
    print("Reversed List:", reversed_list)
```

def reverse_list_manual(input_list):
    """
    Returns the reverse of the input list (manual implementation).

    Args:
        input_list (list): The list to be reversed.

    Returns:
        list: The reversed list.
    """
    left = 0
    right = len(input_list) - 1
    while left < right:
        # Swap elements at left and right indices
        input_list[left], input_list[right] = input_list[right], input_list[left]
        left += 1
        right -= 1
    return input_list

# Example usage:
if __name__ == "__main__":
    # Test the function with a sample list
    sample_list = [1, 2, 3, 4, 5]
    print("Original List:", sample_list)
    reversed_list = reverse_list_manual(sample_list)
    print("Reversed List:", reversed_list)
