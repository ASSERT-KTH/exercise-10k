## Exercise
Write a program that takes a list and an element as input, and returns the index of the first occurrence of the element in the list.

## Reference Solution
```python
# Example function to find the index of the first occurrence of an element in a list
def find_index(lst, elem):
    try:
        index = lst.index(elem)
    except ValueError:
        index = None
    return index

# Another example function to find the index of the first occurrence of an element in a list
def find_index(lst, elem):
    for i, x in enumerate(lst):
        if x == elem:
            return i
    return None

# Example usage:
list1 = ['a', 'b', 'c', 'd', 'e']
element = 'c'
index = find_index(list1, element)
if index is not None:
    print(f"The index of {element} in {list1} is {index}.")
else:
    print(f"{element} is not found in {list1}.")

# Another example usage:
list2 = [1, 2, 3, 4, 5]
element = 3
index = find_index(list2, element)
if index is not None:
    print(f"The index of {element} in {list2} is {index}.")
else:
    print(f"{element} is not found in {list2}.")
```
