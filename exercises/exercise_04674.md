## Exercise
Write a program that takes a string as input and returns the least frequent character in the string.

## Reference Solution
```python
from collections import Counter

def least_frequent_char(s):
    """
    Returns the least frequent character in the input string.

    Args:
        s (str): The input string.

    Returns:
        str: The least frequent character in the string.
    """
    # Remove whitespace and convert to lowercase
    s = s.replace(" ", "").lower()

    # Count the frequency of each character
    char_count = Counter(s)

    # Find the minimum frequency
    min_freq = min(char_count.values())

    # Find all characters with the minimum frequency
    least_frequent_chars = [char for char, freq in char_count.items() if freq == min_freq]

    # Return the first least frequent character
    return least_frequent_chars[0]

# Example usage
def main():
    input_str = "Hello, World!"
    least_frequent = least_frequent_char(input_str)
    print(f"The least frequent character in '{input_str}' is: {least_frequent}")

if __name__ == "__main__":
    main()
```
