## Exercise
Write a program that uses dynamic programming to solve a given problem. The problem should be: find the minimum number of coins needed to make change for a given amount.

def min_coins(coins, amount):
    """
    Find the minimum number of coins needed to make change for a given amount.

    Args:
        coins (list): A list of coin denominations.
        amount (int): The target amount.

    Returns:
        int: The minimum number of coins needed.
    """
    # Create a table to store the minimum number of coins needed for each amount
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # Base case: 0 coins needed to make 0 amount

    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] = min(dp[i], dp[i - coin] + 1)

    return dp[amount] if dp[amount] != float('inf') else -1

# Example usage:
coins = [1, 2, 5]
amount = 11
print(min_coins(coins, amount))  # Output: 3 (11 = 5 + 5 + 1)
