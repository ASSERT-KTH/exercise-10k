Write a program to identify whether a given string is a valid LEI (Local Event Identifier) using a while loop.

def is_valid_lei(lei: str) -> bool:
    """
    Validate a given string as a Local Event Identifier (LEI).

    Args:
    lei (str): The string to be validated.

    Returns:
    bool: True if the string is a valid LEI, False otherwise.
    """
    # Check if the length of the string is 20 characters
    if len(lei) != 20:
        return False

    # Check if the string contains only alphanumeric characters
    if not lei.isalnum():
        return False

    # Check the prefix (first 4 characters)
    prefix = lei[:4]
    # Check if the prefix is a valid alphanumeric string
    if not prefix.isalnum():
        return False

    # Check the entity identifier (next 9 characters)
    entity_id = lei[4:13]
    # Check if the entity identifier is a valid alphanumeric string
    if not entity_id.isalnum():
        return False

    # Check the check digit (next 2 characters)
    check_digit = lei[13:15]
    # Check if the check digit is a valid alphanumeric string
    if not check_digit.isalnum():
        return False

    # Check the reserved field (last 5 characters)
    reserved_field = lei[15:]
    # Check if the reserved field is a valid alphanumeric string
    if not reserved_field.isalnum():
        return False

    # Check the check digit using the ISO 17442 algorithm
    check_digit_calculated = calculate_check_digit(lei[:18])
    if check_digit != check_digit_calculated:
        return False

    return True


def calculate_check_digit(lei_prefix: str) -> str:
    """
    Calculate the check digit using the ISO 17442 algorithm.

    Args:
    lei_prefix (str): The first 18 characters of the LEI.

    Returns:
    str: The calculated check digit.
    """
    weights = [1, 3, 7, 5, 3, 7, 5, 3, 7, 5, 3, 7, 5, 3, 7, 5, 3, 7]
    sum_of_products = 0
    for i in range(18):
        sum_of_products += weights[i] * get_digit_value(lei_prefix[i])

    remainder = sum_of_products % 97
    check_digit = 98 - remainder
    if check_digit < 10:
        return str(check_digit)
    else:
        return str(check_digit)[1]


def get_digit_value(char: str) -> int:
    """
    Get the digit value of a character.

    Args:
    char (str): The character to get the digit value for.

    Returns:
    int: The digit value of the character.
    """
    if char.isdigit():
        return int(char)
    elif char.isupper():
        return ord(char) - ord('A') + 10
    else:
        raise ValueError("Invalid character")


def main():
    lei = input("Enter an LEI: ")
    i = 0
    while i < len(lei):
        if not lei[i].isalnum():
            print("Invalid LEI: Contains non-alphanumeric characters")
            break
        i += 1
    else:
        if is_valid_lei(lei):
            print("Valid LEI")
        else:
            print("Invalid LEI")


if __name__ == "__main__":
    main()
