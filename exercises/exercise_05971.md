## Exercise
Write a program that takes an array and an index as input, and inserts a new element at the specified index.

## Reference Solution

#include <stdio.h>

void insert(int array[], int index, int value) {
    for (int i = 4; i > index; i--) {
        array[i] = array[i - 1];
    }
    array[index] = value;
}

int main() {
    int array[5] = {1, 2, 3, 4, 5};
    int index = 2;
    int value = 10;

    insert(array, index, value);

    for (int i = 0; i < 5; i++) {
        printf("%d\n", array[i]);
    }

    return 0;
}
