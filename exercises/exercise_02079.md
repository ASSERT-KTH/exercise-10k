## Exercise
Write a program that finds the longest increasing subsequence of a given list of numbers.

## Reference Solution

def longest_increasing_subsequence(numbers):
    """
    Find the longest increasing subsequence of a given list of numbers.

    Args:
    numbers (list): A list of integers.

    Returns:
    list: The longest increasing subsequence.
    """
    if not numbers:
        return []

    # Initialize lists to store the lengths and predecessors of the longest increasing subsequences
    lengths = [1] * len(numbers)
    predecessors = [None] * len(numbers)

    # Compute the lengths and predecessors of the longest increasing subsequences
    for i in range(1, len(numbers)):
        for j in range(i):
            if numbers[i] > numbers[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                predecessors[i] = j

    # Find the index of the longest increasing subsequence
    max_length_idx = max(range(len(lengths)), key=lambda i: lengths[i])

    # Reconstruct the longest increasing subsequence
    subsequence = []
    while max_length_idx is not None:
        subsequence.append(numbers[max_length_idx])
        max_length_idx = predecessors[max_length_idx]

    # Return the longest increasing subsequence in the correct order
    return subsequence[::-1]

    print("Input list:", numbers)
    print("Longest increasing subsequence:", longest_increasing_subsequence(numbers))
