## Exercise
Write a program that generates a random number between 1 and 100, then asks the user to guess the number. After each guess, print out a hint, such as 'too high' or 'too low'.

## Reference Solution
```python
import random

def generate_random_number(min_value, max_value):
    """Generate a random number between min_value and max_value (inclusive)"""
    return random.randint(min_value, max_value)

def get_user_guess(min_value, max_value):
    """Get the user's guess and validate it"""
    while True:
        try:
            guess = int(input(f"Guess a number between {min_value} and {max_value}: "))
            if min_value <= guess <= max_value:
                return guess
            else:
                print(f"Please enter a number between {min_value} and {max_value}.")
        except ValueError:
            print("Invalid input. Please enter a whole number.")

def provide_hint(guess, target):
    """Provide a hint based on the user's guess"""
    if guess < target:
        return "Too low"
    elif guess > target:
        return "Too high"
    else:
        return "Congratulations, you've guessed the number!"

def main():
    min_value = 1
    max_value = 100
    target_number = generate_random_number(min_value, max_value)
    attempts = 0

    print("Welcome to the Random Number Guessing Game!")
    print(f"I'm thinking of a number between {min_value} and {max_value}.")

    while True:
        guess = get_user_guess(min_value, max_value)
        attempts += 1
        hint = provide_hint(guess, target_number)
        print(hint)

        if hint.startswith("Congratulations"):
            print(f"It took you {attempts} attempts to guess the number.")
            break

if __name__ == "__main__":
    main()
```
