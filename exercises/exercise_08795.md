## Exercise
Write a program that takes the bases and height of a trapezoid as input and returns its area.

## Reference Solution
```python
def trapezoid_area(base1, base2, height):
    """
    Calculate the area of a trapezoid given the lengths of its two bases and its height.

    Args:
        base1 (float): The length of the first base.
        base2 (float): The length of the second base.
        height (float): The height of the trapezoid.

    Returns:
        float: The area of the trapezoid.
    """
    if base1 <= 0 or base2 <= 0 or height <= 0:
        raise ValueError("All input values must be positive numbers.")

    # The formula for the area of a trapezoid is: area = (base1 + base2) * height / 2
    area = (base1 + base2) * height / 2

    return area


def main():
    # Get the bases and height from the user
    base1 = float(input("Enter the length of the first base: "))
    base2 = float(input("Enter the length of the second base: "))
    height = float(input("Enter the height of the trapezoid: "))

    try:
        # Calculate and print the area
        area = trapezoid_area(base1, base2, height)
        print(f"The area of the trapezoid is: {area}")
    except ValueError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
