## Exercise
Create a program that replaces all occurrences of a substring in a string with another substring.

## Reference Solution
```python
def replace_substring(original_string, substring_to_replace, replacement_substring):
    """
    Replaces all occurrences of a substring in a string with another substring.

    Args:
        original_string (str): The string in which to replace the substring.
        substring_to_replace (str): The substring to be replaced.
        replacement_substring (str): The substring to replace with.

    Returns:
        str: The modified string with all occurrences of the substring replaced.
    """
    return original_string.replace(substring_to_replace, replacement_substring)


def main():
    # Example usage:
    original_string = "Hello, world! Hello again!"
    substring_to_replace = "Hello"
    replacement_substring = "Hi"

    modified_string = replace_substring(original_string, substring_to_replace, replacement_substring)
    print(f"Original String: {original_string}")
    print(f"Modified String: {modified_string}")


if __name__ == "__main__":
    main()
```
