## Exercise
Write a function that uses a context manager to manage resources

## Reference Solution
```python
from contextlib import contextmanager

@contextmanager
def managed_file(filename):
    """Context manager for managing a file resource."""
    try:
        # Open the file in read mode
        file = open(filename, 'r')
        # Yield the file to the context
        yield file
    except FileNotFoundError:
        print(f"File '{filename}' not found.")
    except Exception as e:
        print(f"An error occurred: {e}")
    finally:
        # Close the file if it was opened successfully
        if 'file' in locals() and not file.closed:
            file.close()

def main():
    filename = 'example.txt'
    with managed_file(filename) as file:
        # Read and print the contents of the file
        print(file.read())

if __name__ == "__main__":
    main()
```
