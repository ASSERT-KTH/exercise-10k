## Exercise
Write a program to calculate the compound interest on an investment.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, time, frequency):
    """
    Calculate compound interest on an investment.

    Args:
        principal (float): Initial investment amount
        rate (float): Annual interest rate (in decimal)
        time (int): Time period (in years)
        frequency (int): Compounding frequency (1 = annually, 2 = semi-annually, 4 = quarterly, 12 = monthly)

    Returns:
        float: Total amount after compound interest
    """
    total_amount = principal * (1 + rate / frequency) ** (frequency * time)
    return total_amount

def main():
    print("Compound Interest Calculator")
    print("--------------------------------")

    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))
    frequency = int(input("Enter the compounding frequency (1 = annually, 2 = semi-annually, 4 = quarterly, 12 = monthly): "))

    total_amount = calculate_compound_interest(principal, rate, time, frequency)
    interest_earned = total_amount - principal

    print(f"\nTotal Amount: ${total_amount:.2f}")
    print(f"Interest Earned: ${interest_earned:.2f}")

if __name__ == "__main__":
    main()
```
