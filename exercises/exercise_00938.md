## Exercise
Write a program that creates a simple dictionary using if/else statement

## Reference Solution
```python
# Define a dictionary
dictionary = {
    "hello": "a greeting",
    "world": "the earth and its inhabitants",
    "python": "a programming language"
}

def get_definition(word):
    """
    Returns the definition of a word if it exists in the dictionary.
    """
    if word.lower() in dictionary:
        return dictionary[word.lower()]
    else:
        return "Word not found in dictionary."

def main():
    print("Simple Dictionary Program")
    print("-------------------------")
    
    while True:
        word = input("Enter a word (or 'quit' to exit): ")
        
        if word.lower() == "quit":
            break
        elif word.lower() in dictionary:
            print(f"{word.capitalize()}: {dictionary[word.lower()]}")
        else:
            print(f"{word.capitalize()}: Word not found in dictionary.")

if __name__ == "__main__":
    main()
```
