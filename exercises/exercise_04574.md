## Exercise
Write a function that takes a dictionary and returns a new dictionary with the key-value pairs inverted

## Reference Solution
```python
def invert_dictionary(input_dict):
    """
    Inverts a dictionary, swapping its keys and values.

    Args:
        input_dict (dict): The dictionary to invert.

    Returns:
        dict: A new dictionary with the key-value pairs inverted.

    Raises:
        ValueError: If the input dictionary has non-unique values.
    """

    # Check for non-unique values
    if len(input_dict.values()) != len(set(input_dict.values())):
        raise ValueError("Input dictionary has non-unique values")

    # Invert the dictionary
    inverted_dict = {v: k for k, v in input_dict.items()}

    return inverted_dict


# Example usage
if __name__ == "__main__":
    input_dict = {"a": 1, "b": 2, "c": 3}
    print("Original Dictionary:", input_dict)

    inverted_dict = invert_dictionary(input_dict)
    print("Inverted Dictionary:", inverted_dict)
```
