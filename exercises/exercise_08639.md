## Exercise
Create a function that calculates the average of all numbers in a given list.

## Reference Solution
```python
def calculate_average(num_list):
    """
    Calculate the average of all numbers in a given list.

    Args:
        num_list (list): A list of numbers.

    Returns:
        float: The average of all numbers in the list.

    Raises:
        ValueError: If the input list is empty.
        TypeError: If the input is not a list or if the list contains non-numeric values.
    """
    # Check if input is a list
    if not isinstance(num_list, list):
        raise TypeError("Input must be a list.")

    # Check if list is not empty
    if len(num_list) == 0:
        raise ValueError("Input list cannot be empty.")

    # Check if all elements in the list are numeric
    if not all(isinstance(num, (int, float)) for num in num_list):
        raise TypeError("All elements in the list must be numbers.")

    # Calculate the sum of all numbers in the list
    total_sum = sum(num_list)

    # Calculate the average
    average = total_sum / len(num_list)

    return average


# Example usage
if __name__ == "__main__":
    numbers = [1, 2, 3, 4, 5]
    try:
        avg = calculate_average(numbers)
        print(f"The average of the numbers is: {avg}")
    except (ValueError, TypeError) as e:
        print(f"Error: {e}")
```
