## Exercise
Write a function that finds all possible subsequences of a given string.

## Reference Solution
```python
def generate_subsequences(s):
    """
    Generate all possible subsequences of a given string.

    Args:
        s (str): The input string.

    Returns:
        list: A list of all possible subsequences.
    """
    # Base case: if the string is empty, return a list with an empty string
    if not s:
        return [""]

    # Recursive case: get all subsequences of the substring without the first character
    subsequences = generate_subsequences(s[1:])

    # Add the first character to each subsequence
    new_subsequences = [s[0] + sub for sub in subsequences]

    # Combine the new subsequences with the original ones
    return subsequences + new_subsequences


def main():
    # Example usage:
    input_string = "abc"
    all_subsequences = generate_subsequences(input_string)

    print("Input String:", input_string)
    print("All Subsequences:")
    for i, sub in enumerate(all_subsequences):
        print(f"{i+1}. {sub}")


if __name__ == "__main__":
    main()
```

def generate_subsequences_iterative(s):
    n = len(s)
    total = 1 << n  # 2^n

    subsequences = []
    for i in range(total):
        sub = "".join([s[j] for j in range(n) if (i & (1 << j))])
        subsequences.append(sub)

    return subsequences


def main_iterative():
    input_string = "abc"
    all_subsequences = generate_subsequences_iterative(input_string)

    print("Input String:", input_string)
    print("All Subsequences:")
    for i, sub in enumerate(all_subsequences):
        print(f"{i+1}. {sub}")


if __name__ == "__main__":
