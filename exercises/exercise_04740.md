## Exercise
Write a function that uses asynchronous programming to perform multiple tasks concurrently.

## Reference Solution
```python
import asyncio
import time

async def make_breakfast():
    """Simulate making breakfast"""
    print("Starting to make breakfast")
    await asyncio.sleep(2)  # Simulate cooking time
    print("Breakfast is ready")

async def make_lunch():
    """Simulate making lunch"""
    print("Starting to make lunch")
    await asyncio.sleep(3)  # Simulate cooking time
    print("Lunch is ready")

async def make_dinner():
    """Simulate making dinner"""
    print("Starting to make dinner")
    await asyncio.sleep(4)  # Simulate cooking time
    print("Dinner is ready")

async def main():
    """Run all tasks concurrently"""
    start_time = time.time()
    await asyncio.gather(make_breakfast(), make_lunch(), make_dinner())
    print(f"Total time taken: {time.time() - start_time} seconds")

# Run the main function
asyncio.run(main())
```
