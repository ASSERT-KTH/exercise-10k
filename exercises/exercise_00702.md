## Exercise
Write a function to access the value of a specific key in a dictionary

## Reference Solution
```python
def get_dict_key_value(dictionary, key):
    """
    Retrieves the value associated with a specific key in a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key (any): The key to search for.

    Returns:
        any: The value associated with the key if found, otherwise None.
    """
    return dictionary.get(key)

# Example usage:
my_dict = {"name": "John", "age": 30, "city": "New York"}
key_to_find = "age"

value = get_dict_key_value(my_dict, key_to_find)
print(f"The value of '{key_to_find}' is: {value}")
```

```python
def get_dict_key_value(dictionary, key):
    """
    Retrieves the value associated with a specific key in a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key (any): The key to search for.

    Returns:
        any: The value associated with the key if found.

    Raises:
        KeyError: If the key is not found in the dictionary.
    """
    return dictionary[key]

# Example usage with error handling:
my_dict = {"name": "John", "age": 30, "city": "New York"}
key_to_find = "age"

try:
    value = get_dict_key_value(my_dict, key_to_find)
    print(f"The value of '{key_to_find}' is: {value}")
except KeyError:
    print(f"The key '{key_to_find}' was not found in the dictionary.")
```
