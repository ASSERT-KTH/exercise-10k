## Exercise
Develop a program that determines a hotel room type based on the number of guests and room preferences. The program should use if/else statements to determine the room type.

## Reference Solution
```python
def determine_room_type(num_guests, room_preferences):
    """
    Determine a hotel room type based on the number of guests and room preferences.

    Parameters:
    num_guests (int): The number of guests.
    room_preferences (dict): A dictionary containing room preferences.
        - 'smoking': bool, whether the guest prefers a smoking room
        - 'view': str, the preferred view (e.g. 'ocean', 'city', 'garden')
        - 'bed_type': str, the preferred bed type (e.g. 'king', 'queen', 'twin')

    Returns:
    str: The determined hotel room type.
    """

    # Check the number of guests
    if num_guests == 1:
        room_type = 'Single'
    elif num_guests == 2:
        room_type = 'Double'
    elif 3 <= num_guests <= 4:
        room_type = 'Family'
    else:
        room_type = 'Suite'

    # Consider room preferences
    if room_preferences.get('smoking', False):
        room_type += ' Smoking'
    if room_preferences.get('view'):
        room_type += f' with {room_preferences["view"].capitalize()} View'
    if room_preferences.get('bed_type'):
        room_type += f' and {room_preferences["bed_type"].capitalize()} Bed'

    return room_type


def main():
    # Example usage
    num_guests = 3
    room_preferences = {
        'smoking': False,
        'view': 'ocean',
        'bed_type': 'king'
    }

    room_type = determine_room_type(num_guests, room_preferences)
    print(f'Recommended room type: {room_type}')


if __name__ == '__main__':
    main()
```
