## Exercise
Create a program that simulates a phonebook using a dictionary. The program should have the following functionality: add a new contact, remove a contact, update a contact's phone number, and search for a contact by name. The program should continue to run until the user chooses to exit.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone_number):
        """Add a new contact to the phonebook."""
        if name in self.contacts:
            print("Contact already exists. Please try again with a different name.")
        else:
            self.contacts[name] = phone_number
            print(f"Contact '{name}' added successfully.")

    def remove_contact(self, name):
        """Remove a contact from the phonebook."""
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact '{name}' removed successfully.")
        else:
            print(f"Contact '{name}' not found.")

    def update_contact(self, name, new_phone_number):
        """Update a contact's phone number."""
        if name in self.contacts:
            self.contacts[name] = new_phone_number
            print(f"Contact '{name}' updated successfully.")
        else:
            print(f"Contact '{name}' not found.")

    def search_contact(self, name):
        """Search for a contact by name."""
        if name in self.contacts:
            print(f"Phone number of '{name}': {self.contacts[name]}")
        else:
            print(f"Contact '{name}' not found.")

    def display_contacts(self):
        """Display all contacts in the phonebook."""
        if self.contacts:
            print("Contacts:")
            for name, phone_number in self.contacts.items():
                print(f"{name}: {phone_number}")
        else:
            print("No contacts in the phonebook.")

def main():
    phonebook = Phonebook()
    while True:
        print("\nPhonebook Menu:")
        print("1. Add a new contact")
        print("2. Remove a contact")
        print("3. Update a contact's phone number")
        print("4. Search for a contact")
        print("5. Display all contacts")
        print("6. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            name = input("Enter name: ")
            phone_number = input("Enter phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter name: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter name: ")
            new_phone_number = input("Enter new phone number: ")
            phonebook.update_contact(name, new_phone_number)
        elif choice == "4":
            name = input("Enter name: ")
            phonebook.search_contact(name)
        elif choice == "5":
            phonebook.display_contacts()
        elif choice == "6":
            print("Exiting phonebook.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
