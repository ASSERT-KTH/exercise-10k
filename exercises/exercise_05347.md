## Exercise
Create a function that takes two 2D arrays (matrices) as input, and returns the product of the two matrices.

## Reference Solution
```python
import numpy as np

def matrix_multiply(A, B):
    """
    Compute the product of two matrices A and B.

    Args:
    A (numpy.ndarray): The first matrix.
    B (numpy.ndarray): The second matrix.

    Returns:
    numpy.ndarray: The product of A and B.

    Raises:
    ValueError: If the number of columns in A does not match the number of rows in B.
    """

    # Check if the input matrices are numpy arrays
    if not isinstance(A, np.ndarray) or not isinstance(B, np.ndarray):
        raise ValueError("Input matrices must be numpy arrays")

    # Check if the input matrices are 2D
    if len(A.shape) != 2 or len(B.shape) != 2:
        raise ValueError("Input matrices must be 2D")

    # Check if the number of columns in A matches the number of rows in B
    if A.shape[1] != B.shape[0]:
        raise ValueError("Number of columns in A must match the number of rows in B")

    # Compute the matrix product using numpy's matmul function
    return np.matmul(A, B)


# Example usage
if __name__ == "__main__":
    # Define two example matrices
    A = np.array([[1, 2], [3, 4]])
    B = np.array([[5, 6], [7, 8]])

    # Compute and print the matrix product
    product = matrix_multiply(A, B)
    print("Matrix A:")
    print(A)
    print("Matrix B:")
    print(B)
    print("Matrix Product:")
    print(product)
```
