## Exercise
Create a program that uses a for loop to simulate the Game of Life. The program should display the initial configuration and then simulate the next 5 generations.

## Reference Solution
```python
import pygame
import numpy as np
import copy

# Window size
WIDTH, HEIGHT = 800, 600

# Cell size
CELL_SIZE = 10

# Colors
ALIVE_COLOR = (255, 255, 255)
DEAD_COLOR = (0, 0, 0)

class GameOfLife:
    def __init__(self, width, height, cell_size):
        self.width = width
        self.height = height
        self.cell_size = cell_size
        self.grid = np.random.choice([0,1], size=(width // cell_size, height // cell_size), p=[0.5, 0.5])

    def draw(self, screen):
        screen.fill(DEAD_COLOR)
        for i in range(self.grid.shape[0]):
            for j in range(self.grid.shape[1]):
                if self.grid[i, j] == 1:
                    pygame.draw.rect(screen, ALIVE_COLOR, (i * self.cell_size, j * self.cell_size, self.cell_size, self.cell_size))

    def count_neighbors(self, x, y):
        count = 0
        for i in range(-1, 2):
            for j in range(-1, 2):
                col = (x + i + self.grid.shape[0]) % self.grid.shape[0]
                row = (y + j + self.grid.shape[1]) % self.grid.shape[1]
                count += self.grid[col, row]
        count -= self.grid[x, y]
        return count

    def next_generation(self):
        new_grid = copy.deepcopy(self.grid)
        for i in range(self.grid.shape[0]):
            for j in range(self.grid.shape[1]):
                live_neighbors = self.count_neighbors(i, j)
                if self.grid[i, j] == 1 and (live_neighbors < 2 or live_neighbors > 3):
                    new_grid[i, j] = 0
                if self.grid[i, j] == 0 and live_neighbors == 3:
                    new_grid[i, j] = 1
        self.grid = new_grid

def main():
    pygame.init()
    screen = pygame.display.set_mode((WIDTH, HEIGHT))
    clock = pygame.time.Clock()
    game = GameOfLife(WIDTH, HEIGHT, CELL_SIZE)

    # Display initial configuration
    game.draw(screen)
    pygame.display.update()
    pygame.time.wait(1000)

    # Simulate next 5 generations
    for _ in range(5):
        game.next_generation()
        game.draw(screen)
        pygame.display.update()
        pygame.time.wait(1000)

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False
        clock.tick(60)

    pygame.quit()

if __name__ == "__main__":
    main()
```
