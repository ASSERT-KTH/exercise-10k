## Exercise
Write a program to decrypt a given string that was encrypted using a Caesar cipher.

## Reference Solution
```python
def caesar_decrypt(ciphertext, shift):
    """
    Decrypts a given string that was encrypted using a Caesar cipher.

    Args:
        ciphertext (str): The encrypted string.
        shift (int): The shift used for encryption.

    Returns:
        str: The decrypted string.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    decrypted_text = ''

    for char in ciphertext:
        if char.isalpha():
            # Find the position in the alphabet for the character
            position = alphabet.index(char.lower())

            # Apply the shift in reverse
            new_position = (position - shift) % 26

            # Add the decrypted character to the result
            if char.isupper():
                decrypted_text += alphabet[new_position].upper()
            else:
                decrypted_text += alphabet[new_position]
        else:
            # Non-alphabet characters are left unchanged
            decrypted_text += char

    return decrypted_text


def main():
    ciphertext = input("Enter the encrypted string: ")
    shift = int(input("Enter the shift used for encryption: "))

    decrypted_text = caesar_decrypt(ciphertext, shift)
    print("Decrypted string:", decrypted_text)


if __name__ == "__main__":
    main()
```
