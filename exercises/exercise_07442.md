## Exercise
Implement an optimized version of the Bubble Sort algorithm that can sort a list of integers in ascending order. The optimization should include stopping the algorithm if no swaps are made in a pass, indicating that the list is already sorted.

## Reference Solution
```python
def bubble_sort(A):
    """
    Sorts a list of integers in ascending order using the Bubble Sort algorithm.
    
    Args:
        A (list): A list of integers to be sorted.
    
    Returns:
        list: The sorted list of integers.
    """
    n = len(A)
    
    # Traverse through all array elements
    for i in range(n):
        
        # Last i elements are already in place
        for j in range(0, n - i - 1):
            
            # Traverse the array from 0 to n-i-1
            # Swap if the element found is greater than the next element
            if A[j] > A[j + 1]:
                A[j], A[j + 1] = A[j + 1], A[j]
                
    return A

# Example usage:
arr = [64, 34, 25, 12, 22, 11, 90]
print("Original array:", arr)
print("Sorted array:", bubble_sort(arr))
```

```python
def bubble_sort(A):
    """
    Sorts a list of integers in ascending order using the Bubble Sort algorithm.
    
    Args:
        A (list): A list of integers to be sorted.
    
    Returns:
        list: The sorted list of integers.
    """
    n = len(A)
    swapped = True
    
    # Traverse through all array elements
    while swapped:
        swapped = False
        for j in range(0, n - 1):
            if A[j] > A[j + 1]:
                A[j], A[j + 1] = A[j + 1], A[j]
                swapped = True
        n -= 1
                
    return A

# Example usage:
arr = [64, 34, 25, 12, 22, 11, 90]
print("Original array:", arr)
print("Sorted array:", bubble_sort(arr))
```
