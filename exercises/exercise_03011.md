def parse_string_to_number(input_string):
    """

        input_string (str): The input string to be parsed.

    Returns:

    """

    # First, try to parse the string into an integer
    try:
        # Attempt to convert the string to an integer

    except ValueError:

        try:
            # Attempt to convert the string to a float

        except ValueError:
            # If the string cannot be converted to a float, return None
            return None
