## Exercise
Create a program that takes two numbers and an arithmetic operation as input and performs the operation. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
def arithmetic_operation(num1, num2, operation):
    """
    Performs arithmetic operation based on user input.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to be performed.

    Returns:
        float: The result of the arithmetic operation.
    """
    if operation == "addition" or operation == "+":
        return num1 + num2
    elif operation == "subtraction" or operation == "-":
        return num1 - num2
    elif operation == "multiplication" or operation == "*":
        return num1 * num2
    elif operation == "division" or operation == "/":
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero.")
    else:
        raise ValueError("Invalid operation. Please choose from addition, subtraction, multiplication, or division.")


def main():
    print("Arithmetic Operation Program")
    print("-------------------------------")
    print("Operations:")
    print("1. Addition (+)")
    print("2. Subtraction (-)")
    print("3. Multiplication (*)")
    print("4. Division (/)")

    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    operation_choice = input("Enter the operation (1/2/3/4 or +/-/*): ")
    if operation_choice == "1" or operation_choice == "+":
        operation = "addition"
    elif operation_choice == "2" or operation_choice == "-":
        operation = "subtraction"
    elif operation_choice == "3" or operation_choice == "*":
        operation = "multiplication"
    elif operation_choice == "4" or operation_choice == "/":
        operation = "division"
    else:
        print("Invalid operation choice. Exiting program.")
        return

    try:
        result = arithmetic_operation(num1, num2, operation)
        print(f"{num1} {get_operation_symbol(operation)} {num2} = {result}")
    except ZeroDivisionError as e:
        print(str(e))
    except ValueError as e:
        print(str(e))


def get_operation_symbol(operation):
    """
    Returns the symbol for the given operation.

    Args:
        operation (str): The arithmetic operation.

    Returns:
        str: The symbol for the operation.
    """
    if operation == "addition":
        return "+"
    elif operation == "subtraction":
        return "-"
    elif operation == "multiplication":
        return "*"
    elif operation == "division":
        return "/"


if __name__ == "__main__":
    main()
```
