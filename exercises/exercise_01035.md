## Exercise
Write a program to generate all permutations of a given list.

## Reference Solution
```python
def generate_permutations(lst):
    """
    Generate all permutations of a given list.

    Args:
        lst (list): The input list.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    # Base case: if the list is empty or contains only one element, return the list itself
    if len(lst) <= 1:
        return [lst]

    # Initialize an empty list to store the permutations
    permutations = []

    # Iterate over each element in the list
    for i, elem in enumerate(lst):
        # Create a new list that contains all elements except the current one
        remaining_lst = lst[:i] + lst[i+1:]

        # Recursively generate permutations for the remaining list
        for perm in generate_permutations(remaining_lst):
            # Add the current element to the beginning of each permutation
            permutations.append([elem] + perm)

    return permutations

# Example usage
def main():
    input_list = [1, 2, 3]
    permutations = generate_permutations(input_list)
    print("Permutations of the input list:")
    for i, perm in enumerate(permutations):
        print(f"Permutation {i+1}: {perm}")

if __name__ == "__main__":
    main()
```
