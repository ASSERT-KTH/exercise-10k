## Exercise
Write a program to add or subtract a given number of hours, minutes, and seconds from a given time

## Reference Solution
```python
from datetime import datetime, timedelta

def modify_time(time_str, hours=0, minutes=0, seconds=0, add=True):
    """
    Modify a given time by adding or subtracting hours, minutes, and seconds.

    Args:
        time_str (str): The initial time in the format "HH:MM:SS".
        hours (int, optional): The number of hours to modify. Defaults to 0.
        minutes (int, optional): The number of minutes to modify. Defaults to 0.
        seconds (int, optional): The number of seconds to modify. Defaults to 0.
        add (bool, optional): Whether to add or subtract the time. Defaults to True.

    Returns:
        str: The modified time in the format "HH:MM:SS".
    """
    time_format = "%H:%M:%S"
    time = datetime.strptime(time_str, time_format)
    modification = timedelta(hours=hours, minutes=minutes, seconds=seconds)

    if add:
        modified_time = time + modification
    else:
        modified_time = time - modification

    return modified_time.strftime(time_format)

def main():
    time_str = input("Enter the initial time (HH:MM:SS): ")
    hours = int(input("Enter the number of hours to modify: "))
    minutes = int(input("Enter the number of minutes to modify: "))
    seconds = int(input("Enter the number of seconds to modify: "))
    add = input("Do you want to add or subtract the time? (add/subtract): ")

    if add.lower() == "add":
        add = True
    elif add.lower() == "subtract":
        add = False
    else:
        print("Invalid input. Defaulting to addition.")
        add = True

    modified_time = modify_time(time_str, hours, minutes, seconds, add)
    print(f"The modified time is: {modified_time}")

if __name__ == "__main__":
    main()
```
