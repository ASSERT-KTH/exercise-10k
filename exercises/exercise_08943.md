## Exercise
Write a program to match a string against a wildcard pattern.

def wildcard_pattern_matching(string, pattern):
    """
    Matches a string against a wildcard pattern.

    Args:
    string (str): The input string to match.
    pattern (str): The wildcard pattern to match against.

    Returns:
    bool: True if the string matches the pattern, False otherwise.
    """

    # Initialize a 2D table to store the dynamic programming results
    dp = [[False] * (len(pattern) + 1) for _ in range(len(string) + 1)]

    # An empty string matches an empty pattern
    dp[0][0] = True

    # A '*' in the pattern can match an empty string
    for j in range(1, len(pattern) + 1):
        if pattern[j - 1] == '*':
            dp[0][j] = dp[0][j - 1]

    # Fill in the rest of the table
    for i in range(1, len(string) + 1):
        for j in range(1, len(pattern) + 1):
            if pattern[j - 1] == '*':
                # '*' can match any sequence of characters
                dp[i][j] = dp[i][j - 1] or dp[i - 1][j]
            elif pattern[j - 1] == '?':
                # '?' can match any single character
                dp[i][j] = dp[i - 1][j - 1]
            else:
                # Exact character match
                dp[i][j] = dp[i - 1][j - 1] and string[i - 1] == pattern[j - 1]

    # The result is stored in the bottom-right corner of the table
    return dp[len(string)][len(pattern)]
