## Exercise
Ask the user for the current price and the previous price of an item, and print the inflation rate.

## Reference Solution
```python
def calculate_inflation_rate(current_price, previous_price):
    """
    Calculate the inflation rate given the current and previous prices.

    Args:
        current_price (float): The current price of the item.
        previous_price (float): The previous price of the item.

    Returns:
        float: The inflation rate as a percentage.
    """
    if previous_price == 0:
        raise ValueError("Previous price cannot be zero.")

    inflation_rate = ((current_price - previous_price) / previous_price) * 100
    return inflation_rate


def main():
    # Get user input
    current_price = float(input("Enter the current price: $"))
    previous_price = float(input("Enter the previous price: $"))

    try:
        # Calculate and print the inflation rate
        inflation_rate = calculate_inflation_rate(current_price, previous_price)
        print(f"The inflation rate is: {inflation_rate:.2f}%")
    except ValueError as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
