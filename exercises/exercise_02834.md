## Exercise
Create a program to determine whether a person is eligible for a job based on their age, experience, and qualifications. If the person's age is above 18, experience is above 2 years, and qualifications are relevant, they are eligible.

## Reference Solution
```python
class JobEligibility:
    def __init__(self, age, experience, qualifications):
        """
        Initialize the job eligibility criteria.

        Args:
            age (int): The age of the person.
            experience (int): The experience of the person in years.
            qualifications (list): A list of relevant qualifications.
        """
        self.age = age
        self.experience = experience
        self.qualifications = qualifications

    def is_eligible(self):
        """
        Determine whether the person is eligible for the job.

        Returns:
            bool: True if the person is eligible, False otherwise.
        """
        # Check if the person's age is above 18
        age_eligible = self.age >= 18

        # Check if the person's experience is above 2 years
        experience_eligible = self.experience > 2

        # Check if the person has relevant qualifications
        qualifications_eligible = len(self.qualifications) > 0

        # The person is eligible if all criteria are met
        return age_eligible and experience_eligible and qualifications_eligible


def main():
    # Example usage
    age = 25
    experience = 5
    qualifications = ["Bachelor's degree", "Master's degree"]

    eligibility = JobEligibility(age, experience, qualifications)

    if eligibility.is_eligible():
        print(f"A person with age {age}, experience {experience} years, and qualifications {qualifications} is eligible for the job.")
    else:
        print(f"A person with age {age}, experience {experience} years, and qualifications {qualifications} is not eligible for the job.")


if __name__ == "__main__":
    main()
```
