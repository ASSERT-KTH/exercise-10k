## Exercise
Write a program that rotates a given string by a specified number of positions.

## Reference Solution
```python
def rotate_string(input_string, positions, direction='clockwise'):
    """
    Rotate a given string by a specified number of positions.

    Args:
        input_string (str): The string to be rotated.
        positions (int): The number of positions to rotate the string.
        direction (str, optional): The direction of rotation. Defaults to 'clockwise'.

    Returns:
        str: The rotated string.
    """
    if direction == 'clockwise':
        return input_string[-positions:] + input_string[:-positions]
    elif direction == 'counter-clockwise':
        return input_string[positions:] + input_string[:positions]
    else:
        raise ValueError("Invalid direction. Please specify 'clockwise' or 'counter-clockwise'.")

def main():
    input_string = "Hello, World!"
    positions = 5
    direction = 'clockwise'

    rotated_string = rotate_string(input_string, positions, direction)
    print(f"Rotated string: {rotated_string}")

    direction = 'counter-clockwise'
    rotated_string = rotate_string(input_string, positions, direction)
    print(f"Rotated string (counter-clockwise): {rotated_string}")

if __name__ == "__main__":
    main()
```
