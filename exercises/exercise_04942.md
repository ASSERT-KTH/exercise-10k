## Exercise
Create a program that checks a student's grade based on their score and prints out the corresponding grade using conditional statements.

## Reference Solution
```python
def check_grade(score):
    """
    This function checks the grade of a student based on their score.
    
    Args:
        score (float): The score of the student.
    
    Returns:
        str: The grade of the student.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the score from the user
    score = float(input("Enter the student's score: "))

    # Check if the score is valid (between 0 and 100)
    if 0 <= score <= 100:
        # Get the grade
        grade = check_grade(score)
        
        # Print the grade
        print(f"The student's grade is: {grade}")
    else:
        print("Invalid score. Please enter a score between 0 and 100.")

if __name__ == "__main__":
    main()
```
