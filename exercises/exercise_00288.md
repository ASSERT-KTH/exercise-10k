## Exercise
Create a program that converts a given amount of money from one currency to another.

## Reference Solution
```python
from forex_python.converter import CurrencyRates, CurrencyCodes

def convert_currency(amount, from_currency, to_currency):
    """
    Converts a given amount of money from one currency to another.

    Args:
        amount (float): The amount of money to convert.
        from_currency (str): The currency to convert from (e.g. 'USD', 'EUR', etc.).
        to_currency (str): The currency to convert to (e.g. 'USD', 'EUR', etc.).

    Returns:
        float: The converted amount.
    """
    c = CurrencyRates()
    result = c.convert(from_currency, to_currency, amount)
    return result

def get_currency_symbol(currency_code):
    """
    Gets the symbol for a given currency code.

    Args:
        currency_code (str): The currency code (e.g. 'USD', 'EUR', etc.).

    Returns:
        str: The currency symbol.
    """
    cc = CurrencyCodes()
    symbol = cc.get_symbol(currency_code)
    return symbol

def main():
    amount = float(input("Enter the amount to convert: "))
    from_currency = input("Enter the currency to convert from (e.g. USD, EUR, etc.): ").upper()
    to_currency = input("Enter the currency to convert to (e.g. USD, EUR, etc.): ").upper()

    converted_amount = convert_currency(amount, from_currency, to_currency)
    from_symbol = get_currency_symbol(from_currency)
    to_symbol = get_currency_symbol(to_currency)

    print(f"{amount} {from_symbol}{from_currency} is equal to {converted_amount} {to_symbol}{to_currency}")

if __name__ == "__main__":
    main()
```
