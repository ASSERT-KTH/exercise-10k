## Exercise
Write a program that uses a while loop to calculate the number of coins of each denomination needed to make a given amount of change.

## Reference Solution
```python
def calculate_change(amount):
    # Define the coin denominations
    coins = [25, 10, 5, 1]
    coin_names = ["quarters", "dimes", "nickels", "pennies"]
    change = {}

    # Iterate over each coin denomination
    for i, coin in enumerate(coins):
        # Calculate the number of coins needed for this denomination
        count = amount // coin
        amount %= coin
        change[coin_names[i]] = count

    return change

def main():
    # Get the amount of change to make from the user
    amount = int(input("Enter the amount of change to make (in cents): "))

    # Calculate the change
    change = calculate_change(amount)

    # Print the change
    print("Change:")
    for coin, count in change.items():
        if count > 0:
            print(f"{coin.capitalize()}: {count}")

if __name__ == "__main__":
    main()
```
