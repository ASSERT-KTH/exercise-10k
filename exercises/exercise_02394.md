## Exercise
Write a program that calculates the least common multiple (LCM) of two given numbers using a while loop.

## Reference Solution
```python
#include <stdio.h>
#include <conio.h>

void main() {
    int num1, num2, temp, rem;
    int big, small, lcm;

    clrscr();

    printf("Enter first number: ");
    scanf("%d", &num1);

    printf("Enter second number: ");
    scanf("%d", &num2);

    if (num1 > num2) {
        big = num1;
        small = num2;
    } else {
        big = num2;
        small = num1;
    }

    temp = big;

    while (temp % small != 0) {
        rem = temp % small;
        temp = small;
        small = rem;
    }

    lcm = big * small;

    printf("The LCM of %d and %d is: %d\n", num1, num2, lcm);

    getch();
}
```
