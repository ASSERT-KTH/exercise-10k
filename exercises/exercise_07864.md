## Exercise
Create a program that checks if a number is an integer using conditional statements.

## Reference Solution
```python
def check_integer(num):
    """
    Checks if a number is an integer.

    Args:
        num (float): The number to check.

    Returns:
        bool: True if the number is an integer, False otherwise.
    """
    if isinstance(num, int):
        return True
    elif isinstance(num, float) and num.is_integer():
        return True
    else:
        return False

def main():
    # Test the function with different numbers
    numbers = [5, 3.14, 10.0, "hello", None]
    
    for num in numbers:
        try:
            result = check_integer(num)
            print(f"{num} is an integer: {result}")
        except TypeError:
            print(f"Cannot check if '{num}' is an integer")

if __name__ == "__main__":
    main()
```
