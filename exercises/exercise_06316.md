def unflatten_dict(flattened_dict):
    """
    Unflatten a single-level dictionary into a nested dictionary.

    Args:
        flattened_dict (dict): The single-level dictionary to unflatten.

    Returns:
        dict: The unflattened nested dictionary.
    """
    unflattened_dict = {}
    for key, value in flattened_dict.items():
        # Split the key by dot notation
        key_parts = key.split(".")
        # Initialize the current dictionary
        current_dict = unflattened_dict
        # Iterate over the key parts
        for i, key_part in enumerate(key_parts):
            # If this is not the last key part, create a new dictionary if it doesn't exist
            if i < len(key_parts) - 1:
                current_dict.setdefault(key_part, {})
                current_dict = current_dict[key_part]
            # If this is the last key part, set the value
            else:
                current_dict[key_part] = value
    return unflattened_dict

# Example usage
flattened_dict = {
    "a.b.c": 1,
    "a.b.d": 2,
    "e.f.g": 3,
    "h": 4
}
unflattened_dict = unflatten_dict(flattened_dict)
print(unflattened_dict)
