# Comments explaining the code
# This is a simple Python program that calculates the volume of a cube.
# The formula for the volume of a cube is V = s^3, where s is the side length.
# This program will take the side length as input from the user and print the volume.

# Importing modules
import math

# Defining functions
def cube_volume(s):
    return s ** 3  # Calculate the volume of a cube

# Main program
s = float(input("Enter the side length of the cube: "))  # Take side length as input from the user
print("The volume of the cube is:", cube_volume(s))  # Print the volume of the cube
