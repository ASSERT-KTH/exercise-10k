## Exercise
Write a program that takes in a list of variables and their corresponding data types, and validates if the assigned value matches the declared data type. The variables and their data types will be given as a dictionary where the keys are the variable names and the values are the data types. The assigned values will be given as a list in the order they appear in the dictionary.

## Reference Solution
```python
def validate_types(types_dict, values):
    """
    Validate if the assigned values match the declared data types in a dictionary.

    Args:
        types_dict (dict): Dictionary with variable names as keys and data types as values.
        values (list): List of assigned values in the order they appear in the dictionary.

    Returns:
        bool: True if all values match their declared data types, False otherwise.
    """
    # Get a list of variable names from the dictionary
    variables = list(types_dict.keys())

    # Iterate over the variables and their corresponding values
    for var, val in zip(variables, values):
        # Get the declared data type for the current variable
        data_type = types_dict[var]

        # Validate the value based on its declared data type
        if data_type == "str":
            if not isinstance(val, str):
                return False
        elif data_type == "int":
            if not isinstance(val, int):
                return False
        elif data_type == "float":
            if not isinstance(val, float):
                return False
        elif data_type == "bool":
            if not isinstance(val, bool):
                return False
        elif data_type == "list":
            if not isinstance(val, list):
                return False
        elif data_type == "dict":
            if not isinstance(val, dict):
                return False
        elif data_type == "tuple":
            if not isinstance(val, tuple):
                return False
        elif data_type == "set":
            if not isinstance(val, set):
                return False

    # If all values have been validated successfully, return True
    return True

# Example usage:
types_dict = {"name": "str", "age": "int", "height": "float", "is_admin": "bool"}
values = ["John", 25, 1.75, True]

if validate_types(types_dict, values):
    print("All values match their declared data types.")
else:
    print("At least one value does not match its declared data type.")
```
