## Exercise
Write a program that classifies a given list of variables into different categories (e.g., integers, floats, strings) based on their data types.

## Reference Solution
```python
def classify_variables(variables):
    """
    Classify a list of variables into different categories based on their data types.

    Args:
        variables (list): A list of variables.

    Returns:
        dict: A dictionary where the keys are the data types and the values are lists of variables.
    """
    classified_variables = {
        "integers": [],
        "floats": [],
        "strings": [],
        "lists": [],
        "tuples": [],
        "dictionaries": [],
        "sets": [],
        "booleans": [],
        "other": []
    }

    for variable in variables:
        if isinstance(variable, int):
            classified_variables["integers"].append(variable)
        elif isinstance(variable, float):
            classified_variables["floats"].append(variable)
        elif isinstance(variable, str):
            classified_variables["strings"].append(variable)
        elif isinstance(variable, list):
            classified_variables["lists"].append(variable)
        elif isinstance(variable, tuple):
            classified_variables["tuples"].append(variable)
        elif isinstance(variable, dict):
            classified_variables["dictionaries"].append(variable)
        elif isinstance(variable, set):
            classified_variables["sets"].append(variable)
        elif isinstance(variable, bool):
            classified_variables["booleans"].append(variable)
        else:
            classified_variables["other"].append(variable)

    return classified_variables


def main():
    variables = [
        1, 2, 3, 4, 5,
        1.5, 2.5, 3.5,
        "hello", "world",
        [1, 2, 3], [4, 5, 6],
        (1, 2, 3), (4, 5, 6),
        {"a": 1, "b": 2}, {"c": 3, "d": 4},
        {1, 2, 3}, {4, 5, 6},
        True, False,
        None
    ]

    classified_variables = classify_variables(variables)

    for data_type, variables in classified_variables.items():
        print(f"{data_type.capitalize()}: {variables}")


if __name__ == "__main__":
    main()
```
