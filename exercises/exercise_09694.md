## Exercise
Write a program that takes a user's height in feet and inches as input, and then converts it to meters and centimeters. The program should prompt the user for their height, parse the input to extract the feet and inches, perform the conversion, and then print the result to the console.

## Reference Solution
```python
#include <stdio.h>

// Structure to represent a height in feet and inches
typedef struct {
    int feet;
    int inches;
} Height;

// Function to convert a height in feet and inches to meters and centimeters
Height convertToMetric(int feet, int inches) {
    int totalInches = feet * 12 + inches;
    double totalCentimeters = totalInches * 2.54;
    int meters = (int) (totalCentimeters / 100);
    int centimeters = (int) (totalCentimeters - (meters * 100));
    Height metricHeight;
    metricHeight.feet = meters;
    metricHeight.inches = centimeters;
    return metricHeight;
}

// Function to parse a string input into a height in feet and inches
Height parseInput(const char* input) {
    Height height;
    sscanf(input, "%d %d", &height.feet, &height.inches);
    return height;
}

// Function to get the user's height from the user
Height getUserHeight() {
    printf("Enter your height in feet and inches (e.g. 5 10): ");
    char input[20];
    fgets(input, sizeof(input), stdin);
    return parseInput(input);
}

// Function to print a height in meters and centimeters
void printMetricHeight(Height height) {
    printf("Your height is %d meters and %d centimeters.\n", height.feet, height.inches);
}

int main() {
    Height height = getUserHeight();
    Height metricHeight = convertToMetric(height.feet, height.inches);
    printMetricHeight(metricHeight);
    return 0;
}
```
