## Exercise
Create a function that finds the maximum element in a list of lists.

## Reference Solution
```python
def find_max_element(list_of_lists):
    """
    Finds the maximum element in a list of lists.

    Args:
    list_of_lists (list): A list of lists containing elements.

    Returns:
    The maximum element in the list of lists.
    """
    # Check if the input is a list
    if not isinstance(list_of_lists, list):
        raise ValueError("Input must be a list")

    # Check if the list is empty
    if not list_of_lists:
        raise ValueError("Input list is empty")

    # Check if all elements in the list are lists
    if not all(isinstance(sublist, list) for sublist in list_of_lists):
        raise ValueError("All elements in the list must be lists")

    # Flatten the list of lists into a single list
    flat_list = [element for sublist in list_of_lists for element in sublist]

    # Check if the flattened list is empty
    if not flat_list:
        raise ValueError("No elements found in the list of lists")

    # Return the maximum element in the flattened list
    return max(flat_list)


# Example usage:
if __name__ == "__main__":
    list_of_lists = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    max_element = find_max_element(list_of_lists)
    print(f"Maximum element: {max_element}")
```
