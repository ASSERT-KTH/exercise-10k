    """
    Returns the input parameter value.

    Args:
        param (str, optional): The input parameter. Defaults to "Default Value".

    Returns:
        str: The input parameter value.
    """
    return param

# Example usage
print(return_parameter())  # Outputs: Default Value
print(return_parameter("Custom Value"))  # Outputs: Custom Value
