## Exercise
Create a program that simulates the Game of Life using a for loop to iterate over the grid.

## Reference Solution
```python
import random
import time
import os

def create_grid(rows, cols):
    """Create a grid with random values."""
    grid = [[random.choice([0, 1]) for _ in range(cols)] for _ in range(rows)]
    return grid

def print_grid(grid):
    """Print the grid."""
    for row in grid:
        print(' '.join(['*' if cell else ' ' for cell in row]))

def count_neighbours(grid, row, col):
    """Count the number of live neighbours for a cell."""
    rows, cols = len(grid), len(grid[0])
    count = 0
    for i in range(max(0, row-1), min(rows, row+2)):
        for j in range(max(0, col-1), min(cols, col+2)):
            if (i, j) != (row, col) and grid[i][j]:
                count += 1
    return count

def next_generation(grid):
    """Generate the next generation of the grid."""
    rows, cols = len(grid), len(grid[0])
    next_grid = [[0 for _ in range(cols)] for _ in range(rows)]
    for row in range(rows):
        for col in range(cols):
            live_neighbours = count_neighbours(grid, row, col)
            if grid[row][col]:
                if live_neighbours < 2 or live_neighbours > 3:
                    next_grid[row][col] = 0
                else:
                    next_grid[row][col] = 1
            else:
                if live_neighbours == 3:
                    next_grid[row][col] = 1
    return next_grid

def main():
    rows, cols = 20, 50
    grid = create_grid(rows, cols)
    generations = 100
    for _ in range(generations):
        os.system('cls' if os.name == 'nt' else 'clear')
        print_grid(grid)
        grid = next_generation(grid)
        time.sleep(0.5)

if __name__ == "__main__":
    main()
```
