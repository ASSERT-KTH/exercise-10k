## Exercise
Write a program that takes in basic math expressions and evaluates them

## Reference Solution
```python
# Import the operator module for operator functions
import operator

# Define a dictionary to map operator symbols to their corresponding functions
operators = {
    '+': operator.add,
    '-': operator.sub,
    '*': operator.mul,
    '/': operator.truediv
}

def evaluate_expression(expression):
    """
    Evaluate a basic math expression.

    Args:
    expression (str): The math expression to evaluate.

    Returns:
    float: The result of the evaluation.
    """
    # Split the expression into operands and operators
    elements = expression.split()
    operand1 = float(elements[0])
    operator_symbol = elements[1]
    operand2 = float(elements[2])

    # Get the operator function from the dictionary
    operator_function = operators.get(operator_symbol)

    # Check if the operator is valid
    if operator_function is None:
        raise ValueError("Invalid operator")

    # Evaluate the expression
    result = operator_function(operand1, operand2)

    return result

def main():
    # Example usage
    expression = input("Enter a math expression (e.g., 10 + 5): ")
    try:
        result = evaluate_expression(expression)
        print(f"The result is: {result}")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
