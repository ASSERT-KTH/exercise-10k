## Exercise
Write a program to use looping statements to iterate over a sequence of values.

## Reference Solution
```python
# Define a list of values
values = [1, 2, 3, 4, 5]

# Using a for loop to iterate over the list
def iterate_using_for(values):
    print("Using a for loop:")
    for value in values:
        print(value)

# Using a while loop to iterate over the list
def iterate_using_while(values):
    print("\nUsing a while loop:")
    index = 0
    while index < len(values):
        print(values[index])
        index += 1

# Using a for loop with enumerate to iterate over the list
def iterate_using_enumerate(values):
    print("\nUsing a for loop with enumerate:")
    for index, value in enumerate(values):
        print(f"Index: {index}, Value: {value}")

# Main function
def main():
    iterate_using_for(values)
    iterate_using_while(values)
    iterate_using_enumerate(values)

if __name__ == "__main__":
    main()
```
