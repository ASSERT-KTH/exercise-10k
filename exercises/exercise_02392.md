## Exercise
Write a program that checks if a given number is an Armstrong number using a while loop.

## Reference Solution

def is_armstrong(num):
    """
    Checks if a number is an Armstrong number using a while loop.

    Args:
        num (int): The number to check.

    Returns:
        bool: True if the number is an Armstrong number, False otherwise.
    """
    # Calculate the number of digits
    original_num = num
    num_digits = 0
    while num > 0:
        num_digits += 1
        num //= 10

    # Calculate the sum of each digit raised to the power of the number of digits
    num = original_num
    sum_of_digits = 0
    while num > 0:
        digit = num % 10
        sum_of_digits += digit ** num_digits
        num //= 10

    # Check if the sum is equal to the original number
    return sum_of_digits == original_num

# Example usage:
num = 371
if is_armstrong(num):
    print(f"{num} is an Armstrong number.")
else:
    print(f"{num} is not an Armstrong number.")
