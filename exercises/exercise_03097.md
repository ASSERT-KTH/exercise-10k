## Exercise
Create a function to rotate a given list by a specified number of positions.

## Reference Solution
```python
def rotate_list(input_list, num_positions, direction='clockwise'):
    """
    Rotate a given list by a specified number of positions.

    Args:
        input_list (list): The list to be rotated.
        num_positions (int): The number of positions to rotate the list.
        direction (str, optional): The direction of rotation. Defaults to 'clockwise'.

    Returns:
        list: The rotated list.
    """
    if direction == 'clockwise':
        num_positions = -num_positions
    return input_list[num_positions:] + input_list[:num_positions]

# Example usage:
if __name__ == "__main__":
    my_list = [1, 2, 3, 4, 5]
    print("Original list:", my_list)
    
    # Rotate the list 2 positions clockwise
    rotated_list_clockwise = rotate_list(my_list, 2)
    print("Rotated list (clockwise):", rotated_list_clockwise)
    
    # Rotate the list 2 positions counter-clockwise
    rotated_list_counter_clockwise = rotate_list(my_list, 2, direction='counter-clockwise')
    print("Rotated list (counter-clockwise):", rotated_list_counter_clockwise)
```
