## Exercise
Write a program that takes an array of integers and a shift value as input, and returns a new array containing the elements shifted by the specified amount.

## Reference Solution
```python
#include <bits/stdc++.h>
using namespace std;

vector<int> shiftElements(vector<int> &a, int shift) {
    shift = shift % a.size();

    vector<int> temp(shift);
    for (int i = 0; i < shift; i++) {
        temp[i] = a[a.size() - shift + i];
    }

    for (int i = a.size() - 1; i >= shift; i--) {
        a[i] = a[i - shift];
    }

    for (int i = 0; i < shift; i++) {
        a[i] = temp[i];
    }

    return a;
}

int main() {
    vector<int> a = {1, 2, 3, 4, 5};
    int shift = 2;

    vector<int> result = shiftElements(a, shift);

    for (int num : result) {
        cout << num << " ";
    }

    return 0;
}
```
