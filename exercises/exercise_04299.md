## Exercise
Write a function to group a dictionary's values by a given key function.

## Reference Solution
```python
def group_dict_values(dictionary, key_func):
    """
    This function groups a dictionary's values by a given key function.
    
    Args:
        dictionary (dict): The input dictionary.
        key_func (function): A function that takes a value and returns a key.
        
    Returns:
        dict: A dictionary where the keys are the results of the key function and the values are lists of the corresponding values from the input dictionary.
    """

    # Initialize an empty dictionary to store the grouped values
    grouped_dict = {}

    # Iterate over each key-value pair in the input dictionary
    for key, value in dictionary.items():
        # Apply the key function to the current value to get the group key
        group_key = key_func(value)
        
        # If the group key is not already in the grouped dictionary, add it with an empty list as its value
        if group_key not in grouped_dict:
            # This is a new group, so create a new list for it
            grouped_dict[group_key] = []
        
        # Append the current value to the list of its group
        grouped_dict[group_key].append(value)

    # Return the dictionary with the grouped values
    return grouped_dict

# Example usage:
# Define a dictionary
my_dict = {'a': 1, 'b': 2, 'c': 3, 'd': 4}

# Define a key function that groups even and odd numbers
def is_even(n):
    return 'even' if n % 2 == 0 else 'odd'

# Group the dictionary's values by the key function
grouped = group_dict_values(my_dict, is_even)

# Print the result
print(grouped)
```
