## Exercise
Write a program that implements a chatbot using a while loop.

## Reference Solution
```python
class Chatbot:
    def __init__(self):
        self.greetings = ["hello", "hi", "hey"]
        self.farewells = ["bye", "quit", "exit"]

    def respond(self, message):
        """Return a response to the user's message."""
        message = message.lower()
        if any(greeting in message for greeting in self.greetings):
            return "Hello! How can I assist you today?"
        elif any(farewell in message for farewell in self.farewells):
            return "It was nice chatting with you. Goodbye!"
        elif "how are you" in message:
            return "I'm doing well, thanks for asking!"
        else:
            return "I'm not sure I understand. Could you please rephrase?"

def main():
    chatbot = Chatbot()
    print("Welcome to the chatbot! Type 'quit' to exit.")
    while True:
        user_input = input("You: ")
        response = chatbot.respond(user_input)
        print("Chatbot: ", response)
        if any(farewell in user_input.lower() for farewell in chatbot.farewells):
            break

if __name__ == "__main__":
    main()
```
