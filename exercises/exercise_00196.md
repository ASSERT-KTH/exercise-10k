## Exercise
Create a program that uses a for loop to encrypt and decrypt a message using the Caesar Cipher algorithm.

## Reference Solution
```python
def caesar_cipher(message, shift, mode):
    """
    Encrypts or decrypts a message using the Caesar Cipher algorithm.

    Parameters:
    message (str): The message to encrypt or decrypt.
    shift (int): The number of positions to shift the letters.
    mode (str): 'encrypt' or 'decrypt' mode.

    Returns:
    str: The encrypted or decrypted message.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in message:
        if char.isalpha():
            # Find the position in the alphabet for the character
            pos = alphabet.index(char.lower())

            if mode == 'encrypt':
                # Shift the character to the right
                new_pos = (pos + shift) % 26
            elif mode == 'decrypt':
                # Shift the character to the left
                new_pos = (pos - shift) % 26

            # Add the encrypted or decrypted character to the result
            if char.isupper():
                result += alphabet[new_pos].upper()
            else:
                result += alphabet[new_pos]
        else:
            # If the character is not a letter, add it to the result as is
            result += char

    return result


def main():
    message = input("Enter a message: ")
    shift = int(input("Enter the shift value: "))

    encrypted_message = caesar_cipher(message, shift, 'encrypt')
    print("Encrypted message:", encrypted_message)

    decrypted_message = caesar_cipher(encrypted_message, shift, 'decrypt')
    print("Decrypted message:", decrypted_message)


if __name__ == "__main__":
    main()
```
