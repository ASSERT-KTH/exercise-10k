## Exercise
Convert a string input to an integer, then perform arithmetic operations on it

## Reference Solution
```python
def convert_and_operate(input_str):
    """
    Attempts to convert input string to integer and perform arithmetic operations.
    
    Args:
        input_str (str): The input string to be converted.
    
    Returns:
        dict: A dictionary containing the results of arithmetic operations.
    """
    try:
        # Attempt to convert string to integer
        num = int(input_str)
        
        # Perform arithmetic operations
        result = {
            "addition": num + 5,
            "subtraction": num - 3,
            "multiplication": num * 2,
            "division": num / 2
        }
        
        return result
    
    except ValueError:
        # Handle the case when the input string cannot be converted to an integer
        return "Error: Input string cannot be converted to an integer."

# Example usage
def main():
    input_str = input("Enter a number as a string: ")
    result = convert_and_operate(input_str)
    
    if isinstance(result, dict):
        print("Results of arithmetic operations:")
        for operation, value in result.items():
            print(f"{operation.capitalize()}: {value}")
    else:
        print(result)

if __name__ == "__main__":
    main()
```
