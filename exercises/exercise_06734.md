## Exercise
Write a program that parses input data in JSON format

## Reference Solution
```python
import json

def parse_json(data):
    """
    Parse JSON data and return a Python dictionary.

    Args:
        data (str): JSON data as a string

    Returns:
        dict: Parsed JSON data as a Python dictionary
    """
    try:
        # Attempt to parse the JSON data
        parsed_data = json.loads(data)
        return parsed_data
    except json.JSONDecodeError as e:
        # Handle JSON decoding errors
        print(f"Error parsing JSON: {e}")
        return None

def main():
    # Example usage
    json_data = '''
    {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }
    '''
    parsed_data = parse_json(json_data)

    if parsed_data:
        print("Parsed JSON Data:")
        print(json.dumps(parsed_data, indent=4))

if __name__ == "__main__":
    main()
```
