## Exercise
Write a program that takes an amount as input and converts it to the fewest number of coins possible based on the following denominations: 1 cent, 5 cents, 10 cents, 25 cents.

## Reference Solution
```python
def coin_converter(amount):
    """
    Converts an amount to the fewest number of coins possible.

    Args:
    amount (int): The amount in cents to convert.

    Returns:
    dict: A dictionary with the coin denominations as keys and the number of coins as values.
    """
    # Define the available denominations
    denominations = [25, 10, 5, 1]

    # Initialize a dictionary to store the number of coins for each denomination
    coins = {denomination: 0 for denomination in denominations}

    # Iterate over the denominations in descending order
    for denomination in denominations:
        # Calculate the number of coins for the current denomination
        coins[denomination] = amount // denomination
        # Update the remaining amount
        amount %= denomination

    return coins

def main():
    # Get the amount from the user
    amount = int(input("Enter an amount in cents: "))

    # Convert the amount to coins
    coins = coin_converter(amount)

    # Print the result
    print("Coin Conversion:")
    for denomination, count in coins.items():
        if count > 0:
            print(f"{denomination} cents: {count}")

if __name__ == "__main__":
    main()
```
