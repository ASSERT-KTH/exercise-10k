## Exercise
Create a function that returns another function that has access to its own scope.

# We start by defining a function that will return another function.
def outer_function():
    """
    This function returns another function with access to its own scope.

    Returns:
        function: A function that has access to its own scope.
    """

    # We define a variable in the outer function's scope.
    # This variable will be accessible to the inner function.
    outer_variable = "I'm in the outer scope"

    # We define an inner function that has access to the outer function's scope.
    def inner_function():
        # The inner function can print the outer variable.
        print(outer_variable)

    # The outer function returns the inner function.
    return inner_function

# We call the outer function to get the inner function.
inner_func = outer_function()
