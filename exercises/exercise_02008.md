#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int m;
        cin >> m;

        if (m >= 100) {
            int h = m / 100;
            m %= 100;
            cout << h << " ";
        } else {
            cout << "0 ";
        }

        if (m >= 50) {
            int f = m / 50;
            m %= 50;
            cout << f << " ";
        } else {
            cout << "0 ";
        }

        if (m >= 10) {
            int te = m / 10;
            m %= 10;
            cout << te << " ";
        } else {
            cout << "0 ";
        }

        if (m >= 5) {
            int fi = m / 5;
            m %= 5;
            cout << fi << " ";
        } else {
            cout << "0 ";
        }

        if (m >= 2) {
            int tw = m / 2;
            m %= 2;
            cout << tw << " ";
        } else {
            cout << "0 ";
        }

        if (m >= 1) {
            int o = m / 1;
            cout << o << " ";
        } else {
            cout << "0 ";
        }

        cout << endl;
    }

    return 0;
}
