## Exercise
Write a function to remove and return the value associated with a given key from a dictionary

## Reference Solution
```python
def remove_and_return(dictionary, key):
    """
    Removes and returns the value associated with a given key from a dictionary.

    Args:
        dictionary (dict): The dictionary to remove the key-value pair from.
        key: The key to remove and return the value for.

    Returns:
        The value associated with the key if it exists, otherwise None.
    """
    if key in dictionary:
        value = dictionary[key]
        del dictionary[key]
        return value
    else:
        return None

# Example usage
if __name__ == "__main__":
    example_dict = {"a": 1, "b": 2, "c": 3}
    print("Original dictionary:", example_dict)
    
    key_to_remove = "b"
    removed_value = remove_and_return(example_dict, key_to_remove)
    print(f"Removed value for key '{key_to_remove}': {removed_value}")
    print("Dictionary after removal:", example_dict)

    non_existent_key = "d"
    removed_value = remove_and_return(example_dict, non_existent_key)
    print(f"Removed value for key '{non_existent_key}': {removed_value}")
    print("Dictionary after attempting to remove non-existent key:", example_dict)
```
