## Exercise
Create a program to generate a sequence of numbers (e.g. Fibonacci, prime numbers).

## Reference Solution
```python
def fibonacci(n):
    """
    Generate a Fibonacci sequence up to the nth term.

    Args:
        n (int): The number of terms in the sequence.

    Returns:
        list: A list of Fibonacci numbers.
    """
    fib_sequence = [0, 1]
    while len(fib_sequence) < n:
        fib_sequence.append(fib_sequence[-1] + fib_sequence[-2])
    return fib_sequence[:n]

def prime_numbers(n):
    """
    Generate a sequence of prime numbers up to the nth term.

    Args:
        n (int): The number of terms in the sequence.

    Returns:
        list: A list of prime numbers.
    """
    primes = []
    num = 2
    while len(primes) < n:
        if all(num % p > 0 for p in primes):
            primes.append(num)
        num += 1
    return primes

def main():
    print("Sequence Generator Program")
    print("1. Fibonacci Sequence")
    print("2. Prime Number Sequence")
    choice = input("Enter your choice (1/2): ")
    n = int(input("Enter the number of terms: "))

    if choice == "1":
        sequence = fibonacci(n)
        print("Fibonacci Sequence:", sequence)
    elif choice == "2":
        sequence = prime_numbers(n)
        print("Prime Number Sequence:", sequence)
    else:
        print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
