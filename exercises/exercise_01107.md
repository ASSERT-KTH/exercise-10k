## Exercise
Write a program to calculate the percent increase between two numbers

## Reference Solution
```python
def calculate_percent_increase(original_value, new_value):
    """
    Calculate the percent increase between two numbers.

    Args:
        original_value (float): The original value.
        new_value (float): The new value.

    Returns:
        float: The percent increase.
    """
    if original_value == 0:
        raise ValueError("Original value cannot be zero")

    percent_increase = ((new_value - original_value) / original_value) * 100
    return percent_increase

def main():
    original_value = float(input("Enter the original value: "))
    new_value = float(input("Enter the new value: "))

    try:
        percent_increase = calculate_percent_increase(original_value, new_value)
        print(f"The percent increase is: {percent_increase:.2f}%")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
