## Exercise
Write a program that takes an array as input, and returns the mean, median, and standard deviation of the array.

## Reference Solution
```python
import numpy as np

def calculate_statistics(array):
    """
    Calculate the mean, median, and standard deviation of an array.

    Parameters:
    array (list): A list of numbers.

    Returns:
    tuple: A tuple containing the mean, median, and standard deviation of the array.
    """
    # Convert the input to a NumPy array
    array = np.array(array)

    # Calculate the mean of the array
    mean = np.mean(array)

    # Calculate the median of the array
    median = np.median(array)

    # Calculate the standard deviation of the array
    std_dev = np.std(array)

    return mean, median, std_dev

def main():
    # Example usage
    array = [1, 2, 3, 4, 5]
    mean, median, std_dev = calculate_statistics(array)

    print("Mean:", mean)
    print("Median:", median)
    print("Standard Deviation:", std_dev)

if __name__ == "__main__":
    main()
```
