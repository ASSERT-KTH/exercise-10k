## Exercise
Create a function that takes a string as an argument and returns True if it is a valid password (at least 8 characters, contains at least one uppercase and one lowercase letter, and at least one number) and False otherwise

## Reference Solution
```python
import re

def is_valid_password(password: str) -> bool:
    """
    Checks if a given string is a valid password.

    A valid password must be at least 8 characters long, contain at least one uppercase and one lowercase letter, and at least one number.

    Args:
        password (str): The password to check.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Check if password is at least 8 characters long
    if len(password) < 8:
        return False

    # Check if password contains at least one uppercase letter, one lowercase letter, and one number
    if not re.search(r"[A-Z]", password) or not re.search(r"[a-z]", password) or not re.search(r"\d", password):
        return False

    return True

# Example usage:
if __name__ == "__main__":
    passwords = ["Invalid", "ValidPassword123", "invalidpassword123", "VALIDPASSWORD123"]
    for password in passwords:
        print(f"Password: {password}, Valid: {is_valid_password(password)}")
```
