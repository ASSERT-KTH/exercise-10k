## Exercise
Create a program that takes user input for two numbers and an operation (+, -, *, /). Validate the input to ensure it meets the correct format, then perform the specified operation and print the result. Include error handling for division by zero and invalid operations.

## Reference Solution
```python
def get_user_input():
    """Get user input for two numbers and an operation."""
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            num2 = float(input("Enter the second number: "))
            operation = input("Enter the operation (+, -, *, /): ")
            return num1, num2, operation
        except ValueError:
            print("Invalid input. Please enter a number.")

def validate_operation(operation):
    """Validate the operation to ensure it's one of the four basic arithmetic operations."""
    valid_operations = ['+', '-', '*', '/']
    while operation not in valid_operations:
        print("Invalid operation. Please enter one of the following: +, -, *, /")
        operation = input("Enter the operation: ")
    return operation

def perform_operation(num1, num2, operation):
    """Perform the specified operation on the two numbers."""
    if operation == '+':
        result = num1 + num2
    elif operation == '-':
        result = num1 - num2
    elif operation == '*':
        result = num1 * num2
    elif operation == '/':
        try:
            result = num1 / num2
        except ZeroDivisionError:
            print("Error: Division by zero is not allowed.")
            return None
    return result

def main():
    num1, num2, operation = get_user_input()
    operation = validate_operation(operation)
    result = perform_operation(num1, num2, operation)
    if result is not None:
        print(f"{num1} {operation} {num2} = {result}")

if __name__ == "__main__":
    main()
```
