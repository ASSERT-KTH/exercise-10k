## Exercise
Write a program that simulates a To-Do List app, where the user can add, remove, and mark tasks as completed

## Reference Solution
```python
class Task:
    """Represents a task in the To-Do List"""
    def __init__(self, name):
        self.name = name
        self.completed = False

    def mark_completed(self):
        """Marks the task as completed"""
        self.completed = True

    def mark_incomplete(self):
        """Marks the task as incomplete"""
        self.completed = False


class ToDoList:
    """Simulates a To-Do List app"""
    def __init__(self):
        self.tasks = {}

    def add_task(self, name):
        """Adds a new task to the list"""
        if name not in self.tasks:
            self.tasks[name] = Task(name)
            print(f"Task '{name}' added successfully.")
        else:
            print(f"Task '{name}' already exists.")

    def remove_task(self, name):
        """Removes a task from the list"""
        if name in self.tasks:
            del self.tasks[name]
            print(f"Task '{name}' removed successfully.")
        else:
            print(f"Task '{name}' does not exist.")

    def mark_task_completed(self, name):
        """Marks a task as completed"""
        if name in self.tasks:
            self.tasks[name].mark_completed()
            print(f"Task '{name}' marked as completed.")
        else:
            print(f"Task '{name}' does not exist.")

    def mark_task_incomplete(self, name):
        """Marks a task as incomplete"""
        if name in self.tasks:
            self.tasks[name].mark_incomplete()
            print(f"Task '{name}' marked as incomplete.")
        else:
            print(f"Task '{name}' does not exist.")

    def display_tasks(self):
        """Displays all tasks in the list"""
        print("\nTo-Do List:")
        for task in self.tasks.values():
            status = "Completed" if task.completed else "Incomplete"
            print(f"- {task.name}: {status}")


def main():
    todo_list = ToDoList()

    while True:
        print("\nOptions:")
        print("1. Add task")
        print("2. Remove task")
        print("3. Mark task as completed")
        print("4. Mark task as incomplete")
        print("5. Display tasks")
        print("6. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter task name: ")
            todo_list.add_task(name)
        elif choice == "2":
            name = input("Enter task name: ")
            todo_list.remove_task(name)
        elif choice == "3":
            name = input("Enter task name: ")
            todo_list.mark_task_completed(name)
        elif choice == "4":
            name = input("Enter task name: ")
            todo_list.mark_task_incomplete(name)
        elif choice == "5":
            todo_list.display_tasks()
        elif choice == "6":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please try again.")


if __name__ == "__main__":
    main()
```
