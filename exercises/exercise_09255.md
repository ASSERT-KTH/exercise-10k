## Exercise
Write a program that demonstrates data type conversion errors and handles them accordingly.

## Reference Solution
```python
def convert_to_int(input_str):
    """
    Attempts to convert a string to an integer.

    Args:
        input_str (str): The input string to convert.

    Returns:
        int: The converted integer if successful.
        None: If the conversion fails.
    """
    try:
        return int(input_str)
    except ValueError:
        print(f"Error: Cannot convert '{input_str}' to integer.")
        return None


def convert_to_float(input_str):
    """
    Attempts to convert a string to a floating-point number.

    Args:
        input_str (str): The input string to convert.

    Returns:
        float: The converted float if successful.
        None: If the conversion fails.
    """
    try:
        return float(input_str)
    except ValueError:
        print(f"Error: Cannot convert '{input_str}' to float.")
        return None


def convert_to_bool(input_str):
    """
    Attempts to convert a string to a boolean value.

    Args:
        input_str (str): The input string to convert.

    Returns:
        bool: The converted boolean if successful.
        None: If the conversion fails.
    """
    if input_str.lower() == "true":
        return True
    elif input_str.lower() == "false":
        return False
    else:
        print(f"Error: Cannot convert '{input_str}' to boolean.")
        return None


def main():
    # Test cases for integer conversion
    print("Integer Conversion:")
    print(convert_to_int("123"))  # Expected output: 123
    print(convert_to_int("abc"))  # Expected output: Error message and None

    # Test cases for float conversion
    print("\nFloat Conversion:")
    print(convert_to_float("123.45"))  # Expected output: 123.45
    print(convert_to_float("abc"))  # Expected output: Error message and None

    # Test cases for boolean conversion
    print("\nBoolean Conversion:")
    print(convert_to_bool("true"))  # Expected output: True
    print(convert_to_bool("false"))  # Expected output: False
    print(convert_to_bool("abc"))  # Expected output: Error message and None


if __name__ == "__main__":
    main()
```
