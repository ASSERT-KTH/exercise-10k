## Exercise
Write a program to find the maximum flow of a flow network using the Edmonds-Karp algorithm.

## Reference Solution

from collections import deque

def bfs(rGraph, s, t, parent):
    """
    Performs a BFS traversal of the residual graph to find an augmenting path.

    Args:
    rGraph (list): The residual graph.
    s (int): The source node.
    t (int): The sink node.
    parent (list): A list to store the parent of each node in the augmenting path.

    Returns:
    bool: True if an augmenting path is found, False otherwise.
    """
    visited = [False] * len(rGraph)
    queue = deque()

    queue.append(s)
    visited[s] = True

    while queue:
        u = queue.popleft()

        for ind, val in enumerate(rGraph[u]):
            if not visited[ind] and val > 0:
                queue.append(ind)
                visited[ind] = True
                parent[ind] = u

                if ind == t:
                    return True

    return False


def edmonds_karp(graph, source, sink):
    """
    Finds the maximum flow in a flow network using the Edmonds-Karp algorithm.

    Args:
    graph (list): The flow network.
    source (int): The source node.
    sink (int): The sink node.

    Returns:
    int: The maximum flow.
    """
    parent = [-1] * len(graph)
    max_flow = 0
    rGraph = [i[:] for i in graph]

    while bfs(rGraph, source, sink, parent):
        path_flow = float("Inf")
        s = sink

        while s != source:
            path_flow = min(path_flow, rGraph[parent[s]][s])
            s = parent[s]

        max_flow += path_flow

        v = sink
        while v != source:
            u = parent[v]
            rGraph[u][v] -= path_flow
            rGraph[v][u] += path_flow
            v = parent[v]

    return max_flow


# Example usage:
graph = [[0, 16, 13, 0, 0, 0],
         [0, 0, 10, 12, 0, 0],
         [0, 4, 0, 0, 14, 0],
         [0, 0, 9, 0, 0, 20],
         [0, 0, 0, 7, 0, 4],
         [0, 0, 0, 0, 0, 0]]

source = 0
sink = 5

print("The maximum possible flow is %d " % edmonds_karp(graph, source, sink))
