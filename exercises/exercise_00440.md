## Exercise
Ask the user for a text and print out various statistics about it, such as the number of words, characters, and sentences.

## Reference Solution
```python
import re

def calculate_text_statistics(text):
    """
    Calculate various statistics about the given text.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary containing the text statistics.
    """
    # Calculate the number of characters
    num_chars = len(text)

    # Calculate the number of words
    words = re.findall(r'\b\w+\b', text)
    num_words = len(words)

    # Calculate the number of sentences
    sentences = re.split(r'[.!?]\s*', text)
    num_sentences = len([s for s in sentences if s])

    # Calculate the average word length
    total_word_length = sum(len(word) for word in words)
    avg_word_length = total_word_length / num_words if num_words > 0 else 0

    # Calculate the average sentence length
    total_sentence_length = sum(len(sentence.split()) for sentence in sentences if sentence)
    avg_sentence_length = total_sentence_length / num_sentences if num_sentences > 0 else 0

    return {
        'num_chars': num_chars,
        'num_words': num_words,
        'num_sentences': num_sentences,
        'avg_word_length': avg_word_length,
        'avg_sentence_length': avg_sentence_length
    }

def main():
    # Ask the user for a text
    text = input("Please enter a text: ")

    # Calculate the text statistics
    stats = calculate_text_statistics(text)

    # Print the text statistics
    print("\nText Statistics:")
    print("-----------------")
    print(f"Number of characters: {stats['num_chars']}")
    print(f"Number of words: {stats['num_words']}")
    print(f"Number of sentences: {stats['num_sentences']}")
    print(f"Average word length: {stats['avg_word_length']:.2f}")
    print(f"Average sentence length: {stats['avg_sentence_length']:.2f}")

if __name__ == "__main__":
    main()
```
