## Exercise
Write a program to calculate the midpoint of a line

## Reference Solution
```python
def calculate_midpoint(point1, point2):
    """
    Calculate the midpoint of a line given two points.

    Args:
        point1 (tuple): Coordinates of the first point (x1, y1)
        point2 (tuple): Coordinates of the second point (x2, y2)

    Returns:
        tuple: Coordinates of the midpoint (x, y)
    """
    x1, y1 = point1
    x2, y2 = point2
    x_midpoint = (x1 + x2) / 2
    y_midpoint = (y1 + y2) / 2
    return (x_midpoint, y_midpoint)


def main():
    # Example usage
    point1 = (1, 2)
    point2 = (4, 6)
    midpoint = calculate_midpoint(point1, point2)
    print(f"The midpoint of the line between {point1} and {point2} is {midpoint}")


if __name__ == "__main__":
    main()
```
