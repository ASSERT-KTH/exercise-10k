## Exercise
Create a program that simulates a bank account. The program should take in the account balance and output the updated balance based on the following conditions: if the user deposits money, add the deposit amount to the balance, if the user withdraws money, subtract the withdrawal amount from the balance, if the user checks the balance, output the current balance.

## Reference Solution
```python
class BankAccount:
    def __init__(self, balance=0):
        """
        Initialize the BankAccount class with an initial balance.

        Args:
            balance (float): The initial balance of the account. Defaults to 0.
        """
        self.balance = balance

    def deposit(self, amount):
        """
        Deposit money into the account.

        Args:
            amount (float): The amount to deposit.
        """
        if amount > 0:
            self.balance += amount
            print(f"Deposited ${amount:.2f}. Current balance: ${self.balance:.2f}")
        else:
            print("Invalid deposit amount. Please enter a positive number.")

    def withdraw(self, amount):
        """
        Withdraw money from the account.

        Args:
            amount (float): The amount to withdraw.
        """
        if amount > 0 and amount <= self.balance:
            self.balance -= amount
            print(f"Withdrew ${amount:.2f}. Current balance: ${self.balance:.2f}")
        elif amount <= 0:
            print("Invalid withdrawal amount. Please enter a positive number.")
        else:
            print("Insufficient funds.")

    def check_balance(self):
        """
        Check the current balance of the account.
        """
        print(f"Current balance: ${self.balance:.2f}")


def main():
    # Create a new bank account with an initial balance
    balance = float(input("Enter the initial account balance: $"))
    account = BankAccount(balance)

    while True:
        # Display menu options
        print("\nBank Account Menu:")
        print("1. Deposit money")
        print("2. Withdraw money")
        print("3. Check balance")
        print("4. Exit")

        # Get user input
        choice = input("Enter your choice: ")

        if choice == "1":
            amount = float(input("Enter the amount to deposit: $"))
            account.deposit(amount)
        elif choice == "2":
            amount = float(input("Enter the amount to withdraw: $"))
            account.withdraw(amount)
        elif choice == "3":
            account.check_balance()
        elif choice == "4":
            print("Exiting program.")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
