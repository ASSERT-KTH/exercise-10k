Use map data structure with variables

## Reference Solution
```python
#include <iostream>
#include <map>
#include <string>

int main() {
    // Create a map with string keys and integer values
    std::map<std::string, int> studentGrades;

    // Add elements to the map
    studentGrades["John"] = 85;
    studentGrades["Alice"] = 92;
    studentGrades["Bob"] = 78;

    // Access elements in the map
    std::cout << "John's grade: " << studentGrades["John"] << std::endl;
    std::cout << "Alice's grade: " << studentGrades["Alice"] << std::endl;
    std::cout << "Bob's grade: " << studentGrades["Bob"] << std::endl;

    // Update an element in the map
    studentGrades["John"] = 90;
    std::cout << "John's updated grade: " << studentGrades["John"] << std::endl;

    // Remove an element from the map
    studentGrades.erase("Bob");
    std::cout << "Is Bob in the map? " << (studentGrades.find("Bob") != studentGrades.end()) << std::endl;

    return 0;
}
```
