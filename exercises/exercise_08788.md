## Exercise
Write a program that takes two numbers as input and returns their absolute difference.

## Reference Solution
```python
def absolute_difference(num1: float, num2: float) -> float:
    """
    Calculate the absolute difference between two numbers.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.

    Returns:
    float: The absolute difference between num1 and num2.
    """
    return abs(num1 - num2)


def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    # Calculate and display the absolute difference
    difference = absolute_difference(num1, num2)
    print(f"The absolute difference is: {difference}")


if __name__ == "__main__":
    main()
```
