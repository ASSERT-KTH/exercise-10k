## Exercise
Create a function that can handle different types of input and perform the correct operation.

## Reference Solution
```python
def handle_input(input1, input2, operation='add'):
    """
    This function handles different types of input and performs the correct operation.
    
    Args:
        input1 (int or float or list or tuple): The first input.
        input2 (int or float or list or tuple): The second input.
        operation (str, optional): The operation to be performed. Defaults to 'add'.
        
    Returns:
        int or float or list or tuple: The result of the operation.
    """
    
    # Check if both inputs are numbers
    if isinstance(input1, (int, float)) and isinstance(input2, (int, float)):
        # Perform addition
        if operation == 'add':
            # Return the sum of the two numbers
            return input1 + input2
        # Perform multiplication
        elif operation == 'multiply':
            # Return the product of the two numbers
            return input1 * input2
        # If the operation is neither 'add' nor 'multiply', return an error message
        else:
            return "Error: Invalid operation"
    
    # Check if both inputs are lists or tuples
    elif isinstance(input1, (list, tuple)) and isinstance(input2, (list, tuple)):
        # Perform concatenation
        if operation == 'concatenate':
            # Return the concatenation of the two lists or tuples
            return input1 + input2
        # Perform addition (element-wise)
        elif operation == 'add':
            # Check if both inputs have the same length
            if len(input1) != len(input2):
                return "Error: Inputs must have the same length"
            # Initialize an empty list to store the result
            result = []
            # Iterate over the elements of the inputs
            for i in range(len(input1)):
                # Append the sum of the corresponding elements to the result
                result.append(input1[i] + input2[i])
            # Return the result
            return result
        # If the operation is neither 'concatenate' nor 'add', return an error message
        else:
            return "Error: Invalid operation"
    
    # If the inputs are not of the same type, return an error message
    else:
        return "Error: Inputs must be of the same type"

# Test the function
print(handle_input(5, 7))  # Output: 12
print(handle_input(5, 7, 'multiply'))  # Output: 35
print(handle_input([1, 2, 3], [4, 5, 6], 'concatenate'))  # Output: [1, 2, 3, 4, 5, 6]
print(handle_input([1, 2, 3], [4, 5, 6], 'add'))  # Output: [5, 7, 9]
```
