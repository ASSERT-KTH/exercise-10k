## Exercise
Create a program that uses a while loop to simulate a quiz game, where the user can answer questions until they choose to quit.

## Reference Solution
```python
# Define a dictionary with questions, options, and answers
quiz_questions = {
    "What is the capital of France?": {
        "A": "Berlin",
        "B": "Paris",
        "C": "London",
        "D": "Rome",
        "answer": "B"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Which of the following programming languages is known for its simplicity and readability?": {
        "A": "Java",
        "B": "C++",
        "C": "Python",
        "D": "Ruby",
        "answer": "C"
    }
}

def quiz_game():
    # Initialize the score
    score = 0

    # Initialize the question counter
    question_counter = 0

    # Loop through the quiz questions
    while True:
        # Ask the user if they want to continue playing
        play_again = input("Do you want to play the quiz game? (yes/no): ")
        
        if play_again.lower() != "yes":
            break

        # Ask the user a question
        for question, options in quiz_questions.items():
            print(question)
            for option, value in options.items():
                if option != "answer":
                    print(f"{option}: {value}")
            answer = input("Enter the letter of your answer: ")
            
            # Check if the answer is correct
            if answer.upper() == options["answer"]:
                print("Correct!\n")
                score += 1
            else:
                print(f"Sorry, that's incorrect. The correct answer was {options['answer']}.\n")
            
            # Increment the question counter
            question_counter += 1

        # Print the final score
        print(f"Quiz game over. You answered {score} out of {question_counter} questions correctly.")

# Start the quiz game
quiz_game()
```
