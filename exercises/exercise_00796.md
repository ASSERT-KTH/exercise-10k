## Exercise
Create a program to calculate the correlation coefficient given two lists of numbers.

## Reference Solution
```python
import numpy as np

def calculate_correlation_coefficient(list1, list2):
    """
    Calculate the correlation coefficient between two lists of numbers.

    Args:
        list1 (list): The first list of numbers.
        list2 (list): The second list of numbers.

    Returns:
        float: The correlation coefficient.
    """
    # Convert lists to numpy arrays
    array1 = np.array(list1)
    array2 = np.array(list2)

    # Calculate the means of the arrays
    mean1 = np.mean(array1)
    mean2 = np.mean(array2)

    # Calculate the deviations from the means
    deviations1 = array1 - mean1
    deviations2 = array2 - mean2

    # Calculate the covariance and variances
    covariance = np.sum(deviations1 * deviations2)
    variance1 = np.sum(deviations1 ** 2)
    variance2 = np.sum(deviations2 ** 2)

    # Calculate the correlation coefficient
    correlation_coefficient = covariance / np.sqrt(variance1 * variance2)

    return correlation_coefficient

def main():
    # Example usage
    list1 = [1, 2, 3, 4, 5]
    list2 = [2, 3, 5, 7, 11]
    correlation_coefficient = calculate_correlation_coefficient(list1, list2)
    print(f"Correlation coefficient: {correlation_coefficient}")

if __name__ == "__main__":
    main()
```
