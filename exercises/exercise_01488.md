## Exercise
Write a program that demonstrates the use of string operations in programming.

## Reference Solution
```python
def main():
    # Define two strings
    str1 = "Hello"
    str2 = "World"

    # Concatenation
    print("Concatenation:")
    print(str1 + " " + str2)

    # Substring extraction
    print("\nSubstring Extraction:")
    print(str1[1:3])  # Extracts "el"

    # String length
    print("\nString Length:")
    print(len(str1))  # Prints the length of str1

    # String upper and lower case
    print("\nCase Conversion:")
    print(str1.upper())  # Converts str1 to upper case
    print(str1.lower())  # Converts str1 to lower case

    # String replacement
    print("\nString Replacement:")
    print(str1.replace("H", "J"))  # Replaces "H" with "J" in str1

    # String splitting
    print("\nString Splitting:")
    str3 = "apple,banana,cherry"
    print(str3.split(","))  # Splits str3 into a list of substrings

    # String joining
    print("\nString Joining:")
    fruits = ["apple", "banana", "cherry"]
    print(",".join(fruits))  # Joins the list of fruits into a single string

if __name__ == "__main__":
    main()
```
