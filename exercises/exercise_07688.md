## Exercise
Write a function that removes duplicates from a list.

## Reference Solution
```python
def remove_duplicates(input_list):
    """
    Removes duplicates from a list and returns the resulting list.

    Args:
        input_list (list): The list from which to remove duplicates.

    Returns:
        list: The list with duplicates removed.
    """
    return list(set(input_list))

# Example usage:
def main():
    my_list = [1, 2, 2, 3, 4, 4, 5, 6, 6]
    print("Original List:", my_list)
    print("List without duplicates:", remove_duplicates(my_list))

if __name__ == "__main__":
    main()
```

def remove_duplicates_ordered(input_list):
    """
    Removes duplicates from a list while preserving the original order.

    Args:
        input_list (list): The list from which to remove duplicates.

    Returns:
        list: The list with duplicates removed, preserving the original order.
    """
    seen = set()
    return [x for x in input_list if not (x in seen or seen.add(x))]

# Example usage:
def main():
    my_list = [1, 2, 2, 3, 4, 4, 5, 6, 6]
    print("Original List:", my_list)
    print("List without duplicates (ordered):", remove_duplicates_ordered(my_list))

if __name__ == "__main__":
    main()
