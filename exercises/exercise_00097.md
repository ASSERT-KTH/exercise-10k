## Exercise
Given a list of integers, find the second largest element

## Reference Solution

def find_second_largest(nums):
    """
    Find the second largest element in a list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The second largest element. If the list has less than 2 elements, returns None.
    """
    if len(nums) < 2:
        return None

    max_element = second_max_element = float('-inf')
    for num in nums:
        if num > max_element:
            second_max_element = max_element
            max_element = num
        elif num < max_element and num > second_max_element:
            second_max_element = num

    if second_max_element == float('-inf'):
        return None  # No second largest element found

    return second_max_element

# Example usage
numbers = [4, 2, 9, 6, 5, 1, 8, 3, 7]
print(find_second_largest(numbers))  # Output: 8
