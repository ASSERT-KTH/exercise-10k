## Exercise
Write a function that takes two lists as input, finds their union, and returns the union list.

## Reference Solution
```python
def union(list1, list2):
    """
    Returns the union of two input lists.

    Args:
        list1 (list): The first input list.
        list2 (list): The second input list.

    Returns:
        list: The union of the two input lists.
    """
    return list(set(list1 + list2))

# Example usage:
if __name__ == "__main__":
    list1 = [1, 2, 3, 4]
    list2 = [3, 4, 5, 6]
    print("List 1:", list1)
    print("List 2:", list2)
    print("Union:", union(list1, list2))
```

```python
def union_ordered(list1, list2):
    """
    Returns the union of two input lists, preserving the original order.

    Args:
        list1 (list): The first input list.
        list2 (list): The second input list.

    Returns:
        list: The union of the two input lists, preserving the original order.
    """
    seen = set()
    result = []
    for elem in list1 + list2:
        if elem not in seen:
            result.append(elem)
            seen.add(elem)
    return result

# Example usage:
if __name__ == "__main__":
    list1 = [1, 2, 3, 4]
    list2 = [3, 4, 5, 6]
    print("List 1:", list1)
    print("List 2:", list2)
    print("Union (ordered):", union_ordered(list1, list2))
```
