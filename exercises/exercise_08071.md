## Exercise
Write a function that composes two functions together

## Reference Solution
```python
def compose(f, g):
    """
    Compose two functions together.

    Args:
        f (function): The outer function.
        g (function): The inner function.

    Returns:
        function: A new function that is the composition of f and g.
    """
    def h(x):
        return f(g(x))
    return h

# Example usage:
def double(x):
    return 2 * x

def add_one(x):
    return x + 1

# Compose the functions
double_then_add_one = compose(add_one, double)

# Test the composed function
print(double_then_add_one(5))  # Output: 11
```

```python
def compose(f, g):
    return lambda x: f(g(x))

# Example usage:

add_one = lambda x: x + 1
```

```python
def compose(*functions):
    def h(x):
        result = x
        for f in reversed(functions):
            result = f(result)
        return result
    return h

# Example usage:
double = lambda x: 2 * x
add_one = lambda x: x + 1
square = lambda x: x ** 2

double_then_add_one_then_square = compose(square, add_one, double)
print(double_then_add_one_then_square(5))  # Output: 121
```
