## Exercise
Write a program to find the element at a given index in a array.

## Reference Solution
```python
def find_element_at_index(array, index):
    """
    Finds the element at a given index in an array.

    Args:
        array (list): The input array.
        index (int): The index of the element to find.

    Returns:
        The element at the given index if it exists, otherwise None.
    """
    if index < 0 or index >= len(array):
        return None
    return array[index]

# Example usage
def main():
    array = [1, 2, 3, 4, 5]
    index = 2
    element = find_element_at_index(array, index)
    if element is not None:
        print(f"The element at index {index} is {element}")
    else:
        print(f"No element found at index {index}")

if __name__ == "__main__":
    main()
```
