## Exercise
Generate all permutations of a given list of integers

## Reference Solution
```python
def generate_permutations(nums):
    """
    Generate all permutations of a given list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        list: A list of lists, where each sublist is a permutation of the input list.
    """
    # Base case: if the list is empty or contains only one element, return the list
    if len(nums) <= 1:
        return [nums]

    # Initialize an empty list to store the permutations
    permutations = []

    # For each element in the list
    for i, num in enumerate(nums):
        # Create a new list that contains all elements except the current one
        remaining_nums = nums[:i] + nums[i + 1:]

        # Generate all permutations of the remaining list
        for perm in generate_permutations(remaining_nums):
            # Append the current element to the front of each permutation
            permutations.append([num] + perm)

    return permutations


def main():
    # Example usage
    nums = [1, 2, 3]
    permutations = generate_permutations(nums)

    print("Permutations:")
    for i, perm in enumerate(permutations):
        print(f"Permutation {i + 1}: {perm}")


if __name__ == "__main__":
    main()
```
