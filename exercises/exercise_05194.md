## Exercise
Write a program that asks the user for their interests and then recommends a list of products based on their interests.

## Reference Solution
```python
# Define a dictionary of products with their corresponding interests
products = {
    "books": ["Harry Potter", "The Lord of the Rings", "Pride and Prejudice"],
    "music": ["Guitar", "Piano", "Vinyl Records"],
    "sports": ["Basketball", "Football", "Tennis Racket"],
    "art": ["Paints", "Canvas", "Sketchbook"],
    "travel": ["Backpack", "Suitcase", "Travel Guide"]
}

def get_interests():
    """Asks the user for their interests and returns a list of interests"""
    print("What are your interests? (separate with comma)")
    print("Available interests: books, music, sports, art, travel")
    interests = input().lower().split(",")
    interests = [interest.strip() for interest in interests]
    return interests

def recommend_products(interests):
    """Recommends products based on the user's interests"""
    recommended_products = []
    for interest in interests:
        if interest in products:
            recommended_products.extend(products[interest])
    return recommended_products

def main():
    interests = get_interests()
    recommended_products = recommend_products(interests)
    if recommended_products:
        print("Based on your interests, we recommend the following products:")
        for product in recommended_products:
            print(product)
    else:
        print("Sorry, we couldn't find any products that match your interests.")

if __name__ == "__main__":
    main()
```
