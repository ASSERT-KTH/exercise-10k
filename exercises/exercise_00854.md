## Exercise
Write a function that uses tail recursion to calculate the factorial of a number.

## Reference Solution
```python
def factorial(n, accumulator=1):
    """
    Calculate the factorial of a number using tail recursion.

    Args:
    n (int): The number to calculate the factorial of.
    accumulator (int): The accumulator for the factorial calculation. Defaults to 1.

    Returns:
    int: The factorial of n.
    """
    # Base case: if n is 0 or 1, return the accumulator
    if n == 0 or n == 1:
        return accumulator
    # Recursive case: call the function with n-1 and the accumulator multiplied by n
    else:
        return factorial(n-1, accumulator * n)

# Example usage
def main():
    num = 5
    result = factorial(num)
    print(f"The factorial of {num} is {result}")

if __name__ == "__main__":
    main()
```
