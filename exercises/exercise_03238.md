## Exercise
Write a program to search for an element in a rotated sorted array.

## Reference Solution

def search_rotated_array(arr, target):
    """
    Searches for an element in a rotated sorted array.

    Args:
        arr (list): The rotated sorted array.
        target (int): The element to search for.

    Returns:
        int: The index of the target element if found, -1 otherwise.
    """
    if not arr:
        return -1

    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid

        # Check if the left half is sorted
        if arr[left] <= arr[mid]:
            # If the target is in the sorted left half, update the right pointer
            if arr[left] <= target < arr[mid]:
                right = mid - 1
            else:
                left = mid + 1
        # If the left half is not sorted, the right half must be sorted
        else:
            # If the target is in the sorted right half, update the left pointer
            if arr[mid] < target <= arr[right]:
                left = mid + 1
            else:
                right = mid - 1

    return -1


# Example usage:
if __name__ == "__main__":
    arr = [4, 5, 6, 7, 0, 1, 2]
    target = 0
    result = search_rotated_array(arr, target)
    if result != -1:
        print(f"Target {target} found at index {result}.")
    else:
        print(f"Target {target} not found in the array.")
