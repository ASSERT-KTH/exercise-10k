## Exercise
Create a function that takes two functions as arguments and returns their composition

## Reference Solution
```python
def compose(f, g):
    """
    Returns the composition of two functions.

    Args:
        f (function): The outer function.
        g (function): The inner function.

    Returns:
        function: The composition of f and g.
    """
    def h(x):
        return f(g(x))
    return h

# Example usage:
def double(x):
    return 2 * x

def add_one(x):
    return x + 1

# Create a new function that doubles and then adds one
double_and_add_one = compose(add_one, double)

print(double_and_add_one(5))  # Output: 11
```

```python
def compose(f, g):
    return lambda x: f(g(x))

# Example usage:
double = lambda x: 2 * x
add_one = lambda x: x + 1

double_and_add_one = compose(add_one, double)
print(double_and_add_one(5))  # Output: 11
```

```python
import functools

def compose(*funcs):
    return functools.reduce(lambda f, g: lambda x: f(g(x)), funcs)

# Example usage:
double = lambda x: 2 * x
add_one = lambda x: x + 1
triple = lambda x: 3 * x

double_add_one_triple = compose(triple, add_one, double)
print(double_add_one_triple(5))  # Output: 33
```
