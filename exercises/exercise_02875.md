## Exercise
Write a function to rotate a given string by a specified number of positions

## Reference Solution
```python
def rotate_string(input_str, num_positions):
    """
    Rotate a given string by a specified number of positions.

    Args:
        input_str (str): The input string to be rotated.
        num_positions (int): The number of positions to rotate the string.

    Returns:
        str: The rotated string.
    """
    num_positions = num_positions % len(input_str)  # Handle cases where num_positions is greater than the length of the string
    return input_str[-num_positions:] + input_str[:-num_positions]

# Example usage
def main():
    input_str = "HelloWorld"
    num_positions = 3
    print(f"Original string: {input_str}")
    print(f"Rotated string: {rotate_string(input_str, num_positions)}")

if __name__ == "__main__":
    main()
```
