## Exercise
Create a program that calculates the distance between two points given their coordinates.

## Reference Solution
```python
import math

def calculate_distance(x1, y1, z1=None, x2=None, y2=None, z2=None):
    """
    Calculate the distance between two points.

    Parameters:
    x1 (float): The x-coordinate of the first point.
    y1 (float): The y-coordinate of the first point.
    z1 (float, optional): The z-coordinate of the first point. Defaults to None.
    x2 (float, optional): The x-coordinate of the second point. Defaults to None.
    y2 (float, optional): The y-coordinate of the second point. Defaults to None.
    z2 (float, optional): The z-coordinate of the second point. Defaults to None.

    Returns:
    float: The distance between the two points.
    """
    if z1 is None and z2 is None:
        # 2D distance calculation
        distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
    else:
        # 3D distance calculation
        distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2 + (z2 - z1)**2)

    return distance

def main():
    print("Distance Calculator")
    print("-------------------")

    # Get the coordinates of the two points
    x1 = float(input("Enter the x-coordinate of the first point: "))
    y1 = float(input("Enter the y-coordinate of the first point: "))

    x2 = float(input("Enter the x-coordinate of the second point: "))
    y2 = float(input("Enter the y-coordinate of the second point: "))

    # Ask if the points are in 3D space
    is_3d = input("Are the points in 3D space? (yes/no): ")

    if is_3d.lower() == "yes":
        z1 = float(input("Enter the z-coordinate of the first point: "))
        z2 = float(input("Enter the z-coordinate of the second point: "))

        distance = calculate_distance(x1, y1, z1, x2, y2, z2)
    else:
        distance = calculate_distance(x1, y1, x2=x2, y2=y2)

    print(f"The distance between the two points is: {distance}")

if __name__ == "__main__":
    main()
```
