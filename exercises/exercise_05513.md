## Exercise
Write a program to find the shortest path between two nodes in a graph using Dijkstra's algorithm.

## Reference Solution
```python
import sys
import heapq

def dijkstra(graph, start_node, end_node):
    # Create a dictionary to store the distance to each node
    distances = {node: float('infinity') for node in graph}
    distances[start_node] = 0

    # Create a dictionary to store the previous node in the shortest path
    previous_nodes = {node: None for node in graph}

    # Create a priority queue to store the nodes to visit
    priority_queue = [(0, start_node)]

    while priority_queue:
        # Get the node with the smallest distance
        current_distance, current_node = heapq.heappop(priority_queue)

        # If the current node is the end node, we can stop
        if current_node == end_node:
            break

        # If the current distance is greater than the stored distance, skip this node
        if current_distance > distances[current_node]:
            continue

        # Visit all the neighbors of the current node
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight

            # If the calculated distance is smaller than the stored distance, update it
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                previous_nodes[neighbor] = current_node
                heapq.heappush(priority_queue, (distance, neighbor))

    # Build the shortest path
    path = []
    current_node = end_node
    while current_node is not None:
        path.append(current_node)
        current_node = previous_nodes[current_node]
    path.reverse()

    return path, distances[end_node]

# Example usage:
graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

start_node = 'A'
end_node = 'D'

path, distance = dijkstra(graph, start_node, end_node)
print(f"The shortest path from {start_node} to {end_node} is: {path}")
print(f"The shortest distance from {start_node} to {end_node} is: {distance}")
```
