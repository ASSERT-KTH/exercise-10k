## Exercise
Remove and return the value of a specific key.

## Reference Solution

def remove_and_return_key_value(dictionary, key):
    """
    Removes and returns the value of a specific key from a dictionary.
    
    Args:
        dictionary (dict): The dictionary to remove the key from.
        key (any): The key to remove and return its value.
    
    Returns:
        any: The value of the removed key, or None if the key does not exist.
    """
    return dictionary.pop(key, None)

# Example usage:
my_dict = {"a": 1, "b": 2, "c": 3}
print(remove_and_return_key_value(my_dict, "b"))  # Output: 2
print(my_dict)  # Output: {'a': 1, 'c': 3}
