## Exercise
Write a program to create a simple chatbot that responds to basic user queries.

## Reference Solution
```python
import random

# Define a dictionary with intents and responses
intents = {
    'greeting': {
        'patterns': ['hello', 'hi', 'hey'],
        'responses': ['Hi, how are you?', 'Hello! How can I assist you?', 'Hey, what\'s up?']
    },
    'introduction': {
        'patterns': ['who are you', 'what is your name'],
        'responses': ['I\'m a simple chatbot designed to assist with basic queries.', 'My name is Chatbot, and I\'m here to help.']
    },
    'farewell': {
        'patterns': ['bye', 'goodbye', 'see you later'],
        'responses': ['Goodbye! It was nice chatting with you.', 'See you later! Have a great day.', 'Bye for now!']
    }
}

def match_intent(message):
    """
    Match the user's message with an intent.
    """
    for intent, values in intents.items():
        for pattern in values['patterns']:
            if pattern.lower() in message.lower():
                return intent
    return None

def get_response(intent):
    """
    Get a random response for the matched intent.
    """
    if intent:
        return random.choice(intents[intent]['responses'])
    else:
        return 'Sorry, I didn\'t understand that.'

def main():
    print('Welcome to the chatbot! Type "quit" to exit.')
    while True:
        message = input('You: ')
        if message.lower() == 'quit':
            break
        intent = match_intent(message)
        response = get_response(intent)
        print('Chatbot:', response)

if __name__ == '__main__':
    main()
```
