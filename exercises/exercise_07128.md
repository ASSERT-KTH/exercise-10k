## Exercise
Implement a basic sorting algorithm to arrange a list of numbers in ascending order. The algorithm should be able to handle a list of any size and should be case-insensitive when comparing strings.

## Reference Solution

def bubble_sort(arr):
    """
    Sorts a list of elements in ascending order using the Bubble Sort algorithm.

    Args:
        arr (list): The list of elements to be sorted.

    Returns:
        list: The sorted list of elements.
    """
    n = len(arr)

    for i in range(n):
        # Create a flag that will allow the function to terminate early if there's nothing left to sort
        swapped = False

        for j in range(0, n - i - 1):
            # Check if the current element is greater than the next element
            if isinstance(arr[j], str) and isinstance(arr[j + 1], str):
                # If both elements are strings, compare them case-insensitively
                if arr[j].lower() > arr[j + 1].lower():
                    arr[j], arr[j + 1] = arr[j + 1], arr[j]
                    swapped = True
            elif arr[j] > arr[j + 1]:
                # If the elements are not strings, compare them normally
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True

        # If no two elements were swapped in the inner loop, the list is sorted and we can terminate early
        if not swapped:
            break

    return arr


# Example usage:
if __name__ == "__main__":
    numbers = [64, 34, 25, 12, 22, 11, 90]
    print("Original list:", numbers)
    print("Sorted list:", bubble_sort(numbers))

    strings = ["dog", "Cat", "apple", "Banana", "elephant"]
    print("\nOriginal list:", strings)
    print("Sorted list:", bubble_sort(strings))
