## Exercise
Write a program to analyze a piece of text and print out the number of words, characters, and sentences.

## Reference Solution
```python
import re

def analyze_text(text):
    """
    Analyze a piece of text and return the number of words, characters, and sentences.

    Args:
        text (str): The text to be analyzed.

    Returns:
        dict: A dictionary containing the number of words, characters, and sentences.
    """
    # Remove leading and trailing whitespace
    text = text.strip()

    # Count the number of words
    words = re.findall(r'\b\w+\b', text)
    num_words = len(words)

    # Count the number of characters
    num_chars = len(text)

    # Count the number of sentences
    sentences = re.split(r'[.!?]', text)
    # Remove empty strings from the list
    sentences = [s for s in sentences if s]
    num_sentences = len(sentences)

    return {
        'words': num_words,
        'characters': num_chars,
        'sentences': num_sentences
    }

def main():
    # Example usage
    text = "This is a sample text. It has multiple sentences. And some punctuation!"
    analysis = analyze_text(text)

    print("Text Analysis Results:")
    print("------------------------")
    print(f"Number of words: {analysis['words']}")
    print(f"Number of characters: {analysis['characters']}")
    print(f"Number of sentences: {analysis['sentences']}")

if __name__ == "__main__":
    main()
```
