## Exercise
Write a program to calculate the median of a list of numbers

## Reference Solution
```python
def calculate_median(numbers):
    """
    Calculate the median of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The median of the list of numbers.
    """
    # Sort the list of numbers
    numbers.sort()

    # Get the length of the list
    n = len(numbers)

    # If the list is empty, return None
    if n == 0:
        return None

    # If the list has an odd length, the median is the middle value
    if n % 2 == 1:
        return numbers[n // 2]

    # If the list has an even length, the median is the average of the two middle values
    else:
        return (numbers[n // 2 - 1] + numbers[n // 2]) / 2


def main():
    # Example usage
    numbers = [1, 3, 5, 7, 9]
    median = calculate_median(numbers)
    print("Median:", median)

    numbers = [1, 2, 3, 4, 5, 6]
    median = calculate_median(numbers)
    print("Median:", median)


if __name__ == "__main__":
    main()
```
