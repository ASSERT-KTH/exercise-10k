## Exercise
Create a function to initialize an empty dictionary and add key-value pairs to it.

## Reference Solution
```python
def create_and_populate_dict(keys, values):
    """
    Initialize an empty dictionary and add key-value pairs to it.

    Args:
        keys (list): A list of keys for the dictionary.
        values (list): A list of values for the dictionary.

    Returns:
        dict: The populated dictionary.

    Raises:
        ValueError: If the lengths of the keys and values lists do not match.
    """
    if len(keys) != len(values):
        raise ValueError("The lengths of the keys and values lists must match.")

    # Initialize an empty dictionary
    populated_dict = {}

    # Add key-value pairs to the dictionary
    for key, value in zip(keys, values):
        populated_dict[key] = value

    return populated_dict


# Example usage:
if __name__ == "__main__":
    keys = ["name", "age", "city"]
    values = ["John Doe", 30, "New York"]
    result = create_and_populate_dict(keys, values)
    print(result)
```
