## Exercise
Develop a program that asks the user a series of subtraction questions and keeps track of the score.

## Reference Solution
```python
import random

def subtraction_quiz(num_questions):
    """
    Ask the user a series of subtraction questions and keep track of the score.

    Args:
        num_questions (int): The number of questions to ask.

    Returns:
        None
    """
    score = 0
    for i in range(num_questions):
        num1 = random.randint(1, 100)
        num2 = random.randint(1, num1)  # Ensure num2 is less than or equal to num1
        correct_answer = num1 - num2

        user_answer = input(f"What is {num1} - {num2}? ")
        try:
            user_answer = int(user_answer)
            if user_answer == correct_answer:
                print("Correct!\n")
                score += 1
            else:
                print(f"Sorry, that's incorrect. The correct answer is {correct_answer}.\n")
        except ValueError:
            print(f"Sorry, that's not a valid answer. The correct answer is {correct_answer}.\n")

    print(f"Quiz complete! You scored {score} out of {num_questions}.")

def main():
    num_questions = int(input("How many questions would you like to answer? "))
    subtraction_quiz(num_questions)

if __name__ == "__main__":
    main()
```
