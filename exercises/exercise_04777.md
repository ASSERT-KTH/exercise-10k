## Exercise
Write a program that converts a month from a number to a name or vice versa. If the user inputs '1', the program prints 'January'. If the user inputs 'January', the program prints '1'.

## Reference Solution
```python
def month_converter():
    """
    Converts a month from a number to a name or vice versa.
    """
    # Dictionary mapping month numbers to names
    month_dict = {
        "1": "January", "2": "February", "3": "March", "4": "April",
        "5": "May", "6": "June", "7": "July", "8": "August",
        "9": "September", "10": "October", "11": "November", "12": "December"
    }

    # Dictionary mapping month names to numbers
    month_dict_inv = {v: k for k, v in month_dict.items()}

    # Get user input
    user_input = input("Enter a month number or name: ")

    # Check if input is a number
    if user_input.isdigit() and 1 <= int(user_input) <= 12:
        print(month_dict[user_input])
    # Check if input is a month name
    elif user_input.capitalize() in month_dict_inv:
        print(month_dict_inv[user_input.capitalize()])
    else:
        print("Invalid input. Please enter a valid month number or name.")

if __name__ == "__main__":
    month_converter()
```
