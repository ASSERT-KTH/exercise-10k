## Exercise
Write a program to find the majority element in a given list of integers, where the majority element occurs more than n/2 times.

## Reference Solution
```python
def find_majority_element(nums):
    """
    Find the majority element in a given list of integers.

    Args:
    nums (list): A list of integers.

    Returns:
    int: The majority element if it exists, otherwise None.
    """
    if not nums:
        return None

    # Initialize the majority element and count
    majority_element = nums[0]
    count = 1

    # Iterate through the list
    for num in nums[1:]:
        # If the count is zero, change the majority element
        if count == 0:
            majority_element = num
            count = 1
        # Increment the count if we see the majority element
        elif num == majority_element:
            count += 1
        # Decrement the count if we see a different element
        else:
            count -= 1

    # Check if the majority element occurs more than n/2 times
    if nums.count(majority_element) > len(nums) / 2:
        return majority_element
    else:
        return None


# Example usage
if __name__ == "__main__":
    nums = [3, 3, 4, 2, 4, 4, 2, 4, 4]
    majority_element = find_majority_element(nums)

    if majority_element is not None:
        print(f"The majority element is {majority_element}")
    else:
        print("No majority element found")
```
