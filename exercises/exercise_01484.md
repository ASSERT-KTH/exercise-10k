## Exercise
Write a program that compares the data types of two variables and prints out the result.

## Reference Solution
```python
def compare_data_types(var1, var2):
    """
    This function compares the data types of two variables and prints out the result.

    Args:
        var1 (any): The first variable to compare.
        var2 (any): The second variable to compare.
    """

    # Get the data types of the variables
    type_var1 = type(var1)
    type_var2 = type(var2)

    # Print the data types
    print(f"Data type of var1: {type_var1}")
    print(f"Data type of var2: {type_var2}")

    # Compare the data types
    if type_var1 == type_var2:
        print("The data types are the same.")
    else:
        print("The data types are different.")

# Test the function
var1 = 10
var2 = "Hello"
compare_data_types(var1, var2)

var3 = 20.5
var4 = 30.7
compare_data_types(var3, var4)

var5 = True
var6 = False
compare_data_types(var5, var6)
```
