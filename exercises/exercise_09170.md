## Exercise
Create a string with Unicode characters and print out the Unicode code points for each character, then print out the code points.

## Reference Solution
```python
def print_unicode_code_points(input_string):
    """
    Prints the Unicode code points for each character in the input string.

    Args:
        input_string (str): The input string containing Unicode characters.
    """
    # Iterate over each character in the input string
    for char in input_string:
        # Print the character and its Unicode code point
        print(f"Character: {char}, Unicode Code Point: {ord(char)}")


def main():
    # Create a string with Unicode characters
    unicode_string = "Hello, àçé"

    # Print the input string
    print("Input String:")
    print(unicode_string)

    # Print the Unicode code points for each character
    print("\nUnicode Code Points:")
    print_unicode_code_points(unicode_string)


if __name__ == "__main__":
    main()
```
