## Exercise
Read a name and age from the user and print out a greeting message with the name and age.

## Reference Solution
```python
def get_user_info():
    """Get user's name and age."""
    name = input("Please enter your name: ")
    while True:
        try:
            age = int(input("Please enter your age: "))
            break
        except ValueError:
            print("Invalid input. Age must be an integer.")
    return name, age

def print_greeting(name, age):
    """Print a greeting message with the user's name and age."""
    print(f"Hello, {name}! You are {age} years old.")

def main():
    name, age = get_user_info()
    print_greeting(name, age)

if __name__ == "__main__":
    main()
```
