## Exercise
Write a program that takes a time interval as input and counts down to zero

## Reference Solution
```python
import time

def countdown(t):
    """
    Counts down from the given time interval to zero.

    Args:
        t (int): Time interval in seconds.
    """
    while t:
        mins, secs = divmod(t, 60)
        timer = '{:02d}:{:02d}'.format(mins, secs)
        print(timer, end="\r")
        time.sleep(1)
        t -= 1
    print('Time Up!!')

def main():
    # Get the time interval from the user
    t = input("Enter the time in seconds: ")

    # Validate the input
    try:
        t = int(t)
        if t < 0:
            print("Time cannot be negative.")
            return
    except ValueError:
        print("Invalid input. Please enter a whole number.")
        return

    # Start the countdown
    countdown(t)

if __name__ == "__main__":
    main()
```
