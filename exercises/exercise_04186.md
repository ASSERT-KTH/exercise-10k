## Exercise
Write a function that generates a Mad Libs story based on user input.

## Reference Solution
```python
def get_word(prompt):
    """Get a word from the user."""
    while True:
        word = input(prompt)
        if word:
            return word
        print("Please enter a word.")

def generate_mad_libs_story():
    """Generate a Mad Libs story based on user input."""
    print("Welcome to the Mad Libs story generator!")
    print("Please provide the following words:")

    # Get words from the user
    noun1 = get_word("Enter a noun (e.g., book, car, apple): ")
    verb1 = get_word("Enter a verb (e.g., run, jump, read): ")
    adjective1 = get_word("Enter an adjective (e.g., happy, tall, green): ")
    noun2 = get_word("Enter another noun (e.g., dog, city, pencil): ")
    verb2 = get_word("Enter another verb (e.g., dance, sing, write): ")
    adjective2 = get_word("Enter another adjective (e.g., big, soft, blue): ")

    # Story template
    story_template = """
    Yesterday, I saw a {adjective1} {noun1} walking down the street.
    I was so surprised that I started to {verb1} all the way home.
    When I got home, I saw my {adjective2} {noun2} waiting for me.
    I was so happy to see it that I started to {verb2} with joy.
    """

    # Fill in the story template with the user's words
    story = story_template.format(
        adjective1=adjective1,
        noun1=noun1,
        verb1=verb1,
        adjective2=adjective2,
        noun2=noun2,
        verb2=verb2
    )

    # Print the generated story
    print("\nHere is your Mad Libs story:")
    print(story)

# Example usage
if __name__ == "__main__":
    generate_mad_libs_story()
```
