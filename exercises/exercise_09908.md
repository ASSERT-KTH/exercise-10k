## Exercise
Write a program that allows users to create and manage a grocery list.

## Reference Solution
```python
class GroceryList:
    def __init__(self):
        self.grocery_list = {}

    def add_item(self):
        """Add item to the grocery list."""
        item_name = input("Enter item name: ")
        quantity = int(input("Enter quantity: "))
        if item_name in self.grocery_list:
            print(f"{item_name} already exists in the list. Updating quantity.")
            self.grocery_list[item_name] += quantity
        else:
            self.grocery_list[item_name] = quantity
        print(f"{item_name} added to the list.")

    def remove_item(self):
        """Remove item from the grocery list."""
        item_name = input("Enter item name: ")
        if item_name in self.grocery_list:
            quantity = int(input("Enter quantity to remove: "))
            if quantity <= self.grocery_list[item_name]:
                self.grocery_list[item_name] -= quantity
                if self.grocery_list[item_name] == 0:
                    del self.grocery_list[item_name]
                print(f"{item_name} removed from the list.")
            else:
                print(f"Invalid quantity. You can't remove more than {self.grocery_list[item_name]} {item_name}.")
        else:
            print(f"{item_name} does not exist in the list.")

    def view_list(self):
        """View the grocery list."""
        print("Grocery List:")
        for item, quantity in self.grocery_list.items():
            print(f"{item}: {quantity}")

    def update_quantity(self):
        """Update quantity of an item in the grocery list."""
        item_name = input("Enter item name: ")
        if item_name in self.grocery_list:
            new_quantity = int(input("Enter new quantity: "))
            self.grocery_list[item_name] = new_quantity
            print(f"Quantity of {item_name} updated.")
        else:
            print(f"{item_name} does not exist in the list.")

def main():
    grocery_list = GroceryList()
    while True:
        print("\nGrocery List Manager Menu:")
        print("1. Add item to the list")
        print("2. Remove item from the list")
        print("3. View the list")
        print("4. Update quantity of an item")
        print("5. Quit")
        choice = input("Enter your choice: ")
        if choice == "1":
            grocery_list.add_item()
        elif choice == "2":
            grocery_list.remove_item()
        elif choice == "3":
            grocery_list.view_list()
        elif choice == "4":
            grocery_list.update_quantity()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
