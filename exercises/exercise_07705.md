## Exercise
Convert a given string to uppercase or lowercase.

## Reference Solution
```python
def convert_case(input_string, case):
    """
    Convert a given string to uppercase or lowercase.

    Args:
        input_string (str): The input string to convert.
        case (str): The case to convert to. Can be 'upper' or 'lower'.

    Returns:
        str: The converted string.
    """
    if case == 'upper':
        return input_string.upper()
    elif case == 'lower':
        return input_string.lower()
    else:
        raise ValueError("Invalid case. Please specify 'upper' or 'lower'.")

# Example usage:
input_string = "Hello World"
print(convert_case(input_string, 'upper'))  # Output: HELLO WORLD
print(convert_case(input_string, 'lower'))  # Output: hello world
```

def convert_case(input_string, case):
    case_methods = {
        'upper': str.upper,
        'lower': str.lower
    }
    if case in case_methods:
        return case_methods[case](input_string)
    else:
        raise ValueError("Invalid case. Please specify 'upper' or 'lower'.")

# Example usage:
input_string = "Hello World"
print(convert_case(input_string, 'upper'))  # Output: HELLO WORLD
print(convert_case(input_string, 'lower'))  # Output: hello world
