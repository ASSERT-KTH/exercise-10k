## Exercise
Write a function to check if two dictionaries are equal

def are_dictionaries_equal(dict1, dict2):
    """
    Check if two dictionaries are equal.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        bool: True if the dictionaries are equal, False otherwise.
    """
    return dict1 == dict2

# Example usage:
dict1 = {"a": 1, "b": 2}
dict2 = {"a": 1, "b": 2}
dict3 = {"a": 1, "b": 3}

print(are_dictionaries_equal(dict1, dict2))  # Output: True
print(are_dictionaries_equal(dict1, dict3))  # Output: False

import collections.abc

def are_dictionaries_deep_equal(dict1, dict2):
    """
    Recursively check if two dictionaries are equal.

    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.

    Returns:
        bool: True if the dictionaries are equal, False otherwise.
    """
    if not isinstance(dict1, collections.abc.Mapping) or not isinstance(dict2, collections.abc.Mapping):
        return dict1 == dict2

    if set(dict1.keys()) != set(dict2.keys()):
        return False

    for key in dict1:
        if isinstance(dict1[key], collections.abc.Mapping) and isinstance(dict2[key], collections.abc.Mapping):
            if not are_dictionaries_deep_equal(dict1[key], dict2[key]):
                return False
        elif dict1[key] != dict2[key]:
            return False

    return True

# Example usage:
dict1 = {"a": 1, "b": {"c": 2, "d": 3}}
dict2 = {"a": 1, "b": {"c": 2, "d": 3}}
dict3 = {"a": 1, "b": {"c": 2, "d": 4}}

print(are_dictionaries_deep_equal(dict1, dict2))  # Output: True
print(are_dictionaries_deep_equal(dict1, dict3))  # Output: False
