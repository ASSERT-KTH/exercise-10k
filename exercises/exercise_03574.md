## Exercise
Write a function that expands a flat dictionary into a nested dictionary

## Reference Solution
```python
def expand_dict(flat_dict, separator='.'):
    """
    Expands a flat dictionary into a nested dictionary.

    Args:
        flat_dict (dict): The flat dictionary to expand.
        separator (str): The separator character used to denote the path in the flat dictionary. Defaults to '.'.

    Returns:
        dict: The expanded nested dictionary.
    """
    nested_dict = {}
    for key, value in flat_dict.items():
        path = key.split(separator)
        current_dict = nested_dict
        for part in path[:-1]:
            if part not in current_dict:
                current_dict[part] = {}
            current_dict = current_dict[part]
        current_dict[path[-1]] = value
    return nested_dict

# Example usage:
flat_dict = {
    'a.b.c': 1,
    'a.b.d': 2,
    'a.e': 3,
    'f': 4
}

nested_dict = expand_dict(flat_dict)
print(nested_dict)
```
