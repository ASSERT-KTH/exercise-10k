## Exercise
Write a function to filter a dictionary based on a given condition.

## Reference Solution
```python
def filter_dictionary(dictionary, condition):
    """
    Filters a dictionary based on a given condition.

    Args:
        dictionary (dict): The dictionary to be filtered.
        condition (function): A function that takes a key-value pair and returns a boolean.

    Returns:
        dict: A new dictionary containing only the key-value pairs that meet the condition.
    """
    return {key: value for key, value in dictionary.items() if condition(key, value)}

# Example usage:
if __name__ == "__main__":
    # Define a dictionary
    dictionary = {
        "a": 1,
        "b": 2,
        "c": 3,
        "d": 4
    }

    # Define a condition function
    def condition(key, value):
        return value > 2

    # Filter the dictionary
    filtered_dictionary = filter_dictionary(dictionary, condition)

    # Print the filtered dictionary
    print(filtered_dictionary)
```
