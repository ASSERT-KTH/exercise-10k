## Exercise
Write a function that unregisters a previously registered function with a given name.

## Reference Solution
```python
registered_functions = {}

def register_function(name, func):
    """
    Registers a function with a given name.

    Args:
        name (str): The name of the function to be registered.
        func (function): The function to be registered.

    Returns:
        None
    """
    # Add the function to the dictionary
    registered_functions[name] = func

def unregister_function(name):
    """
    Unregisters a previously registered function with a given name.

    Args:
        name (str): The name of the function to be unregistered.

    Returns:
        bool: True if the function was unregistered successfully, False otherwise.
    """
    # Check if the function is registered
    if name in registered_functions:
        # If it is, remove it from the dictionary
        del registered_functions[name]
        return True
    else:
        # If it's not, return False
        return False

# Example usage:
def example_function():
    print("This is an example function.")

# Register the example function
register_function("example", example_function)

# Unregister the example function
print(unregister_function("example"))  # Should print: True
print(unregister_function("example"))  # Should print: False
```
