## Exercise
Create an expression with logical operators and evaluate it using short-circuit evaluation, then print out the result.

## Reference Solution
```python
def main():
    # Define two functions to be used in the expression
    def is_positive(x):
        print(f"Checking if {x} is positive...")
        return x > 0

    def is_even(x):
        print(f"Checking if {x} is even...")
        return x % 2 == 0

    # Use short-circuit evaluation with logical operators
    x = 5
    result1 = is_positive(x) and is_even(x)
    print(f"Result 1: {result1}")

    x = 4
    result2 = is_positive(x) and is_even(x)
    print(f"Result 2: {result2}")

    x = -4
    result3 = is_positive(x) or is_even(x)
    print(f"Result 3: {result3}")

if __name__ == "__main__":
    main()
```
