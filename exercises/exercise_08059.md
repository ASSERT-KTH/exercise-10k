## Exercise
Write a program that uses an if/else statement to manage personal finances with investments based on user input.

## Reference Solution
```python
class PersonalFinanceManager:
    def __init__(self):
        self.cash = 0
        self.investments = {}

    def add_cash(self, amount):
        self.cash += amount

    def add_investment(self, name, amount, rate):
        self.investments[name] = {'amount': amount, 'rate': rate}

    def calculate_total(self):
        total = self.cash
        for investment in self.investments.values():
            total += investment['amount'] * (1 + investment['rate'] / 100)
        return total

    def get_recommendation(self, goal):
        total = self.calculate_total()
        if total < 1000:
            return "You may want to consider saving more to reach your goal of {}".format(goal)
        elif total < 5000:
            return "You're getting closer to your goal of {}. Consider investing in a low-risk option to grow your wealth".format(goal)
        else:
            return "Congratulations, you've reached a significant milestone! Consider diversifying your investments to achieve your goal of {}".format(goal)


def main():
    manager = PersonalFinanceManager()

    print("Welcome to the Personal Finance Manager!")
    print("Please enter your current cash balance:")
    cash = float(input())
    manager.add_cash(cash)

    print("Do you have any investments? (yes/no)")
    response = input()
    if response.lower() == "yes":
        while True:
            print("Enter the name of the investment:")
            name = input()
            print("Enter the amount invested:")
            amount = float(input())
            print("Enter the expected rate of return (in %):")
            rate = float(input())
            manager.add_investment(name, amount, rate)
            print("Do you have another investment? (yes/no)")
            response = input()
            if response.lower() != "yes":
                break

    print("What is your financial goal?")
    goal = input()

    print("Your current total wealth is: ${:.2f}".format(manager.calculate_total()))
    print(manager.get_recommendation(goal))


if __name__ == "__main__":
    main()
```
