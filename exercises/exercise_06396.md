## Exercise
Develop a program that takes in the first number, last number, and increment, then generates the number sequence. Use if/else statements to check the increment and generate the sequence

## Reference Solution
```python
def generate_sequence(first_num, last_num, increment):
    """
    Generates a number sequence based on the first number, last number, and increment.

    Args:
        first_num (int): The first number in the sequence.
        last_num (int): The last number in the sequence.
        increment (int): The increment between numbers in the sequence.

    Returns:
        list: A list of numbers in the sequence.
    """
    sequence = []

    if increment > 0:
        # Generate sequence in ascending order
        num = first_num
        while num <= last_num:
            sequence.append(num)
            num += increment
    elif increment < 0:
        # Generate sequence in descending order
        num = first_num
        while num >= last_num:
            sequence.append(num)
            num += increment
    else:
        print("Error: Increment cannot be zero.")
        return None

    return sequence


def main():
    first_num = int(input("Enter the first number: "))
    last_num = int(input("Enter the last number: "))
    increment = int(input("Enter the increment: "))

    sequence = generate_sequence(first_num, last_num, increment)

    if sequence is not None:
        print("Number Sequence:")
        print(sequence)


if __name__ == "__main__":
    main()
```
