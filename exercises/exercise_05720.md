## Exercise
Create a program that displays the current date and allows the user to navigate to different dates.

## Reference Solution
```python
import datetime

def get_current_date():
    """Returns the current date."""
    return datetime.date.today()

def get_date_from_user():
    """Asks the user for a date and returns it."""
    while True:
        try:
            year = int(input("Enter the year: "))
            month = int(input("Enter the month (1-12): "))
            day = int(input("Enter the day (1-31): "))
            return datetime.date(year, month, day)
        except ValueError:
            print("Invalid date. Please try again.")

def display_date(date):
    """Displays the given date."""
    print(f"Current date: {date}")

def navigate_to_date(date):
    """Asks the user how to navigate from the given date."""
    while True:
        print("\nNavigation options:")
        print("1. Go to previous day")
        print("2. Go to next day")
        print("3. Go to previous month")
        print("4. Go to next month")
        print("5. Go to previous year")
        print("6. Go to next year")
        print("7. Go to a specific date")
        print("8. Quit")
        
        choice = input("Choose an option: ")
        
        if choice == "1":
            date = date - datetime.timedelta(days=1)
        elif choice == "2":
            date = date + datetime.timedelta(days=1)
        elif choice == "3":
            if date.month == 1:
                date = date.replace(year=date.year-1, month=12)
            else:
                date = date.replace(month=date.month-1)
        elif choice == "4":
            if date.month == 12:
                date = date.replace(year=date.year+1, month=1)
            else:
                date = date.replace(month=date.month+1)
        elif choice == "5":
            date = date.replace(year=date.year-1)
        elif choice == "6":
            date = date.replace(year=date.year+1)
        elif choice == "7":
            date = get_date_from_user()
        elif choice == "8":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")
        
        display_date(date)

def main():
    """The main function."""
    current_date = get_current_date()
    display_date(current_date)
    navigate_to_date(current_date)

if __name__ == "__main__":
    main()
```
