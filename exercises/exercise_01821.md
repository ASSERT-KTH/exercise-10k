```python
def calculate_sum(array):
    """
    This function calculates the sum of all elements in a given array.
    
    Parameters:
    array (list): A list of numbers.
    
    Returns:
    int: The sum of all elements in the array.
    
    Raises:
    ValueError: If the array contains non-numeric values.
    """
    try:
        return sum(array)
    except TypeError:
        raise ValueError("Array contains non-numeric values")

# Example usage:
array = [1, 2, 3, 4, 5]
print("Array:", array)
print("Sum of array elements:", calculate_sum(array))

# Example usage with non-numeric values:
array = [1, 2, '3', 4, 5]
try:
    print("Array:", array)
    print("Sum of array elements:", calculate_sum(array))
except ValueError as e:
    print("Error:", e)
```
