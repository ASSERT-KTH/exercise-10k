## Exercise
Write a program to declare and use boolean variables.

## Reference Solution
```python
# Define a function to demonstrate boolean variables
def demonstrate_boolean_variables():
    # Declare boolean variables
    is_admin = True
    is_user_logged_in = False

    # Print the values of the boolean variables
    print("Is Admin:", is_admin)
    print("Is User Logged In:", is_user_logged_in)

    # Use boolean variables in conditional statements
    if is_admin:
        print("You are an admin.")
    else:
        print("You are not an admin.")

    if is_user_logged_in:
        print("You are logged in.")
    else:
        print("You are not logged in.")

    # Use boolean variables in logical operations
    is_user_creator = True
    is_user_editor = False

    print("Is User Creator and Editor:", is_user_creator and is_user_editor)
    print("Is User Creator or Editor:", is_user_creator or is_user_editor)

# Call the function to demonstrate boolean variables
demonstrate_boolean_variables()
```
