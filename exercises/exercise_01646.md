def are_almost_equal(a, b, epsilon=1e-9):
    """
    Check if two floating-point numbers are almost equal.

    Args:
    a (float): The first number.
    b (float): The second number.
    epsilon (float, optional): The tolerance value. Defaults to 1e-9.

    Returns:
    bool: True if the numbers are almost equal, False otherwise.
    """
    return abs(a - b) < epsilon
