## Exercise
Ask the user for a list of numbers and print the mean, median, and standard deviation.

## Reference Solution
```python
import math
import statistics

def get_numbers_from_user():
    """Get a list of numbers from the user."""
    while True:
        try:
            numbers = [float(x) for x in input("Enter a list of numbers separated by spaces: ").split()]
            if len(numbers) == 0:
                print("Please enter at least one number.")
                continue
            return numbers
        except ValueError:
            print("Invalid input. Please enter numbers only.")

def calculate_statistics(numbers):
    """Calculate mean, median, and standard deviation of a list of numbers."""
    mean = statistics.mean(numbers)
    median = statistics.median(numbers)
    std_dev = statistics.stdev(numbers) if len(numbers) > 1 else 0
    return mean, median, std_dev

def main():
    numbers = get_numbers_from_user()
    mean, median, std_dev = calculate_statistics(numbers)
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Standard Deviation: {std_dev}")

if __name__ == "__main__":
    main()
```
