## Exercise
Create a function that takes a dictionary of student names and their corresponding grades, then calculate and return the average grade of all students. Also, return the names of students who scored above the average grade.

## Reference Solution
```python
def calculate_average_grade(grades: dict) -> tuple:
    """
    Calculate the average grade of all students and identify the students who scored above the average grade.

    Args:
        grades (dict): A dictionary where keys are student names and values are their corresponding grades.

    Returns:
        tuple: A tuple containing the average grade and a list of students who scored above the average grade.
    """

    # Check if the input dictionary is empty
    if not grades:
        return None, []

    # Calculate the sum of all grades
    total_grade = sum(grades.values())

    # Calculate the average grade
    average_grade = total_grade / len(grades)

    # Identify the students who scored above the average grade
    above_average_students = [name for name, grade in grades.items() if grade > average_grade]

    return average_grade, above_average_students


# Example usage:
def main():
    student_grades = {
        "Alice": 90,
        "Bob": 80,
        "Charlie": 95,
        "David": 85,
        "Eve": 92
    }

    average, above_average = calculate_average_grade(student_grades)

    print(f"Average Grade: {average:.2f}")
    print("Students who scored above the average grade:")
    for student in above_average:
        print(student)


if __name__ == "__main__":
    main()
```
