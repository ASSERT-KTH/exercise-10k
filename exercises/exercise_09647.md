## Exercise
Write a function that uses the threading module to run tasks concurrently.

## Reference Solution
```python
import threading
import time
from random import randint

def execute_task(task_name):
    """
    Simulates a task by sleeping for a random duration between 1 and 5 seconds.
    
    Args:
        task_name (str): The name of the task being executed.
    """
    sleep_duration = randint(1, 5)
    print(f"Task '{task_name}' started. Sleeping for {sleep_duration} seconds.")
    time.sleep(sleep_duration)
    print(f"Task '{task_name}' completed.")

def run_tasks_concurrently(task_names):
    """
    Runs a list of tasks concurrently using the threading module.
    
    Args:
        task_names (list): A list of task names to be executed concurrently.
    """
    threads = []
    
    # Create and start a thread for each task
    for task_name in task_names:
        thread = threading.Thread(target=execute_task, args=(task_name,))
        threads.append(thread)
        thread.start()
    
    # Wait for all threads to complete
    for thread in threads:
        thread.join()

# Example usage
if __name__ == "__main__":
    task_names = ["Task A", "Task B", "Task C", "Task D"]
    start_time = time.time()
    run_tasks_concurrently(task_names)
    end_time = time.time()
    print(f"Total execution time: {end_time - start_time} seconds")
```
