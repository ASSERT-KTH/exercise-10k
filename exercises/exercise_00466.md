## Exercise
Write a program that takes a list of strings as input and returns the longest common prefix

## Reference Solution
```python
def longest_common_prefix(strs):
    """
    Returns the longest common prefix of a list of strings.

    Args:
        strs (list): A list of strings.

    Returns:
        str: The longest common prefix.
    """
    if not strs:
        return ""

    shortest_str = min(strs, key=len)

    for i, char in enumerate(shortest_str):
        for other in strs:
            if other[i] != char:
                return shortest_str[:i]

    return shortest_str

# Example usage
def main():
    strs = ["flower", "flow", "flight"]
    print(longest_common_prefix(strs))  # Output: "fl"

    strs = ["dog", "racecar", "car"]
    print(longest_common_prefix(strs))  # Output: ""

    strs = ["a", "a", "a"]
    print(longest_common_prefix(strs))  # Output: "a"

if __name__ == "__main__":
    main()
```
