## Exercise
Write a program that finds the longest substring without repeating characters in a given string.

## Reference Solution

def longest_substring_without_repeating_chars(s: str) -> str:
    """
    Finds the longest substring without repeating characters in a given string.

    Args:
    s (str): The input string.

    Returns:
    str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    left = 0  # Left pointer of the sliding window
    max_length = 0  # Maximum length of substring without repeating characters
    max_substring = ""  # Longest substring without repeating characters
    char_set = set()  # Set to store unique characters in the current window

    for right in range(len(s)):
        # Move the right pointer to the right
        while s[right] in char_set:
            # If the character is in the set, remove the leftmost character
            char_set.remove(s[left])
            left += 1

        char_set.add(s[right])  # Add the current character to the set

        # Update the maximum length and substring if necessary
        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring


def main():
    # Example usage
    input_string = "abcabcbb"
    longest_substring = longest_substring_without_repeating_chars(input_string)
    print(f"Longest substring without repeating characters: {longest_substring}")


if __name__ == "__main__":
    main()
