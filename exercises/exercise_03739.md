## Exercise
Write a program that reads a string from the user and prints it out with quotes added around each line.

## Reference Solution
```python
def add_quotes_around_lines(input_string):
    """
    Adds quotes around each line of the input string.
    
    Args:
    input_string (str): The input string from the user.
    
    Returns:
    str: The input string with quotes added around each line.
    """
    
    # Split the input string into lines
    lines = input_string.split('\n')
    
    # Initialize an empty list to store the quoted lines
    quoted_lines = []
    
    # Iterate over each line
    for line in lines:
        # Add quotes around the line and append it to the quoted_lines list
        quoted_lines.append(f'"{line}"')
    
    # Join the quoted lines back into a string with newline characters in between
    quoted_string = '\n'.join(quoted_lines)
    
    return quoted_string

def main():
    """
    Reads a string from the user, adds quotes around each line, and prints the result.
    """
    
    # Read a string from the user
    input_string = input("Please enter a string: ")
    
    # Add quotes around each line of the input string
    quoted_string = add_quotes_around_lines(input_string)
    
    # Print the result
    print(quoted_string)

if __name__ == "__main__":
    main()
```
