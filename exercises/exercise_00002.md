## Exercise
Write a program to convert a string to an integer and vice versa.

## Reference Solution
```python
def string_to_int(s):
    """
    Converts a string to an integer.

    Args:
        s (str): The string to convert.

    Returns:
        int: The integer representation of the string.

    Raises:
        ValueError: If the string is not a valid integer.
    """
    try:
        return int(s)
    except ValueError:
        print("Invalid input. Please enter a valid integer string.")
        return None

def int_to_string(n):
    """
    Converts an integer to a string.

    Args:
        n (int): The integer to convert.

    Returns:
        str: The string representation of the integer.
    """
    return str(n)

def main():
    while True:
        print("1. Convert string to integer")
        print("2. Convert integer to string")
        print("3. Quit")
        choice = input("Enter your choice: ")

        if choice == "1":
            s = input("Enter a string: ")
            result = string_to_int(s)
            if result is not None:
                print(f"The integer representation of '{s}' is {result}")
        elif choice == "2":
            try:
                n = int(input("Enter an integer: "))
                print(f"The string representation of {n} is {int_to_string(n)}")
            except ValueError:
                print("Invalid input. Please enter a valid integer.")
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
