## Exercise
Write a function to check if a given binary tree is symmetric.

## Reference Solution
```python
class Node:
    """Represents a node in the binary tree."""
    def __init__(self, value, left=None, right=None):
        self.value = value
        self.left = left
        self.right = right

def is_symmetric(root):
    """
    Checks if a given binary tree is symmetric.

    Args:
    root (Node): The root node of the binary tree.

    Returns:
    bool: True if the binary tree is symmetric, False otherwise.
    """
    if root is None:
        return True

    def is_mirror(node1, node2):
        """
        Checks if two trees are mirror images of each other.

        Args:
        node1 (Node): The root node of the first tree.
        node2 (Node): The root node of the second tree.

        Returns:
        bool: True if the two trees are mirror images, False otherwise.
        """
        if node1 is None and node2 is None:
            return True
        if node1 is None or node2 is None:
            return False
        return (node1.value == node2.value and
                is_mirror(node1.left, node2.right) and
                is_mirror(node1.right, node2.left))

    return is_mirror(root.left, root.right)

# Example usage
if __name__ == "__main__":
    # Create a symmetric binary tree
    #       1
    #      / \
    #     2   2
    #    / \ / \
    #   3  4 4  3
    root1 = Node(1)
    root1.left = Node(2)
    root1.right = Node(2)
    root1.left.left = Node(3)
    root1.left.right = Node(4)
    root1.right.left = Node(4)
    root1.right.right = Node(3)

    print(is_symmetric(root1))  # Output: True

    # Create an asymmetric binary tree
    #       1
    #      / \
    #     2   2
    #    / \
    #   3  4
    root2 = Node(1)
    root2.left = Node(2)
    root2.right = Node(2)
    root2.left.left = Node(3)
    root2.left.right = Node(4)

    print(is_symmetric(root2))  # Output: False
```
