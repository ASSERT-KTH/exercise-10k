## Exercise
Write a program that takes a list of tasks as input and lets the user add, remove, and mark tasks as completed.

## Reference Solution
```python
class TaskManager:
    def __init__(self):
        self.tasks = []

    def add_task(self):
        """Add a task to the task list."""
        task_name = input("Enter the task name: ")
        self.tasks.append({"name": task_name, "completed": False})
        print(f"Task '{task_name}' added successfully.")

    def remove_task(self):
        """Remove a task from the task list."""
        if not self.tasks:
            print("No tasks available to remove.")
            return

        print("Available tasks:")
        for index, task in enumerate(self.tasks, start=1):
            print(f"{index}. {task['name']}")

        task_index = input("Enter the task number to remove: ")
        try:
            task_index = int(task_index) - 1
            if 0 <= task_index < len(self.tasks):
                task_name = self.tasks.pop(task_index)["name"]
                print(f"Task '{task_name}' removed successfully.")
            else:
                print("Invalid task number.")
        except ValueError:
            print("Invalid input. Please enter a number.")

    def mark_task_completed(self):
        """Mark a task as completed."""
        if not self.tasks:
            print("No tasks available to mark as completed.")
            return

        print("Available tasks:")
        for index, task in enumerate(self.tasks, start=1):
            status = "Completed" if task["completed"] else "Not completed"
            print(f"{index}. {task['name']} - {status}")

        task_index = input("Enter the task number to mark as completed: ")
        try:
            task_index = int(task_index) - 1
            if 0 <= task_index < len(self.tasks):
                self.tasks[task_index]["completed"] = True
                task_name = self.tasks[task_index]["name"]
                print(f"Task '{task_name}' marked as completed.")
            else:
                print("Invalid task number.")
        except ValueError:
            print("Invalid input. Please enter a number.")

    def display_tasks(self):
        """Display all tasks."""
        if not self.tasks:
            print("No tasks available.")
            return

        print("Available tasks:")
        for index, task in enumerate(self.tasks, start=1):
            status = "Completed" if task["completed"] else "Not completed"
            print(f"{index}. {task['name']} - {status}")


def main():
    task_manager = TaskManager()

    while True:
        print("\nTask Manager Menu:")
        print("1. Add task")
        print("2. Remove task")
        print("3. Mark task as completed")
        print("4. Display tasks")
        print("5. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            task_manager.add_task()
        elif choice == "2":
            task_manager.remove_task()
        elif choice == "3":
            task_manager.mark_task_completed()
        elif choice == "4":
            task_manager.display_tasks()
        elif choice == "5":
            print("Exiting the task manager. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
