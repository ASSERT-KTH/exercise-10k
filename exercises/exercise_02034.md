## Exercise
Develop a program that determines a person's credit score based on their credit history and income. The program should use if/else statements to determine the credit score.

## Reference Solution
```python
def calculate_credit_score(credit_history, income):
    """
    Calculates a person's credit score based on their credit history and income.

    Args:
        credit_history (str): 'good', 'fair', or 'poor'
        income (int): Annual income

    Returns:
        int: Credit score (300-850)
    """
    if credit_history == 'good':
        if income >= 50000:
            credit_score = 750
        elif income >= 30000:
            credit_score = 700
        else:
            credit_score = 650
    elif credit_history == 'fair':
        if income >= 50000:
            credit_score = 650
        elif income >= 30000:
            credit_score = 600
        else:
            credit_score = 550
    else:
        if income >= 50000:
            credit_score = 550
        elif income >= 30000:
            credit_score = 500
        else:
            credit_score = 450

    return credit_score

def main():
    credit_history = input("Enter your credit history (good, fair, or poor): ")
    income = int(input("Enter your annual income: "))

    credit_score = calculate_credit_score(credit_history, income)
    print(f"Your credit score is: {credit_score}")

if __name__ == "__main__":
    main()
```
