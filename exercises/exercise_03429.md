## Exercise
Write a function that takes two strings as input and returns the index of the first occurrence of the second string in the first string

## Reference Solution
```python
def find_string_index(main_string: str, sub_string: str) -> int:
    """
    This function finds the index of the first occurrence of the sub_string in the main_string.

    Args:
        main_string (str): The string to search in.
        sub_string (str): The string to search for.

    Returns:
        int: The index of the first occurrence of the sub_string. Returns -1 if not found.
    """
    try:
        # Use the index method to find the index of the sub_string
        index = main_string.index(sub_string)
        return index
    except ValueError:
        # If the sub_string is not found, return -1
        return -1

# Example usage:
if __name__ == "__main__":
    main_str = "Hello, World!"
    sub_str = "World"
    index = find_string_index(main_str, sub_str)
    print(f"The index of '{sub_str}' in '{main_str}' is: {index}")
```
