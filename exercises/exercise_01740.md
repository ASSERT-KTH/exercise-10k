## Exercise
Write a program that takes two strings as input and returns the minimum window substring that contains all the characters of the second string.

## Reference Solution

from collections import Counter

def min_window_substring(s1, s2):
    if not s1 or not s2:
        return None

    s2_counter = Counter(s2)
    required_chars = len(s2_counter)
    formed_chars = 0

    window_counts = {}
    ans = float("inf"), None, None
    window_start = 0

    for window_end in range(len(s1)):
        char = s1[window_end]
        window_counts[char] = window_counts.get(char, 0) + 1

        if char in s2_counter and window_counts[char] == s2_counter[char]:
            formed_chars += 1

        while window_start <= window_end and formed_chars == required_chars:
            char = s1[window_start]

            if window_end - window_start + 1 < ans[0]:
                ans = (window_end - window_start + 1, window_start, window_end)

            window_counts[char] -= 1
            if char in s2_counter and window_counts[char] < s2_counter[char]:
                formed_chars -= 1

            window_start += 1

    return ans[1:] if ans[0] != float("inf") else None
