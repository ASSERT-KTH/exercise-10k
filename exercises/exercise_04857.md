## Exercise
Write a program that takes an array and an element as input, and returns the index of the element if it exists in the array using a linear search algorithm.

## Reference Solution
```python
#include <iostream>
using namespace std;

int linear_search(int arr[], int n, int x) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == x) {
            return i;
        }
    }
    return -1; // Return -1 if the element is not found in the array
}

int main() {
    int arr[] = {2, 3, 4, 10, 40};
    int x = 10;
    int n = sizeof(arr) / sizeof(arr[0]);

    int result = linear_search(arr, n, x);

    if (result == -1) {
        cout << "Element is not present in array" << endl;
    } else {
        cout << "Element is present at index " << result << endl;
    }

    return 0;
}
```
