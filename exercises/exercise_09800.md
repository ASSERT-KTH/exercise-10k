## Exercise
Write a program that takes an input and validates it using recursion.

## Reference Solution
```python
def is_palindrome(s: str) -> bool:
    """
    Recursively checks if the input string is a palindrome.

    Args:
    s (str): The input string.

    Returns:
    bool: True if the string is a palindrome, False otherwise.
    """
    # Base case: if the string is empty or has only one character, it's a palindrome
    if len(s) <= 1:
        return True

    # If the first and last characters don't match, it's not a palindrome
    if s[0].lower() != s[-1].lower():
        return False

    # Recursive case: check the substring without the first and last characters
    return is_palindrome(s[1:-1])


def main():
    # Get the input string from the user
    input_str = input("Enter a string: ")

    # Validate the input string using recursion
    if is_palindrome(input_str):
        print(f"'{input_str}' is a palindrome.")
    else:
        print(f"'{input_str}' is not a palindrome.")


if __name__ == "__main__":
    main()
```
