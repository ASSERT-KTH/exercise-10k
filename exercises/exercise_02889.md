## Exercise
Design a program to calculate the result of nested loops that perform arithmetic operations.

## Reference Solution
```python
def nested_loop_arithmetic(num_loops, num_iterations, operation, initial_value):
    """
    Calculate the result of nested loops that perform arithmetic operations.

    Args:
    - num_loops (int): The number of loops.
    - num_iterations (int): The number of iterations per loop.
    - operation (str): The arithmetic operation to perform. Can be '+', '-', '*', or '/'.
    - initial_value (int): The initial value for the calculation.

    Returns:
    - result (int): The result of the nested loops.
    """
    result = initial_value
    for _ in range(num_loops):
        for _ in range(num_iterations):
            if operation == '+':
                result += 1
            elif operation == '-':
                result -= 1
            elif operation == '*':
                result *= 2
            elif operation == '/':
                if result != 0:
                    result /= 2
                else:
                    raise ZeroDivisionError("Cannot divide by zero")
            else:
                raise ValueError("Invalid operation. Must be '+', '-', '*', or '/'")
    return result

def main():
    num_loops = int(input("Enter the number of loops: "))
    num_iterations = int(input("Enter the number of iterations per loop: "))
    operation = input("Enter the arithmetic operation (+, -, *, /): ")
    initial_value = int(input("Enter the initial value: "))

    try:
        result = nested_loop_arithmetic(num_loops, num_iterations, operation, initial_value)
        print(f"The result is: {result}")
    except (ZeroDivisionError, ValueError) as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
