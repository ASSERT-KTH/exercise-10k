## Exercise
Write a program that converts a numerical grade to a letter grade (e.g., A, B, C, etc.) based on a simple grading system

## Reference Solution
```python
def convert_grade(num_grade):
    """
    Convert a numerical grade to a letter grade.
    
    Parameters:
    num_grade (float): The numerical grade to be converted.
    
    Returns:
    str: The corresponding letter grade.
    """
    if num_grade >= 90:
        return "A"
    elif num_grade >= 80:
        return "B"
    elif num_grade >= 70:
        return "C"
    elif num_grade >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the numerical grade from the user
    while True:
        try:
            num_grade = float(input("Enter your numerical grade (0-100): "))
            if 0 <= num_grade <= 100:
                break
            else:
                print("Please enter a grade between 0 and 100.")
        except ValueError:
            print("Invalid input. Please enter a numerical value.")

    # Convert the numerical grade to a letter grade
    letter_grade = convert_grade(num_grade)

    # Print the result
    print(f"Your letter grade is: {letter_grade}")

if __name__ == "__main__":
    main()
```
