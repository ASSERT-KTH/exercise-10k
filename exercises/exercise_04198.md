## Exercise
Write a function that analyzes a piece of text and provides statistics, such as word count and sentence count.

## Reference Solution
```python
import re

def analyze_text(text):
    """
    Analyzes a piece of text and provides statistics.

    Args:
    text (str): The text to be analyzed.

    Returns:
    dict: A dictionary containing the text statistics.
    """

    # Remove leading and trailing whitespace
    text = text.strip()

    # Calculate word count
    word_count = len(re.findall(r'\b\w+\b', text))

    # Calculate sentence count
    sentence_count = len(re.split(r'[.!?]', text)) - 1 if text else 0

    # Calculate average sentence length
    avg_sentence_length = word_count / sentence_count if sentence_count > 0 else 0

    # Create a dictionary to store the statistics
    stats = {
        'word_count': word_count,
        'sentence_count': sentence_count,
        'avg_sentence_length': avg_sentence_length
    }

    return stats

def main():
    # Example usage
    text = "This is an example sentence. This is another example sentence."
    stats = analyze_text(text)

    print("Text Statistics:")
    print(f"Word Count: {stats['word_count']}")
    print(f"Sentence Count: {stats['sentence_count']}")
    print(f"Average Sentence Length: {stats['avg_sentence_length']}")

if __name__ == "__main__":
    main()
```
