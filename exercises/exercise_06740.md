## Exercise
Write a program that processes input data in real-time

## Reference Solution
```python
import threading
import queue
import time

class RealTimeDataProcessor:
    def __init__(self):
        self.data_queue = queue.Queue()
        self.processing_thread = threading.Thread(target=self.process_data)
        self.processing_thread.daemon = True  # Allow the program to exit even if the thread is still running

    def start_processing(self):
        self.processing_thread.start()

    def add_data(self, data):
        self.data_queue.put(data)

    def process_data(self):
        while True:
            if not self.data_queue.empty():
                data = self.data_queue.get()
                # Process the data here
                print(f"Processing data: {data}")
                time.sleep(1)  # Simulate processing time
                print(f"Data processed: {data}")
            else:
                time.sleep(0.1)  # Check the queue every 0.1 seconds

def main():
    processor = RealTimeDataProcessor()
    processor.start_processing()

    while True:
        data = input("Enter data to process (or 'quit' to exit): ")
        if data.lower() == 'quit':
            break
        processor.add_data(data)

if __name__ == "__main__":
    main()
```
