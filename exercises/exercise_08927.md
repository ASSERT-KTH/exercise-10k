## Exercise
Write a program to find the lowest common ancestor of two nodes in a given binary tree.

## Reference Solution
```python
class Node:
    """Represents a node in the binary tree."""
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None

def find_path(root, target):
    """
    Finds the path from the root to the target node.

    Args:
        root (Node): The root of the binary tree.
        target (int): The value of the target node.

    Returns:
        list: A list of node values representing the path from the root to the target node.
    """
    if root is None:
        return None

    if root.value == target:
        return [root.value]

    left_path = find_path(root.left, target)
    if left_path is not None:
        return [root.value] + left_path

    right_path = find_path(root.right, target)
    if right_path is not None:
        return [root.value] + right_path

    return None

def find_lca(root, node1, node2):
    """
    Finds the lowest common ancestor of two nodes in the binary tree.

    Args:
        root (Node): The root of the binary tree.
        node1 (int): The value of the first node.
        node2 (int): The value of the second node.

    Returns:
        int: The value of the lowest common ancestor node.
    """
    path1 = find_path(root, node1)
    path2 = find_path(root, node2)

    if path1 is None or path2 is None:
        return None

    i = 0
    while i < len(path1) and i < len(path2) and path1[i] == path2[i]:
        i += 1

    return path1[i - 1]

# Example usage:
root = Node(1)
root.left = Node(2)
root.right = Node(3)
root.left.left = Node(4)
root.left.right = Node(5)
root.right.left = Node(6)
root.right.right = Node(7)

node1 = 4
node2 = 5
lca = find_lca(root, node1, node2)
print("Lowest common ancestor of {} and {} is {}".format(node1, node2, lca))
```
