## Exercise
Create a program that takes a series of events as input and generates a story based on the events.

## Reference Solution
```python
class Story:
    def __init__(self):
        self.events = []

    def add_event(self, event):
        self.events.append(event)

    def generate_story(self):
        story = ""
        for i, event in enumerate(self.events):
            if i == 0:
                story += f"Once upon a time, {event}. "
            elif i == len(self.events) - 1:
                story += f"Finally, {event}. "
            else:
                story += f"Then, {event}. "

        return story


def main():
    print("Welcome to the story generator!")
    print("Enter the series of events you'd like to use for your story, one event per line.")
    print("When you're finished entering events, type 'done'.")

    story = Story()
    while True:
        event = input("Enter an event: ")
        if event.lower() == 'done':
            break
        else:
            story.add_event(event)

    print("\nHere's your story:")
    print(story.generate_story())


if __name__ == "__main__":
    main()
```
