## Exercise
Write a program that takes a list of integers as input, and returns a new list containing only the unique integers from the original list.

// https://www.hackerrank.com/challenges/30-review-debugging

#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    int n;
    std::cin >> n;

    std::vector<int> ar(n);
    for (int i = 0; i < n; i++) {
        std::cin >> ar[i];
    }

    std::vector<int> uniqueAr(n);
    int count = 0;

    for (int i = 0; i < n; i++) {
        bool found = false;
        for (int j = 0; j < count; j++) {
            if (ar[i] == uniqueAr[j]) {
                found = true;
                break;
            }
        }
        if (!found) {
            uniqueAr[count] = ar[i];
            count++;
        }
    }

    std::sort(uniqueAr.begin(), uniqueAr.begin() + count);

    for (int i = 0; i < count; i++) {
        std::cout << uniqueAr[i] << " ";
    }

    return 0;
}
