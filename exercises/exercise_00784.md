## Exercise
Create a program to calculate the tip and total cost of a meal given the cost of the meal and the tip percentage.

## Reference Solution
```python
def calculate_tip(meal_cost, tip_percentage):
    """
    Calculate the tip amount based on the meal cost and tip percentage.

    Args:
        meal_cost (float): The cost of the meal.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The tip amount.
    """
    tip_amount = (tip_percentage / 100) * meal_cost
    return tip_amount

def calculate_total_cost(meal_cost, tip_amount):
    """
    Calculate the total cost of the meal including the tip.

    Args:
        meal_cost (float): The cost of the meal.
        tip_amount (float): The tip amount.

    Returns:
        float: The total cost of the meal.
    """
    total_cost = meal_cost + tip_amount
    return total_cost

def main():
    print("Tip and Total Cost Calculator")
    print("--------------------------------")

    # Get the meal cost from the user
    while True:
        try:
            meal_cost = float(input("Enter the cost of the meal: $"))
            if meal_cost < 0:
                print("Meal cost cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip percentage from the user
    while True:
        try:
            tip_percentage = float(input("Enter the tip percentage (%): "))
            if tip_percentage < 0:
                print("Tip percentage cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the tip amount
    tip_amount = calculate_tip(meal_cost, tip_percentage)

    # Calculate the total cost
    total_cost = calculate_total_cost(meal_cost, tip_amount)

    # Print the results
    print(f"Meal cost: ${meal_cost:.2f}")
    print(f"Tip percentage: {tip_percentage}%")
    print(f"Tip amount: ${tip_amount:.2f}")
    print(f"Total cost: ${total_cost:.2f}")

if __name__ == "__main__":
    main()
```
