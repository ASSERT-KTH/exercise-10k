## Exercise
Write a program that takes the income of a person as input and outputs their tax amount based on a given tax rate.

## Reference Solution
```python
def calculate_tax(income, tax_rate):
    """
    Calculate the tax amount based on the income and tax rate.

    Args:
        income (float): The income of the person.
        tax_rate (float): The tax rate as a decimal.

    Returns:
        float: The tax amount.
    """
    tax_amount = income * tax_rate
    return tax_amount

def main():
    # Get the income from the user
    income = float(input("Enter your income: "))

    # Define the tax rates
    tax_rates = {
        "single": 0.24,
        "joint": 0.22,
        "head_of_household": 0.20
    }

    # Get the filing status from the user
    print("Choose your filing status:")
    print("1. Single")
    print("2. Joint")
    print("3. Head of Household")
    choice = int(input("Enter your choice (1/2/3): "))

    # Get the tax rate based on the filing status
    if choice == 1:
        tax_rate = tax_rates["single"]
    elif choice == 2:
        tax_rate = tax_rates["joint"]
    elif choice == 3:
        tax_rate = tax_rates["head_of_household"]
    else:
        print("Invalid choice. Exiting program.")
        return

    # Calculate and print the tax amount
    tax_amount = calculate_tax(income, tax_rate)
    print(f"Your tax amount is: ${tax_amount:.2f}")

if __name__ == "__main__":
    main()
```
