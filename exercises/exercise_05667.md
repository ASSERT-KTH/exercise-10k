## Exercise
Write a program that takes two numbers as input and outputs their exponentiation result.

## Reference Solution
```python
def exponentiation(base, exponent):
    """
    This function calculates the exponentiation result of two given numbers.
    
    Parameters:
    base (float): The base number.
    exponent (float): The exponent to which the base is raised.
    
    Returns:
    float: The result of the exponentiation.
    """
    try:
        # Check if the inputs are numbers
        if not isinstance(base, (int, float)) or not isinstance(exponent, (int, float)):
            raise TypeError("Both inputs must be numbers.")
        
        # Calculate the exponentiation result
        result = base ** exponent
        
        return result
    
    except TypeError as e:
        print(f"Error: {e}")
        return None

def main():
    # Get the base and exponent from the user
    base = float(input("Enter the base: "))
    exponent = float(input("Enter the exponent: "))
    
    # Calculate and print the exponentiation result
    result = exponentiation(base, exponent)
    
    if result is not None:
        print(f"{base} raised to the power of {exponent} is {result}")

if __name__ == "__main__":
    main()
```
