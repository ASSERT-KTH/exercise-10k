## Exercise
Write a function that takes a string as input and returns the compressed string using run-length encoding.

## Reference Solution
```python
def compress_string(input_str: str) -> str:
    """
    Compress a string using run-length encoding.

    Args:
        input_str (str): The input string to be compressed.

    Returns:
        str: The compressed string.
    """
    if not input_str:
        return ""

    compressed = []
    count = 1

    for i in range(1, len(input_str)):
        if input_str[i] == input_str[i - 1]:
            count += 1
        else:
            compressed.append(input_str[i - 1] + str(count))
            count = 1

    compressed.append(input_str[-1] + str(count))
    return "".join(compressed)


# Example usage
if __name__ == "__main__":
    input_str = "AAABBBCCCDDDD"
    compressed_str = compress_string(input_str)
    print(f"Input String: {input_str}")
    print(f"Compressed String: {compressed_str}")
```
