## Exercise
Write a program that encrypts and decrypts text using a Caesar cipher, taking the user's input as text and a shift value.

## Reference Solution
```python
# Author: [Your Name]
# Description: A program that uses a Caesar cipher to encrypt and decrypt text.

import string

def encrypt(text, shift):
    result = ""

    for char in text:
        if char in string.ascii_lowercase:
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        elif char in string.ascii_uppercase:
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        else:
            result += char

    return result

def decrypt(text, shift):
    result = ""

    for char in text:
        if char in string.ascii_lowercase:
            result += chr((ord(char) - 97 - shift) % 26 + 97)
        elif char in string.ascii_uppercase:
            result += chr((ord(char) - 65 - shift) % 26 + 65)
        else:
            result += char

    return result

def test():
    text = input("Enter a text: ")
    shift = int(input("Enter a shift value: "))

    encrypted_text = encrypt(text, shift)
    print("Encrypted text: ", encrypted_text)

    decrypted_text = decrypt(encrypted_text, shift)
    print("Decrypted text: ", decrypted_text)

    print("Original text and decrypted text are the same: ", text == decrypted_text)

test()
```
