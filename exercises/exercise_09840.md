## Exercise
Write a program that takes an array of integers as input and returns the maximum value in the array. The program should use a basic algorithm to iterate through the array and find the maximum value.

## Reference Solution
```python
#include <iostream>
using namespace std;

int maxim(int v[], int n) {
    int maximul = v[0];
    for (int i = 1; i < n; i++) {
        if (v[i] > maximul) {
            maximul = v[i];
        }
    }
    return maximul;
}

int main() {
    int v[] = {5, 3, 7, 1, 9, 2};
    int n = sizeof(v) / sizeof(v[0]);
    cout << "Maximul este: " << maxim(v, n) << endl;
    return 0;
}
```
