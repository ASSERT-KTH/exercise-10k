## Exercise
Write a function that takes a string as input and returns all possible permutations of the string.

## Reference Solution
```python
from itertools import permutations

def generate_permutations(input_string):
    """
    Generate all possible permutations of the input string.

    Args:
        input_string (str): The input string.

    Returns:
        list: A list of all possible permutations.
    """
    # Use itertools.permutations to generate all permutations
    perms = permutations(input_string)

    # Convert the permutations to a list of strings
    perms_list = [''.join(p) for p in perms]

    return perms_list

# Example usage
def main():
    input_str = "abc"
    perms = generate_permutations(input_str)
    print("Permutations of '{}'".format(input_str))
    for perm in perms:
        print(perm)

if __name__ == "__main__":
    main()
```
