## Exercise
Create a program that checks if a password meets certain criteria (e.g. length, contains uppercase and lowercase letters, contains digits). Use a while loop to keep asking for input until a valid password is entered.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Checks if a password meets the required criteria.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the regular expression patterns for each criterion
    patterns = {
        "uppercase": r"[A-Z]",
        "lowercase": r"[a-z]",
        "digit": r"\d"
    }

    # Check if the password meets each criterion
    if len(password) < 8:
        return False
    if not re.search(patterns["uppercase"], password):
        return False
    if not re.search(patterns["lowercase"], password):
        return False
    if not re.search(patterns["digit"], password):
        return False

    return True

def main():
    while True:
        password = input("Enter a password: ")
        if validate_password(password):
            print("Password is valid.")
            break
        else:
            print("Password is invalid. Please try again.")
            print("Requirements:")
            print("* Minimum length of 8 characters")
            print("* Contains at least one uppercase letter")
            print("* Contains at least one lowercase letter")
            print("* Contains at least one digit")

if __name__ == "__main__":
    main()
```
