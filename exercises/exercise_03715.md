## Exercise
Create a function that generates a random matrix of a specified size.

## Reference Solution
```python
import numpy as np
import random

def generate_random_matrix(rows: int, cols: int, data_type: str = 'int', min_val: int = 0, max_val: int = 100) -> np.ndarray:
    """
    Generates a random matrix of a specified size.

    Args:
    - rows (int): The number of rows in the matrix.
    - cols (int): The number of columns in the matrix.
    - data_type (str, optional): The type of data in the matrix. Defaults to 'int'.
    - min_val (int, optional): The minimum value in the matrix. Defaults to 0.
    - max_val (int, optional): The maximum value in the matrix. Defaults to 100.

    Returns:
    - np.ndarray: A random matrix of the specified size.
    """

    if data_type == 'int':
        # Generate random integer matrix
        return np.random.randint(min_val, max_val + 1, size=(rows, cols))
    elif data_type == 'float':
        # Generate random float matrix
        return np.random.uniform(min_val, max_val, size=(rows, cols))
    else:
        raise ValueError("Invalid data type. Supported data types are 'int' and 'float'.")

# Example usage
if __name__ == "__main__":
    random_int_matrix = generate_random_matrix(3, 3, 'int', 1, 10)
    print("Random Integer Matrix:")
    print(random_int_matrix)

    random_float_matrix = generate_random_matrix(3, 3, 'float', 1.0, 10.0)
    print("\nRandom Float Matrix:")
    print(random_float_matrix)
```
