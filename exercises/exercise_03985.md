## Exercise
Write a program that implements a Caesar cipher. A Caesar cipher is a type of substitution cipher in which each character in the plaintext is 'shifted' a certain number of places down the alphabet.

## Reference Solution
```python
def caesar_cipher(text, shift, direction):
    """
    Encrypts or decrypts a given text using the Caesar cipher algorithm.

    Args:
    - text (str): The input text to be encrypted or decrypted.
    - shift (int): The number of positions to shift the characters.
    - direction (str): The direction of the shift, either 'encrypt' or 'decrypt'.

    Returns:
    - str: The encrypted or decrypted text.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in text:
        if char.isalpha():
            # Find the position of the character in the alphabet
            index = alphabet.index(char.lower())

            # Shift the character based on the direction
            if direction == 'encrypt':
                new_index = (index + shift) % 26
            elif direction == 'decrypt':
                new_index = (index - shift) % 26

            # Preserve the original case of the character
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            # Non-alphabetic characters are left unchanged
            result += char

    return result


def main():
    # Example usage
    text = "Hello, World!"
    shift = 3

    encrypted_text = caesar_cipher(text, shift, 'encrypt')
    decrypted_text = caesar_cipher(encrypted_text, shift, 'decrypt')

    print(f"Original Text: {text}")
    print(f"Encrypted Text: {encrypted_text}")
    print(f"Decrypted Text: {decrypted_text}")


if __name__ == "__main__":
    main()
```
