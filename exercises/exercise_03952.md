## Exercise
Write a program to calculate the total bill for a customer based on their purchases.

## Reference Solution
```python
# Define a class to represent a purchase item
class PurchaseItem:
    def __init__(self, name, price, quantity):
        self.name = name
        self.price = price
        self.quantity = quantity

# Define a function to calculate the total bill
def calculate_total_bill(items, tax_rate, discount):
    """
    Calculate the total bill based on the purchase items, tax rate, and discount.

    Args:
        items (list): List of PurchaseItem objects
        tax_rate (float): Tax rate as a decimal (e.g., 8% = 0.08)
        discount (float): Discount as a decimal (e.g., 10% = 0.10)

    Returns:
        float: Total bill amount
    """
    subtotal = sum(item.price * item.quantity for item in items)
    tax = subtotal * tax_rate
    discount_amount = subtotal * discount
    total = subtotal + tax - discount_amount
    return total

# Define a main function to demonstrate the program
def main():
    # Create a list of purchase items
    items = [
        PurchaseItem("Item 1", 10.99, 2),
        PurchaseItem("Item 2", 5.99, 3),
        PurchaseItem("Item 3", 7.99, 1)
    ]

    # Set the tax rate and discount
    tax_rate = 0.08  # 8% tax rate
    discount = 0.10  # 10% discount

    # Calculate the total bill
    total_bill = calculate_total_bill(items, tax_rate, discount)

    # Print the receipt
    print("Receipt:")
    for item in items:
        print(f"{item.name} x {item.quantity} = ${item.price * item.quantity:.2f}")
    print(f"Subtotal: ${sum(item.price * item.quantity for item in items):.2f}")
    print(f"Tax ({tax_rate*100}%): ${sum(item.price * item.quantity for item in items) * tax_rate:.2f}")
    print(f"Discount ({discount*100}%): -${sum(item.price * item.quantity for item in items) * discount:.2f}")
    print(f"Total: ${total_bill:.2f}")

# Run the main function
if __name__ == "__main__":
    main()
```
