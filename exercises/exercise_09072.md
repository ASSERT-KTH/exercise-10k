## Exercise
Create a program that converts units of measurement, such as length or weight, using if/else statements

## Reference Solution
```python
def convert_length():
    print("Length Conversion")
    print("1. Kilometers to Miles")
    print("2. Miles to Kilometers")
    print("3. Meters to Feet")
    print("4. Feet to Meters")
    
    choice = int(input("Enter your choice (1/2/3/4): "))
    
    if choice == 1:
        kilometers = float(input("Enter length in kilometers: "))
        miles = kilometers * 0.621371
        print(f"{kilometers} kilometers is equal to {miles} miles")
    elif choice == 2:
        miles = float(input("Enter length in miles: "))
        kilometers = miles * 1.60934
        print(f"{miles} miles is equal to {kilometers} kilometers")
    elif choice == 3:
        meters = float(input("Enter length in meters: "))
        feet = meters * 3.28084
        print(f"{meters} meters is equal to {feet} feet")
    elif choice == 4:
        feet = float(input("Enter length in feet: "))
        meters = feet * 0.3048
        print(f"{feet} feet is equal to {meters} meters")
    else:
        print("Invalid choice")

def convert_weight():
    print("Weight Conversion")
    print("1. Kilograms to Pounds")
    print("2. Pounds to Kilograms")
    print("3. Grams to Ounces")
    print("4. Ounces to Grams")
    
    choice = int(input("Enter your choice (1/2/3/4): "))
    
    if choice == 1:
        kilograms = float(input("Enter weight in kilograms: "))
        pounds = kilograms * 2.20462
        print(f"{kilograms} kilograms is equal to {pounds} pounds")
    elif choice == 2:
        pounds = float(input("Enter weight in pounds: "))
        kilograms = pounds * 0.453592
        print(f"{pounds} pounds is equal to {kilograms} kilograms")
    elif choice == 3:
        grams = float(input("Enter weight in grams: "))
        ounces = grams * 0.035274
        print(f"{grams} grams is equal to {ounces} ounces")
    elif choice == 4:
        ounces = float(input("Enter weight in ounces: "))
        grams = ounces * 28.3495
        print(f"{ounces} ounces is equal to {grams} grams")
    else:
        print("Invalid choice")

def main():
    print("Unit Conversion Program")
    print("1. Length Conversion")
    print("2. Weight Conversion")
    
    choice = int(input("Enter your choice (1/2): "))
    
    if choice == 1:
        convert_length()
    elif choice == 2:
        convert_weight()
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
