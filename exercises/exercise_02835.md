## Exercise
Develop a program to validate a credit card number. The credit card number should be 16 digits long and should pass the Luhn test.

## Reference Solution
```python
def validate_credit_card_number(card_number):
    """
    Validates a credit card number by checking its length and performing the Luhn test.

    Args:
        card_number (str): The credit card number to validate.

    Returns:
        bool: True if the credit card number is valid, False otherwise.
    """
    # Check if the card number is 16 digits long
    if not card_number.isdigit() or len(card_number) != 16:
        return False

    # Perform the Luhn test
    sum_of_digits = 0
    for i, digit in enumerate(reversed(card_number)):
        digit = int(digit)
        if i % 2 == 1:
            digit *= 2
            if digit > 9:
                digit -= 9
        sum_of_digits += digit

    # Check if the sum of digits is divisible by 10
    return sum_of_digits % 10 == 0


def main():
    # Example usage
    card_number = input("Enter a credit card number: ")
    if validate_credit_card_number(card_number):
        print("The credit card number is valid.")
    else:
        print("The credit card number is invalid.")


if __name__ == "__main__":
    main()
```
