def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:
        dict: A new dictionary containing all key-value pairs from both input dictionaries.
    """
    # Use the built-in dictionary update method to merge the dictionaries
    merged_dict = dict1.copy()  # Create a copy of dict1 to avoid modifying the original
    merged_dict.update(dict2)
    return merged_dict

def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries into one.

    Args:
        dict1 (dict): The first dictionary to merge.
        dict2 (dict): The second dictionary to merge.

    Returns:

    """

# Example usage:
