## Exercise
Create a program that asks the user a series of questions and keeps track of the score. The user should be able to input their answers and the program should then print out the final score.

## Reference Solution
```python
# Question and Answer Program

# Dictionary containing questions, options, and answers
questions = {
    "What is the capital of France?": {
        "A": "Paris",
        "B": "London",
        "C": "Berlin",
        "D": "Rome",
        "answer": "A"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "answer": "C"
    },
    "Which programming language is used for this program?": {
        "A": "Java",
        "B": "Python",
        "C": "C++",
        "D": "JavaScript",
        "answer": "B"
    }
}

def ask_questions(questions):
    """Asks the user a series of questions and keeps track of the score."""
    score = 0
    for question, options in questions.items():
        print(question)
        for option, value in options.items():
            if option != "answer":
                print(f"{option}: {value}")
        answer = input("Enter the letter of your answer: ").upper()
        if answer == options["answer"]:
            print("Correct!\n")
            score += 1
        else:
            print(f"Incorrect. The correct answer was {options['answer']}: {options[options['answer']]}\n")
    return score

def main():
    """Main function to run the program."""
    print("Welcome to the Question and Answer Program!")
    score = ask_questions(questions)
    print(f"Your final score is {score} out of {len(questions)}.")

if __name__ == "__main__":
    main()
```
