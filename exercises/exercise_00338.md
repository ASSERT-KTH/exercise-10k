## Exercise
Write a program to map a variable to a new name, creating an alias for the original variable.

## Reference Solution
```python
def create_alias(original_variable, new_name):
    """
    Creates an alias for the original variable.

    Args:
        original_variable (any): The original variable to be aliased.
        new_name (str): The new name for the alias.

    Returns:
        dict: A dictionary containing the alias.
    """
    alias = {new_name: original_variable}
    return alias

# Example usage:
if __name__ == "__main__":
    original_var = 10
    new_name = "my_alias"
    
    alias = create_alias(original_var, new_name)
    
    print(f"Original variable: {original_var}")
    print(f"Alias: {alias[new_name]}")

    # Update the original variable and verify the alias
    original_var = 20
    print(f"\nUpdated original variable: {original_var}")
    print(f"Alias after update: {alias[new_name]}")

    # To keep the alias updated with the original variable,
    # you would need to create a class or use a different approach.
    # Below is an example using a class:
    class VariableAlias:
        def __init__(self, original_variable, new_name):
            self.original_variable = original_variable
            self.new_name = new_name

        def get_alias(self):
            return {self.new_name: self.original_variable}

        def update_original(self, new_value):
            self.original_variable = new_value

    # Using the class:
    alias_class = VariableAlias(10, "my_alias_class")
    print(f"\nAlias class (initial): {alias_class.get_alias()[alias_class.new_name]}")
    alias_class.update_original(30)
    print(f"Alias class (after update): {alias_class.get_alias()[alias_class.new_name]}")
```
