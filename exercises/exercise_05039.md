## Exercise
Write a program that simulates a customer order system using for loops and while loops.

## Reference Solution
```python
# Define a dictionary to store menu items
menu = {
    "1": {"item": "Burger", "price": 10.99},
    "2": {"item": "Pizza", "price": 12.99},
    "3": {"item": "Salad", "price": 8.99},
    "4": {"item": "Pasta", "price": 11.99},
}

# Define a list to store customer orders
orders = []

# Function to display menu
def display_menu():
    print("\nMenu:")
    for key, value in menu.items():
        print(f"{key}. {value['item']} - ${value['price']}")

# Function to take customer order
def take_order():
    order = {}
    while True:
        display_menu()
        choice = input("\nEnter the number of the item you'd like to order (or 'q' to quit): ")
        if choice.lower() == 'q':
            break
        elif choice in menu:
            quantity = int(input("Enter the quantity: "))
            if choice not in order:
                order[choice] = quantity
            else:
                order[choice] += quantity
        else:
            print("Invalid choice. Please try again.")
    return order

# Function to calculate total cost
def calculate_total(order):
    total = 0
    for key, value in order.items():
        total += menu[key]["price"] * value
    return total

# Main program
while True:
    print("\nCustomer Order System")
    print("1. Place an order")
    print("2. View all orders")
    print("3. Quit")
    choice = input("Enter your choice: ")
    if choice == "1":
        order = take_order()
        orders.append(order)
        print("\nOrder summary:")
        for key, value in order.items():
            print(f"{menu[key]['item']} x {value} = ${menu[key]['price'] * value:.2f}")
        print(f"Total: ${calculate_total(order):.2f}")
    elif choice == "2":
        print("\nAll orders:")
        for i, order in enumerate(orders, start=1):
            print(f"Order {i}:")
            for key, value in order.items():
                print(f"{menu[key]['item']} x {value} = ${menu[key]['price'] * value:.2f}")
            print(f"Total: ${calculate_total(order):.2f}\n")
    elif choice == "3":
        print("Goodbye!")
        break
    else:
        print("Invalid choice. Please try again.")
```
