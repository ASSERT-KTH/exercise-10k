## Exercise
Write a program that calculates the price of a book based on its original price and a discount percentage.

## Reference Solution
```python
def calculate_book_price(original_price, discount_percentage):
    """
    Calculate the price of a book after applying a discount.

    Args:
        original_price (float): The original price of the book.
        discount_percentage (float): The discount percentage to be applied.

    Returns:
        float: The price of the book after applying the discount.
    """
    if original_price < 0 or discount_percentage < 0:
        raise ValueError("Original price and discount percentage cannot be negative")

    discount_amount = (original_price / 100) * discount_percentage
    price_after_discount = original_price - discount_amount

    return price_after_discount


def main():
    original_price = float(input("Enter the original price of the book: "))
    discount_percentage = float(input("Enter the discount percentage: "))

    try:
        price_after_discount = calculate_book_price(original_price, discount_percentage)
        print(f"The price of the book after applying a {discount_percentage}% discount is: ${price_after_discount:.2f}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
