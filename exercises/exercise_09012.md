## Exercise
Write a function that demonstrates function overloading using default arguments.

## Reference Solution
```python
def calculate_area(length, width=None, radius=None):
    """
    Calculate the area of different shapes.

    Args:
        length (float): The length of the shape.
        width (float, optional): The width of the shape. Defaults to None.
        radius (float, optional): The radius of the shape. Defaults to None.

    Returns:
        float: The calculated area.
    """
    if width is not None:
        # Calculate area of a rectangle
        return length * width
    elif radius is not None:
        # Calculate area of a circle
        import math
        return math.pi * (radius ** 2)
    else:
        # Calculate area of a square
        return length ** 2

# Example usage:
print(calculate_area(5))  # Calculate area of a square
print(calculate_area(4, 5))  # Calculate area of a rectangle
print(calculate_area(5, radius=5))  # Calculate area of a circle
```

```python
def calculate_area(*args, **kwargs):
    """
    Calculate the area of different shapes.

    Args:
        *args: Variable number of arguments.
        **kwargs: Keyword arguments.

    Returns:
        float: The calculated area.
    """
    if len(args) == 1:
        # Calculate area of a square
        return args[0] ** 2
    elif len(args) == 2:
        # Calculate area of a rectangle
        return args[0] * args[1]
    elif 'radius' in kwargs:
        # Calculate area of a circle
        import math
        return math.pi * (kwargs['radius'] ** 2)
    else:
        raise ValueError("Invalid input")

# Example usage:
print(calculate_area(5))  # Calculate area of a square
print(calculate_area(4, 5))  # Calculate area of a rectangle
print(calculate_area(radius=5))  # Calculate area of a circle
```
