## Exercise
Write a function to validate a dictionary against a set of rules (e.g., required keys, value types).

## Reference Solution
```python
from typing import Dict, Any

def validate_dict(data: Dict[str, Any], rules: Dict[str, Any]) -> bool:
    """
    Validate a dictionary against a set of rules.

    Args:
    - data (Dict[str, Any]): The dictionary to be validated.
    - rules (Dict[str, Any]): A dictionary containing the validation rules.
        - 'required': A list of required keys.
        - 'types': A dictionary where keys are the data keys and values are the expected types.

    Returns:
    - bool: True if the dictionary is valid, False otherwise.
    """
    # Check if all required keys are present in the data
    if 'required' in rules:
        required_keys = rules['required']
        if not all(key in data for key in required_keys):
            return False

    # Check if all values have the expected type
    if 'types' in rules:
        types = rules['types']
        for key, expected_type in types.items():
            if key in data and not isinstance(data[key], expected_type):
                return False

    return True


# Example usage
if __name__ == "__main__":
    data = {
        'name': 'John Doe',
        'age': 30,
        'email': 'john@example.com'
    }

    rules = {
        'required': ['name', 'age', 'email'],
        'types': {
            'name': str,
            'age': int,
            'email': str
        }
    }

    is_valid = validate_dict(data, rules)
    print(f"Is data valid? {is_valid}")

    # Test with invalid data
    invalid_data = {
        'name': 'Jane Doe',
        'age': 'twenty-five'  # invalid type
    }

    is_valid = validate_dict(invalid_data, rules)
    print(f"Is invalid data valid? {is_valid}")
```
