## Exercise
Write a program to simulate a weather forecast system that provides the user with the current weather and a 5-day forecast.

## Reference Solution
```python
import random

# Define a dictionary to store the current weather and forecast
weather_data = {
    "current_weather": {
        "temperature": 25,
        "condition": "Sunny"
    },
    "forecast": [
        {"day": "Monday", "temperature": 22, "condition": "Partly Cloudy"},
        {"day": "Tuesday", "temperature": 28, "condition": "Sunny"},
        {"day": "Wednesday", "temperature": 20, "condition": "Rainy"},
        {"day": "Thursday", "temperature": 24, "condition": "Windy"},
        {"day": "Friday", "temperature": 26, "condition": "Cloudy"}
    ]
}

def get_current_weather():
    """Return the current weather"""
    return weather_data["current_weather"]

def get_forecast():
    """Return the 5-day forecast"""
    return weather_data["forecast"]

def update_weather():
    """Update the current weather and forecast"""
    weather_data["current_weather"]["temperature"] = random.randint(15, 30)
    weather_data["current_weather"]["condition"] = random.choice(["Sunny", "Partly Cloudy", "Rainy", "Windy", "Cloudy"])
    
    for day in weather_data["forecast"]:
        day["temperature"] = random.randint(15, 30)
        day["condition"] = random.choice(["Sunny", "Partly Cloudy", "Rainy", "Windy", "Cloudy"])

def print_current_weather():
    """Print the current weather"""
    current_weather = get_current_weather()
    print(f"Current Temperature: {current_weather['temperature']}°C")
    print(f"Current Condition: {current_weather['condition']}")

def print_forecast():
    """Print the 5-day forecast"""
    forecast = get_forecast()
    print("5-Day Forecast:")
    for day in forecast:
        print(f"{day['day']}: {day['temperature']}°C, {day['condition']}")

def main():
    while True:
        print("\nWeather Forecast System")
        print("1. Get Current Weather")
        print("2. Get 5-Day Forecast")
        print("3. Update Weather")
        print("4. Exit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            print_current_weather()
        elif choice == "2":
            print_forecast()
        elif choice == "3":
            update_weather()
            print("Weather updated successfully!")
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
