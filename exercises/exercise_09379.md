## Exercise
Write a program that demonstrates object iteration techniques.

## Reference Solution
```python
# Define a dictionary
person = {
    "name": "John Doe",
    "age": 30,
    "city": "New York"
}

# Define a list
fruits = ["apple", "banana", "cherry"]

# Define a set
colors = {"red", "green", "blue"}

def iterate_dictionary(obj):
    """
    Iterate over a dictionary using .items(), .keys(), and .values().
    """
    print("Dictionary Iteration:")
    print("Using .items():")
    for key, value in obj.items():
        print(f"{key}: {value}")

    print("\nUsing .keys():")
    for key in obj.keys():
        print(key)

    print("\nUsing .values():")
    for value in obj.values():
        print(value)

def iterate_list(obj):
    """
    Iterate over a list using for loop and enumerate().
    """
    print("\nList Iteration:")
    print("Using for loop:")
    for item in obj:
        print(item)

    print("\nUsing enumerate():")
    for index, item in enumerate(obj):
        print(f"{index}: {item}")

def iterate_set(obj):
    """
    Iterate over a set using for loop.
    """
    print("\nSet Iteration:")
    print("Using for loop:")
    for item in obj:
        print(item)

def main():
    iterate_dictionary(person)
    iterate_list(fruits)
    iterate_set(colors)

if __name__ == "__main__":
    main()
```
