## Exercise
Develop a function that justifies text by adding spaces between words to make all lines the same length.

## Reference Solution
```python
def justify_text(text, line_length):
    """
    Justifies text by adding spaces between words to make all lines the same length.

    Args:
        text (str): The input text.
        line_length (int): The desired length of each line.

    Returns:
        list: A list of justified lines.
    """
    words = text.split()
    lines = []
    current_line = []

    for word in words:
        if len(' '.join(current_line)) + len(word) + 1 > line_length:
            # Add spaces to the current line to justify it
            lines.append(justify_line(current_line, line_length))
            current_line = [word]
        else:
            current_line.append(word)

    # Add the last line
    lines.append(justify_last_line(current_line, line_length))

    return lines


def justify_line(words, line_length):
    """
    Justifies a single line by adding spaces between words.

    Args:
        words (list): A list of words in the line.
        line_length (int): The desired length of the line.

    Returns:
        str: The justified line.
    """
    total_length = sum(len(word) for word in words)
    num_spaces = line_length - total_length
    num_gaps = len(words) - 1

    if num_gaps == 0:
        return words[0] + ' ' * num_spaces

    base_spaces = num_spaces // num_gaps
    extra_spaces = num_spaces % num_gaps

    line = ''
    for i, word in enumerate(words):
        line += word
        if i < len(words) - 1:
            line += ' ' * (base_spaces + (1 if i < extra_spaces else 0))

    return line


def justify_last_line(words, line_length):
    """
    Justifies the last line by adding spaces to the right.

    Args:
        words (list): A list of words in the line.
        line_length (int): The desired length of the line.

    Returns:
        str: The justified line.
    """
    line = ' '.join(words)
    return line + ' ' * (line_length - len(line))


# Example usage
text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
line_length = 50

justified_lines = justify_text(text, line_length)

for line in justified_lines:
    print(line)
```
