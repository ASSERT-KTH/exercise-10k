## Exercise
Implement a Caesar cipher to encrypt and decrypt a given string

## Reference Solution
```python
class CaesarCipher:
    """
    A class used to encrypt and decrypt strings using the Caesar cipher algorithm.
    """

    def __init__(self, shift):
        """
        Initializes the CaesarCipher object with a shift value.

        Args:
            shift (int): The number of positions to shift the alphabet.
        """
        self.shift = shift

    def _shift_char(self, char, direction):
        """
        Shifts a character by the specified number of positions.

        Args:
            char (str): The character to shift.
            direction (int): The direction of the shift (1 for encryption, -1 for decryption).

        Returns:
            str: The shifted character.
        """
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            return chr((ord(char) - ascii_offset + direction * self.shift) % 26 + ascii_offset)
        else:
            return char

    def encrypt(self, plaintext):
        """
        Encrypts the given plaintext using the Caesar cipher algorithm.

        Args:
            plaintext (str): The text to encrypt.

        Returns:
            str: The encrypted text.
        """
        return ''.join(self._shift_char(char, 1) for char in plaintext)

    def decrypt(self, ciphertext):
        """
        Decrypts the given ciphertext using the Caesar cipher algorithm.

        Args:
            ciphertext (str): The text to decrypt.

        Returns:
            str: The decrypted text.
        """
        return ''.join(self._shift_char(char, -1) for char in ciphertext)


def main():
    # Example usage:
    cipher = CaesarCipher(3)
    plaintext = "Hello, World!"
    print(f"Plaintext: {plaintext}")

    ciphertext = cipher.encrypt(plaintext)
    print(f"Ciphertext: {ciphertext}")

    decrypted_text = cipher.decrypt(ciphertext)
    print(f"Decrypted Text: {decrypted_text}")


if __name__ == "__main__":
    main()
```
