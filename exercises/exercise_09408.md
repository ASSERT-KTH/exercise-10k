## Exercise
Create a program that calculates the factorial of a given integer.

## Reference Solution
```python
def factorial_recursive(n):
    """
    Calculate the factorial of a given integer using recursion.

    Args:
    n (int): The input integer.

    Returns:
    int: The factorial of the input integer.

    Raises:
    ValueError: If the input integer is negative.
    """
    if n < 0:
        raise ValueError("Input integer must be non-negative.")
    elif n == 0 or n == 1:
        return 1
    else:
        return n * factorial_recursive(n - 1)


def factorial_iterative(n):
    """
    Calculate the factorial of a given integer using iteration.

    Args:
    n (int): The input integer.

    Returns:
    int: The factorial of the input integer.

    Raises:
    ValueError: If the input integer is negative.
    """
    if n < 0:
        raise ValueError("Input integer must be non-negative.")
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result


def main():
    num = int(input("Enter a non-negative integer: "))
    try:
        recursive_result = factorial_recursive(num)
        iterative_result = factorial_iterative(num)
        print(f"Recursive factorial: {recursive_result}")
        print(f"Iterative factorial: {iterative_result}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
