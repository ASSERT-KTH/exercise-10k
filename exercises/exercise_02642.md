## Exercise
Write a function to access a value in a dictionary by its key

## Reference Solution
```python
def access_dict_value(dictionary, key):
    """
    Access a value in a dictionary by its key.

    Args:
        dictionary (dict): The dictionary to access.
        key: The key of the value to access.

    Returns:
        The value associated with the key if it exists.
        None if the dictionary or key does not exist.

    Raises:
        TypeError: If the input is not a dictionary.
    """
    # Check if the input is a dictionary
    if not isinstance(dictionary, dict):
        raise TypeError("Input must be a dictionary.")

    # Check if the key exists in the dictionary
    if key in dictionary:
        return dictionary[key]
    else:
        return None

# Example usage:
if __name__ == "__main__":

    key_to_access = "name"
    print(access_dict_value(example_dict, key_to_access))  # Output: John

    # Test error handling
    print(access_dict_value(example_dict, "nonexistent_key"))  # Output: None
    try:
        print(access_dict_value("not a dictionary", key_to_access))
    except TypeError as e:
        print(e)  # Output: Input must be a dictionary.
```
