## Exercise
Given a list of integers, find the next greater element for each element

## Reference Solution

def next_greater_element(nums):
    """
    Find the next greater element for each element in the list.

    Args:
    nums (list): A list of integers.

    Returns:
    list: A list of next greater elements. If no greater element is found, returns -1.
    """
    stack = []
    result = [-1] * len(nums)

    for i in range(len(nums)):
        # Pop all elements from the stack that are smaller than the current element
        while stack and nums[stack[-1]] < nums[i]:
            # Update the next greater element for the popped element
            result[stack.pop()] = nums[i]
        # Push the current element onto the stack
        stack.append(i)

    return result

# Example usage
nums = [4, 5, 2, 25]
print(next_greater_element(nums))  # Output: [5, 25, 25, -1]
