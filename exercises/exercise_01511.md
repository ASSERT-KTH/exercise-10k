## Exercise
Write a program that implements a Caesar cipher, a type of substitution cipher where each letter in the plaintext is 'shifted' a certain number of places down the alphabet.

## Reference Solution
```python
def caesar_encrypt(plaintext: str, shift: int) -> str:
    """
    Encrypts a plaintext message using a Caesar cipher with a specified shift value.

    Args:
        plaintext (str): The message to be encrypted.
        shift (int): The number of positions each letter in the plaintext should be shifted down the alphabet.

    Returns:
        str: The encrypted message.
    """
    encrypted_text = ""
    for char in plaintext:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            encrypted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encrypted_text += encrypted_char
        else:
            encrypted_text += char
    return encrypted_text


def caesar_decrypt(ciphertext: str, shift: int) -> str:
    """
    Decrypts a ciphertext message using a Caesar cipher with a specified shift value.

    Args:
        ciphertext (str): The message to be decrypted.
        shift (int): The number of positions each letter in the ciphertext should be shifted up the alphabet.

    Returns:
        str: The decrypted message.
    """
    decrypted_text = ""
    for char in ciphertext:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            decrypted_char = chr((ord(char) - ascii_offset - shift) % 26 + ascii_offset)
            decrypted_text += decrypted_char
        else:
            decrypted_text += char
    return decrypted_text


def main():
    # Example usage:
    plaintext = "Hello, World!"
    shift = 3

    encrypted_text = caesar_encrypt(plaintext, shift)
    print(f"Plaintext: {plaintext}")
    print(f"Shift: {shift}")
    print(f"Encrypted Text: {encrypted_text}")

    decrypted_text = caesar_decrypt(encrypted_text, shift)
    print(f"Decrypted Text: {decrypted_text}")


if __name__ == "__main__":
    main()
```
