## Exercise
Write a program to find the first duplicate in a given list of integers.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Find the first duplicate in a given list of integers.

    Args:
        nums (list): A list of integers.

    Returns:
        int: The first duplicate in the list. If no duplicates are found, returns None.
    """
    seen = set()
    for num in nums:
        if num in seen:
            return num
        seen.add(num)
    return None

# Example usage
def main():
    nums = [1, 2, 3, 4, 5, 2, 6]
    duplicate = find_first_duplicate(nums)
    if duplicate is not None:
        print(f"The first duplicate is: {duplicate}")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
