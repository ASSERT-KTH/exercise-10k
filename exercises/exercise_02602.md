## Exercise
Write a program that accesses the value of a specific key in a dictionary

## Reference Solution
```python
def access_dict_value(dictionary, key):
    """
    Accesses the value of a specific key in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key (any): The key to access.

    Returns:
        any: The value associated with the key, or None if the key is not found.

    Raises:
        TypeError: If the input is not a dictionary.
    """
    if not isinstance(dictionary, dict):
        raise TypeError("Input must be a dictionary")

    try:
        # Attempt to access the value of the key
        value = dictionary[key]
        return value
    except KeyError:
        # If the key is not found, return None
        return None


# Example usage:
my_dict = {"name": "John", "age": 30, "city": "New York"}
key_to_access = "age"

value = access_dict_value(my_dict, key_to_access)

if value is not None:
    print(f"The value of '{key_to_access}' is: {value}")
else:
    print(f"The key '{key_to_access}' was not found in the dictionary.")
```
