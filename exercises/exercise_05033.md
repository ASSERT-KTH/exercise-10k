## Exercise
Write a program that removes duplicates from a given list of numbers using for loops.

## Reference Solution
```python
def remove_duplicates(num_list):
    """
    Removes duplicates from a given list of numbers.

    Args:
        num_list (list): The list of numbers.

    Returns:
        list: The list of numbers with duplicates removed.
    """
    unique_list = []
    for num in num_list:
        found = False
        for unique_num in unique_list:
            if num == unique_num:
                found = True
                break
        if not found:
            unique_list.append(num)
    return unique_list


def main():
    num_list = [1, 2, 2, 3, 4, 4, 5, 6, 6, 7, 8, 8, 9]
    print("Original list: ", num_list)
    unique_list = remove_duplicates(num_list)
    print("List after removing duplicates: ", unique_list)


if __name__ == "__main__":
    main()
```
