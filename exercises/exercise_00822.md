## Exercise
Given a string, find all unique palindromic substrings and return them in a list.

## Reference Solution
```python
def find_palindromic_substrings(s):
    """
    Finds all unique palindromic substrings in a given string.

    Args:
        s (str): The input string.

    Returns:
        list: A list of unique palindromic substrings.
    """
    def is_palindrome(substring):
        """Checks if a substring is a palindrome."""
        return substring == substring[::-1]

    palindromic_substrings = set()
    for length in range(1, len(s) + 1):
        for i in range(len(s) - length + 1):
            substring = s[i:i + length]
            if is_palindrome(substring):
                palindromic_substrings.add(substring)

    return list(palindromic_substrings)


# Example usage:
def main():
    input_string = "abba"
    result = find_palindromic_substrings(input_string)
    print("Unique palindromic substrings:", result)


if __name__ == "__main__":
    main()
```
