## Exercise
Create a program that plays a game of Rock, Paper, Scissors with the user. The program should generate a random choice (rock, paper, or scissors) and ask the user for their choice. The program should then determine the winner based on the game's rules.

## Reference Solution
```python
import random

def get_computer_choice():
    """Generate a random choice for the computer"""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def get_user_choice():
    """Get the user's choice and validate it"""
    while True:
        user_choice = input("Enter your choice (rock, paper, or scissors): ").lower()
        if user_choice in ['rock', 'paper', 'scissors']:
            return user_choice
        print("Invalid choice. Please enter rock, paper, or scissors.")

def determine_winner(computer_choice, user_choice):
    """Determine the winner based on the game's rules"""
    if computer_choice == user_choice:
        return "It's a tie!"
    if (computer_choice == 'rock' and user_choice == 'scissors') or \
       (computer_choice == 'scissors' and user_choice == 'paper') or \
       (computer_choice == 'paper' and user_choice == 'rock'):
        return "Computer wins!"
    return "User wins!"

def main():
    print("Welcome to Rock, Paper, Scissors!")
    play_again = 'y'
    while play_again.lower() == 'y':
        computer_choice = get_computer_choice()
        user_choice = get_user_choice()
        print(f"\nComputer chose: {computer_choice}")
        print(f"You chose: {user_choice}\n")
        print(determine_winner(computer_choice, user_choice))
        play_again = input("\nDo you want to play again? (y/n): ")

if __name__ == "__main__":
    main()
```
