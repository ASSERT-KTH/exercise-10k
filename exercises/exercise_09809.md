## Exercise
Create a program that takes a user's name, age, and location as input and then prints out a greeting message with the user's information. The program should handle invalid inputs and ensure that the age is a positive integer.

## Reference Solution
```python
import re

def get_input():
    name = input("Please enter your name: ")
    age = input("Please enter your age: ")
    location = input("Please enter your location: ")
    return name, age, location

def validate_input(name, age, location):
    if not name:
        print("Error: Name cannot be empty.")
        return False
    if not age:
        print("Error: Age cannot be empty.")
        return False
    if not location:
        print("Error: Location cannot be empty.")
        return False
    if not age.isdigit():
        print("Error: Age must be a whole number.")
        return False
    if int(age) <= 0:
        print("Error: Age must be a positive number.")
        return False
    return True

def process(name, age, location):
    greeting = f"Hello, {name}! You are {age} years old and from {location}."
    return greeting

def main():
    name, age, location = get_input()
    if validate_input(name, age, location):
        greeting = process(name, age, location)
        print(greeting)

if __name__ == "__main__":
    main()
```
