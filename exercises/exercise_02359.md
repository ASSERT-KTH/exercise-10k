## Exercise
Write a program to take a text as input and analyze its content, including word frequency, sentence length, and character frequency

## Reference Solution
```python
import re
from collections import Counter

def analyze_text(text):
    # Remove punctuation and convert to lower case
    text = re.sub(r'[^\w\s]', '', text).lower()

    # Split text into words
    words = text.split()

    # Calculate word frequency
    word_frequency = Counter(words)

    # Calculate sentence length
    sentences = re.split(r'[.!?]', text)
    sentence_length = [len(sentence.split()) for sentence in sentences if sentence]

    # Calculate character frequency
    character_frequency = Counter(text)

    return word_frequency, sentence_length, character_frequency

def print_analysis(word_frequency, sentence_length, character_frequency):
    # Print word frequency
    print("Word Frequency:")
    for word, frequency in word_frequency.most_common():
        print(f"{word}: {frequency}")

    # Print sentence length
    print("\nSentence Length:")
    for i, length in enumerate(sentence_length):
        print(f"Sentence {i+1}: {length} words")

    # Print character frequency
    print("\nCharacter Frequency:")
    for char, frequency in character_frequency.most_common():
        print(f"{char}: {frequency}")

def main():
    text = input("Enter a text: ")
    word_frequency, sentence_length, character_frequency = analyze_text(text)
    print_analysis(word_frequency, sentence_length, character_frequency)

if __name__ == "__main__":
    main()
```
