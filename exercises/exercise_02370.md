## Exercise
Develop a program that calculates the compound interest on an investment.

## Reference Solution
```python
def calculate_compound_interest(principal, interest_rate, time_period, compounding_frequency):
    """
    Calculate the compound interest on an investment.

    Args:
        principal (float): The initial investment amount.
        interest_rate (float): The annual interest rate (in decimal form, e.g., 4% = 0.04).
        time_period (int): The time period of the investment (in years).
        compounding_frequency (int): The frequency of compounding per year (e.g., 1 for annual, 2 for semi-annual, 4 for quarterly).

    Returns:
        float: The final amount after compound interest has been applied.
    """
    return principal * (1 + interest_rate / compounding_frequency) ** (compounding_frequency * time_period)


def main():
    print("Compound Interest Calculator")
    print("--------------------------------")

    principal = float(input("Enter the principal amount: $"))
    interest_rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time_period = int(input("Enter the time period (in years): "))
    compounding_frequency = int(input("Enter the compounding frequency per year (1 for annual, 2 for semi-annual, 4 for quarterly): "))

    final_amount = calculate_compound_interest(principal, interest_rate, time_period, compounding_frequency)
    interest_earned = final_amount - principal

    print(f"\nFinal Amount: ${final_amount:.2f}")
    print(f"Interest Earned: ${interest_earned:.2f}")


if __name__ == "__main__":
    main()
```
