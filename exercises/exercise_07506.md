## Exercise
Create a grade calculator program using variables and conditional statements.

## Reference Solution
```python
# Define a function to calculate the grade
def calculate_grade(score):
    """
    Calculate the grade based on the score.

    Args:
        score (float): The student's score.

    Returns:
        str: The corresponding letter grade.
    """
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

# Define a function to get the student's score
def get_score():
    """
    Get the student's score from the user.

    Returns:
        float: The student's score.
    """
    while True:
        try:
            score = float(input("Enter the student's score (0-100): "))
            if score < 0 or score > 100:
                print("Invalid score. Please enter a score between 0 and 100.")
            else:
                return score
        except ValueError:
            print("Invalid input. Please enter a number.")

# Define a function to display the grade
def display_grade(score, grade):
    """
    Display the student's score and corresponding letter grade.

    Args:
        score (float): The student's score.
        grade (str): The corresponding letter grade.
    """
    print(f"Score: {score}")
    print(f"Grade: {grade}")

# Main function
def main():
    score = get_score()
    grade = calculate_grade(score)
    display_grade(score, grade)

# Run the program
if __name__ == "__main__":
    main()
```
