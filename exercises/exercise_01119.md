## Exercise
Write a program to calculate the interest charged on a credit card given the outstanding balance, interest rate, and time period

## Reference Solution
```python
def calculate_credit_card_interest(outstanding_balance, interest_rate, time_period):
    """
    Calculate the interest charged on a credit card.

    Args:
        outstanding_balance (float): The outstanding balance on the credit card.
        interest_rate (float): The interest rate as a decimal (e.g., 18% = 0.18).
        time_period (int): The time period in months.

    Returns:
        float: The total interest charged.
    """
    # Calculate the monthly interest rate
    monthly_interest_rate = interest_rate / 12

    # Calculate the total interest charged
    total_interest = outstanding_balance * monthly_interest_rate * time_period

    return total_interest

def main():
    # Example usage
    outstanding_balance = 1000.0
    interest_rate = 0.18  # 18% interest rate
    time_period = 6  # 6 months

    total_interest = calculate_credit_card_interest(outstanding_balance, interest_rate, time_period)

    print(f"Outstanding Balance: ${outstanding_balance:.2f}")
    print(f"Interest Rate: {interest_rate*100}%")
    print(f"Time Period: {time_period} months")
    print(f"Total Interest Charged: ${total_interest:.2f}")

if __name__ == "__main__":
    main()
```
