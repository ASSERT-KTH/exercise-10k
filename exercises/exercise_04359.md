## Exercise
Write a program that calculates the MPG (miles per gallon) of a vehicle based on the distance traveled and the amount of fuel consumed

## Reference Solution
```python
def calculate_mpg(distance, fuel_consumed):
    """
    Calculates the miles per gallon (MPG) of a vehicle.

    Args:
        distance (float): The distance traveled in miles.
        fuel_consumed (float): The amount of fuel consumed in gallons.

    Returns:
        float: The miles per gallon (MPG) of the vehicle.
    """
    if fuel_consumed == 0:
        return 0
    mpg = distance / fuel_consumed
    return mpg

def main():
    print("MPG Calculator Program")
    print("------------------------")

    distance = float(input("Enter the distance traveled in miles: "))
    fuel_consumed = float(input("Enter the amount of fuel consumed in gallons: "))

    mpg = calculate_mpg(distance, fuel_consumed)

    print(f"MPG: {mpg:.2f}")

if __name__ == "__main__":
    main()
```
