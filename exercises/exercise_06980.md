## Exercise
Create a function that takes a graph as an argument and returns the maximum flow between two nodes.

from collections import deque

def bfs(graph, source, sink, parent):
    """
    Breadth-first search to find an augmenting path.

    Args:
    graph (list): Residual graph.
    source (int): Source node.
    sink (int): Sink node.
    parent (list): Parent nodes for the augmenting path.

    Returns:
    bool: Whether an augmenting path is found.
    """
    visited = [False] * len(graph)
    queue = deque()

    queue.append(source)
    visited[source] = True

    while queue:
        u = queue.popleft()

        for ind, val in enumerate(graph[u]):
            if not visited[ind] and val > 0:
                queue.append(ind)
                visited[ind] = True
                parent[ind] = u

                if ind == sink:
                    return True

    return False


def ford_fulkerson(graph, source, sink):
    """
    Ford-Fulkerson algorithm with Edmonds-Karp implementation.

    Args:
    graph (list): Flow network.
    source (int): Source node.
    sink (int): Sink node.

    Returns:
    int: Maximum flow.
    """
    parent = [-1] * len(graph)
    max_flow = 0

    while bfs(graph, source, sink, parent):
        path_flow = float("inf")
        s = sink

        while s != source:
            path_flow = min(path_flow, graph[parent[s]][s])
            s = parent[s]

        max_flow += path_flow

        v = sink
        while v != source:
            u = parent[v]
            graph[u][v] -= path_flow
            graph[v][u] += path_flow
            v = parent[v]

    return max_flow


# Example usage
if __name__ == "__main__":
    graph = [[0, 16, 13, 0, 0, 0],
             [0, 0, 10, 12, 0, 0],
             [0, 4, 0, 0, 14, 0],
             [0, 0, 9, 0, 0, 20],
             [0, 0, 0, 7, 0, 4],
             [0, 0, 0, 0, 0, 0]]

    source = 0
    sink = 5

    max_flow = ford_fulkerson(graph, source, sink)
    print("Maximum flow:", max_flow)
