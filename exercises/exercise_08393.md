## Exercise
Create a program that takes two numbers and an operation as input and performs the specified operation. The operation can be one of addition, subtraction, multiplication, or division.

## Reference Solution
```python
# Define a function for each operation
def add(num1, num2):
    return num1 + num2

def subtract(num1, num2):
    return num1 - num2

def multiply(num1, num2):
    return num1 * num2

def divide(num1, num2):
    # Check for division by zero
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return num1 / num2

# Create a dictionary mapping operation names to their corresponding functions
operations = {
    'addition': add,
    'subtraction': subtract,
    'multiplication': multiply,
    'division': divide,
    '+': add,
    '-': subtract,
    '*': multiply,
    '/': divide
}

def calculate(num1, num2, operation):
    """
    Perform the specified operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The operation to perform. Can be one of 'addition', 'subtraction', 'multiplication', 'division', '+', '-', '*', '/'.

    Returns:
        float: The result of the operation.
    """
    # Check if the operation is valid
    if operation not in operations:
        raise ValueError("Invalid operation")

    # Perform the operation
    return operations[operation](num1, num2)

# Test the function
def main():
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the operation (+, -, *, /, addition, subtraction, multiplication, division): ")

    try:
        result = calculate(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except ZeroDivisionError as e:
        print(e)
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
