## Exercise
Write a program that finds the maximum subarray of a given list of numbers.

## Reference Solution

def max_subarray(nums):
    """
    Finds the maximum subarray of a given list of numbers.

    Args:
    - nums (list): A list of integers.

    Returns:
    - max_sum (int): The sum of the maximum subarray.
    - max_subarray (list): The maximum subarray.
    """
    if not nums:
        return 0, []

    max_current = max_global = nums[0]
    start = end = temp_start = 0

    for i in range(1, len(nums)):
        if nums[i] > max_current + nums[i]:
            max_current = nums[i]
            temp_start = i
        else:
            max_current += nums[i]

        if max_current > max_global:
            max_global = max_current
            start = temp_start
            end = i

    max_subarray = nums[start:end+1]
    return max_global, max_subarray

# Example usage:
if __name__ == "__main__":
    nums = [-2, -3, 4, -1, -2, 1, 5, -3]
    max_sum, max_subarray = max_subarray(nums)
    print("Maximum subarray sum:", max_sum)
    print("Maximum subarray:", max_subarray)
