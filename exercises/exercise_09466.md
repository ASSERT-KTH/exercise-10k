## Exercise
Create a program that finds the longest palindromic substring in a given string.

## Reference Solution
```python
def longest_palindromic_substring(s: str) -> str:
    """
    Finds the longest palindromic substring in a given string.

    Args:
        s (str): The input string.

    Returns:
        str: The longest palindromic substring.
    """

    def expand_around_center(s: str, left: int, right: int) -> str:
        """
        Expands around the center of a potential palindrome.

        Args:
            s (str): The input string.
            left (int): The left index.
            right (int): The right index.

        Returns:
            str: The longest palindromic substring centered at the given indices.
        """
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
        return s[left + 1:right]

    longest = ""
    for i in range(len(s)):
        # Odd-length palindrome
        odd_palindrome = expand_around_center(s, i, i)
        if len(odd_palindrome) > len(longest):
            longest = odd_palindrome

        # Even-length palindrome
        even_palindrome = expand_around_center(s, i, i + 1)
        if len(even_palindrome) > len(longest):
            longest = even_palindrome

    return longest

    input_string = "cbbd"
    print(f"\nInput string: {input_string}")
    print(f"Longest palindromic substring: {longest_palindromic_substring(input_string)}")
```
