## Exercise
Write a program that calculates the symmetric difference of two lists of numbers.

## Reference Solution
```python
def symmetric_difference(list1, list2):
    """
    Calculate the symmetric difference of two lists.

    Args:
        list1 (list): The first list of numbers.
        list2 (list): The second list of numbers.

    Returns:
        list: The symmetric difference of the two lists.
    """
    return list(set(list1) ^ set(list2))

def main():
    # Example usage
    list1 = [1, 2, 3, 4, 5]
    list2 = [4, 5, 6, 7, 8]

    print("List 1:", list1)
    print("List 2:", list2)

    symmetric_diff = symmetric_difference(list1, list2)
    print("Symmetric Difference:", symmetric_diff)

if __name__ == "__main__":
    main()
```
