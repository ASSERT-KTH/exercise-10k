## Exercise
Create a dictionary with 5 key-value pairs. Delete a specific key-value pair.

## Reference Solution
```python
def create_and_delete_dict():
    # Create a dictionary with 5 key-value pairs
    my_dict = {
        "apple": 1,
        "banana": 2,
        "cherry": 3,
        "date": 4,
        "elderberry": 5
    }

    print("Original Dictionary:")
    print(my_dict)

    # Delete a specific key-value pair
    key_to_delete = "cherry"
    if key_to_delete in my_dict:
        del my_dict[key_to_delete]
    else:
        print(f"Key '{key_to_delete}' not found in the dictionary.")

    print("\nDictionary after deletion:")
    print(my_dict)


if __name__ == "__main__":
    create_and_delete_dict()
```
