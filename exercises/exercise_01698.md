## Exercise
Write a function to check if a given list is sorted.

## Reference Solution
```python
def is_sorted(lst, order='asc'):
    """
    Check if a list is sorted.

    Args:
    - lst (list): The list to check.
    - order (str, optional): The order of sorting. Defaults to 'asc'. Can be 'asc' or 'desc'.

    Returns:
    - bool: True if the list is sorted, False otherwise.
    """
    if order == 'asc':
        return lst == sorted(lst)
    elif order == 'desc':
        return lst == sorted(lst, reverse=True)
    else:
        raise ValueError("Invalid order. Use 'asc' or 'desc'.")

# Example usage:
if __name__ == "__main__":
    print(is_sorted([1, 2, 3, 4, 5]))  # True
    print(is_sorted([5, 4, 3, 2, 1], order='desc'))  # True
    print(is_sorted([1, 3, 2, 4, 5]))  # False
```
