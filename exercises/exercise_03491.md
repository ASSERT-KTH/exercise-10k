## Exercise
Write a program that takes the coordinates of two points as input and returns the distance between them.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the Euclidean distance between two points.

    Args:
    point1 (list): Coordinates of the first point (e.g., [x, y] or [x, y, z])
    point2 (list): Coordinates of the second point (e.g., [x, y] or [x, y, z])

    Returns:
    float: The distance between the two points
    """
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of dimensions")

    squared_diffs = [(a - b) ** 2 for a, b in zip(point1, point2)]
    return math.sqrt(sum(squared_diffs))

def main():
    # Example usage:
    point1 = [1, 2]  # 2D point
    point2 = [4, 6]
    distance = calculate_distance(point1, point2)
    print(f"The distance between {point1} and {point2} is {distance:.2f}")

    point1 = [1, 2, 3]  # 3D point
    point2 = [4, 5, 6]
    distance = calculate_distance(point1, point2)
    print(f"The distance between {point1} and {point2} is {distance:.2f}")

if __name__ == "__main__":
    main()
```
