## Exercise
Write a program that takes in a time in one unit (e.g. seconds) and converts it to another unit (e.g. minutes) using conditional statements.

## Reference Solution
```python
# Function to convert time
def convert_time(time, from_unit, to_unit):
    # Convert time to seconds
    if from_unit == "seconds":
        time_in_seconds = time
    elif from_unit == "minutes":
        time_in_seconds = time * 60
    elif from_unit == "hours":
        time_in_seconds = time * 3600
    elif from_unit == "days":
        time_in_seconds = time * 86400
    else:
        return "Invalid from_unit"

    # Convert time from seconds to desired unit
    if to_unit == "seconds":
        return time_in_seconds
    elif to_unit == "minutes":
        return time_in_seconds / 60
    elif to_unit == "hours":
        return time_in_seconds / 3600
    elif to_unit == "days":
        return time_in_seconds / 86400
    else:
        return "Invalid to_unit"

# Main function
def main():
    time = float(input("Enter time: "))
    from_unit = input("Enter unit of time (seconds, minutes, hours, days): ")
    to_unit = input("Enter unit to convert to (seconds, minutes, hours, days): ")

    result = convert_time(time, from_unit, to_unit)
    if "Invalid" not in str(result):
        print(f"{time} {from_unit} is equal to {result} {to_unit}")
    else:
        print(result)

# Run the main function
if __name__ == "__main__":
    main()
```
