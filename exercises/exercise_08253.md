## Exercise
Write a program that administers a quiz using if/else statements.

## Reference Solution
```python
# Define the quiz questions and answers
quiz_questions = {
    "What is the capital of France?": "Paris",
    "What is the largest planet in our solar system?": "Jupiter",
    "Who painted the Starry Night?": "Van Gogh",
    "What is the chemical symbol for gold?": "Au"
}

def administer_quiz(questions):
    """Administers the quiz and returns the final score."""
    score = 0
    for question, answer in questions.items():
        user_answer = input(question + " ")
        if user_answer.lower() == answer.lower():
            print("Correct!")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer is {answer}.")
    return score

def main():
    print("Welcome to the quiz!")
    score = administer_quiz(quiz_questions)
    print(f"\nQuiz complete! Your final score is {score} out of {len(quiz_questions)}.")

if __name__ == "__main__":
    main()
```
