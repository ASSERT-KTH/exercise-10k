Write a program that takes two strings as input and returns the concatenated string.

def concatenate_strings(str1, str2):
    """
    This function takes two strings as input and returns the concatenated string.

    Args:
        str1 (str): The first string to be concatenated.
        str2 (str): The second string to be concatenated.

    Returns:
        str: The concatenated string.
    """
    return str1 + str2

# Example usage:
if __name__ == "__main__":
