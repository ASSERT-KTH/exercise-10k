## Exercise
Write a program to format a given string with placeholders for variables.

## Reference Solution
```python
def format_string(template, **kwargs):
    """
    Formats a given string with placeholders for variables.

    Args:
        template (str): The string template with placeholders.
        **kwargs: Keyword arguments to fill the placeholders.

    Returns:
        str: The formatted string.
    """
    try:
        formatted_string = template.format(**kwargs)
        return formatted_string
    except KeyError as e:
        print(f"Error: Missing placeholder value for '{e.args[0]}'")
        return None

def main():
    # Example usage
    template = "Hello, my name is {name} and I am {age} years old."
    name = "John"
    age = 30

    formatted_string = format_string(template, name=name, age=age)
    print(formatted_string)

    # Example with missing placeholder value
    template = "Hello, my name is {name} and I am {age} years old."
    name = "John"

    formatted_string = format_string(template, name=name)
    print(formatted_string)

if __name__ == "__main__":
    main()
```
