## Exercise
Determine if a given string contains any duplicate characters.

## Reference Solution
```python
def has_duplicates(s: str) -> bool:
    """
    Checks if a given string contains any duplicate characters.

    Args:
        s (str): The input string.

    Returns:
        bool: True if the string contains duplicates, False otherwise.
    """
    seen = set()
    for char in s:
        if char in seen:
            return True
        seen.add(char)
    return False

# Example usage:
if __name__ == "__main__":
    strings = ["hello", "world", "abcdefg", "aabbcc"]
    for s in strings:
        print(f"String '{s}' has duplicates: {has_duplicates(s)}")
```
