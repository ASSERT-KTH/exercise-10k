## Exercise
Create a function that simulates a game of rock, paper, scissors and returns the winner

## Reference Solution
```python
import random

def rock_paper_scissors(player1, player2):
    """
    Simulates a game of rock, paper, scissors between two players.

    Args:
        player1 (str): The choice of player 1. Can be 'rock', 'paper', or 'scissors'.
        player2 (str): The choice of player 2. Can be 'rock', 'paper', or 'scissors'.

    Returns:
        str: The winner of the game. Can be 'player1', 'player2', or 'tie'.
    """
    choices = ['rock', 'paper', 'scissors']
    if player1 not in choices or player2 not in choices:
        raise ValueError("Invalid choice. Must be 'rock', 'paper', or 'scissors'.")

    if player1 == player2:
        return 'tie'
    elif (player1 == 'rock' and player2 == 'scissors') or \
         (player1 == 'scissors' and player2 == 'paper') or \
         (player1 == 'paper' and player2 == 'rock'):
        return 'player1'
    else:
        return 'player2'

def play_game():
    """
    Plays a game of rock, paper, scissors between two players.

    Returns:
        None
    """
    print("Welcome to Rock, Paper, Scissors!")
    player1 = input("Enter player 1's choice (rock, paper, scissors): ")
    player2 = input("Enter player 2's choice (rock, paper, scissors): ")
    winner = rock_paper_scissors(player1.lower(), player2.lower())
    if winner == 'tie':
        print("It's a tie!")
    elif winner == 'player1':
        print("Player 1 wins!")
    else:
        print("Player 2 wins!")

def play_computer_game():
    """
    Plays a game of rock, paper, scissors between a player and the computer.

    Returns:
        None
    """
    print("Welcome to Rock, Paper, Scissors!")
    player = input("Enter your choice (rock, paper, scissors): ")
    computer = random.choice(['rock', 'paper', 'scissors'])
    print(f"Computer chose: {computer}")
    winner = rock_paper_scissors(player.lower(), computer)
    if winner == 'tie':
        print("It's a tie!")
    elif winner == 'player1':
        print("You win!")
    else:
        print("Computer wins!")

# Example usage:
play_game()
play_computer_game()
```
