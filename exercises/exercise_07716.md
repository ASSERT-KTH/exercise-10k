## Exercise
Find the most frequent character in a given string.

## Reference Solution
```python
def most_frequent_char(s):
    """
    Find the most frequent character in a string.

    Args:
        s (str): The input string.

    Returns:
        str: The most frequent character.
    """
    # Create a dictionary to store the frequency of each character
    freq_dict = {}

    # Iterate through each character in the string
    for char in s:
        # If the character is already in the dictionary, increment its count
        if char in freq_dict:
            freq_dict[char] += 1
        # Otherwise, add it to the dictionary with a count of 1
        else:
            freq_dict[char] = 1

    # Find the character with the maximum frequency
    max_freq = max(freq_dict.values())
    most_frequent_chars = [char for char, freq in freq_dict.items() if freq == max_freq]

    # Return the most frequent character(s)
    if len(most_frequent_chars) == 1:
        return most_frequent_chars[0]
    else:
        return most_frequent_chars

# Example usage
s = "hello world"
print(most_frequent_char(s))
```
