## Exercise
Write a program that takes a JSON object as input and processes it, then outputs the result.

## Reference Solution
```python
# Import the json module
import json

def process_json(json_object):
    """
    Process a JSON object.

    Args:
    - json_object (dict or str): A dictionary or a string representing a JSON object.

    Returns:
    - dict: The processed JSON object.
    """
    # If the input is a string, parse it into a dictionary
    if isinstance(json_object, str):
        json_object = json.loads(json_object)

    # Process the JSON object (for example, convert all strings to uppercase)
    processed_object = {key: value.upper() if isinstance(value, str) else value for key, value in json_object.items()}

    return processed_object

def main():
    # Example usage
    json_string = '''
    {
        "name": "John Doe",
        "age": 30,
        "city": "New York"
    }
    '''

    result = process_json(json_string)
    print("Processed JSON object:")
    print(json.dumps(result, indent=4))

if __name__ == "__main__":
    main()
```
