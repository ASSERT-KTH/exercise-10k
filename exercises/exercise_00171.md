## Exercise
Write a program to understand the scope of variables (local, global) and how to access them.

## Reference Solution
```python
# Global variable
global_variable = 10

def outer_function():
    # Nonlocal variable (local to outer_function)
    outer_variable = 20

    def inner_function():
        # Local variable (local to inner_function)
        inner_variable = 30

        # Accessing global variable
        print("Global variable:", global_variable)

        # Accessing outer variable (nonlocal)
        print("Outer variable:", outer_variable)

        # Accessing local variable
        print("Inner variable:", inner_variable)

        # Modifying global variable
        global global_variable
        global_variable += 10
        print("Modified global variable:", global_variable)

        # Modifying outer variable (nonlocal)
        nonlocal outer_variable
        outer_variable += 10
        print("Modified outer variable:", outer_variable)

    inner_function()
    print("Outer variable after modification:", outer_variable)

outer_function()
print("Global variable after modification:", global_variable)
```
