## Exercise
Create a password generator that generates a random password based on user input.

## Reference Solution
```python
import string
import secrets

def password_generator(length=12, use_uppercase=True, use_numbers=True, use_special_chars=True):
    """
    Generates a random password based on user input.

    Args:
        length (int): The length of the password. Defaults to 12.
        use_uppercase (bool): Whether to include uppercase letters. Defaults to True.
        use_numbers (bool): Whether to include numbers. Defaults to True.
        use_special_chars (bool): Whether to include special characters. Defaults to True.

    Returns:
        str: The generated password.
    """

    # Define the characters that can be included in the password
    chars = string.ascii_lowercase
    if use_uppercase:
        chars += string.ascii_uppercase
    if use_numbers:
        chars += string.digits
    if use_special_chars:
        chars += string.punctuation

    # Generate the password
    password = ''.join(secrets.choice(chars) for _ in range(length))

    return password

def main():
    print("Password Generator")
    print("-------------------")

    # Ask for user input
    length = int(input("Enter the length of the password (default=12): ") or 12)
    use_uppercase = input("Include uppercase letters? (y/n, default=y): ").lower() != 'n'
    use_numbers = input("Include numbers? (y/n, default=y): ").lower() != 'n'
    use_special_chars = input("Include special characters? (y/n, default=y): ").lower() != 'n'

    # Generate and print the password
    password = password_generator(length, use_uppercase, use_numbers, use_special_chars)
    print("Generated Password : ", password)

if __name__ == "__main__":
    main()
```
