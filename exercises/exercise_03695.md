## Exercise
Write a program to calculate the grade of a student given the marks obtained in different subjects.

## Reference Solution
```python
def calculate_grade(marks):
    """
    Calculate the grade based on the average marks.
    
    Args:
    marks (dict): A dictionary containing the marks obtained in different subjects.
    
    Returns:
    str: The grade of the student.
    """
    # Calculate the average marks
    average_marks = sum(marks.values()) / len(marks)
    
    # Determine the grade based on the average marks
    if average_marks >= 90:
        return "A"
    elif average_marks >= 80:
        return "B"
    elif average_marks >= 70:
        return "C"
    elif average_marks >= 60:
        return "D"
    else:
        return "F"

def main():
    # Get the marks obtained in different subjects from the user
    num_subjects = int(input("Enter the number of subjects: "))
    marks = {}
    for i in range(num_subjects):
        subject = input(f"Enter the name of subject {i+1}: ")
        mark = float(input(f"Enter the marks obtained in {subject}: "))
        marks[subject] = mark
    
    # Calculate and display the grade
    grade = calculate_grade(marks)
    print(f"Grade: {grade}")

if __name__ == "__main__":
    main()
```
