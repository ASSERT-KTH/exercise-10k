## Exercise
Create a program that counts the number of variables of each data type in a given list.

## Reference Solution
```python
from collections import defaultdict

def count_data_types(lst):
    """
    Counts the number of variables of each data type in a given list.

    Args:
        lst (list): The list to count data types from.

    Returns:
        dict: A dictionary with data types as keys and their counts as values.
    """
    data_type_counts = defaultdict(int)

    for item in lst:
        data_type = type(item).__name__
        data_type_counts[data_type] += 1

    return dict(data_type_counts)


def main():
    # Example list
    example_list = [1, 2, 'a', 'b', 3.5, True, None, [1, 2], {'a': 1}, (1, 2)]

    # Count data types
    data_type_counts = count_data_types(example_list)

    # Print data type counts
    print("Data Type Counts:")
    for data_type, count in data_type_counts.items():
        print(f"{data_type}: {count}")


if __name__ == "__main__":
    main()
```
