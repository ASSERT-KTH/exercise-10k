## Exercise
Write a program that asks the user a series of questions and keeps track of the score.

## Reference Solution
```python
# Define the questions, options, and answers
questions = [
    {
        "question": "What is the capital of France?",
        "options": ["Paris", "London", "Berlin", "Rome"],
        "answer": "Paris"
    },
    {
        "question": "What is the largest planet in our solar system?",
        "options": ["Earth", "Saturn", "Jupiter", "Uranus"],
        "answer": "Jupiter"
    },
    {
        "question": "Who painted the Mona Lisa?",
        "options": ["Leonardo da Vinci", "Michelangelo", "Raphael", "Caravaggio"],
        "answer": "Leonardo da Vinci"
    }
]

def quiz(questions):
    score = 0
    for i, question in enumerate(questions, start=1):
        print(f"\nQuestion {i}: {question['question']}")
        for j, option in enumerate(question['options'], start=1):
            print(f"{j}. {option}")
        answer = input("Enter the number of your answer: ")
        if question['options'][int(answer) - 1] == question['answer']:
            print("Correct!")
            score += 1
        else:
            print(f"Sorry, that's incorrect. The correct answer is {question['answer']}.")
    print(f"\nQuiz complete! Your final score is {score} out of {len(questions)}.")

def main():
    print("Welcome to the quiz!")
    quiz(questions)

if __name__ == "__main__":
    main()
```
