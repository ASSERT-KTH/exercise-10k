## Exercise
Create a program that takes the coordinates of two points as input and returns the distance between them.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the distance between two points in a 2D or 3D space.

    Args:
        point1 (tuple): The coordinates of the first point.
        point2 (tuple): The coordinates of the second point.

    Returns:
        float: The distance between the two points.
    """

    # Check if both points have the same number of dimensions
    if len(point1) != len(point2):
        raise ValueError("Both points must have the same number of dimensions")

    # Calculate the squared differences between corresponding coordinates
    squared_differences = [(a - b) ** 2 for a, b in zip(point1, point2)]

    # Calculate the sum of the squared differences
    sum_of_squared_differences = sum(squared_differences)

    # Calculate the distance using the Euclidean distance formula
    distance = math.sqrt(sum_of_squared_differences)

    return distance

def main():
    # Example usage:
    point1 = (1, 2, 3)  # Coordinates of the first point
    point2 = (4, 5, 6)  # Coordinates of the second point

    distance = calculate_distance(point1, point2)
    print(f"The distance between {point1} and {point2} is {distance}")

if __name__ == "__main__":
    main()
```
