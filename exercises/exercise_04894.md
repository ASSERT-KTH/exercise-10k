## Exercise
Remove and return the last inserted key-value pair.

## Reference Solution
```python
from collections import OrderedDict

class KeyValueStore:
    def __init__(self):
        self.store = OrderedDict()

    def insert(self, key, value):
        self.store[key] = value

    def remove_last(self):
        if not self.store:
            return None
        last_key = next(reversed(self.store))
        last_value = self.store.pop(last_key)
        return last_key, last_value

# Example usage:
store = KeyValueStore()
store.insert('key1', 'value1')
store.insert('key2', 'value2')
store.insert('key3', 'value3')

print(store.remove_last())  # Output: ('key3', 'value3')
```
