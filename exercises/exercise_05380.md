## Exercise
Write a program to unzip a list of tuples into two or more separate lists.

## Reference Solution
```python
def unzip_tuples(tuples_list):
    """
    Unzips a list of tuples into separate lists.

    Args:
        tuples_list (list): A list of tuples.

    Returns:
        list: A list of separate lists, one for each tuple element.
    """
    return list(zip(*tuples_list))

def main():
    # Example usage
    tuples_list = [(1, 2, 3), (4, 5, 6), (7, 8, 9)]
    unzipped_lists = unzip_tuples(tuples_list)

    # Convert the zipped result back into separate lists
    list1, list2, list3 = map(list, unzipped_lists)

    print("Unzipped List 1:", list1)
    print("Unzipped List 2:", list2)
    print("Unzipped List 3:", list3)

if __name__ == "__main__":
    main()
```
