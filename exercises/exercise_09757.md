## Exercise
Write a program to calculate the fuel efficiency of a vehicle

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumed):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
        distance (float): The distance traveled in miles.
        fuel_consumed (float): The amount of fuel consumed in gallons.

    Returns:
        float: The fuel efficiency in miles per gallon (mpg).
    """
    if fuel_consumed == 0:
        return 0
    else:
        return distance / fuel_consumed


def main():
    print("Fuel Efficiency Calculator")
    print("---------------------------")

    distance = float(input("Enter the distance traveled in miles: "))
    fuel_consumed = float(input("Enter the amount of fuel consumed in gallons: "))

    fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumed)

    print(f"Fuel efficiency: {fuel_efficiency:.2f} miles per gallon (mpg)")


if __name__ == "__main__":
    main()
```
