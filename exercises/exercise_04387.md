## Exercise
Understand the scope of variables in different contexts

## Reference Solution
```python
def my_function():
    x = 10  # local variable
    print(x)  # prints 10

my_function()
print(x)  # raises a NameError, x is not defined
```

```python
x = 10  # global variable

def my_function():
    print(x)  # prints 10

my_function()
print(x)  # prints 10
```

```python
# mymodule.py
x = 10  # module variable

def my_function():
    print(x)  # prints 10
```

```python
class MyClass:
    x = 10  # class variable

    def my_method(self):
        print(x)  # prints 10
```

```python
class MyClass:
    def __init__(self):
        self.x = 10  # instance variable

    def my_method(self):
        print(self.x)  # prints 10
```

```python
def outer_function():
    x = 10  # closure variable

    def inner_function():
        print(x)  # prints 10

    inner_function()

outer_function()
```

```python
x = 10  # global variable

lambda_function = lambda: print(x)  # lambda variable
lambda_function()  # prints 10
```
