## Exercise
Write a function that uses currying to transform a function with multiple arguments into a sequence of functions, each taking a single argument.

def curry(func):
    """
    This function uses currying to transform a function with multiple arguments into a sequence of functions, each taking a single argument.

    Args:
        func (function): The original function with multiple arguments.

    Returns:
        function: A curried version of the original function.
    """

    # Define a nested function that will be returned by the curry function
    def curried(*args, **kwargs):
        
        # If the number of arguments passed is equal to the number of arguments required by the original function,
        # then we can call the original function and return its result
        if len(args) + len(kwargs) == func.__code__.co_argcount:
            return func(*args, **kwargs)
        
        # If the number of arguments passed is less than the number of arguments required by the original function,
        # then we need to return another function that takes the remaining arguments
        else:
            # Define another nested function that takes the remaining arguments
            def next_curried(*next_args, **next_kwargs):
                return curried(*(args + next_args), **dict(kwargs, **next_kwargs))
            
            # Return the next_curried function
            return next_curried
    
    # Return the curried function
    return curried

# Example usage:
def add(a, b, c):
    return a + b + c

curried_add = curry(add)
print(curried_add(1)(2)(3))  # Output: 6
