## Exercise
Create a list of 3 integers and remove the element at a specific index.

## Reference Solution
```python
def remove_element_at_index(lst, index):
    """
    Removes the element at the specified index from the list.

    Args:
        lst (list): The list of integers.
        index (int): The index of the element to remove.

    Returns:
        list: The updated list after removing the element.
    """
    # Check if the index is within the bounds of the list
    if index < 0 or index >= len(lst):
        raise IndexError("Index out of range")

    # Remove the element at the specified index
    lst.pop(index)
    return lst


# Example usage:
if __name__ == "__main__":
    # Create a list of 3 integers
    lst = [10, 20, 30]
    print("Original list:", lst)

    # Remove the element at index 1
    updated_lst = remove_element_at_index(lst, 1)
    print("Updated list:", updated_lst)
```
