## Exercise
Create a program that plays a text adventure game against the user. The user can choose to move in different directions and interact with objects.

## Reference Solution
```python
# Import the random module for generating random outcomes
import random

# Define a dictionary to store the game world
game_world = {
    'castle': {
        'description': 'You are standing in front of a grand castle.',
        'exits': {'north': 'forest', 'south': 'cave'},
        'objects': ['key', 'sword']
    },
    'forest': {
        'description': 'You are standing in a dense forest.',
        'exits': {'south': 'castle', 'east': 'cave'},
        'objects': ['apple', 'bow']
    },
    'cave': {
        'description': 'You are standing in a dark cave.',
        'exits': {'north': 'castle', 'west': 'forest'},
        'objects': ['gold', 'shield']
    }
}

# Define a dictionary to store the player's inventory
player_inventory = []

# Define a variable to store the player's current location
player_location = 'castle'

# Game loop
while True:
    # Print the description of the player's current location
    print(game_world[player_location]['description'])

    # Print the exits from the player's current location
    print('Exits:')
    for exit in game_world[player_location]['exits']:
        print(f'- {exit.capitalize()}')

    # Print the objects in the player's current location
    if game_world[player_location]['objects']:
        print('Objects:')
        for obj in game_world[player_location]['objects']:
            print(f'- {obj.capitalize()}')
    else:
        print('No objects in this location.')

    # Get the player's input
    action = input('What would you like to do? (type "move" to move, "take" to take an object, "inventory" to view your inventory, or "quit" to quit the game) ').lower()

    # Handle the player's input
    if action == 'move':
        direction = input('Which direction would you like to move? ').lower()
        if direction in game_world[player_location]['exits']:
            player_location = game_world[player_location]['exits'][direction]
        else:
            print('You cannot move that way.')
    elif action == 'take':
        obj = input('Which object would you like to take? ').lower()
        if obj in game_world[player_location]['objects']:
            player_inventory.append(obj)
            game_world[player_location]['objects'].remove(obj)
            print(f'You took the {obj}.')
        else:
            print('There is no such object in this location.')
    elif action == 'inventory':
        print('Your inventory:')
        for item in player_inventory:
            print(f'- {item.capitalize()}')
    elif action == 'quit':
        print('Thanks for playing!')
        break
    else:
        print('Invalid input. Please try again.')
```
