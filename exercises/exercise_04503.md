## Exercise
Create a program that checks if a user-inputted password meets certain criteria: it must be at least 8 characters long, contain at least one uppercase letter, one lowercase letter, and one digit.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the given criteria.

    Args:
        password (str): The password to be validated.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the criteria
    criteria = {
        "length": len(password) >= 8,
        "uppercase": any(char.isupper() for char in password),
        "lowercase": any(char.islower() for char in password),
        "digit": any(char.isdigit() for char in password)
    }

    # Check if all criteria are met
    return all(criteria.values())

def main():
    # Get user input
    password = input("Enter a password: ")

    # Validate the password
    if validate_password(password):
        print("Password is valid.")
    else:
        print("Password is not valid. Please ensure it meets the following criteria:")
        print("* At least 8 characters long")
        print("* Contains at least one uppercase letter")
        print("* Contains at least one lowercase letter")
        print("* Contains at least one digit")

if __name__ == "__main__":
    main()
```
