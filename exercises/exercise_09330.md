## Exercise
Write a program to perform various operations using variables, such as addition, subtraction, multiplication, and division.

## Reference Solution
```python
# Define variables
num1 = 10
num2 = 2

# Addition
addition = num1 + num2
print(f"Addition: {num1} + {num2} = {addition}")

# Subtraction
subtraction = num1 - num2
print(f"Subtraction: {num1} - {num2} = {subtraction}")

# Multiplication
multiplication = num1 * num2
print(f"Multiplication: {num1} * {num2} = {multiplication}")

# Division
if num2 != 0:
    division = num1 / num2
    print(f"Division: {num1} / {num2} = {division}")
else:
    print("Error: Division by zero is not allowed.")

# Ask user for input
def get_user_input():
    try:
        num1 = float(input("Enter the first number: "))
        num2 = float(input("Enter the second number: "))
        return num1, num2
    except ValueError:
        print("Invalid input. Please enter a number.")
        return get_user_input()

# Perform operations with user input
def perform_operations():
    num1, num2 = get_user_input()
    print("\nOperations:")
    print(f"Addition: {num1} + {num2} = {num1 + num2}")
    print(f"Subtraction: {num1} - {num2} = {num1 - num2}")
    print(f"Multiplication: {num1} * {num2} = {num1 * num2}")
    if num2 != 0:
        print(f"Division: {num1} / {num2} = {num1 / num2}")
    else:
        print("Error: Division by zero is not allowed.")

# Main function
def main():
    print("Arithmetic Operations")
    print("---------------------")
    print("Example with predefined numbers:")
    num1 = 10
    num2 = 2
    print(f"Addition: {num1} + {num2} = {num1 + num2}")
    print(f"Subtraction: {num1} - {num2} = {num1 - num2}")
    print(f"Multiplication: {num1} * {num2} = {num1 * num2}")
    if num2 != 0:
        print(f"Division: {num1} / {num2} = {num1 / num2}")
    else:
        print("Error: Division by zero is not allowed.")
    print("\nNow, let's perform operations with user input.")
    perform_operations()

if __name__ == "__main__":
    main()
```
