## Exercise
Write a program to compare two given strings lexicographically.

## Reference Solution
```python
def compare_strings_lexicographically(str1: str, str2: str) -> int:
    """
    Compare two strings lexicographically.

    Args:
    str1 (str): The first string to compare.
    str2 (str): The second string to compare.

    Returns:
    int: A negative integer if str1 is lexicographically less than str2,
         zero if str1 is equal to str2,
         a positive integer if str1 is lexicographically greater than str2.
    """
    if str1 < str2:
        return -1  # str1 is lexicographically less than str2
    elif str1 > str2:
        return 1  # str1 is lexicographically greater than str2
    else:
        return 0  # str1 is equal to str2


def main():
    # Example usage
    str1 = input("Enter the first string: ")
    str2 = input("Enter the second string: ")

    result = compare_strings_lexicographically(str1, str2)

    if result < 0:
        print(f"'{str1}' is lexicographically less than '{str2}'")
    elif result > 0:
        print(f"'{str1}' is lexicographically greater than '{str2}'")
    else:
        print(f"'{str1}' is equal to '{str2}'")


if __name__ == "__main__":
    main()
```
