## Exercise
Create a program that takes two integers as input and performs addition, subtraction, multiplication, and division operations based on the user's choice. The program should handle division by zero and invalid input.

## Reference Solution
```python
#include <iostream>
#include <stdexcept>

using namespace std;

int add(int a, int b) {
    return a + b;
}

int sub(int a, int b) {
    return a - b;
}

int mul(int a, int b) {
    return a * b;
}

int div(int a, int b) {
    if (b == 0) {
        throw runtime_error("Division by zero");
    }
    return a / b;
}

int main() {
    int n1, n2;
    char op;

    cout << "Enter two integers: ";
    cin >> n1 >> n2;

    cout << "Choose an operator (+, -, *, /): ";
    cin >> op;

    try {
        switch (op) {
            case '+':
                cout << n1 << " + " << n2 << " = " << add(n1, n2) << endl;
                break;
            case '-':
                cout << n1 << " - " << n2 << " = " << sub(n1, n2) << endl;
                break;
            case '*':
                cout << n1 << " * " << n2 << " = " << mul(n1, n2) << endl;
                break;
            case '/':
                cout << n1 << " / " << n2 << " = " << div(n1, n2) << endl;
                break;
            default:
                cout << "Invalid operator" << endl;
                break;
        }
    } catch (const exception& e) {
        cerr << "Error: " << e.what() << endl;
    }

    return 0;
}
```
