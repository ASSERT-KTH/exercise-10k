## Exercise
Develop a program that calculates the insurance premium based on the policyholder's age and health status. The program should use if/else statements to determine the insurance premium.

## Reference Solution
```python
def calculate_premium(age, health_status):
    """
    Calculates the insurance premium based on age and health status.

    Args:
    - age (int): The policyholder's age.
    - health_status (str): The policyholder's health status ('good', 'average', or 'poor').

    Returns:
    - premium (float): The calculated insurance premium.
    """

    # Base premium amount
    base_premium = 500.0

    # Age-based premium adjustments
    if age < 25:
        age_adjustment = 0.2  # 20% increase for young adults
    elif age < 40:
        age_adjustment = 0.1  # 10% increase for adults
    elif age < 60:
        age_adjustment = 0.0  # No adjustment for middle-aged adults
    else:
        age_adjustment = -0.1  # 10% decrease for seniors

    # Health-based premium adjustments
    if health_status == 'good':
        health_adjustment = -0.2  # 20% decrease for good health
    elif health_status == 'average':
        health_adjustment = 0.0  # No adjustment for average health
    elif health_status == 'poor':
        health_adjustment = 0.3  # 30% increase for poor health
    else:
        raise ValueError("Invalid health status. Please enter 'good', 'average', or 'poor'.")

    # Calculate the total premium adjustment
    total_adjustment = age_adjustment + health_adjustment

    # Calculate the final premium amount
    premium = base_premium * (1 + total_adjustment)

    return premium

def main():
    # Example usage:
    age = 35
    health_status = 'average'
    premium = calculate_premium(age, health_status)
    print(f"The insurance premium for a {age} year old with {health_status} health is: ${premium:.2f}")

if __name__ == "__main__":
    main()
```
