## Exercise
Write a program to solve the fractional knapsack problem

## Reference Solution
```python
class Item:
    """Represents an item with a weight and a value."""

    def __init__(self, weight, value):
        self.weight = weight
        self.value = value
        self.ratio = value / weight

    def __repr__(self):
        return f"Item(weight={self.weight}, value={self.value}, ratio={self.ratio})"


def fractional_knapsack(items, capacity):
    """
    Solves the fractional knapsack problem.

    Args:
    items (list of Item): The items to consider.
    capacity (float): The capacity of the knapsack.

    Returns:
    float: The maximum total value.
    list of Item: The items to include in the knapsack.
    """
    # Sort the items by their value-to-weight ratio in descending order
    items.sort(key=lambda item: item.ratio, reverse=True)

    total_value = 0.0
    included_items = []

    for item in items:
        if capacity >= item.weight:
            # Include the entire item
            capacity -= item.weight
            total_value += item.value
            included_items.append((item, 1.0))  # (item, fraction)
        else:
            # Include a fraction of the item
            fraction = capacity / item.weight
            total_value += item.value * fraction
            included_items.append((item, fraction))
            break

    return total_value, included_items


# Example usage
if __name__ == "__main__":
    items = [
        Item(10, 60),
        Item(20, 100),
        Item(30, 120),
    ]
    capacity = 50.0

    max_value, included_items = fractional_knapsack(items, capacity)

    print("Maximum total value:", max_value)
    print("Included items:")
    for item, fraction in included_items:
        print(f"  - {item} (fraction: {fraction})")
```
