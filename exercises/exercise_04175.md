## Exercise
Create a program that generates a random word and allows the user to guess it by suggesting letters.

## Reference Solution
```python
import random

# List of words to choose from
word_list = ['apple', 'banana', 'cherry', 'date', 'elderberry']

def word_guessing_game():
    # Choose a random word from the list
    word = random.choice(word_list)
    word_length = len(word)
    
    # Initialize the guessed word with underscores
    guessed_word = ['_'] * word_length
    
    # Initialize the number of attempts
    attempts = 6
    
    print("Welcome to the Word Guessing Game!")
    print(f"You have {attempts} attempts to guess the word.")
    
    while attempts > 0 and '_' in guessed_word:
        # Print the current state of the guessed word
        print(' '.join(guessed_word))
        
        # Ask the user for a letter
        letter = input("Guess a letter: ").lower()
        
        # Check if the letter is in the word
        if letter in word:
            # Check if the letter is in the correct position
            for i, char in enumerate(word):
                if char == letter:
                    guessed_word[i] = letter
        else:
            # Decrement the number of attempts
            attempts -= 1
            print(f"Incorrect! You have {attempts} attempts left.")
    
    # Check if the user guessed the word
    if '_' not in guessed_word:
        print(' '.join(guessed_word))
        print("Congratulations! You guessed the word.")
    else:
        print(f"Sorry, you didn't guess the word. The word was {word}.")

if __name__ == "__main__":
    word_guessing_game()
```
