## Exercise
Write a program that checks if a given password meets certain criteria (e.g. length, characters, etc.)

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the following criteria:
    - Length: at least 8 characters
    - Characters: at least one uppercase letter, one lowercase letter, one digit, and one special character
    """
    # Define the regular expression pattern
    pattern = r"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*#?&])[A-Za-z\d@$!%*#?&]{8,}$"

    # Check if the password matches the pattern
    if re.match(pattern, password):
        return True
    else:
        return False

def get_password_errors(password):
    """
    Return a list of errors for the given password
    """
    errors = []
    if len(password) < 8:
        errors.append("Password must be at least 8 characters long")
    if not any(char.isupper() for char in password):
        errors.append("Password must contain at least one uppercase letter")
    if not any(char.islower() for char in password):
        errors.append("Password must contain at least one lowercase letter")
    if not any(char.isdigit() for char in password):
        errors.append("Password must contain at least one digit")
    if not any(char in "@$!%*#?&" for char in password):
        errors.append("Password must contain at least one special character")
    return errors

def main():
    password = input("Enter a password: ")
    if validate_password(password):
        print("Password is valid")
    else:
        print("Password is not valid")
        errors = get_password_errors(password)
        for error in errors:
            print(error)

if __name__ == "__main__":
    main()
```
