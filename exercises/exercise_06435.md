## Exercise
Create a program to manage a to-do list. Use a while loop to keep adding tasks and displaying the list until the user decides to quit.

## Reference Solution
```python
class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self):
        task = input("Enter a task: ")
        self.tasks.append(task)
        print(f"Task '{task}' added successfully!")

    def display_tasks(self):
        if not self.tasks:
            print("No tasks in the list.")
        else:
            print("To-Do List:")
            for i, task in enumerate(self.tasks, start=1):
                print(f"{i}. {task}")

    def run(self):
        while True:
            print("\nOptions:")
            print("1. Add Task")
            print("2. Display Tasks")
            print("3. Quit")
            choice = input("Choose an option: ")

            if choice == "1":
                self.add_task()
            elif choice == "2":
                self.display_tasks()
            elif choice == "3":
                print("Goodbye!")
                break
            else:
                print("Invalid option. Please choose again.")


if __name__ == "__main__":
    todo_list = ToDoList()
    todo_list.run()
```
