## Exercise
Implement a function to sort an array using the insertion sort algorithm

## Reference Solution
```python
def insertion_sort(arr):
    """
    Sorts an array using the insertion sort algorithm.

    Args:
        arr (list): The input array to be sorted.

    Returns:
        list: The sorted array.
    """
    # Iterate over the array starting from the second element (index 1)
    for i in range(1, len(arr)):
        # Store the current element to be inserted
        current_element = arr[i]
        # Initialize the index of the previous element
        j = i - 1
        
        # Shift elements in the sorted region to make room for the current element
        while j >= 0 and arr[j] > current_element:
            arr[j + 1] = arr[j]
            j -= 1
        
        # Insert the current element into the sorted region
        arr[j + 1] = current_element
    
    return arr

# Example usage:
if __name__ == "__main__":
    arr = [64, 34, 25, 12, 22, 11, 90]
    print("Original array:", arr)
    sorted_arr = insertion_sort(arr)
    print("Sorted array:", sorted_arr)
```
