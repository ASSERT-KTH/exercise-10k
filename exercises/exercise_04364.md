## Exercise
Write a program that adds a new key-value pair to a dictionary.

## Reference Solution
```python
def add_key_value_pair(dictionary, key, value):
    """
    Adds a new key-value pair to a dictionary.

    Args:
        dictionary (dict): The dictionary to add the key-value pair to.
        key (any): The key to add to the dictionary.
        value (any): The value associated with the key.

    Returns:
        dict: The dictionary with the new key-value pair.
    """
    dictionary[key] = value
    return dictionary


def main():
    # Create a dictionary
    my_dict = {"name": "John", "age": 30}

    # Print the original dictionary
    print("Original dictionary:")
    print(my_dict)

    # Add a new key-value pair
    new_key = "city"
    new_value = "New York"
    updated_dict = add_key_value_pair(my_dict, new_key, new_value)

    # Print the updated dictionary
    print("\nUpdated dictionary:")
    print(updated_dict)


if __name__ == "__main__":
    main()
```
