## Exercise
Write a program to divide two numbers and print the quotient and remainder

## Reference Solution
```python
def divide_numbers(dividend: int, divisor: int) -> None:
    """
    Divide two numbers and print the quotient and remainder.
    
    Args:
    dividend (int): The number being divided.
    divisor (int): The number by which we are dividing.
    """
    if divisor == 0:
        print("Error: Division by zero is not allowed.")
        return
    
    quotient = dividend // divisor
    remainder = dividend % divisor
    
    print(f"Quotient: {quotient}")
    print(f"Remainder: {remainder}")


def main() -> None:
    # Example usage
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))
    
    divide_numbers(dividend, divisor)


if __name__ == "__main__":
    main()
```
