## Exercise
Write a program that calculates the permutation of a given number of objects taken a given number at a time.

## Reference Solution
```python
import math

def calculate_permutation(n, r):
    """
    Calculate the permutation of a given number of objects taken a given number at a time.

    Args:
        n (int): The total number of objects.
        r (int): The number of objects taken at a time.

    Returns:
        int: The permutation of n objects taken r at a time.
    """
    if r > n:
        raise ValueError("r cannot be greater than n")
    permutation = math.factorial(n) // math.factorial(n - r)
    return permutation

def main():
    # Example usage
    n = int(input("Enter the total number of objects: "))
    r = int(input("Enter the number of objects taken at a time: "))
    try:
        permutation = calculate_permutation(n, r)
        print(f"The permutation of {n} objects taken {r} at a time is: {permutation}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
