## Exercise
Write a program to convert a string to an integer and vice versa.

## Reference Solution
```python
def string_to_int(s):
    """Convert a string to an integer."""
    try:
        return int(s)
    except ValueError:
        return "Invalid input. Please enter a valid integer string."

def int_to_string(n):
    """Convert an integer to a string."""
    return str(n)

def main():
    while True:
        print("\nOptions:")
        print("1. Convert string to integer")
        print("2. Convert integer to string")
        print("3. Quit")
        
        choice = input("Choose an option: ")

        if choice == "1":
            s = input("Enter a string: ")
            result = string_to_int(s)
            print(f"Result: {result}")
        elif choice == "2":
            try:
                n = int(input("Enter an integer: "))
                result = int_to_string(n)
                print(f"Result: {result}")
            except ValueError:
                print("Invalid input. Please enter a valid integer.")
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
