## Exercise
Write a program that asks the user for a number and then prints out the number in words.

## Reference Solution
```python
class NumberToWords:
    def __init__(self):
        self.ones = ['', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine']
        self.teens = ['ten', 'eleven', 'twelve', 'thirteen', 'fourteen', 'fifteen', 'sixteen', 'seventeen', 'eighteen', 'nineteen']
        self.tens = ['', '', 'twenty', 'thirty', 'forty', 'fifty', 'sixty', 'seventy', 'eighty', 'ninety']
        self.thousands = ['', 'thousand', 'million', 'billion']

    def convert(self, num):
        if num == 0:
            return 'zero'
        result = ''
        i = 0
        while num > 0:
            if num % 1000 != 0:
                result = self.helper(num % 1000) + ' ' + self.thousands[i] + ' ' + result
            num //= 1000
            i += 1
        return result.strip()

    def helper(self, num):
        if num == 0:
            return ''
        elif num < 10:
            return self.ones[num]
        elif num < 20:
            return self.teens[num - 10]
        elif num < 100:
            return self.tens[num // 10] + ('' if num % 10 == 0 else ' ' + self.ones[num % 10])
        else:
            return self.ones[num // 100] + ' hundred' + ('' if num % 100 == 0 else ' and ' + self.helper(num % 100))


def main():
    converter = NumberToWords()
    num = int(input("Enter a number: "))
    print(converter.convert(num))


if __name__ == "__main__":
    main()
```
