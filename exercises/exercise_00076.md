## Exercise
Write a function to remove consecutive duplicate characters from a given string

## Reference Solution
```python
def remove_consecutive_duplicates(input_string: str) -> str:
    """
    Remove consecutive duplicate characters from a given string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The string with consecutive duplicate characters removed.
    """
    if not input_string:  # Check if the input string is empty
        return ""

    result = [input_string[0]]  # Initialize the result with the first character
    for char in input_string[1:]:  # Iterate over the rest of the input string
        if char != result[-1]:  # Check if the current character is different from the last character in the result
            result.append(char)  # Add the current character to the result

    return "".join(result)  # Join the result list into a string and return it


# Example usage
if __name__ == "__main__":
    input_string = "aaabbbcccdddeee"
    print("Input string:", input_string)
    print("Result string:", remove_consecutive_duplicates(input_string))
```
