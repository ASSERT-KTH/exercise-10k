## Exercise
Write a program that interacts with the user through a terminal

## Reference Solution
```python
def get_user_name():
    """Ask the user for their name and greet them."""
    name = input("What is your name? ")
    print(f"Hello, {name}!")

def get_user_age():
    """Ask the user for their age and tell them how old they will be in a year."""
    while True:
        try:
            age = int(input("How old are you? "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    print(f"In a year, you will be {age + 1} years old.")

def display_menu():
    """Display a menu for the user to choose from."""
    print("\nMenu:")
    print("1. Get your name")
    print("2. Get your age")
    print("3. Exit")

def main():
    while True:
        display_menu()
        choice = input("What would you like to do? ")

        if choice == "1":
            get_user_name()
        elif choice == "2":
            get_user_age()
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
