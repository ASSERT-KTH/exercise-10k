## Exercise
Demonstrate the concept of variable scope by declaring variables in different scopes

## Reference Solution
```python
# Global variable
global_variable = "I am a global variable"

def print_global_variable():
    print(global_variable)

print_global_variable()  # Outputs: I am a global variable
```

```python
def print_local_variable():
    # Local variable
    local_variable = "I am a local variable"
    print(local_variable)

print_local_variable()  # Outputs: I am a local variable

# Trying to access local_variable outside its scope will raise an error
try:
    print(local_variable)
except NameError:
    print("Error: local_variable is not defined")
```

```python
def outer_function():
    # Enclosing variable
    enclosing_variable = "I am an enclosing variable"

    def inner_function():
        print(enclosing_variable)

    inner_function()

outer_function()  # Outputs: I am an enclosing variable
```

```python
# Built-in function
print(len("Hello, World!"))  # Outputs: 13

# Built-in variable
import math
print(math.pi)  # Outputs: 3.14159265359
```

```python
x = 10  # Global variable

def outer_function():
    x = 20  # Enclosing variable

    def inner_function():
        x = 30  # Local variable
        print("Local:", x)

    inner_function()
    print("Enclosing:", x)

outer_function()
print("Global:", x)
```
