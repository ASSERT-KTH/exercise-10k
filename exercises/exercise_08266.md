## Exercise
Develop a program that validates an email address using if/else statements.

## Reference Solution
```python
def validate_email(email):
    """
    Validates an email address.

    Args:
        email (str): The email address to be validated.

    Returns:
        bool: True if the email is valid, False otherwise.
    """

    # Check if the email contains the "@" symbol
    if "@" not in email:
        return False

    # Split the email into local part and domain
    local_part, domain = email.split("@")

    # Check if the local part is not empty
    if not local_part:
        return False

    # Check if the domain contains at least one "."
    if "." not in domain:
        return False

    # Check if the domain ends with a valid top-level domain (TLD)
    valid_tlds = [".com", ".org", ".net", ".edu", ".gov"]
    if not any(domain.endswith(tld) for tld in valid_tlds):
        return False

    # If all checks pass, the email is valid
    return True


def main():
    email = input("Enter an email address: ")
    if validate_email(email):
        print("The email address is valid.")
    else:
        print("The email address is invalid.")


if __name__ == "__main__":
    main()
```
