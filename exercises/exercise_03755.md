## Exercise
Write a program to generate a Caesar cipher for a given string and shift value using a basic algorithm

## Reference Solution

def caesar_cipher(text, shift):
    """
    Generates a Caesar cipher for a given string and shift value.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The number of positions to shift the alphabet.

    Returns:
        str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        # Check if character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift value
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        # If character is not a letter, leave it as is
        else:
            result += char

    return result


def main():
    text = input("Enter the text to be encrypted: ")
    shift = int(input("Enter the shift value: "))

    encrypted_text = caesar_cipher(text, shift)
    print("Encrypted text: ", encrypted_text)


if __name__ == "__main__":
    main()
