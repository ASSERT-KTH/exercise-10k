## Exercise
Write a program that takes a student's grades as input and calculates their average grade.

## Reference Solution
```python
def calculate_average_grade():
    # Get the number of grades from the user
    num_grades = int(input("Enter the number of grades: "))

    # Initialize a list to store the grades
    grades = []

    # Get each grade from the user and add it to the list
    for i in range(num_grades):
        while True:
            try:
                grade = float(input(f"Enter grade {i+1}: "))
                if 0 <= grade <= 100:
                    grades.append(grade)
                    break
                else:
                    print("Grade must be between 0 and 100.")
            except ValueError:
                print("Invalid input. Please enter a number.")

    # Calculate the average grade
    average_grade = sum(grades) / num_grades

    # Print the average grade
    print(f"Your average grade is: {average_grade:.2f}")

calculate_average_grade()
```
