## Exercise
Ask the user for a password and check if it meets certain requirements like length and character types.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the following rules:
    - Minimum length of 8 characters
    - At least one lowercase letter
    - At least one uppercase letter
    - At least one digit
    - At least one special character
    """
    errors = []

    # Check length
    if len(password) < 8:
        errors.append("Password must be at least 8 characters long.")

    # Check for lowercase letter
    if not re.search("[a-z]", password):
        errors.append("Password must contain at least one lowercase letter.")

    # Check for uppercase letter
    if not re.search("[A-Z]", password):
        errors.append("Password must contain at least one uppercase letter.")

    # Check for digit
    if not re.search("[0-9]", password):
        errors.append("Password must contain at least one digit.")

    # Check for special character
    if not re.search("[^A-Za-z0-9]", password):
        errors.append("Password must contain at least one special character.")

    return errors

def main():
    password = input("Enter a password: ")
    errors = validate_password(password)

    if errors:
        print("Password is invalid. Please fix the following errors:")
        for error in errors:
            print(f"- {error}")
    else:
        print("Password is valid.")

if __name__ == "__main__":
    main()
```
