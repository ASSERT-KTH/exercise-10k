## Exercise
Write a program that converts a time from one unit to another (e.g. seconds to minutes, hours to days).

## Reference Solution
```python
# Time conversion program

def seconds_to_minutes(seconds):
    """Converts seconds to minutes"""
    return seconds / 60

def seconds_to_hours(seconds):
    """Converts seconds to hours"""
    return seconds / 3600

def seconds_to_days(seconds):
    """Converts seconds to days"""
    return seconds / 86400

def minutes_to_seconds(minutes):
    """Converts minutes to seconds"""
    return minutes * 60

def minutes_to_hours(minutes):
    """Converts minutes to hours"""
    return minutes / 60

def minutes_to_days(minutes):
    """Converts minutes to days"""
    return minutes / 1440

def hours_to_seconds(hours):
    """Converts hours to seconds"""
    return hours * 3600

def hours_to_minutes(hours):
    """Converts hours to minutes"""
    return hours * 60

def hours_to_days(hours):
    """Converts hours to days"""
    return hours / 24

def days_to_seconds(days):
    """Converts days to seconds"""
    return days * 86400

def days_to_minutes(days):
    """Converts days to minutes"""
    return days * 1440

def days_to_hours(days):
    """Converts days to hours"""
    return days * 24

def main():
    print("Time Conversion Program")
    print("1. Convert seconds to minutes, hours, or days")
    print("2. Convert minutes to seconds, hours, or days")
    print("3. Convert hours to seconds, minutes, or days")
    print("4. Convert days to seconds, minutes, or hours")

    choice = int(input("Enter your choice: "))

    if choice == 1:
        seconds = float(input("Enter the time in seconds: "))
        print(f"{seconds} seconds is equal to {seconds_to_minutes(seconds)} minutes, {seconds_to_hours(seconds)} hours, or {seconds_to_days(seconds)} days")
    elif choice == 2:
        minutes = float(input("Enter the time in minutes: "))
        print(f"{minutes} minutes is equal to {minutes_to_seconds(minutes)} seconds, {minutes_to_hours(minutes)} hours, or {minutes_to_days(minutes)} days")
    elif choice == 3:
        hours = float(input("Enter the time in hours: "))
        print(f"{hours} hours is equal to {hours_to_seconds(hours)} seconds, {hours_to_minutes(hours)} minutes, or {hours_to_days(hours)} days")
    elif choice == 4:
        days = float(input("Enter the time in days: "))
        print(f"{days} days is equal to {days_to_seconds(days)} seconds, {days_to_minutes(days)} minutes, or {days_to_hours(days)} hours")
    else:
        print("Invalid choice")

if __name__ == "__main__":
    main()
```
