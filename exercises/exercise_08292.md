## Exercise
Create a program that calculates the points of intersection between two circles.

## Reference Solution
```python
import math

def circle_intersection(circle1, circle2):
    """
    Calculate the points of intersection between two circles.

    Args:
        circle1 (tuple): (x, y, r) of the first circle
        circle2 (tuple): (x, y, r) of the second circle

    Returns:
        list: Points of intersection, or empty list if no intersection
    """
    x1, y1, r1 = circle1
    x2, y2, r2 = circle2

    # Calculate the distance between the centers of the circles
    dx = x2 - x1
    dy = y2 - y1
    d = math.sqrt(dx**2 + dy**2)

    # Check if the circles are separate, tangent, or overlapping
    if d > r1 + r2:
        # No intersection
        return []
    elif d < abs(r1 - r2):
        # One circle is inside the other
        return []
    elif d == 0 and r1 == r2:
        # Circles are coincident
        return []

    # Calculate the points of intersection
    a = (r1**2 - r2**2 + d**2) / (2 * d)
    h = math.sqrt(r1**2 - a**2)
    xm = x1 + a * dx / d
    ym = y1 + a * dy / d
    xs1 = xm + h * dy / d
    xs2 = xm - h * dy / d
    ys1 = ym - h * dx / d
    ys2 = ym + h * dx / d

    # Return the points of intersection
    if d == r1 + r2:
        # Tangent circles
        return [(xs1, ys1)]
    else:
        # Overlapping circles
        return [(xs1, ys1), (xs2, ys2)]

# Example usage
circle1 = (0, 0, 5)
circle2 = (3, 4, 4)
print(circle_intersection(circle1, circle2))
```
