## Exercise
Write a program to implement a Caesar cipher to encrypt and decrypt messages.

## Reference Solution
```python
def caesar_encrypt(plaintext, shift):
    """
    Encrypts a message using a Caesar cipher.

    Args:
        plaintext (str): The message to be encrypted.
        shift (int): The number of positions to shift each character.

    Returns:
        str: The encrypted message.
    """
    ciphertext = ""
    for char in plaintext:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            ciphertext += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            ciphertext += char
    return ciphertext

def caesar_decrypt(ciphertext, shift):
    """
    Decrypts a message using a Caesar cipher.

    Args:
        ciphertext (str): The message to be decrypted.
        shift (int): The number of positions to shift each character.

    Returns:
        str: The decrypted message.
    """
    return caesar_encrypt(ciphertext, -shift)

def main():
    while True:
        print("1. Encrypt a message")
        print("2. Decrypt a message")
        print("3. Quit")
        choice = input("Choose an option: ")
        
        if choice == "1":
            plaintext = input("Enter a message to encrypt: ")
            shift = int(input("Enter a shift value: "))
            ciphertext = caesar_encrypt(plaintext, shift)
            print(f"Encrypted message: {ciphertext}")
        elif choice == "2":
            ciphertext = input("Enter a message to decrypt: ")
            shift = int(input("Enter a shift value: "))
            plaintext = caesar_decrypt(ciphertext, shift)
            print(f"Decrypted message: {plaintext}")
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
