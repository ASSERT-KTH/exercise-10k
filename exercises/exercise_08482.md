## Exercise
Write a program that takes an array of integers as input, and outputs the sum of all elements, the average of all elements, and the maximum element in the array.

## Reference Solution
```python
#include <iostream>
#include <climits>

using namespace std;

int main() {
    int n, max, sum = 0, average;
    cin >> n;

    int input;
    max = INT_MIN;
    for (int i = 0; i < n; i++) {
        cin >> input;
        sum += input;
        if (input > max) {
            max = input;
        }
    }

    average = sum / n;

    cout << "Sum: " << sum << endl;
    cout << "Average: " << average << endl;
    cout << "Maximum: " << max << endl;

    return 0;
}
```
