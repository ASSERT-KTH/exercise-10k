## Exercise
Write a program to calculate the GCD of two given numbers

## Reference Solution
```python
import math

# Function to calculate GCD using Euclidean Algorithm
def calculate_gcd(a, b):
    """
    Calculate the Greatest Common Divisor (GCD) of two numbers.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The GCD of a and b.
    """
    while b != 0:
        a, b = b, a % b
    return abs(a)

# Function to calculate GCD using math.gcd
def calculate_gcd_math(a, b):
    """
    Calculate the Greatest Common Divisor (GCD) of two numbers using math.gcd.

    Args:
        a (int): The first number.
        b (int): The second number.

    Returns:
        int: The GCD of a and b.
    """
    return math.gcd(a, b)

# Main function
def main():
    # Test the functions
    num1 = 48
    num2 = 18

    gcd_euclidean = calculate_gcd(num1, num2)
    gcd_math = calculate_gcd_math(num1, num2)

    print(f"GCD of {num1} and {num2} using Euclidean Algorithm: {gcd_euclidean}")
    print(f"GCD of {num1} and {num2} using math.gcd: {gcd_math}")

if __name__ == "__main__":
    main()
```
