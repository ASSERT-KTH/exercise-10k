## Exercise
Write a program that takes a string as input and prints out the string split into substrings based on a delimiter.

## Reference Solution
```python
#include <iostream>
#include <string>
#include <vector>

int main() {
    std::string str;
    std::cin >> str;

    std::vector<std::string> res;
    std::string tmp;
    for (auto i = str.begin(); i != str.end(); i++) {
        if (*i == ',') {
            res.push_back(tmp);
            tmp = "";
        } else {
            tmp += *i;
        }
    }

    res.push_back(tmp);
    for (auto i = res.begin(); i != res.end(); i++) {
        std::cout << *i << " ";
    }

    std::cout << std::endl;

    return 0;
}
```
