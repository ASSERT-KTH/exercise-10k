## Exercise
Create a program to sort a given dictionary by its keys or values.

## Reference Solution
```python
def sort_dict(input_dict, sort_by='keys', reverse=False):
    """
    Sorts a dictionary by its keys or values.

    Args:
        input_dict (dict): The dictionary to be sorted.
        sort_by (str, optional): Whether to sort by 'keys' or 'values'. Defaults to 'keys'.
        reverse (bool, optional): Whether to sort in reverse order. Defaults to False.

    Returns:
        dict: The sorted dictionary.
    """

    if sort_by == 'keys':
        # Sort by keys
        sorted_dict = dict(sorted(input_dict.items()))
    elif sort_by == 'values':
        # Sort by values
        sorted_dict = dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))
    else:
        raise ValueError("Invalid sort_by parameter. It should be either 'keys' or 'values'.")

    return sorted_dict


def main():
    # Example usage
    input_dict = {'apple': 5, 'banana': 2, 'cherry': 8, 'date': 1}

    # Sort by keys
    sorted_dict_keys = sort_dict(input_dict, sort_by='keys')
    print("Sorted by keys:", sorted_dict_keys)

    # Sort by values
    sorted_dict_values = sort_dict(input_dict, sort_by='values')
    print("Sorted by values:", sorted_dict_values)

    # Sort by values in reverse order
    sorted_dict_values_reverse = sort_dict(input_dict, sort_by='values', reverse=True)
    print("Sorted by values in reverse order:", sorted_dict_values_reverse)


if __name__ == "__main__":
    main()
```
