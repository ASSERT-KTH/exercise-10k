## Exercise
Write a program to estimate the cost of materials for a construction project given the quantity and price of each material.

## Reference Solution
```python
class Material:
    def __init__(self, name, quantity, price):
        self.name = name
        self.quantity = quantity
        self.price = price

    def calculate_cost(self):
        return self.quantity * self.price


class ConstructionProject:
    def __init__(self):
        self.materials = []

    def add_material(self, material):
        self.materials.append(material)

    def calculate_total_cost(self):
        total_cost = 0
        for material in self.materials:
            total_cost += material.calculate_cost()
        return total_cost

    def display_materials(self):
        for i, material in enumerate(self.materials, start=1):
            print(f"{i}. {material.name}: {material.quantity} units @ ${material.price:.2f} = ${material.calculate_cost():.2f}")


def main():
    project = ConstructionProject()

    while True:
        print("\n1. Add Material")
        print("2. Calculate Total Cost")
        print("3. Display Materials")
        print("4. Quit")
        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter material name: ")
            quantity = float(input("Enter material quantity: "))
            price = float(input("Enter material price per unit: "))
            project.add_material(Material(name, quantity, price))
        elif choice == "2":
            print(f"\nTotal Cost: ${project.calculate_total_cost():.2f}")
        elif choice == "3":
            print("\nMaterials:")
            project.display_materials()
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
