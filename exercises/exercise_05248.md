## Exercise
Write a program to calculate the tip for a given total bill and tip percentage.

## Reference Solution
```python
def calculate_tip(total_bill, tip_percentage):
    """
    Calculate the tip for a given total bill and tip percentage.

    Args:
        total_bill (float): The total bill amount.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The calculated tip amount.
    """
    tip_amount = (tip_percentage / 100) * total_bill
    return tip_amount

def main():
    # Get user input
    total_bill = float(input("Enter the total bill amount: $"))
    tip_percentage = float(input("Enter the tip percentage (%): "))

    # Calculate the tip
    tip_amount = calculate_tip(total_bill, tip_percentage)

    # Calculate the total amount to pay
    total_amount = total_bill + tip_amount

    # Print the results
    print(f"Total Bill: ${total_bill:.2f}")
    print(f"Tip ({tip_percentage}%): ${tip_amount:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
