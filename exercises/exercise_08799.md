## Exercise
Write a program that takes two numbers as input and returns the result of the exponentiation.

## Reference Solution
```python
# Function to calculate the exponentiation of two numbers
def calculate_exponentiation(base, exponent):
    """
    This function calculates the exponentiation of two numbers.

    Args:
        base (float): The base number.
        exponent (float): The exponent to which the base is raised.

    Returns:
        float: The result of the exponentiation.
    """
    try:
        # Calculate the exponentiation
        result = base ** exponent
        return result
    except TypeError:
        # Handle the case when the input is not a number
        return "Error: Both inputs must be numbers."
    except Exception as e:
        # Handle any other exception
        return f"An error occurred: {str(e)}"


# Main function
def main():
    # Get the base and exponent from the user
    base = float(input("Enter the base number: "))
    exponent = float(input("Enter the exponent: "))

    # Calculate the exponentiation
    result = calculate_exponentiation(base, exponent)

    # Print the result
    print(f"The result of {base} raised to the power of {exponent} is: {result}")


if __name__ == "__main__":
    main()
```
