## Exercise
Write a program that searches for a given substring in a text using for loops.

## Reference Solution
```python
def find_substring(text, substring):
    """
    Searches for a given substring in a text and returns the indices of all occurrences.

    Args:
        text (str): The text to search in.
        substring (str): The substring to search for.

    Returns:
        list: A list of indices where the substring was found.
    """
    indices = []
    for i in range(len(text)):
        match = True
        for j in range(len(substring)):
            if i + j >= len(text) or text[i + j] != substring[j]:
                match = False
                break
        if match:
            indices.append(i)
    return indices


def main():
    text = "Hello, world! Hello again!"
    substring = "Hello"
    indices = find_substring(text, substring)

    if indices:
        print(f"The substring '{substring}' was found at indices: {indices}")
    else:
        print(f"The substring '{substring}' was not found in the text.")


if __name__ == "__main__":
    main()
```
