## Exercise
Write a program to calculate the maximum and minimum of a set of numbers.

## Reference Solution
```python
def calculate_max_min(numbers):
    """
    Calculate the maximum and minimum values from a set of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        tuple: A tuple containing the maximum and minimum values.
    """
    if not numbers:
        raise ValueError("Input list is empty")

    max_value = max(numbers)
    min_value = min(numbers)

    return max_value, min_value

def main():
    # Get user input
    user_input = input("Enter a set of numbers separated by spaces: ")
    numbers = [float(num) for num in user_input.split()]

    # Calculate maximum and minimum values
    max_value, min_value = calculate_max_min(numbers)

    # Print results
    print(f"Maximum value: {max_value}")
    print(f"Minimum value: {min_value}")

if __name__ == "__main__":
    main()
```
