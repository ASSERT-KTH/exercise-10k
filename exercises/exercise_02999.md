## Exercise
Write a program that checks if a given string contains at least one digit.

## Reference Solution
```python
def contains_digit(input_string):
    """
    Checks if a given string contains at least one digit.

    Args:
        input_string (str): The input string to check.

    Returns:
        bool: True if the string contains at least one digit, False otherwise.
    """
    return any(char.isdigit() for char in input_string)

def main():
    # Example usage:
    input_strings = ["HelloWorld", "Hello123", "abcdefg"]

    for input_string in input_strings:
        if contains_digit(input_string):
            print(f"'{input_string}' contains at least one digit.")
        else:
            print(f"'{input_string}' does not contain any digits.")

if __name__ == "__main__":
    main()
```
