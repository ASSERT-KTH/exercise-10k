## Exercise
Write a function that determines the winner of a Rock, Paper, Scissors game.

## Reference Solution

def determine_winner(player1_choice, player2_choice):
    """
    Determine the winner of a Rock, Paper, Scissors game.

    Args:
        player1_choice (str): The choice of player 1. Can be "rock", "paper", or "scissors".
        player2_choice (str): The choice of player 2. Can be "rock", "paper", or "scissors".

    Returns:
        str: The winner of the game. Can be "player 1", "player 2", or "tie".
    """

    # Define the winning conditions
    winning_conditions = {
        "rock": "scissors",
        "paper": "rock",
        "scissors": "paper"
    }

    # Check if the inputs are valid
    if player1_choice not in winning_conditions or player2_choice not in winning_conditions:
        raise ValueError("Invalid input. Please enter 'rock', 'paper', or 'scissors'.")

    # Check if it's a tie
    if player1_choice == player2_choice:
        return "tie"

    # Check if player 1 wins
    if winning_conditions[player1_choice] == player2_choice:
        return "player 1"

    # If none of the above conditions are met, player 2 wins
    return "player 2"


# Example usage
if __name__ == "__main__":
    print(determine_winner("rock", "scissors"))  # Should print "player 1"
    print(determine_winner("paper", "rock"))  # Should print "player 1"
    print(determine_winner("scissors", "paper"))  # Should print "player 1"
    print(determine_winner("rock", "rock"))  # Should print "tie"
    print(determine_winner("scissors", "rock"))  # Should print "player 2"
