## Exercise
Write a program that converts a string to an integer and then to a float. The input string will be a numeric value.

## Reference Solution
```python
def convert_string_to_float(input_string):
    """
    Converts a numeric string to an integer and then to a float.

    Args:
        input_string (str): A string representing a numeric value.

    Returns:
        float: The input string converted to an integer and then to a float.
    """

    # First, we convert the string to an integer
    try:
        integer_value = int(input_string)
    except ValueError:
        print("Error: The input string is not a valid numeric value.")
        return None

    # Then, we convert the integer to a float
    float_value = float(integer_value)

    return float_value


def main():
    # Example usage:
    input_string = "123"
    result = convert_string_to_float(input_string)

    if result is not None:
        print(f"The string '{input_string}' converted to an integer and then to a float is: {result}")


if __name__ == "__main__":
    main()
```
