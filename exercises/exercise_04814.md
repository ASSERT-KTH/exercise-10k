## Exercise
Given a linked list of integers, sort the linked list in ascending order. The function should take in a linked list of integers as input and return the sorted linked list.

class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

def sortList(head):
    if not head or not head.next:
        return head

    mid = getMiddle(head)
    midNext = mid.next

    mid.next = None

    left = sortList(head)
    right = sortList(midNext)

    sortedList = sortedList(left, right)
    return sortedList

def getMiddle(head):
    slow = head
    fast = head

    while fast.next and fast.next.next:
        slow = slow.next
        fast = fast.next.next

    return slow

def sortedList(l1, l2):
    dummyRoot = ListNode(0)
    ptr = dummyRoot

    while l1 and l2:
        if l1.val < l2.val:
            ptr.next = l1
            l1 = l1.next
        else:
            ptr.next = l2
            l2 = l2.next
        ptr = ptr.next

    ptr.next = l1 if l1 else l2
    return dummyRoot.next
