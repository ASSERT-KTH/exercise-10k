## Exercise
Write a program that takes the radius of a circle as input and returns the area of the circle.

#include <iostream>

int main() {
    double radius;
    std::cout << "Enter the radius of a circle: ";
    std::cin >> radius;
    double area = 3.14 * radius * radius;
    std::cout << "The area of the circle is: " << area << std::endl;
    return 0;
}
