## Exercise
Write a program that checks the flight status using if/else statements

## Reference Solution
```python
def check_flight_status(flight_number):
    """
    Checks the flight status based on the flight number.

    Args:
        flight_number (str): The flight number.

    Returns:
        str: The flight status.
    """
    flight_status = {
        "AA101": "On Time",
        "AA102": "Delayed",
        "AA103": "Cancelled",
        "UA201": "On Time",
        "UA202": "Delayed",
        "UA203": "Cancelled"
    }

    if flight_number in flight_status:
        if flight_status[flight_number] == "On Time":
            return f"Flight {flight_number} is on time."
        elif flight_status[flight_number] == "Delayed":
            return f"Flight {flight_number} is delayed. Please check with the airline for the latest updates."
        else:
            return f"Flight {flight_number} is cancelled. Please contact the airline to rebook or refund your ticket."
    else:
        return f"Flight {flight_number} not found. Please check the flight number and try again."

def main():
    flight_number = input("Enter the flight number: ")
    print(check_flight_status(flight_number))

if __name__ == "__main__":
    main()
```
