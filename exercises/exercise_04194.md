## Exercise
Write a function that encrypts and decrypts a message using a simple substitution cipher.

## Reference Solution

def substitution_cipher(message, shift, mode):
    """
    Encrypts or decrypts a message using a simple substitution cipher.

    Args:
    - message (str): The message to be encrypted or decrypted.
    - shift (int): The shift value for the substitution cipher.
    - mode (str): 'encrypt' or 'decrypt' mode.

    Returns:
    - str: The encrypted or decrypted message.
    """
    alphabet = 'abcdefghijklmnopqrstuvwxyz'
    result = ''

    for char in message:
        if char.isalpha():
            index = alphabet.index(char.lower())
            if mode == 'encrypt':
                new_index = (index + shift) % 26
            elif mode == 'decrypt':
                new_index = (index - shift) % 26
            if char.isupper():
                result += alphabet[new_index].upper()
            else:
                result += alphabet[new_index]
        else:
            result += char

    return result

# Example usage:
if __name__ == "__main__":
    message = "Hello, World!"
    shift = 3

    encrypted_message = substitution_cipher(message, shift, 'encrypt')
    decrypted_message = substitution_cipher(encrypted_message, shift, 'decrypt')

    print(f"Original Message: {message}")
    print(f"Encrypted Message: {encrypted_message}")
    print(f"Decrypted Message: {decrypted_message}")
