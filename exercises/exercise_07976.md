## Exercise
Write a program that performs a binary search on a sorted list to find a target value.

## Reference Solution
```python
def binary_search(sorted_list, target):
    """
    Performs a binary search on a sorted list to find a target value.

    Args:
        sorted_list (list): A sorted list of values.
        target: The target value to search for.

    Returns:
        int: The index of the target value in the list if found, -1 otherwise.
    """
    low = 0
    high = len(sorted_list) - 1

    while low <= high:
        mid = (low + high) // 2
        if sorted_list[mid] == target:
            return mid
        elif sorted_list[mid] < target:
            low = mid + 1
        else:
            high = mid - 1

    return -1


def main():
    # Example usage
    sorted_list = [1, 3, 5, 7, 9, 11, 13, 15]
    target = 9

    result = binary_search(sorted_list, target)

    if result != -1:
        print(f"Target value {target} found at index {result}.")
    else:
        print(f"Target value {target} not found in the list.")


if __name__ == "__main__":
    main()
```
