## Exercise
Write a program that simulates a simple bank account. The program should take the account balance and a transaction (deposit or withdrawal) as input. Based on the transaction, update the account balance and print the new balance.

## Reference Solution
```python
class BankAccount:
    def __init__(self, balance=0):
        """
        Initialize the bank account with an initial balance.

        Args:
            balance (float, optional): The initial balance. Defaults to 0.
        """
        self.balance = balance

    def deposit(self, amount):
        """
        Deposit an amount into the bank account.

        Args:
            amount (float): The amount to deposit.

        Raises:
            ValueError: If the amount to deposit is negative.
        """
        if amount < 0:
            raise ValueError("Deposit amount cannot be negative.")
        self.balance += amount
        print(f"Deposited ${amount:.2f}. New balance: ${self.balance:.2f}")

    def withdraw(self, amount):
        """
        Withdraw an amount from the bank account.

        Args:
            amount (float): The amount to withdraw.

        Raises:
            ValueError: If the amount to withdraw is negative or exceeds the balance.
        """
        if amount < 0:
            raise ValueError("Withdrawal amount cannot be negative.")
        if amount > self.balance:
            raise ValueError("Insufficient funds.")
        self.balance -= amount
        print(f"Withdrew ${amount:.2f}. New balance: ${self.balance:.2f}")


def main():
    # Get the initial account balance from the user
    while True:
        try:
            balance = float(input("Enter the initial account balance: $"))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Create a BankAccount object with the initial balance
    account = BankAccount(balance)

    # Get the transaction type from the user
    while True:
        transaction = input("Enter the transaction type (deposit/withdrawal): ").lower()
        if transaction in ["deposit", "withdrawal"]:
            break
        print("Invalid input. Please enter 'deposit' or 'withdrawal'.")

    # Get the transaction amount from the user
    while True:
        try:
            amount = float(input("Enter the transaction amount: $"))
            if amount < 0:
                print("Invalid input. Please enter a non-negative number.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Perform the transaction
    if transaction == "deposit":
        account.deposit(amount)
    else:
        try:
            account.withdraw(amount)
        except ValueError as e:
            print(e)


if __name__ == "__main__":
    main()
```
