## Exercise
Write a program that performs arithmetic operations with complex numbers (addition, subtraction, multiplication, division).

class ComplexNumber:
    """Class representing a complex number."""

    def __init__(self, real=0, imaginary=0):
        """
        Initialize a complex number.

        Args:
            real (float, optional): Real part of the complex number. Defaults to 0.
            imaginary (float, optional): Imaginary part of the complex number. Defaults to 0.
        """
        self.real = real
        self.imaginary = imaginary

    def __str__(self):
        """Return a string representation of the complex number."""
        return f"{self.real} + {self.imaginary}i"

    def __add__(self, other):
        """
        Add two complex numbers.

        Args:
            other (ComplexNumber): Complex number to add.

        Returns:
            ComplexNumber: Sum of the two complex numbers.
        """
        real_part = self.real + other.real
        imaginary_part = self.imaginary + other.imaginary
        return ComplexNumber(real_part, imaginary_part)

    def __sub__(self, other):
        """
        Subtract one complex number from another.

        Args:
            other (ComplexNumber): Complex number to subtract.

        Returns:
            ComplexNumber: Difference of the two complex numbers.
        """
        real_part = self.real - other.real
        imaginary_part = self.imaginary - other.imaginary
        return ComplexNumber(real_part, imaginary_part)

    def __mul__(self, other):
        """
        Multiply two complex numbers.

        Args:
            other (ComplexNumber): Complex number to multiply.

        Returns:
            ComplexNumber: Product of the two complex numbers.
        """
        real_part = self.real * other.real - self.imaginary * other.imaginary
        imaginary_part = self.real * other.imaginary + self.imaginary * other.real
        return ComplexNumber(real_part, imaginary_part)

    def __truediv__(self, other):
        """
        Divide one complex number by another.

        Args:
            other (ComplexNumber): Complex number to divide by.

        Returns:
            ComplexNumber: Quotient of the two complex numbers.
        """
        denominator = other.real ** 2 + other.imaginary ** 2
        real_part = (self.real * other.real + self.imaginary * other.imaginary) / denominator
        imaginary_part = (self.imaginary * other.real - self.real * other.imaginary) / denominator
        return ComplexNumber(real_part, imaginary_part)


def main():
    # Example usage:
    num1 = ComplexNumber(3, 4)
    num2 = ComplexNumber(1, 2)

    print(f"Number 1: {num1}")
    print(f"Number 2: {num2}")

    print("\nArithmetic Operations:")
    print(f"Addition: {num1 + num2}")
    print(f"Subtraction: {num1 - num2}")
    print(f"Multiplication: {num1 * num2}")
    print(f"Division: {num1 / num2}")


if __name__ == "__main__":
    main()
