## Exercise
Write a program to find the median of all numbers in an array.

## Reference Solution
```python
def calculate_median(array):
    """
    Calculate the median of all numbers in an array.

    Args:
        array (list): A list of numbers.

    Returns:
        float: The median of the array.
    """
    # Sort the array in ascending order
    sorted_array = sorted(array)

    # Get the length of the array
    n = len(sorted_array)

    # If the array has an odd number of elements, return the middle value
    if n % 2 != 0:
        return sorted_array[n // 2]
    # If the array has an even number of elements, return the average of the two middle values
    else:
        mid1 = sorted_array[n // 2 - 1]
        mid2 = sorted_array[n // 2]
        return (mid1 + mid2) / 2


def main():
    # Example usage
    array = [1, 3, 5, 7, 9]
    median = calculate_median(array)
    print(f"The median of the array is: {median}")

    array = [1, 2, 3, 4, 5, 6]
    median = calculate_median(array)
    print(f"The median of the array is: {median}")


if __name__ == "__main__":
    main()
```
