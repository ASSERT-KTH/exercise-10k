## Exercise
Given two strings, check if the first string is a subsequence of the second string

## Reference Solution

```python
def is_subsequence(s1: str, s2: str) -> bool:
    """
    Checks if s1 is a subsequence of s2.

    Args:
        s1 (str): The potential subsequence.
        s2 (str): The string to check against.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """
    i = 0  # Pointer for s1
    for j in range(len(s2)):  # Iterate through s2
        if i < len(s1) and s1[i] == s2[j]:  # If current characters match
            i += 1  # Move to the next character in s1
    return i == len(s1)  # Return True if we've found all characters of s1

# Example usage
if __name__ == "__main__":
    print(is_subsequence("ace", "abcde"))  # True
    print(is_subsequence("aec", "abcde"))  # False
    print(is_subsequence("", "abcde"))  # True
    print(is_subsequence("abcde", ""))  # False
```
