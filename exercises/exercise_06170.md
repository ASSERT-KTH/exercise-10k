## Exercise
Write a function that takes another function as input and applies it to a given value.

## Reference Solution
```python
# Define a function that takes another function and a value as input
def apply_function(func, value):
    """
    Applies a given function to a given value.

    Args:
        func (function): The function to be applied.
        value: The value to which the function will be applied.

    Returns:
        The result of applying the function to the value.
    """

    # Check if the input is a function
    if not callable(func):
        raise TypeError("Input is not a function")

    # Apply the function to the value and return the result
    # We use try-except block to catch any potential exceptions that might occur during function application
    try:
        # Apply the function to the value
        result = func(value)
        
        # Return the result
        return result
    
    # Catch and re-raise any exceptions that occur during function application
    except Exception as e:
        raise Exception(f"An error occurred while applying the function: {str(e)}")


# Example usage:
# Define a simple function that squares a number
def square(x):
    return x ** 2

# Apply the square function to the value 5
result = apply_function(square, 5)

# Print the result
print(result)  # Output: 25
```
