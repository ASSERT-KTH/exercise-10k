## Exercise
Create a dictionary with 3 key-value pairs and pop a key-value pair.

## Reference Solution
```python
def create_and_pop_dict():
    # Create a dictionary with 3 key-value pairs
    my_dict = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }
    
    print("Original Dictionary:")
    print(my_dict)

    # Pop a key-value pair from the dictionary
    popped_key = "age"
    popped_value = my_dict.pop(popped_key)

    print("\nPopped Key-Value Pair:")
    print(f"Key: {popped_key}, Value: {popped_value}")

    print("\nDictionary after popping:")
    print(my_dict)

# Example usage
create_and_pop_dict()
```
