## Exercise
Create a program that uses a for loop to find the minimum number in a list of numbers.

## Reference Solution
```python
def find_minimum(numbers):
    """
    Find the minimum number in a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        int: The minimum number in the list.

    Raises:
        ValueError: If the list is empty.
    """
    if not numbers:
        raise ValueError("List cannot be empty")

    min_number = numbers[0]
    for num in numbers:
        if num < min_number:
            min_number = num

    return min_number

def main():
    # Example usage
    numbers = [10, 5, 8, 3, 9, 1]
    try:
        min_number = find_minimum(numbers)
        print(f"The minimum number in the list is: {min_number}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
