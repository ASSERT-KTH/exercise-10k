## Exercise
Develop a program that simulates a simple game of chance where the user can bet on the outcome of a random event (e.g., coin flip, roll of a die).

## Reference Solution
```python
import random

def coin_flip_game():
    """Simulates a coin flip game."""
    print("Welcome to the coin flip game!")
    bet = input("Do you want to bet on heads or tails? ")
    outcome = random.choice(["heads", "tails"])
    print(f"The coin landed on {outcome}.")
    if bet.lower() == outcome:
        print("You win!")
    else:
        print("You lose!")

def die_roll_game():
    """Simulates a die roll game."""
    print("Welcome to the die roll game!")
    bet = int(input("Do you want to bet on a number between 1 and 6? "))
    outcome = random.randint(1, 6)
    print(f"The die rolled a {outcome}.")
    if bet == outcome:
        print("You win!")
    else:
        print("You lose!")

def main():
    print("Game of Chance Simulator")
    print("1. Coin Flip Game")
    print("2. Die Roll Game")
    choice = input("Which game do you want to play? (1/2) ")
    if choice == "1":
        coin_flip_game()
    elif choice == "2":
        die_roll_game()
    else:
        print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
