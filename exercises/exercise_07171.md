## Exercise
Write a program that encrypts a string by shifting each character by a fixed amount

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Encrypts a string by shifting each character by a fixed amount.

    Args:
        text (str): The input string to be encrypted.
        shift (int): The number of positions to shift each character.

    Returns:
        str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - 65 + shift) % 26 + 65)
        # Check if character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character
            # and add the shift
            result += chr((ord(char) - 97 + shift) % 26 + 97)
        # If the character is not a letter, leave it unchanged
        else:
            result += char

    return result


def main():
    text = input("Enter the string to be encrypted: ")
    shift = int(input("Enter the shift value: "))

    encrypted_text = caesar_cipher(text, shift)
    print(f"Encrypted text: {encrypted_text}")


if __name__ == "__main__":
    main()
```
