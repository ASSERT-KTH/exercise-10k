## Exercise
Write a program that checks if a password meets certain criteria (length, uppercase, etc.) using a for loop.

## Reference Solution
```python
def validate_password(password):
    """
    Validate a password based on certain criteria.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """

    # Define the criteria
    criteria = [
        {"name": "Length", "func": lambda x: len(x) >= 8},
        {"name": "Uppercase", "func": lambda x: any(c.isupper() for c in x)},
        {"name": "Lowercase", "func": lambda x: any(c.islower() for c in x)},
        {"name": "Digits", "func": lambda x: any(c.isdigit() for c in x)},
        {"name": "Special characters", "func": lambda x: any(not c.isalnum() for c in x)},
    ]

    # Check each criterion
    for criterion in criteria:
        if not criterion["func"](password):
            print(f"Password does not meet {criterion['name']} criterion.")
            return False

    print("Password is valid.")
    return True


def main():
    password = input("Enter a password: ")
    validate_password(password)


if __name__ == "__main__":
    main()
```
