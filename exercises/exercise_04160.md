## Exercise
Write a function to reduce a dictionary to a single value.

## Reference Solution
```python
from functools import reduce

def reduce_dict(dictionary, reduction_func):
    """
    Reduces a dictionary to a single value using a given reduction function.

    Args:
        dictionary (dict): The dictionary to reduce.
        reduction_func (function): The function to use for reduction.

    Returns:
        The reduced value.
    """
    # Get the dictionary values
    values = dictionary.values()
    
    # Check if the dictionary is empty
    if not values:
        return None
    
    # Apply the reduction function to the dictionary values
    return reduce(reduction_func, values)

# Example usage
my_dict = {'a': 1, 'b': 2, 'c': 3, 'd': 4}
print(reduce_dict(my_dict, sum))  # Output: 10
print(reduce_dict(my_dict, max))  # Output: 4
print(reduce_dict(my_dict, min))  # Output: 1
```
