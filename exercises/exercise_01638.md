## Exercise
Develop a program that takes a list of numbers as input and returns the index of the maximum value.

## Reference Solution
```python
def find_max_index(num_list):
    """
    This function finds the index of the maximum value in a given list of numbers.

    Args:
        num_list (list): A list of numbers.

    Returns:
        int: The index of the maximum value in the list.

    Raises:
        ValueError: If the input list is empty.
    """

    # Check if the list is empty
    if not num_list:
        raise ValueError("Input list is empty")

    # Initialize the max_index and max_value with the first element of the list
    max_index = 0
    max_value = num_list[0]

    # Iterate over the list to find the maximum value and its index
    for i in range(1, len(num_list)):
        if num_list[i] > max_value:
            max_index = i
            max_value = num_list[i]

    return max_index


# Example usage
if __name__ == "__main__":
    numbers = [4, 2, 9, 6, 5, 1]
    max_index = find_max_index(numbers)
    print(f"The index of the maximum value is: {max_index}")
```
