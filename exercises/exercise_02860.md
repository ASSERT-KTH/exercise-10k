## Exercise
Write a function to remove and return the last inserted key-value pair from a dictionary.

## Reference Solution
```python
def remove_last_inserted_pair(dictionary):
    """
    Removes and returns the last inserted key-value pair from a dictionary.

    Args:
        dictionary (dict): The dictionary to remove the last inserted pair from.

    Returns:
        tuple: A tuple containing the key and value of the removed pair, or None if the dictionary is empty.
    """
    if not dictionary:
        return None

    # Since dictionaries in Python 3.7+ maintain insertion order, we can use this to our advantage
    last_key = list(dictionary.keys())[-1]
    last_value = dictionary.pop(last_key)

    return last_key, last_value


# Example usage:
if __name__ == "__main__":
    my_dict = {}
    my_dict['apple'] = 1
    my_dict['banana'] = 2
    my_dict['cherry'] = 3

    print("Original Dictionary:", my_dict)
    removed_pair = remove_last_inserted_pair(my_dict)
    print("Removed Pair:", removed_pair)
    print("Updated Dictionary:", my_dict)
```
