## Exercise
Write a program that finds all duplicate characters in a given string.

## Reference Solution
```python
def find_duplicate_chars(input_string):
    """
    Finds all duplicate characters in a given string.

    Args:
        input_string (str): The input string to search for duplicates.

    Returns:
        dict: A dictionary where the keys are the duplicate characters and the values are their frequencies.
    """
    char_frequency = {}
    duplicate_chars = {}

    # Count the frequency of each character
    for char in input_string:
        if char in char_frequency:
            char_frequency[char] += 1
        else:
            char_frequency[char] = 1

    # Find the duplicate characters
    for char, frequency in char_frequency.items():
        if frequency > 1:
            duplicate_chars[char] = frequency

    return duplicate_chars


def main():
    input_string = input("Enter a string: ")
    duplicate_chars = find_duplicate_chars(input_string)

    if duplicate_chars:
        print("Duplicate characters:")
        for char, frequency in duplicate_chars.items():
            print(f"{char}: {frequency}")
    else:
        print("No duplicate characters found.")


if __name__ == "__main__":
    main()
```
