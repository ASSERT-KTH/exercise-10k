## Exercise
Write a function to merge two dictionaries with common keys

## Reference Solution
```python
def merge_dictionaries(dict1, dict2):
    """
    Merge two dictionaries with common keys.
    
    For common keys, the values are combined into lists.
    
    Parameters:
    dict1 (dict): The first dictionary to merge.
    dict2 (dict): The second dictionary to merge.
    
    Returns:
    dict: The merged dictionary.
    """
    merged_dict = dict1.copy()  # Start with a copy of the first dictionary
    
    for key, value in dict2.items():
        if key in merged_dict:
            # If the key is already in the merged dictionary, combine the values
            if isinstance(merged_dict[key], list):
                # If the existing value is already a list, append to it
                merged_dict[key].append(value)
            else:
                # If the existing value is not a list, convert it to a list and append
                merged_dict[key] = [merged_dict[key], value]
        else:
            # If the key is not in the merged dictionary, add it
            merged_dict[key] = value
    
    return merged_dict

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 20, 'c': 30, 'd': 40}

merged = merge_dictionaries(dict1, dict2)
print(merged)
```
