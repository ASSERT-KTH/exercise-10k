## Exercise
Create a program to predict the outcome of a game based on the scores of the two teams using conditional statements

## Reference Solution
```python
def predict_game_outcome(team1_score, team2_score):
    """
    Predicts the outcome of a game based on the scores of the two teams.

    Args:
        team1_score (int): The score of the first team.
        team2_score (int): The score of the second team.

    Returns:
        str: The predicted outcome of the game.
    """
    if team1_score > team2_score:
        return "Team 1 wins"
    elif team2_score > team1_score:
        return "Team 2 wins"
    else:
        return "It's a tie"

def main():
    team1_score = int(input("Enter the score of Team 1: "))
    team2_score = int(input("Enter the score of Team 2: "))

    outcome = predict_game_outcome(team1_score, team2_score)
    print("The predicted outcome of the game is:", outcome)

if __name__ == "__main__":
    main()
```
