## Exercise
Write a program that takes a starting number and a sequence type as input (e.g., odd, even, prime). Use if/else statements to generate the sequence.

def generate_sequence(start, seq_type, length):
    """
    Generate a sequence based on the input type.

    Args:
        start (int): Starting number of the sequence.
        seq_type (str): Type of sequence to generate (e.g., odd, even, prime).
        length (int): Length of the sequence.

    Returns:
        list: Generated sequence.
    """

    # Initialize an empty list to store the sequence
    sequence = []

    # Initialize a counter variable to keep track of the current number
    num = start

    # Loop until the sequence reaches the desired length
    while len(sequence) < length:
        # Check the sequence type
        if seq_type == "odd":
            # Check if the current number is odd
            if num % 2 != 0:
                # Append the current number to the sequence
                sequence.append(num)
        elif seq_type == "even":
            # Check if the current number is even
            if num % 2 == 0:
                # Append the current number to the sequence
                sequence.append(num)
        elif seq_type == "prime":
            # Check if the current number is prime
            if is_prime(num):
                # Append the current number to the sequence
                sequence.append(num)
        else:
            # Handle invalid sequence type
            print("Invalid sequence type. Please enter 'odd', 'even', or 'prime'.")
            return []

        # Increment the counter variable
        num += 1

    # Return the generated sequence
    return sequence


def is_prime(n):
    """
    Check if a number is prime.

    Args:
        n (int): Number to check.

    Returns:
        bool: True if the number is prime, False otherwise.
    """

    # Handle edge cases
    if n <= 1:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False

    # Check for factors
    for i in range(3, int(n ** 0.5) + 1, 2):
        if n % i == 0:
            return False

    # If no factors are found, the number is prime
    return True


def main():
    # Get the starting number and sequence type from the user
    start = int(input("Enter the starting number: "))
    seq_type = input("Enter the sequence type (odd, even, prime): ")
    length = int(input("Enter the sequence length: "))

    # Generate the sequence
    sequence = generate_sequence(start, seq_type, length)

    # Print the sequence
    print("Generated Sequence:")
    print(sequence)


if __name__ == "__main__":
    main()
