## Exercise
Create a program that calculates the area of a trapezoid given the lengths of its two parallel sides and its height.

## Reference Solution
```python
def calculate_trapezoid_area(base1, base2, height):
    """
    Calculate the area of a trapezoid.

    Args:
        base1 (float): Length of the first parallel side.
        base2 (float): Length of the second parallel side.
        height (float): Height of the trapezoid.

    Returns:
        float: Area of the trapezoid.
    """
    if base1 <= 0 or base2 <= 0 or height <= 0:
        raise ValueError("All input values must be positive.")

    area = 0.5 * (base1 + base2) * height
    return area


def main():
    # Example usage:
    base1 = 5  # length of the first parallel side
    base2 = 10  # length of the second parallel side
    height = 6  # height of the trapezoid

    area = calculate_trapezoid_area(base1, base2, height)
    print(f"The area of the trapezoid is: {area:.2f}")


if __name__ == "__main__":
    main()
```
