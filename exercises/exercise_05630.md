## Exercise
Create a program to find the maximum sum of a subarray in a given list of integers.

## Reference Solution
```python
def max_subarray_sum(nums):
    """
    Find the maximum sum of a subarray in a given list of integers.

    Args:
    - nums (list): A list of integers.

    Returns:
    - max_sum (int): The maximum sum of a subarray.
    """
    if not nums:
        return 0

    max_sum = float('-inf')
    current_sum = 0

    for num in nums:
        current_sum = max(num, current_sum + num)
        max_sum = max(max_sum, current_sum)

    return max_sum

# Example usage
def main():
    nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
    max_sum = max_subarray_sum(nums)
    print("Maximum subarray sum:", max_sum)

if __name__ == "__main__":
    main()
```
