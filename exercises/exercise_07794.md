## Exercise
Write a program that asks the user for a number and prints out whether it's even or odd.

## Reference Solution
```python
def check_parity(num):
    """Checks if a number is even or odd."""
    if num % 2 == 0:
        return "even"
    else:
        return "odd"

def main():
    # Ask the user for a number
    while True:
        try:
            num = int(input("Enter a number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a valid integer.")

    # Check the parity of the number
    parity = check_parity(num)

    # Print the result
    print(f"The number {num} is {parity}.")

if __name__ == "__main__":
    main()
```
