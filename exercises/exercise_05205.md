## Exercise
Create a program that converts a measurement from one unit to another (e.g., inches to centimeters).

## Reference Solution
```python
def convert_units(value, from_unit, to_unit):
    """
    Converts a measurement from one unit to another.

    Args:
        value (float): The value to convert.
        from_unit (str): The unit to convert from.
        to_unit (str): The unit to convert to.

    Returns:
        float: The converted value.
    """

    # Define conversion factors
    conversions = {
        'inches': {'centimeters': 2.54, 'feet': 0.0833333, 'yards': 0.0277778, 'meters': 0.0254},
        'centimeters': {'inches': 0.393701, 'feet': 0.0328084, 'yards': 0.0109361, 'meters': 0.01},
        'feet': {'inches': 12, 'centimeters': 30.48, 'yards': 0.333333, 'meters': 0.3048},
        'yards': {'inches': 36, 'centimeters': 91.44, 'feet': 3, 'meters': 0.9144},
        'meters': {'inches': 39.3701, 'centimeters': 100, 'feet': 3.28084, 'yards': 1.09361}
    }

    # Check if from_unit and to_unit are valid
    if from_unit not in conversions or to_unit not in conversions:
        raise ValueError("Invalid unit")

    # Convert the value
    return value * conversions[from_unit][to_unit]


def main():
    print("Unit Conversion Program")
    print("------------------------")

    # Get user input
    value = float(input("Enter the value to convert: "))
    from_unit = input("Enter the unit to convert from (inches, centimeters, feet, yards, meters): ")
    to_unit = input("Enter the unit to convert to (inches, centimeters, feet, yards, meters): ")

    # Convert the value and print the result
    try:
        result = convert_units(value, from_unit, to_unit)
        print(f"{value} {from_unit} is equal to {result} {to_unit}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
