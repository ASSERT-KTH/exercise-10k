## Exercise
Create a program that checks if a ransom note can be constructed from a given magazine. The ransom note is constructed by cutting words from the magazine.

## Reference Solution
```python
from collections import Counter

def can_construct_ransom_note(magazine, ransom_note):
    """
    Checks if a ransom note can be constructed from a given magazine.

    Args:
        magazine (str): The text from which words can be cut.
        ransom_note (str): The text that needs to be constructed.

    Returns:
        bool: True if the ransom note can be constructed, False otherwise.
    """
    # Split the magazine and ransom note into words
    magazine_words = magazine.split()
    ransom_note_words = ransom_note.split()

    # Count the occurrences of each word in the magazine and ransom note
    magazine_word_counts = Counter(magazine_words)
    ransom_note_word_counts = Counter(ransom_note_words)

    # Check if each word in the ransom note can be found in the magazine
    for word, count in ransom_note_word_counts.items():
        if word not in magazine_word_counts or count > magazine_word_counts[word]:
            return False

    return True

# Example usage
if __name__ == "__main__":
    magazine = "give me one grand today night"
    ransom_note = "give one grand today"
    print(can_construct_ransom_note(magazine, ransom_note))  # Output: True

    magazine = "two times two is four"
    ransom_note = "two times two is four five"
    print(can_construct_ransom_note(magazine, ransom_note))  # Output: False
```
