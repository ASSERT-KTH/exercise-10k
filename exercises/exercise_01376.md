## Exercise
Create a program to calculate the interest earned on a bank deposit.

## Reference Solution
```python
def calculate_interest(principal, rate, time):
    """
    Calculate the interest earned on a bank deposit.

    Args:
        principal (float): The initial deposit amount.
        rate (float): The annual interest rate (in decimal form, e.g., 4% = 0.04).
        time (int): The time period in years.

    Returns:
        float: The total interest earned.
    """
    interest = principal * rate * time
    return interest

def main():
    print("Bank Deposit Interest Calculator")
    print("---------------------------------")

    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the annual interest rate (in %): ")) / 100
    time = int(input("Enter the time period (in years): "))

    # Calculate interest
    interest = calculate_interest(principal, rate, time)

    # Calculate total amount
    total_amount = principal + interest

    # Display results
    print("\nResults:")
    print(f"Principal Amount: ${principal:.2f}")
    print(f"Interest Rate: {rate * 100}%")
    print(f"Time Period: {time} years")
    print(f"Total Interest: ${interest:.2f}")
    print(f"Total Amount: ${total_amount:.2f}")

if __name__ == "__main__":
    main()
```
