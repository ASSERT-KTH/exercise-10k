## Exercise
Write a program that takes a bytes object as input and returns the string decoded in a specified encoding (e.g. ASCII, UTF-8).

## Reference Solution
```python
def decode_bytes(bytes_obj, encoding):
    """
    Decodes a bytes object into a string using a specified encoding.

    Args:
        bytes_obj (bytes): The bytes object to be decoded.
        encoding (str): The encoding to use for decoding.

    Returns:
        str: The decoded string.
    """
    try:
        # Attempt to decode the bytes object using the specified encoding
        decoded_str = bytes_obj.decode(encoding)
        return decoded_str
    except LookupError:
        # If the encoding is not found, raise a ValueError
        raise ValueError(f"Unknown encoding: {encoding}")
    except UnicodeDecodeError as e:
        # If the decoding fails, raise a ValueError with the error message
        raise ValueError(f"Failed to decode bytes: {e}")

# Example usage:
if __name__ == "__main__":
    # Define a bytes object
    bytes_obj = b"Hello, World!"

    # Define the encoding to use
    encoding = "ascii"

    # Decode the bytes object
    try:
        decoded_str = decode_bytes(bytes_obj, encoding)
        print(f"Decoded string: {decoded_str}")
    except ValueError as e:
        print(f"Error: {e}")
```
