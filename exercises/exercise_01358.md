## Exercise
Write a program to demonstrate the use of environment variables

## Reference Solution
```python
import os

def main():
    # Set an environment variable
    env_var_name = "DEMO_VAR"
    env_var_value = "Hello, World!"
    os.environ[env_var_name] = env_var_value
    print(f"Set environment variable {env_var_name} to {env_var_value}")

    # Get the value of an environment variable
    print(f"Getting environment variable {env_var_name}...")
    print(f"Value: {os.environ.get(env_var_name)}")

    # Update an environment variable
    new_value = "Updated value"
    os.environ[env_var_name] = new_value
    print(f"Updated environment variable {env_var_name} to {new_value}")

    # Delete an environment variable
    del os.environ[env_var_name]
    print(f"Deleted environment variable {env_var_name}")

    # Try to get the deleted environment variable
    print(f"Trying to get deleted environment variable {env_var_name}...")
    print(f"Value: {os.environ.get(env_var_name)}")

if __name__ == "__main__":
    main()
```
