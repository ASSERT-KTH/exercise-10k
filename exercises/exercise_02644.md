## Exercise
Write a function to delete a key-value pair from a dictionary

## Reference Solution
```python
def delete_key_value_pair(dictionary, key):
    """
    Deletes a key-value pair from a dictionary.

    Args:
    dictionary (dict): The dictionary from which to delete the key-value pair.
    key: The key of the key-value pair to delete.

    Returns:
    dict: The dictionary with the key-value pair deleted.

    Raises:
    KeyError: If the key is not found in the dictionary.
    """

    # Check if the key exists in the dictionary
    if key in dictionary:
        # Delete the key-value pair
        del dictionary[key]
    else:
        # Raise a KeyError if the key is not found
        raise KeyError(f"The key '{key}' does not exist in the dictionary.")

    return dictionary

# Example usage:
if __name__ == "__main__":
    dictionary = {"name": "John", "age": 30, "city": "New York"}
    print("Original dictionary:", dictionary)

    key_to_delete = "age"
    updated_dictionary = delete_key_value_pair(dictionary, key_to_delete)
    print(f"Dictionary after deleting '{key_to_delete}':", updated_dictionary)
```
