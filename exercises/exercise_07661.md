## Exercise
Create a function to find the least common multiple of two given numbers.

## Reference Solution
```python
import math

def find_lcm(num1, num2):
    """
    Calculate the least common multiple of two numbers.

    Args:
        num1 (int): The first number.
        num2 (int): The second number.

    Returns:
        int: The least common multiple of num1 and num2.
    """
    # Calculate the GCD using the math module
    gcd = math.gcd(num1, num2)
    
    # Calculate the LCM using the formula: lcm(a, b) = |a*b| / gcd(a, b)
    lcm = abs(num1 * num2) // gcd
    
    return lcm

# Example usage
if __name__ == "__main__":
    num1 = 12
    num2 = 15
    print(f"The LCM of {num1} and {num2} is: {find_lcm(num1, num2)}")
```
