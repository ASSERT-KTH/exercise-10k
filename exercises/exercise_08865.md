## Exercise
Create a function to check if a given number is a Pronic number or not

## Reference Solution

import math

def is_pronic(n):
    """
    Checks if a given number is a Pronic number.

    Args:
        n (int): The number to check.

    Returns:
        bool: True if the number is Pronic, False otherwise.
    """
    if n < 0:
        return False
    for i in range(int(math.sqrt(n)) + 1):
        if i * (i + 1) == n:
            return True
    return False

# Example usage:
numbers_to_check = [2, 6, 12, 20, 30]
for num in numbers_to_check:
    if is_pronic(num):
        print(f"{num} is a Pronic number.")
    else:
        print(f"{num} is not a Pronic number.")
