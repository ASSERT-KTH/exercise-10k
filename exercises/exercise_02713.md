## Exercise
Write a program to find the longest common suffix of two or more strings.

def longest_common_suffix(*strings):
    """
    Find the longest common suffix of two or more strings.

    Args:
        *strings (str): Variable number of strings to compare.

    Returns:
        str: The longest common suffix.
    """
    if not strings:
        return ""

    # Reverse each string to compare suffixes
    reversed_strings = [s[::-1] for s in strings]

    # Find the length of the shortest string
    min_len = min(len(s) for s in reversed_strings)

    # Initialize the common suffix
    common_suffix = ""

    # Compare characters from the start of each reversed string
    for chars in zip(*reversed_strings):
        # If all characters match, add to the common suffix
        if len(set(chars)) == 1:
            common_suffix += chars[0]
        # If characters don't match, break the loop
        else:
            break

    # Reverse the common suffix to get the original order
    common_suffix = common_suffix[::-1]

    return common_suffix
