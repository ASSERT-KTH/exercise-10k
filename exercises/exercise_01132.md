## Exercise
Write a program that takes a piece of text as input and analyzes its content (e.g. word frequency, sentiment)

## Reference Solution
```python
import nltk
from nltk.sentiment.vader import SentimentIntensityAnalyzer
from nltk.corpus import stopwords
from collections import Counter
import re

# Initialize NLTK data
nltk.download('vader_lexicon')
nltk.download('stopwords')

def analyze_text(text):
    """
    Analyze the content of the input text.

    Args:
    text (str): The input text.

    Returns:
    dict: A dictionary containing the analysis results.
    """

    # Remove punctuation and convert to lowercase
    text = re.sub(r'[^\w\s]', '', text).lower()

    # Tokenize the text into words
    words = nltk.word_tokenize(text)

    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    words = [word for word in words if word not in stop_words]

    # Calculate word frequency
    word_freq = Counter(words)

    # Calculate sentiment
    sia = SentimentIntensityAnalyzer()
    sentiment = sia.polarity_scores(text)

    # Create a dictionary to store the analysis results
    analysis_results = {
        'word_frequency': dict(word_freq.most_common(10)),  # Top 10 most frequent words
        'sentiment': sentiment
    }

    return analysis_results

def main():
    text = input("Enter a piece of text: ")
    analysis_results = analyze_text(text)

    print("Analysis Results:")
    print("------------------")
    print("Word Frequency:")
    for word, freq in analysis_results['word_frequency'].items():
        print(f"{word}: {freq}")
    print()
    print("Sentiment:")
    print(f"Positive: {analysis_results['sentiment']['pos']:.2f}")
    print(f"Neutral: {analysis_results['sentiment']['neu']:.2f}")
    print(f"Negative: {analysis_results['sentiment']['neg']:.2f}")
    print(f"Compound: {analysis_results['sentiment']['compound']:.2f}")

if __name__ == "__main__":
    main()
```
