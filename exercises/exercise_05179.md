## Exercise
Write a function to check if two dictionaries are equal.

## Reference Solution
```python
def are_dictionaries_equal(dict1, dict2):
    """
    Checks if two dictionaries are equal.
    
    Args:
        dict1 (dict): The first dictionary to compare.
        dict2 (dict): The second dictionary to compare.
    
    Returns:
        bool: True if the dictionaries are equal, False otherwise.
    """
    return dict1 == dict2

# Example usage
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2}
    dict2 = {"a": 1, "b": 2}
    dict3 = {"a": 1, "b": 3}
    
    print(are_dictionaries_equal(dict1, dict2))  # Output: True
    print(are_dictionaries_equal(dict1, dict3))  # Output: False
```

def are_dictionaries_equal_manual(dict1, dict2):
    """
    Manually checks if two dictionaries are equal.
    
    Args:
        dict1 (dict): The first dictionary to compare.
        dict2 (dict): The second dictionary to compare.
    
    Returns:
        bool: True if the dictionaries are equal, False otherwise.
    """
    # Check if the number of keys is equal
    if len(dict1) != len(dict2):
        return False
    
    # Check if all key-value pairs are equal
    for key, value in dict1.items():
        if key not in dict2 or dict2[key] != value:
            return False
    
    return True

# Example usage
if __name__ == "__main__":
    dict1 = {"a": 1, "b": 2}
    dict2 = {"a": 1, "b": 2}
    dict3 = {"a": 1, "b": 3}
    
    print(are_dictionaries_equal_manual(dict1, dict2))  # Output: True
    print(are_dictionaries_equal_manual(dict1, dict3))  # Output: False
