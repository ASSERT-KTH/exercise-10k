## Exercise
Write a program to calculate the sum of all integers from 1 to a given number.

## Reference Solution
```python
def sum_of_integers(n):
    """
    Calculate the sum of all integers from 1 to a given number.

    Args:
        n (int): The given number.

    Returns:
        int: The sum of all integers from 1 to n.
    """
    if not isinstance(n, int) or n < 1:
        raise ValueError("Input must be a positive integer.")

    # Calculate the sum using the formula for the sum of an arithmetic series
    total_sum = (n * (n + 1)) // 2

    return total_sum


def main():
    num = int(input("Enter a positive integer: "))
    try:
        result = sum_of_integers(num)
        print(f"The sum of all integers from 1 to {num} is: {result}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
