## Exercise
Write a function to encode a given string using Huffman coding. The function should return the encoded string.

from collections import defaultdict
from heapq import heapify, heappop, heappush

def huffmanEncoding(string):
    # Step 1: Calculate the frequency of each character in the string
    frequency = defaultdict(int)
    for char in string:
        frequency[char] += 1

    # Step 2: Create a priority queue to store characters and their frequencies
    priority_queue = [[weight, [char, ""]] for char, weight in frequency.items()]
    heapify(priority_queue)

    # Step 3: Build the Huffman tree
    while len(priority_queue) > 1:
        lo = heappop(priority_queue)
        hi = heappop(priority_queue)
        for pair in lo[1:]:
            pair[1] = '0' + pair[1]
        for pair in hi[1:]:
            pair[1] = '1' + pair[1]
        heappush(priority_queue, [lo[0] + hi[0]] + lo[1:] + hi[1:])

    # Step 4: Build the Huffman codes
    huffman_codes = sorted(heapq.heappop(priority_queue)[1:], key=lambda p: (len(p[-1]), p))

    # Step 5: Build the encoded string
    encoded_string = ""
    huffman_dict = {}
    for char, code in huffman_codes:
        huffman_dict[char] = code
    for char in string:
        encoded_string += huffman_dict[char]

    return encoded_string

# Example usage:
string = "this is an example for huffman encoding"
encoded_string = huffmanEncoding(string)
print(encoded_string)
