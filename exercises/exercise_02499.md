## Exercise
Write a program to find the longest increasing subsequence in a list of numbers

## Reference Solution

def longest_increasing_subsequence(numbers):
    """
    Find the longest increasing subsequence in a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        list: The longest increasing subsequence.
    """
    if not numbers:
        return []

    # Initialize lists to store lengths and subsequences
    lengths = [1] * len(numbers)
    subsequences = [[num] for num in numbers]

    # Compute lengths and subsequences
    for i in range(1, len(numbers)):
        for j in range(i):
            if numbers[i] > numbers[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                subsequences[i] = subsequences[j] + [numbers[i]]

    # Find the longest increasing subsequence
    max_length = max(lengths)
    max_index = lengths.index(max_length)
    return subsequences[max_index]


def main():
    # Example usage
    numbers = [10, 22, 9, 33, 21, 50, 41, 60, 80]
    print("Input list:", numbers)
    print("Longest increasing subsequence:", longest_increasing_subsequence(numbers))


if __name__ == "__main__":
    main()
