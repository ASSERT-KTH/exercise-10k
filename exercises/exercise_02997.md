## Exercise
Write a program that checks if a given list is a rotation of another given list.

## Reference Solution
```python
def is_rotation(list1, list2):
    """
    Checks if list2 is a rotation of list1.

    Args:
        list1 (list): The original list.
        list2 (list): The list to check.

    Returns:
        bool: True if list2 is a rotation of list1, False otherwise.
    """
    if len(list1) != len(list2):
        return False

    # Concatenate list1 with itself to simulate rotation
    double_list1 = list1 + list1

    # Check if list2 is a sublist of the concatenated list
    for i in range(len(list1)):
        if double_list1[i:i + len(list1)] == list2:
            return True

    return False


def main():
    # Example usage
    list1 = [1, 2, 3, 4, 5]
    list2 = [3, 4, 5, 1, 2]
    list3 = [5, 4, 3, 2, 1]

    print(f"Is {list2} a rotation of {list1}? {is_rotation(list1, list2)}")
    print(f"Is {list3} a rotation of {list1}? {is_rotation(list1, list3)}")


if __name__ == "__main__":
    main()
```
