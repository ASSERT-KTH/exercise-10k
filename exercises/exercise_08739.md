## Exercise
Validate variables against a complex set of rules to ensure they are correct.

## Reference Solution
```python
import re

# Define the rules
class ValidationRules:
    def __init__(self):
        self.rules = {
            'name': self.validate_name,
            'email': self.validate_email,
            'age': self.validate_age,
        }

    def validate_name(self, value):
        """Validate name"""
        if not isinstance(value, str):
            return False
        if len(value) < 2 or len(value) > 50:
            return False
        return True

    def validate_email(self, value):
        """Validate email"""
        email_regex = r"[^@]+@[^@]+\.[^@]+"
        if not re.match(email_regex, value):
            return False
        return True

    def validate_age(self, value):
        """Validate age"""
        if not isinstance(value, int):
            return False
        if value < 18 or value > 100:
            return False
        return True

    def validate_variable(self, variable_name, value):
        """Validate a variable against the rules"""
        if variable_name in self.rules:
            return self.rules[variable_name](value)
        else:
            return False

# Create a validation instance
validation_rules = ValidationRules()

# Define the variables to validate
variables = {
    'name': 'John Doe',
    'email': 'john@example.com',
    'age': 30,
}

# Validate the variables
for variable_name, value in variables.items():
    if validation_rules.validate_variable(variable_name, value):
        print(f"{variable_name} is valid")
    else:
        print(f"{variable_name} is invalid")
```
