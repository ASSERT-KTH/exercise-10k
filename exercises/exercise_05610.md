## Exercise
Write a program that simulates a simple chatbot, taking the user's input as messages and responding with pre-defined responses.

## Reference Solution
```python
# Import the required libraries
import random

# Define a dictionary with pre-defined responses
responses = {
    'hello': ['Hi, how are you?', 'Hello! What can I help you with?', 'Hey, what\'s up?'],
    'how are you': ['I\'m doing great, thanks for asking!', 'I\'m just a chatbot, I don\'t have feelings.', 'I\'m good, thanks!'],
    'what is your purpose': ['My purpose is to assist and provide information.', 'I\'m here to help answer your questions.', 'My purpose is to chat with you and provide helpful responses.'],
    'default': ['I didn\'t understand that. Can you please rephrase?', 'I\'m not sure I understand. Can you explain?', 'I didn\'t catch that. Can you try again?']
}

def get_response(message):
    """
    This function takes in a message and returns a pre-defined response.
    
    Args:
        message (str): The user's input message.
    
    Returns:
        str: A pre-defined response to the user's message.
    """
    message = message.lower()
    
    # Check if the message contains any of the pre-defined keywords
    for keyword in responses:
        if keyword in message:
            # Return a random response from the list of pre-defined responses
            return random.choice(responses[keyword])
    
    # If the message doesn't contain any of the pre-defined keywords, return a default response
    return random.choice(responses['default'])

def main():
    print("Welcome to the chatbot! Type 'quit' to exit.")
    
    while True:
        # Get the user's input message
        message = input("You: ")
        
        # Check if the user wants to quit
        if message.lower() == 'quit':
            print("Goodbye!")
            break
        
        # Get the chatbot's response
        response = get_response(message)
        
        # Print the chatbot's response
        print("Chatbot: ", response)

if __name__ == "__main__":
    main()
```
