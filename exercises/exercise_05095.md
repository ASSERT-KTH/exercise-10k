## Exercise
Write a program that analyzes a piece of text and prints out statistics about the text, such as the number of words and characters.

## Reference Solution
```python
def analyze_text(text):
    """
    Analyzes a piece of text and returns statistics about the text.

    Args:
        text (str): The text to analyze.

    Returns:
        dict: A dictionary containing statistics about the text.
    """
    # Remove leading and trailing whitespace
    text = text.strip()

    # Calculate the number of characters
    num_chars = len(text)

    # Calculate the number of words
    words = text.split()
    num_words = len(words)

    # Calculate the number of sentences
    num_sentences = text.count('.') + text.count('!') + text.count('?')

    # Calculate the average word length
    avg_word_len = sum(len(word) for word in words) / num_words if num_words > 0 else 0

    # Return the statistics
    return {
        'num_chars': num_chars,
        'num_words': num_words,
        'num_sentences': num_sentences,
        'avg_word_len': avg_word_len
    }

def print_statistics(stats):
    """
    Prints out statistics about the text.

    Args:
        stats (dict): A dictionary containing statistics about the text.
    """
    print("Text Statistics:")
    print("----------------")
    print(f"Number of Characters: {stats['num_chars']}")
    print(f"Number of Words: {stats['num_words']}")
    print(f"Number of Sentences: {stats['num_sentences']}")
    print(f"Average Word Length: {stats['avg_word_len']:.2f}")

def main():
    # Get the text from the user
    text = input("Enter a piece of text: ")

    # Analyze the text
    stats = analyze_text(text)

    # Print the statistics
    print_statistics(stats)

if __name__ == "__main__":
    main()
```
