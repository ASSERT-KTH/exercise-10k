## Exercise
Write a program that takes a temperature as input and prints the equivalent temperature in Celsius or Fahrenheit, depending on the user's choice.

## Reference Solution
```python
#include <iostream>

int main() {
    float temperature;
    int option;

    std::cout << "Enter temperature: ";
    std::cin >> temperature;

    std::cout << "1. Celsius to Fahrenheit\n";
    std::cout << "2. Fahrenheit to Celsius\n";
    std::cout << "Enter your choice: ";
    std::cin >> option;

    switch (option) {
        case 1:
            temperature = (temperature * 9/5) + 32;
            std::cout << "Temperature in Fahrenheit: " << temperature << std::endl;
            break;
        case 2:
            temperature = (temperature - 32) * 5/9;
            std::cout << "Temperature in Celsius: " << temperature << std::endl;
            break;
        default:
            std::cout << "Invalid option. Please try again." << std::endl;
            break;
    }

    return 0;
}
```
