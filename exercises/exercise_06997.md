## Exercise
Write a program that takes the sides of a triangle as input and returns its perimeter.

## Reference Solution
```python
def calculate_perimeter(a, b, c):
    """
    Calculate the perimeter of a triangle.

    Parameters:
    a (float): The length of the first side.
    b (float): The length of the second side.
    c (float): The length of the third side.

    Returns:
    float: The perimeter of the triangle.
    """
    # Check if the sides can form a valid triangle
    if a + b <= c or a + c <= b or b + c <= a:
        raise ValueError("The sides cannot form a valid triangle")

    # Calculate the perimeter
    perimeter = a + b + c
    return perimeter


def main():
    # Take the sides of the triangle as input
    a = float(input("Enter the length of the first side: "))
    b = float(input("Enter the length of the second side: "))
    c = float(input("Enter the length of the third side: "))

    try:
        # Calculate and print the perimeter
        perimeter = calculate_perimeter(a, b, c)
        print(f"The perimeter of the triangle is: {perimeter}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
```
