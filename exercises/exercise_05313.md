## Exercise
Create a function that checks the type of its input and returns an error message if it's not valid

## Reference Solution
```python
def validate_input(input_value, expected_type):
    """
    Validate the type of the input value.

    Args:
        input_value (any): The input value to be validated.
        expected_type (type): The expected type of the input value.

    Returns:
        str: An error message if the input type is not valid, otherwise None.
    """
    if not isinstance(input_value, expected_type):
        return f"Error: Expected {expected_type.__name__}, but got {type(input_value).__name__}"
    return None

# Example usage
def main():
    # Valid input
    print(validate_input(10, int))  # Should return None

    # Invalid input
    print(validate_input("hello", int))  # Should return an error message

    # Valid string input
    print(validate_input("hello", str))  # Should return None

    # Invalid string input
    print(validate_input(10, str))  # Should return an error message

if __name__ == "__main__":
    main()
```
