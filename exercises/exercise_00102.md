## Exercise
Write a program to find the maximum value in an array.

## Reference Solution
```python
def find_max_value(array):
    """
    Finds the maximum value in an array.

    Args:
        array (list): A list of numbers.

    Returns:
        max_value (int/float): The maximum value in the array.

    Raises:
        ValueError: If the array is empty.
    """
    if not array:
        raise ValueError("Array is empty")

    max_value = array[0]
    for value in array:
        if value > max_value:
            max_value = value

    return max_value

def main():
    # Example usage
    array = [12, 45, 7, 23, 56, 89, 34]
    try:
        max_value = find_max_value(array)
        print(f"Maximum value in the array: {max_value}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
