## Exercise
Create a dictionary with 5 key-value pairs. Access a specific key-value pair using the key.

## Reference Solution
```python
# Create a dictionary with 5 key-value pairs
my_dict = {
    "name": "John Doe",
    "age": 30,
    "city": "New York",
    "country": "USA",
    "occupation": "Software Engineer"
}

# Access a specific key-value pair using the key
def access_key_value_pair(dictionary, key):
    """
    Access a specific key-value pair in a dictionary.

    Args:
        dictionary (dict): The dictionary to access.
        key (str): The key of the desired key-value pair.

    Returns:
        The value associated with the given key if it exists, otherwise None.
    """
    return dictionary.get(key)

# Example usage:
key_to_access = "age"
value = access_key_value_pair(my_dict, key_to_access)

if value is not None:
    print(f"The value associated with the key '{key_to_access}' is: {value}")
else:
    print(f"The key '{key_to_access}' does not exist in the dictionary.")
```
