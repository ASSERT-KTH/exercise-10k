## Exercise
Write a program that converts a variable from one data type to another.

## Reference Solution
```python
def convert_data_type(var, target_type):
    """
    Converts a variable from one data type to another.

    Args:
        var (any): The variable to be converted.
        target_type (str): The target data type.

    Returns:
        any: The converted variable.

    Raises:
        ValueError: If the conversion is not supported.
    """
    if target_type == "int":
        try:
            return int(var)
        except ValueError:
            print("Error: Cannot convert to integer.")
            return None
    elif target_type == "float":
        try:
            return float(var)
        except ValueError:
            print("Error: Cannot convert to float.")
            return None
    elif target_type == "str":
        return str(var)
    else:
        print("Error: Unsupported target data type.")
        return None


# Example usage
if __name__ == "__main__":
    # Convert integer to float
    var = 10
    converted_var = convert_data_type(var, "float")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")

    # Convert float to integer
    var = 10.5
    converted_var = convert_data_type(var, "int")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")

    # Convert string to integer
    var = "10"
    converted_var = convert_data_type(var, "int")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")

    # Convert string to float
    var = "10.5"
    converted_var = convert_data_type(var, "float")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")

    # Convert integer to string
    var = 10
    converted_var = convert_data_type(var, "str")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")

    # Convert float to string
    var = 10.5
    converted_var = convert_data_type(var, "str")
    print(f"Converted {var} to {converted_var} ({type(converted_var).__name__})")
```
