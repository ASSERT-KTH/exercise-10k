## Exercise
Find the youngest student in a dictionary where keys are student names and values are their corresponding ages

## Reference Solution
```python
def find_youngest_student(students):
    """
    Find the youngest student(s) in a dictionary.

    Args:
        students (dict): A dictionary where keys are student names and values are their corresponding ages.

    Returns:
        list: A list of the youngest student(s) and their age.
    """
    # Check if the dictionary is empty
    if not students:
        return None

    # Find the minimum age
    min_age = min(students.values())

    # Find the students with the minimum age
    youngest_students = [(name, age) for name, age in students.items() if age == min_age]

    return youngest_students

# Example usage
students = {
    "John": 20,
    "Alice": 22,
    "Bob": 20,
    "Charlie": 25
}

youngest_students = find_youngest_student(students)

if youngest_students:
    print("The youngest student(s) is/are:")
    for student in youngest_students:
        print(f"{student[0]} - {student[1]} years old")
else:
    print("No students found.")
```
