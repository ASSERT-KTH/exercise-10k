## Exercise
Create a program that responds to basic user input with pre-defined responses.

## Reference Solution
```python
# Import the required libraries
import time

# Define a dictionary with pre-defined responses
responses = {
    'hello': 'Hello! How are you today?',
    'hi': 'Hi! What can I help you with?',
    'how are you': 'I\'m doing great, thanks for asking!',
    'what is your name': 'My name is Chatbot.',
    'bye': 'Goodbye! It was nice chatting with you.',
    'default': 'I didn\'t understand that. Please try again.'
}

def get_response(user_input):
    """
    Returns a pre-defined response based on the user's input.
    
    Args:
        user_input (str): The user's input.
    
    Returns:
        str: A pre-defined response.
    """
    # Convert the user's input to lowercase for case-insensitive comparison
    user_input = user_input.lower()
    
    # Check if the user's input matches any of the pre-defined responses
    for key in responses:
        if key in user_input:
            return responses[key]
    
    # If no match is found, return the default response
    return responses['default']

def main():
    print("Welcome to the chatbot program!")
    print("Type 'bye' to exit the program.")
    
    while True:
        # Get the user's input
        user_input = input("You: ")
        
        # Check if the user wants to exit the program
        if user_input.lower() == 'bye':
            print("Chatbot: ", get_response(user_input))
            break
        
        # Print the chatbot's response
        print("Chatbot: ", get_response(user_input))
        
        # Add a short delay to simulate a conversation
        time.sleep(0.5)

if __name__ == "__main__":
    main()
```
