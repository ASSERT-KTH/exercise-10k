## Exercise
Write a program that performs set operations (union, intersection, difference, etc.) and prints the result.

## Reference Solution
```python
class SetOperations:
    def __init__(self, set1, set2):
        self.set1 = set1
        self.set2 = set2

    def union(self):
        return self.set1.union(self.set2)

    def intersection(self):
        return self.set1.intersection(self.set2)

    def difference(self):
        return self.set1.difference(self.set2)

    def symmetric_difference(self):
        return self.set1.symmetric_difference(self.set2)

    def is_subset(self):
        return self.set1.issubset(self.set2)

    def is_superset(self):
        return self.set1.issuperset(self.set2)

    def is_disjoint(self):
        return self.set1.isdisjoint(self.set2)


def main():
    # Define two sets
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}

    # Create an instance of the SetOperations class
    set_operations = SetOperations(set1, set2)

    # Perform set operations
    print("Set 1:", set1)
    print("Set 2:", set2)
    print("Union:", set_operations.union())
    print("Intersection:", set_operations.intersection())
    print("Difference (Set 1 - Set 2):", set_operations.difference())
    print("Symmetric Difference:", set_operations.symmetric_difference())
    print("Is Set 1 a subset of Set 2?", set_operations.is_subset())
    print("Is Set 1 a superset of Set 2?", set_operations.is_superset())
    print("Is Set 1 disjoint from Set 2?", set_operations.is_disjoint())


if __name__ == "__main__":
    main()
```
