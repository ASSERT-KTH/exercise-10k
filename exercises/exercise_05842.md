## Exercise
Create a program that thinks of a random number between 1 and 10, and then asks the user to guess it. After each guess, tell the user if their guess is higher or lower than the number

## Reference Solution
```python
import random

def get_user_guess():
    """Gets the user's guess and converts it to an integer."""
    while True:
        try:
            return int(input("Guess a number between 1 and 10: "))
        except ValueError:
            print("Invalid input. Please enter a number.")

def check_guess(guess, target):
    """Checks if the user's guess is correct, higher, or lower than the target number."""
    if guess < target:
        return "Your guess is too low. Try again!"
    elif guess > target:
        return "Your guess is too high. Try again!"
    else:
        return "Congratulations! You guessed the number correctly."

def main():
    # Generate a random number between 1 and 10
    target_number = random.randint(1, 10)
    
    # Initialize the number of attempts
    attempts = 0
    
    print("Welcome to the number guessing game!")
    print("I'm thinking of a number between 1 and 10.")
    
    while True:
        # Get the user's guess
        user_guess = get_user_guess()
        
        # Check if the guess is within the valid range
        if user_guess < 1 or user_guess > 10:
            print("Please enter a number between 1 and 10.")
            continue
        
        # Increment the number of attempts
        attempts += 1
        
        # Check the guess
        result = check_guess(user_guess, target_number)
        print(result)
        
        # If the guess is correct, exit the loop
        if "Congratulations" in result:
            print(f"You guessed the number in {attempts} attempts.")
            break

if __name__ == "__main__":
    main()
```
