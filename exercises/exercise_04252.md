## Exercise
Create a program that allows users to manage their bank accounts. The program should take the account type, balance, and transaction type as input, and print the updated balance.

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_type, balance=0):
        """
        Initialize a BankAccount object.

        Args:
        - account_type (str): The type of bank account (e.g., "Checking", "Savings").
        - balance (float, optional): The initial balance of the account. Defaults to 0.
        """
        self.account_type = account_type
        self.balance = balance

    def deposit(self, amount):
        """
        Deposit money into the account.

        Args:
        - amount (float): The amount to deposit.

        Returns:
        - float: The updated balance.
        """
        self.balance += amount
        return self.balance

    def withdraw(self, amount):
        """
        Withdraw money from the account.

        Args:
        - amount (float): The amount to withdraw.

        Returns:
        - float: The updated balance if the withdrawal is successful, otherwise None.
        """
        if amount > self.balance:
            print("Insufficient funds.")
            return None
        self.balance -= amount
        return self.balance

    def print_balance(self):
        """
        Print the current balance of the account.
        """
        print(f"Current balance: ${self.balance:.2f}")


def main():
    print("Welcome to the Bank Account Manager!")
    
    # Get account type from user
    account_type = input("Enter the account type (e.g., Checking, Savings): ")
    
    # Get initial balance from user
    while True:
        try:
            balance = float(input("Enter the initial balance: $"))
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")
    
    # Create a BankAccount object
    account = BankAccount(account_type, balance)
    
    # Main loop
    while True:
        print("\nOptions:")
        print("1. Deposit")
        print("2. Withdraw")
        print("3. Check balance")
        print("4. Quit")
        
        # Get user's choice
        while True:
            try:
                choice = int(input("Enter your choice (1-4): "))
                if 1 <= choice <= 4:
                    break
                else:
                    print("Invalid choice. Please enter a number between 1 and 4.")
            except ValueError:
                print("Invalid input. Please enter a valid number.")
        
        # Handle user's choice
        if choice == 1:
            while True:
                try:
                    amount = float(input("Enter the amount to deposit: $"))
                    if amount > 0:
                        account.deposit(amount)
                        account.print_balance()
                        break
                    else:
                        print("Invalid amount. Please enter a positive number.")
                except ValueError:
                    print("Invalid input. Please enter a valid number.")
        elif choice == 2:
            while True:
                try:
                    amount = float(input("Enter the amount to withdraw: $"))
                    if amount > 0:
                        updated_balance = account.withdraw(amount)
                        if updated_balance is not None:
                            account.print_balance()
                        break
                    else:
                        print("Invalid amount. Please enter a positive number.")
                except ValueError:
                    print("Invalid input. Please enter a valid number.")
        elif choice == 3:
            account.print_balance()
        elif choice == 4:
            print("Thank you for using the Bank Account Manager!")
            break


if __name__ == "__main__":
    main()
```
