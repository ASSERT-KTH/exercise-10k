## Exercise
Create a function that takes two boolean values as arguments and returns True if both are True, False otherwise

## Reference Solution
```python
def boolean_and(a: bool, b: bool) -> bool:
    """
    Returns True if both input boolean values are True, False otherwise.

    Args:
        a (bool): The first boolean value.
        b (bool): The second boolean value.

    Returns:
        bool: The result of the logical AND operation.
    """
    return a and b

# Example usage:
if __name__ == "__main__":
    print(boolean_and(True, True))  # Expected output: True
    print(boolean_and(True, False))  # Expected output: False
    print(boolean_and(False, True))  # Expected output: False
    print(boolean_and(False, False))  # Expected output: False
```
