## Exercise
Develop a program that determines an employee's benefits based on their job title and years of service. The program should use if/else statements to determine the employee's benefits.

## Reference Solution
```python
class Employee:
    def __init__(self, name, job_title, years_of_service):
        self.name = name
        self.job_title = job_title
        self.years_of_service = years_of_service

    def determine_benefits(self):
        benefits = []

        # Check for health insurance eligibility
        if self.years_of_service >= 1:
            benefits.append("Health Insurance")

        # Check for retirement plan eligibility
        if self.years_of_service >= 5:
            benefits.append("Retirement Plan")

        # Check for life insurance eligibility
        if self.job_title == "Manager" or self.job_title == "Executive":
            benefits.append("Life Insurance")

        # Check for paid time off eligibility
        if self.years_of_service >= 3:
            benefits.append("Paid Time Off")

        # Check for education assistance eligibility
        if self.job_title == "Software Engineer" or self.job_title == "Data Scientist":
            benefits.append("Education Assistance")

        # Check for bonus eligibility
        if self.years_of_service >= 10:
            benefits.append("Bonus")

        return benefits


def main():
    print("Employee Benefits Program")
    print("---------------------------")

    name = input("Enter employee name: ")
    job_title = input("Enter employee job title: ")
    years_of_service = int(input("Enter employee years of service: "))

    employee = Employee(name, job_title, years_of_service)
    benefits = employee.determine_benefits()

    print(f"\nEmployee: {employee.name}")
    print(f"Job Title: {employee.job_title}")
    print(f"Years of Service: {employee.years_of_service}")
    print("Benefits:")
    for benefit in benefits:
        print(f"- {benefit}")


if __name__ == "__main__":
    main()
```
