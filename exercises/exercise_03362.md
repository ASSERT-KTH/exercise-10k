## Exercise
Use a while loop to calculate the sum of even numbers from 2 to 100

## Reference Solution
```python
def sum_even_numbers(start, end):
    """
    Calculate the sum of even numbers within a given range.

    Args:
    start (int): The starting number of the range (inclusive).
    end (int): The ending number of the range (inclusive).

    Returns:
    int: The sum of even numbers within the range.
    """
    total = 0
    current = start
    while current <= end:
        if current % 2 == 0:
            total += current
        current += 1
    return total


def main():
    start_range = 2
    end_range = 100
    result = sum_even_numbers(start_range, end_range)
    print(f"The sum of even numbers from {start_range} to {end_range} is: {result}")


if __name__ == "__main__":
    main()
```
