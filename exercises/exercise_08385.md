## Exercise
Write a program that calculates the nth Catalan number using dynamic programming.

def catalan(n):
    """
    Calculate the nth Catalan number using dynamic programming.

    The Catalan numbers are a sequence of natural numbers that occur in various counting problems,
    often involving recursive structures.

    :param n: The index of the Catalan number to calculate.
    :return: The nth Catalan number.
    """
    # Create a table to store the Catalan numbers
    catalan_table = [0] * (n + 1)

    # Base case: C(0) = 1
    catalan_table[0] = 1

    # Calculate the Catalan numbers iteratively
    for i in range(1, n + 1):
        catalan_table[i] = 0
        for j in range(i):
            catalan_table[i] += catalan_table[j] * catalan_table[i - j - 1]

    return catalan_table[n]

# Example usage:
n = 5
result = catalan(n)
print(f"The {n}th Catalan number is: {result}")
