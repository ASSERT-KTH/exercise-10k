## Exercise
Write a program that finds the maximum value in an array of integers. The program should take an array of integers as input, iterate through the array to find the maximum value, and then return this maximum value.

## Reference Solution
```python
class Solution {
public:
    int maximumWealth(vector<vector<int>>& accounts) {
        int ans = 0;
        int n = accounts.size();
        for (int i = 0; i < n; i++) {
            int sum = 0;
            for (int j = 0; j < accounts[i].size(); j++) {
                sum += accounts[i][j];
            }
            ans = max(ans, sum);
        }
        return ans;
    }
};
```
