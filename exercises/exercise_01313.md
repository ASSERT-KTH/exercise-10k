## Exercise
Write a function to find the minimum value in a dictionary

## Reference Solution
```python
def find_min_value(dictionary):
    """
    This function finds the minimum value in a dictionary.

    Args:
        dictionary (dict): The input dictionary.

    Returns:
        The minimum value in the dictionary.

    Raises:
        ValueError: If the dictionary is empty.
    """
    if not dictionary:
        raise ValueError("Dictionary is empty")

    return min(dictionary.values())

# Example usage:
if __name__ == "__main__":
    example_dict = {"a": 10, "b": 5, "c": 15}
    print(find_min_value(example_dict))  # Output: 5

    empty_dict = {}
    try:
        print(find_min_value(empty_dict))
    except ValueError as e:
        print(e)  # Output: Dictionary is empty
```
