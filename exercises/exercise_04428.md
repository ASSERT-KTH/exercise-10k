## Exercise
Create a program that allows the user to add and remove items from a to-do list using a while loop.

## Reference Solution
```python
def main():
    # Initialize an empty to-do list
    todo_list = []

    while True:
        # Print the current to-do list
        print("\nCurrent To-Do List:")
        for i, item in enumerate(todo_list, start=1):
            print(f"{i}. {item}")

        # Print the menu options
        print("\nOptions:")
        print("1. Add item to to-do list")
        print("2. Remove item from to-do list")
        print("3. Exit")

        # Get the user's choice
        choice = input("Enter your choice: ")

        if choice == "1":
            # Add item to to-do list
            item = input("Enter the item to add: ")
            todo_list.append(item)
            print(f"Added '{item}' to the to-do list.")
        elif choice == "2":
            # Remove item from to-do list
            if not todo_list:
                print("The to-do list is empty.")
            else:
                item_number = input("Enter the number of the item to remove: ")
                try:
                    item_number = int(item_number)
                    if 1 <= item_number <= len(todo_list):
                        item = todo_list.pop(item_number - 1)
                        print(f"Removed '{item}' from the to-do list.")
                    else:
                        print("Invalid item number.")
                except ValueError:
                    print("Invalid input. Please enter a number.")
        elif choice == "3":
            # Exit the program
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
