## Exercise
Develop a program to mix colors, given two colors and their proportions, using a for loop to handle multiple color combinations.

## Reference Solution
```python
def mix_colors(color1, color2, proportion):
    """
    Mix two colors in a given proportion.

    Args:
        color1 (tuple): RGB values of the first color.
        color2 (tuple): RGB values of the second color.
        proportion (float): Proportion of the first color.

    Returns:
        tuple: RGB values of the resulting color.
    """
    r = color1[0] * proportion + color2[0] * (1 - proportion)
    g = color1[1] * proportion + color2[1] * (1 - proportion)
    b = color1[2] * proportion + color2[2] * (1 - proportion)
    return (r, g, b)

def main():
    # Define colors
    colors = {
        "red": (1, 0, 0),
        "blue": (0, 0, 1),
        "green": (0, 1, 0),
        "yellow": (1, 1, 0),
    }

    # Define color combinations and proportions
    combinations = [
        {"color1": "red", "color2": "blue", "proportion": 0.5},
        {"color1": "green", "color2": "yellow", "proportion": 0.3},
        {"color1": "red", "color2": "green", "proportion": 0.7},
    ]

    # Mix colors
    for combination in combinations:
        color1 = colors[combination["color1"]]
        color2 = colors[combination["color2"]]
        proportion = combination["proportion"]
        resulting_color = mix_colors(color1, color2, proportion)
        print(f"Mixing {combination['color1']} and {combination['color2']} in proportion {proportion}:")
        print(f"Resulting color: RGB({resulting_color[0]:.2f}, {resulting_color[1]:.2f}, {resulting_color[2]:.2f})")
        print()

if __name__ == "__main__":
    main()
```
