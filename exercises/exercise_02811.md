## Exercise
Write a program to ask the user for temperature, humidity, and wind speed, and then print out a report of the meteorological data.

## Reference Solution
```python
def get_meteorological_data():
    """
    Asks the user for temperature, humidity, and wind speed.
    
    Returns:
        dict: A dictionary containing the user-inputted meteorological data.
    """
    data = {}
    
    # Get temperature from user
    while True:
        try:
            temperature = float(input("Enter temperature (°C): "))
            data["temperature"] = temperature
            break
        except ValueError:
            print("Invalid input. Please enter a number.")
    
    # Get humidity from user
    while True:
        try:
            humidity = float(input("Enter humidity (%): "))
            if humidity < 0 or humidity > 100:
                print("Invalid input. Humidity must be between 0 and 100.")
            else:
                data["humidity"] = humidity
                break
        except ValueError:
            print("Invalid input. Please enter a number.")
    
    # Get wind speed from user
    while True:
        try:
            wind_speed = float(input("Enter wind speed (km/h): "))
            if wind_speed < 0:
                print("Invalid input. Wind speed cannot be negative.")
            else:
                data["wind_speed"] = wind_speed
                break
        except ValueError:
            print("Invalid input. Please enter a number.")
    
    return data

def print_meteorological_report(data):
    """
    Prints a report of the meteorological data.
    
    Args:
        data (dict): A dictionary containing the meteorological data.
    """
    print("\nMeteorological Report:")
    print("------------------------")
    print(f"Temperature: {data['temperature']}°C")
    print(f"Humidity: {data['humidity']}%")
    print(f"Wind Speed: {data['wind_speed']} km/h")

def main():
    data = get_meteorological_data()
    print_meteorological_report(data)

if __name__ == "__main__":
    main()
```
