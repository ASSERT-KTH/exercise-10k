## Exercise
Write a function to move all zeroes to the end of an array

## Reference Solution
```python
def move_zeroes_to_end(arr):
    """
    Move all zeroes to the end of an array.

    Args:
        arr (list): The input array.

    Returns:
        list: The array with all zeroes at the end.
    """
    # Initialize two pointers, one at the beginning of the array
    non_zero_ptr = 0

    # Iterate over the array
    for i in range(len(arr)):
        # If the current element is not zero, swap it with the element at the non_zero_ptr
        if arr[i] != 0:
            arr[non_zero_ptr], arr[i] = arr[i], arr[non_zero_ptr]
            # Move the non_zero_ptr forward
            non_zero_ptr += 1

    return arr

# Example usage:
arr = [0, 1, 0, 3, 12, 0]
print("Original array:", arr)
print("Array with zeroes at the end:", move_zeroes_to_end(arr))
```
