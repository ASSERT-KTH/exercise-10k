## Exercise
Write a function to remove duplicate characters from a given string.

## Reference Solution
```python
def remove_duplicates(input_string):
    """
    Removes duplicate characters from a given string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The input string with no duplicate characters.
    """
    seen = set()
    output_string = ""

    for char in input_string:
        if char not in seen:
            output_string += char
            seen.add(char)

    return output_string

# Example usage:
input_str = "hello world"
output_str = remove_duplicates(input_str)
print("Input String:", input_str)
print("Output String:", output_str)
```

from collections import OrderedDict

def remove_duplicates_ordered(input_string):
    """
    Removes duplicate characters from a given string while preserving order.

    Args:
        input_string (str): The input string.

    Returns:
        str: The input string with no duplicate characters.
    """
    return "".join(OrderedDict.fromkeys(input_string))

# Example usage:
input_str = "hello world"
output_str = remove_duplicates_ordered(input_str)
