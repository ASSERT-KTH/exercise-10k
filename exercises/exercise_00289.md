## Exercise
Write a program that converts a given temperature from Celsius to Fahrenheit or vice versa.

## Reference Solution
```python
def celsius_to_fahrenheit(celsius):
    """Converts Celsius to Fahrenheit"""
    return (celsius * 9/5) + 32

def fahrenheit_to_celsius(fahrenheit):
    """Converts Fahrenheit to Celsius"""
    return (fahrenheit - 32) * 5/9

def main():
    while True:
        print("\nTemperature Converter")
        print("1. Celsius to Fahrenheit")
        print("2. Fahrenheit to Celsius")
        print("3. Quit")
        
        choice = input("Choose an option: ")
        
        if choice == "1":
            celsius = float(input("Enter temperature in Celsius: "))
            print(f"{celsius}째C is equal to {celsius_to_fahrenheit(celsius)}째F")
        elif choice == "2":
            fahrenheit = float(input("Enter temperature in Fahrenheit: "))
            print(f"{fahrenheit}째F is equal to {fahrenheit_to_celsius(fahrenheit)}째C")
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
