## Exercise
Write a program that takes the scores of a student as input and returns the average grade.

## Reference Solution
```python
def calculate_average_grade(scores):
    """
    Calculate the average grade of a student.

    Args:
    scores (dict): A dictionary where keys are subject names and values are scores.

    Returns:
    float: The average grade of the student.
    """
    total_score = sum(scores.values())
    average_grade = total_score / len(scores)
    return average_grade


def main():
    # Get the number of subjects
    num_subjects = int(input("Enter the number of subjects: "))

    # Initialize an empty dictionary to store scores
    scores = {}

    # Get scores for each subject
    for i in range(num_subjects):
        subject = input(f"Enter subject {i+1} name: ")
        score = float(input(f"Enter {subject} score: "))
        scores[subject] = score

    # Calculate and display the average grade
    average_grade = calculate_average_grade(scores)
    print(f"\nAverage Grade: {average_grade:.2f}")


if __name__ == "__main__":
    main()
```
