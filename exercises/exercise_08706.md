## Exercise
Create a simple converter that converts between different units of measurement.

## Reference Solution
```python
def length_conversion():
    """Converts length between meters and feet."""
    print("Length Conversion")
    print("1. Meters to Feet")
    print("2. Feet to Meters")
    choice = int(input("Enter your choice: "))

    if choice == 1:
        meters = float(input("Enter length in meters: "))
        feet = meters * 3.28084
        print(f"{meters} meters is equal to {feet} feet")
    elif choice == 2:
        feet = float(input("Enter length in feet: "))
        meters = feet / 3.28084
        print(f"{feet} feet is equal to {meters} meters")
    else:
        print("Invalid choice")

def weight_conversion():
    """Converts weight between kilograms and pounds."""
    print("Weight Conversion")
    print("1. Kilograms to Pounds")
    print("2. Pounds to Kilograms")
    choice = int(input("Enter your choice: "))

    if choice == 1:
        kilograms = float(input("Enter weight in kilograms: "))
        pounds = kilograms * 2.20462
        print(f"{kilograms} kilograms is equal to {pounds} pounds")
    elif choice == 2:
        pounds = float(input("Enter weight in pounds: "))
        kilograms = pounds / 2.20462
        print(f"{pounds} pounds is equal to {kilograms} kilograms")
    else:
        print("Invalid choice")

def temperature_conversion():
    """Converts temperature between Celsius and Fahrenheit."""
    print("Temperature Conversion")
    print("1. Celsius to Fahrenheit")
    print("2. Fahrenheit to Celsius")
    choice = int(input("Enter your choice: "))

    if choice == 1:
        celsius = float(input("Enter temperature in Celsius: "))
        fahrenheit = (celsius * 9/5) + 32
        print(f"{celsius} Celsius is equal to {fahrenheit} Fahrenheit")
    elif choice == 2:
        fahrenheit = float(input("Enter temperature in Fahrenheit: "))
        celsius = (fahrenheit - 32) * 5/9
        print(f"{fahrenheit} Fahrenheit is equal to {celsius} Celsius")
    else:
        print("Invalid choice")

def main():
    while True:
        print("Unit Converter")
        print("1. Length Conversion")
        print("2. Weight Conversion")
        print("3. Temperature Conversion")
        print("4. Quit")
        choice = int(input("Enter your choice: "))

        if choice == 1:
            length_conversion()
        elif choice == 2:
            weight_conversion()
        elif choice == 3:
            temperature_conversion()
        elif choice == 4:
            break
        else:
            print("Invalid choice")

if __name__ == "__main__":
    main()
```
