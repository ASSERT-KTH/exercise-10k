## Exercise
Write a program to check if a given string is a rotation of another string

def is_rotation(s1, s2):
    """
    Checks if s2 is a rotation of s1.

    Args:
        s1 (str): The original string.
        s2 (str): The string to check.

    Returns:
        bool: True if s2 is a rotation of s1, False otherwise.
    """
    if len(s1) != len(s2):
        return False
    return s2 in s1 + s1

def main():
    # Example usage
    s1 = "abcde"
    s2 = "cdeab"
    s3 = "abced"

    print(f"Is '{s2}' a rotation of '{s1}'? {is_rotation(s1, s2)}")
    print(f"Is '{s3}' a rotation of '{s1}'? {is_rotation(s1, s3)}")

if __name__ == "__main__":
    main()
