## Exercise
Write a function that checks if a given key exists in a dictionary

def check_key(dictionary, key):
    """
    Checks if a given key exists in a dictionary.

    Args:
        dictionary (dict): The dictionary to search in.
        key: The key to search for.

    Returns:
        bool: True if the key exists, False otherwise.
    """
    return key in dictionary

# Example usage:
def main():
    example_dict = {"name": "John", "age": 30, "city": "New York"}
    key_to_check = "age"
    result = check_key(example_dict, key_to_check)
    print(f"Key '{key_to_check}' exists: {result}")

    key_to_check = "country"
    result = check_key(example_dict, key_to_check)
    print(f"Key '{key_to_check}' exists: {result}")

if __name__ == "__main__":
    main()
