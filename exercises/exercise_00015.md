## Exercise
Write a program to use a data structure such as a list or dictionary to store and retrieve data.

## Reference Solution
```python
class DataStorage:
    def __init__(self):
        """
        Initialize an empty dictionary to store data.
        """
        self.data = {}

    def add_data(self, name, age):
        """
        Add a new entry to the dictionary.

        Args:
            name (str): The name of the person.
            age (int): The age of the person.
        """
        self.data[name] = age
        print(f"Added {name} with age {age} to the database.")

    def retrieve_data(self, name):
        """
        Retrieve the age of a person from the dictionary.

        Args:
            name (str): The name of the person.

        Returns:
            int: The age of the person if found, otherwise None.
        """
        if name in self.data:
            return self.data[name]
        else:
            print(f"{name} not found in the database.")
            return None

    def update_data(self, name, age):
        """
        Update the age of a person in the dictionary.

        Args:
            name (str): The name of the person.
            age (int): The new age of the person.
        """
        if name in self.data:
            self.data[name] = age
            print(f"Updated {name}'s age to {age}.")
        else:
            print(f"{name} not found in the database.")

    def delete_data(self, name):
        """
        Delete a person's data from the dictionary.

        Args:
            name (str): The name of the person.
        """
        if name in self.data:
            del self.data[name]
            print(f"Deleted {name} from the database.")
        else:
            print(f"{name} not found in the database.")

    def display_data(self):
        """
        Display all the data in the dictionary.
        """
        print("Current Database:")
        for name, age in self.data.items():
            print(f"{name}: {age}")


def main():
    storage = DataStorage()

    while True:
        print("\n1. Add Data")
        print("2. Retrieve Data")
        print("3. Update Data")
        print("4. Delete Data")
        print("5. Display Data")
        print("6. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter name: ")
            age = int(input("Enter age: "))
            storage.add_data(name, age)
        elif choice == "2":
            name = input("Enter name: ")
            age = storage.retrieve_data(name)
            if age is not None:
                print(f"{name} is {age} years old.")
        elif choice == "3":
            name = input("Enter name: ")
            age = int(input("Enter new age: "))
            storage.update_data(name, age)
        elif choice == "4":
            name = input("Enter name: ")
            storage.delete_data(name)
        elif choice == "5":
            storage.display_data()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please choose a valid option.")


if __name__ == "__main__":
    main()
```
