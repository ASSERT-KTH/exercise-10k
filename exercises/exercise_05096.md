## Exercise
Create a program that simulates a conversation with the user.

## Reference Solution
```python
import random

# Define a dictionary of responses
responses = {
    'hello': ['Hi, how are you?', 'Hello! What brings you here today?', 'Hey, what\'s up?'],
    'how are you': ['I\'m doing great, thanks for asking!', 'I\'m just a computer program, I don\'t have feelings.', 'I\'m good, thanks!'],
    'what is your name': ['My name is Conversation Bot.', 'I don\'t have a personal name, but I\'m here to chat with you.', 'You can call me CB for short.'],
    'default': ['I didn\'t quite understand that.', 'Can you please rephrase?', 'I\'m not sure what you mean.']
}

def get_response(user_input):
    """Get a response based on the user's input"""
    user_input = user_input.lower()
    for keyword, response_list in responses.items():
        if keyword in user_input:
            return random.choice(response_list)
    return random.choice(responses['default'])

def main():
    print("Welcome to the conversation simulator!")
    while True:
        user_input = input("You: ")
        if user_input.lower() == "quit":
            break
        response = get_response(user_input)
        print("Conversation Bot: ", response)

if __name__ == "__main__":
    main()
```
