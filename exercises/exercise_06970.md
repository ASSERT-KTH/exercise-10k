## Exercise
Create a function that takes a list of numbers as an argument and returns the variance.

## Reference Solution
```python
import statistics

def calculate_variance(numbers):
    """
    Calculate the variance of a list of numbers.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The variance of the input list.
    """
    if len(numbers) < 2:
        raise ValueError("Input list must contain at least two numbers.")

    return statistics.variance(numbers)

# Example usage
numbers = [1, 2, 3, 4, 5]
variance = calculate_variance(numbers)
print(f"The variance of the input list is: {variance}")
```

def calculate_variance_manual(numbers):
    """
    Calculate the variance of a list of numbers manually.

    Args:
        numbers (list): A list of numbers.

    Returns:
        float: The variance of the input list.
    """
    if len(numbers) < 2:
        raise ValueError("Input list must contain at least two numbers.")

    mean = sum(numbers) / len(numbers)
    variance = sum((x - mean) ** 2 for x in numbers) / (len(numbers) - 1)
    return variance

# Example usage
numbers = [1, 2, 3, 4, 5]
variance = calculate_variance_manual(numbers)
print(f"The variance of the input list is: {variance}")
