## Exercise
Create a program that responds to basic user input with pre-defined responses and remembers the conversation history.

## Reference Solution
```python
# conversation_bot.py

class ConversationBot:
    def __init__(self):
        self.responses = {
            "hello": "Hi! How can I assist you today?",
            "goodbye": "Goodbye! It was nice talking to you.",
            "how are you": "I'm doing well, thanks for asking!",
            "what is your name": "My name is Conversation Bot.",
            "default": "I didn't quite understand that. Please try again."
        }
        self.conversation_history = []

    def get_response(self, user_input):
        user_input = user_input.lower()
        for key in self.responses:
            if key in user_input:
                return self.responses[key]
        return self.responses["default"]

    def remember_conversation(self, user_input, response):
        self.conversation_history.append(f"User: {user_input}")
        self.conversation_history.append(f"Bot: {response}")

    def display_conversation_history(self):
        print("\nConversation History:")
        for line in self.conversation_history:
            print(line)


def main():
    bot = ConversationBot()
    while True:
        user_input = input("User: ")
        response = bot.get_response(user_input)
        print(f"Bot: {response}")
        bot.remember_conversation(user_input, response)
        if user_input.lower() == "goodbye":
            bot.display_conversation_history()
            break


if __name__ == "__main__":
    main()
```
