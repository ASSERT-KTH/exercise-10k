## Exercise
Remove and return a value from a dictionary using the pop method

## Reference Solution
```python
def remove_and_return_value(dictionary, key, default=None):
    """
    Removes and returns the value of a specified key from a dictionary.
    
    Args:
    dictionary (dict): The dictionary to remove the value from.
    key: The key to remove and return the value for.
    default: The default value to return if the key is not found. Defaults to None.
    
    Returns:
    The value of the removed key, or the default value if the key is not found.
    """
    return dictionary.pop(key, default)

# Example usage:
my_dict = {'a': 1, 'b': 2, 'c': 3}

# Remove and return a value for an existing key
print(remove_and_return_value(my_dict, 'a'))  # Output: 1
print(my_dict)  # Output: {'b': 2, 'c': 3}

# Remove and return a value for a non-existent key with a default value
print(remove_and_return_value(my_dict, 'd', 'Key not found'))  # Output: Key not found
print(my_dict)  # Output: {'b': 2, 'c': 3}

# Remove and return a value for a non-existent key without a default value
try:
    print(remove_and_return_value(my_dict, 'd'))
except KeyError as e:
    print(e)  # Output: 'd'
print(my_dict)  # Output: {'b': 2, 'c': 3}
```
