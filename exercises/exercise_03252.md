## Exercise
Create a program that generates a geometric sequence given the first term, common ratio, and number of terms.

## Reference Solution
```python
def generate_geometric_sequence(first_term, common_ratio, num_terms):
    """
    Generate a geometric sequence.

    Args:
        first_term (float): The first term of the sequence.
        common_ratio (float): The common ratio of the sequence.
        num_terms (int): The number of terms in the sequence.

    Returns:
        list: A list of numbers representing the geometric sequence.
    """
    sequence = [first_term]
    for _ in range(1, num_terms):
        next_term = sequence[-1] * common_ratio
        sequence.append(next_term)
    return sequence

def main():
    # Example usage:
    first_term = float(input("Enter the first term: "))
    common_ratio = float(input("Enter the common ratio: "))
    num_terms = int(input("Enter the number of terms: "))

    sequence = generate_geometric_sequence(first_term, common_ratio, num_terms)
    print("Geometric Sequence:")
    for i, term in enumerate(sequence):
        print(f"Term {i+1}: {term}")

if __name__ == "__main__":
    main()
```
