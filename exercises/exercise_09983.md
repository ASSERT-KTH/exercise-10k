## Exercise
Use a boolean variable to determine if a person is eligible to vote based on their age

## Reference Solution
```python
def is_eligible_to_vote(age):
    """
    Checks if a person is eligible to vote based on their age.

    Args:
        age (int): The person's age.

    Returns:
        bool: True if the person is eligible to vote, False otherwise.
    """
    # Define the minimum voting age
    minimum_voting_age = 18

    # Use a boolean variable to store the eligibility status
    is_eligible = age >= minimum_voting_age

    return is_eligible

# Example usage
def main():
    person_name = "John Doe"
    person_age = 25

    eligibility_status = is_eligible_to_vote(person_age)

    if eligibility_status:
        print(f"{person_name} is eligible to vote.")
    else:
        print(f"{person_name} is not eligible to vote.")

if __name__ == "__main__":
    main()
```
