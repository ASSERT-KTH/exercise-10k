## Exercise
Write a program that calculates the mean, median, and standard deviation of a given list of numbers.

## Reference Solution
```python
import math
import statistics

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and standard deviation of a list of numbers.
    
    Args:
        numbers (list): A list of numbers.
    
    Returns:
        tuple: A tuple containing the mean, median, and standard deviation.
    """
    # Calculate the mean
    mean = sum(numbers) / len(numbers)
    
    # Calculate the median
    median = statistics.median(numbers)
    
    # Calculate the standard deviation
    variance = sum((x - mean) ** 2 for x in numbers) / len(numbers)
    std_dev = math.sqrt(variance)
    
    return mean, median, std_dev

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    mean, median, std_dev = calculate_statistics(numbers)
    
    print("Mean:", mean)
    print("Median:", median)
    print("Standard Deviation:", std_dev)

if __name__ == "__main__":
    main()
```
