## Exercise
Create a program to group a given dictionary by its values.

## Reference Solution
```python
from collections import defaultdict

def group_by_values(dictionary):
    """
    Group a dictionary by its values.

    Args:
        dictionary (dict): The dictionary to be grouped.

    Returns:
        dict: A dictionary where the keys are the original dictionary's values,
              and the values are lists of keys from the original dictionary.
    """
    grouped_dict = defaultdict(list)
    for key, value in dictionary.items():
        grouped_dict[value].append(key)
    return dict(grouped_dict)


# Example usage:
if __name__ == "__main__":
    input_dict = {
        'a': 1,
        'b': 2,
        'c': 1,
        'd': 3,
        'e': 2,
        'f': 1
    }

    grouped_dict = group_by_values(input_dict)
    print("Input Dictionary:")
    print(input_dict)
    print("\nGrouped Dictionary:")
    print(grouped_dict)
```
