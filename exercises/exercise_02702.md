## Exercise
Write a program to check the strength of a given password. A password is strong if it has at least 8 characters, 1 uppercase letter, 1 lowercase letter, and 1 digit.

## Reference Solution
```python
def check_password_strength(password):
    """
    Check the strength of a given password.

    Args:
        password (str): The password to check.

    Returns:
        bool: True if the password is strong, False otherwise.
    """
    # Check if password has at least 8 characters
    if len(password) < 8:
        return False

    # Check if password has at least 1 uppercase letter
    if not any(char.isupper() for char in password):
        return False

    # Check if password has at least 1 lowercase letter
    if not any(char.islower() for char in password):
        return False

    # Check if password has at least 1 digit
    if not any(char.isdigit() for char in password):
        return False

    return True

def main():
    password = input("Enter a password: ")
    if check_password_strength(password):
        print("The password is strong.")
    else:
        print("The password is not strong. Please ensure it has at least 8 characters, 1 uppercase letter, 1 lowercase letter, and 1 digit.")

if __name__ == "__main__":
    main()
```
