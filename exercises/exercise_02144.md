## Exercise
Write a program that reads a file, counts the number of lines, and writes the count to a new file.

## Reference Solution
```python
def count_lines(input_filename, output_filename):
    """
    Reads a file, counts the number of lines, and writes the count to a new file.

    Args:
        input_filename (str): The name of the file to read from.
        output_filename (str): The name of the file to write to.

    Returns:
        None
    """
    try:
        # Open the input file in read mode
        with open(input_filename, 'r') as input_file:
            # Count the number of lines
            line_count = sum(1 for _ in input_file)

        # Open the output file in write mode
        with open(output_filename, 'w') as output_file:
            # Write the line count to the output file
            output_file.write(str(line_count))

        print(f"Line count written to {output_filename} successfully.")

    except FileNotFoundError:
        print(f"File {input_filename} not found.")
    except Exception as e:
        print(f"An error occurred: {e}")


# Example usage
if __name__ == "__main__":
    input_filename = "input.txt"
    output_filename = "output.txt"
    count_lines(input_filename, output_filename)
```
