## Exercise
Write a program that inverts a dictionary, i.e., swaps keys and values

## Reference Solution
```python
def invert_dict(original_dict):
    """
    Invert a dictionary, swapping its keys and values.

    Args:
        original_dict (dict): The dictionary to invert.

    Returns:
        dict: The inverted dictionary.

    Raises:
        ValueError: If the original dictionary has duplicate values.
    """
    # Check for duplicate values
    if len(original_dict.values()) != len(set(original_dict.values())):
        raise ValueError("Original dictionary has duplicate values.")

    # Invert the dictionary
    inverted_dict = {v: k for k, v in original_dict.items()}

    return inverted_dict

# Example usage:
original_dict = {"a": 1, "b": 2, "c": 3}
inverted_dict = invert_dict(original_dict)
print(inverted_dict)  # Output: {1: 'a', 2: 'b', 3: 'c'}
```

def invert_dict_with_duplicates(original_dict):
    """
    Invert a dictionary, swapping its keys and values, and handling duplicate values.

    Args:
        original_dict (dict): The dictionary to invert.

    Returns:
        dict: The inverted dictionary with lists of keys for duplicate values.
    """
    inverted_dict = {}
    for k, v in original_dict.items():
        if v in inverted_dict:
            inverted_dict[v].append(k)
        else:
            inverted_dict[v] = [k]

    return inverted_dict

# Example usage:
original_dict = {"a": 1, "b": 2, "c": 1}
inverted_dict = invert_dict_with_duplicates(original_dict)
print(inverted_dict)  # Output: {1: ['a', 'c'], 2: ['b']}
