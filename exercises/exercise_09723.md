def add_binary(a, b):
    # Convert binary numbers to integers, add them, and then convert back to binary
    sum = int(a, 2) + int(b, 2)
    return bin(sum)[2:]

def add_binary(a, b):
    # Find the maximum length between a and b
    max_len = max(len(a), len(b))
    a = a.zfill(max_len)
    b = b.zfill(max_len)

    result = ''
    carry = 0

    # Iterate over the binary numbers from right to left
    for i in range(max_len-1, -1, -1):
        bit_sum = carry
        bit_sum += 1 if a[i] == '1' else 0
        bit_sum += 1 if b[i] == '1' else 0

        result = ('1' if bit_sum % 2 == 1 else '0') + result
        carry = 0 if bit_sum < 2 else 1

    if carry != 0:
        result = '1' + result

    return result
