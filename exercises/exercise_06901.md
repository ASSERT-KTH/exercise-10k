## Exercise
Create a program that takes in three numbers and performs addition, subtraction, multiplication, and division on them. The program should then print out the results of each operation.

## Reference Solution
```python
def perform_operations(num1, num2, num3):
    """
    This function takes in three numbers and performs addition, subtraction, 
    multiplication, and division on them. It then prints out the results of each operation.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        num3 (float): The third number.
    """
    # Perform operations
    add_result = num1 + num2 + num3
    sub_result_1 = num1 - num2 - num3
    sub_result_2 = num2 - num1 - num3
    sub_result_3 = num3 - num1 - num2
    mul_result = num1 * num2 * num3
    div_result_1 = num1 / num2 / num3 if num2 != 0 and num3 != 0 else "Cannot divide by zero"
    div_result_2 = num2 / num1 / num3 if num1 != 0 and num3 != 0 else "Cannot divide by zero"
    div_result_3 = num3 / num1 / num2 if num1 != 0 and num2 != 0 else "Cannot divide by zero"

    # Print results
    print("Results:")
    print(f"Addition: {num1} + {num2} + {num3} = {add_result}")
    print(f"Subtraction: {num1} - {num2} - {num3} = {sub_result_1}")
    print(f"Subtraction: {num2} - {num1} - {num3} = {sub_result_2}")
    print(f"Subtraction: {num3} - {num1} - {num2} = {sub_result_3}")
    print(f"Multiplication: {num1} * {num2} * {num3} = {mul_result}")
    print(f"Division: {num1} / {num2} / {num3} = {div_result_1}")
    print(f"Division: {num2} / {num1} / {num3} = {div_result_2}")
    print(f"Division: {num3} / {num1} / {num2} = {div_result_3}")

def main():
    # Get user input
    num1 = float(input("Enter first number: "))
    num2 = float(input("Enter second number: "))
    num3 = float(input("Enter third number: "))

    # Perform operations
    perform_operations(num1, num2, num3)

if __name__ == "__main__":
    main()
```
