## Exercise
Write a program that declares and initializes date and time variables, then prints their values.

## Reference Solution
```python
#include <iostream>
#include <ctime>

using namespace std;

int main() {
    tm date = {};
    time_t date_time;
    time(&date_time);
    localtime_r(&date_time, &date);

    cout << date.tm_year << endl;
    cout << date.tm_mon << endl;
    cout << date.tm_mday << endl;
    cout << date.tm_hour << endl;
    cout << date.tm_min << endl;
    cout << date.tm_sec << endl;

    return 0;
}
```
