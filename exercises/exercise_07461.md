## Exercise
Write a program that checks the data type of a given variable and prints it out

## Reference Solution
```python
# Define a function to check the data type
def check_data_type(var):
    """
    Checks the data type of a given variable and prints it out.

    Args:
        var (any): The variable to check the data type for.
    """
    print(f"The data type of the variable is: {type(var)}")


# Example usage
if __name__ == "__main__":
    # Test with different data types
    check_data_type(10)  # integer
    check_data_type(10.5)  # float
    check_data_type("Hello, World!")  # string
    check_data_type(True)  # boolean
    check_data_type([1, 2, 3])  # list
    check_data_type((1, 2, 3))  # tuple
    check_data_type({"a": 1, "b": 2})  # dictionary
    check_data_type({1, 2, 3})  # set
    check_data_type(None)  # NoneType
```
