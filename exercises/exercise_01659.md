## Exercise
Work with date and time objects to perform basic operations.

## Reference Solution
```python
from datetime import datetime, timedelta, date
```

```python
# Create a date and time object
now = datetime.now()
print(now)

# Create a date object
today = date.today()
print(today)

# Create a time object
time_now = datetime.now().time()

```

```python
# Calculate the date 30 days from now
future_date = now + timedelta(days=30)
print(future_date)

# Calculate the date 1 year ago
past_date = now - timedelta(days=365)
print(past_date)
```

```python

print(formatted_now)

# Format the date as a string
formatted_today = today.strftime("%Y-%m-%d")

# Format the time as a string

print(formatted_time_now)
```

```python
# Parse a date and time string
parsed_now = datetime.strptime("2022-01-01 12:00:00", "%Y-%m-%d %H:%M:%S")
print(parsed_now)

# Parse a date string
parsed_today = datetime.strptime("2022-01-01", "%Y-%m-%d").date()
print(parsed_today)

# Parse a time string
parsed_time_now = datetime.strptime("12:00:00", "%H:%M:%S").time()
print(parsed_time_now)
```

```python
from datetime import datetime

def calculate_days_between_dates(date1, date2):
    date1 = datetime.strptime(date1, "%Y-%m-%d")
    date2 = datetime.strptime(date2, "%Y-%m-%d")
    return abs((date2 - date1).days)

date1 = "2022-01-01"
date2 = "2022-01-15"
days_between_dates = calculate_days_between_dates(date1, date2)
print(f"The number of days between {date1} and {date2} is {days_between_dates}.")
```
