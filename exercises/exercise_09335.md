## Exercise
Write a program to demonstrate the difference between pass by value and pass by reference.

## Reference Solution
```python
def pass_by_value(x):
    """Pass by value example"""
    print("Inside pass_by_value function:")
    print("Initial value:", x)
    x = 20  # Changing the value
    print("Updated value:", x)

def pass_by_reference(lst):
    """Pass by reference example"""
    print("Inside pass_by_reference function:")
    print("Initial list:", lst)
    lst.append(20)  # Modifying the list
    print("Updated list:", lst)

def main():
    # Pass by value demonstration
    x = 10
    print("Outside pass_by_value function:")
    print("Initial value:", x)
    pass_by_value(x)
    print("Final value:", x)

    # Pass by reference demonstration
    lst = [1, 2, 3]
    print("\nOutside pass_by_reference function:")
    print("Initial list:", lst)
    pass_by_reference(lst)
    print("Final list:", lst)

if __name__ == "__main__":
    main()
```
