## Exercise
Write a function to find all occurrences of a pattern in a text.

## Reference Solution

def compute_prefix_function(pattern):
    """
    Compute the prefix function used in the KMP algorithm.

    Args:
    pattern (str): The pattern to search for.

    Returns:
    list: The prefix function.
    """
    prefix = [0] * len(pattern)
    j = 0
    for i in range(1, len(pattern)):
        while j > 0 and pattern[j] != pattern[i]:
            j = prefix[j - 1]
        if pattern[j] == pattern[i]:
            j += 1
        prefix[i] = j
    return prefix


def find_pattern(text, pattern):
    """
    Find all occurrences of a pattern in a text.

    Args:
    text (str): The text to search in.
    pattern (str): The pattern to search for.

    Returns:
    list: A list of indices where the pattern occurs.
    """
    prefix = compute_prefix_function(pattern)
    occurrences = []
    j = 0
    for i in range(len(text)):
        while j > 0 and text[i] != pattern[j]:
            j = prefix[j - 1]
        if text[i] == pattern[j]:
            j += 1
        if j == len(pattern):
            occurrences.append(i - j + 1)
            j = prefix[j - 1]
    return occurrences


# Example usage
text = "abxabcabcaby"
pattern = "abc"
occurrences = find_pattern(text, pattern)
print(f"Pattern '{pattern}' found at indices: {occurrences}")
