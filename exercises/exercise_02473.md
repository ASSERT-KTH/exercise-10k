## Exercise
Create a function that checks if a given string is a valid credit card number.

def validate_credit_card_number(card_number: str) -> bool:
    """
    Validate a credit card number using the Luhn algorithm.

    Args:
    card_number (str): The credit card number to validate.

    Returns:
    bool: True if the credit card number is valid, False otherwise.
    """

    # Remove any non-digit characters from the card number
    card_number = ''.join(filter(str.isdigit, card_number))

    # Check if the card number is empty or has less than 13 digits
    if len(card_number) < 13 or len(card_number) > 16:
        return False

    # Apply the Luhn algorithm
    sum = 0
    for i, digit in enumerate(reversed(card_number)):
        digit = int(digit)
        if i % 2 == 1:
            digit *= 2
            if digit > 9:
                digit -= 9
        sum += digit

    # Check if the sum is divisible by 10
    return sum % 10 == 0

if __name__ == "__main__":
    card_number = "4532015112830366"
    if validate_credit_card_number(card_number):
        print(f"The credit card number {card_number} is valid.")
    else:
        print(f"The credit card number {card_number} is not valid.")
