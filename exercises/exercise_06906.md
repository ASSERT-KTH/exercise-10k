## Exercise
Write a Python program that takes a dictionary as input where keys are student names and values are dictionaries containing their grades in different subjects. The program should calculate the average grade for each student and the overall average grade for the class. It should also find the student with the highest average grade.

## Reference Solution
```python
def calculate_average_grade(student_grades):
    """
    Calculate the average grade for each student and the overall average grade for the class.
    
    Args:
    student_grades (dict): A dictionary where keys are student names and values are dictionaries containing their grades in different subjects.
    
    Returns:
    dict: A dictionary containing the average grade for each student.
    float: The overall average grade for the class.
    str: The student with the highest average grade.
    """
    # Initialize variables to store the overall sum of grades and the student with the highest average grade
    overall_sum = 0
    overall_count = 0
    highest_average_grade = 0
    highest_average_grade_student = ""

    # Initialize a dictionary to store the average grade for each student
    average_grades = {}

    # Iterate over each student in the input dictionary
    for student, grades in student_grades.items():
        # Calculate the sum of grades for the current student
        grade_sum = sum(grades.values())
        
        # Calculate the average grade for the current student
        average_grade = grade_sum / len(grades)
        
        # Update the overall sum of grades and the count of grades
        overall_sum += grade_sum
        overall_count += len(grades)
        
        # Update the average grade for the current student in the dictionary
        average_grades[student] = average_grade
        
        # Check if the current student has the highest average grade
        if average_grade > highest_average_grade:
            highest_average_grade = average_grade
            highest_average_grade_student = student

    # Calculate the overall average grade for the class
    overall_average_grade = overall_sum / overall_count

    return average_grades, overall_average_grade, highest_average_grade_student


def main():
    # Example usage
    student_grades = {
        "John": {"Math": 90, "Science": 85, "English": 95},
        "Alice": {"Math": 95, "Science": 90, "English": 85},
        "Bob": {"Math": 80, "Science": 75, "English": 90}
    }

    average_grades, overall_average_grade, highest_average_grade_student = calculate_average_grade(student_grades)

    print("Average Grades:")
    for student, average_grade in average_grades.items():
        print(f"{student}: {average_grade:.2f}")

    print(f"\nOverall Average Grade: {overall_average_grade:.2f}")
    print(f"Student with the Highest Average Grade: {highest_average_grade_student}")


if __name__ == "__main__":
    main()
```
