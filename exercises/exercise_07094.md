## Exercise
Read a positive integer from the user and print a sequence of numbers from that number down to 1

## Reference Solution
```python
def get_positive_integer_from_user():
    """Get a positive integer from the user."""
    while True:
        try:
            num = int(input("Enter a positive integer: "))
            if num > 0:
                return num
            else:
                print("Please enter a positive integer.")
        except ValueError:
            print("Invalid input. Please enter an integer.")

def print_count_down_sequence(num):
    """Print a sequence of numbers from num down to 1."""
    for i in range(num, 0, -1):
        print(i)

def main():
    num = get_positive_integer_from_user()
    print_count_down_sequence(num)

if __name__ == "__main__":
    main()
```
