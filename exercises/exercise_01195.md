## Exercise
Write a program that stores student grades in a dictionary and calculates the average grade.

## Reference Solution
```python
def calculate_average_grade(grades):
    """
    Calculate the average grade for a given dictionary of student grades.

    Args:
        grades (dict): Dictionary where keys are student names and values are lists of grades.

    Returns:
        dict: Dictionary with student names as keys and average grades as values.
    """
    average_grades = {}
    for student, grade_list in grades.items():
        average_grade = sum(grade_list) / len(grade_list)
        average_grades[student] = average_grade
    return average_grades


def main():
    # Example usage:
    student_grades = {
        "John": [90, 85, 95],
        "Alice": [92, 88, 91],
        "Bob": [78, 82, 85]
    }

    average_grades = calculate_average_grade(student_grades)

    print("Student Grades:")
    for student, grade_list in student_grades.items():
        print(f"{student}: {grade_list}")

    print("\nAverage Grades:")
    for student, average_grade in average_grades.items():
        print(f"{student}: {average_grade:.2f}")


if __name__ == "__main__":
    main()
```
