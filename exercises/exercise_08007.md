## Exercise
Write a program that takes in a user's name and age as input, then prints out a personalized greeting message along with the user's age and a message indicating whether the user is eligible to vote or not, assuming the voting age is 18.

## Reference Solution
```python
# Importing the built-in `input` function
input = input

# Defining the `get_user_name` function
def get_user_name():
    return input("Enter your name: ")

# Defining the `greet_user` function
def greet_user(name):
    print(f"Hello, {name}!")

# Defining the `get_user_age` function
def get_user_age():
    return int(input("Enter your age: "))

# Defining the `print_user_age` function
def print_user_age(age):
    print(f"You are {age} years old.")

# Defining the `is_eligible_to_vote` function
def is_eligible_to_vote(age):
    return age >= 18

# Defining the `print_voting_status` function
def print_voting_status(is_eligible):
    if is_eligible:
        print("You are eligible to vote.")
    else:
        print("You are not eligible to vote.")

# Defining the `main` function
def main():
    user_name = get_user_name()
    user_age = get_user_age()

    greet_user(user_name)
    print_user_age(user_age)
    print_voting_status(is_eligible_to_vote(user_age))

# Calling the `main` function
main()
```
