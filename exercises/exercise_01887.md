## Exercise
Perform basic string manipulation operations (concatenation, substring) on a given string

```python
def concatenate_strings(str1, str2):
    """Concatenate two strings using the + operator."""
    result = str1 + str2
    return result

# Example usage:
str1 = "Hello, "
str2 = "World!"
result = concatenate_strings(str1, str2)
print(result)  # Output: Hello, World!
```

```python
def concatenate_strings(str1, str2):
    """Concatenate two strings using the join() method."""
    result = "".join([str1, str2])
    return result

# Example usage:
str1 = "Hello, "
str2 = "World!"
result = concatenate_strings(str1, str2)
print(result)  # Output: Hello, World!
```

```python
def extract_substring(input_str, start, end):
    """Extract a substring from the input string using slicing."""
    result = input_str[start:end]
    return result

# Example usage:
input_str = "Hello, World!"
start = 7
end = 12
result = extract_substring(input_str, start, end)
print(result)  # Output: World
```
