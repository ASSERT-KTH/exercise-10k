## Exercise
Create a program that searches for a book in a library using if/else statements

## Reference Solution
```python
# Library Management System

# List of books in the library
books = [
    {"title": "To Kill a Mockingbird", "author": "Harper Lee", "available": True},
    {"title": "1984", "author": "George Orwell", "available": False},
    {"title": "Pride and Prejudice", "author": "Jane Austen", "available": True},
    {"title": "The Great Gatsby", "author": "F. Scott Fitzgerald", "available": True},
    {"title": "The Catcher in the Rye", "author": "J.D. Salinger", "available": False},
]

def search_book_by_title(title):
    """Search for a book by title."""
    for book in books:
        if book["title"].lower() == title.lower():
            return book
    return None

def search_book_by_author(author):
    """Search for a book by author."""
    for book in books:
        if book["author"].lower() == author.lower():
            return book
    return None

def main():
    print("Welcome to the Library Management System!")
    while True:
        print("\nOptions:")
        print("1. Search for a book by title")
        print("2. Search for a book by author")
        print("3. Exit")
        
        option = input("Choose an option: ")
        
        if option == "1":
            title = input("Enter the title of the book: ")
            book = search_book_by_title(title)
            if book:
                print(f"\nBook found: {book['title']} by {book['author']}")
                if book["available"]:
                    print("The book is available.")
                else:
                    print("The book is not available.")
            else:
                print("\nBook not found.")
        elif option == "2":
            author = input("Enter the author of the book: ")
            book = search_book_by_author(author)
            if book:
                print(f"\nBook found: {book['title']} by {book['author']}")
                if book["available"]:
                    print("The book is available.")
                else:
                    print("The book is not available.")
            else:
                print("\nBook not found.")
        elif option == "3":
            print("\nExiting the system. Goodbye!")
            break
        else:
            print("\nInvalid option. Please choose a valid option.")

if __name__ == "__main__":
    main()
```
