## Exercise
Write a program that takes in two numbers and performs addition, subtraction, multiplication, and division based on the user's choice. The program should handle cases where the user chooses division and the second number is zero.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int a, b, n;
    char ch;

    cout << "Enter first number: ";
    cin >> a;

    cout << "Enter second number: ";
    cin >> b;

    cout << "Enter your choice (+, -, *, /): ";
    cin >> ch;

    switch (ch) {
        case '+':
            cout << "Result: " << a + b << endl;
            break;
        case '-':
            cout << "Result: " << a - b << endl;
            break;
        case '*':
            cout << "Result: " << a * b << endl;
            break;
        case '/':
            if (b == 0) {
                cout << "Error: Division by zero is not allowed." << endl;
            } else {
                cout << "Result: " << (float)a / b << endl;
            }
            break;
        default:
            cout << "Invalid choice. Please choose +, -, *, or /." << endl;
            break;
    }

    return 0;
}
```
