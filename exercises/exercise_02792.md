## Exercise
Write a program that performs basic matrix operations such as addition and multiplication on two given arrays.

## Reference Solution
```python
import numpy as np

def add_matrices(matrix1, matrix2):
    # Check if the matrices have the same dimensions
    if matrix1.shape != matrix2.shape:
        raise ValueError("Matrices must have the same dimensions for addition")
    
    return np.add(matrix1, matrix2)

def multiply_matrices(matrix1, matrix2):
    # Check if the number of columns in the first matrix is equal to the number of rows in the second matrix
    if matrix1.shape[1] != matrix2.shape[0]:
        raise ValueError("Number of columns in the first matrix must be equal to the number of rows in the second matrix for multiplication")
    
    return np.matmul(matrix1, matrix2)

def main():
    # Define two example matrices
    matrix1 = np.array([[1, 2, 3], [4, 5, 6]])
    matrix2 = np.array([[7, 8, 9], [10, 11, 12]])

    print("Matrix 1:")
    print(matrix1)
    print("\nMatrix 2:")
    print(matrix2)

    # Perform addition
    try:
        result_addition = add_matrices(matrix1, matrix2)
        print("\nResult of addition:")
        print(result_addition)
    except ValueError as e:
        print(f"Error: {e}")

    # Perform multiplication
    try:
        result_multiplication = multiply_matrices(matrix1, matrix2)
        print("\nResult of multiplication:")
        print(result_multiplication)
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
