## Exercise
Write a program that uses an if/else statement to validate a password based on certain criteria.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validates a password based on certain criteria.

    Args:
        password (str): The password to validate.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Define the criteria
    criteria = {
        "length": len(password) >= 8,
        "uppercase": any(char.isupper() for char in password),
        "lowercase": any(char.islower() for char in password),
        "digit": any(char.isdigit() for char in password),
        "special_char": bool(re.search(r"[^A-Za-z0-9]", password))
    }

    # Check if all criteria are met
    if all(criteria.values()):
        print("Password is valid.")
        return True
    else:
        print("Password is not valid.")
        # Print the unmet criteria
        for key, value in criteria.items():
            if not value:
                print(f"  - {key.capitalize().replace('_', ' ')} is required.")
        return False

# Example usage
def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
