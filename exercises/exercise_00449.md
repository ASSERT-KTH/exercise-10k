## Exercise
Write a program that asks the user for their location, then prints out the current weather for that location.

## Reference Solution
```python
import requests
import json

def get_current_weather(api_key, location):
    """
    Retrieves the current weather for a given location.

    Args:
        api_key (str): OpenWeatherMap API key
        location (str): City or zip code

    Returns:
        dict: Current weather conditions
    """
    base_url = "http://api.openweathermap.org/data/2.5/weather"
    params = {
        "q": location,
        "appid": api_key,
        "units": "imperial"  # Use imperial units (Fahrenheit)
    }
    response = requests.get(base_url, params=params)
    return response.json()

def main():
    api_key = input("Enter your OpenWeatherMap API key: ")
    location = input("Enter your location (city or zip code): ")
    weather_data = get_current_weather(api_key, location)
    print("Current Weather:")
    print(f"Location: {weather_data['name']}, {weather_data['sys']['country']}")
    print(f"Weather: {weather_data['weather'][0]['description']}")

if __name__ == "__main__":
```
