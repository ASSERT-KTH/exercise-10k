## Exercise
Write a program that redirects input from one source to another

## Reference Solution
```python
import sys

def redirect_input(src, dst):
    """
    Redirects input from src to dst.

    Args:
        src (str): The source file path.
        dst (str): The destination file path.

    Returns:
        None
    """
    try:
        # Open the source file in read mode
        with open(src, 'r') as f_in:
            # Read the contents of the source file
            data = f_in.read()
        
        # Open the destination file in write mode
        with open(dst, 'w') as f_out:
            # Write the contents of the source file to the destination file
            f_out.write(data)
        
        print(f"Input successfully redirected from {src} to {dst}")
    
    except FileNotFoundError:
        print(f"Source file {src} not found.")
    except Exception as e:
        print(f"An error occurred: {e}")

def main():
    # Check if the correct number of arguments are provided
    if len(sys.argv) != 3:
        print("Usage: python input_redirect.py <source_file> <destination_file>")
        return
    
    # Get the source and destination file paths from the command line arguments
    src = sys.argv[1]
    dst = sys.argv[2]
    
    # Call the redirect_input function
    redirect_input(src, dst)

if __name__ == "__main__":
    main()
```
