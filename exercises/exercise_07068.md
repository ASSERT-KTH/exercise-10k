## Exercise
Write a program that takes a string as input and encrypts it using a simple encryption algorithm.

## Reference Solution
```python
def encrypt_string(input_string, shift):
    """
    Encrypts a string using a Caesar cipher algorithm.

    Args:
        input_string (str): The string to be encrypted.
        shift (int): The number of positions to shift each character.

    Returns:
        str: The encrypted string.
    """
    encrypted_string = ""
    for char in input_string:
        if char.isalpha():
            ascii_offset = 65 if char.isupper() else 97
            encrypted_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encrypted_string += encrypted_char
        else:
            encrypted_string += char
    return encrypted_string

def main():
    input_string = input("Enter a string to encrypt: ")
    shift = int(input("Enter the shift value: "))
    encrypted_string = encrypt_string(input_string, shift)
    print("Encrypted string:", encrypted_string)

if __name__ == "__main__":
    main()
```
