## Exercise
Create a program to check whether a given substring is present in a string or not using a basic algorithm

## Reference Solution
```python
def check_substring(main_string, sub_string):
    """
    Checks if a substring is present in a main string.

    Args:
        main_string (str): The main string to search in.
        sub_string (str): The substring to search for.

    Returns:
        bool: True if the substring is found, False otherwise.
    """
    main_string_len = len(main_string)
    sub_string_len = len(sub_string)

    for i in range(main_string_len - sub_string_len + 1):
        match = True
        for j in range(sub_string_len):
            if main_string[i + j] != sub_string[j]:
                match = False
                break
        if match:
            return True

    return False

def main():
    main_string = input("Enter the main string: ")
    sub_string = input("Enter the substring to search for: ")

    if check_substring(main_string, sub_string):
        print(f"'{sub_string}' is present in '{main_string}'")
    else:
        print(f"'{sub_string}' is not present in '{main_string}'")

if __name__ == "__main__":
    main()
```
