## Exercise
Create a function to group a dictionary by a common key (e.g., group by 'category').

## Reference Solution
```python
from typing import List, Dict, Any

def group_dict(dict_list: List[Dict[str, Any]], key: str) -> Dict[Any, List[Dict[str, Any]]]:
    """
    Groups a list of dictionaries by a common key.

    Args:
    - dict_list (List[Dict[str, Any]]): The list of dictionaries to group.
    - key (str): The key to group by.

    Returns:
    - Dict[Any, List[Dict[str, Any]]]: A new dictionary where the keys are the unique values from the specified key and the values are lists of dictionaries that have the same key value.
    """
    grouped_dict = {}
    for dictionary in dict_list:
        if key in dictionary:
            if dictionary[key] in grouped_dict:
                grouped_dict[dictionary[key]].append(dictionary)
            else:
                grouped_dict[dictionary[key]] = [dictionary]
    return grouped_dict

# Example usage
if __name__ == "__main__":
    dict_list = [
        {"id": 1, "category": "A", "name": "Item 1"},
        {"id": 2, "category": "B", "name": "Item 2"},
        {"id": 3, "category": "A", "name": "Item 3"},
        {"id": 4, "category": "B", "name": "Item 4"},
        {"id": 5, "category": "C", "name": "Item 5"}
    ]

    key = "category"
    grouped = group_dict(dict_list, key)
    print(grouped)
```
