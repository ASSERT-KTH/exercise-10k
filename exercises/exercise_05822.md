## Exercise
Extract a substring from a given string based on a start and end index

def extract_substring(input_string, start_index, end_index):
    """
    Extracts a substring from the input string based on the start and end index.

    Args:
        input_string (str): The input string.
        start_index (int): The start index of the substring.
        end_index (int): The end index of the substring.

    Returns:
        str: The extracted substring.

    Raises:
        ValueError: If the start index is greater than the end index.
        IndexError: If the start or end index is out of range.
    """
    if start_index > end_index:
        raise ValueError("Start index cannot be greater than end index")

    try:
        return input_string[start_index:end_index + 1]
    except IndexError:
        raise IndexError("Start or end index is out of range")

# Example usage
input_string = "Hello, World!"
start_index = 7
end_index = 12

try:
    substring = extract_substring(input_string, start_index, end_index)
    print(substring)  # Output: "World"
except (ValueError, IndexError) as e:
    print(e)
