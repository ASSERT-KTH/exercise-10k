## Exercise
Write a program that takes two arrays as input and returns a new array containing the elements from the first array paired with the corresponding elements from the second array.

## Reference Solution
```python
#include <iostream>
using namespace std;

int main() {
    int ar1[] = {1, 2, 3, 4, 5};
    int ar2[] = {6, 7, 8, 9, 10};

    int n = sizeof(ar1) / sizeof(ar1[0]);

    for (int i = 0; i < n; i++) {
        cout << "(" << ar1[i] << ", " << ar2[i] << ")" << endl;
    }

    return 0;
}
```
