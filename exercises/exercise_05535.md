## Exercise
Write a program to use regular expressions for pattern matching in strings.

## Reference Solution
```python
import re

def match_pattern(string, pattern):
    """
    Match a regular expression pattern in a string.

    Args:
        string (str): The input string to search.
        pattern (str): The regular expression pattern to match.

    Returns:
        list: A list of matches found in the string.
    """
    try:
        # Compile the regular expression pattern
        regex = re.compile(pattern)

        # Find all matches in the string
        matches = regex.findall(string)

        return matches
    except re.error as e:
        print(f"Invalid regular expression: {e}")
        return []

def main():
    # Example usage
    string = "Hello, my email is example@example.com and my phone number is 123-456-7890."
    pattern = r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b"  # Email pattern
    matches = match_pattern(string, pattern)

    print("Matches:")
    for match in matches:
        print(match)

    # Example usage with phone number pattern
    pattern = r"\d{3}-\d{3}-\d{4}"  # Phone number pattern
    matches = match_pattern(string, pattern)

    print("\nMatches:")
    for match in matches:
        print(match)

if __name__ == "__main__":
    main()
```
