## Exercise
Create a function that takes in a dictionary and returns a new dictionary containing the same key-value pairs, but with the values grouped by a common key

## Reference Solution
```python
def group_dict_values(input_dict, common_key):
    """
    This function groups dictionary values by a common key.
    
    Args:
        input_dict (dict): The input dictionary.
        common_key (str): The common key to group by.
    
    Returns:
        dict: A new dictionary with values grouped by the common key.
    """
    # Initialize an empty dictionary to store the grouped values
    grouped_dict = {}
    
    # Iterate over the key-value pairs in the input dictionary
    for key, value in input_dict.items():
        # Check if the value is a dictionary and has the common key
        if isinstance(value, dict) and common_key in value:
            # Get the value of the common key
            common_value = value[common_key]
            
            # If the common value is not already a key in the grouped dictionary, add it
            if common_value not in grouped_dict:
                grouped_dict[common_value] = []
            
            # Append the key-value pair to the list of values for the common key
            grouped_dict[common_value].append({key: value})
    
    return grouped_dict

# Example usage
input_dict = {
    'item1': {'category': 'A', 'price': 10},
    'item2': {'category': 'B', 'price': 20},
    'item3': {'category': 'A', 'price': 30},
    'item4': {'category': 'B', 'price': 40}
}

common_key = 'category'

grouped_dict = group_dict_values(input_dict, common_key)

print(grouped_dict)
```
