## Exercise
Write a program to reduce an array to a single value.

## Reference Solution
```python
import functools
import operator

def reduce_array(array, reduction_function):
    """
    Reduces an array to a single value using a specified reduction function.

    Args:
        array (list): The input array.
        reduction_function (function): The reduction function to apply.

    Returns:
        The reduced value.
    """
    return functools.reduce(reduction_function, array)

# Example usage
if __name__ == "__main__":
    # Define an example array
    array = [1, 2, 3, 4, 5]

    # Reduce the array using the sum function
    reduced_sum = reduce_array(array, operator.add)
    print(f"Reduced sum: {reduced_sum}")

    # Reduce the array using the product function
    reduced_product = reduce_array(array, operator.mul)
    print(f"Reduced product: {reduced_product}")

    # Reduce the array using the maximum function
    reduced_max = reduce_array(array, max)
    print(f"Reduced max: {reduced_max}")

    # Reduce the array using the minimum function
    reduced_min = reduce_array(array, min)
    print(f"Reduced min: {reduced_min}")
```
