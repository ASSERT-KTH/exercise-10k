## Exercise
Write a program that declares and operates on complex numbers.

## Reference Solution
```python
class ComplexNumber:
    def __init__(self, real=0, imaginary=0):
        """
        Initialize a complex number.

        Args:
            real (float): The real part of the complex number.
            imaginary (float): The imaginary part of the complex number.
        """
        self.real = real
        self.imaginary = imaginary

    def __add__(self, other):
        """
        Add two complex numbers.

        Args:
            other (ComplexNumber): The other complex number to add.

        Returns:
            ComplexNumber: The sum of the two complex numbers.
        """
        real = self.real + other.real
        imaginary = self.imaginary + other.imaginary
        return ComplexNumber(real, imaginary)

    def __sub__(self, other):
        """
        Subtract two complex numbers.

        Args:
            other (ComplexNumber): The other complex number to subtract.

        Returns:
            ComplexNumber: The difference of the two complex numbers.
        """
        real = self.real - other.real
        imaginary = self.imaginary - other.imaginary
        return ComplexNumber(real, imaginary)

    def __mul__(self, other):
        """
        Multiply two complex numbers.

        Args:
            other (ComplexNumber): The other complex number to multiply.

        Returns:
            ComplexNumber: The product of the two complex numbers.
        """
        real = self.real * other.real - self.imaginary * other.imaginary
        imaginary = self.real * other.imaginary + self.imaginary * other.real
        return ComplexNumber(real, imaginary)

    def __truediv__(self, other):
        """
        Divide two complex numbers.

        Args:
            other (ComplexNumber): The other complex number to divide.

        Returns:
            ComplexNumber: The quotient of the two complex numbers.

        Raises:
            ZeroDivisionError: If the other complex number is zero.
        """
        if other.real == 0 and other.imaginary == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        denominator = other.real ** 2 + other.imaginary ** 2
        real = (self.real * other.real + self.imaginary * other.imaginary) / denominator
        imaginary = (self.imaginary * other.real - self.real * other.imaginary) / denominator
        return ComplexNumber(real, imaginary)

    def __str__(self):
        """
        Return a string representation of the complex number.

        Returns:
            str: A string representation of the complex number.
        """
        if self.imaginary >= 0:
            return f"{self.real} + {self.imaginary}i"
        else:
            return f"{self.real} - {abs(self.imaginary)}i"


def main():
    # Create two complex numbers
    num1 = ComplexNumber(3, 4)
    num2 = ComplexNumber(1, 2)

    # Add the complex numbers
    sum_num = num1 + num2
    print(f"Sum: {sum_num}")

    # Subtract the complex numbers
    diff_num = num1 - num2
    print(f"Difference: {diff_num}")

    # Multiply the complex numbers
    prod_num = num1 * num2
    print(f"Product: {prod_num}")

    # Divide the complex numbers
    quot_num = num1 / num2
    print(f"Quotient: {quot_num}")


if __name__ == "__main__":
    main()
```
