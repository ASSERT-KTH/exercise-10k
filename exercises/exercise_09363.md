## Exercise
Write a program that checks the data type of a variable.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a variable.

    Args:
        var: The variable to check.

    Returns:
        The data type of the variable.
    """
    return type(var)

def main():
    # Example usage
    var_int = 10
    var_float = 10.5
    var_str = "Hello"
    var_bool = True
    var_list = [1, 2, 3]
    var_dict = {"a": 1, "b": 2}

    print(f"Data type of {var_int}: {check_data_type(var_int)}")
    print(f"Data type of {var_float}: {check_data_type(var_float)}")
    print(f"Data type of {var_str}: {check_data_type(var_str)}")
    print(f"Data type of {var_bool}: {check_data_type(var_bool)}")
    print(f"Data type of {var_list}: {check_data_type(var_list)}")
    print(f"Data type of {var_dict}: {check_data_type(var_dict)}")

if __name__ == "__main__":
    main()
```
