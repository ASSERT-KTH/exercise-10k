## Exercise
Create a Personal Finance Tracker program using variables and data structures.

## Reference Solution
```python
# Import the datetime module for date and time functionality
import datetime

# Define a class for transactions
class Transaction:
    def __init__(self, date, description, amount, type):
        self.date = date
        self.description = description
        self.amount = amount
        self.type = type

# Define a class for the finance tracker
class FinanceTracker:
    def __init__(self):
        self.transactions = []
        self.balance = 0.0

    # Method to add a transaction
    def add_transaction(self, description, amount, type):
        date = datetime.date.today()
        transaction = Transaction(date, description, amount, type)
        self.transactions.append(transaction)
        if type == "income":
            self.balance += amount
        elif type == "expense":
            self.balance -= amount
        print(f"Transaction added: {description} - {amount}")

    # Method to remove a transaction
    def remove_transaction(self, index):
        if index < len(self.transactions):
            transaction = self.transactions.pop(index)
            if transaction.type == "income":
                self.balance -= transaction.amount
            elif transaction.type == "expense":
                self.balance += transaction.amount
            print(f"Transaction removed: {transaction.description} - {transaction.amount}")
        else:
            print("Invalid transaction index")

    # Method to view all transactions
    def view_transactions(self):
        print("Transactions:")
        for i, transaction in enumerate(self.transactions):
            print(f"{i+1}. {transaction.date} - {transaction.description} - {transaction.amount} ({transaction.type})")

    # Method to view the current balance
    def view_balance(self):
        print(f"Current balance: {self.balance}")

# Create a finance tracker instance
tracker = FinanceTracker()

# Main program loop
while True:
    print("\nPersonal Finance Tracker")
    print("1. Add transaction")
    print("2. Remove transaction")
    print("3. View transactions")
    print("4. View balance")
    print("5. Quit")
    choice = input("Enter your choice: ")

    if choice == "1":
        description = input("Enter transaction description: ")
        amount = float(input("Enter transaction amount: "))
        type = input("Enter transaction type (income/expense): ")
        tracker.add_transaction(description, amount, type)
    elif choice == "2":
        tracker.view_transactions()
        index = int(input("Enter transaction index to remove: ")) - 1
        tracker.remove_transaction(index)
    elif choice == "3":
        tracker.view_transactions()
    elif choice == "4":
        tracker.view_balance()
    elif choice == "5":
        break
    else:
        print("Invalid choice")
```
