## Exercise
Write a program that takes a bill amount as input and prints out the calculated tip

## Reference Solution
```python
def calculate_tip(bill_amount, tip_percentage):
    """
    Calculates the tip based on the bill amount and tip percentage.

    Args:
        bill_amount (float): The total bill amount.
        tip_percentage (float): The tip percentage.

    Returns:
        float: The calculated tip.
    """
    tip = (bill_amount / 100) * tip_percentage
    return tip

def main():
    print("Tip Calculator Program")
    print("------------------------")

    # Get the bill amount from the user
    while True:
        try:
            bill_amount = float(input("Enter the bill amount: $"))
            if bill_amount < 0:
                print("Bill amount cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Get the tip percentage from the user
    while True:
        try:
            tip_percentage = float(input("Enter the tip percentage (e.g., 15 for 15%): "))
            if tip_percentage < 0:
                print("Tip percentage cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    # Calculate the tip
    tip = calculate_tip(bill_amount, tip_percentage)

    # Print the result
    print(f"Bill Amount: ${bill_amount:.2f}")
    print(f"Tip Percentage: {tip_percentage}%")
    print(f"Calculated Tip: ${tip:.2f}")
    print(f"Total Amount: ${bill_amount + tip:.2f}")

if __name__ == "__main__":
    main()
```
