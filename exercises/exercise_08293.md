## Exercise
Write a program that calculates the point of intersection between two line segments.

def calculate_intersection(line1, line2):
    """
    Calculate the intersection point of two line segments.

    Args:
        line1 (list): Two points defining the first line segment, 
                      where each point is a tuple of two floats.
        line2 (list): Two points defining the second line segment, 
                      where each point is a tuple of two floats.

    Returns:
        tuple or None: The intersection point if the lines intersect, 
                       otherwise None.
    """

    # Unpack the points
    (x1, y1), (x2, y2) = line1
    (x3, y3), (x4, y4) = line2

    # Calculate the denominator
    denominator = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)

    # If the denominator is zero, the lines are parallel
    if denominator == 0:
        return None

    # Calculate the numerators
    t_numerator = (x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)
    u_numerator = (x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3)

    # Calculate the parameters
    t = t_numerator / denominator
    u = u_numerator / denominator

    # Check if the intersection point lies within the line segments
    if 0 <= t <= 1 and 0 <= u <= 1:
        # Calculate the intersection point
        intersection_x = x1 + t * (x2 - x1)
        intersection_y = y1 + t * (y2 - y1)
        return (intersection_x, intersection_y)
    else:
        return None


# Example usage
line1 = [(1, 1), (3, 3)]
line2 = [(1, 3), (3, 1)]

intersection = calculate_intersection(line1, line2)
if intersection:
    print(f"The lines intersect at {intersection}")
else:
    print("The lines do not intersect")
