## Exercise
Develop a program that analyzes a given text and prints out statistics (e.g., word count, character count, most frequent words). The program should ask for the text, and then print out the statistics.

## Reference Solution
```python
import re
from collections import Counter

def calculate_statistics(text):
    # Remove punctuation
    text_without_punctuation = re.sub(r'[^\w\s]', '', text)

    # Calculate word count
    words = text_without_punctuation.split()
    word_count = len(words)

    # Calculate character count
    character_count = len(text)

    # Calculate most frequent words
    word_frequencies = Counter(words)
    most_frequent_words = word_frequencies.most_common(10)

    return word_count, character_count, most_frequent_words

def print_statistics(word_count, character_count, most_frequent_words):
    print("Text Statistics:")
    print("----------------")
    print(f"Word Count: {word_count}")
    print(f"Character Count: {character_count}")
    print("Most Frequent Words:")
    for word, frequency in most_frequent_words:
        print(f"{word}: {frequency}")

def main():
    text = input("Please enter the text: ")
    word_count, character_count, most_frequent_words = calculate_statistics(text)
    print_statistics(word_count, character_count, most_frequent_words)

if __name__ == "__main__":
    main()
```
