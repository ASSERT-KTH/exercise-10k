## Exercise
Create a program that simulates a phone book. The program should take in the name and phone number and output the updated phone book based on the following conditions: if the user adds a contact, add the contact to the phone book, if the user removes a contact, remove the contact from the phone book, if the user checks the phone book, output the current phone book.

## Reference Solution
```python
class PhoneBook:
    def __init__(self):
        self.phone_book = {}

    def add_contact(self, name, phone_number):
        """Add a contact to the phone book"""
        if name in self.phone_book:
            print(f"Contact {name} already exists. Updating phone number.")
        self.phone_book[name] = phone_number
        print(f"Contact {name} added successfully.")

    def remove_contact(self, name):
        """Remove a contact from the phone book"""
        if name in self.phone_book:
            del self.phone_book[name]
            print(f"Contact {name} removed successfully.")
        else:
            print(f"Contact {name} not found in phone book.")

    def check_phone_book(self):
        """Print the current phone book"""
        if not self.phone_book:
            print("Phone book is empty.")
        else:
            print("Current Phone Book:")
            for name, phone_number in self.phone_book.items():
                print(f"{name}: {phone_number}")


def main():
    phone_book = PhoneBook()
    while True:
        print("\nPhone Book Options:")
        print("1. Add Contact")
        print("2. Remove Contact")
        print("3. Check Phone Book")
        print("4. Quit")
        option = input("Choose an option: ")
        if option == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phone_book.add_contact(name, phone_number)
        elif option == "2":
            name = input("Enter contact name: ")
            phone_book.remove_contact(name)
        elif option == "3":
            phone_book.check_phone_book()
        elif option == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
