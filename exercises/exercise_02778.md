## Exercise
Write a program that takes user input and responds accordingly (e.g. greeting, goodbye, help).

## Reference Solution
```python
def get_user_input():
    """Get user input from the console."""
    user_input = input("You: ")
    return user_input.lower()

def respond_to_input(user_input):
    """Respond to user input."""
    responses = {
        "hello": "Hi, how are you?",
        "hi": "Hello! How can I assist you?",
        "hey": "Hey, what's up?",
        "goodbye": "Goodbye, have a great day!",
        "bye": "See you later!",
        "help": "I can assist you with basic conversations. Type 'hello', 'goodbye', or 'help' to get started."
    }

    for keyword, response in responses.items():
        if keyword in user_input:
            return response

    return "I didn't understand that. Type 'help' for more information."

def main():
    print("Welcome to the user input response program. Type 'help' for more information.")
    while True:
        user_input = get_user_input()
        response = respond_to_input(user_input)
        print("Bot:", response)

        if "goodbye" in user_input or "bye" in user_input:
            break

if __name__ == "__main__":
    main()
```
