## Exercise
Write a program to calculate the simple interest given the principal amount, rate, and time.

## Reference Solution
```python
def calculate_simple_interest(principal, rate, time):
    """
    Calculates the simple interest.

    Args:
        principal (float): The principal amount.
        rate (float): The interest rate (in percent).
        time (float): The time period (in years).

    Returns:
        float: The simple interest.
    """
    # Convert rate from percent to decimal
    rate_decimal = rate / 100

    # Calculate simple interest
    simple_interest = principal * rate_decimal * time

    return simple_interest

def main():
    # Get user input
    principal = float(input("Enter the principal amount: "))
    rate = float(input("Enter the interest rate (in percent): "))
    time = float(input("Enter the time period (in years): "))

    # Calculate simple interest
    simple_interest = calculate_simple_interest(principal, rate, time)

    # Print result
    print(f"Simple Interest: {simple_interest:.2f}")
    print(f"Total Amount: {principal + simple_interest:.2f}")

if __name__ == "__main__":
    main()
```
