## Exercise
Perform bitwise operations on two numbers and print out the result.

## Reference Solution
```python
def bitwise_operations(num1, num2):
    """
    Performs bitwise operations on two numbers.

    Args:
    num1 (int): The first number.
    num2 (int): The second number.

    Returns:
    None
    """
    # Bitwise AND
    and_result = num1 & num2
    print(f"Bitwise AND of {num1} and {num2}: {and_result}")

    # Bitwise OR
    or_result = num1 | num2
    print(f"Bitwise OR of {num1} and {num2}: {or_result}")

    # Bitwise XOR
    xor_result = num1 ^ num2
    print(f"Bitwise XOR of {num1} and {num2}: {xor_result}")

    # Bitwise NOT
    not_result1 = ~num1
    not_result2 = ~num2
    print(f"Bitwise NOT of {num1}: {not_result1}")
    print(f"Bitwise NOT of {num2}: {not_result2}")

    # Left shift
    left_shift_result1 = num1 << 2
    left_shift_result2 = num2 << 2
    print(f"Left shift of {num1} by 2: {left_shift_result1}")
    print(f"Left shift of {num2} by 2: {left_shift_result2}")

    # Right shift
    right_shift_result1 = num1 >> 2
    right_shift_result2 = num2 >> 2
    print(f"Right shift of {num1} by 2: {right_shift_result1}")
    print(f"Right shift of {num2} by 2: {right_shift_result2}")

# Example usage
if __name__ == "__main__":
    num1 = 5   # Binary: 101
    num2 = 3   # Binary: 011
    bitwise_operations(num1, num2)
```
