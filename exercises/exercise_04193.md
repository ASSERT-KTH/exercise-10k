## Exercise
Create a function that generates a word scramble game, giving the user a set of letters to unscramble.

## Reference Solution
```python
import random

def generate_word_scramble(words):
    """
    Generates a word scramble game.

    Args:
        words (list): A list of words to choose from.

    Returns:
        tuple: A tuple containing the scrambled word and the original word.
    """
    # Select a random word from the list
    original_word = random.choice(words)
    
    # Convert the word to a list of letters
    letters = list(original_word)
    
    # Shuffle the letters
    scrambled_word = ''.join(random.sample(letters, len(letters)))
    
    return scrambled_word, original_word

def play_word_scramble(words):
    """
    Plays a word scramble game.

    Args:
        words (list): A list of words to choose from.
    """
    scrambled_word, original_word = generate_word_scramble(words)
    
    print(f"Unscramble the letters: {scrambled_word}")
    
    # Ask the user for their answer
    user_answer = input("Enter your answer: ")
    
    # Check if the user's answer is correct
    if user_answer.lower() == original_word.lower():
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {original_word}.")

# Example usage
words = ["apple", "banana", "cherry", "date", "elderberry"]
play_word_scramble(words)
```
