## Exercise
Create a function to check if a given number is an armorable number.

## Reference Solution
```python
def count_divisors(n):
    """Count the number of divisors of a given number."""
    count = 0
    for i in range(1, int(n**0.5) + 1):
        if n % i == 0:
            if n // i == i:
                count += 1
            else:
                count += 2
    return count

def is_armorable(n):
    """Check if a given number is an armorable number."""
    num_divisors = count_divisors(n)
    num_digits = len(str(n))
    return num_divisors == num_digits

# Example usage:
print(is_armorable(24))  # True
print(is_armorable(25))  # False
```
