## Exercise
Write a function to sort a dictionary by its keys or values.

## Reference Solution
```python
def sort_dict(input_dict, sort_by='keys', reverse=False):
    """
    Sorts a dictionary by its keys or values.

    Args:
        input_dict (dict): The dictionary to be sorted.
        sort_by (str, optional): Whether to sort by 'keys' or 'values'. Defaults to 'keys'.
        reverse (bool, optional): Whether to sort in reverse order. Defaults to False.

    Returns:
        dict: The sorted dictionary.
    """
    if sort_by == 'keys':
        return dict(sorted(input_dict.items()))
    elif sort_by == 'values':
        return dict(sorted(input_dict.items(), key=lambda item: item[1], reverse=reverse))
    else:
        raise ValueError("Invalid sort_by parameter. It should be either 'keys' or 'values'.")

# Example usage:
if __name__ == "__main__":
    input_dict = {'c': 3, 'b': 2, 'a': 1}
    print("Original dictionary:", input_dict)

    sorted_by_keys = sort_dict(input_dict)
    print("Sorted by keys:", sorted_by_keys)

    sorted_by_values = sort_dict(input_dict, sort_by='values')
    print("Sorted by values:", sorted_by_values)

    sorted_by_values_reverse = sort_dict(input_dict, sort_by='values', reverse=True)
    print("Sorted by values in reverse order:", sorted_by_values_reverse)
```
