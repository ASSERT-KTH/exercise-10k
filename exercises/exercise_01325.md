## Exercise
Write a function that takes two strings as input and returns True if the first string is a subsequence of the second string, False otherwise.

## Reference Solution

def is_subsequence(s1: str, s2: str) -> bool:
    """
    Checks if s1 is a subsequence of s2.

    Args:
        s1 (str): The potential subsequence.
        s2 (str): The string to check against.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """
    # Initialize two pointers, one for each string
    i = 0  # Pointer for s1
    j = 0  # Pointer for s2

    # Iterate through s2
    while j < len(s2):
        # If the current characters in s1 and s2 match, move the pointer for s1
        if i < len(s1) and s1[i] == s2[j]:
            i += 1
        # Move the pointer for s2
        j += 1

    # If we've iterated through all characters in s1, it's a subsequence of s2
    return i == len(s1)


# Example usage
if __name__ == "__main__":
    print(is_subsequence("abc", "ahbgdc"))  # True
    print(is_subsequence("axc", "ahbgdc"))  # False
