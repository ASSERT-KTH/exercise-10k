## Exercise
Write a program to find the next greater element for each element in a given array of numbers.

## Reference Solution
```python
def next_greater_element(arr):
    """
    Find the next greater element for each element in the given array.

    Args:
        arr (list): The input array of numbers.

    Returns:
        list: A list of next greater elements for each element in the input array.
    """
    stack = []
    result = [-1] * len(arr)

    for i in range(len(arr)):
        # While the stack is not empty and the current element is greater than the top of the stack
        while stack and arr[i] > arr[stack[-1]]:
            # Pop the top of the stack and update the result
            index = stack.pop()
            result[index] = arr[i]
        # Push the current index onto the stack
        stack.append(i)

    return result


# Example usage
def main():
    arr = [4, 5, 2, 10, 8]
    next_greater = next_greater_element(arr)
    print("Next greater elements:", next_greater)


if __name__ == "__main__":
    main()
```
