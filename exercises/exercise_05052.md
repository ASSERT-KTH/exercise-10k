## Exercise
Write a program that generates a geometric progression with a given first term, common ratio, and number of terms.

## Reference Solution
```python
def generate_geometric_progression(first_term, common_ratio, num_terms):
    """
    Generates a geometric progression with a given first term, common ratio, and number of terms.

    Args:
        first_term (float): The first term of the geometric progression.
        common_ratio (float): The common ratio of the geometric progression.
        num_terms (int): The number of terms in the geometric progression.

    Returns:
        list: A list of numbers representing the geometric progression.
    """
    progression = [first_term * (common_ratio ** i) for i in range(num_terms)]
    return progression

def main():
    # Example usage:
    first_term = 2
    common_ratio = 2
    num_terms = 10

    progression = generate_geometric_progression(first_term, common_ratio, num_terms)
    print(f"Geometric Progression: {progression}")

if __name__ == "__main__":
    main()
```
