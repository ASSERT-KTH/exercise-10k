## Exercise
Create a program that calculates the sine, cosine, and tangent of an angle.

## Reference Solution
```python
import math

def calculate_trigonometric_values(angle_in_degrees):
    """
    Calculate the sine, cosine, and tangent of an angle.

    Args:
        angle_in_degrees (float): The angle in degrees.

    Returns:
        tuple: A tuple containing the sine, cosine, and tangent of the angle.
    """
    # Convert the angle from degrees to radians
    angle_in_radians = math.radians(angle_in_degrees)

    # Calculate the sine, cosine, and tangent of the angle
    sine = math.sin(angle_in_radians)
    cosine = math.cos(angle_in_radians)
    tangent = math.tan(angle_in_radians)

    return sine, cosine, tangent

def main():
    # Get the angle from the user
    angle_in_degrees = float(input("Enter an angle in degrees: "))

    # Calculate the trigonometric values
    sine, cosine, tangent = calculate_trigonometric_values(angle_in_degrees)

    # Print the results
    print(f"Sine of {angle_in_degrees} degrees: {sine:.4f}")
    print(f"Cosine of {angle_in_degrees} degrees: {cosine:.4f}")
    print(f"Tangent of {angle_in_degrees} degrees: {tangent:.4f}")

if __name__ == "__main__":
    main()
```
