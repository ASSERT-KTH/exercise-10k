## Exercise
Create a program that takes two player's choices (rock, paper, scissors) as input and determines the winner based on the following criteria: rock beats scissors, paper beats rock, scissors beats paper using if/else statements

## Reference Solution
```python
def determine_winner(player1_choice, player2_choice):
    """
    This function determines the winner of a rock-paper-scissors game.

    Args:
        player1_choice (str): The choice of player 1. Can be 'rock', 'paper', or 'scissors'.
        player2_choice (str): The choice of player 2. Can be 'rock', 'paper', or 'scissors'.

    Returns:
        str: The winner of the game. Can be 'player1', 'player2', or 'tie'.
    """
    # Check if both players have made the same choice
    if player1_choice == player2_choice:
        return 'tie'

    # Check if player 1 wins
    if (player1_choice == 'rock' and player2_choice == 'scissors') or \
       (player1_choice == 'paper' and player2_choice == 'rock') or \
       (player1_choice == 'scissors' and player2_choice == 'paper'):
        return 'player1'

    # If player 1 does not win and it's not a tie, then player 2 must win
    return 'player2'


def main():
    # Get the choices of the two players
    player1_choice = input("Enter player 1's choice (rock, paper, scissors): ")
    player2_choice = input("Enter player 2's choice (rock, paper, scissors): ")

    # Validate the input
    valid_choices = ['rock', 'paper', 'scissors']
    while player1_choice not in valid_choices or player2_choice not in valid_choices:
        print("Invalid choice. Please enter rock, paper, or scissors.")
        player1_choice = input("Enter player 1's choice (rock, paper, scissors): ")
        player2_choice = input("Enter player 2's choice (rock, paper, scissors): ")

    # Determine the winner
    winner = determine_winner(player1_choice, player2_choice)

    # Print the result
    if winner == 'tie':
        print("It's a tie!")
    elif winner == 'player1':
        print("Player 1 wins!")
    else:
        print("Player 2 wins!")


if __name__ == "__main__":
    main()
```
