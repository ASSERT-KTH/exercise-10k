## Exercise
Create a function that partitions a list into two sublists based on a given condition.

## Reference Solution
```python
def partition_list(input_list, condition):
    """
    Partitions a list into two sublists based on a given condition.

    Args:
        input_list (list): The list to be partitioned.
        condition (function): A function that takes one argument and returns a boolean value.

    Returns:
        tuple: A tuple containing two lists. The first list contains elements for which the condition is True,
               and the second list contains elements for which the condition is False.
    """

    # Initialize two empty lists to store the partitioned elements
    true_list = []
    
    # This list will store elements for which the condition is False
    false_list = []

    # Iterate over each element in the input list
    for element in input_list:
        # Check if the condition is met for the current element
        if condition(element):
            # If the condition is True, append the element to the true_list
            true_list.append(element)
        else:
            # If the condition is False, append the element to the false_list
            false_list.append(element)

    # Return a tuple containing the two partitioned lists
    return true_list, false_list


# Example usage:
# Define a condition function that checks if a number is even
def is_even(num):
    return num % 2 == 0

# Partition a list of numbers based on the is_even condition
numbers = [1, 2, 3, 4, 5, 6]
even_numbers, odd_numbers = partition_list(numbers, is_even)

print("Even numbers:", even_numbers)
print("Odd numbers:", odd_numbers)
```
