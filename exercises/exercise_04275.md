## Exercise
Write a function that takes a dictionary as input and returns a shallow copy of the dictionary using the copy() method

## Reference Solution
```python
def shallow_copy_dict(input_dict):
    """
    Returns a shallow copy of the input dictionary.

    Args:
        input_dict (dict): The dictionary to be copied.

    Returns:
        dict: A shallow copy of the input dictionary.
    """
    return input_dict.copy()

# Example usage
if __name__ == "__main__":
    original_dict = {"a": 1, "b": 2, "c": [3, 4]}
    copied_dict = shallow_copy_dict(original_dict)

    print("Original Dictionary:", original_dict)
    print("Copied Dictionary:", copied_dict)

    # Modify the original dictionary to demonstrate shallow copy
    original_dict["c"].append(5)
    print("Modified Original Dictionary:", original_dict)
    print("Copied Dictionary after modification:", copied_dict)
```
