## Exercise
Write a program to access and print the elements of an array at specific indices.

## Reference Solution
```python
def access_array_elements(array, indices):
    """
    Access and print the elements of an array at specific indices.

    Args:
        array (list): The input array.
        indices (list): A list of indices to access.

    Returns:
        None
    """
    try:
        # Iterate over each index
        for index in indices:
            # Check if the index is within the array bounds
            if 0 <= index < len(array):
                # Print the element at the current index
                print(f"Element at index {index}: {array[index]}")
            else:
                print(f"Index {index} is out of bounds.")
    except TypeError:
        print("Invalid input type. Please use a list for the array and indices.")

def main():
    # Example array
    array = [10, 20, 30, 40, 50]

    # Example indices
    indices = [0, 2, 4]

    print("Array:", array)
    print("Indices:", indices)

    access_array_elements(array, indices)

if __name__ == "__main__":
    main()
```
