## Exercise
Write a program that uses loops (for, while, etc.) to iterate over a sequence and perform tasks.

## Reference Solution
```python
def sequence_iterator(sequence):
    """
    Iterate over a sequence, perform calculations, and print the results.
    
    Args:
        sequence (list): A list of numbers.
    """
    # Initialize variables
    total = 0
    max_value = float('-inf')
    min_value = float('inf')

    # Iterate over the sequence using a for loop
    print("For Loop:")
    for num in sequence:
        print(f"Number: {num}")
        total += num
        if num > max_value:
            max_value = num
        if num < min_value:
            min_value = num

    # Print the results
    print(f"Total: {total}")
    print(f"Max Value: {max_value}")
    print(f"Min Value: {min_value}")

    # Iterate over the sequence using a while loop
    print("\nWhile Loop:")
    i = 0
    while i < len(sequence):
        num = sequence[i]
        print(f"Number: {num}")
        i += 1

def main():
    # Create a sample sequence
    sequence = [1, 2, 3, 4, 5]

    # Call the sequence iterator function
    sequence_iterator(sequence)

if __name__ == "__main__":
    main()
```
