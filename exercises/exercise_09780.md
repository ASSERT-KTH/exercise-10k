## Exercise
Write a program that reads input from a file and processes it.

## Reference Solution
```python
def process_file(filename):
    """
    Reads input from a file and processes it.

    Args:
        filename (str): The name of the file to process.

    Returns:
        tuple: A tuple containing the number of lines, words, and characters in the file.
    """
    try:
        with open(filename, 'r') as file:
            lines = file.readlines()
            line_count = len(lines)
            word_count = sum(len(line.split()) for line in lines)
            char_count = sum(len(line) for line in lines)
            return line_count, word_count, char_count
    except FileNotFoundError:
        print(f"File '{filename}' not found.")
        return None

def main():
    filename = input("Enter the filename: ")
    result = process_file(filename)
    if result:
        line_count, word_count, char_count = result
        print(f"Lines: {line_count}")
        print(f"Words: {word_count}")
        print(f"Characters: {char_count}")

if __name__ == "__main__":
    main()
```
