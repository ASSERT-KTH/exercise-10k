## Exercise
Write a program that reads input from a database and prints out the contents.

## Reference Solution
```python
import sqlite3

def read_database(db_name, table_name):
    """
    Reads input from a SQLite database and prints out the contents.

    Args:
        db_name (str): Name of the database file
        table_name (str): Name of the table to read from

    Returns:
        None
    """
    try:
        # Connect to the database
        conn = sqlite3.connect(db_name)
        cursor = conn.cursor()

        # Read data from the table
        cursor.execute(f"SELECT * FROM {table_name}")

        # Fetch all rows from the table
        rows = cursor.fetchall()

        # Print the column names
        column_names = [description[0] for description in cursor.description]
        print("Column Names:")
        print(column_names)

        # Print the data
        print("\nData:")
        for row in rows:
            print(row)

    except sqlite3.Error as e:
        print(f"Error: {e}")

    finally:
        # Close the connection
        if conn:
            conn.close()

# Example usage
if __name__ == "__main__":
    db_name = "example.db"
    table_name = "users"
    read_database(db_name, table_name)
```
