## Exercise
Write a program to group a list of elements by a common attribute.

## Reference Solution
```python
from itertools import groupby
from operator import itemgetter
from dataclasses import dataclass

# Define a data class for the elements
@dataclass
class Element:
    name: str
    category: str

def group_elements(elements, attribute):
    """
    Group a list of elements by a common attribute.

    Args:
        elements (list): A list of elements.
        attribute (str): The attribute to group by.

    Returns:
        dict: A dictionary where the keys are the unique attribute values and the values are lists of elements.
    """
    # Sort the elements by the attribute
    sorted_elements = sorted(elements, key=itemgetter(attribute))

    # Group the elements by the attribute
    grouped_elements = {}
    for key, group in groupby(sorted_elements, key=itemgetter(attribute)):
        grouped_elements[key] = list(group)

    return grouped_elements

# Example usage
if __name__ == "__main__":
    # Create a list of elements
    elements = [
        Element("Apple", "Fruit"),
        Element("Banana", "Fruit"),
        Element("Carrot", "Vegetable"),
        Element("Potato", "Vegetable"),
        Element("Orange", "Fruit"),
    ]

    # Group the elements by category
    grouped_elements = group_elements(elements, "category")

    # Print the grouped elements
    for category, elements in grouped_elements.items():
        print(f"Category: {category}")
        for element in elements:
            print(f"- {element.name}")
        print()
```
