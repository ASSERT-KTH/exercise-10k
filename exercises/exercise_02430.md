## Exercise
Develop a program to validate passwords, ensuring they meet certain criteria (e.g., length, containing both uppercase and lowercase letters, containing digits), using a while loop.

## Reference Solution
```python
import re

def validate_password(password):
    """
    Validate a password based on the following criteria:
    - Length: at least 8 characters
    - Uppercase letters: at least one
    - Lowercase letters: at least one
    - Digits: at least one
    """
    while True:
        # Check length
        if len(password) < 8:
            print("Password must be at least 8 characters long.")
            password = input("Enter a new password: ")
            continue

        # Check for uppercase letters
        if not re.search("[A-Z]", password):
            print("Password must contain at least one uppercase letter.")
            password = input("Enter a new password: ")
            continue

        # Check for lowercase letters
        if not re.search("[a-z]", password):
            print("Password must contain at least one lowercase letter.")
            password = input("Enter a new password: ")
            continue

        # Check for digits
        if not re.search("[0-9]", password):
            print("Password must contain at least one digit.")
            password = input("Enter a new password: ")
            continue

        # If all checks pass, break out of the loop
        break

    print("Password is valid.")

def main():
    password = input("Enter a password: ")
    validate_password(password)

if __name__ == "__main__":
    main()
```
