## Exercise
Create a program that calculates the mean, median, and mode of a list of numbers.

## Reference Solution
```python
from statistics import mean, median, multimode

def calculate_statistics(numbers):
    """
    Calculate the mean, median, and mode of a list of numbers.

    Args:
    - numbers (list): A list of numbers.

    Returns:
    - A dictionary containing the mean, median, and mode.
    """
    if not numbers:
        raise ValueError("Input list is empty")

    mean_value = mean(numbers)
    median_value = median(numbers)
    mode_value = multimode(numbers)

    return {
        "mean": mean_value,
        "median": median_value,
        "mode": mode_value
    }

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5, 2, 2]
    stats = calculate_statistics(numbers)

    print("Mean:", stats["mean"])
    print("Median:", stats["median"])
    print("Mode:", stats["mode"])

if __name__ == "__main__":
    main()
```
