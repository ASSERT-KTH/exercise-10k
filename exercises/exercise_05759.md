## Exercise
Use built-in functions to convert data types and print the results

## Reference Solution
```python
# Data Type Conversion and Printing

def convert_and_print():
    # Define variables of different data types
    integer = 10
    float_num = 20.5
    string = "Hello, World!"
    boolean = True

    # Convert data types using built-in functions
    integer_to_float = float(integer)
    float_to_integer = int(float_num)
    string_to_integer = int(string.split(",")[0].strip().replace("Hello", "").replace(" ", ""))
    boolean_to_integer = int(boolean)

    # Print the results
    print("Original Values:")
    print(f"Integer: {integer}")
    print(f"Float: {float_num}")
    print(f"String: {string}")
    print(f"Boolean: {boolean}")

    print("\nConverted Values:")
    print(f"Integer to Float: {integer_to_float}")
    print(f"Float to Integer: {float_to_integer}")
    print(f"String to Integer: {string_to_integer}")
    print(f"Boolean to Integer: {boolean_to_integer}")


if __name__ == "__main__":
    convert_and_print()
```
