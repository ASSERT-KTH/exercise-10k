## Exercise
Write a function that takes a string as input and returns the shortest word in the string.

## Reference Solution
```python
def shortest_word(input_string):
    """
    Returns the shortest word in the input string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The shortest word in the input string.
    """
    # Split the input string into words
    words = input_string.split()

    # Check if the input string is empty
    if not words:
        return None

    # Initialize the shortest word with the first word
    shortest = words[0]

    # Iterate over the words to find the shortest one
    for word in words:
        # Remove punctuation from the word
        word_no_punct = ''.join(e for e in word if e.isalnum())
        
        # Update the shortest word if the current word is shorter
        if len(word_no_punct) < len(''.join(e for e in shortest if e.isalnum())):
            shortest = word

    return shortest

# Example usage
def main():
    input_string = "I love programming in Python"
    print(shortest_word(input_string))

if __name__ == "__main__":
    main()
```
