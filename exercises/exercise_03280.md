## Exercise
Create a program that uses a for loop to find the mode (most frequently occurring number) in a list of numbers.

## Reference Solution
```python
def find_mode(num_list):
    # Create an empty dictionary to store the frequency of each number
    frequency_dict = {}

    # Iterate through the list to count the frequency of each number
    for num in num_list:
        if num in frequency_dict:
            frequency_dict[num] += 1
        else:
            frequency_dict[num] = 1

    # Initialize the mode and its frequency
    mode = None
    max_frequency = 0

    # Iterate through the dictionary to find the mode
    for num, frequency in frequency_dict.items():
        if frequency > max_frequency:
            max_frequency = frequency
            mode = num

    return mode

# Example usage
numbers = [1, 2, 3, 4, 2, 2, 3, 1, 4, 4, 4]
mode = find_mode(numbers)
print(f"The mode is: {mode}")
```

```python
def find_mode(num_list):
    if not num_list:
        raise ValueError("Input list is empty")

    frequency_dict = {}
    for num in num_list:
        if num in frequency_dict:
            frequency_dict[num] += 1
        else:
            frequency_dict[num] = 1

    max_frequency = max(frequency_dict.values())
    modes = [num for num, frequency in frequency_dict.items() if frequency == max_frequency]

    if len(modes) == 1:
        return modes[0]
    else:
        return modes

# Example usage
numbers = [1, 2, 3, 4, 2, 2, 3, 1, 4, 4, 4]
mode = find_mode(numbers)
print(f"The mode(s) is/are: {mode}")
```
