## Exercise
Create two dictionaries with names of countries as keys and their capitals as values. Check if one dictionary is a superset of the other and print the result.

## Reference Solution
```python
def is_superset(dict1, dict2):
    """
    Checks if dict1 is a superset of dict2.
    
    Args:
        dict1 (dict): The potential superset dictionary.
        dict2 (dict): The dictionary to check against.
    
    Returns:
        bool: True if dict1 is a superset of dict2, False otherwise.
    """
    return all(key in dict1 and dict1[key] == value for key, value in dict2.items())

def main():
    # Create dictionaries
    countries1 = {
        "France": "Paris",
        "Germany": "Berlin",
        "Italy": "Rome",
        "Spain": "Madrid"
    }
    
    countries2 = {
        "France": "Paris",
        "Germany": "Berlin",
        "Italy": "Rome"
    }
    
    # Check if countries1 is a superset of countries2
    if is_superset(countries1, countries2):
        print("countries1 is a superset of countries2")
    else:
        print("countries1 is not a superset of countries2")

if __name__ == "__main__":
    main()
```
