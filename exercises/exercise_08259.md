## Exercise
Write a program that counts the number of words in a sentence using if/else statements.

## Reference Solution
```python
def count_words(sentence):
    """
    Counts the number of words in a sentence.

    Args:
        sentence (str): The input sentence.

    Returns:
        int: The number of words in the sentence.
    """
    words = sentence.split()
    word_count = 0

    if words:  # Check if the list of words is not empty
        word_count = len(words)
    else:
        print("The sentence is empty.")

    return word_count

def main():
    sentence = input("Enter a sentence: ")
    word_count = count_words(sentence)

    if word_count > 0:
        print(f"The sentence contains {word_count} word{'s' if word_count > 1 else ''}.")
    else:
        print("No words found.")

if __name__ == "__main__":
    main()
```
