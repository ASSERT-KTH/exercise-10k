## Exercise
Create a program that takes two numbers and an arithmetic operation as input and performs the operation on the numbers. The operation can be addition, subtraction, multiplication, or division.

## Reference Solution
```python
def perform_arithmetic_operation(num1, num2, operation):
    """
    Perform an arithmetic operation on two numbers.

    Args:
    num1 (float): The first number.
    num2 (float): The second number.
    operation (str): The arithmetic operation to perform. Can be 'add', 'subtract', 'multiply', or 'divide'.

    Returns:
    float: The result of the arithmetic operation.
    """
    if operation == 'add':
        return num1 + num2
    elif operation == 'subtract':
        return num1 - num2
    elif operation == 'multiply':
        return num1 * num2
    elif operation == 'divide':
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero")
    else:
        raise ValueError("Invalid operation. Supported operations are 'add', 'subtract', 'multiply', and 'divide'.")

def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the arithmetic operation (add, subtract, multiply, divide): ").lower()

    # Perform the arithmetic operation
    try:
        result = perform_arithmetic_operation(num1, num2, operation)
        print(f"The result of {operation}ing {num1} and {num2} is {result}")
    except (ZeroDivisionError, ValueError) as e:
        print(str(e))

if __name__ == "__main__":
    main()
```
