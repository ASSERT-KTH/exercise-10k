## Exercise
Write a program to demonstrate the use of constant variables

## Reference Solution
```python
# Define constant variables
PI = 3.14159  # Mathematical constant
GRAVITY = 9.81  # m/s^2, acceleration due to gravity on Earth
MAX_STUDENTS = 30  # Maximum number of students in a class

# Function to calculate the area of a circle
def calculate_circle_area(radius):
    """Calculate the area of a circle using the formula: area = Ï€ * r^2"""
    return PI * (radius ** 2)

# Function to calculate the force of gravity on an object
def calculate_force(mass):
    """Calculate the force of gravity on an object using the formula: force = m * g"""
    return mass * GRAVITY

# Main function
def main():
    print("Constant Variables Demonstration")
    print("----------------------------------")

    # Calculate the area of a circle with radius 5
    radius = 5
    area = calculate_circle_area(radius)
    print(f"The area of a circle with radius {radius} is {area:.2f}")

    # Calculate the force of gravity on an object with mass 10 kg
    mass = 10
    force = calculate_force(mass)
    print(f"The force of gravity on an object with mass {mass} kg is {force:.2f} N")

    # Check if the number of students is within the maximum limit
    num_students = 25
    if num_students <= MAX_STUDENTS:
        print(f"The number of students ({num_students}) is within the maximum limit ({MAX_STUDENTS})")
    else:
        print(f"The number of students ({num_students}) exceeds the maximum limit ({MAX_STUDENTS})")

if __name__ == "__main__":
    main()
```
