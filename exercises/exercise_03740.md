## Exercise
Create a program that reads a text file from the user, processes the data, and prints out the results.

## Reference Solution
```python
import os

def read_file(file_path):
    """
    Reads the contents of a text file.

    Args:
        file_path (str): Path to the text file.

    Returns:
        str: File contents.
    """
    try:
        with open(file_path, 'r') as file:
            return file.read()
    except FileNotFoundError:
        print(f"File not found: {file_path}")
        return None

def process_data(data):
    """
    Processes the text data.

    Args:
        data (str): Text data.

    Returns:
        dict: Processed data.
    """
    words = data.split()
    word_count = len(words)
    unique_words = set(words)
    unique_word_count = len(unique_words)

    return {
        "word_count": word_count,
        "unique_word_count": unique_word_count,
        "unique_words": unique_words
    }

def print_results(data):
    """
    Prints the processed data.

    Args:
        data (dict): Processed data.
    """
    print("Text File Statistics:")
    print("------------------------")
    print(f"Word Count: {data['word_count']}")
    print(f"Unique Word Count: {data['unique_word_count']}")
    print("Unique Words:")
    for word in data["unique_words"]:
        print(word)

def main():
    file_path = input("Enter the text file path: ")
    data = read_file(file_path)

    if data:
        processed_data = process_data(data)
        print_results(processed_data)

if __name__ == "__main__":
    main()
```
