    """

    Returns:
        float: The temperature in Fahrenheit.
    """

    # First, we need to check if the input is a valid number. We'll use a try-except block for this.
    try:

        fahrenheit = (celsius * 9) / 5
        
        # Then, we'll add 32 to the result to get the final Fahrenheit temperature.
        fahrenheit += 32

    # If the input is not a valid number, we'll catch the exception and raise a ValueError with a meaningful error message.
