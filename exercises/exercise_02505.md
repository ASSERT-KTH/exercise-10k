## Exercise
Write a program that takes two numbers as input and returns their modulus result.

## Reference Solution
```python
# Function to calculate modulus
def calculate_modulus(dividend, divisor):
    """
    Calculate the modulus of two numbers.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        int: The modulus result.
    """
    if divisor == 0:
        raise ZeroDivisionError("Divisor cannot be zero")
    return dividend % divisor

# Main function
def main():
    try:
        # Get user input for dividend and divisor
        dividend = int(input("Enter the dividend: "))
        divisor = int(input("Enter the divisor: "))

        # Calculate and display the modulus result
        modulus = calculate_modulus(dividend, divisor)
        print(f"The modulus of {dividend} and {divisor} is: {modulus}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")
    except ValueError:
        print("Invalid input. Please enter integers only.")

if __name__ == "__main__":
    main()
```
