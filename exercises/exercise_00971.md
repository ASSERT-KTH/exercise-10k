## Exercise
Given a dictionary and a set of required keys, write a function to check if the dictionary contains all the required keys.

## Reference Solution
```python
def check_required_keys(dictionary, required_keys):
    """
    Checks if a dictionary contains all the required keys.

    Args:
        dictionary (dict): The dictionary to be checked.
        required_keys (set): A set of required keys.

    Returns:
        bool: True if the dictionary contains all the required keys, False otherwise.
    """

    # First, we need to check if the dictionary is actually a dictionary
    if not isinstance(dictionary, dict):
        raise TypeError("The input 'dictionary' must be a dictionary.")

    # Then, we need to check if the required keys are a set
    if not isinstance(required_keys, set):
        raise TypeError("The input 'required_keys' must be a set.")

    # Now, we can use the built-in 'issubset' method of sets in Python
    # This method returns True if all elements of the set are present in the dictionary's keys
    # We can get the dictionary's keys using the 'keys' method
    return required_keys.issubset(dictionary.keys())

# Example usage:
my_dict = {"a": 1, "b": 2, "c": 3}
required_keys = {"a", "b"}
print(check_required_keys(my_dict, required_keys))  # Output: True

my_dict = {"a": 1, "b": 2}
required_keys = {"a", "b", "c"}
print(check_required_keys(my_dict, required_keys))  # Output: False
```
