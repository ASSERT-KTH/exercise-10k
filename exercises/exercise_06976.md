## Exercise
Create a function that takes two strings as arguments and returns True if the first string is a subsequence of the second string and False otherwise.

def is_subsequence(s1, s2):
    """
    Checks if the first string is a subsequence of the second string.

    Args:
        s1 (str): The potential subsequence string.
        s2 (str): The main string.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """

    # Initialize two pointers, one for each string, at the beginning of the strings
    i = 0  # Pointer for the first string
    j = 0  # Pointer for the second string

    # Continue the loop until we have checked all characters in the first string
    while i < len(s1) and j < len(s2):
        # If the current characters in both strings are equal, move the pointer for the first string forward
        if s1[i] == s2[j]:
            i += 1  # Move the pointer for the first string forward
        # Move the pointer for the second string forward regardless of whether the characters match or not
        j += 1  # Move the pointer for the second string forward

    # If we have checked all characters in the first string, it is a subsequence of the second string
    # Otherwise, it is not a subsequence
    return i == len(s1)
