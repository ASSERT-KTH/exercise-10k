## Exercise
Write a function to reverse the order of elements in an array.

## Reference Solution
```python
def reverse_array(arr):
    """
    Reverses the order of elements in an array.

    Args:
        arr (list): The input array.

    Returns:
        list: A new array with the elements in reverse order.
    """
    return arr[::-1]

# Example usage
if __name__ == "__main__":
    arr = [1, 2, 3, 4, 5]
    reversed_arr = reverse_array(arr)
    print("Original array:", arr)
    print("Reversed array:", reversed_arr)
```

def reverse_array_manual(arr):
    left = 0
    right = len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1
    return arr

# Example usage
if __name__ == "__main__":
    arr = [1, 2, 3, 4, 5]
    reversed_arr = reverse_array_manual(arr)
    print("Original array:", [1, 2, 3, 4, 5])
    print("Reversed array:", reversed_arr)
