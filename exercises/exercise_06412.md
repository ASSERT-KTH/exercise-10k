## Exercise
Write a program that generates a Mad Libs story with user input. Use if/else statements to determine the story.

## Reference Solution
```python
# Import the required libraries
import random

# Define a function to get user input
def get_user_input():
    print("Please enter the following:")
    noun = input("Enter a noun: ")
    adjective = input("Enter an adjective: ")
    verb = input("Enter a verb: ")
    place = input("Enter a place: ")
    name = input("Enter a name: ")
    return noun, adjective, verb, place, name

# Define a function to generate the Mad Libs story
def generate_story(noun, adjective, verb, place, name):
    # Use if/else statements to determine the story
    if random.randint(0, 1) == 0:
        story = f"One day, {name} was walking through the {adjective} forest when they stumbled upon a {noun}. " \
                f"They decided to {verb} it and take it to the {place}. " \
                f"When they arrived, they were greeted by a group of friendly {noun}s."
    else:
        story = f"{name} had always dreamed of visiting the {place}. " \
                f"So, they packed their bags and set off on an adventure to {verb} through the {adjective} mountains. " \
                f"Along the way, they encountered a {noun} who became their trusty companion."
    return story

# Define a main function to run the program
def main():
    noun, adjective, verb, place, name = get_user_input()
    story = generate_story(noun, adjective, verb, place, name)
    print("\nHere is your Mad Libs story:")
    print(story)

# Run the main function
if __name__ == "__main__":
    main()
```
