## Exercise
Write a program that simulates user input

## Reference Solution

import unittest.mock as mock
from unittest.mock import MagicMock
import sys

def simulate_user_input(inputs):
    """
    Simulate user input by returning a mock object that can be used to simulate user input.

    Args:
        inputs (list): A list of user inputs.

    Returns:
        MagicMock: A mock object that can be used to simulate user input.
    """
    mock_input = MagicMock(side_effect=inputs)
    mock_input.return_value = inputs.pop(0)
    return mock_input

def main():
    # Define the user inputs
    user_inputs = ["hello", "world", "123"]

    # Simulate user input
    mock_input = simulate_user_input(user_inputs)

    # Use the mock object to simulate user input
    with mock.patch('builtins.input', mock_input):
        for _ in range(len(user_inputs)):
            print("User input:", input())

if __name__ == "__main__":
    main()
