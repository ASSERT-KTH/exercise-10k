## Exercise
Write a program to take a string input from the user and print it to the screen, but handle any errors that may occur during input.

## Reference Solution
```python
def get_user_input():
    """Gets user input and handles exceptions."""
    try:
        user_input = input("Please enter a string: ")
        return user_input
    except KeyboardInterrupt:
        print("\nError: Input interrupted by user.")
        return None
    except EOFError:
        print("Error: End of file reached unexpectedly.")
        return None
    except Exception as e:
        print(f"An unexpected error occurred: {str(e)}")
        return None


def print_user_input(user_input):
    """Prints the user's input to the screen."""
    if user_input is not None:
        print(f"You entered: {user_input}")
    else:
        print("No input to print.")


def main():
    user_input = get_user_input()
    print_user_input(user_input)


if __name__ == "__main__":
    main()
```
