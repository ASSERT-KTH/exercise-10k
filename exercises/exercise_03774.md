## Exercise
Sort a dictionary by its keys.

## Reference Solution
```python
def sort_dict_by_keys(input_dict):
    """
    Sorts a dictionary by its keys.

    Args:
        input_dict (dict): The dictionary to be sorted.

    Returns:
        dict: A new dictionary sorted by keys.
    """
    return dict(sorted(input_dict.items()))

# Example usage:
my_dict = {'c': 3, 'b': 2, 'a': 1, 'd': 4}
sorted_dict = sort_dict_by_keys(my_dict)
print(sorted_dict)
```

```python
from collections import OrderedDict

def sort_dict_by_keys(input_dict):
    """
    Sorts a dictionary by its keys.

    Args:
        input_dict (dict): The dictionary to be sorted.

    Returns:

    """

# Example usage:
```
