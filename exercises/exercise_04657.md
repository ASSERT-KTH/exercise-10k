## Exercise
Create a program that manages a person's finances. The user can add or remove transactions from the account, and the program displays the current balance.

## Reference Solution
```python
class FinancialAccount:
    def __init__(self):
        self.transactions = []
        self.balance = 0.0

    def add_transaction(self, amount, description):
        """Add a transaction to the account."""
        self.transactions.append((amount, description))
        self.balance += amount
        print(f"Transaction added: {amount} for {description}")

    def remove_transaction(self, index):
        """Remove a transaction from the account."""
        if index < len(self.transactions):
            amount, _ = self.transactions.pop(index)
            self.balance -= amount
            print(f"Transaction removed: {amount}")
        else:
            print("Invalid transaction index")

    def display_balance(self):
        """Display the current balance."""
        print(f"Current balance: ${self.balance:.2f}")

    def display_transactions(self):
        """Display all transactions."""
        print("Transactions:")
        for i, (amount, description) in enumerate(self.transactions):
            print(f"{i+1}. {amount} for {description}")


def main():
    account = FinancialAccount()

    while True:
        print("\nFinancial Management Program")
        print("1. Add transaction")
        print("2. Remove transaction")
        print("3. Display balance")
        print("4. Display transactions")
        print("5. Quit")

        choice = input("Choose an option: ")

        if choice == "1":
            amount = float(input("Enter transaction amount: $"))
            description = input("Enter transaction description: ")
            account.add_transaction(amount, description)
        elif choice == "2":
            account.display_transactions()
            index = int(input("Enter transaction number to remove: ")) - 1
            account.remove_transaction(index)
        elif choice == "3":
            account.display_balance()
        elif choice == "4":
            account.display_transactions()
        elif choice == "5":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")


if __name__ == "__main__":
    main()
```
