## Exercise
Write a function to curry a given function, allowing it to be called with multiple arguments one at a time.

def curry(func):
    """
    This function curries a given function, allowing it to be called with multiple arguments one at a time.

    Args:
        func (function): The function to be curried.

    Returns:
        function: A curried version of the input function.
    """

    # Define a helper function to store the arguments and the original function
    def helper(*args, **kwargs):
        
        # If the number of arguments is less than the required number of arguments for the function
        if len(args) + len(kwargs) < func.__code__.co_argcount:
            
            # Return a new function that takes the remaining arguments
            def new_func(*new_args, **new_kwargs):
                # Combine the old arguments with the new arguments
                return helper(*(args + new_args), **dict(kwargs, **new_kwargs))
            
            # Return the new function
            return new_func
        
        # If the number of arguments is equal to or more than the required number of arguments for the function
        else:
            
            # Call the original function with the arguments and return the result
            return func(*args, **kwargs)
    
    # Return the helper function
    return helper

# Example usage:
def add(a, b, c):
    return a + b + c

curried_add = curry(add)
print(curried_add(1)(2)(3))  # Outputs: 6
