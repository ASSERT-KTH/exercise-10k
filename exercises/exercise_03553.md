## Exercise
Write a program that takes two arrays as input, and returns a new array containing all unique elements.

## Reference Solution
```python
def unique_elements(arr1, arr2):
    """
    This function takes two arrays as input and returns a new array containing all unique elements.
    
    Parameters:
    arr1 (list): The first input array.
    arr2 (list): The second input array.
    
    Returns:
    list: A new array containing all unique elements from the input arrays.
    """
    
    # Combine the two arrays into one
    combined_array = arr1 + arr2
    
    # Convert the combined array to a set to remove duplicates
    unique_set = set(combined_array)
    
    # Convert the set back to a list
    unique_list = list(unique_set)
    
    return unique_list

# Example usage
arr1 = [1, 2, 3, 4]
arr2 = [3, 4, 5, 6]
print(unique_elements(arr1, arr2))  # Output: [1, 2, 3, 4, 5, 6]
```

def unique_elements_ordered(arr1, arr2):
    """
    This function takes two arrays as input and returns a new array containing all unique elements, preserving the original order.
    
    Parameters:
    arr1 (list): The first input array.
    arr2 (list): The second input array.
    
    Returns:
    list: A new array containing all unique elements from the input arrays, in the original order.
    """
    
    # Create an empty set to keep track of seen elements
    seen = set()
    
    # Create an empty list to store the unique elements
    unique_list = []
    
    # Combine the two arrays into one
    combined_array = arr1 + arr2
    
    # Iterate over the combined array
    for element in combined_array:
        # Check if the element is not in the seen set
        if element not in seen:
            # Add the element to the seen set
            seen.add(element)
            # Add the element to the unique list
            unique_list.append(element)
    
    return unique_list

# Example usage
arr1 = [1, 2, 3, 4]
arr2 = [3, 4, 5, 6]
print(unique_elements_ordered(arr1, arr2))  # Output: [1, 2, 3, 4, 5, 6]
