## Exercise
Create a program that allows the user to add and remove items from a to-do list.

## Reference Solution
```python
# To-Do List Program

# Initialize an empty list to store to-do items
todo_list = []

def display_list():
    """Displays the current to-do list."""
    print("Current To-Do List:")
    for i, item in enumerate(todo_list, start=1):
        print(f"{i}. {item}")

def add_item():
    """Adds a new item to the to-do list."""
    item = input("Enter a new item: ")
    todo_list.append(item)
    print(f"Added '{item}' to the list.")

def remove_item():
    """Removes an item from the to-do list."""
    if not todo_list:
        print("The list is empty.")
        return

    display_list()
    try:
        item_number = int(input("Enter the number of the item to remove: "))
        if 1 <= item_number <= len(todo_list):
            item = todo_list.pop(item_number - 1)
            print(f"Removed '{item}' from the list.")
        else:
            print("Invalid item number.")
    except ValueError:
        print("Invalid input. Please enter a number.")

def main():
    """Main function to run the to-do list program."""
    while True:
        print("\nOptions:")
        print("1. Display To-Do List")
        print("2. Add Item")
        print("3. Remove Item")
        print("4. Quit")
        choice = input("Enter your choice: ")
        
        if choice == "1":
            display_list()
        elif choice == "2":
            add_item()
        elif choice == "3":
            remove_item()
        elif choice == "4":
            print("Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
