## Exercise
Create a program that checks if two strings are anagrams of each other using a while loop.

## Reference Solution
```python
def are_anagrams(str1, str2):
    """
    Checks if two strings are anagrams of each other.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        bool: True if the strings are anagrams, False otherwise.
    """
    # Remove any whitespace and convert to lowercase
    str1 = str1.replace(" ", "").lower()
    str2 = str2.replace(" ", "").lower()

    # If the lengths of the strings are not equal, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Create a copy of the first string to modify
    str1_copy = list(str1)

    # Iterate over the characters in the second string
    i = 0
    while i < len(str2):
        # Check if the current character is in the copy of the first string
        if str2[i] in str1_copy:
            # Remove the character from the copy of the first string
            str1_copy.remove(str2[i])
        else:
            # If the character is not found, the strings are not anagrams
            return False
        i += 1

    # If all characters were found and removed, the strings are anagrams
    return True


# Example usage
def main():
    str1 = "Listen"
    str2 = "Silent"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")

    str1 = "Hello"
    str2 = "World"
    print(f"Are '{str1}' and '{str2}' anagrams? {are_anagrams(str1, str2)}")


if __name__ == "__main__":
    main()
```
