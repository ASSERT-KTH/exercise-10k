## Exercise
Write a program that takes a list as input and returns the sum, mean, median, and standard deviation of the list elements.

## Reference Solution
```python
import numpy as np

def calculate_statistics(input_list):
    """
    Calculate and return the sum, mean, median, and standard deviation of the input list.
    
    Parameters:
    input_list (list): A list of numbers.
    
    Returns:
    dict: A dictionary containing the sum, mean, median, and standard deviation of the input list.
    """
    # Check if the input list is empty
    if not input_list:
        return {
            "sum": None,
            "mean": None,
            "median": None,
            "std_dev": None,
        }
    
    # Calculate the sum of the list elements
    total_sum = np.sum(input_list)
    
    # Calculate the mean of the list elements
    mean = np.mean(input_list)
    
    # Calculate the median of the list elements
    median = np.median(input_list)
    
    # Calculate the standard deviation of the list elements
    std_dev = np.std(input_list)
    
    # Return the calculated statistics as a dictionary
    return {
        "sum": total_sum,
        "mean": mean,
        "median": median,
        "std_dev": std_dev,
    }

def main():
    # Example usage:
    input_list = [1, 2, 3, 4, 5]
    stats = calculate_statistics(input_list)
    print("Input List:", input_list)
    print("Statistics:")
    for key, value in stats.items():
        print(f"{key.capitalize().replace('_', ' ')}: {value}")

if __name__ == "__main__":
    main()
```
