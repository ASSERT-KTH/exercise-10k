```python
# String concatenation
str1 = "Hello"
str2 = "World"
print("Concatenation:", str1 + " " + str2)  # Output: Hello World

# String repetition
print("Repetition:", "Hello" * 3)  # Output: HelloHelloHello

# String indexing
str3 = "Python"
print("Indexing:", str3[0])  # Output: P

# String slicing
print("Slicing:", str3[1:3])  # Output: yth

# String methods
print("Upper case:", str3.upper())  # Output: PYTHON
print("Lower case:", str3.lower())  # Output: python
print("Strip:", "   Hello World   ".strip())  # Output: Hello World
print("Split:", "Hello World".split())  # Output: ['Hello', 'World']
```

```python
import datetime

# Create datetime objects
dt1 = datetime.datetime(2022, 1, 1, 12, 0, 0)
dt2 = datetime.datetime(2022, 1, 1, 13, 0, 0)

# Add and subtract datetime objects
print("Addition:", dt1 + datetime.timedelta(hours=1))  # Output: 2022-01-01 13:00:00
print("Subtraction:", dt2 - dt1)  # Output: 1:00:00

# Compare datetime objects
print("Equal:", dt1 == dt2)  # Output: False
print("Not equal:", dt1 != dt2)  # Output: True
print("Greater than:", dt2 > dt1)  # Output: True
print("Less than:", dt1 < dt2)  # Output: True
print("Greater than or equal:", dt2 >= dt1)  # Output: True
print("Less than or equal:", dt1 <= dt2)  # Output: True
```
