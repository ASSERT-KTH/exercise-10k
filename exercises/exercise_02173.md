## Exercise
Ask the user for the principal amount, interest rate, and number of years, and print the compound interest.

## Reference Solution
```python
def calculate_compound_interest(principal, rate, years):
    """
    Calculate compound interest.

    Args:
        principal (float): The initial amount of money.
        rate (float): The interest rate (in %).
        years (int): The number of years.

    Returns:
        float: The compound interest.
    """
    rate = rate / 100  # Convert rate from % to decimal
    compound_interest = principal * (1 + rate) ** years - principal
    return compound_interest

def main():
    # Get user input
    principal = float(input("Enter the principal amount: $"))
    rate = float(input("Enter the interest rate (%): "))
    years = int(input("Enter the number of years: "))

    # Calculate compound interest
    compound_interest = calculate_compound_interest(principal, rate, years)

    # Print the result
    print(f"The compound interest is: ${compound_interest:.2f}")
    print(f"The total amount after {years} years is: ${principal + compound_interest:.2f}")

if __name__ == "__main__":
    main()
```
