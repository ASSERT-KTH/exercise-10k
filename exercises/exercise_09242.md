## Exercise
Write a program that uses conditional statements (if-else, switch, etc.) to make decisions based on conditions.

## Reference Solution
```python
def get_decision(age, score):
    """
    Make a decision based on age and score.

    Args:
        age (int): The age of the person.
        score (int): The score of the person.

    Returns:
        str: The decision made based on the conditions.
    """
    if age < 18:
        return "You are not eligible to participate."
    elif score < 50:
        return "You need to improve your score."
    elif score < 80:
        return "You are eligible to participate, but need to practice more."
    else:
        return "You are eligible to participate and have a good chance of winning."

def get_day_of_week(day_number):
    """
    Get the day of the week based on the day number.

    Args:
        day_number (int): The day number (1-7).

    Returns:
        str: The day of the week.
    """
    match day_number:
        case 1:
            return "Monday"
        case 2:
            return "Tuesday"
        case 3:
            return "Wednesday"
        case 4:
            return "Thursday"
        case 5:
            return "Friday"
        case 6:
            return "Saturday"
        case 7:
            return "Sunday"
        case _:
            return "Invalid day number"

def main():
    print("Decision Maker Program")
    print("------------------------")

    age = int(input("Enter your age: "))
    score = int(input("Enter your score: "))

    decision = get_decision(age, score)
    print(f"Decision: {decision}")

    day_number = int(input("Enter the day number (1-7): "))
    day_of_week = get_day_of_week(day_number)
    print(f"Day of the week: {day_of_week}")

if __name__ == "__main__":
    main()
```
