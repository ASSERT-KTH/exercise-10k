## Exercise
Write a program that calculates the speed given the distance and time.

## Reference Solution
```python
def calculate_speed(distance, time):
    """
    Calculate the speed of an object given the distance and time.

    Args:
        distance (float): The distance traveled by the object in meters.
        time (float): The time taken by the object to travel the distance in seconds.

    Returns:
        float: The speed of the object in meters per second.
    """
    if time <= 0:
        raise ValueError("Time must be greater than zero")
    speed = distance / time
    return speed

def main():
    # Example usage
    distance = float(input("Enter the distance traveled in meters: "))
    time = float(input("Enter the time taken in seconds: "))

    try:
        speed = calculate_speed(distance, time)
        print(f"The speed of the object is {speed:.2f} meters per second")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
