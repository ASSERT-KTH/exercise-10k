## Exercise
Write a program that takes two numbers as input and prints their quotient and remainder.

## Reference Solution
```python
def calculate_quotient_and_remainder(dividend, divisor):
    """
    Calculate the quotient and remainder of the division of two numbers.

    Args:
        dividend (int): The number being divided.
        divisor (int): The number by which we are dividing.

    Returns:
        tuple: A tuple containing the quotient and remainder.
    """
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")

    quotient = dividend // divisor
    remainder = dividend % divisor

    return quotient, remainder


def main():
    # Get the dividend and divisor from the user
    dividend = int(input("Enter the dividend: "))
    divisor = int(input("Enter the divisor: "))

    try:
        quotient, remainder = calculate_quotient_and_remainder(dividend, divisor)
        print(f"The quotient is: {quotient}")
        print(f"The remainder is: {remainder}")
    except ZeroDivisionError as e:
        print(e)


if __name__ == "__main__":
    main()
```
