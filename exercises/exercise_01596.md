## Exercise
Create a program to create a phonebook that allows users to add, remove, and search for contacts.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, number):
        if name in self.contacts:
            print(f"Contact with name '{name}' already exists. Would you like to update the number? (yes/no)")
            choice = input().lower()
            if choice == "yes":
                self.contacts[name] = number
                print(f"Contact with name '{name}' updated successfully.")
            else:
                print("Contact not updated.")
        else:
            self.contacts[name] = number
            print(f"Contact with name '{name}' added successfully.")

    def remove_contact(self, name):
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact with name '{name}' removed successfully.")
        else:
            print(f"Contact with name '{name}' not found.")

    def search_contact(self, name):
        if name in self.contacts:
            print(f"Contact with name '{name}' found. Number: {self.contacts[name]}")
        else:
            print(f"Contact with name '{name}' not found.")

    def display_contacts(self):
        if not self.contacts:
            print("Phonebook is empty.")
        else:
            print("Phonebook:")
            for name, number in self.contacts.items():
                print(f"Name: {name}, Number: {number}")


def main():
    phonebook = Phonebook()
    while True:
        print("\nPhonebook Menu:")
        print("1. Add Contact")
        print("2. Remove Contact")
        print("3. Search Contact")
        print("4. Display Contacts")
        print("5. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            name = input("Enter contact name: ")
            number = input("Enter contact number: ")
            phonebook.add_contact(name, number)
        elif choice == "2":
            name = input("Enter contact name: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter contact name: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Exiting phonebook.")
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
