## Exercise
Create a program that finds the highest score among a list of scores using if/else statements

## Reference Solution
```python
def find_highest_score(scores):
    """
    Finds the highest score among a list of scores.

    Args:
        scores (list): A list of scores.

    Returns:
        int: The highest score.
    """
    if len(scores) == 0:
        return None  # Return None if the list is empty

    highest_score = scores[0]  # Initialize the highest score as the first score

    for score in scores[1:]:
        if score > highest_score:
            highest_score = score  # Update the highest score if a higher score is found

    return highest_score


def main():
    # Example usage
    scores = [80, 90, 70, 95, 85]
    highest_score = find_highest_score(scores)

    if highest_score is not None:
        print(f"The highest score is: {highest_score}")
    else:
        print("The list of scores is empty.")


if __name__ == "__main__":
    main()
```
