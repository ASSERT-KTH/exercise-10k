## Exercise
Write a function that partitions a given list into two separate lists based on a specified condition.

## Reference Solution
```python
def partition_list(input_list, condition):
    """
    Partitions the input list into two separate lists based on a specified condition.

    Args:
        input_list (list): The list to be partitioned.
        condition (function): A function that takes an element and returns a boolean value.

    Returns:
        tuple: A tuple containing two lists. The first list contains elements for which the condition is True,
               and the second list contains elements for which the condition is False.
    """

    # Initialize two empty lists to store the partitioned elements
    true_list = []
    false_list = []

    # Iterate over each element in the input list
    for element in input_list:
        # Check the condition for the current element
        if condition(element):
            # If the condition is True, append the element to the true list
            true_list.append(element)
        else:
            # If the condition is False, append the element to the false list
            false_list.append(element)

    # Return the partitioned lists as a tuple
    return true_list, false_list


# Example usage:
if __name__ == "__main__":
    # Define a condition function that checks if a number is even
    def is_even(num):
        return num % 2 == 0

    # Create a sample list of numbers
    numbers = [1, 2, 3, 4, 5, 6]

    # Partition the list based on the condition
    even_numbers, odd_numbers = partition_list(numbers, is_even)

    # Print the partitioned lists
    print("Even numbers:", even_numbers)
    print("Odd numbers:", odd_numbers)
```
