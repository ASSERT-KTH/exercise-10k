## Exercise
Write a program that takes input from a file and prints out the output to another file.

## Reference Solution
```python
def read_input_file(input_filename):
    """
    Reads input from a file.

    Args:
        input_filename (str): The name of the input file.

    Returns:
        str: The contents of the input file.
    """
    try:
        with open(input_filename, 'r') as input_file:
            return input_file.read()
    except FileNotFoundError:
        print(f"Error: Input file '{input_filename}' not found.")
        return None


def write_output_file(output_filename, output_data):
    """
    Writes output to a file.

    Args:
        output_filename (str): The name of the output file.
        output_data (str): The data to be written to the output file.
    """
    try:
        with open(output_filename, 'w') as output_file:
            output_file.write(output_data)
    except Exception as e:
        print(f"Error: Unable to write to output file '{output_filename}'. {str(e)}")


def process_input_data(input_data):
    """
    Processes the input data.

    Args:
        input_data (str): The input data to be processed.

    Returns:
        str: The processed output data.
    """
    # Example processing: Convert input data to uppercase
    return input_data.upper()


def main():
    input_filename = 'input.txt'
    output_filename = 'output.txt'

    input_data = read_input_file(input_filename)
    if input_data is not None:
        output_data = process_input_data(input_data)
        write_output_file(output_filename, output_data)
        print(f"Output written to '{output_filename}'")


if __name__ == '__main__':
    main()
```
