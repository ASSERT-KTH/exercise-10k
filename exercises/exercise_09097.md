## Exercise
Given a sequence of integers, find the length of the longest increasing subsequence. A subsequence is a sequence that can be derived from another sequence by deleting some elements without changing the order of the remaining elements.

## Reference Solution
```python
def longest_increasing_subsequence(sequence):
    """
    Find the length of the longest increasing subsequence in a sequence of integers.

    Args:
        sequence (list): A list of integers.

    Returns:
        int: The length of the longest increasing subsequence.
    """
    if not sequence:
        return 0

    n = len(sequence)
    dp = [1] * n

    for i in range(1, n):
        for j in range(i):
            if sequence[i] > sequence[j]:
                dp[i] = max(dp[i], dp[j] + 1)

    return max(dp)


# Example usage
sequence = [10, 22, 9, 33, 21, 50, 41, 60, 80]
print("Length of LIS:", longest_increasing_subsequence(sequence))
```

```python
from bisect import bisect_left

def longest_increasing_subsequence(sequence):
    """
    Find the length of the longest increasing subsequence in a sequence of integers.

    Args:
        sequence (list): A list of integers.

    Returns:
        int: The length of the longest increasing subsequence.
    """
    if not sequence:
        return 0

    dp = [sequence[0]]

    for num in sequence[1:]:
        if num > dp[-1]:
            dp.append(num)
        else:
            idx = bisect_left(dp, num)
            dp[idx] = num

    return len(dp)


# Example usage
sequence = [10, 22, 9, 33, 21, 50, 41, 60, 80]
print("Length of LIS:", longest_increasing_subsequence(sequence))
```
