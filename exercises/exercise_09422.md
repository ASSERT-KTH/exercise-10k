## Exercise
Create a function that takes a tuple and returns the first element

## Reference Solution
```python
def get_first_element(input_tuple):
    """
    Returns the first element of a tuple.

    Args:
        input_tuple (tuple): The input tuple.

    Returns:
        The first element of the tuple.

    Raises:
        IndexError: If the tuple is empty.
    """
    if len(input_tuple) == 0:
        raise IndexError("Tuple is empty")
    return input_tuple[0]

# Example usage:
def main():
    example_tuple = (1, 2, 3)
    print(get_first_element(example_tuple))  # Output: 1

    empty_tuple = ()
    try:
        print(get_first_element(empty_tuple))
    except IndexError as e:
        print(e)  # Output: Tuple is empty

if __name__ == "__main__":
    main()
```
