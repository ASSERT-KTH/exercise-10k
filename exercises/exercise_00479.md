## Exercise
Write a program that takes a string as input and returns a spiral pattern of the string

## Reference Solution

def generate_spiral(string, size):
    """
    Generate a spiral pattern of the input string.

    Args:
    - string (str): The input string.
    - size (int): The size of the spiral pattern.

    Returns:
    - spiral (list): A 2D list representing the spiral pattern.
    """
    # Initialize the spiral pattern with empty spaces
    spiral = [[' ' for _ in range(size)] for _ in range(size)]

    # Initialize the directions (right, down, left, up)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]

    # Initialize the current position and direction
    row, col = 0, 0
    dir_index = 0

    # Fill the spiral pattern with the input string
    for char in string:
        spiral[row][col] = char

        # Move to the next position
        next_row, next_col = row + directions[dir_index][0], col + directions[dir_index][1]

        # Check if the next position is within the bounds and not occupied
        if (0 <= next_row < size and 0 <= next_col < size and spiral[next_row][next_col] == ' '):
            row, col = next_row, next_col
        else:
            # Change direction if the next position is not valid
            dir_index = (dir_index + 1) % 4
            row, col = row + directions[dir_index][0], col + directions[dir_index][1]

    return spiral

def print_spiral(spiral):
    """
    Print the spiral pattern.

    Args:
    - spiral (list): A 2D list representing the spiral pattern.
    """
    for row in spiral:
        print(' '.join(row))

# Example usage
string = "HELLO WORLD"
size = 10

spiral = generate_spiral(string, size)
print_spiral(spiral)
