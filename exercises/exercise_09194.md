## Exercise
Use built-in functions to convert between data types

## Reference Solution
```python
# Convert a string to an integer
num_str = "10"
num_int = int(num_str)
print(num_int)  # Output: 10

# Convert a float to an integer
num_float = 10.5
num_int = int(num_float)
print(num_int)  # Output: 10

# Convert an integer to a float
num_int = 10
num_float = float(num_int)
print(num_float)  # Output: 10.0

# Convert an integer to a complex number
num_int = 10
num_complex = complex(num_int)
print(num_complex)  # Output: (10+0j)
```

```python
# Convert an integer to a string

num_str = str(num_int)

# Convert a float to a string
num_float = 10.5
num_str = str(num_float)
print(num_str)  # Output: "10.5"

# Convert a list to a string
my_list = [1, 2, 3]

```

```python
# Convert an integer to a boolean
num_int = 10
bool_value = bool(num_int)
print(bool_value)  # Output: True

# Convert a string to a boolean
my_str = "Hello"
bool_value = bool(my_str)
print(bool_value)  # Output: True

# Convert an empty string to a boolean
my_str = ""
bool_value = bool(my_str)
print(bool_value)  # Output: False
```

```python
# Convert a string to a list
my_str = "Hello"
list_chars = list(my_str)
print(list_chars)  # Output: ["H", "e", "l", "l", "o"]

# Convert a list to a tuple
my_list = [1, 2, 3]
my_tuple = tuple(my_list)
print(my_tuple)  # Output: (1, 2, 3)

# Convert a tuple to a list
my_tuple = (1, 2, 3)
my_list = list(my_tuple)
print(my_list)  # Output: [1, 2, 3]
```

```python
# Convert a list of tuples to a dictionary
my_list = [(1, "a"), (2, "b"), (3, "c")]
my_dict = dict(my_list)
print(my_dict)  # Output: {1: "a", 2: "b", 3: "c"}
```
