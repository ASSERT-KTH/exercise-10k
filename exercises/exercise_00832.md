## Exercise
Given a string and a maximum line width, justify the text by adding spaces between the words to fill the lines.

## Reference Solution

def full_justify(text, max_width):
    """
    Justify the given text by adding spaces between words to fill lines.

    Args:
    text (str): The input text.
    max_width (int): The maximum line width.

    Returns:
    list: A list of justified lines.
    """
    words = text.split()
    justified_lines = []
    current_line = []
    current_width = 0

    for word in words:
        # Check if adding the current word exceeds the max width
        if current_width + len(word) + len(current_line) > max_width:
            # Justify the current line
            justified_lines.append(justify_line(current_line, max_width))
            # Reset the current line and width
            current_line = [word]
            current_width = len(word)
        else:
            # Add the word to the current line
            current_line.append(word)
            current_width += len(word)

    # Justify the last line
    justified_lines.append(justify_last_line(current_line, max_width))

    return justified_lines


def justify_line(words, max_width):
    """
    Justify a line of text by adding spaces between words.

    Args:
    words (list): A list of words.
    max_width (int): The maximum line width.

    Returns:
    str: The justified line.
    """
    total_width = sum(len(word) for word in words)
    num_spaces = max_width - total_width
    num_gaps = len(words) - 1

    if num_gaps == 0:
        # If there's only one word, add all spaces to the end
        return words[0] + ' ' * num_spaces

    space_per_gap = num_spaces // num_gaps
    extra_spaces = num_spaces % num_gaps

    justified_line = ''
    for i, word in enumerate(words):
        justified_line += word
        if i < len(words) - 1:
            justified_line += ' ' * (space_per_gap + (1 if i < extra_spaces else 0))

    return justified_line


def justify_last_line(words, max_width):
    """
    Justify the last line of text by adding spaces between words and to the end.

    Args:
    words (list): A list of words.
    max_width (int): The maximum line width.

    Returns:
    str: The justified last line.
    """
    justified_line = ' '.join(words)
    return justified_line + ' ' * (max_width - len(justified_line))


# Example usage
text = "This is an example of text justification."
max_width = 16
justified_lines = full_justify(text, max_width)

for line in justified_lines:
    print(line)
