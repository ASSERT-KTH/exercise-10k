## Exercise
Write a program that asks the user for a number and prints out its square root.

## Reference Solution
```python
import math

def get_user_input():
    """Get a number from the user."""
    while True:
        try:
            num = float(input("Enter a number: "))
            if num < 0:
                print("Error: Square root of negative numbers is not a real number.")
                continue
            return num
        except ValueError:
            print("Error: Invalid input. Please enter a number.")

def calculate_square_root(num):
    """Calculate the square root of a number."""
    return math.sqrt(num)

def main():
    num = get_user_input()
    square_root = calculate_square_root(num)
    print(f"The square root of {num} is {square_root:.4f}")

if __name__ == "__main__":
    main()
```
