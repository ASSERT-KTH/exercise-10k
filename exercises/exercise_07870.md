## Exercise
Create a program that checks if a list of words is in alphabetical order using conditional statements.

## Reference Solution
```python
def is_alphabetical_order(word_list):
    """
    Checks if a list of words is in alphabetical order.

    Args:
        word_list (list): A list of words.

    Returns:
        bool: True if the list is in alphabetical order, False otherwise.
    """
    # Iterate over the list of words
    for i in range(len(word_list) - 1):
        # Check if the current word is greater than the next word
        if word_list[i].lower() > word_list[i + 1].lower():
            # If it is, return False
            return False
    # If the loop completes without finding any words out of order, return True
    return True


# Example usage:
words = ["apple", "banana", "cherry", "date", "elderberry"]
if is_alphabetical_order(words):
    print("The list is in alphabetical order.")
else:
    print("The list is not in alphabetical order.")

# Test with an unordered list
unordered_words = ["cherry", "banana", "apple", "date", "elderberry"]
if is_alphabetical_order(unordered_words):
    print("The list is in alphabetical order.")
else:
    print("The list is not in alphabetical order.")
```
