## Exercise
Create a function that formats a given phone number by adding parentheses and a hyphen.

## Reference Solution
```python
def format_phone_number(phone_number: str) -> str:
    """
    Formats a given phone number by adding parentheses and a hyphen.
    
    Args:
        phone_number (str): The phone number to be formatted. It should be 10 digits long.
    
    Returns:
        str: The formatted phone number.
    """
    # Check if the phone number is 10 digits long
    if len(phone_number) != 10:
        raise ValueError("Phone number should be 10 digits long")

    # Remove any non-digit characters from the phone number
    phone_number = ''.join(filter(str.isdigit, phone_number))

    # Format the phone number
    formatted_phone_number = f"({phone_number[:3]}) {phone_number[3:6]}-{phone_number[6:]}"

    return formatted_phone_number

# Example usage
if __name__ == "__main__":
    phone_number = "1234567890"
    formatted_phone_number = format_phone_number(phone_number)
    print(f"Formatted phone number: {formatted_phone_number}")
```
