## Exercise
Write a program to perform basic matrix operations (e.g. addition, multiplication).

## Reference Solution
```python
import numpy as np

def add_matrices(mat1, mat2):
    """
    Adds two matrices element-wise.

    Args:
        mat1 (numpy.ndarray): The first matrix.
        mat2 (numpy.ndarray): The second matrix.

    Returns:
        numpy.ndarray: The sum of the two matrices.
    """
    return np.add(mat1, mat2)

def multiply_matrices(mat1, mat2):
    """
    Multiplies two matrices using the dot product.

    Args:
        mat1 (numpy.ndarray): The first matrix.
        mat2 (numpy.ndarray): The second matrix.

    Returns:
        numpy.ndarray: The product of the two matrices.
    """
    return np.dot(mat1, mat2)

def transpose_matrix(mat):
    """
    Transposes a matrix.

    Args:
        mat (numpy.ndarray): The matrix to transpose.

    Returns:
        numpy.ndarray: The transposed matrix.
    """
    return np.transpose(mat)

def main():
    # Example matrices
    mat1 = np.array([[1, 2], [3, 4]])
    mat2 = np.array([[5, 6], [7, 8]])

    print("Matrix 1:")
    print(mat1)
    print("\nMatrix 2:")
    print(mat2)

    # Perform matrix operations
    sum_mat = add_matrices(mat1, mat2)
    product_mat = multiply_matrices(mat1, mat2)
    transpose_mat1 = transpose_matrix(mat1)

    print("\nSum of Matrices:")
    print(sum_mat)
    print("\nProduct of Matrices:")
    print(product_mat)
    print("\nTranspose of Matrix 1:")
    print(transpose_mat1)

if __name__ == "__main__":
    main()
```
