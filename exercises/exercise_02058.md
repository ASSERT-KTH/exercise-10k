## Exercise
Write a function to merge two dictionaries with overlapping keys.

## Reference Solution
```python
def merge_dicts(dict1, dict2):
    """
    Merge two dictionaries with overlapping keys.
    
    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.
    
    Returns:
        dict: The merged dictionary.
    """
    # Create a copy of the first dictionary
    merged_dict = dict1.copy()
    
    # Update the copied dictionary with the key-value pairs from the second dictionary
    merged_dict.update(dict2)
    
    return merged_dict

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 20, 'c': 30, 'd': 40}

merged_dict = merge_dicts(dict1, dict2)
print(merged_dict)  # Output: {'a': 1, 'b': 20, 'c': 30, 'd': 40}
```

```python
def merge_dicts(dict1, dict2):
    """
    Merge two dictionaries with overlapping keys, combining values into lists.
    
    Args:
        dict1 (dict): The first dictionary.
        dict2 (dict): The second dictionary.
    
    Returns:
        dict: The merged dictionary.
    """
    # Create a set of all keys from both dictionaries
    all_keys = set(dict1.keys()) | set(dict2.keys())
    
    # Create a new dictionary with combined values
    merged_dict = {key: [dict1.get(key), dict2.get(key)] for key in all_keys}
    
    # Remove None values from the lists
    merged_dict = {key: [value for value in values if value is not None] for key, values in merged_dict.items()}
    
    return merged_dict

# Example usage
dict1 = {'a': 1, 'b': 2, 'c': 3}
dict2 = {'b': 20, 'c': 30, 'd': 40}

merged_dict = merge_dicts(dict1, dict2)
print(merged_dict)  # Output: {'a': [1], 'b': [2, 20], 'c': [3, 30], 'd': [40]}
```
