## Exercise
Write a program that checks the data type of user input and prints it out.

## Reference Solution
```python
def check_data_type(user_input):
    """Checks the data type of the user input."""
    try:
        # Try to convert the input to an integer
        int(user_input)
        return "Integer"
    except ValueError:
        try:
            # Try to convert the input to a float
            float(user_input)
            return "Float"
        except ValueError:
            # If the input is not a number, check if it's a boolean
            if user_input.lower() == "true" or user_input.lower() == "false":
                return "Boolean"
            else:
                # If the input is not a number or a boolean, it's a string
                return "String"

def main():
    user_input = input("Please enter something: ")
    data_type = check_data_type(user_input)
    print(f"The data type of your input is: {data_type}")

if __name__ == "__main__":
    main()
```
