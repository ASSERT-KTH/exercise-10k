## Exercise
Create a program that uses a while loop to simulate a bouncing ball.

## Reference Solution
```python
import time

# Constants
GRAVITY = 9.81  # m/s^2
ELASTICITY = 0.7  # coefficient of restitution (0 = perfectly inelastic, 1 = perfectly elastic)

class Ball:
    def __init__(self, initial_height, initial_velocity):
        self.height = initial_height
        self.velocity = initial_velocity

    def update(self, dt):
        self.velocity += -GRAVITY * dt
        self.height += self.velocity * dt

        if self.height < 0:
            self.height = 0
            self.velocity = -self.velocity * ELASTICITY

    def print_state(self):
        print(f"Height: {self.height:.2f} m, Velocity: {self.velocity:.2f} m/s")

def main():
    initial_height = 10  # m
    initial_velocity = 0  # m/s
    dt = 0.01  # s (time step)

    ball = Ball(initial_height, initial_velocity)

    while ball.height > 0 and ball.velocity != 0:
        ball.update(dt)
        ball.print_state()
        time.sleep(dt)

if __name__ == "__main__":
    main()
```
