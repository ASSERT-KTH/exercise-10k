## Exercise
Write a function to count the frequency of each character in a given string

## Reference Solution
```python
def count_character_frequencies(input_string):
    """
    Counts the frequency of each character in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where the keys are the unique characters in the string
              and the values are their corresponding frequencies.
    """
    frequency_dict = {}
    for char in input_string:
        if char in frequency_dict:
            frequency_dict[char] += 1
        else:
            frequency_dict[char] = 1
    return frequency_dict

# Example usage:
def main():
    input_str = "Hello, World!"
    frequency_dict = count_character_frequencies(input_str)
    for char, frequency in frequency_dict.items():
        print(f"Character '{char}': {frequency}")

if __name__ == "__main__":
    main()
```

from collections import Counter

def count_character_frequencies(input_string):
    """
    Counts the frequency of each character in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        dict: A dictionary where the keys are the unique characters in the string
              and the values are their corresponding frequencies.
    """
    return dict(Counter(input_string))

# Example usage:

    frequency_dict = count_character_frequencies(input_str)
