## Exercise
Write a program that takes a string as input and checks if it is a valid password. A valid password must be at least 8 characters long, contain at least one uppercase letter, one lowercase letter, and one digit.

## Reference Solution
```python
import re

def is_valid_password(password: str) -> bool:
    """
    Checks if a given string is a valid password.

    A valid password must be at least 8 characters long, contain at least one uppercase letter, one lowercase letter, and one digit.

    Args:
        password (str): The password to be checked.

    Returns:
        bool: True if the password is valid, False otherwise.
    """
    # Check if password is at least 8 characters long
    if len(password) < 8:
        return False

    # Check if password contains at least one uppercase letter, one lowercase letter, and one digit
    if not re.search(r"[A-Z]", password) or not re.search(r"[a-z]", password) or not re.search(r"\d", password):
        return False

    return True

def main():
    # Example usage
    passwords = [
        "short",  # Too short
        "alllowercase",  # No uppercase letter
        "ALLUPPERCASE",  # No lowercase letter
        "nodash1",  # No digit
        "ValidP@ssw0rd"  # Valid password
    ]

    for password in passwords:
        print(f"Password: {password}, Valid: {is_valid_password(password)}")

if __name__ == "__main__":
    main()
```
