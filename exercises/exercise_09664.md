## Exercise
Write a program that asks for a sentence and prints out the longest word.

## Reference Solution
```python
def find_longest_word(sentence):
    """Split the sentence into words and return the longest word."""
    words = sentence.split()
    longest_word = max(words, key=len)
    return longest_word

def main():
    # Get a sentence from the user
    sentence = input("Please enter a sentence: ")

    # Find and print the longest word
    longest_word = find_longest_word(sentence)
    print(f"The longest word is: {longest_word}")

if __name__ == "__main__":
    main()
```

```python
def find_longest_word(sentence):
    """Split the sentence into words and return all longest words."""
    words = sentence.split()
    max_length = max(len(word) for word in words)
    longest_words = [word for word in words if len(word) == max_length]
    return longest_words
```
