## Exercise
Write a program to calculate the GPA of a student based on their grades.

## Reference Solution
```python
def calculate_gpa(grades, credits):
    """
    Calculates the GPA of a student based on their grades.

    Args:
        grades (list): A list of grade points for each subject.
        credits (list): A list of credits for each subject.

    Returns:
        float: The calculated GPA.
    """
    total_points = sum([grade * credit for grade, credit in zip(grades, credits)])
    total_credits = sum(credits)
    return total_points / total_credits

def main():
    # Example usage:
    subjects = ["Math", "Science", "English", "History"]
    grades = [4.0, 3.5, 4.5, 3.0]  # Grade points for each subject
    credits = [3, 4, 3, 4]  # Credits for each subject

    print("GPA Calculator")
    print("----------------")
    for subject, grade, credit in zip(subjects, grades, credits):
        print(f"{subject}: Grade = {grade}, Credits = {credit}")

    gpa = calculate_gpa(grades, credits)
    print(f"\nGPA: {gpa:.2f}")

if __name__ == "__main__":
    main()
```
