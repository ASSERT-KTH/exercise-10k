## Exercise
Create a program that takes a story template with placeholders for different types of words (e.g. noun, verb, adjective), then asks the user to input words to fill in the placeholders and prints out the completed story.

## Reference Solution
```python
# Import the required libraries
import re

def get_word_from_user(prompt):
    """Get a word from the user."""
    while True:
        word = input(prompt)
        if word:  # Check if the input is not empty
            return word
        print("Please enter a word.")

def fill_in_template(template):
    """Fill in the template with user input."""
    # Define a dictionary to store the word types and their corresponding prompts
    word_types = {
        "noun": "Enter a noun: ",
        "verb": "Enter a verb: ",
        "adjective": "Enter an adjective: ",
        "adverb": "Enter an adverb: ",
        "place": "Enter a place: ",
        "person": "Enter a person's name: "
    }

    # Use regular expression to find the placeholders in the template
    placeholders = re.findall(r"\{(.*?)\}", template)

    # Create a dictionary to store the user input
    user_input = {}

    # Ask the user to input words for each placeholder
    for placeholder in placeholders:
        word_type = placeholder.lower()
        if word_type in word_types:
            user_input[placeholder] = get_word_from_user(word_types[word_type])
        else:
            print(f"Unknown word type: {word_type}")

    # Fill in the template with the user input
    completed_story = template
    for placeholder, word in user_input.items():
        completed_story = completed_story.replace("{" + placeholder + "}", word)

    return completed_story

def main():
    # Define a story template with placeholders
    template = "Once upon a time, in a {adjective} {place}, there lived a {adjective} {person} who loved to {verb} {adverb}. One day, while wandering through the {place}, they stumbled upon a {noun} and decided to take it home."

    # Fill in the template with user input and print the completed story
    completed_story = fill_in_template(template)
    print("\nCompleted Story:")
    print(completed_story)

if __name__ == "__main__":
    main()
```
