## Exercise
Write a program that calculates the total score given a list of scores and their respective weights.

## Reference Solution
```python
def calculate_weighted_score(scores, weights):
    """
    Calculate the total weighted score.

    Args:
        scores (list): A list of scores.
        weights (list): A list of weights corresponding to the scores.

    Returns:
        float: The total weighted score.
    """
    if len(scores) != len(weights):
        raise ValueError("The number of scores and weights must be equal")

    total_weighted_score = sum(score * weight for score, weight in zip(scores, weights))
    total_weight = sum(weights)

    if total_weight == 0:
        raise ZeroDivisionError("The total weight cannot be zero")

    return total_weighted_score / total_weight


def main():
    # Example usage
    scores = [90, 80, 70, 60]
    weights = [0.3, 0.2, 0.2, 0.3]

    try:
        total_weighted_score = calculate_weighted_score(scores, weights)
        print(f"Total Weighted Score: {total_weighted_score:.2f}")
    except (ValueError, ZeroDivisionError) as e:
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
