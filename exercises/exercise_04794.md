## Exercise
Write a program that asks the user for words and then fills in the blanks in a story with the words.

## Reference Solution
```python
# Import the required libraries
import re

def get_words_from_user():
    """Get words from the user."""
    print("Please provide the following words:")
    words = {
        "noun": input("Enter a noun: "),
        "adjective": input("Enter an adjective: "),
        "verb": input("Enter a verb: "),
        "place": input("Enter a place: ")
    }
    return words

def fill_in_the_blanks(words):
    """Fill in the blanks in the story with the words."""
    story = f"One day, a very {words['adjective']} {words['noun']} was walking down the street. " \
            f"Suddenly, the {words['noun']} decided to {words['verb']} to the {words['place']}. " \
            f"When the {words['noun']} arrived at the {words['place']}, it was amazed by the beauty of the place."
    return story

def main():
    """Main function."""
    words = get_words_from_user()
    story = fill_in_the_blanks(words)
    print("\nHere is the story with the blanks filled in:")
    print(story)

if __name__ == "__main__":
    main()
```
