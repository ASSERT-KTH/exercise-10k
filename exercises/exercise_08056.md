## Exercise
Write a program that uses an if/else statement to manage a to-do list with tags based on user input.

## Reference Solution
```python
# Define a dictionary to store the to-do list
todo_list = {}

def add_task():
    """Add a task to the to-do list."""
    task_name = input("Enter the task name: ")
    tags = input("Enter the tags (comma-separated): ").split(',')
    tags = [tag.strip() for tag in tags]
    
    # Check if the task already exists
    if task_name in todo_list:
        print("Task already exists. Do you want to update the tags? (yes/no)")
        response = input().lower()
        if response == "yes":
            todo_list[task_name] = tags
            print("Tags updated successfully.")
        else:
            print("Operation cancelled.")
    else:
        todo_list[task_name] = tags
        print("Task added successfully.")

def delete_task():
    """Delete a task from the to-do list."""
    task_name = input("Enter the task name: ")
    
    # Check if the task exists
    if task_name in todo_list:
        del todo_list[task_name]
        print("Task deleted successfully.")
    else:
        print("Task does not exist.")

def update_task():
    """Update a task in the to-do list."""
    task_name = input("Enter the task name: ")
    
    # Check if the task exists
    if task_name in todo_list:
        print("Enter the new task name (press enter to skip): ")
        new_task_name = input()
        if new_task_name:
            todo_list[new_task_name] = todo_list.pop(task_name)
            task_name = new_task_name
        print("Enter the new tags (comma-separated, press enter to skip): ")
        new_tags = input()
        if new_tags:
            todo_list[task_name] = [tag.strip() for tag in new_tags.split(',')]
        print("Task updated successfully.")
    else:
        print("Task does not exist.")

def view_tasks():
    """View all tasks in the to-do list."""
    if todo_list:
        for task, tags in todo_list.items():
            print(f"Task: {task}, Tags: {', '.join(tags)}")
    else:
        print("No tasks in the to-do list.")

def main():
    while True:
        print("\nTo-Do List Menu:")
        print("1. Add Task")
        print("2. Delete Task")
        print("3. Update Task")
        print("4. View Tasks")
        print("5. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            add_task()
        elif choice == "2":
            delete_task()
        elif choice == "3":
            update_task()
        elif choice == "4":
            view_tasks()
        elif choice == "5":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
