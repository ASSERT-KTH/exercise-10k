## Exercise
Create a program that uses a dictionary to store student names as keys and their corresponding grades as values. The program should allow users to add, remove, and update student grades. It should also be able to calculate the average grade of all students and display the grades of a specific student.

## Reference Solution
```python
# Student Gradebook Program

# Initialize an empty dictionary to store student grades
student_grades = {}

def display_menu():
    """Displays the main menu of the program"""
    print("\nStudent Gradebook Program")
    print("1. Add Student Grade")
    print("2. Remove Student Grade")
    print("3. Update Student Grade")
    print("4. Calculate Average Grade")
    print("5. Display Student Grade")
    print("6. Exit")

def add_student_grade():
    """Adds a new student grade to the dictionary"""
    name = input("Enter student name: ")
    while True:
        try:
            grade = float(input("Enter student grade: "))
            if 0 <= grade <= 100:
                student_grades[name] = grade
                print(f"Grade added for {name}")
                break
            else:
                print("Invalid grade. Please enter a grade between 0 and 100.")
        except ValueError:
            print("Invalid input. Please enter a number.")

def remove_student_grade():
    """Removes a student grade from the dictionary"""
    name = input("Enter student name: ")
    if name in student_grades:
        del student_grades[name]
        print(f"Grade removed for {name}")
    else:
        print(f"No grade found for {name}")

def update_student_grade():
    """Updates a student grade in the dictionary"""
    name = input("Enter student name: ")
    if name in student_grades:
        while True:
            try:
                grade = float(input("Enter new grade: "))
                if 0 <= grade <= 100:
                    student_grades[name] = grade
                    print(f"Grade updated for {name}")
                    break
                else:
                    print("Invalid grade. Please enter a grade between 0 and 100.")
            except ValueError:
                print("Invalid input. Please enter a number.")
    else:
        print(f"No grade found for {name}")

def calculate_average_grade():
    """Calculates the average grade of all students"""
    if student_grades:
        average_grade = sum(student_grades.values()) / len(student_grades)
        print(f"Average grade: {average_grade:.2f}")
    else:
        print("No grades available")

def display_student_grade():
    """Displays the grade of a specific student"""
    name = input("Enter student name: ")
    if name in student_grades:
        print(f"Grade for {name}: {student_grades[name]}")
    else:
        print(f"No grade found for {name}")

def main():
    while True:
        display_menu()
        choice = input("Enter your choice: ")
        if choice == "1":
            add_student_grade()
        elif choice == "2":
            remove_student_grade()
        elif choice == "3":
            update_student_grade()
        elif choice == "4":
            calculate_average_grade()
        elif choice == "5":
            display_student_grade()
        elif choice == "6":
            print("Exiting program...")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
