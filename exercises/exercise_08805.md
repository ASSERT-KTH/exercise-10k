## Exercise
Write a program that takes an angle in radians as input and returns its sine, cosine, and tangent.

## Reference Solution
```python
#include <iostream>
#include <cmath>

using namespace std;

void sin_cos_tan(double angle) {
    double sin_x = sin(angle);
    double cos_x = cos(angle);
    double tan_x = tan(angle);

    cout << "sin(" << angle << ") = " << sin_x << endl;
    cout << "cos(" << angle << ") = " << cos_x << endl;
    cout << "tan(" << angle << ") = " << tan_x << endl;
}

int main() {
    double angle = 1.0;
    sin_cos_tan(angle);

    return 0;
}
```
