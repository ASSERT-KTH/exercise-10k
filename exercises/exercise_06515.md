## Exercise
Write a program to understand the concept of variable scope (local, global).

## Reference Solution
```python
# Global variable
x = 10

def local_scope_example():
    """
    Function to demonstrate local variable scope.
    """
    # Local variable
    y = 20
    print("Local variable y:", y)

def global_scope_example():
    """
    Function to demonstrate global variable scope.
    """
    global x  # Access global variable x
    x = 30  # Modify global variable x
    print("Global variable x (modified):", x)

def nested_scope_example():
    """
    Function to demonstrate nested variable scope.
    """
    z = 40  # Local variable

    def inner_scope():
        nonlocal z  # Access outer local variable z
        z = 50  # Modify outer local variable z
        print("Local variable z (modified):", z)

    inner_scope()
    print("Local variable z (after modification):", z)

# Example usage
if __name__ == "__main__":
    print("Initial global variable x:", x)
    local_scope_example()
    print("Global variable x (unchanged):", x)
    global_scope_example()
    print("Global variable x (final):", x)
    nested_scope_example()
```
