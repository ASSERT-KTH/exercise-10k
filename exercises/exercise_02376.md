## Exercise
Develop a program that converts time from one unit to another (seconds to minutes, hours to days).

def convert_time(value, from_unit, to_unit):
    """
    Converts time from one unit to another.

    Args:
        value (float): The time value to convert.
        from_unit (str): The unit of the input time. Can be 'seconds', 'minutes', 'hours', 'days', 'weeks', 'months', 'years'.
        to_unit (str): The unit to convert the time to. Can be 'seconds', 'minutes', 'hours', 'days', 'weeks', 'months', 'years'.

    Returns:
        float: The converted time value.
    """

    # Define conversion factors
    conversion_factors = {
        'seconds': {'minutes': 60, 'hours': 3600, 'days': 86400, 'weeks': 604800, 'months': 2629743, 'years': 31556952},
        'minutes': {'seconds': 1/60, 'hours': 60, 'days': 1440, 'weeks': 10080, 'months': 43800, 'years': 525600},
        'hours': {'seconds': 1/3600, 'minutes': 1/60, 'days': 24, 'weeks': 168, 'months': 720, 'years': 8760},
        'days': {'seconds': 1/86400, 'minutes': 1/1440, 'hours': 1/24, 'weeks': 7, 'months': 30.44, 'years': 365.25},
        'weeks': {'seconds': 1/604800, 'minutes': 1/10080, 'hours': 1/168, 'days': 1/7, 'months': 4.33, 'years': 52.14},
        'months': {'seconds': 1/2629743, 'minutes': 1/43800, 'hours': 1/720, 'days': 1/30.44, 'weeks': 1/4.33, 'years': 12},
        'years': {'seconds': 1/31556952, 'minutes': 1/525600, 'hours': 1/8760, 'days': 1/365.25, 'weeks': 1/52.14, 'months': 1/12}
    }

    # Check if from_unit and to_unit are valid
    if from_unit not in conversion_factors or to_unit not in conversion_factors:
        raise ValueError("Invalid from_unit or to_unit")

    # Check if from_unit is the same as to_unit
    if from_unit == to_unit:
        return value

    # Convert time
    return value * conversion_factors[from_unit][to_unit]


def main():
    print("Time Unit Converter Program")
    print("----------------------------")

    # Get user input
    value = float(input("Enter the time value: "))
    from_unit = input("Enter the unit of the input time (seconds, minutes, hours, days, weeks, months, years): ")
    to_unit = input("Enter the unit to convert the time to (seconds, minutes, hours, days, weeks, months, years): ")

    # Convert time
    try:
        converted_value = convert_time(value, from_unit, to_unit)
        print(f"{value} {from_unit} is equal to {converted_value} {to_unit}")
    except ValueError as e:
        print(e)


if __name__ == "__main__":
    main()
