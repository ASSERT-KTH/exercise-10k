## Exercise
Write a program that takes an array and a number of positions as input, and rotates the array to the right by the given number of positions. For example, if the input array is [1, 2, 3, 4, 5] and the number of positions is 2, the output should be [4, 5, 1, 2, 3].

## Reference Solution
```python
#include <stdio.h>
#include <stdlib.h>

int* rotate(int* nums, int numsSize, int k, int* returnSize) {
    int* res = (int*)malloc(numsSize * sizeof(int));
    int i = 0;
    k = k % numsSize;
    for (i = 0; i < k; i++) {
        res[i] = nums[numsSize - k + i];
    }
    for (i = k; i < numsSize; i++) {
        res[i] = nums[i - k];
    }
    *returnSize = numsSize;
    return res;
}

int main() {
    int nums[] = {1, 2, 3, 4, 5};
    int numsSize = sizeof(nums) / sizeof(nums[0]);
    int k = 2;
    int returnSize;
    int* res = rotate(nums, numsSize, k, &returnSize);
    for (int i = 0; i < returnSize; i++) {
        printf("%d ", res[i]);
    }
    printf("\n");
    free(res);
    return 0;
}
```
