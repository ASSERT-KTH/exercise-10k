## Exercise
Write a function that takes a dictionary and a key as input and returns True if the key is in the dictionary and False otherwise

## Reference Solution

def check_key(dictionary, key):
    """
    Checks if a key exists in a dictionary.

    Args:
        dictionary (dict): The dictionary to check.
        key (any): The key to look for.

    Returns:
        bool: True if the key is in the dictionary, False otherwise.
    """
    return key in dictionary

# Example usage:
def main():
    example_dict = {"apple": 1, "banana": 2, "cherry": 3}
    print(check_key(example_dict, "apple"))  # Output: True
    print(check_key(example_dict, "grape"))   # Output: False

if __name__ == "__main__":
    main()
