## Exercise
Write a program that uses pipes to connect input and output streams

## Reference Solution
```python
import subprocess

def connect_streams(input_stream, output_stream):
    """
    Connects the input and output streams using a pipe.
    
    Args:
        input_stream (str): The input stream to read from.
        output_stream (str): The output stream to write to.
    
    Returns:
        None
    """
    # Create a pipe
    pipe = subprocess.Popen(['cat'], stdin=subprocess.PIPE, stdout=subprocess.PIPE)
    
    # Write to the input stream
    with open(input_stream, 'r') as f:
        input_data = f.read()
        pipe.stdin.write(input_data.encode())
        pipe.stdin.close()
    
    # Read from the output stream
    output_data = pipe.stdout.read().decode()
    with open(output_stream, 'w') as f:
        f.write(output_data)

def main():
    # Define the input and output streams
    input_stream = 'input.txt'
    output_stream = 'output.txt'
    
    # Create the input stream
    with open(input_stream, 'w') as f:
        f.write('Hello, World!')
    
    # Connect the input and output streams
    connect_streams(input_stream, output_stream)
    
    # Print the contents of the output stream
    with open(output_stream, 'r') as f:
        print(f.read())

if __name__ == '__main__':
    main()
```
