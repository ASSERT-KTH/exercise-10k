## Exercise
Write a program that takes the radius of a circle as input and outputs its area, circumference, and diameter.

## Reference Solution
```python
import math

def calculate_circle_properties(radius):
    """
    Calculate the area, circumference, and diameter of a circle.

    Args:
        radius (float): The radius of the circle.

    Returns:
        tuple: A tuple containing the area, circumference, and diameter of the circle.
    """
    if radius < 0:
        raise ValueError("Radius cannot be negative")

    area = math.pi * (radius ** 2)
    circumference = 2 * math.pi * radius
    diameter = 2 * radius

    return area, circumference, diameter

def main():
    # Get the radius from the user
    while True:
        try:
            radius = float(input("Enter the radius of the circle: "))
            if radius < 0:
                print("Radius cannot be negative. Please try again.")
            else:
                break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Calculate and display the circle properties
    area, circumference, diameter = calculate_circle_properties(radius)
    print(f"Area: {area:.2f}")
    print(f"Circumference: {circumference:.2f}")
    print(f"Diameter: {diameter:.2f}")

if __name__ == "__main__":
    main()
```
