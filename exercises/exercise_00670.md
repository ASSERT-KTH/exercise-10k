## Exercise
Given a dictionary {'name': 'Hannah', 'age': 20, 'city': 'Adelaide'}, write a program to create a copy of the dictionary.

## Reference Solution
```python
def create_dict_copy():
    # Original dictionary
    original_dict = {'name': 'Hannah', 'age': 20, 'city': 'Adelaide'}
    
    # Method 1: Using the copy() method
    dict_copy_method1 = original_dict.copy()
    
    # Method 2: Using the dict() constructor
    dict_copy_method2 = dict(original_dict)
    
    # Print the copied dictionaries
    print("Original Dictionary:", original_dict)
    print("Copied Dictionary (Method 1):", dict_copy_method1)
    print("Copied Dictionary (Method 2):", dict_copy_method2)

# Example usage
create_dict_copy()
```
