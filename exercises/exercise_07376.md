## Exercise
Create a program that encrypts a given string by shifting each character's ASCII value by 3 places to the right. If the character is a letter, it should wrap around the alphabet if necessary. Non-alphabetic characters should remain unchanged.

## Reference Solution
```python
public class Main {
    public static String encrypt(String s) {
        StringBuilder encrypted = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (Character.isLetter(c)) {
                if (Character.isUpperCase(c)) {
                    encrypted.append((char) ((c - 'A' + 3) % 26 + 'A'));
                } else {
                    encrypted.append((char) ((c - 'a' + 3) % 26 + 'a'));
                }
            } else {
                encrypted.append(c);
            }
        }
        return encrypted.toString();
    }

    public static void main(String[] args) {
        String s = "This is a test string.";
        System.out.println("Encrypted string: " + encrypt(s));
    }
}
```
