## Exercise
Write a program that demonstrates array manipulation techniques.

## Reference Solution
```python
import numpy as np

def main():
    # Create two arrays
    array1 = np.array([1, 2, 3, 4, 5])
    array2 = np.array([6, 7, 8, 9, 10])

    # Print the original arrays
    print("Original Array 1: ", array1)
    print("Original Array 2: ", array2)

    # Append array2 to array1
    appended_array = np.append(array1, array2)
    print("\nAppended Array: ", appended_array)

    # Insert elements into array1
    inserted_array = np.insert(array1, 2, [11, 12])
    print("\nInserted Array: ", inserted_array)

    # Delete elements from array1
    deleted_array = np.delete(array1, [0, 2])
    print("\nDeleted Array: ", deleted_array)

    # Reshape array1
    reshaped_array = array1.reshape(1, 5)
    print("\nReshaped Array: \n", reshaped_array)

    # Concatenate two arrays
    concatenated_array = np.concatenate((array1, array2))
    print("\nConcatenated Array: ", concatenated_array)

    # Split array1 into two parts
    split_arrays = np.split(array1, 2)
    print("\nSplit Arrays: ", split_arrays)

    # Find the index of the first occurrence of a value in array1
    index = np.where(array1 == 3)
    print("\nIndex of value 3: ", index)

    # Sort array1
    sorted_array = np.sort(array1)
    print("\nSorted Array: ", sorted_array)

if __name__ == "__main__":
    main()
```
