## Exercise
Write a program that takes two strings as input and returns whether the first string is a subsequence of the second string or not.

## Reference Solution
```python
def is_subsequence(s1: str, s2: str) -> bool:
    """
    Checks if s1 is a subsequence of s2.

    Args:
        s1 (str): The string to check.
        s2 (str): The string to check against.

    Returns:
        bool: True if s1 is a subsequence of s2, False otherwise.
    """
    i = 0  # Index for s1
    j = 0  # Index for s2

    while i < len(s1) and j < len(s2):
        if s1[i] == s2[j]:
            i += 1
        j += 1

    return i == len(s1)


def main():
    # Example usage
    s1 = "abc"
    s2 = "ahbgdc"
    print(f"'{s1}' is a subsequence of '{s2}': {is_subsequence(s1, s2)}")

    s1 = "axc"
    s2 = "ahbgdc"
    print(f"'{s1}' is a subsequence of '{s2}': {is_subsequence(s1, s2)}")


if __name__ == "__main__":
    main()
```
