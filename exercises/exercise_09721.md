## Exercise
Write a program that takes two rational numbers as input and returns their product.

## Reference Solution
```python
class Rational:
    def __init__(self, numerator, denominator):
        if denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        
        self.numerator = numerator
        self.denominator = denominator

    def __str__(self):
        return f"{self.numerator}/{self.denominator}"

    def multiply(self, other):
        numerator = self.numerator * other.numerator
        denominator = self.denominator * other.denominator
        return Rational(numerator, denominator)

    def simplify(self):
        def gcd(a, b):
            while b:
                a, b = b, a % b
            return a

        common_divisor = gcd(self.numerator, self.denominator)
        return Rational(self.numerator // common_divisor, self.denominator // common_divisor)


def multiply_rationals(r1, r2):
    product = r1.multiply(r2)
    return product.simplify()


# Example usage
if __name__ == "__main__":
    r1 = Rational(1, 2)
    r2 = Rational(1, 3)
    
    result = multiply_rationals(r1, r2)
    print(f"Product of {r1} and {r2} is {result}")
```
