Create a function that implements a dynamic programming algorithm to solve a problem. The function should take a problem as an argument and return the solution.

def knapsack(weights, values, capacity):
    """
    This function solves the 0/1 Knapsack problem using dynamic programming.
    
    Parameters:
    weights (list): A list of weights of the items.
    values (list): A list of values of the items.
    capacity (int): The maximum capacity of the knapsack.
    
    Returns:
    int: The maximum value that can be obtained with the given capacity.
    """

    # Check if weights and values are None
    if weights is None or values is None:
        return None

    # Check if the lengths of weights and values lists are equal
    if len(weights) != len(values):
        raise ValueError("The lengths of weights and values lists must be equal")

    # Initialize a 2D table to store the maximum value that can be obtained with a given capacity
    # The table has n+1 rows and w+1 columns, where n is the number of items and w is the capacity
    n = len(values)
    table = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]

    # Iterate over each item
    for i in range(1, n + 1):
        # Iterate over each possible capacity from 1 to the maximum capacity
        for w in range(1, capacity + 1):
            # If the weight of the current item is greater than the current capacity, 
            # then we cannot include this item in the knapsack
            if weights[i - 1] > w:
                # So, the maximum value that can be obtained with the current capacity is the same as 
                # the maximum value that can be obtained without the current item
                table[i][w] = table[i - 1][w]
            else:
                # Otherwise, we have two options: either include the current item in the knapsack or not
                # We choose the option that gives us the maximum value
                table[i][w] = max(table[i - 1][w], table[i - 1][w - weights[i - 1]] + values[i - 1])

    # The maximum value that can be obtained with the given capacity is stored in the last cell of the table
    return table[n][capacity]

# Example usage:
weights = [1, 2, 4, 2, 5]
values = [5, 3, 5, 3, 2]
capacity = 10
print(knapsack(weights, values, capacity))  # Output: 14
