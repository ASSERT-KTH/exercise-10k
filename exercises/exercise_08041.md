## Exercise
Write a program that uses an if/else statement to simulate the Game of Life based on the initial cell configuration.

## Reference Solution
```python
import numpy as np
import time
import os

def game_of_life(grid):
    """
    Simulates the Game of Life based on the initial cell configuration.

    Args:
        grid (numpy array): A 2D numpy array representing the initial cell configuration.
                             0 represents a dead cell and 1 represents a live cell.

    Returns:
        numpy array: A 2D numpy array representing the next generation of cells.
    """
    rows, cols = grid.shape
    next_grid = np.copy(grid)

    for i in range(rows):
        for j in range(cols):
            # Count the number of live neighbours
            live_neighbours = 0
            for x in range(max(0, i-1), min(rows, i+2)):
                for y in range(max(0, j-1), min(cols, j+2)):
                    if (x, y) != (i, j) and grid[x, y] == 1:
                        live_neighbours += 1

            # Apply the rules of the Game of Life
            if grid[i, j] == 1 and (live_neighbours < 2 or live_neighbours > 3):
                next_grid[i, j] = 0  # Underpopulation or overpopulation
            elif grid[i, j] == 0 and live_neighbours == 3:
                next_grid[i, j] = 1  # Reproduction
            # Else, the cell remains in its current state

    return next_grid

def print_grid(grid):
    """
    Prints the current state of the grid.

    Args:
        grid (numpy array): A 2D numpy array representing the current cell configuration.
    """
    os.system('cls' if os.name == 'nt' else 'clear')  # Clear the console
    for row in grid:
        print(' '.join(['*' if cell else ' ' for cell in row]))

def main():
    # Initialize the grid with a random configuration
    np.random.seed(0)
    grid = np.random.choice([0, 1], size=(20, 20), p=[0.5, 0.5])

    while True:
        print_grid(grid)
        grid = game_of_life(grid)
        time.sleep(0.5)  # Pause for 0.5 seconds between generations

if __name__ == '__main__':
    main()
```
