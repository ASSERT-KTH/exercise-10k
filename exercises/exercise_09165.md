Create a program with poor code readability (e.g. long lines, complex logic) and refactor it to improve readability, then print out the refactored code.

# Original code with poor readability
def calculate_total_cost(order):
    if order['discount'] == 'high': return order['price'] * order['quantity'] * 0.8
    elif order['discount'] == 'medium': return order['price'] * order['quantity'] * 0.9
    else: return order['price'] * order['quantity']

# Refactored code with improved readability
def calculate_total_cost_refactored(order):
    """
    Calculate the total cost of an order based on the discount level.
    
    Args:
        order (dict): A dictionary containing the order details.
            - 'price' (float): The price of the item.
            - 'quantity' (int): The quantity of the item.
            - 'discount' (str): The discount level. Can be 'high', 'medium', or 'low'.
    
    Returns:
        float: The total cost of the order.
    """
    discount_levels = {
        'high': 0.8,
        'medium': 0.9,
        'low': 1.0
    }
    discount_factor = discount_levels.get(order['discount'], 1.0)
    total_cost = order['price'] * order['quantity'] * discount_factor
    return total_cost

# Test the function
order = {
    'price': 100.0,
    'quantity': 2,
    'discount': 'high'
}
print(calculate_total_cost_refactored(order))
