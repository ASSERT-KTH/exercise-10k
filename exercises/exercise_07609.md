## Exercise
Write a program that uses membership operators to check if a variable is in a sequence.

## Reference Solution
```python
def check_membership(sequence, variable):
    if variable in sequence:
        print(f"'{variable}' is in the sequence.")
    else:
        print(f"'{variable}' is not in the sequence.")

def main():
    # Test with a list
    my_list = [1, 2, 3, 4, 5]
    variable = 3
    print("Checking membership in a list:")
    check_membership(my_list, variable)

    # Test with a tuple
    my_tuple = (1, 2, 3, 4, 5)
    print("\nChecking membership in a tuple:")
    check_membership(my_tuple, variable)

    # Test with a string
    my_string = "hello"
    variable = 'l'
    print("\nChecking membership in a string:")
    check_membership(my_string, variable)

    # Test with a variable not in the sequence
    variable = 6
    print("\nChecking membership with a variable not in the sequence:")
    check_membership(my_list, variable)

if __name__ == "__main__":
    main()
```
