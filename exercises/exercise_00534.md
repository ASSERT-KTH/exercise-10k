## Exercise
Implement a function to answer range queries on an array of integers.

## Reference Solution
```python
class SegmentTree:
    def __init__(self, nums):
        """
        Initialize the Segment Tree.

        Args:
        nums (list): The input array of integers.
        """
        self.n = len(nums)
        self.tree = [0] * (4 * self.n)
        self.build_tree(nums, 0, 0, self.n - 1)

    def build_tree(self, nums, node, start, end):
        """
        Build the Segment Tree recursively.

        Args:
        nums (list): The input array of integers.
        node (int): The current node index.
        start (int): The start index of the current range.
        end (int): The end index of the current range.
        """
        if start == end:
            self.tree[node] = nums[start]
        else:
            mid = (start + end) // 2
            self.build_tree(nums, 2 * node + 1, start, mid)
            self.build_tree(nums, 2 * node + 2, mid + 1, end)
            self.tree[node] = self.tree[2 * node + 1] + self.tree[2 * node + 2]

    def query(self, node, start, end, left, right):
        """
        Answer a range query.

        Args:
        node (int): The current node index.
        start (int): The start index of the current range.
        end (int): The end index of the current range.
        left (int): The start index of the query range.
        right (int): The end index of the query range.

        Returns:
        int: The sum of the elements in the query range.
        """
        if right < start or end < left:
            return 0
        if left <= start and end <= right:
            return self.tree[node]
        mid = (start + end) // 2
        p1 = self.query(2 * node + 1, start, mid, left, right)
        p2 = self.query(2 * node + 2, mid + 1, end, left, right)
        return p1 + p2

    def range_query(self, left, right):
        """
        Answer a range query.

        Args:
        left (int): The start index of the query range.
        right (int): The end index of the query range.

        Returns:
        int: The sum of the elements in the query range.
        """
        return self.query(0, 0, self.n - 1, left, right)


# Example usage:
if __name__ == "__main__":
    nums = [1, 3, 5, 7, 9, 11]
    segment_tree = SegmentTree(nums)
    print(segment_tree.range_query(1, 4))  # Output: 3 + 5 + 7 + 9 = 24
```
