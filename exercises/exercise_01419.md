## Exercise
Write a program that converts a number from decimal to binary or vice versa. If the user inputs 'dec', convert to binary, if 'bin', convert to decimal.

## Reference Solution
```python
def dec_to_bin(decimal):
    """Convert decimal to binary."""
    return bin(int(decimal))[2:]


def bin_to_dec(binary):
    """Convert binary to decimal."""
    return str(int(binary, 2))


def main():
    print("Decimal-Binary Converter Program")
    print("--------------------------------")

    while True:
        conversion_type = input("Enter 'dec' to convert decimal to binary or 'bin' to convert binary to decimal: ")
        
        if conversion_type.lower() == 'dec':
            decimal = input("Enter a decimal number: ")
            try:
                binary = dec_to_bin(decimal)
                print(f"The binary representation of {decimal} is: {binary}")
            except ValueError:
                print("Invalid decimal number. Please try again.")
                
        elif conversion_type.lower() == 'bin':
            binary = input("Enter a binary number: ")
            try:
                decimal = bin_to_dec(binary)
                print(f"The decimal representation of {binary} is: {decimal}")
            except ValueError:
                print("Invalid binary number. Please try again.")
                
        else:
            print("Invalid conversion type. Please enter 'dec' or 'bin'.")

        cont = input("Do you want to continue? (y/n): ")
        if cont.lower() != 'y':
            break


if __name__ == "__main__":
    main()
```
