## Exercise
Write a function that uses async functions to perform concurrent tasks.

## Reference Solution
```python
import asyncio
import time

# Define an async function for task1
async def task1():
    """Simulate task1 that takes 2 seconds to complete"""
    print("Task 1 started")
    await asyncio.sleep(2)  # Simulate task1 taking 2 seconds
    print("Task 1 completed")
    return "Task 1 result"

# Define an async function for task2
async def task2():
    """Simulate task2 that takes 3 seconds to complete"""
    print("Task 2 started")
    await asyncio.sleep(3)  # Simulate task2 taking 3 seconds
    print("Task 2 completed")
    return "Task 2 result"

# Define the main async function
async def main():
    """Run task1 and task2 concurrently"""
    start_time = time.time()
    # Run task1 and task2 concurrently using asyncio.gather
    results = await asyncio.gather(task1(), task2())
    print(f"Results: {results}")
    print(f"Total execution time: {time.time() - start_time} seconds")

# Run the main async function
if __name__ == "__main__":
    asyncio.run(main())
```
