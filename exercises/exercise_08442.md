## Exercise
Write a program that takes in a price and calculates the corresponding discount using conditional statements.

## Reference Solution
```python
# Define a function to calculate the discount
def calculate_discount(price):
    # Initialize the discount rate
    discount_rate = 0

    # Check the price and apply the corresponding discount rate
    if price < 10:
        discount_rate = 0
    elif price >= 10 and price < 20:
        discount_rate = 0.05
    elif price >= 20 and price < 50:
        discount_rate = 0.10
    elif price >= 50 and price < 100:
        discount_rate = 0.15
    elif price >= 100:
        discount_rate = 0.20

    # Calculate the discount amount
    discount_amount = price * discount_rate

    # Calculate the price after discount
    price_after_discount = price - discount_amount

    # Print the results
    print(f"Original Price: ${price:.2f}")
    print(f"Discount Rate: {discount_rate*100}%")
    print(f"Discount Amount: ${discount_amount:.2f}")
    print(f"Price after Discount: ${price_after_discount:.2f}")

# Main function
def main():
    # Get the price from the user
    price = float(input("Enter the price: "))

    # Calculate the discount
    calculate_discount(price)

# Run the main function
if __name__ == "__main__":
    main()
```
