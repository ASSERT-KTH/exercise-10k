## Exercise
Write a program to implement bubble sort on a given list.

```python
def bubble_sort(lst):
    """
    Sorts a list in ascending order using the bubble sort algorithm.

    Args:
    lst (list): The list to be sorted.

    Returns:
    list: The sorted list.
    """
    n = len(lst)
    
    for i in range(n):
        # Create a flag that will allow the function to terminate early if there's nothing left to sort
        swapped = False
        
        # Start looking at each item of the list one by one, comparing it with its adjacent value
        for j in range(n - i - 1):
            # If we find an element that is greater than the next element, then swap them
            if lst[j] > lst[j + 1]:
                # Swap the elements
                lst[j], lst[j + 1] = lst[j + 1], lst[j]
                # Set the flag to True so we'll loop again
                swapped = True
        
        # If there were no swaps during the last iteration, the list is already sorted, and we can terminate
        if not swapped:
            break
    
    return lst


# Example usage:
if __name__ == "__main__":
    lst = [64, 34, 25, 12, 22, 11, 90]
    print("Original list:", lst)
    sorted_lst = bubble_sort(lst)
    print("Sorted list:", sorted_lst)
```
