## Exercise
Create a program to manage student information using dictionaries. The program should allow users to add, remove, and update student records. Each student record should include the student's name, age, and grade level.

## Reference Solution
```python
# Student Information Management System

# Initialize an empty dictionary to store student records
student_records = {}

def add_student():
    """
    Adds a new student record to the dictionary.
    """
    name = input("Enter student's name: ")
    age = int(input("Enter student's age: "))
    grade_level = input("Enter student's grade level: ")
    
    # Create a new student record as a dictionary
    student_record = {
        "name": name,
        "age": age,
        "grade_level": grade_level
    }
    
    # Add the student record to the dictionary
    student_records[name] = student_record
    
    print(f"Student '{name}' added successfully!")

def remove_student():
    """
    Removes a student record from the dictionary.
    """
    name = input("Enter student's name: ")
    
    # Check if the student record exists
    if name in student_records:
        del student_records[name]
        print(f"Student '{name}' removed successfully!")
    else:
        print(f"Student '{name}' not found!")

def update_student():
    """
    Updates a student record in the dictionary.
    """
    name = input("Enter student's name: ")
    
    # Check if the student record exists
    if name in student_records:
        print("Enter new information (press Enter to keep current value):")
        student_records[name]["name"] = input(f"Name ({student_records[name]['name']}): ") or student_records[name]["name"]
        student_records[name]["age"] = int(input(f"Age ({student_records[name]['age']}): ") or student_records[name]["age"])
        student_records[name]["grade_level"] = input(f"Grade Level ({student_records[name]['grade_level']}): ") or student_records[name]["grade_level"]
        
        print(f"Student '{name}' updated successfully!")
    else:
        print(f"Student '{name}' not found!")

def display_students():
    """
    Displays all student records in the dictionary.
    """
    if student_records:
        for name, student_record in student_records.items():
            print(f"Name: {student_record['name']}")
            print(f"Age: {student_record['age']}")
            print(f"Grade Level: {student_record['grade_level']}\n")
    else:
        print("No student records found!")

def main():
    while True:
        print("Student Information Management System")
        print("1. Add Student")
        print("2. Remove Student")
        print("3. Update Student")
        print("4. Display Students")
        print("5. Exit")
        
        choice = input("Enter your choice: ")
        
        if choice == "1":
            add_student()
        elif choice == "2":
            remove_student()
        elif choice == "3":
            update_student()
        elif choice == "4":
            display_students()
        elif choice == "5":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
