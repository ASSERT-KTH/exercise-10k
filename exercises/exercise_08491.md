## Exercise
Create a program that uses functions to perform basic arithmetic operations such as addition, subtraction, multiplication, and division. The program should prompt the user to select an operation and input two numbers, then it should call the corresponding function to perform the operation and display the result.

## Reference Solution
```python
# Define a function for addition
def add(num1, num2):
    """Return the sum of two numbers."""
    return num1 + num2

# Define a function for subtraction
def subtract(num1, num2):
    """Return the difference of two numbers."""
    return num1 - num2

# Define a function for multiplication
def multiply(num1, num2):
    """Return the product of two numbers."""
    return num1 * num2

# Define a function for division
def divide(num1, num2):
    """Return the quotient of two numbers. Raise a ZeroDivisionError if num2 is zero."""
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero.")
    return num1 / num2

def main():
    # Create a dictionary to map operation choices to their corresponding functions
    operations = {
        "1": {"name": "Addition", "func": add},
        "2": {"name": "Subtraction", "func": subtract},
        "3": {"name": "Multiplication", "func": multiply},
        "4": {"name": "Division", "func": divide},
    }

    # Prompt the user to select an operation
    print("Basic Arithmetic Operations")
    print("-------------------------------")
    for key, value in operations.items():
        print(f"{key}. {value['name']}")

    # Get the user's operation choice
    while True:
        choice = input("Enter the number of your chosen operation: ")
        if choice in operations:
            break
        print("Invalid choice. Please enter a number from the list.")

    # Get the two numbers from the user
    while True:
        try:
            num1 = float(input("Enter the first number: "))
            num2 = float(input("Enter the second number: "))
            break
        except ValueError:
            print("Invalid input. Please enter a number.")

    # Perform the chosen operation and display the result
    try:
        result = operations[choice]["func"](num1, num2)
        print(f"The result of {operations[choice]['name']} is: {num1} {get_operator(choice)} {num2} = {result}")
    except ZeroDivisionError as e:
        print(str(e))

def get_operator(choice):
    """Return the operator symbol for the given operation choice."""
    operators = {
        "1": "+",
        "2": "-",
        "3": "*",
        "4": "/",
    }
    return operators[choice]

if __name__ == "__main__":
    main()
```
