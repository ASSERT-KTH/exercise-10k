## Exercise
Write a program that converts kitchen recipes between different units, using conditional statements to determine the conversion rules

def convert_units(value, input_unit, output_unit):
    """
    Converts a value from one unit to another.

    Args:
        value (float): The value to be converted.
        input_unit (str): The unit of the input value.
        output_unit (str): The desired output unit.

    Returns:
        float: The converted value.
    """
    # Define conversion factors
    conversion_factors = {
        'teaspoon': {'tablespoon': 0.333333, 'cup': 0.0208333, 'ounce': 0.208333, 'milliliter': 4.92892},
        'tablespoon': {'teaspoon': 3, 'cup': 0.0625, 'ounce': 0.5, 'milliliter': 14.7868},
        'cup': {'teaspoon': 48, 'tablespoon': 16, 'ounce': 8, 'milliliter': 236.588},
        'ounce': {'teaspoon': 6, 'tablespoon': 2, 'cup': 0.125, 'milliliter': 29.5735},
        'milliliter': {'teaspoon': 0.202884, 'tablespoon': 0.067628, 'cup': 0.00422675, 'ounce': 0.033814}
    }

    # Check if input and output units are valid
    if input_unit not in conversion_factors or output_unit not in conversion_factors:
        raise ValueError("Invalid input or output unit")

    # Convert value to desired output unit
    if input_unit == output_unit:
        return value
    else:
        return value * conversion_factors[input_unit][output_unit]

def main():
    print("Recipe Unit Converter Program")
    print("--------------------------------")

    # Get user input
    value = float(input("Enter the value to be converted: "))
    input_unit = input("Enter the input unit (teaspoon, tablespoon, cup, ounce, milliliter): ")
    output_unit = input("Enter the desired output unit (teaspoon, tablespoon, cup, ounce, milliliter): ")

    # Convert value and print result
    try:
        converted_value = convert_units(value, input_unit, output_unit)
        print(f"{value} {input_unit}(s) is equal to {converted_value} {output_unit}(s)")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
