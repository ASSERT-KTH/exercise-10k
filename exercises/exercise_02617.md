## Exercise
Write a program that finds all keys in a dictionary that have a common value

## Reference Solution
```python
def find_common_value_keys(dictionary):
    """
    Finds all keys in a dictionary that have a common value.

    Args:
        dictionary (dict): The input dictionary.

    Returns:
        dict: A dictionary where the keys are the common values and the values are lists of keys that have that common value.
    """
    common_value_keys = {}
    for key, value in dictionary.items():
        if value in common_value_keys:
            common_value_keys[value].append(key)
        else:
            common_value_keys[value] = [key]

    # Filter out values that only have one key
    common_value_keys = {value: keys for value, keys in common_value_keys.items() if len(keys) > 1}

    return common_value_keys

def main():
    # Example usage
    dictionary = {
        'a': 1,
        'b': 2,
        'c': 1,
        'd': 3,
        'e': 2,
        'f': 1
    }

    common_value_keys = find_common_value_keys(dictionary)
    print("Common value keys:")
    for value, keys in common_value_keys.items():
        print(f"Value: {value}, Keys: {keys}")

if __name__ == "__main__":
    main()
```
