## Exercise
Write a function that checks if a string starts or ends with a given substring

## Reference Solution
```python
def check_substring(string, substring, position='both'):
    """
    Checks if a string starts or ends with a given substring.

    Args:
        string (str): The main string to check.
        substring (str): The substring to look for.
        position (str): The position to check, can be 'start', 'end', or 'both'. Defaults to 'both'.

    Returns:
        bool: True if the string starts or ends with the substring, False otherwise.
    """

    # Check the position argument
    if position not in ['start', 'end', 'both']:
        raise ValueError("Position must be 'start', 'end', or 'both'")

    # Check if the string starts with the substring
    if position in ['start', 'both'] and string.startswith(substring):
        return True

    # Check if the string ends with the substring
    if position in ['end', 'both'] and string.endswith(substring):
        return True

    # If none of the above conditions are met, return False
    return False


# Example usage
if __name__ == "__main__":
    print(check_substring("Hello World", "Hello"))  # True
    print(check_substring("Hello World", "World", position='end'))  # True
    print(check_substring("Hello World", "Universe", position='start'))  # False
```
