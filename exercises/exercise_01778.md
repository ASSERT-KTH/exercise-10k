## Exercise
Write a function to find the topological sort of a given directed acyclic graph (DAG).

from collections import defaultdict, deque

def topological_sort(graph):
    """
    Performs a topological sort on a directed acyclic graph (DAG).

    Args:
    graph (dict): A dictionary representing the DAG, where each key is a node
                  and its corresponding value is a list of its neighbors.

    Returns:
    list: A list of nodes in topological order.
    """
    # Initialize the in-degree of each node to 0
    in_degree = {node: 0 for node in graph}

    # Calculate the in-degree of each node
    for node in graph:
        for neighbor in graph[node]:
            in_degree[neighbor] += 1

    # Initialize a queue with nodes having in-degree 0
    queue = deque([node for node in in_degree if in_degree[node] == 0])

    # Initialize an empty list to store the topological sort
    topological_order = []

    # Perform the topological sort
    while queue:
        node = queue.popleft()
        topological_order.append(node)

        # Decrease the in-degree of each neighbor by 1
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # Check if the graph has a cycle (i.e., it's not a DAG)
    if len(topological_order) != len(graph):
        raise ValueError("The graph has a cycle and cannot be topologically sorted")

    return topological_order

    try:
        topological_order = topological_sort(graph)
