#include <stdio.h>

// Function to increment a static variable
void increment_static_variable() {
    // Declare a static variable
    static int static_var = 0;

    // Increment the static variable
    static_var++;

    // Print the value of the static variable
    printf("Static variable value: %d\n", static_var);
}

int main() {
    // Call the function to increment the static variable
    for (int i = 0; i < 5; i++) {
        increment_static_variable();
    }

    return 0;
}
