## Exercise
You are given a dictionary where the keys are names of cities and the values are dictionaries containing information about the city, such as population, area, and climate. Your task is to write a program that calculates and prints the average population and area of all the cities in the dictionary.

## Reference Solution
```python
def calculate_average_city_stats(city_dict):
    """
    Calculate the average population and area of all cities in the dictionary.

    Args:
        city_dict (dict): A dictionary where the keys are city names and the values are dictionaries containing city information.

    Returns:
        tuple: A tuple containing the average population and area of all cities.
    """
    total_population = 0
    total_area = 0
    city_count = 0

    # Iterate over each city in the dictionary
    for city, info in city_dict.items():
        # Check if the city has population and area information
        if 'population' in info and 'area' in info:
            total_population += info['population']
            total_area += info['area']
            city_count += 1

    # Check if there are any cities with population and area information
    if city_count == 0:
        return None, None

    # Calculate the average population and area
    average_population = total_population / city_count
    average_area = total_area / city_count

    return average_population, average_area


def print_average_city_stats(city_dict):
    """
    Calculate and print the average population and area of all cities in the dictionary.

    Args:
        city_dict (dict): A dictionary where the keys are city names and the values are dictionaries containing city information.
    """
    average_population, average_area = calculate_average_city_stats(city_dict)

    if average_population is None or average_area is None:
        print("No cities with population and area information found.")
    else:
        print(f"Average population: {average_population:.2f}")
        print(f"Average area: {average_area:.2f}")
```
