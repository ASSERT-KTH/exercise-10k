## Exercise
Write a program that takes a list of questions as input and asks the user to answer them.

## Reference Solution
```python
def ask_questions(questions):
    """
    Asks the user to answer a list of questions.

    Args:
        questions (list): A list of questions to ask the user.

    Returns:
        list: A list of answers from the user.
    """
    answers = []
    for i, question in enumerate(questions):
        answer = input(f"Question {i + 1}: {question}\nYour answer: ")
        answers.append(answer)
    return answers


def main():
    # List of questions
    questions = [
        "What is your name?",
        "How old are you?",
        "Where are you from?",
        # Add more questions as needed
    ]

    print("Please answer the following questions:")
    answers = ask_questions(questions)

    # Print out the answers
    print("\nYour answers:")
    for i, (question, answer) in enumerate(zip(questions, answers)):
        print(f"Question {i + 1}: {question}")
        print(f"Your answer: {answer}")
        print()


if __name__ == "__main__":
    main()
```
