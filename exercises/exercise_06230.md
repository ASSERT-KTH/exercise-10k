## Exercise
Write a function that checks if a given list is sorted in ascending order.

## Reference Solution
```python
def is_sorted_ascending(lst):
    """
    Checks if a given list is sorted in ascending order.

    Args:
        lst (list): The list to check.

    Returns:
        bool: True if the list is sorted in ascending order, False otherwise.
    """
    return lst == sorted(lst)

# Example usage:
print(is_sorted_ascending([1, 2, 3, 4, 5]))  # True
print(is_sorted_ascending([5, 4, 3, 2, 1]))  # False
print(is_sorted_ascending([1, 2, 3, 2, 1]))  # False
```

def is_sorted_ascending_efficient(lst):
    """
    Checks if a given list is sorted in ascending order.

    Args:
        lst (list): The list to check.

    Returns:
        bool: True if the list is sorted in ascending order, False otherwise.
    """
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))

# Example usage:
print(is_sorted_ascending_efficient([1, 2, 3, 4, 5]))  # True
print(is_sorted_ascending_efficient([5, 4, 3, 2, 1]))  # False
print(is_sorted_ascending_efficient([1, 2, 3, 2, 1]))  # False
