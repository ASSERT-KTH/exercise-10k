## Exercise
Write a program to reverse an array.

```python
def reverse_array(arr):
    """
    Reverses the input array.

    Args:
        arr (list): The input array to be reversed.

    Returns:
        list: The reversed array.
    """
    return arr[::-1]

# Example usage:
def main():
    # Test array
    arr = [1, 2, 3, 4, 5]
    print("Original array:", arr)

    # Reverse the array
```

```python
def reverse_array(arr):
    """
    Reverses the input array.

    Args:
        arr (list): The input array to be reversed.

    Returns:
        list: The reversed array.
    """
    left = 0
    right = len(arr) - 1

    while left < right:
        # Swap elements at left and right indices
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1

    return arr

# Example usage:
def main():
    # Test array
    arr = [1, 2, 3, 4, 5]
    print("Original array:", arr)

    # Reverse the array
    reversed_arr = reverse_array(arr)
    print("Reversed array:", reversed_arr)

if __name__ == "__main__":
    main()
```
