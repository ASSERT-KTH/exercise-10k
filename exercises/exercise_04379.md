## Exercise
Write a program that aggregates the values in multiple dictionaries.

## Reference Solution
```python
def aggregate_dictionaries(dict_list):
    """
    Aggregate the values in multiple dictionaries.

    Args:
        dict_list (list): A list of dictionaries to aggregate.

    Returns:
        dict: A dictionary with aggregated values.
    """
    if not dict_list:
        return {}

    # Get the keys from the first dictionary
    keys = dict_list[0].keys()

    # Initialize the aggregated dictionary
    aggregated_dict = {key: 0 for key in keys}

    # Aggregate the values
    for dictionary in dict_list:
        for key, value in dictionary.items():
            aggregated_dict[key] += value

    return aggregated_dict


def main():
    # Example usage
    dict1 = {"a": 1, "b": 2, "c": 3}
    dict2 = {"a": 4, "b": 5, "c": 6}
    dict3 = {"a": 7, "b": 8, "c": 9}

    dict_list = [dict1, dict2, dict3]

    aggregated_dict = aggregate_dictionaries(dict_list)

    print("Aggregated Dictionary:")
    for key, value in aggregated_dict.items():
        print(f"{key}: {value}")


if __name__ == "__main__":
    main()
```
