## Exercise
Write a function that takes a string as input and returns True if it matches a given pattern

## Reference Solution
```python
import re

def match_pattern(input_string, pattern):
    """
    Checks if the input string matches the given pattern.

    Args:
        input_string (str): The string to check.
        pattern (str): The pattern to match.

    Returns:
        bool: True if the string matches the pattern, False otherwise.
    """
    try:
        # Compile the pattern into a regular expression object
        regex = re.compile(pattern)
        
        # Check if the entire string matches the pattern
        if regex.fullmatch(input_string):
            return True
        else:
            return False
    except re.error as e:
        # Handle invalid patterns
        print(f"Invalid pattern: {e}")
        return False

# Example usage:
if __name__ == "__main__":
    # Test 1: Matching a simple string
    input_string = "hello"
    pattern = "hello"
    print(f"Input: {input_string}, Pattern: {pattern}, Match: {match_pattern(input_string, pattern)}")

    # Test 2: Matching a pattern with a wildcard
    input_string = "hello world"
    pattern = "hello .*"
    print(f"Input: {input_string}, Pattern: {pattern}, Match: {match_pattern(input_string, pattern)}")

    # Test 3: Non-matching string
    input_string = "goodbye"
    pattern = "hello"
    print(f"Input: {input_string}, Pattern: {pattern}, Match: {match_pattern(input_string, pattern)}")
```
