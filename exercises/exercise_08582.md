## Exercise
Write a function to clear all key-value pairs from a dictionary

## Reference Solution
```python
def clear_dictionary(input_dict):
    """
    Clears all key-value pairs from a dictionary.

    Args:
        input_dict (dict): The dictionary to be cleared.

    Returns:
        dict: The cleared dictionary.
    """
    # Check if input is a dictionary
    if not isinstance(input_dict, dict):
        raise TypeError("Input must be a dictionary")

    # Clear the dictionary
    input_dict.clear()

    return input_dict

# Example usage:
my_dict = {"key1": "value1", "key2": "value2"}
print("Before clearing:", my_dict)
cleared_dict = clear_dictionary(my_dict)
print("After clearing:", cleared_dict)
```
