## Exercise
Write a program that takes two strings as input and returns the edit distance between them.

def edit_distance(str1, str2):
    """
    Calculate the edit distance between two strings.

    Args:
        str1 (str): The first string.
        str2 (str): The second string.

    Returns:
        int: The edit distance between the two strings.
    """

    # Initialize a matrix to store the edit distances between substrings
    dp = [[0 for _ in range(len(str2) + 1)] for _ in range(len(str1) + 1)]

    # Initialize the base cases
    for i in range(len(str1) + 1):
        dp[i][0] = i
    for j in range(len(str2) + 1):
        dp[0][j] = j

    # Fill in the rest of the matrix
    for i in range(1, len(str1) + 1):
        for j in range(1, len(str2) + 1):
            # Check if the current characters match
            if str1[i - 1] == str2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                # Calculate the edit distance for the current cell
                dp[i][j] = 1 + min(
                    dp[i - 1][j],  # Deletion
                    dp[i][j - 1],  # Insertion
                    dp[i - 1][j - 1]  # Substitution
                )

    # The edit distance is stored in the bottom-right cell of the matrix
    return dp[len(str1)][len(str2)]


# Example usage:
if __name__ == "__main__":
    str1 = "kitten"
    str2 = "sitting"
    distance = edit_distance(str1, str2)
    print(f"The edit distance between '{str1}' and '{str2}' is {distance}.")
