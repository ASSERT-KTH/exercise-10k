## Exercise
Create a program that generates a sequence of numbers based on a given starting number and increment using conditional statements.

## Reference Solution
```python
def generate_sequence(start, increment, length):
    """
    Generates a sequence of numbers based on a given starting number and increment.
    
    Args:
    start (int): The starting number of the sequence.
    increment (int): The increment between each number in the sequence.
    length (int): The length of the sequence.
    
    Returns:
    list: A list of numbers representing the sequence.
    """
    
    # Check if the length is a positive integer
    if not isinstance(length, int) or length <= 0:
        raise ValueError("Length must be a positive integer.")
    
    # Initialize an empty list to store the sequence
    sequence = []
    
    # Initialize the current number to the start
    current = start
    
    # Loop until the sequence has reached the desired length
    for _ in range(length):
        # Append the current number to the sequence
        sequence.append(current)
        
        # Check if the increment is positive or negative
        if increment > 0:
            # If the increment is positive, add it to the current number
            current += increment
        elif increment < 0:
            # If the increment is negative, subtract it from the current number
            current += increment
        else:
            # If the increment is zero, keep the current number the same
            current = current
    
    # Return the generated sequence
    return sequence

# Example usage:
start = 1
increment = 2
length = 10

sequence = generate_sequence(start, increment, length)
print(f"Sequence starting from {start} with an increment of {increment} and a length of {length}: {sequence}")
```
