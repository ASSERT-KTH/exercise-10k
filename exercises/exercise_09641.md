## Exercise
Write a function that uses the csv module to read and write CSV files.

## Reference Solution
```python
import csv

def read_csv_file(file_path):
    """
    Reads a CSV file and returns its contents as a list of lists.

    Args:
        file_path (str): The path to the CSV file.

    Returns:
        list: A list of lists, where each sublist represents a row in the CSV file.
    """
    with open(file_path, 'r') as file:
        csv_reader = csv.reader(file)
        return list(csv_reader)

def write_csv_file(file_path, data):
    """
    Writes data to a CSV file.

    Args:
        file_path (str): The path to the CSV file.
        data (list): A list of lists, where each sublist represents a row in the CSV file.
    """
    with open(file_path, 'w', newline='') as file:
        csv_writer = csv.writer(file)
        csv_writer.writerows(data)

# Example usage:
if __name__ == "__main__":
    # Read a CSV file
    file_path = 'example.csv'
    data = read_csv_file(file_path)
    print("Read Data:")
    for row in data:
        print(row)

    # Write to a CSV file
    new_file_path = 'new_example.csv'
    new_data = [
        ['Name', 'Age'],
        ['John', '25'],
        ['Alice', '30']
    ]
    write_csv_file(new_file_path, new_data)
    print("\nWrote to", new_file_path)
```
