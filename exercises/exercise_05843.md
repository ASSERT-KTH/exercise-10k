## Exercise
Write a program that plays Rock, Paper, Scissors with the user. The program should randomly choose one of rock, paper, or scissors, and then ask the user for their choice. Determine the winner based on the game's rules

## Reference Solution
```python
import random

def get_computer_choice():
    """Randomly selects the computer's choice"""
    choices = ['rock', 'paper', 'scissors']
    return random.choice(choices)

def determine_winner(computer_choice, user_choice):
    """Determines the winner based on the game's rules"""
    if computer_choice == user_choice:
        return "It's a tie!"
    if (computer_choice == 'rock' and user_choice == 'scissors') or \
       (computer_choice == 'scissors' and user_choice == 'paper') or \
       (computer_choice == 'paper' and user_choice == 'rock'):
        return "Computer wins!"
    return "User wins!"

def main():
    print("Welcome to Rock, Paper, Scissors!")
    while True:
        computer_choice = get_computer_choice()
        user_choice = input("Enter your choice (rock, paper, or scissors), or 'q' to quit: ").lower()
        if user_choice == 'q':
            break
        if user_choice not in ['rock', 'paper', 'scissors']:
            print("Invalid choice. Please try again.")
            continue
        print(f"\nComputer chose: {computer_choice}")
        print(f"User chose: {user_choice}\n")
        print(determine_winner(computer_choice, user_choice))

if __name__ == "__main__":
    main()
```
