## Exercise
Write a program that reads a string from the user, reverses the order of its characters, and then prints the reversed string. Additionally, the program should count the number of vowels in the original string and print this count.

## Reference Solution
```python
#include <iostream>
#include <string>

using namespace std;

int main() {
    string str;
    int count = 0;

    cout << "Enter a string: ";
    getline(cin, str);

    for (int i = 0; i < str.length(); i++) {
        if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u' ||
            str[i] == 'A' || str[i] == 'E' || str[i] == 'I' || str[i] == 'O' || str[i] == 'U') {
            count++;
        }
    }

    cout << "Reversed string: ";
    for (int i = str.length() - 1; i >= 0; i--) {
        cout << str[i];
    }
    cout << endl;

    cout << "Number of vowels: " << count << endl;

    return 0;
}
```
