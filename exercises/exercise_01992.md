## Exercise
Develop a program to calculate a person's age based on their birth year using conditional statements

## Reference Solution
```python
def calculate_age(current_year, birth_year):
    """
    Calculate a person's age based on their birth year.

    Args:
        current_year (int): The current year.
        birth_year (int): The person's birth year.

    Returns:
        int: The person's age.
    """
    if current_year < birth_year:
        raise ValueError("Birth year cannot be in the future")
    elif current_year == birth_year:
        return 0
    else:
        age = current_year - birth_year
        return age

def main():
    # Get the current year
    from datetime import datetime
    current_year = datetime.now().year

    # Get the person's birth year
    birth_year = int(input("Enter your birth year: "))

    try:
        # Calculate the person's age
        age = calculate_age(current_year, birth_year)
        print(f"Your age is: {age}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
