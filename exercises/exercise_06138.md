## Exercise
Ask the user for two words and check if they are anagrams of each other.

## Reference Solution
```python
def are_anagrams(word1, word2):
    """
    Checks if two words are anagrams of each other.

    Args:
        word1 (str): The first word.
        word2 (str): The second word.

    Returns:
        bool: True if the words are anagrams, False otherwise.
    """
    # Remove any whitespace and convert to lowercase
    word1 = word1.replace(" ", "").lower()
    word2 = word2.replace(" ", "").lower()

    # Check if the sorted words are equal
    return sorted(word1) == sorted(word2)

def main():
    # Ask the user for two words
    word1 = input("Enter the first word: ")
    word2 = input("Enter the second word: ")

    # Check if the words are anagrams
    if are_anagrams(word1, word2):
        print(f"'{word1}' and '{word2}' are anagrams.")
    else:
        print(f"'{word1}' and '{word2}' are not anagrams.")

if __name__ == "__main__":
    main()
```
