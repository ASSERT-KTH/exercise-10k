## Exercise
Write a function that takes a dictionary and a list of key-value pairs, then updates the dictionary with the new pairs

## Reference Solution
```python
def update_dictionary(dictionary, new_pairs):
    """
    Updates a dictionary with new key-value pairs.

    Args:
    dictionary (dict): The dictionary to be updated.
    new_pairs (list): A list of tuples containing the new key-value pairs.

    Returns:
    dict: The updated dictionary.
    """
    # Iterate over each pair in the list
    for pair in new_pairs:
        # Check if the pair is a tuple with two elements (key and value)
        if isinstance(pair, tuple) and len(pair) == 2:
            # Update the dictionary with the new key-value pair
            dictionary[pair[0]] = pair[1]
        else:
            # Raise an error if the pair is not a tuple with two elements
            raise ValueError("Invalid key-value pair: {}".format(pair))

    # Return the updated dictionary
    return dictionary


# Example usage
if __name__ == "__main__":
    # Create a dictionary
    my_dict = {"a": 1, "b": 2}

    # Create a list of new key-value pairs
    new_pairs = [("c", 3), ("d", 4), ("a", 5)]  # Update existing key "a" with new value 5

    # Update the dictionary
    updated_dict = update_dictionary(my_dict, new_pairs)

    # Print the updated dictionary
    print(updated_dict)  # Output: {'a': 5, 'b': 2, 'c': 3, 'd': 4}
```

```python
def update_dictionary(dictionary, new_pairs):
    """
    Updates a dictionary with new key-value pairs.

    Args:
    dictionary (dict): The dictionary to be updated.
    new_pairs (list): A list of tuples containing the new key-value pairs.

    Returns:
    dict: The updated dictionary.
    """
    # Convert the list of tuples to a dictionary
    new_dict = dict(new_pairs)

    # Update the dictionary
    dictionary.update(new_dict)

    # Return the updated dictionary
    return dictionary
```
