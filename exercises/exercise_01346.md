## Exercise
Write a program to demonstrate the use of boolean variables in conditional statements

## Reference Solution
```python
# Define a function to demonstrate boolean variables in conditional statements
def boolean_variables_demo():
    # Initialize boolean variables
    is_admin = True
    is_logged_in = False
    is_user_active = True

    # Conditional statement using boolean variables
    if is_admin and is_logged_in:
        print("Welcome, Admin! You are logged in and active.")
    elif is_admin and not is_logged_in:
        print("Welcome, Admin! However, you need to log in to access your account.")
    elif not is_admin and is_logged_in:
        if is_user_active:
            print("Welcome, User! You are logged in and active.")
        else:
            print("Welcome, User! However, your account is inactive.")
    else:
        print("You need to log in to access your account.")

    # Conditional statement using boolean variables in a loop
    users = [
        {"name": "John", "is_admin": True, "is_logged_in": True, "is_active": True},
        {"name": "Jane", "is_admin": False, "is_logged_in": True, "is_active": True},
        {"name": "Bob", "is_admin": True, "is_logged_in": False, "is_active": True},
    ]

    for user in users:
        if user["is_admin"] and user["is_logged_in"]:
            print(f"Welcome, Admin {user['name']}! You are logged in and active.")
        elif user["is_admin"] and not user["is_logged_in"]:
            print(f"Welcome, Admin {user['name']}! However, you need to log in to access your account.")
        elif not user["is_admin"] and user["is_logged_in"]:
            if user["is_active"]:
                print(f"Welcome, User {user['name']}! You are logged in and active.")
            else:
                print(f"Welcome, User {user['name']}! However, your account is inactive.")
        else:
            print(f"You need to log in to access your account, {user['name']}.")

# Call the function to demonstrate boolean variables in conditional statements
boolean_variables_demo()
```
