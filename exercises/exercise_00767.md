## Exercise
Write a program that checks the data type of a variable and prints out a message based on the type.

## Reference Solution
```python
def check_data_type(var):
    """
    Checks the data type of a variable and prints out a message based on the type.
    
    Parameters:
    var (any): The variable to be checked.
    """
    
    # Use isinstance() function to check the type of the variable
    if isinstance(var, int):
        print(f"{var} is an integer.")
    elif isinstance(var, float):
        print(f"{var} is a float.")
    elif isinstance(var, str):
        print(f"{var} is a string.")
    elif isinstance(var, list):
        print(f"{var} is a list.")
    elif isinstance(var, tuple):
        print(f"{var} is a tuple.")
    elif isinstance(var, dict):
        print(f"{var} is a dictionary.")
    elif isinstance(var, set):
        print(f"{var} is a set.")
    else:
        print(f"{var} is of type {type(var).__name__}.")

def main():
    # Test the function with different data types
    check_data_type(10)         # Integer
    check_data_type(10.5)       # Float
    check_data_type("Hello")    # String
    check_data_type([1, 2, 3])  # List
    check_data_type((1, 2, 3))  # Tuple
    check_data_type({"a": 1, "b": 2})  # Dictionary
    check_data_type({1, 2, 3})  # Set
    check_data_type(True)       # Boolean

if __name__ == "__main__":
    main()
```
