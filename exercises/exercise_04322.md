## Exercise
Write a program to find the last element of an array.

## Reference Solution
```python
def get_last_element(array):
    """
    Returns the last element of the given array.

    Args:
        array (list): The input array.

    Returns:
        any: The last element of the array.

    Raises:
        IndexError: If the array is empty.
    """
    if not array:
        raise IndexError("Array is empty")
    return array[-1]

# Example usage
def main():
    array = [1, 2, 3, 4, 5]
    print("Last element:", get_last_element(array))

    # Test with empty array
    try:
        print(get_last_element([]))
    except IndexError as e:
        print(e)

if __name__ == "__main__":
    main()
```
