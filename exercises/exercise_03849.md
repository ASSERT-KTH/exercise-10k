## Exercise
Create a program that uses a for loop to implement a Caesar cipher, shifting each letter in a given string by a specified number of positions.

## Reference Solution
```python
def caesar_cipher(text, shift):
    """
    Shifts each letter in the given text by the specified number of positions.

    Args:
        text (str): The input text to be encrypted.
        shift (int): The number of positions to shift each letter.

    Returns:
        str: The encrypted text.
    """

    result = ""

    # Iterate over each character in the input text
    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Calculate the ASCII value of the shifted character
            # The ASCII value of 'A' is 65, so we subtract 65 to get the position of the letter in the alphabet
            # We add the shift value, take the modulus with 26 to wrap around the alphabet, and then add 65 back
            shifted_char = chr((ord(char) - 65 + shift) % 26 + 65)
            result += shifted_char
        # Check if the character is a lowercase letter
        elif char.islower():
            # Calculate the ASCII value of the shifted character
            # The ASCII value of 'a' is 97, so we subtract 97 to get the position of the letter in the alphabet
            # We add the shift value, take the modulus with 26 to wrap around the alphabet, and then add 97 back
            shifted_char = chr((ord(char) - 97 + shift) % 26 + 97)
            result += shifted_char
        # If the character is not a letter, add it to the result as is
        else:
            result += char

    return result


# Test the function
text = "Hello, World!"
shift = 3
print(f"Original text: {text}")
print(f"Shift: {shift}")
print(f"Encrypted text: {caesar_cipher(text, shift)}")
```
