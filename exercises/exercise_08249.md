## Exercise
Create a program that guesses a number between 1 and 100 using if/else statements.

## Reference Solution
```python
import random

def guess_number():
    # Initialize the range of possible numbers
    low = 1
    high = 100

    # Initialize the number of attempts
    attempts = 0

    # Loop until the number is guessed
    while True:
        # Calculate the middle of the range
        guess = (low + high) // 2

        # Ask the user if the guess is correct
        print(f"Is your number {guess}?")
        response = input("(h)igher, (l)ower, or (c)orrect: ")

        # Increment the number of attempts
        attempts += 1

        # Check the user's response
        if response == "h":
            # If the guess is too low, update the range
            low = guess + 1
        elif response == "l":
            # If the guess is too high, update the range
            high = guess - 1
        elif response == "c":
            # If the guess is correct, print the result and exit
            print(f"Yay! I guessed the number in {attempts} attempts.")
            break
        else:
            # If the user enters an invalid response, ask again
            print("Invalid response. Please enter (h)igher, (l)ower, or (c)orrect.")

if __name__ == "__main__":
    guess_number()
```
