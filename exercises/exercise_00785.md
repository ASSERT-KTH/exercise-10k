## Exercise
Write a program to generate an arithmetic sequence given the first term, common difference, and number of terms.

## Reference Solution
```python
def generate_arithmetic_sequence(first_term, common_difference, num_terms):
    """
    Generate an arithmetic sequence given the first term, common difference, and number of terms.

    Args:
        first_term (int): The first term of the sequence.
        common_difference (int): The common difference between each term.
        num_terms (int): The number of terms in the sequence.

    Returns:
        list: A list of integers representing the arithmetic sequence.
    """
    sequence = [first_term + i * common_difference for i in range(num_terms)]
    return sequence


def main():
    first_term = int(input("Enter the first term: "))
    common_difference = int(input("Enter the common difference: "))
    num_terms = int(input("Enter the number of terms: "))

    sequence = generate_arithmetic_sequence(first_term, common_difference, num_terms)
    print("Arithmetic Sequence:")
    print(sequence)


if __name__ == "__main__":
    main()
```
