## Exercise
Write a program that demonstrates explicit type conversion.

## Reference Solution
```python
def main():
    # Implicit conversion
    print("# Implicit Conversion #")
    a = 5
    b = 3.5
    print("a =", a)
    print("b =", b)
    print("a + b =", a + b)

    # Explicit conversion
    print("\n# Explicit Conversion #")
    c = 10
    d = 2.5
    print("c =", c)
    print("d =", d)

    # Convert float to int
    e = int(d)
    print("d as int:", e)

    # Convert int to float
    f = float(c)
    print("c as float:", f)

    # Convert int to string
    g = str(c)
    print("c as string:", g)

    # Convert string to int
    h = "20"
    i = int(h)
    print("h as int:", i)

    # Convert string to float
    j = "30.5"
    k = float(j)
    print("j as float:", k)

if __name__ == "__main__":
    main()
```
