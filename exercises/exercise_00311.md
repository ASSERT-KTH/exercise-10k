## Exercise
Write a function that deletes an element at a specified position in a given array.

## Reference Solution
```python
def delete_element_at_position(array, position):
    """
    Deletes an element at a specified position in a given array.

    Args:
        array (list): The input array.
        position (int): The position of the element to be deleted.

    Returns:
        list: The modified array with the element deleted.

    Raises:
        IndexError: If the position is out of range.
    """
    if position < 0 or position >= len(array):
        raise IndexError("Position out of range")

    # Use list slicing to delete the element at the specified position
    return array[:position] + array[position + 1:]

# Example usage:
def main():
    array = [1, 2, 3, 4, 5]
    position = 2
    print("Original array:", array)
    print("Position to delete:", position)
    modified_array = delete_element_at_position(array, position)
    print("Modified array:", modified_array)

if __name__ == "__main__":
    main()
```
