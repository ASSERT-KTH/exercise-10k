## Exercise
Write a program to find all duplicate elements in an array.

## Reference Solution
```python
def find_duplicates(array):
    """
    Finds all duplicate elements in the given array.

    Args:
        array (list): The input array.

    Returns:
        dict: A dictionary where the keys are the duplicate elements and the values are their frequencies.
    """
    frequency_dict = {}
    duplicates = {}

    # Count the frequency of each element
    for element in array:
        if element in frequency_dict:
            frequency_dict[element] += 1
        else:
            frequency_dict[element] = 1

    # Find the duplicate elements
    for element, frequency in frequency_dict.items():
        if frequency > 1:
            duplicates[element] = frequency

    return duplicates


def main():
    # Example usage
    array = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
    duplicates = find_duplicates(array)

    print("Duplicate elements:")
    for element, frequency in duplicates.items():
        print(f"{element}: {frequency} times")


if __name__ == "__main__":
    main()
```
