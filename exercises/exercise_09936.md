def combine_dictionaries(dict1, dict2):
    """
    This function combines two dictionaries into one. If there are any duplicate keys, 
    the values from the second dictionary override the values from the first dictionary.

    Args:

    Returns:

    # First, we'll create a new dictionary to store the combined key-value pairs.
    combined_dict = {}

    # We'll start by copying all the key-value pairs from the first dictionary into our new dictionary.
    # This is because we want to include all the key-value pairs from both dictionaries in our result.
    for key, value in dict1.items():
        # For each key-value pair in the first dictionary, we'll add it to our new dictionary.
        combined_dict[key] = value

    # Now, we'll copy all the key-value pairs from the second dictionary into our new dictionary.
    # If there are any duplicate keys, the values from the second dictionary will override the values from the first dictionary.
    for key, value in dict2.items():
        # For each key-value pair in the second dictionary, we'll add it to our new dictionary.
        # If the key already exists in our new dictionary, its value will be updated to the value from the second dictionary.
        combined_dict[key] = value

    # Finally, we'll return our new dictionary that combines the key-value pairs from both input dictionaries.
    return combined_dict
