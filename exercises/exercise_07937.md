## Exercise
Write a program to calculate the acceleration given the initial and final velocities and time.

## Reference Solution
```python
def calculate_acceleration(initial_velocity, final_velocity, time):
    """
    Calculate the acceleration given the initial and final velocities and time.

    Args:
    - initial_velocity (float): The initial velocity of the object (m/s)
    - final_velocity (float): The final velocity of the object (m/s)
    - time (float): The time over which the acceleration occurs (s)

    Returns:
    - float: The acceleration of the object (m/s^2)
    """
    if time == 0:
        raise ZeroDivisionError("Time cannot be zero")

    acceleration = (final_velocity - initial_velocity) / time
    return acceleration

def main():
    # Example usage
    initial_velocity = float(input("Enter the initial velocity (m/s): "))
    final_velocity = float(input("Enter the final velocity (m/s): "))
    time = float(input("Enter the time (s): "))

    try:
        acceleration = calculate_acceleration(initial_velocity, final_velocity, time)
        print(f"The acceleration is {acceleration:.2f} m/s^2")
    except ZeroDivisionError as e:
        print(e)

if __name__ == "__main__":
    main()
```
