## Exercise
Write a program that demonstrates implicit type conversion.

```python
def implicit_type_conversion():
    # Implicit conversion of integer to float
    a = 5
    b = 2.0
    print("Implicit conversion of integer to float: ")
    print(f"a = {a}, type(a) = {type(a)}")
    print(f"b = {b}, type(b) = {type(b)}")
    result = a + b
    print(f"a + b = {result}, type(a + b) = {type(result)}")

    # Implicit conversion of integer to float in operations
    print("\nImplicit conversion of integer to float in operations: ")
    a = 5
    b = 2.0
    print(f"a = {a}, type(a) = {type(a)}")
    print(f"b = {b}, type(b) = {type(b)}")
    result = a * b
    print(f"a * b = {result}, type(a * b) = {type(result)}")

    # Implicit conversion of boolean to integer
    print("\nImplicit conversion of boolean to integer: ")
    a = True
    b = 2
    print(f"a = {a}, type(a) = {type(a)}")
    print(f"b = {b}, type(b) = {type(b)}")
    result = a + b
    print(f"a + b = {result}, type(a + b) = {type(result)}")


if __name__ == "__main__":
    implicit_type_conversion()
```
