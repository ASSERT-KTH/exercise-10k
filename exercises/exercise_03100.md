## Exercise
Implement a function to generate a Caesar cipher for a given string with a specified shift.

## Reference Solution
```python
def caesar_cipher(text: str, shift: int) -> str:
    """
    Generates a Caesar cipher for a given string with a specified shift.

    Args:
    - text (str): The input string to be encrypted.
    - shift (int): The number of positions to shift the alphabet.

    Returns:
    - str: The encrypted string.
    """
    result = ""

    for char in text:
        # Check if the character is an uppercase letter
        if char.isupper():
            # Find the position in the alphabet for the character and add the shift
            # Use the modulo operator to ensure the result is within the alphabet
            result += chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
        # Check if the character is a lowercase letter
        elif char.islower():
            # Find the position in the alphabet for the character and add the shift
            # Use the modulo operator to ensure the result is within the alphabet
            result += chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
        # If the character is not a letter, leave it unchanged
        else:
            result += char

    return result

def main():
    # Example usage:
    text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_cipher(text, shift)
    decrypted_text = caesar_cipher(encrypted_text, -shift)

    print(f"Original Text: {text}")
    print(f"Encrypted Text: {encrypted_text}")
    print(f"Decrypted Text: {decrypted_text}")

if __name__ == "__main__":
    main()
```
