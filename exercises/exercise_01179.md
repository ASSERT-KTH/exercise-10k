## Exercise
Write a function to check the strength of a password

## Reference Solution
```python
import re

def check_password_strength(password: str) -> dict:
    """
    Checks the strength of a password.

    Args:
        password (str): The password to check.

    Returns:
        dict: A dictionary containing the results of the password strength check.
    """

    # Initialize the results dictionary
    results = {
        "length": False,
        "uppercase": False,
        "lowercase": False,
        "digits": False,
        "special_chars": False,
        "strong": False
    }

    # Check the length of the password
    if len(password) >= 8:
        results["length"] = True

    # Check for uppercase letters
    if re.search(r"[A-Z]", password):
        results["uppercase"] = True

    # Check for lowercase letters
    if re.search(r"[a-z]", password):
        results["lowercase"] = True

    # Check for digits
    if re.search(r"\d", password):
        results["digits"] = True

    # Check for special characters
    if re.search(r"[^A-Za-z0-9]", password):
        results["special_chars"] = True

    # Check if the password is strong
    if (results["length"] and results["uppercase"] and results["lowercase"] and
            results["digits"] and results["special_chars"]):
        results["strong"] = True

    return results


def print_password_strength_results(results: dict) -> None:
    """
    Prints the results of the password strength check.

    Args:
        results (dict): The results dictionary.
    """

    print("Password Strength Check Results:")
    print(f"Length: {'OK' if results['length'] else 'Too short'}")
    print(f"Uppercase letters: {'OK' if results['uppercase'] else 'Missing'}")
    print(f"Lowercase letters: {'OK' if results['lowercase'] else 'Missing'}")
    print(f"Digits: {'OK' if results['digits'] else 'Missing'}")
    print(f"Special characters: {'OK' if results['special_chars'] else 'Missing'}")
    print(f"Password strength: {'Strong' if results['strong'] else 'Weak'}")


# Example usage:
if __name__ == "__main__":
    password = input("Enter a password: ")
    results = check_password_strength(password)
    print_password_strength_results(results)
```
