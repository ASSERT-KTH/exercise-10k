## Exercise
Develop a program to create a simple chatbot using conditional statements to respond to user input

## Reference Solution
```python
# Import the required library
import random

# Define a function to handle greetings
def handle_greeting(message):
    greetings = ["Hello", "Hi", "Hey", "Hi there"]
    if any(greeting in message for greeting in greetings):
        return random.choice(["Hello!", "Hi!", "Hey!", "Hi there!"])
    else:
        return None

# Define a function to handle goodbyes
def handle_goodbye(message):
    goodbyes = ["Bye", "Goodbye", "See you later", "See ya"]
    if any(goodbye in message for goodbye in goodbyes):
        return random.choice(["Goodbye!", "See you later!", "Bye!", "See ya!"])
    else:
        return None

# Define a function to handle basic questions
def handle_questions(message):
    questions = {
        "what is your name": "My name is ChatBot",
        "how are you": "I'm doing well, thanks!",
        "what can you do": "I can have a conversation with you",
    }
    for question, answer in questions.items():
        if question in message.lower():
            return answer
    return None

# Define a function to handle unknown messages
def handle_unknown(message):
    return random.choice(["I didn't understand that.", "Can you please rephrase?", "I'm not sure what you mean."])

# Define the main function
def main():
    print("Welcome to the chatbot! Type 'quit' to exit.")
    while True:
        message = input("You: ")
        if message.lower() == "quit":
            break
        greeting_response = handle_greeting(message)
        if greeting_response:
            print("ChatBot:", greeting_response)
            continue
        goodbye_response = handle_goodbye(message)
        if goodbye_response:
            print("ChatBot:", goodbye_response)
            continue
        question_response = handle_questions(message)
        if question_response:
            print("ChatBot:", question_response)
            continue
        unknown_response = handle_unknown(message)
        print("ChatBot:", unknown_response)

if __name__ == "__main__":
    main()
```
