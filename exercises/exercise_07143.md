## Exercise
Write a program that takes in a variable and its data type as input, and checks if the variable is of the correct data type. The program should then print out whether the variable is valid or not.

## Reference Solution
```python
def check_data_type(variable, data_type):
    """
    Checks if a variable is of the correct data type.

    Args:
        variable (any): The variable to be checked.
        data_type (type): The expected data type of the variable.

    Returns:
        bool: True if the variable is of the correct data type, False otherwise.
    """
    try:
        # Check if the variable is of the correct data type
        if isinstance(variable, data_type):
            print(f"The variable {variable} is of type {type(variable).__name__} and is valid.")
            return True
        else:
            print(f"The variable {variable} is of type {type(variable).__name__} but should be of type {data_type.__name__}.")
            return False
    except Exception as e:
        print(f"An error occurred: {e}")
        return False


def main():
    # Test the function with some examples
    check_data_type(10, int)
    check_data_type("hello", str)
    check_data_type(10.5, float)
    check_data_type([1, 2, 3], list)
    check_data_type({"a": 1, "b": 2}, dict)
    check_data_type((1, 2, 3), tuple)
    check_data_type(True, bool)
    check_data_type(None, type(None))
    check_data_type("hello", int)  # This should return False


if __name__ == "__main__":
    main()
```
