## Exercise
Write a program that calculates the price of movie tickets based on the number of adult and child tickets purchased.

## Reference Solution
```python
def calculate_ticket_price(adult_tickets, child_tickets):
    """
    Calculate the total price of movie tickets.

    Args:
        adult_tickets (int): Number of adult tickets purchased.
        child_tickets (int): Number of child tickets purchased.

    Returns:
        float: Total price of movie tickets.
    """
    adult_price = 10.0
    child_price = 5.0

    total_price = (adult_tickets * adult_price) + (child_tickets * child_price)

    return total_price


def main():
    print("Movie Ticket Price Calculator")
    print("--------------------------------")

    adult_tickets = int(input("Enter the number of adult tickets: "))
    child_tickets = int(input("Enter the number of child tickets: "))

    total_price = calculate_ticket_price(adult_tickets, child_tickets)

    print(f"Total price: ${total_price:.2f}")


if __name__ == "__main__":
    main()
```
