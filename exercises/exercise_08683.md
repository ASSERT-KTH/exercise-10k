## Exercise
Write a program that scrambles a word and asks the user to unscramble it.

## Reference Solution
```python
import random

# List of words to scramble
words = ["apple", "banana", "cherry", "date", "elderberry"]

def scramble_word(word):
    """Scramble a word by shuffling its characters."""
    word_list = list(word)
    random.shuffle(word_list)
    return "".join(word_list)

def main():
    # Choose a random word from the list
    word = random.choice(words)
    scrambled_word = scramble_word(word)

    print(f"Unscramble the word: {scrambled_word}")
    user_answer = input("Enter your answer: ")

    if user_answer.lower() == word:
        print("Correct!")
    else:
        print(f"Sorry, the correct answer was {word}.")

if __name__ == "__main__":
    main()
```
