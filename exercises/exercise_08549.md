## Exercise
Create a dictionary with 3 key-value pairs and reduce the values to a single value.

## Reference Solution
```python
import math

def reduce_dict_values(dictionary, method='sum'):
    """
    Reduces dictionary values to a single value.

    Args:
    - dictionary (dict): Input dictionary.
    - method (str): Reduction method. Can be 'sum', 'product', or 'average'.

    Returns:
    - result (float): Reduced value.
    """
    if method == 'sum':
        result = sum(dictionary.values())
    elif method == 'product':
        result = math.prod(dictionary.values())
    elif method == 'average':
        result = sum(dictionary.values()) / len(dictionary)
    else:
        raise ValueError("Invalid method. Choose 'sum', 'product', or 'average'.")

    return result

# Example dictionary
example_dict = {
    'a': 10,
    'b': 20,
    'c': 30
}

# Reduce values to a single value using sum, product, and average
sum_result = reduce_dict_values(example_dict, 'sum')
product_result = reduce_dict_values(example_dict, 'product')
average_result = reduce_dict_values(example_dict, 'average')

print(f"Sum: {sum_result}")
print(f"Product: {product_result}")
print(f"Average: {average_result}")
```
