## Exercise
Write a program that takes a user's income and expenses as input and prints out a summary of their finances.

## Reference Solution
```python
class FinanceTracker:
    def __init__(self):
        self.income = 0
        self.expenses = {}

    def get_income(self):
        while True:
            try:
                self.income = float(input("Enter your monthly income: $"))
                if self.income < 0:
                    print("Income cannot be negative. Please enter a valid income.")
                else:
                    break
            except ValueError:
                print("Invalid input. Please enter a valid number.")

    def get_expenses(self):
        while True:
            expense_name = input("Enter the name of an expense (or 'done' to finish): ")
            if expense_name.lower() == 'done':
                break
            while True:
                try:
                    expense_amount = float(input(f"Enter the monthly amount for {expense_name}: $"))
                    if expense_amount < 0:
                        print("Expense amount cannot be negative. Please enter a valid amount.")
                    else:
                        self.expenses[expense_name] = expense_amount
                        break
                except ValueError:
                    print("Invalid input. Please enter a valid number.")

    def calculate_summary(self):
        total_expenses = sum(self.expenses.values())
        remaining_balance = self.income - total_expenses
        return total_expenses, remaining_balance

    def print_summary(self):
        total_expenses, remaining_balance = self.calculate_summary()
        print("\nSummary of Finances:")
        print(f"Income: ${self.income:.2f}")
        print("Expenses:")
        for expense, amount in self.expenses.items():
            print(f"  - {expense}: ${amount:.2f}")
        print(f"Total Expenses: ${total_expenses:.2f}")
        print(f"Remaining Balance: ${remaining_balance:.2f}")


def main():
    tracker = FinanceTracker()
    tracker.get_income()
    tracker.get_expenses()
    tracker.print_summary()


if __name__ == "__main__":
    main()
```
