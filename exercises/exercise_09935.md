## Exercise
Create a program that takes in two numbers and an arithmetic operation (+, -, *, /) as input, and then performs the specified operation on the two numbers. The program should handle invalid operations and division by zero.

## Reference Solution
```python
import sys

def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    op = input("Enter the operation (+, -, *, /): ")

    # Define operations
    def add(x, y):
        return x + y

    def sub(x, y):
        return x - y

    def mul(x, y):
        return x * y

    def div(x, y):
        if y == 0:
            print("Error: Division by zero")
            sys.exit()
        return x / y

    # Map operations to functions
    operations = {
        '+': add,
        '-': sub,
        '*': mul,
        '/': div
    }

    # Check if operation is valid
    if op not in operations:
        print("Error: Invalid operation")
        sys.exit()

    # Perform operation and print result
    result = operations[op](num1, num2)
    print("Result:", result)

if __name__ == "__main__":
    main()
```
