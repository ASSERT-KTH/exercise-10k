## Exercise
Create a program that uses a dictionary to store student names as keys and their corresponding grades as values. The program should allow users to add, remove, and update student grades. It should also be able to calculate the average grade of all students.

## Reference Solution
```python
class StudentGrades:
    def __init__(self):
        self.grades = {}

    def add_student(self, name, grade):
        """Add a student to the dictionary."""
        if name in self.grades:
            print(f"Student {name} already exists. Please update their grade instead.")
        else:
            self.grades[name] = grade
            print(f"Student {name} added with grade {grade}.")

    def remove_student(self, name):
        """Remove a student from the dictionary."""
        if name in self.grades:
            del self.grades[name]
            print(f"Student {name} removed.")
        else:
            print(f"Student {name} does not exist.")

    def update_grade(self, name, new_grade):
        """Update a student's grade in the dictionary."""
        if name in self.grades:
            self.grades[name] = new_grade
            print(f"Student {name}'s grade updated to {new_grade}.")
        else:
            print(f"Student {name} does not exist. Please add them first.")

    def calculate_average(self):
        """Calculate the average grade of all students."""
        if not self.grades:
            print("No students exist.")
            return
        average_grade = sum(self.grades.values()) / len(self.grades)
        print(f"The average grade is {average_grade:.2f}.")

    def display_grades(self):
        """Display all students and their grades."""
        if not self.grades:
            print("No students exist.")
        else:
            for name, grade in self.grades.items():
                print(f"{name}: {grade}")


def main():
    student_grades = StudentGrades()
    while True:
        print("\n1. Add student")
        print("2. Remove student")
        print("3. Update grade")
        print("4. Calculate average grade")
        print("5. Display grades")
        print("6. Quit")
        choice = input("Choose an option: ")
        if choice == "1":
            name = input("Enter student name: ")
            grade = float(input("Enter student grade: "))
            student_grades.add_student(name, grade)
        elif choice == "2":
            name = input("Enter student name: ")
            student_grades.remove_student(name)
        elif choice == "3":
            name = input("Enter student name: ")
            new_grade = float(input("Enter new grade: "))
            student_grades.update_grade(name, new_grade)
        elif choice == "4":
            student_grades.calculate_average()
        elif choice == "5":
            student_grades.display_grades()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
