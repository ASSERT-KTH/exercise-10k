## Exercise
Write a program that asks the user for their grades and then calculates their GPA based on the following scale: A (4.0), B (3.0), C (2.0), D (1.0), F (0.0).

## Reference Solution
```python
def calculate_gpa():
    """
    Asks the user for their grades and calculates their GPA.
    """
    # Initialize variables
    total_credits = 0
    total_points = 0

    # Ask the user for the number of courses
    num_courses = int(input("Enter the number of courses: "))

    # Create a dictionary to map grades to GPA points
    grade_points = {
        'A': 4.0,
        'B': 3.0,
        'C': 2.0,
        'D': 1.0,
        'F': 0.0
    }

    # Loop through each course
    for i in range(num_courses):
        print(f"\nCourse {i+1}:")
        
        # Ask the user for the grade
        while True:
            grade = input("Enter your grade (A, B, C, D, F): ").upper()
            if grade in grade_points:
                break
            else:
                print("Invalid grade. Please enter A, B, C, D, or F.")
        
        # Ask the user for the number of credits
        while True:
            try:
                credits = int(input("Enter the number of credits: "))
                if credits > 0:
                    break
                else:
                    print("Invalid number of credits. Please enter a positive integer.")
            except ValueError:
                print("Invalid input. Please enter a positive integer.")
        
        # Update the total credits and points
        total_credits += credits
        total_points += grade_points[grade] * credits

    # Calculate the GPA
    if total_credits == 0:
        print("No courses entered. Cannot calculate GPA.")
    else:
        gpa = total_points / total_credits
        print(f"\nYour GPA is: {gpa:.2f}")

# Run the program
calculate_gpa()
```
