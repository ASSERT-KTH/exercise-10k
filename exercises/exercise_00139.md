## Exercise
Write a program that performs basic arithmetic operations (addition, subtraction, multiplication, division) on complex numbers.

## Reference Solution
```python
class ComplexNumber:
    def __init__(self, real=0, imaginary=0):
        """
        Initializes a ComplexNumber object.

        Args:
            real (float): The real part of the complex number. Defaults to 0.
            imaginary (float): The imaginary part of the complex number. Defaults to 0.
        """
        self.real = real
        self.imaginary = imaginary

    def __str__(self):
        """
        Returns a string representation of the complex number.

        Returns:
            str: A string in the format 'a + bj' or 'a - bj' where a is the real part and b is the imaginary part.
        """
        sign = '+' if self.imaginary >= 0 else '-'
        return f"{self.real} {sign} {abs(self.imaginary)}j"

    def add(self, other):
        """
        Adds two complex numbers.

        Args:
            other (ComplexNumber): The complex number to add.

        Returns:
            ComplexNumber: The sum of the two complex numbers.
        """
        real = self.real + other.real
        imaginary = self.imaginary + other.imaginary
        return ComplexNumber(real, imaginary)

    def subtract(self, other):
        """
        Subtracts one complex number from another.

        Args:
            other (ComplexNumber): The complex number to subtract.

        Returns:
            ComplexNumber: The difference of the two complex numbers.
        """
        real = self.real - other.real
        imaginary = self.imaginary - other.imaginary
        return ComplexNumber(real, imaginary)

    def multiply(self, other):
        """
        Multiplies two complex numbers.

        Args:
            other (ComplexNumber): The complex number to multiply.

        Returns:
            ComplexNumber: The product of the two complex numbers.
        """
        real = self.real * other.real - self.imaginary * other.imaginary
        imaginary = self.real * other.imaginary + self.imaginary * other.real
        return ComplexNumber(real, imaginary)

    def divide(self, other):
        """
        Divides one complex number by another.

        Args:
            other (ComplexNumber): The complex number to divide by.

        Returns:
            ComplexNumber: The quotient of the two complex numbers.

        Raises:
            ZeroDivisionError: If the divisor is zero.
        """
        if other.real == 0 and other.imaginary == 0:
            raise ZeroDivisionError("Cannot divide by zero")

        denominator = other.real ** 2 + other.imaginary ** 2
        real = (self.real * other.real + self.imaginary * other.imaginary) / denominator
        imaginary = (self.imaginary * other.real - self.real * other.imaginary) / denominator
        return ComplexNumber(real, imaginary)


def main():
    # Example usage:
    num1 = ComplexNumber(3, 4)
    num2 = ComplexNumber(1, 2)

    print("Number 1:", num1)
    print("Number 2:", num2)

    print("Addition:", num1.add(num2))
    print("Subtraction:", num1.subtract(num2))
    print("Multiplication:", num1.multiply(num2))
    print("Division:", num1.divide(num2))


if __name__ == "__main__":
    main()
```
