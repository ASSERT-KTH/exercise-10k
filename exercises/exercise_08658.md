## Exercise
Write a function that calculates the average of all elements in a list of lists.

## Reference Solution
```python
def calculate_average(list_of_lists):
    """
    Calculate the average of all elements in a list of lists.

    Args:
        list_of_lists (list): A list containing lists of numbers.

    Returns:
        float: The average of all elements in the list of lists.

    Raises:
        ValueError: If the list of lists is empty or contains non-numeric values.
    """
    # Check if the list of lists is empty
    if not list_of_lists or not any(list_of_lists):
        raise ValueError("The list of lists is empty.")

    # Flatten the list of lists
    flattened_list = [num for sublist in list_of_lists for num in sublist]

    # Check if the flattened list contains non-numeric values
    if not all(isinstance(num, (int, float)) for num in flattened_list):
        raise ValueError("The list of lists contains non-numeric values.")

    # Calculate the average
    average = sum(flattened_list) / len(flattened_list)

    return average


# Example usage
if __name__ == "__main__":
    list_of_lists = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    try:
        average = calculate_average(list_of_lists)
        print(f"The average is: {average}")
    except ValueError as e:
        print(f"Error: {e}")
```
