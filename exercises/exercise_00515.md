## Exercise
Write a program to create a new list by applying a specific operation to each element in a given list.

## Reference Solution
```python
def apply_operation(input_list, operation, value):
    """
    Applies a specific operation to each element in the input list.

    Args:
        input_list (list): The input list of numbers.
        operation (str): The operation to apply. Can be 'add', 'subtract', 'multiply', or 'divide'.
        value (float): The value to use for the operation.

    Returns:
        list: The new list with the operation applied to each element.
    """
    new_list = []
    for num in input_list:
        if operation == 'add':
            new_list.append(num + value)
        elif operation == 'subtract':
            new_list.append(num - value)
        elif operation == 'multiply':
            new_list.append(num * value)
        elif operation == 'divide':
            if value != 0:
                new_list.append(num / value)
            else:
                raise ZeroDivisionError("Cannot divide by zero")
        else:
            raise ValueError("Invalid operation")
    return new_list

def main():
    # Example usage
    input_list = [1, 2, 3, 4, 5]
    operation = 'multiply'
    value = 2

    new_list = apply_operation(input_list, operation, value)
    print(f"Input List: {input_list}")
    print(f"Operation: {operation} by {value}")
    print(f"New List: {new_list}")

if __name__ == "__main__":
    main()
```
