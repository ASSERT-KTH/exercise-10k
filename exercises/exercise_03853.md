## Exercise
Create a program that uses a for loop to print each node in a tree data structure.

## Reference Solution
```python
class Node:
    """Represents a node in the tree."""
    def __init__(self, value):
        self.value = value
        self.children = []

class Tree:
    """Represents the tree data structure."""
    def __init__(self, root):
        self.root = Node(root)

    def add_child(self, parent_value, child_value):
        """Adds a child node to the tree."""
        parent = self.find_node(parent_value)
        if parent:
            parent.children.append(Node(child_value))

    def find_node(self, value, node=None):
        """Finds a node in the tree by value."""
        if node is None:
            node = self.root
        if node.value == value:
            return node
        for child in node.children:
            found_node = self.find_node(value, child)
            if found_node:
                return found_node
        return None

    def print_tree(self):
        """Prints each node in the tree using a for loop."""
        nodes_to_visit = [self.root]
        while nodes_to_visit:
            node = nodes_to_visit.pop(0)
            print(node.value)
            nodes_to_visit.extend(node.children)

# Example usage:
tree = Tree("Root")
tree.add_child("Root", "Child 1")
tree.add_child("Root", "Child 2")
tree.add_child("Child 1", "Grandchild 1")
tree.add_child("Child 1", "Grandchild 2")

print("Tree nodes:")
tree.print_tree()
```
