## Exercise
Write a program to take a message as input and encode it using a Caesar cipher, then decode the encoded message

## Reference Solution
```python
def caesar_encode(message, shift):
    """
    Encode a message using a Caesar cipher with the specified shift.

    Args:
        message (str): The message to encode.
        shift (int): The shift value for the Caesar cipher.

    Returns:
        str: The encoded message.
    """
    encoded_message = ""
    for char in message:
        if char.isalpha():
            ascii_offset = 97 if char.islower() else 65
            encoded_char = chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
            encoded_message += encoded_char
        else:
            encoded_message += char
    return encoded_message

def caesar_decode(encoded_message, shift):
    """
    Decode an encoded message using a Caesar cipher with the specified shift.

    Args:
        encoded_message (str): The encoded message to decode.
        shift (int): The shift value for the Caesar cipher.

    Returns:
        str: The decoded message.
    """
    decoded_message = ""
    for char in encoded_message:
        if char.isalpha():
            ascii_offset = 97 if char.islower() else 65
            decoded_char = chr((ord(char) - ascii_offset - shift) % 26 + ascii_offset)
            decoded_message += decoded_char
        else:
            decoded_message += char
    return decoded_message

def main():
    message = input("Enter a message: ")
    shift = int(input("Enter a shift value: "))

    encoded_message = caesar_encode(message, shift)
    print(f"Encoded message: {encoded_message}")

    decoded_message = caesar_decode(encoded_message, shift)
    print(f"Decoded message: {decoded_message}")

if __name__ == "__main__":
    main()
```
