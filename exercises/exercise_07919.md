## Exercise
Write a program to calculate the distance between two points in a 2D plane.

## Reference Solution
```python
import math

def calculate_distance(point1, point2):
    """
    Calculate the distance between two points in a 2D plane.

    Args:
        point1 (tuple): The coordinates of the first point as (x, y).
        point2 (tuple): The coordinates of the second point as (x, y).

    Returns:
        float: The distance between the two points.
    """
    x1, y1 = point1
    x2, y2 = point2
    return math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)

def main():
    # Define the coordinates of the two points
    point1 = (1, 2)
    point2 = (4, 6)

    # Calculate the distance between the two points
    distance = calculate_distance(point1, point2)

    # Print the result
    print(f"The distance between {point1} and {point2} is {distance:.2f} units.")

if __name__ == "__main__":
    main()
```
