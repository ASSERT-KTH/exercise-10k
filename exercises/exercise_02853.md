## Exercise
Write a function to create a copy of a dictionary.

## Reference Solution
```python
def copy_dictionary(dictionary):
    """
    Creates a copy of a dictionary.

    Args:
        dictionary (dict): The dictionary to be copied.

    Returns:
        dict: A copy of the dictionary.
    """
    return dictionary.copy()

# Example usage:
original_dict = {"a": 1, "b": 2, "c": [3, 4]}
copied_dict = copy_dictionary(original_dict)

# Modify the original dictionary
original_dict["c"].append(5)

print("Original dictionary:", original_dict)
print("Copied dictionary:", copied_dict)
```

```python
import copy

def deep_copy_dictionary(dictionary):
    """
    Creates a deep copy of a dictionary.

    Args:
        dictionary (dict): The dictionary to be copied.

    Returns:
        dict: A deep copy of the dictionary.
    """
    return copy.deepcopy(dictionary)

# Example usage:
original_dict = {"a": 1, "b": 2, "c": [3, 4]}
copied_dict = deep_copy_dictionary(original_dict)

# Modify the original dictionary
original_dict["c"].append(5)

print("Original dictionary:", original_dict)
print("Copied dictionary:", copied_dict)
```
