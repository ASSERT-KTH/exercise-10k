## Exercise
Create a program that allows users to track their workouts and nutrition using conditional statements, providing fitness recommendations.

## Reference Solution
```python
class WorkoutTracker:
    def __init__(self):
        self.workouts = []
        self.nutrition = []

    def add_workout(self):
        """Add a new workout to the tracker"""
        workout_type = input("Enter workout type (e.g., cardio, strength, yoga): ")
        duration = int(input("Enter workout duration (in minutes): "))
        calories_burned = int(input("Enter calories burned (approximate): "))
        self.workouts.append({"type": workout_type, "duration": duration, "calories_burned": calories_burned})

    def add_nutrition(self):
        """Add a new nutrition entry to the tracker"""
        food_type = input("Enter food type (e.g., protein, carbohydrate, fat): ")
        calories_consumed = int(input("Enter calories consumed (approximate): "))
        self.nutrition.append({"type": food_type, "calories_consumed": calories_consumed})

    def view_workouts(self):
        """View all workouts in the tracker"""
        print("Workouts:")
        for i, workout in enumerate(self.workouts, start=1):
            print(f"{i}. Type: {workout['type']}, Duration: {workout['duration']} minutes, Calories Burned: {workout['calories_burned']}")

    def view_nutrition(self):
        """View all nutrition entries in the tracker"""
        print("Nutrition:")
        for i, nutrition in enumerate(self.nutrition, start=1):
            print(f"{i}. Type: {nutrition['type']}, Calories Consumed: {nutrition['calories_consumed']}")

    def provide_recommendations(self):
        """Provide fitness recommendations based on user input"""
        total_calories_burned = sum(workout["calories_burned"] for workout in self.workouts)
        total_calories_consumed = sum(nutrition["calories_consumed"] for nutrition in self.nutrition)

        if total_calories_burned > total_calories_consumed:
            print("You are burning more calories than you are consuming. Consider increasing your caloric intake to support muscle growth and recovery.")
        elif total_calories_burned < total_calories_consumed:
            print("You are consuming more calories than you are burning. Consider increasing your physical activity to burn off excess calories.")
        else:
            print("You are in a caloric balance. Continue to monitor your workouts and nutrition to maintain a healthy balance.")

def main():
    tracker = WorkoutTracker()

    while True:
        print("1. Add Workout")
        print("2. Add Nutrition")
        print("3. View Workouts")
        print("4. View Nutrition")
        print("5. Provide Recommendations")
        print("6. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            tracker.add_workout()
        elif choice == "2":
            tracker.add_nutrition()
        elif choice == "3":
            tracker.view_workouts()
        elif choice == "4":
            tracker.view_nutrition()
        elif choice == "5":
            tracker.provide_recommendations()
        elif choice == "6":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
