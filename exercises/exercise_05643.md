## Exercise
Create a function that plays a game of Rock Paper Scissors against the computer. The function should take in the user's choice and return the result of the game.

## Reference Solution
```python
import random

def play_rock_paper_scissors(user_choice):
    """
    This function plays a game of Rock Paper Scissors against the computer.
    
    Args:
        user_choice (str): The user's choice. It can be 'rock', 'paper', or 'scissors'.
    
    Returns:
        str: The result of the game.
    """

    # First, we define the possible choices in a list
    choices = ['rock', 'paper', 'scissors']

    # We check if the user's choice is valid
    if user_choice not in choices:
        return "Invalid choice. Please choose 'rock', 'paper', or 'scissors'."

    # Then, we generate the computer's choice randomly
    computer_choice = random.choice(choices)

    # We print out the user's choice and the computer's choice
    print(f"\nUser chose: {user_choice}")
    print(f"Computer chose: {computer_choice}\n")

    # We determine the winner based on the game's rules
    if user_choice == computer_choice:
        # If both choices are the same, it's a tie
        return "It's a tie!"
    elif (user_choice == 'rock' and computer_choice == 'scissors') or \
         (user_choice == 'scissors' and computer_choice == 'paper') or \
         (user_choice == 'paper' and computer_choice == 'rock'):
        # If the user's choice beats the computer's choice, the user wins
        return "User wins!"
    else:
        # Otherwise, the computer wins
        return "Computer wins!"

# Example usage:
print(play_rock_paper_scissors('rock'))
```
