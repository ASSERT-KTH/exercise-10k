## Exercise
Write a program that reads a JSON object from a file and prints out the object's values.

## Reference Solution
```python
import json

def print_json_values(filename):
    """
    Reads a JSON object from a file and prints out its values.

    Args:
        filename (str): The path to the JSON file.
    """
    try:
        with open(filename, 'r') as file:
            # Load the JSON data from the file
            data = json.load(file)

            # Print the JSON values
            print_json_values_recursive(data)

    except FileNotFoundError:
        print(f"File '{filename}' not found.")
    except json.JSONDecodeError as e:
        print(f"Failed to parse JSON: {e}")

def print_json_values_recursive(data):
    """
    Recursively prints the values of a JSON object.

    Args:
        data (dict or list): The JSON object to print.
    """
    if isinstance(data, dict):
        # If the data is a dictionary, print its values
        for key, value in data.items():
            print(f"{key}: {value}")

            # If the value is a dictionary or list, print its values recursively
            if isinstance(value, (dict, list)):
                print_json_values_recursive(value)

    elif isinstance(data, list):
        # If the data is a list, print its values
        for i, value in enumerate(data):
            print(f"[{i}]: {value}")

            # If the value is a dictionary or list, print its values recursively
            if isinstance(value, (dict, list)):
                print_json_values_recursive(value)

# Example usage:
if __name__ == "__main__":
    filename = "example.json"
    print_json_values(filename)
```
