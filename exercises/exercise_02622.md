## Exercise
Create a program that takes two numbers and an operator as input and performs the operation.

## Reference Solution
```python
def calculator(num1, num2, operator):
    """
    Performs arithmetic operations based on the operator.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operator (str): The arithmetic operator.

    Returns:
        float: The result of the operation.

    Raises:
        ValueError: If the operator is not valid.
    """
    if operator == '+':
        return num1 + num2
    elif operator == '-':
        return num1 - num2
    elif operator == '*':
        return num1 * num2
    elif operator == '/':
        if num2 != 0:
            return num1 / num2
        else:
            raise ZeroDivisionError("Cannot divide by zero")
    else:
        raise ValueError("Invalid operator")

def main():
    # Get user input
    num1 = float(input("Enter the first number: "))
    operator = input("Enter the operator (+, -, *, /): ")
    num2 = float(input("Enter the second number: "))

    # Perform the operation
    try:
        result = calculator(num1, num2, operator)
        print(f"{num1} {operator} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        print(e)

if __name__ == "__main__":
    main()
```
