## Exercise
Write a program that asks the user for a subject, verb, and object, and builds a sentence using them.

## Reference Solution
```python
def get_user_input(prompt):
    """Get user input for a given prompt."""
    while True:
        user_input = input(prompt)
        if user_input:
            return user_input
        print("Please enter a value.")

def build_sentence(subject, verb, object):
    """Build a sentence using the provided subject, verb, and object."""
    return f"{subject} {verb} {object}."

def main():
    # Get user input for subject, verb, and object
    subject = get_user_input("Enter a subject: ")
    verb = get_user_input("Enter a verb: ")
    object = get_user_input("Enter an object: ")

    # Build and print the sentence
    sentence = build_sentence(subject, verb, object)
    print("Here is the sentence:")
    print(sentence)

if __name__ == "__main__":
    main()
```
