## Exercise
Develop a program to count the occurrences of consonants in a given string.

## Reference Solution
```python
def count_consonants(input_string):
    """
    Count the occurrences of consonants in a given string.

    Args:
        input_string (str): The input string to count consonants from.

    Returns:
        int: The total count of consonants in the input string.
    """
    # Initialize count to 0
    consonant_count = 0

    # Define vowels
    vowels = 'aeiou'

    # Convert input string to lowercase
    input_string = input_string.lower()

    # Iterate over each character in the input string
    for char in input_string:
        # Check if character is an alphabet letter and not a vowel
        if char.isalpha() and char not in vowels:
            # Increment consonant count
            consonant_count += 1

    # Return consonant count
    return consonant_count

def main():
    # Example usage
    input_string = "Hello, World!"
    consonant_count = count_consonants(input_string)
    print(f"Consonant count in '{input_string}': {consonant_count}")

if __name__ == "__main__":
    main()
```
