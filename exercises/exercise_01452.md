## Exercise
Write a program that takes the subtotal and sales tax rate as input and outputs the total amount to pay.

## Reference Solution
```python
def calculate_total(subtotal, tax_rate):
    """
    Calculates the total amount to pay based on the subtotal and sales tax rate.

    Args:
        subtotal (float): The subtotal amount.
        tax_rate (float): The sales tax rate as a decimal.

    Returns:
        float: The total amount to pay.
    """
    tax_amount = subtotal * tax_rate
    total = subtotal + tax_amount
    return total

def main():
    # Get user input
    subtotal = float(input("Enter the subtotal: $"))
    tax_rate = float(input("Enter the sales tax rate (in %): ")) / 100

    # Calculate total amount to pay
    total = calculate_total(subtotal, tax_rate)

    # Print result
    print(f"Subtotal: ${subtotal:.2f}")
    print(f"Sales Tax ({tax_rate*100}%): ${subtotal * tax_rate:.2f}")
    print(f"Total: ${total:.2f}")

if __name__ == "__main__":
    main()
```
