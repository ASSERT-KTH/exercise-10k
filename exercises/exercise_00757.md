## Exercise
Write a program that asks the user a series of quiz questions and keeps track of their score

## Reference Solution
```python
# quiz_program.py

# Define a dictionary to store quiz questions and answers
quiz_questions = {
    "What is the capital of France?": "Paris",
    "What is the largest planet in our solar system?": "Jupiter",
    "Who painted the Starry Night?": "Vincent van Gogh",
    "What is the chemical symbol for gold?": "Au",
    "Who wrote Romeo and Juliet?": "William Shakespeare"
}

def ask_question(question, answer):
    """Ask the user a question and check their response."""
    user_response = input(question + " ")
    if user_response.lower() == answer.lower():
        return True
    else:
        return False

def run_quiz(questions):
    """Run the quiz and keep track of the user's score."""
    score = 0
    for question, answer in questions.items():
        if ask_question(question, answer):
            print("Correct!\n")
            score += 1
        else:
            print(f"Sorry, the correct answer was {answer}.\n")
    return score

def main():
    print("Welcome to the quiz program!")
    score = run_quiz(quiz_questions)
    print(f"Quiz complete! Your final score is {score} out of {len(quiz_questions)}.")

if __name__ == "__main__":
    main()
```
