## Exercise
Create a program that takes two numbers as input and performs addition, subtraction, multiplication, and division based on the user's choice. The program should handle invalid inputs and division by zero errors.

## Reference Solution
```python
# Importing libraries
import numpy as np

# Defining the function
def calculadora():
    # Getting user input
    n1 = float(input("Enter the first number: "))
    n2 = float(input("Enter the second number: "))

    # Getting user choice
    print("\nChoose an operation:")
    print("1. Addition")
    print("2. Subtraction")
    print("3. Multiplication")
    print("4. Division")
    choice = input("Enter your choice (1/2/3/4): ")

    # Performing arithmetic operations
    if choice == '1':
        result = n1 + n2
        print(f"\n{n1} + {n2} = {result}")
    elif choice == '2':
        result = n1 - n2
        print(f"\n{n1} - {n2} = {result}")
    elif choice == '3':
        result = n1 * n2
        print(f"\n{n1} * {n2} = {result}")
    elif choice == '4':
        if n2 != 0:
            result = n1 / n2
            print(f"\n{n1} / {n2} = {result}")
        else:
            print("\nError: Division by zero is not allowed.")
    else:
        print("\nError: Invalid choice. Please choose a valid operation.")

# Calling the function
calculadora()
```
