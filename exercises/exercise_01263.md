## Exercise
Write a program to modify the elements of an array.

## Reference Solution
```python
def update_array(array, index, value):
    """
    Updates the element at the specified index with the given value.
    
    Args:
    array (list): The input array.
    index (int): The index of the element to update.
    value: The new value for the element.
    """
    if index < len(array):
        array[index] = value
    else:
        print("Index out of range.")

def insert_into_array(array, index, value):
    """
    Inserts a new element with the given value at the specified index.
    
    Args:
    array (list): The input array.
    index (int): The index where the new element will be inserted.
    value: The value of the new element.
    """
    if index <= len(array):
        array.insert(index, value)
    else:
        print("Index out of range.")

def delete_from_array(array, index):
    """
    Deletes the element at the specified index.
    
    Args:
    array (list): The input array.
    index (int): The index of the element to delete.
    """
    if index < len(array):
        del array[index]
    else:
        print("Index out of range.")

def replace_in_array(array, old_value, new_value):
    """
    Replaces all occurrences of the old value with the new value.
    
    Args:
    array (list): The input array.
    old_value: The value to replace.
    new_value: The replacement value.
    """
    for i in range(len(array)):
        if array[i] == old_value:
            array[i] = new_value

# Example usage
if __name__ == "__main__":
    array = [1, 2, 3, 4, 5]
    print("Original array:", array)

    update_array(array, 2, 10)
    print("After updating the 3rd element:", array)

    insert_into_array(array, 1, 20)
    print("After inserting a new element at index 1:", array)

    delete_from_array(array, 0)
    print("After deleting the 1st element:", array)

    replace_in_array(array, 10, 30)
    print("After replacing 10 with 30:", array)
```
