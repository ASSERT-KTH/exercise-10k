## Exercise
Write a function that finds the longest increasing subsequence in a given list.

def longest_increasing_subsequence(sequence):
    """
    Finds the longest increasing subsequence in a given list.

    Args:
        sequence (list): A list of integers.

    Returns:
        list: The longest increasing subsequence.
    """
    if not sequence:
        return []

    # Initialize a list to store the lengths of the longest increasing subsequences
    lengths = [1] * len(sequence)

    # Initialize a list to store the predecessors in the longest increasing subsequences
    predecessors = [None] * len(sequence)

    # Compute the lengths of the longest increasing subsequences
    for i in range(1, len(sequence)):
        for j in range(i):
            if sequence[i] > sequence[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                predecessors[i] = j

    # Find the index of the last element in the longest increasing subsequence
    max_length_idx = max(range(len(lengths)), key=lambda i: lengths[i])

    # Reconstruct the longest increasing subsequence
    subsequence = []
    while max_length_idx is not None:
        subsequence.append(sequence[max_length_idx])
        max_length_idx = predecessors[max_length_idx]

    # Return the longest increasing subsequence in the correct order
    return subsequence[::-1]


# Example usage
if __name__ == "__main__":
    sequence = [10, 22, 9, 33, 21, 50, 41, 60, 80]
    print("Longest Increasing Subsequence:", longest_increasing_subsequence(sequence))
