## Exercise
Write a program that converts a given amount of money from one currency to another

## Reference Solution
```python
# Import the required libraries
import requests

def get_exchange_rate(base_currency, target_currency):
    """
    Get the exchange rate from the base currency to the target currency.
    
    Parameters:
    base_currency (str): The base currency to convert from.
    target_currency (str): The target currency to convert to.
    
    Returns:
    float: The exchange rate from the base currency to the target currency.
    """
    response = requests.get(f"https://api.exchangerate-api.com/v4/latest/{base_currency}")
    data = response.json()
    return data["rates"][target_currency]

def convert_currency(amount, base_currency, target_currency):
    """
    Convert the given amount of money from the base currency to the target currency.
    
    Parameters:
    amount (float): The amount of money to convert.
    base_currency (str): The base currency to convert from.
    target_currency (str): The target currency to convert to.
    
    Returns:
    float: The converted amount of money.
    """
    exchange_rate = get_exchange_rate(base_currency, target_currency)
    return amount * exchange_rate

def main():
    # Get the user input
    amount = float(input("Enter the amount of money to convert: "))
    base_currency = input("Enter the base currency (e.g. USD, EUR, JPY): ").upper()
    target_currency = input("Enter the target currency (e.g. USD, EUR, JPY): ").upper()
    
    # Convert the currency
    converted_amount = convert_currency(amount, base_currency, target_currency)
    
    # Print the result
    print(f"{amount} {base_currency} is equal to {converted_amount} {target_currency}")

if __name__ == "__main__":
    main()
```
