## Exercise
Write a program that demonstrates the use of dictionary operations in programming.

## Reference Solution
```python
# dictionary_operations.py

def main():
    # Create a dictionary
    student_grades = {
        "Alice": 90,
        "Bob": 85,
        "Charlie": 95
    }

    print("Initial Dictionary:")
    print(student_grades)

    # Add a new key-value pair
    student_grades["David"] = 92
    print("\nDictionary after adding a new key-value pair:")
    print(student_grades)

    # Update a value
    student_grades["Alice"] = 91
    print("\nDictionary after updating a value:")
    print(student_grades)

    # Remove a key-value pair
    del student_grades["Bob"]
    print("\nDictionary after removing a key-value pair:")
    print(student_grades)

    # Iterate over the dictionary
    print("\nIterating over the dictionary:")
    for name, grade in student_grades.items():
        print(f"{name}: {grade}")

    # Check if a key exists
    print("\nChecking if a key exists:")
    print("Alice" in student_grades)  # True
    print("Bob" in student_grades)   # False

    # Get the value of a key (with a default value if the key doesn't exist)
    print("\nGetting the value of a key (with a default value):")
    print(student_grades.get("Charlie", "Not found"))  # 95
    print(student_grades.get("Bob", "Not found"))      # Not found

if __name__ == "__main__":
    main()
```
