## Exercise
Write a program that uses a loop to print the numbers from 1 to 10, then from 10 to 1, using both a for loop and a while loop.

## Reference Solution
```python
#include <stdio.h>

int main() {
    int i;

    // Printing numbers from 1 to 10 using for loop
    printf("Printing numbers from 1 to 10 using for loop:\n");
    for (i = 1; i <= 10; i++) {
        printf("%d ", i);
    }
    printf("\n");

    // Printing numbers from 1 to 10 using while loop
    printf("Printing numbers from 1 to 10 using while loop:\n");
    i = 1;
    while (i <= 10) {
        printf("%d ", i);
        i++;
    }
    printf("\n");

    // Printing numbers from 10 to 1 using for loop
    printf("Printing numbers from 10 to 1 using for loop:\n");
    i = 10;
    for (; i > 0; i--) {
        printf("%d ", i);
    }
    printf("\n");

    // Printing numbers from 10 to 1 using while loop
    printf("Printing numbers from 10 to 1 using while loop:\n");
    i = 10;
    while (i > 0) {
        printf("%d ", i);
        i--;
    }
    printf("\n");

    return 0;
}
```
