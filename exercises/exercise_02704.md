## Exercise
Write a program to analyze a given text and calculate the number of vowels, consonants, and words.

## Reference Solution
```python
def count_vowels(text):
    """
    Count the number of vowels in a given text.

    Args:
        text (str): The input text.

    Returns:
        int: The number of vowels in the text.
    """
    vowels = 'aeiou'
    return sum(1 for char in text.lower() if char in vowels)


def count_consonants(text):
    """
    Count the number of consonants in a given text.

    Args:
        text (str): The input text.

    Returns:
        int: The number of consonants in the text.
    """
    vowels = 'aeiou'
    return sum(1 for char in text.lower() if char.isalpha() and char not in vowels)


def count_words(text):
    """
    Count the number of words in a given text.

    Args:
        text (str): The input text.

    Returns:
        int: The number of words in the text.
    """
    return len(text.split())


def analyze_text(text):
    """
    Analyze a given text and calculate the number of vowels, consonants, and words.

    Args:
        text (str): The input text.

    Returns:
        dict: A dictionary containing the number of vowels, consonants, and words.
    """
    vowels = count_vowels(text)
    consonants = count_consonants(text)
    words = count_words(text)
    return {
        'vowels': vowels,
        'consonants': consonants,
        'words': words
    }


def main():
    text = input("Enter a text: ")
    analysis = analyze_text(text)
    print("Analysis:")
    print(f"Vowels: {analysis['vowels']}")
    print(f"Consonants: {analysis['consonants']}")
    print(f"Words: {analysis['words']}")


if __name__ == "__main__":
    main()
```
