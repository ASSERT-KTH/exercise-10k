## Exercise
Write a program that uses a while loop to keep asking the user for input until they enter a valid grade.

## Reference Solution
```python
def get_valid_grade():
    """
    Continuously asks the user for input until a valid grade is entered.
    """
    while True:
        user_input = input("Please enter a grade (A, B, C, D, F, or a number between 0 and 100): ")
        
        # Check if the input is a letter grade
        if user_input.upper() in ["A", "B", "C", "D", "F"]:
            return user_input.upper()
        
        # Check if the input is a numerical grade
        try:
            numerical_grade = int(user_input)
            if 0 <= numerical_grade <= 100:
                return numerical_grade
            else:
                print("Invalid grade. Please enter a number between 0 and 100.")
        except ValueError:
            print("Invalid input. Please enter a letter grade (A, B, C, D, F) or a number between 0 and 100.")

def main():
    valid_grade = get_valid_grade()
    print(f"You entered a valid grade: {valid_grade}")

if __name__ == "__main__":
    main()
```
