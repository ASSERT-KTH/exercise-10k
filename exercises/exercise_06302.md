## Exercise
Write a program that accesses a specific key in a dictionary and prints its corresponding value

## Reference Solution
```python
def access_dictionary_key(dictionary, key):
    """
    Accesses a specific key in a dictionary and prints its corresponding value.

    Args:
        dictionary (dict): The dictionary to access.
        key (any): The key to access in the dictionary.

    Returns:
        None
    """
    try:
        # Attempt to access the key in the dictionary
        value = dictionary[key]
        print(f"The value for key '{key}' is: {value}")
    except KeyError:
        # Handle the case where the key does not exist in the dictionary
        print(f"Key '{key}' not found in the dictionary.")

# Example usage:
if __name__ == "__main__":
    # Create a sample dictionary
    sample_dict = {
        "name": "John",
        "age": 30,
        "city": "New York"
    }

    # Access a key in the dictionary
    access_dictionary_key(sample_dict, "name")
    access_dictionary_key(sample_dict, "age")
    access_dictionary_key(sample_dict, "nonexistent_key")
```
