## Exercise
Create a program to calculate the fuel efficiency of a vehicle given the distance traveled and fuel consumed.

## Reference Solution
```python
def calculate_fuel_efficiency(distance, fuel_consumed):
    """
    Calculate the fuel efficiency of a vehicle.

    Args:
        distance (float): Distance traveled in kilometers.
        fuel_consumed (float): Fuel consumed in liters.

    Returns:
        float: Fuel efficiency in kilometers per liter (km/L).
    """
    if fuel_consumed <= 0:
        raise ValueError("Fuel consumed must be greater than zero.")

    fuel_efficiency = distance / fuel_consumed
    return fuel_efficiency

def main():
    print("Fuel Efficiency Calculator")
    print("----------------------------")

    distance = float(input("Enter the distance traveled (km): "))
    fuel_consumed = float(input("Enter the fuel consumed (L): "))

    try:
        fuel_efficiency = calculate_fuel_efficiency(distance, fuel_consumed)
        print(f"Fuel efficiency: {fuel_efficiency:.2f} km/L")
    except ValueError as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
