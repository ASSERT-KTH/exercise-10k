## Exercise
Implement an LZW compressor to compress a given string

class LZWCompressor:
    def __init__(self):
        self.dictionary = {}

    def compress(self, input_string):
        """
        Compress the input string using LZW compression.

        Args:
            input_string (str): The input string to be compressed.

        Returns:
            list: The compressed string represented as a list of integers.
        """
        self.dictionary = {chr(i): i for i in range(256)}
        result = []
        phrase = ""

        for char in input_string:
            phrase_char = phrase + char
            if phrase_char in self.dictionary:
                phrase = phrase_char
            else:
                result.append(self.dictionary[phrase])
                self.dictionary[phrase_char] = len(self.dictionary) + 1
                phrase = char

        if phrase:
            result.append(self.dictionary[phrase])

        return result

    def decompress(self, compressed_string):
        """
        Decompress the compressed string using LZW decompression.

        Args:
            compressed_string (list): The compressed string represented as a list of integers.

        Returns:
            str: The decompressed string.
        """
        self.dictionary = {i: chr(i) for i in range(256)}
        result = ""
        phrase = chr(compressed_string[0])

        result += phrase
        for code in compressed_string[1:]:
            if code in self.dictionary:
                phrase_char = self.dictionary[code]
            elif code == len(self.dictionary):
                phrase_char = phrase + phrase[0]
            else:
                raise ValueError("Invalid compressed string")

            result += phrase_char
            self.dictionary[len(self.dictionary)] = phrase + phrase_char[0]
            phrase = phrase_char

        return result

    compressor = LZWCompressor()
    input_string = "TOBEORNOTTOBEORTOBEORNOT"
    compressed_string = compressor.compress(input_string)
    print("Compressed string:", compressed_string)

    decompressed_string = compressor.decompress(compressed_string)
    print("Decompressed string:", decompressed_string)
