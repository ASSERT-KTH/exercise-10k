## Exercise
Write a program to perform basic character operations (ASCII value, character concatenation).

## Reference Solution
```python
def ascii_value(char):
    """
    Returns the ASCII value of a character.

    Args:
        char (str): A single character.

    Returns:
        int: The ASCII value of the character.
    """
    return ord(char)


def concatenate_chars(char1, char2):
    """
    Concatenates two characters.

    Args:
        char1 (str): The first character.
        char2 (str): The second character.

    Returns:
        str: The concatenated characters.
    """
    return char1 + char2


def main():
    # Example usage:
    char1 = 'A'
    char2 = 'B'

    print(f"The ASCII value of '{char1}' is: {ascii_value(char1)}")
    print(f"The ASCII value of '{char2}' is: {ascii_value(char2)}")

    concatenated = concatenate_chars(char1, char2)
    print(f"The concatenation of '{char1}' and '{char2}' is: {concatenated}")


if __name__ == "__main__":
    main()
```
