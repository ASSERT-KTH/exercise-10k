## Exercise
You are given a dictionary containing student information, including names and grades. Write a program to calculate the average grade of all students and then find the student with the highest grade. If there are multiple students with the same highest grade, return all of them.

## Reference Solution
```python
def calculate_average_grade(students):
    """
    Calculate the average grade of all students.

    Args:
        students (dict): A dictionary containing student information, where each key is a student's name and each value is their grade.

    Returns:
        float: The average grade of all students.
    """
    total_grade = sum(students.values())
    average_grade = total_grade / len(students)
    return average_grade


def find_highest_grade_students(students):
    """
    Find the student(s) with the highest grade.

    Args:
        students (dict): A dictionary containing student information, where each key is a student's name and each value is their grade.

    Returns:
        list: A list of student names with the highest grade.
    """
    max_grade = max(students.values())
    highest_grade_students = [name for name, grade in students.items() if grade == max_grade]
    return highest_grade_students


def main():
    # Example usage
    students = {
        "John": 85,
        "Alice": 92,
        "Bob": 78,
        "Charlie": 92,
        "David": 88
    }

    average_grade = calculate_average_grade(students)
    print(f"Average grade: {average_grade:.2f}")

    highest_grade_students = find_highest_grade_students(students)
    print(f"Student(s) with the highest grade: {highest_grade_students}")


if __name__ == "__main__":
    main()
```
