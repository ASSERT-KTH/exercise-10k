## Exercise
Create a program that takes a city name as input and prints the current weather forecast for that city.

## Reference Solution
```python
import requests
import json

def get_weather(city, api_key):
    """
    Fetches the current weather forecast for a given city.

    Args:
        city (str): City name
        api_key (str): OpenWeatherMap API key

    Returns:
        dict: Current weather forecast data
    """
    base_url = "http://api.openweathermap.org/data/2.5/weather"
    params = {
        "q": city,
        "appid": api_key,
        "units": "metric"  # Use metric units (e.g., Celsius)
    }

    response = requests.get(base_url, params=params)

    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error {response.status_code}: {response.text}")
        return None

def print_weather(weather_data):
    """
    Prints the current weather forecast in a human-readable format.

    Args:
        weather_data (dict): Current weather forecast data
    """
    if weather_data:
        print(f"Weather in {weather_data['name']}:")

        print(f"Humidity: {weather_data['main']['humidity']}%")
        print(f"Weather condition: {weather_data['weather'][0]['description']}")
    else:
        print("Failed to retrieve weather data.")

def main():
    api_key = "YOUR_OPENWEATHERMAP_API_KEY"  # Replace with your API key
    city = input("Enter city name: ")

    weather_data = get_weather(city, api_key)
    print_weather(weather_data)

if __name__ == "__main__":
    main()
```
