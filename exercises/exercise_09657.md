## Exercise
Write a program that asks for a string and converts it to uppercase or lowercase.

## Reference Solution
```python
def convert_string_case(input_string, case):
    """
    Converts the input string to the specified case.

    Args:
        input_string (str): The string to be converted.
        case (str): The case to convert to ('upper' or 'lower').

    Returns:
        str: The converted string.
    """
    if case == 'upper':
        return input_string.upper()
    elif case == 'lower':
        return input_string.lower()
    else:
        raise ValueError("Invalid case. Please choose 'upper' or 'lower'.")

def main():
    # Get the input string from the user
    input_string = input("Please enter a string: ")

    # Ask the user for the case conversion option
    print("Choose an option:")
    print("1. Convert to uppercase")
    print("2. Convert to lowercase")
    option = input("Enter your choice (1/2): ")

    # Convert the string based on the user's choice
    if option == '1':
        converted_string = convert_string_case(input_string, 'upper')
    elif option == '2':
        converted_string = convert_string_case(input_string, 'lower')
    else:
        print("Invalid option. Exiting program.")
        return

    # Print the converted string
    print(f"Converted string: {converted_string}")

if __name__ == "__main__":
    main()
```
