## Exercise
Write a program to find the longest common prefix of two strings

## Reference Solution
```python
def longest_common_prefix(str1, str2):
    """
    Finds the longest common prefix of two strings.

    Args:
        str1 (str): The first input string.
        str2 (str): The second input string.

    Returns:
        str: The longest common prefix of the two input strings.
    """
    prefix = ""
    for char1, char2 in zip(str1, str2):
        if char1 == char2:
            prefix += char1
        else:
            break
    return prefix

# Example usage:
def main():
    str1 = "hello"
    str2 = "hello world"
    print("Longest common prefix:", longest_common_prefix(str1, str2))

    str1 = "abc"
    str2 = "abcd"
    print("Longest common prefix:", longest_common_prefix(str1, str2))

    str1 = "def"
    str2 = "ghi"
    print("Longest common prefix:", longest_common_prefix(str1, str2))

if __name__ == "__main__":
    main()
```
