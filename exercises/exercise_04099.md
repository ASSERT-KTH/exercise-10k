## Exercise
Write a function that takes a list of functions as input and applies them in a pipeline

## Reference Solution
```python
from typing import Callable, Any

def apply_pipeline(functions: list[Callable], input_value: Any) -> Any:
    """
    Applies a list of functions in a pipeline manner.

    Args:
        functions (list[Callable]): A list of functions to apply.
        input_value (Any): The initial input value.

    Returns:
        Any: The result of the pipeline.
    """
    result = input_value
    for func in functions:
        result = func(result)
    return result

# Example usage:
def double(x: int) -> int:
    """Doubles the input value."""
    return x * 2

def add_one(x: int) -> int:
    """Adds one to the input value."""
    return x + 1

def square(x: int) -> int:
    """Squares the input value."""
    return x ** 2

if __name__ == "__main__":
    functions = [double, add_one, square]
    input_value = 5
    result = apply_pipeline(functions, input_value)
    print(f"Pipeline result: {result}")  # Output: Pipeline result: 121
```
