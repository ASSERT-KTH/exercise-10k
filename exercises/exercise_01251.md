Write a function that demonstrates function overloading.

def greet(name, msg="Hello"):
    """
    Prints out a personalized greeting message.

    Args:
        name (str): The name of the person to greet.
        msg (str, optional): The greeting message. Defaults to "Hello".
    """
    print(f"{msg}, {name}!")

# Test the function
greet("John")  # Output: Hello, John!
greet("Jane", "Hi")  # Output: Hi, Jane!

```python
from functools import singledispatch

@singledispatch
def fun(arg):
    """
    A generic function that can be overloaded.

    Args:
        arg: The argument to process.
    """
    return arg

@fun.register
def _(arg: int):
    """
    Overloads the fun function for integer arguments.

    Args:
        arg (int): The integer to process.

    Returns:
        int: The square of the input integer.
    """
    return arg ** 2

@fun.register
def _(arg: list):
    """
    Overloads the fun function for list arguments.

    Args:
        arg (list): The list to process.

    Returns:
        int: The length of the input list.
    """
    return len(arg)

# Test the function
print(fun(5))  # Output: 25
print(fun([1, 2, 3]))  # Output: 3
print(fun("hello"))  # Output: hello
```
