## Exercise
Given an array of integers, write a program to calculate statistical measures such as mean, median, and standard deviation.

## Reference Solution
```python
import numpy as np

def calculate_statistical_measures(arr):
    """
    Calculate mean, median, and standard deviation of an array.

    Args:
        arr (list): A list of integers.

    Returns:
        dict: A dictionary containing the mean, median, and standard deviation.
    """
    mean = np.mean(arr)
    median = np.median(arr)
    std_dev = np.std(arr)

    return {
        "mean": mean,
        "median": median,
        "standard_deviation": std_dev
    }

def main():
    # Example usage
    arr = [1, 2, 3, 4, 5]
    stats = calculate_statistical_measures(arr)

    print("Statistical Measures:")
    print(f"Mean: {stats['mean']}")
    print(f"Median: {stats['median']}")
    print(f"Standard Deviation: {stats['standard_deviation']}")

if __name__ == "__main__":
    main()
```
