## Exercise
Create a program that simulates a phonebook using a dictionary. The program should allow users to add, remove, and search for contacts. Each contact should have a name and a phone number.

## Reference Solution
```python
class Phonebook:
    def __init__(self):
        self.contacts = {}

    def add_contact(self, name, phone_number):
        """Add a new contact to the phonebook."""
        if name in self.contacts:
            print(f"Contact with name '{name}' already exists.")
            response = input("Do you want to update the existing contact? (yes/no): ")
            if response.lower() == "yes":
                self.contacts[name] = phone_number
                print(f"Contact with name '{name}' updated successfully.")
            else:
                print("Contact not added.")
        else:
            self.contacts[name] = phone_number
            print(f"Contact with name '{name}' added successfully.")

    def remove_contact(self, name):
        """Remove a contact from the phonebook."""
        if name in self.contacts:
            del self.contacts[name]
            print(f"Contact with name '{name}' removed successfully.")
        else:
            print(f"Contact with name '{name}' does not exist.")

    def search_contact(self, name):
        """Search for a contact in the phonebook."""
        if name in self.contacts:
            print(f"Contact with name '{name}' found. Phone number: {self.contacts[name]}")
        else:
            print(f"Contact with name '{name}' not found.")

    def display_contacts(self):
        """Display all contacts in the phonebook."""
        if not self.contacts:
            print("Phonebook is empty.")
        else:
            print("Contacts:")
            for name, phone_number in self.contacts.items():
                print(f"Name: {name}, Phone number: {phone_number}")


def main():
    phonebook = Phonebook()

    while True:
        print("\nPhonebook Menu:")
        print("1. Add contact")
        print("2. Remove contact")
        print("3. Search contact")
        print("4. Display contacts")
        print("5. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter contact name: ")
            phone_number = input("Enter contact phone number: ")
            phonebook.add_contact(name, phone_number)
        elif choice == "2":
            name = input("Enter contact name: ")
            phonebook.remove_contact(name)
        elif choice == "3":
            name = input("Enter contact name: ")
            phonebook.search_contact(name)
        elif choice == "4":
            phonebook.display_contacts()
        elif choice == "5":
            print("Exiting phonebook.")
            break
        else:
            print("Invalid option. Please choose again.")


if __name__ == "__main__":
    main()
```
