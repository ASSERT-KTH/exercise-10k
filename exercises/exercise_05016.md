## Exercise
Write a program that calculates the result of the expression 10 + 20 * 3. Print the result to the console.

## Reference Solution

def calculate_expression():
    """
    Calculate the result of the expression 10 + 20 * 3.
    """
    # Define the expression
    a = 10
    b = 20
    c = 3
    
    # Calculate the result using the order of operations (PEMDAS/BODMAS)
    result = a + b * c
    
    return result

def main():
    """
    Main function to print the result to the console.
    """
    result = calculate_expression()
    print("The result of the expression 10 + 20 * 3 is:", result)

if __name__ == "__main__":
    main()
