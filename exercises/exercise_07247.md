## Exercise
Create a program that takes in two numbers and an arithmetic operation as input, and then performs the specified operation on the numbers. The program should support addition, subtraction, multiplication, and division.

## Reference Solution
```python
def perform_arithmetic_operation(num1, num2, operation):
    """
    Performs the specified arithmetic operation on two numbers.

    Args:
        num1 (float): The first number.
        num2 (float): The second number.
        operation (str): The arithmetic operation to perform. Can be '+', '-', '*', or '/'.

    Returns:
        float: The result of the arithmetic operation.
    """

    # Check if the operation is valid
    if operation not in ['+', '-', '*', '/']:
        raise ValueError("Invalid operation. Supported operations are '+', '-', '*', and '/'.")

    # Perform the specified operation
    if operation == '+':
        # If the operation is addition, add the two numbers
        result = num1 + num2
    elif operation == '-':
        # If the operation is subtraction, subtract the second number from the first
        result = num1 - num2
    elif operation == '*':
        # If the operation is multiplication, multiply the two numbers
        result = num1 * num2
    elif operation == '/':
        # If the operation is division, divide the first number by the second
        # Check for division by zero
        if num2 == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        result = num1 / num2

    return result


def main():
    # Get the two numbers and the operation from the user
    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))
    operation = input("Enter the arithmetic operation (+, -, *, /): ")

    try:
        # Perform the operation and print the result
        result = perform_arithmetic_operation(num1, num2, operation)
        print(f"{num1} {operation} {num2} = {result}")
    except (ValueError, ZeroDivisionError) as e:
        # Handle any errors that occur
        print(f"Error: {e}")


if __name__ == "__main__":
    main()
```
