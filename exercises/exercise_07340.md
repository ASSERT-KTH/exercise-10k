## Exercise
Write a program that takes input from the user with a graphical user interface and prints out the results.

## Reference Solution
```python
import tkinter as tk
from tkinter import messagebox

class UserInputGUI:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("User Input GUI")

        # Create input fields
        self.name_label = tk.Label(self.window, text="Name:")
        self.name_label.grid(row=0, column=0)
        self.name_entry = tk.Entry(self.window)
        self.name_entry.grid(row=0, column=1)

        self.age_label = tk.Label(self.window, text="Age:")
        self.age_label.grid(row=1, column=0)
        self.age_entry = tk.Entry(self.window)
        self.age_entry.grid(row=1, column=1)

        # Create submit button
        self.submit_button = tk.Button(self.window, text="Submit", command=self.submit_input)
        self.submit_button.grid(row=2, column=0, columnspan=2)

    def submit_input(self):
        # Get input values
        name = self.name_entry.get()
        age = self.age_entry.get()

        # Validate input
        if not name or not age:
            messagebox.showerror("Error", "Please fill out all fields")
            return

        # Print results
        print("Name:", name)
        print("Age:", age)

        # Clear input fields
        self.name_entry.delete(0, tk.END)
        self.age_entry.delete(0, tk.END)

    def run(self):
        self.window.mainloop()

if __name__ == "__main__":
    gui = UserInputGUI()
    gui.run()
```
