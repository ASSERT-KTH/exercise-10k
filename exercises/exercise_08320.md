## Exercise
Create a dictionary to store student grades. The dictionary should have student names as keys and a list of grades as values. Write a program that allows users to add students, add grades to existing students, and calculate the average grade for each student.

## Reference Solution
```python
# student_grades.py

def add_student(grades):
    """Add a new student to the grades dictionary."""
    name = input("Enter the student's name: ")
    if name not in grades:
        grades[name] = []
        print(f"Student '{name}' added successfully.")
    else:
        print(f"Student '{name}' already exists.")

def add_grade(grades):
    """Add a grade to an existing student."""
    name = input("Enter the student's name: ")
    if name in grades:
        while True:
            try:
                grade = float(input("Enter the grade: "))
                if 0 <= grade <= 100:
                    grades[name].append(grade)
                    print(f"Grade '{grade}' added to '{name}' successfully.")
                    break
                else:
                    print("Grade must be between 0 and 100.")
            except ValueError:
                print("Invalid input. Please enter a number.")
    else:
        print(f"Student '{name}' does not exist.")

def calculate_average(grades):
    """Calculate the average grade for each student."""
    for name, grade_list in grades.items():
        if grade_list:
            average = sum(grade_list) / len(grade_list)
            print(f"The average grade for '{name}' is {average:.2f}.")
        else:
            print(f"No grades available for '{name}'.")

def main():
    grades = {}

    while True:
        print("\nStudent Grades Menu:")
        print("1. Add Student")
        print("2. Add Grade")
        print("3. Calculate Average Grade")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            add_student(grades)
        elif choice == "2":
            add_grade(grades)
        elif choice == "3":
            calculate_average(grades)
        elif choice == "4":
            print("Exiting the program. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
