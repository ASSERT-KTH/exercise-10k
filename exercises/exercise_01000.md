## Exercise
Create a program that calculates statistical measures (mean, median, mode, standard deviation) for a list of numbers. The program uses if-else statements to check for valid input.

## Reference Solution
```python
import statistics
import math

def calculate_statistics(numbers):
    """
    Calculate statistical measures (mean, median, mode, standard deviation) for a list of numbers.
    
    Args:
    numbers (list): A list of numbers.
    
    Returns:
    A dictionary containing the mean, median, mode, and standard deviation.
    """
    
    # Check if input is a list
    if not isinstance(numbers, list):
        raise ValueError("Input must be a list of numbers.")
        
    # Check if list is empty
    if len(numbers) == 0:
        raise ValueError("Input list cannot be empty.")
        
    # Check if all elements in the list are numbers
    if not all(isinstance(num, (int, float)) for num in numbers):
        raise ValueError("All elements in the list must be numbers.")
        
    # Calculate mean
    mean = statistics.mean(numbers)
    
    # Calculate median
    median = statistics.median(numbers)
    
    # Calculate mode
    try:
        mode = statistics.mode(numbers)
    except statistics.StatisticsError:
        mode = "No unique mode found"
        
    # Calculate standard deviation
    if len(numbers) > 1:
        std_dev = statistics.stdev(numbers)
    else:
        std_dev = "Standard deviation is not defined for a single data point"
        
    return {
        "mean": mean,
        "median": median,
        "mode": mode,
        "standard_deviation": std_dev
    }

def main():
    # Example usage
    numbers = [1, 2, 3, 4, 5]
    stats = calculate_statistics(numbers)
    
    print("Statistical Measures:")
    print(f"Mean: {stats['mean']}")
    print(f"Median: {stats['median']}")
    print(f"Mode: {stats['mode']}")
    print(f"Standard Deviation: {stats['standard_deviation']}")

if __name__ == "__main__":
    main()
```
