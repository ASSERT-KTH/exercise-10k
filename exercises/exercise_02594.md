## Exercise
Write a program to find the missing number in a given sequence of numbers (e.g. 1, 2, 3, 5, 6).

## Reference Solution
```python
def find_missing_number(sequence):
    """
    Finds the missing number in a given sequence of numbers.

    Args:
        sequence (list): A list of integers representing the sequence.

    Returns:
        int: The missing number in the sequence.
    """
    # Sort the sequence in ascending order
    sequence.sort()

    # Iterate through the sequence to find the missing number
    for i in range(len(sequence) - 1):
        # If the difference between two consecutive numbers is greater than 1, 
        # then there is a missing number
        if sequence[i + 1] - sequence[i] > 1:
            # The missing number is the one that would come after the current number
            return sequence[i] + 1

    # If no missing number is found, return None
    return None


def main():
    # Example usage
    sequence = [1, 2, 3, 5, 6]
    missing_number = find_missing_number(sequence)

    if missing_number is not None:
        print(f"The missing number in the sequence is: {missing_number}")
    else:
        print("No missing number found in the sequence.")


if __name__ == "__main__":
    main()
```
