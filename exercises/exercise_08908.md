## Exercise
Write a program to find the missing number in a given list of consecutive numbers.

## Reference Solution
```python
def find_missing_number(num_list):
    """
    Find the missing number in a list of consecutive numbers.

    Args:
        num_list (list): A list of consecutive numbers with one number missing.

    Returns:
        int: The missing number in the sequence.
    """
    # Calculate the sum of the first and last numbers in the complete sequence
    n = len(num_list) + 1
    total_sum = (n * (num_list[0] + num_list[-1])) // 2

    # Calculate the sum of the given numbers
    given_sum = sum(num_list)

    # The missing number is the difference between the total sum and the given sum
    missing_number = total_sum - given_sum

    return missing_number


def main():
    # Example usage
    num_list = [1, 2, 3, 5, 6]
    missing_number = find_missing_number(num_list)
    print(f"The missing number in the sequence is: {missing_number}")


if __name__ == "__main__":
    main()
```
