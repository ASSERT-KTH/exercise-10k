## Exercise
Write a program that takes an array of numbers as input and returns the first duplicate.

## Reference Solution
```python
def find_first_duplicate(nums):
    """
    Returns the first duplicate in the input array.

    Args:
        nums (list): A list of numbers.

    Returns:
        int: The first duplicate in the list. If no duplicates are found, returns None.
    """
    seen = set()
    for num in nums:
        if num in seen:
            return num
        seen.add(num)
    return None

# Example usage
def main():
    numbers = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(numbers)
    if duplicate is not None:
        print(f"The first duplicate is: {duplicate}")
    else:
        print("No duplicates found.")

if __name__ == "__main__":
    main()
```
