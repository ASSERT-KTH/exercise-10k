## Exercise
Write a function to encrypt a given string using a Caesar cipher with a given shift

```python
def caesar_encrypt(plain_text: str, shift: int) -> str:
    """
    Encrypts a given string using a Caesar cipher with a specified shift.

    Args:
        plain_text (str): The string to be encrypted.
        shift (int): The shift value for the Caesar cipher.

    Returns:
        str: The encrypted string.
    """
    encrypted_text = ""

    for char in plain_text:
        if char.isalpha():
            ascii_offset = ord('a') if char.islower() else ord('A')
            encrypted_text += chr((ord(char) - ascii_offset + shift) % 26 + ascii_offset)
        else:
            encrypted_text += char

    return encrypted_text

# Example usage
if __name__ == "__main__":
    plain_text = "Hello, World!"
    shift = 3
    encrypted_text = caesar_encrypt(plain_text, shift)
    print(f"Plain text: {plain_text}")
    print(f"Shift: {shift}")
    print(f"Encrypted text: {encrypted_text}")
```
