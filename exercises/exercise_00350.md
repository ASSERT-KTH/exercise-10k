## Exercise
Create a function that deletes an element from a list.

## Reference Solution
```python
def delete_element(lst, index):
    """
    Deletes an element from a list at the specified index.

    Args:
        lst (list): The list from which to delete the element.
        index (int): The index of the element to delete.

    Returns:
        list: The updated list with the element deleted.
    """
    try:
        # Check if the index is within the bounds of the list
        if index < 0 or index >= len(lst):
            raise IndexError("Index out of range")
        
        # Delete the element at the specified index
        del lst[index]
        
        return lst
    
    except IndexError as e:
        print(f"Error: {e}")
        return None

# Example usage
if __name__ == "__main__":
    my_list = [1, 2, 3, 4, 5]
    print("Original List:", my_list)
    
    updated_list = delete_element(my_list, 2)
    print("Updated List:", updated_list)
```
