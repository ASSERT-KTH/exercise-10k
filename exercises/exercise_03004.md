## Exercise
Write a program to extract a substring from a given string.

## Reference Solution
```python
def extract_substring(input_string, start_index, end_index):
    """
    Extracts a substring from the input string.

    Args:
        input_string (str): The original string.
        start_index (int): The starting index of the substring.
        end_index (int): The ending index of the substring.

    Returns:
        str: The extracted substring.
    """
    if start_index < 0 or end_index > len(input_string) or start_index > end_index:
        raise ValueError("Invalid indices")

    return input_string[start_index:end_index]

def main():
    input_string = input("Enter the input string: ")
    start_index = int(input("Enter the start index: "))
    end_index = int(input("Enter the end index: "))

    try:
        substring = extract_substring(input_string, start_index, end_index)
        print("Extracted substring:", substring)
    except ValueError as e:
        print("Error:", e)

if __name__ == "__main__":
    main()
```
