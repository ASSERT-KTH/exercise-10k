## Exercise
Use a for loop to find the longest substring without repeating characters in a given string

## Reference Solution
```python
def longest_substring_without_repeating_characters(s: str) -> str:
    """
    Find the longest substring without repeating characters in a given string.

    Args:
    s (str): The input string.

    Returns:
    str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    max_length = 0
    max_substring = ""
    char_set = set()

    left = 0  # Left pointer of the sliding window

    for right in range(len(s)):
        # Move the right pointer to the right
        while s[right] in char_set:
            # If the character is already in the set, remove the leftmost character
            char_set.remove(s[left])
            left += 1  # Move the left pointer to the right

        char_set.add(s[right])

        # Update the maximum length and substring if the current window is larger
        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring


# Example usage:
if __name__ == "__main__":
    string = "abcabcbb"
    print(f"Longest substring without repeating characters: {longest_substring_without_repeating_characters(string)}")
```
