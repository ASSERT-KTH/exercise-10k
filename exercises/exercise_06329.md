## Exercise
Write a program that takes an array as input, calculates and returns its mean, median, and mode.

## Reference Solution
```python
import numpy as np
from scipy import stats

def calculate_statistics(array):
    """
    Calculate the mean, median, and mode of an array.

    Args:
        array (list): Input array.

    Returns:
        tuple: A tuple containing the mean, median, and mode of the array.
    """
    # Calculate the mean
    mean = np.mean(array)

    # Calculate the median
    median = np.median(array)

    # Calculate the mode
    mode = stats.mode(array)[0][0]

    return mean, median, mode

def main():
    # Example usage
    array = [1, 2, 3, 4, 4, 5, 5, 5]
    mean, median, mode = calculate_statistics(array)

    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Mode: {mode}")

if __name__ == "__main__":
    main()
```
