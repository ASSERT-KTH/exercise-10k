## Exercise
Write a program that displays a calendar for a given month and year.

## Reference Solution
```python
import calendar

def display_calendar(year, month):
    """
    Display a calendar for the given month and year.

    Args:
        year (int): The year for which to display the calendar.
        month (int): The month for which to display the calendar.
    """
    # Create a plain text calendar
    cal = calendar.monthcalendar(year, month)

    # Print the month and year
    print(calendar.month_name[month], year)
    print("Mo Tu We Th Fr Sa Su")

    # Print each week in the month
    for week in cal:
        for day in week:
            if day == 0:
                print("  ", end="")
            else:
                print(f"{day:>2}", end=" ")
        print()

def main():
    year = int(input("Enter the year: "))
    month = int(input("Enter the month (1-12): "))

    display_calendar(year, month)

if __name__ == "__main__":
    main()
```
