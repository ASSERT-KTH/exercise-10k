## Exercise
Write a function that takes a size as input and returns a Sudoku board

## Reference Solution

import random

def generate_sudoku_board(size):
    """
    Generate a Sudoku board of a specified size.

    Args:
        size (int): The size of the Sudoku board. Must be a perfect square.

    Returns:
        list: A 2D list representing the Sudoku board.
    """
    # Check if size is a perfect square
    if int(size ** 0.5) ** 2 != size:
        raise ValueError("Size must be a perfect square")

    # Initialize an empty board
    board = [[0 for _ in range(size)] for _ in range(size)]

    def is_valid(board, row, col, num):
        """
        Check if a number can be placed at a given position on the board.

        Args:
            board (list): The Sudoku board.
            row (int): The row index.
            col (int): The column index.
            num (int): The number to check.

        Returns:
            bool: True if the number can be placed, False otherwise.
        """
        # Check the row
        for x in range(size):
            if board[row][x] == num:
                return False

        # Check the column
        for x in range(size):
            if board[x][col] == num:
                return False

        # Check the sub-grid
        subgrid_size = int(size ** 0.5)
        start_row, start_col = row - row % subgrid_size, col - col % subgrid_size
        for i in range(subgrid_size):
            for j in range(subgrid_size):
                if board[i + start_row][j + start_col] == num:
                    return False
        return True

    def solve_sudoku(board):
        """
        Solve the Sudoku puzzle using backtracking.

        Args:
            board (list): The Sudoku board.
        """
        for i in range(size):
            for j in range(size):
                if board[i][j] == 0:
                    numbers = list(range(1, size + 1))
                    random.shuffle(numbers)
                    for num in numbers:
                        if is_valid(board, i, j, num):
                            board[i][j] = num
                            if solve_sudoku(board):
                                return True
                            board[i][j] = 0
                    return False
        return True

    # Generate the Sudoku board
    solve_sudoku(board)
    return board

def print_board(board):
    """
    Print the Sudoku board in a readable format.

    Args:
        board (list): The Sudoku board.
    """
    size = len(board)
    subgrid_size = int(size ** 0.5)
    for i in range(size):
        if i % subgrid_size == 0 and i != 0:
            print("- - - - - - - - - - - -")
        for j in range(size):
            if j % subgrid_size == 0 and j != 0:
                print(" | ", end="")
            if j == size - 1:
                print(board[i][j])
            else:
                print(str(board[i][j]) + " ", end="")

# Example usage
if __name__ == "__main__":
    size = 9
    board = generate_sudoku_board(size)
    print_board(board)
