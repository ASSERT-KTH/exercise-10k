## Exercise
Write a function that takes a function with multiple arguments and returns a curried version of the function.

## Reference Solution

def curry(func):
    """
    This function takes a function with multiple arguments and returns a curried version of the function.
    
    Args:
        func (function): The function to be curried.
    
    Returns:
        function: The curried version of the function.
    """
    
    # We define a new function that will be the curried version of the input function
    def curried(*args, **kwargs):
        
        # If the number of arguments is equal to the number of arguments required by the function, 
        # we can call the function with these arguments
        if len(args) + len(kwargs) == func.__code__.co_argcount:
            # We call the function with the given arguments and return the result
            return func(*args, **kwargs)
        
        # If the number of arguments is less than the number required by the function, 
        # we return a new function that takes the remaining arguments
        else:
            # We define a new function that takes the remaining arguments
            def next_curried(*next_args, **next_kwargs):
                # We call the curried function with the arguments from the previous call and the new arguments
                return curried(*(args + next_args), **dict(kwargs, **next_kwargs))
            
            # We return the new function
            return next_curried
    
    # We return the curried function
    return curried

# Example usage:
def add(a, b, c):
    return a + b + c

curried_add = curry(add)
print(curried_add(1)(2)(3))  # Outputs: 6
