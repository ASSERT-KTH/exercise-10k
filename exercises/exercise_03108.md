## Exercise
Create a dictionary with 10 key-value pairs. Count the occurrences of each value in the dictionary.

## Reference Solution
```python
def count_value_occurrences(dictionary):
    """Count the occurrences of each value in the dictionary."""
    value_counts = {}
    for key, value in dictionary.items():
        if value in value_counts:
            value_counts[value] += 1
        else:
            value_counts[value] = 1
    return value_counts

# Create a dictionary with 10 key-value pairs
data = {
    'a': 1,
    'b': 2,
    'c': 1,
    'd': 3,
    'e': 2,
    'f': 1,
    'g': 4,
    'h': 3,
    'i': 2,
    'j': 1
}

# Count the occurrences of each value
value_counts = count_value_occurrences(data)

# Print the results
print("Dictionary:")
for key, value in data.items():
    print(f"{key}: {value}")

print("\nValue Counts:")
for value, count in value_counts.items():
    print(f"{value}: {count}")
```
