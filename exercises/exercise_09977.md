## Exercise
Write a program that demonstrates the concept of variable lifetime.

## Reference Solution
```python
# Global variable
global_variable = 10

def outer_function():
    """Demonstrates the lifetime of local variables."""
    # Local variable in outer_function
    outer_local_variable = 20

    def inner_function():
        """Demonstrates the lifetime of local variables in nested functions."""
        # Local variable in inner_function
        inner_local_variable = 30
        print("Inner function: global_variable =", global_variable)
        print("Inner function: outer_local_variable =", outer_local_variable)
        print("Inner function: inner_local_variable =", inner_local_variable)

    print("Outer function: global_variable =", global_variable)
    print("Outer function: outer_local_variable =", outer_local_variable)
    inner_function()

def static_variable_example():
    """Demonstrates the concept of static variables."""
    if not hasattr(static_variable_example, "static_variable"):
        static_variable_example.static_variable = 0
    static_variable_example.static_variable += 1
    print("Static variable:", static_variable_example.static_variable)

def main():
    print("Global variable before outer_function:", global_variable)
    outer_function()
    print("Global variable after outer_function:", global_variable)
    static_variable_example()  # Static variable: 1
    static_variable_example()  # Static variable: 2

if __name__ == "__main__":
    main()
```
