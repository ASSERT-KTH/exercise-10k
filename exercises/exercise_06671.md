## Exercise
Write a program to calculate the percentage decrease between two numbers

## Reference Solution
```python
def calculate_percentage_decrease(original_value, new_value):
    """
    Calculate the percentage decrease between two numbers.

    Args:
        original_value (float): The original value.
        new_value (float): The new value.

    Returns:
        float: The percentage decrease.

    Raises:
        ValueError: If the original value is zero.
    """
    if original_value == 0:
        raise ValueError("Original value cannot be zero")

    percentage_decrease = ((original_value - new_value) / original_value) * 100
    return percentage_decrease


def main():
    # Example usage:
    original_value = 100
    new_value = 80
    try:
        percentage_decrease = calculate_percentage_decrease(original_value, new_value)
        print(f"The percentage decrease from {original_value} to {new_value} is {percentage_decrease:.2f}%")
    except ValueError as e:
        print(str(e))


if __name__ == "__main__":
    main()
```
