## Exercise
Write a program that takes a phone number as input and formats it according to a specific pattern (e.g. (123) 456-7890).

## Reference Solution
```python
def format_phone_number(phone_number):
    """
    Formats a given phone number according to the pattern (XXX) XXX-XXXX.
    
    Args:
    phone_number (str): The phone number to be formatted.
    
    Returns:
    str: The formatted phone number.
    """
    
    # Remove all non-digit characters from the phone number
    digits_only = ''.join(filter(str.isdigit, phone_number))
    
    # Check if the phone number has 10 digits
    if len(digits_only) != 10:
        return "Invalid phone number. Please enter a 10-digit phone number."
    
    # Format the phone number
    formatted_phone_number = f"({digits_only[:3]}) {digits_only[3:6]}-{digits_only[6:]}"
    
    return formatted_phone_number

def main():
    phone_number = input("Enter a phone number: ")
    print(format_phone_number(phone_number))

if __name__ == "__main__":
    main()
```
