## Exercise
Create a program that uses a while loop to draw a fractal tree of a given size.

## Reference Solution
```python
import turtle

def draw_tree(t, branch_length, angle, shrink_factor, min_length):
    """
    Recursively draw a fractal tree.

    :param t: Turtle object to use for drawing.
    :param branch_length: Length of the current branch.
    :param angle: Angle to turn at each branch.
    :param shrink_factor: Factor by which to shrink each branch.
    :param min_length: Minimum length of a branch.
    """
    if branch_length < min_length:
        return

    t.forward(branch_length)
    t.left(angle)
    draw_tree(t, branch_length * shrink_factor, angle, shrink_factor, min_length)
    t.right(angle * 2)
    draw_tree(t, branch_length * shrink_factor, angle, shrink_factor, min_length)
    t.left(angle)
    t.backward(branch_length)

def main():
    # Set up the turtle
    window = turtle.Screen()
    window.setup(800, 600)
    t = turtle.Turtle()
    t.speed(0)  # Fastest speed
    t.left(90)  # Start pointing up

    # Draw the tree
    draw_tree(t, 100, 30, 0.7, 5)

    # Keep the window open
    window.mainloop()

if __name__ == "__main__":
    main()
```
