## Exercise
Create a program that analyzes a given list of variables and prints out statistics about their data types (e.g., count, mean, median).

## Reference Solution
```python
import statistics
import numpy as np

def analyze_data(data):
    """
    Analyzes a given list of variables and prints out statistics about their data types.

    Args:
        data (list): A list of variables.
    """

    # Initialize an empty dictionary to store the count of each data type
    data_type_count = {}

    # Initialize empty lists to store numeric data
    numeric_data = []

    # Iterate over each variable in the list
    for var in data:
        # Get the data type of the variable
        var_type = type(var).__name__

        # Increment the count of the data type in the dictionary
        if var_type in data_type_count:
            data_type_count[var_type] += 1
        else:
            data_type_count[var_type] = 1

        # If the variable is numeric, add it to the numeric data list
        if var_type in ['int', 'float']:
            numeric_data.append(var)

    # Print the count of each data type
    print("Data Type Count:")
    for data_type, count in data_type_count.items():
        print(f"{data_type}: {count}")

    # If there is numeric data, calculate and print additional statistics
    if numeric_data:
        print("\nNumeric Data Statistics:")
        print(f"Mean: {np.mean(numeric_data)}")
        print(f"Median: {np.median(numeric_data)}")
        print(f"Min: {min(numeric_data)}")
        print(f"Max: {max(numeric_data)}")
        print(f"Standard Deviation: {np.std(numeric_data)}")

# Example usage:
data = [1, 2, 3, 4, 5, 'a', 'b', 'c', 6.7, 8.9, True, False, None]
analyze_data(data)
```
