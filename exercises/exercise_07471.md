## Exercise
Write a program that checks if a given string is a valid variable name in a programming language

## Reference Solution
```python
import keyword
import re

def is_valid_variable_name(name):
    """
    Checks if a given string is a valid variable name.

    Args:
        name (str): The string to check.

    Returns:
        bool: True if the string is a valid variable name, False otherwise.
    """
    # Check if the name is a reserved keyword
    if keyword.iskeyword(name):
        return False

    # Check if the name matches the valid variable name pattern
    pattern = r'^[a-zA-Z_][a-zA-Z0-9_]*$'
    if re.match(pattern, name):
        return True

    return False

# Example usage
def main():
    names = ["hello_world", "123abc", "_variable", "class", "valid-variable"]
    for name in names:
        if is_valid_variable_name(name):
            print(f"'{name}' is a valid variable name.")
        else:
            print(f"'{name}' is not a valid variable name.")

if __name__ == "__main__":
    main()
```
