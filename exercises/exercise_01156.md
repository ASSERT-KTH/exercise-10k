## Exercise
Create a program that solves the Tower of Hanoi puzzle. The program should move disks from one peg to another, subject to certain rules.

def tower_of_hanoi(n, from_peg, to_peg, aux_peg):
    """
    Solves the Tower of Hanoi puzzle by moving n disks from the from_peg to the to_peg.

    Args:
        n (int): The number of disks.
        from_peg (str): The peg where the disks are initially located.
        to_peg (str): The peg where the disks need to be moved.
        aux_peg (str): The auxiliary peg used to facilitate the move.
    """
    if n == 1:
        # Base case: move a single disk from the from_peg to the to_peg.
        print(f"Move disk 1 from {from_peg} to {to_peg}")
        return

    # Move n-1 disks from the from_peg to the aux_peg, using the to_peg as a temporary storage.
    tower_of_hanoi(n - 1, from_peg, aux_peg, to_peg)

    # Move the nth disk from the from_peg to the to_peg.
    print(f"Move disk {n} from {from_peg} to {to_peg}")

    # Move the n-1 disks from the aux_peg to the to_peg, using the from_peg as a temporary storage.
    tower_of_hanoi(n - 1, aux_peg, to_peg, from_peg)


def main():
    # Define the number of disks and the pegs.
    n = 3
    from_peg = "A"
    to_peg = "C"
    aux_peg = "B"

    # Solve the Tower of Hanoi puzzle.
    print(f"Solving Tower of Hanoi puzzle with {n} disks:")
    tower_of_hanoi(n, from_peg, to_peg, aux_peg)


if __name__ == "__main__":
    main()
