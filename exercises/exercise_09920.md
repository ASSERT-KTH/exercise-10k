## Exercise
Write a program that allows users to track their expenses and income.

## Reference Solution
```python
import datetime

class Transaction:
    def __init__(self, amount, description, type):
        self.amount = amount
        self.description = description
        self.type = type
        self.date = datetime.date.today()

class ExpenseTracker:
    def __init__(self):
        self.transactions = []

    def add_transaction(self):
        amount = float(input("Enter amount: "))
        description = input("Enter description: ")
        type = input("Enter type (income/expense): ")
        transaction = Transaction(amount, description, type)
        self.transactions.append(transaction)
        print("Transaction added successfully!")

    def delete_transaction(self):
        if not self.transactions:
            print("No transactions to delete.")
            return
        for i, transaction in enumerate(self.transactions):
            print(f"{i+1}. {transaction.description} - {transaction.amount} - {transaction.type}")
        choice = int(input("Enter the number of the transaction to delete: ")) - 1
        if choice < 0 or choice >= len(self.transactions):
            print("Invalid choice.")
            return
        del self.transactions[choice]
        print("Transaction deleted successfully!")

    def view_transactions(self):
        if not self.transactions:
            print("No transactions to view.")
            return
        for transaction in self.transactions:
            print(f"{transaction.description} - {transaction.amount} - {transaction.type} - {transaction.date}")

    def calculate_balance(self):
        income = sum(transaction.amount for transaction in self.transactions if transaction.type == "income")
        expense = sum(transaction.amount for transaction in self.transactions if transaction.type == "expense")
        balance = income - expense
        print(f"Income: {income}")
        print(f"Expense: {expense}")
        print(f"Balance: {balance}")

def main():
    tracker = ExpenseTracker()
    while True:
        print("\nExpense Tracker Menu:")
        print("1. Add transaction")
        print("2. Delete transaction")
        print("3. View transactions")
        print("4. Calculate balance")
        print("5. Exit")
        choice = input("Enter your choice: ")
        if choice == "1":
            tracker.add_transaction()
        elif choice == "2":
            tracker.delete_transaction()
        elif choice == "3":
            tracker.view_transactions()
        elif choice == "4":
            tracker.calculate_balance()
        elif choice == "5":
            print("Exiting program.")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
