## Exercise
Write a program that takes two numbers as input and returns their remainder.

## Reference Solution
```python
# Function to calculate remainder
def calculate_remainder(dividend, divisor):
    if divisor == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return dividend % divisor

# Main function
def main():
    try:
        # Take input from user
        dividend = float(input("Enter the dividend: "))
        divisor = float(input("Enter the divisor: "))

        # Calculate remainder
        remainder = calculate_remainder(dividend, divisor)

        # Print remainder
        print(f"The remainder of {dividend} divided by {divisor} is {remainder}")
    except ZeroDivisionError as e:
        print(f"Error: {e}")

# Run the program
if __name__ == "__main__":
    main()
```
