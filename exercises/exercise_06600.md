## Exercise
Write a program to find the longest substring without repeating characters in a given string.

## Reference Solution
```python
def longest_substring(s: str) -> str:
    """
    Finds the longest substring without repeating characters in a given string.

    Args:
        s (str): The input string.

    Returns:
        str: The longest substring without repeating characters.
    """
    if not s:
        return ""

    left = 0
    max_length = 0
    max_substring = ""
    char_set = set()

    for right in range(len(s)):
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1
        char_set.add(s[right])
        if right - left + 1 > max_length:
            max_length = right - left + 1
            max_substring = s[left:right + 1]

    return max_substring


def main():
    # Example usage:
    input_str = "abcabcbb"
    longest = longest_substring(input_str)
    print(f"Longest substring without repeating characters: {longest}")


if __name__ == "__main__":
    main()
```
