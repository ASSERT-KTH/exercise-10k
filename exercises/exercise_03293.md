## Exercise
Write a program that converts Roman numerals to integers using a for loop.

## Reference Solution
```python
def roman_to_int(roman_numeral):
    """
    Converts a Roman numeral to an integer.

    Args:
        roman_numeral (str): The Roman numeral to convert.

    Returns:
        int: The integer equivalent of the Roman numeral.

    Raises:
        ValueError: If the input is not a valid Roman numeral.
    """

    # Dictionary to map Roman numerals to their integer values
    roman_numerals = {
        'I': 1,
        'V': 5,
        'X': 10,
        'L': 50,
        'C': 100,
        'D': 500,
        'M': 1000
    }

    # Initialize the result variable to 0
    result = 0

    # Iterate over the characters in the Roman numeral
    for i in range(len(roman_numeral)):
        # Check if the current numeral is valid
        if roman_numeral[i] not in roman_numerals:
            raise ValueError("Invalid Roman numeral")

        # If this is not the last numeral and the next one is greater,
        # subtract the current numeral's value
        if i < len(roman_numeral) - 1 and roman_numerals[roman_numeral[i]] < roman_numerals[roman_numeral[i + 1]]:
            result -= roman_numerals[roman_numeral[i]]
        # Otherwise, add the current numeral's value
        else:
            result += roman_numerals[roman_numeral[i]]

    return result


# Example usage:
if __name__ == "__main__":
    print(roman_to_int("III"))  # Outputs: 3
    print(roman_to_int("IV"))  # Outputs: 4
    print(roman_to_int("IX"))  # Outputs: 9
    print(roman_to_int("LVIII"))  # Outputs: 58
    print(roman_to_int("MCMXCIV"))  # Outputs: 1994
```
