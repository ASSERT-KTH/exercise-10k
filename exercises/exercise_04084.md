## Exercise
Write two functions, one that doubles a number and another that adds 1 to the result, and then compose them

## Reference Solution
```python
def double_number(n: int) -> int:
    """
    Doubles the input number.

    Args:
        n (int): The number to double.

    Returns:
        int: The doubled number.
    """
    return n * 2


def add_one(n: int) -> int:
    """
    Adds 1 to the input number.

    Args:
        n (int): The number to increment.

    Returns:
        int: The incremented number.
    """
    return n + 1


def double_and_add_one(n: int) -> int:
    """
    Composes double_number and add_one functions.

    Args:
        n (int): The input number.

    Returns:
        int: The result of doubling the number and adding 1.
    """
    return add_one(double_number(n))


# Example usage:
if __name__ == "__main__":
    num = 5
    result = double_and_add_one(num)
    print(f"Result of doubling {num} and adding 1: {result}")
```
