## Exercise
Create a program that uses a while loop to calculate the miles per hour of a car given the distance traveled and the time taken.

## Reference Solution
```python
def calculate_mph(distance, time):
    """
    Calculate miles per hour (mph) given distance and time.

    Args:
        distance (float): Distance traveled in miles.
        time (float): Time taken in hours.

    Returns:
        float: Miles per hour (mph).
    """
    if time == 0:
        raise ValueError("Time cannot be zero.")
    return distance / time

def main():
    print("Miles Per Hour Calculator")
    print("-------------------------")

    while True:
        try:
            distance = float(input("Enter distance traveled (miles): "))
            time = float(input("Enter time taken (hours): "))

            mph = calculate_mph(distance, time)
            print(f"Miles per hour: {mph:.2f} mph")

            cont = input("Do you want to calculate again? (y/n): ")
            if cont.lower() != 'y':
                break
        except ValueError as e:
            print(f"Error: {e}")

if __name__ == "__main__":
    main()
```
