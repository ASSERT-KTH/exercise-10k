Create a function that calculates the solutions to a quadratic equation using the quadratic formula. The function should take the coefficients of the quadratic equation as arguments and return the solutions.

import math

def calculate_quadratic_solutions(a, b, c):
    """
    This function calculates the solutions to a quadratic equation using the quadratic formula.
    
    Parameters:
    a (float): The coefficient of the squared term in the quadratic equation.
    b (float): The coefficient of the linear term in the quadratic equation.
    c (float): The constant term in the quadratic equation.
    
    Returns:
    tuple: A tuple containing the two solutions to the quadratic equation.
    """

    # First, we need to calculate the determinant (b^2 - 4ac) to determine the nature of the solutions
    determinant = b**2 - 4*a*c
    
    # If the determinant is negative, the solutions will be complex numbers
    if determinant < 0:
        # Calculate the real part of the solutions
        real_part = -b / (2*a)
        
        # Calculate the imaginary part of the solutions
        imaginary_part = math.sqrt(-determinant) / (2*a)
        
        # Return the complex solutions
        return complex(real_part, imaginary_part), complex(real_part, -imaginary_part)
    
    # If the determinant is zero, there will be one repeated real solution
    elif determinant == 0:
        # Calculate the repeated real solution
        solution = -b / (2*a)
        
        # Return the repeated real solution
        return solution, solution
    
    # If the determinant is positive, there will be two distinct real solutions
    else:
        # Calculate the first real solution
        solution1 = (-b + math.sqrt(determinant)) / (2*a)
        
        # Calculate the second real solution
        solution2 = (-b - math.sqrt(determinant)) / (2*a)
        
        # Return the two distinct real solutions
        return solution1, solution2
