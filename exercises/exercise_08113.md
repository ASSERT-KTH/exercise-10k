## Exercise
Write a program that takes a string as input and returns the average length of the words in the string.

## Reference Solution
```python
def calculate_average_word_length(input_string):
    """
    Calculate the average length of words in a string.

    Args:
        input_string (str): The input string.

    Returns:
        float: The average length of the words in the string.
    """
    # Remove leading and trailing whitespace
    input_string = input_string.strip()
    
    # Split the string into words
    words = input_string.split()
    
    # Calculate the total length of all words
    total_length = sum(len(word) for word in words)
    
    # Calculate the average length
    average_length = total_length / len(words) if words else 0
    
    return average_length

def main():
    # Example usage
    input_string = "This is an example string"
    average_length = calculate_average_word_length(input_string)
    print(f"The average length of the words in '{input_string}' is {average_length}")

if __name__ == "__main__":
    main()
```
