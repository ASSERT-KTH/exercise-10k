## Exercise
Write a program that checks if a given variable is mutable.

## Reference Solution
```python
import typing

def is_mutable(var: typing.Any) -> bool:
    """
    Checks if a given variable is mutable.

    Args:
    var (typing.Any): The variable to check.

    Returns:
    bool: True if the variable is mutable, False otherwise.
    """
    # List of immutable types
    immutable_types = [int, float, str, tuple, complex, bytes, frozenset]

    # Check if the variable is of an immutable type
    if type(var) in immutable_types:
        return False
    else:
        return True

# Example usage:
if __name__ == "__main__":
    # Test with a mutable variable (list)
    mutable_var = [1, 2, 3]
    print(f"Is {mutable_var} mutable? {is_mutable(mutable_var)}")

    # Test with an immutable variable (tuple)
    immutable_var = (1, 2, 3)
    print(f"Is {immutable_var} mutable? {is_mutable(immutable_var)}")

    # Test with an immutable variable (integer)
    immutable_var = 5
    print(f"Is {immutable_var} mutable? {is_mutable(immutable_var)}")
```
