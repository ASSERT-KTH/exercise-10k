## Exercise
Write a function to split a string into substrings based on a delimiter.

## Reference Solution
```python
def split_string(input_string, delimiter):
    """
    Splits a string into substrings based on a delimiter.

    Args:
        input_string (str): The input string to be split.
        delimiter (str): The delimiter used to split the string.

    Returns:
        list: A list of substrings.
    """
    return input_string.split(delimiter)

# Example usage:
if __name__ == "__main__":
    input_str = "apple,banana,cherry"
    delimiter = ","
    substrings = split_string(input_str, delimiter)
    print(substrings)  # Output: ['apple', 'banana', 'cherry']
```

```python
import re

def split_string_regex(input_string, delimiter):
    """
    Splits a string into substrings based on a regular expression.

    Args:
        input_string (str): The input string to be split.
        delimiter (str): The regular expression used to split the string.

    Returns:
        list: A list of substrings.
    """
    return re.split(delimiter, input_string)

# Example usage:
if __name__ == "__main__":
    input_str = "apple,banana,cherry"
    delimiter = ","
    substrings = split_string_regex(input_str, delimiter)
    print(substrings)  # Output: ['apple', 'banana', 'cherry']
```
