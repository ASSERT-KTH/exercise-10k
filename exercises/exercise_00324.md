## Exercise
Write a program to demonstrate the type conversion of variables from one data type to another.

## Reference Solution
```python
# Type Conversion Program

def main():
    # Integer to Float
    int_val = 10
    float_val = float(int_val)
    print(f"Integer to Float: {int_val} -> {float_val}")

    # Float to Integer
    float_val = 10.5
    int_val = int(float_val)
    print(f"Float to Integer: {float_val} -> {int_val}")

    # String to Integer
    str_val = "10"
    int_val = int(str_val)
    print(f"String to Integer: {str_val} -> {int_val}")

    # Integer to String
    int_val = 10
    str_val = str(int_val)
    print(f"Integer to String: {int_val} -> {str_val}")

    # Float to String
    float_val = 10.5
    str_val = str(float_val)
    print(f"Float to String: {float_val} -> {str_val}")

    # Boolean to Integer
    bool_val = True
    int_val = int(bool_val)
    print(f"Boolean to Integer: {bool_val} -> {int_val}")

    # Boolean to String
    bool_val = False
    str_val = str(bool_val)
    print(f"Boolean to String: {bool_val} -> {str_val}")

if __name__ == "__main__":
    main()
```
