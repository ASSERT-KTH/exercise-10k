## Exercise
Develop a program that calculates the tax amount based on the income and tax rate provided by the user.

## Reference Solution
```python
def calculate_tax(income, tax_rate):
    """
    Calculates the tax amount based on the income and tax rate.

    Args:
        income (float): The income amount.
        tax_rate (float): The tax rate as a decimal.

    Returns:
        float: The calculated tax amount.
    """
    tax_amount = income * tax_rate
    return tax_amount

def main():
    print("Tax Calculator Program")
    print("------------------------")

    # Get user input
    income = float(input("Enter your income: $"))
    tax_rate = float(input("Enter the tax rate (in %): "))

    # Convert tax rate from percentage to decimal
    tax_rate = tax_rate / 100

    # Calculate tax amount
    tax_amount = calculate_tax(income, tax_rate)

    # Display results
    print(f"Your income: ${income:.2f}")
    print(f"Tax rate: {tax_rate * 100:.2f}%")
    print(f"Tax amount: ${tax_amount:.2f}")
    print(f"Net income: ${income - tax_amount:.2f}")

if __name__ == "__main__":
    main()
```
