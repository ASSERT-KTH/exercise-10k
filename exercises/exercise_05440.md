## Exercise
Write a program that administers a quiz with multiple-choice questions. If the user chooses to take the quiz, display the questions and possible answers. If the user chooses to view the score, calculate the number of correct answers and display the score.

## Reference Solution
```python
# Quiz Program

# Define the quiz questions and answers
quiz_questions = {
    "What is the capital of France?": {
        "A": "Paris",
        "B": "London",
        "C": "Berlin",
        "D": "Rome",
        "correct": "A"
    },
    "What is the largest planet in our solar system?": {
        "A": "Earth",
        "B": "Saturn",
        "C": "Jupiter",
        "D": "Uranus",
        "correct": "C"
    },
    "What is the chemical symbol for gold?": {
        "A": "Ag",
        "B": "Au",
        "C": "Hg",
        "D": "Pb",
        "correct": "B"
    }
}

def take_quiz():
    """
    Administer the quiz to the user.
    """
    score = 0
    user_answers = {}

    # Ask the user each question and store their answer
    for question, answers in quiz_questions.items():
        print(question)
        for option, answer in answers.items():
            if option != "correct":
                print(f"{option}: {answer}")
        user_answer = input("Choose your answer (A, B, C, D): ")
        user_answers[question] = user_answer

    # Calculate the score
    for question, answers in quiz_questions.items():
        if user_answers[question] == answers["correct"]:
            score += 1

    # Display the score
    print(f"\nYou scored {score} out of {len(quiz_questions)}")

def view_score():
    """
    View the score of the quiz.
    """
    # This function is not applicable in this case because the quiz is not stored anywhere
    # We would need to store the quiz results in a file or database to view the score later
    print("No score to view.")

def main():
    while True:
        print("\n1. Take the quiz")
        print("2. View score")
        print("3. Quit")
        choice = input("Choose an option: ")

        if choice == "1":
            take_quiz()
        elif choice == "2":
            view_score()
        elif choice == "3":
            break
        else:
            print("Invalid choice. Please choose again.")

if __name__ == "__main__":
    main()
```
