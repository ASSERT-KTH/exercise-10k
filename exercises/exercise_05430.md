## Exercise
Write a program that stores and retrieves student information. If the user chooses to add a student, prompt for name, age, and grade. If the user chooses to retrieve a student's information, prompt for name and display the corresponding information.

## Reference Solution
```python
class Student:
    """Represents a student with name, age, and grade."""
    def __init__(self, name, age, grade):
        self.name = name
        self.age = age
        self.grade = grade

    def __str__(self):
        return f"Name: {self.name}, Age: {self.age}, Grade: {self.grade}"


class StudentDatabase:
    """Manages a collection of students."""
    def __init__(self):
        self.students = {}

    def add_student(self, name, age, grade):
        """Adds a new student to the database."""
        self.students[name] = Student(name, age, grade)
        print(f"Student '{name}' added successfully.")

    def get_student(self, name):
        """Retrieves a student's information from the database."""
        return self.students.get(name)

    def display_student(self, name):
        """Displays a student's information."""
        student = self.get_student(name)
        if student:
            print(student)
        else:
            print(f"Student '{name}' not found.")


def main():
    database = StudentDatabase()

    while True:
        print("\nStudent Information System")
        print("1. Add a student")
        print("2. Retrieve a student's information")
        print("3. Exit")

        choice = input("Choose an option: ")

        if choice == "1":
            name = input("Enter student's name: ")
            age = int(input("Enter student's age: "))
            grade = input("Enter student's grade: ")
            database.add_student(name, age, grade)
        elif choice == "2":
            name = input("Enter student's name: ")
            database.display_student(name)
        elif choice == "3":
            print("Exiting the system.")
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
