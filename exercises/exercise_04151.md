## Exercise
Write a function to use the pop() method to remove and return a value from a dictionary.

## Reference Solution
```python
def remove_and_return(dictionary, key, default=None):
    """
    Removes and returns the value associated with the given key from the dictionary.

    Args:
        dictionary (dict): The dictionary to remove the value from.
        key: The key of the value to remove.
        default: The default value to return if the key is not found. Defaults to None.

    Returns:
        The removed value or the default value if the key is not found.
    """
    return dictionary.pop(key, default)

# Example usage:
my_dict = {"name": "John", "age": 30, "city": "New York"}

# Remove and return a value
removed_value = remove_and_return(my_dict, "age")
print(f"Removed value: {removed_value}")
print(f"Dictionary after removal: {my_dict}")

# Remove and return a non-existent value with default
removed_value = remove_and_return(my_dict, "country", "Not found")
print(f"Removed value: {removed_value}")
print(f"Dictionary after removal: {my_dict}")
```
