## Exercise
Write a program to reverse the order of words in a given string

## Reference Solution
```python
def reverse_word_order(input_string):
    """
    Reverses the order of words in a given string.

    Args:
        input_string (str): The input string.

    Returns:
        str: The input string with the order of words reversed.
    """
    # Split the input string into a list of words
    words = input_string.split()

    # Reverse the list of words
    reversed_words = words[::-1]

    # Join the reversed list of words into a string
    reversed_string = ' '.join(reversed_words)

    return reversed_string

# Example usage
def main():
    input_string = "Hello World this is Python"
    print("Input String:", input_string)
    print("Reversed String:", reverse_word_order(input_string))

if __name__ == "__main__":
    main()
```
