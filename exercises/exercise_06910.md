## Exercise
Write a program that uses a loop to print the numbers from 1 to 10, but only prints the numbers that are divisible by 3. Use a for loop or a while loop to solve this problem.

## Reference Solution
```python
# Using a for loop
def print_numbers_divisible_by_3_for_loop():
    """Prints numbers from 1 to 10 that are divisible by 3 using a for loop."""
    for i in range(1, 11):
        if i % 3 == 0:
            print(i)

# Using a while loop
def print_numbers_divisible_by_3_while_loop():
    """Prints numbers from 1 to 10 that are divisible by 3 using a while loop."""
    i = 1
    while i <= 10:
        if i % 3 == 0:
            print(i)
        i += 1

# Example usage
if __name__ == "__main__":
    print("Numbers divisible by 3 using a for loop:")
    print_numbers_divisible_by_3_for_loop()
    print("\nNumbers divisible by 3 using a while loop:")
    print_numbers_divisible_by_3_while_loop()
```
