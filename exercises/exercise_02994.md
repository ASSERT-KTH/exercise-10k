## Exercise
Write a program that finds the longest subsequence of a given list of numbers that is in increasing order.

## Reference Solution
```python
def longest_increasing_subsequence(nums):
    """
    Finds the longest increasing subsequence of a given list of numbers.

    Args:
    - nums (list): A list of integers.

    Returns:
    - lis (list): The longest increasing subsequence.
    """
    if not nums:
        return []

    # Initialize lists to store lengths and subsequences
    lengths = [1] * len(nums)
    subsequences = [[num] for num in nums]

    # Compute lengths and subsequences
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[i] > nums[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                subsequences[i] = subsequences[j] + [nums[i]]

    # Find the longest subsequence
    max_length = max(lengths)
    longest_subsequence = [sub for sub in subsequences if len(sub) == max_length]

    return longest_subsequence[0]

# Example usage
nums = [10, 22, 9, 33, 21, 50, 41, 60, 80]
print("Original list:", nums)
print("Longest increasing subsequence:", longest_increasing_subsequence(nums))
```
