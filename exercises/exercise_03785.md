## Exercise
Create a function that finds the missing number in a sequence of consecutive integers.

## Reference Solution
```python
def find_missing_number(sequence):
    """
    Finds the missing number in a sequence of consecutive integers.

    Args:
        sequence (list): A list of consecutive integers with one number missing.

    Returns:
        int: The missing number in the sequence.

    Raises:
        ValueError: If the input sequence is not a list or is empty.
    """
    if not isinstance(sequence, list) or len(sequence) == 0:
        raise ValueError("Input sequence must be a non-empty list")

    # Calculate the sum of the sequence if it were complete
    n = len(sequence) + 1
    total_sum = n * (n + 1) // 2

    # Calculate the sum of the given sequence
    sequence_sum = sum(sequence)

    # The missing number is the difference between the total sum and the sequence sum
    missing_number = total_sum - sequence_sum

    return missing_number

# Example usage
if __name__ == "__main__":
    sequence = [1, 2, 3, 5]
    missing_number = find_missing_number(sequence)
    print("Missing number:", missing_number)
```
