Write a program that uses nested conditional statements to make decisions based on multiple conditions.

```python
def check_eligibility(age, gpa, test_score):
    """
    Checks if a student is eligible to join a university based on their age, GPA, and test scores.

    Args:
        age (int): The student's age.
        gpa (float): The student's GPA.
        test_score (int): The student's test score.

    Returns:
        bool: True if the student is eligible, False otherwise.
    """
    if age >= 17:  # Check if the student is at least 17 years old
        if gpa >= 3.0:  # Check if the student's GPA is at least 3.0
            if test_score >= 80:  # Check if the student's test score is at least 80
                return True  # The student is eligible
            else:
                print("Test score is too low. Eligibility denied.")
        else:
            print("GPA is too low. Eligibility denied.")
    else:
        print("Age is too low. Eligibility denied.")
    return False  # The student is not eligible


def main():
    # Example usage
    age = 20
    gpa = 3.5
    test_score = 90
    if check_eligibility(age, gpa, test_score):
        print("Congratulations, you are eligible to join the university!")
    else:
        print("Sorry, you are not eligible to join the university.")


if __name__ == "__main__":
    main()
```
