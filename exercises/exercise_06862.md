## Exercise
Write a program that takes an array of integers and an index as input, and returns the element at the specified index. If the index is out of range, return an error message.

## Reference Solution
```python
#include <iostream>
using namespace std;

int get(int arr[], int index, int size) {
    if (index >= 0 && index < size) {
        return arr[index];
    } else {
        cout << "Index out of range" << endl;
        return -1; // or any other value to indicate an error
    }
}

int main() {
    int arr[] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
    int size = sizeof(arr) / sizeof(arr[0]);
    int index = 3; // for example

    cout << get(arr, index, size) << endl;

    return 0;
}
```
