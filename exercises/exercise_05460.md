## Exercise
Write a program that generates a random word and lets the user guess it by inputting individual letters

## Reference Solution
```python
import random

# List of words to choose from
word_list = ['apple', 'banana', 'cherry', 'date', 'elderberry']

def word_guessing_game():
    # Choose a random word from the list
    word_to_guess = random.choice(word_list)
    word_length = len(word_to_guess)

    # Create a list to store the guessed letters
    guessed_word = ['_'] * word_length
    guessed_letters = []
    tries = 6

    print("Welcome to the Word Guessing Game!")
    print(f"You have {tries} tries to guess the word.")

    while '_' in guessed_word and tries > 0:
        print(' '.join(guessed_word))
        guess = input("Guess a letter: ").lower()

        if len(guess) != 1:
            print("Please guess one letter at a time.")
            continue

        if guess in guessed_letters:
            print("You already guessed this letter. Try another one.")
            continue

        guessed_letters.append(guess)

        if guess not in word_to_guess:
            tries -= 1
            print(f"Incorrect guess. You have {tries} tries left.")
        else:
            for i, letter in enumerate(word_to_guess):
                if letter == guess:
                    guessed_word[i] = guess

    if '_' not in guessed_word:
        print(' '.join(guessed_word))
        print("Congratulations! You guessed the word.")
    else:
        print(f"You didn't guess the word. The word was {word_to_guess}.")

if __name__ == "__main__":
    word_guessing_game()
```
