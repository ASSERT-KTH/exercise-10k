## Exercise
Create a program that takes in three sides of a triangle and checks if they form a valid triangle using conditional statements.

## Reference Solution
```python
def is_valid_triangle(a, b, c):
    """
    Check if three sides form a valid triangle.

    A triangle is valid if the sum of the lengths of any two sides is greater than the length of the third side.

    Args:
        a (float): The length of the first side.
        b (float): The length of the second side.
        c (float): The length of the third side.

    Returns:
        bool: True if the sides form a valid triangle, False otherwise.
    """

    # Check if all sides are positive
    if a <= 0 or b <= 0 or c <= 0:
        return False

    # Check if the sum of the lengths of any two sides is greater than the length of the third side
    if (a + b > c) and (a + c > b) and (b + c > a):
        return True
    else:
        return False


def main():
    # Example usage:
    a = float(input("Enter the length of the first side: "))
    b = float(input("Enter the length of the second side: "))
    c = float(input("Enter the length of the third side: "))

    if is_valid_triangle(a, b, c):
        print("The sides form a valid triangle.")
    else:
        print("The sides do not form a valid triangle.")


if __name__ == "__main__":
    main()
```
