## Exercise
Create a function that uses debugging techniques such as print statements or a debugger to diagnose issues.

## Reference Solution
```python
def debug_function(input_list):
    """
    This function takes a list as an input, checks if all elements are integers, 
    and returns the sum of all integers in the list. It uses print statements 
    to diagnose issues during execution.

    Args:
        input_list (list): A list containing elements to be summed.

    Returns:
        int: The sum of all integers in the input list.
    """

    # First, we check if the input is a list. If not, we print an error message.
    if not isinstance(input_list, list):
        print("Error: Input is not a list.")
        return None

    # Initialize the sum variable to 0.
    total_sum = 0

    # Iterate over each element in the input list.
    for i, element in enumerate(input_list):
        # Print the current element being processed for debugging purposes.
        print(f"Processing element {i+1}: {element}")

        # Check if the current element is an integer.
        if not isinstance(element, int):
            # If not, print an error message and continue to the next element.
            print(f"Error: Element {i+1} is not an integer.")
            continue

        # If the element is an integer, add it to the total sum.
        total_sum += element
        # Print the updated total sum for debugging purposes.
        print(f"Updated total sum: {total_sum}")

    # After processing all elements, print the final total sum.
    print(f"Final total sum: {total_sum}")

    # Return the total sum.
    return total_sum

# Example usage:
print(debug_function([1, 2, 3, 4, 5]))  # Output: 15
print(debug_function([1, 'a', 3, 4, 5]))  # Output: 13
```
