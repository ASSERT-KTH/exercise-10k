#include <iostream>
#include <vector>

void max_pair(std::vector<int> elements) {
    int maximum_index = -1;
    int maximum_sum = -1;  // Initialize with -1 instead of float('-inf')

    for (int i = 0; i < elements.size() - 1; i++) {
        int pair_sum = elements[i] + elements[i + 1];
        if (pair_sum > maximum_sum) {  // Change to > for maximum sum
            maximum_index = i;
            maximum_sum = pair_sum;
        }
    }

    if (maximum_index != -1) {
        std::cout << "Maximum pair sum: " << maximum_sum << " at index " << maximum_index << std::endl;
    } else {
        std::cout << "List is too short" << std::endl;
    }
}

int main() {
    std::vector<int> elements = {1, 2, 3, 4, 5};
    max_pair(elements);

    return 0;
}
