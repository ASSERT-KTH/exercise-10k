## Exercise
Develop a program that takes user input for income and expenses, and calculates the total savings.

## Reference Solution
```python
def get_income():
    """Get user income"""
    while True:
        try:
            income = float(input("Enter your monthly income: $"))
            if income < 0:
                print("Income cannot be negative. Please try again.")
            else:
                return income
        except ValueError:
            print("Invalid input. Please enter a valid number.")

def get_expenses():
    """Get user expenses"""
    expenses = {}
    while True:
        expense_name = input("Enter expense name (or 'done' to finish): ")
        if expense_name.lower() == 'done':
            break
        while True:
            try:
                expense_amount = float(input(f"Enter {expense_name} expense amount: $"))
                if expense_amount < 0:
                    print("Expense amount cannot be negative. Please try again.")
                else:
                    expenses[expense_name] = expense_amount
                    break
            except ValueError:
                print("Invalid input. Please enter a valid number.")
    return expenses

def calculate_savings(income, expenses):
    """Calculate total savings"""
    total_expenses = sum(expenses.values())
    savings = income - total_expenses
    return savings

def main():
    print("Savings Calculator Program")
    print("---------------------------")
    income = get_income()
    expenses = get_expenses()
    savings = calculate_savings(income, expenses)
    print("\nSummary:")
    print(f"Income: ${income:.2f}")
    print("Expenses:")
    for expense, amount in expenses.items():
        print(f"  {expense}: ${amount:.2f}")
    print(f"Total Expenses: ${sum(expenses.values()):.2f}")
    print(f"Savings: ${savings:.2f}")

if __name__ == "__main__":
    main()
```
