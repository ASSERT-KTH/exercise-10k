## Exercise
Write a program that formats a given string using a given format specifier

## Reference Solution
```python
def format_string(format_specifier, *args):
    """
    Formats a given string using a given format specifier.

    Args:
        format_specifier (str): The format specifier string.
        *args: The arguments to be formatted.

    Returns:
        str: The formatted string.
    """
    formatted_string = ""
    arg_index = 0

    for i, char in enumerate(format_specifier):
        if char == "%":
            if i + 1 < len(format_specifier):
                specifier = format_specifier[i + 1]
                if specifier == "s":
                    formatted_string += str(args[arg_index])
                elif specifier == "d":
                    formatted_string += str(int(args[arg_index]))
                elif specifier == "f":
                    formatted_string += str(float(args[arg_index]))
                elif specifier == "x":
                    formatted_string += hex(int(args[arg_index])).replace("0x", "")
                elif specifier == "X":
                    formatted_string += hex(int(args[arg_index])).replace("0x", "").upper()
                arg_index += 1
            else:
                raise ValueError("Invalid format specifier")
        else:
            formatted_string += char

    return formatted_string

# Example usage
if __name__ == "__main__":
    print(format_string("Hello, %s!", "World"))  # Output: Hello, World!
    print(format_string("The answer is %d.", 42))  # Output: The answer is 42.
    print(format_string("The answer is %f.", 42.0))  # Output: The answer is 42.0.
    print(format_string("The answer is %x.", 42))  # Output: The answer is 2a.
    print(format_string("The answer is %X.", 42))  # Output: The answer is 2A.
```
