## Exercise
Write a program to perform basic float operations (addition, subtraction, multiplication, division).

## Reference Solution
```python
def add_floats(num1, num2):
    """Return the sum of two floating-point numbers."""
    return num1 + num2

def subtract_floats(num1, num2):
    """Return the difference of two floating-point numbers."""
    return num1 - num2

def multiply_floats(num1, num2):
    """Return the product of two floating-point numbers."""
    return num1 * num2

def divide_floats(num1, num2):
    """Return the quotient of two floating-point numbers."""
    if num2 == 0:
        raise ZeroDivisionError("Cannot divide by zero")
    return num1 / num2

def main():
    print("Float Operations Program")
    print("-------------------------")

    num1 = float(input("Enter the first number: "))
    num2 = float(input("Enter the second number: "))

    print(f"Addition: {num1} + {num2} = {add_floats(num1, num2)}")
    print(f"Subtraction: {num1} - {num2} = {subtract_floats(num1, num2)}")
    print(f"Multiplication: {num1} * {num2} = {multiply_floats(num1, num2)}")

    try:
        print(f"Division: {num1} / {num2} = {divide_floats(num1, num2)}")
    except ZeroDivisionError as e:
        print(str(e))

if __name__ == "__main__":
    main()
```
