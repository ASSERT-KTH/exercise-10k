## Exercise
Create a program that takes in a student's age and score, and checks if they are eligible for a scholarship using conditional statements.

## Reference Solution
```python
def check_scholarship_eligibility(age, score):
    """
    Checks if a student is eligible for a scholarship based on their age and score.

    Args:
        age (int): The age of the student.
        score (float): The score of the student.

    Returns:
        bool: True if the student is eligible for a scholarship, False otherwise.
    """

    # Define the eligibility criteria
    min_age = 16
    max_age = 25
    min_score = 80.0

    # Check if the student's age is within the eligible range
    if age < min_age or age > max_age:
        return False  # Student's age is not within the eligible range

    # Check if the student's score is above the minimum required score
    if score < min_score:
        return False  # Student's score is below the minimum required score

    # If both conditions are met, the student is eligible for a scholarship
    return True


def main():
    # Get the student's age and score from the user
    age = int(input("Enter the student's age: "))
    score = float(input("Enter the student's score (out of 100): "))

    # Check if the student is eligible for a scholarship
    if check_scholarship_eligibility(age, score):
        print("The student is eligible for a scholarship.")
    else:
        print("The student is not eligible for a scholarship.")


if __name__ == "__main__":
    main()
```
