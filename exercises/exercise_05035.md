## Exercise
Write a program that simulates a bank account manager using for loops and while loops.

## Reference Solution
```python
class BankAccount:
    def __init__(self, account_number, account_name, balance=0):
        self.account_number = account_number
        self.account_name = account_name
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        print(f"Deposited ${amount}. Current balance: ${self.balance}")

    def withdraw(self, amount):
        if amount > self.balance:
            print("Insufficient funds.")
        else:
            self.balance -= amount
            print(f"Withdrew ${amount}. Current balance: ${self.balance}")

    def display_details(self):
        print(f"Account Number: {self.account_number}")
        print(f"Account Name: {self.account_name}")
        print(f"Current Balance: ${self.balance}")


def main():
    accounts = []

    while True:
        print("\nBank Account Manager")
        print("1. Create Account")
        print("2. Manage Account")
        print("3. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            account_number = input("Enter account number: ")
            account_name = input("Enter account name: ")
            balance = float(input("Enter initial balance: "))

            new_account = BankAccount(account_number, account_name, balance)
            accounts.append(new_account)

            print(f"Account created successfully for {account_name}.")

        elif choice == "2":
            if not accounts:
                print("No accounts created.")
                continue

            print("\nAvailable Accounts:")
            for i, account in enumerate(accounts, start=1):
                print(f"{i}. Account Number: {account.account_number}, Account Name: {account.account_name}")

            account_choice = int(input("Enter the account number you want to manage: "))

            if account_choice > len(accounts) or account_choice < 1:
                print("Invalid account choice.")
                continue

            chosen_account = accounts[account_choice - 1]

            while True:
                print("\nAccount Manager")
                print("1. Deposit")
                print("2. Withdraw")
                print("3. Display Details")
                print("4. Back")

                account_manager_choice = input("Enter your choice: ")

                if account_manager_choice == "1":
                    amount = float(input("Enter amount to deposit: "))
                    chosen_account.deposit(amount)
                elif account_manager_choice == "2":
                    amount = float(input("Enter amount to withdraw: "))
                    chosen_account.withdraw(amount)
                elif account_manager_choice == "3":
                    chosen_account.display_details()
                elif account_manager_choice == "4":
                    break
                else:
                    print("Invalid choice. Please choose again.")

        elif choice == "3":
            print("Exiting the program.")
            break
        else:
            print("Invalid choice. Please choose again.")


if __name__ == "__main__":
    main()
```
