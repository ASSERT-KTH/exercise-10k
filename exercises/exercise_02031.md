## Exercise
Create a program that determines a bank account type based on the account balance and customer status. The program should use if/else statements to determine the account type.

## Reference Solution
```python
def determine_account_type(balance, customer_status):
    """
    Determine the bank account type based on the account balance and customer status.

    Args:
        balance (float): The account balance.
        customer_status (str): The customer status. Can be 'premium', 'gold', 'silver', or 'basic'.

    Returns:
        str: The account type.
    """

    # Define the balance thresholds for each account type
    premium_threshold = 100000
    gold_threshold = 50000
    silver_threshold = 10000

    # Check if the customer is premium
    if customer_status == 'premium':
        # If the balance is above the premium threshold, it's a premium account
        if balance >= premium_threshold:
            return 'Premium Account'
        # If the balance is above the gold threshold, it's a gold account
        elif balance >= gold_threshold:
            return 'Gold Account'
        # If the balance is above the silver threshold, it's a silver account
        elif balance >= silver_threshold:
            return 'Silver Account'
        # Otherwise, it's a basic account
        else:
            return 'Basic Account'

    # Check if the customer is gold
    elif customer_status == 'gold':
        # If the balance is above the gold threshold, it's a gold account
        if balance >= gold_threshold:
            return 'Gold Account'
        # If the balance is above the silver threshold, it's a silver account
        elif balance >= silver_threshold:
            return 'Silver Account'
        # Otherwise, it's a basic account
        else:
            return 'Basic Account'

    # Check if the customer is silver
    elif customer_status == 'silver':
        # If the balance is above the silver threshold, it's a silver account
        if balance >= silver_threshold:
            return 'Silver Account'
        # Otherwise, it's a basic account
        else:
            return 'Basic Account'

    # If the customer is basic, it's always a basic account
    else:
        return 'Basic Account'


def main():
    # Test the function
    print(determine_account_type(150000, 'premium'))  # Output: Premium Account
    print(determine_account_type(75000, 'gold'))  # Output: Gold Account
    print(determine_account_type(20000, 'silver'))  # Output: Silver Account
    print(determine_account_type(5000, 'basic'))  # Output: Basic Account


if __name__ == "__main__":
    main()
```
