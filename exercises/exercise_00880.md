## Exercise
Write a program that simulates a library. The program should take in the book title and output the updated book list based on the following conditions: if the user checks out a book, remove the book from the list, if the user returns a book, add the book to the list, if the user checks the book list, output the current book list.

## Reference Solution
```python
class Library:
    def __init__(self):
        # Initialize an empty book list
        self.books = []

    def add_book(self, title):
        """Add a book to the library"""
        self.books.append(title)
        print(f"'{title}' has been added to the library.")

    def remove_book(self, title):
        """Remove a book from the library"""
        if title in self.books:
            self.books.remove(title)
            print(f"'{title}' has been checked out from the library.")
        else:
            print(f"'{title}' is not available in the library.")

    def check_book_list(self):
        """Print the current book list"""
        if not self.books:
            print("The library is currently empty.")
        else:
            print("Current book list:")
            for book in self.books:
                print(book)


def main():
    library = Library()
    library.add_book("Book 1")
    library.add_book("Book 2")
    library.add_book("Book 3")

    while True:
        print("\nLibrary Menu:")
        print("1. Check out a book")
        print("2. Return a book")
        print("3. Check book list")
        print("4. Exit")

        choice = input("Enter your choice: ")

        if choice == "1":
            title = input("Enter the title of the book you want to check out: ")
            library.remove_book(title)
        elif choice == "2":
            title = input("Enter the title of the book you want to return: ")
            library.add_book(title)
        elif choice == "3":
            library.check_book_list()
        elif choice == "4":
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
```
