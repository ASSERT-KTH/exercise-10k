## Exercise
Write a program to calculate the area of a trapezoid given its bases and height.

## Reference Solution
```python
def calculate_trapezoid_area(base1, base2, height):
    """
    Calculate the area of a trapezoid.

    Args:
        base1 (float): The length of the first base.
        base2 (float): The length of the second base.
        height (float): The height of the trapezoid.

    Returns:
        float: The area of the trapezoid.
    """
    if base1 <= 0 or base2 <= 0 or height <= 0:
        raise ValueError("All inputs must be positive numbers.")

    area = 0.5 * (base1 + base2) * height
    return area

def main():
    # Example usage
    base1 = float(input("Enter the length of the first base: "))
    base2 = float(input("Enter the length of the second base: "))
    height = float(input("Enter the height of the trapezoid: "))

    try:
        area = calculate_trapezoid_area(base1, base2, height)
        print(f"The area of the trapezoid is: {area:.2f} square units.")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
