## Exercise
Create a function that reduces a list to a single value by applying a given function.

## Reference Solution
```python
from functools import reduce

def reduce_list(lst, func):
    """
    Reduces a list to a single value by applying a given function.

    Args:
        lst (list): The list to reduce.
        func (function): The function to apply to the list.

    Returns:
        The reduced value.
    """
    if not lst:
        raise ValueError("List is empty")

    return reduce(func, lst)

# Example usage:
numbers = [1, 2, 3, 4, 5]
sum_result = reduce_list(numbers, lambda x, y: x + y)
print(sum_result)  # Output: 15

product_result = reduce_list(numbers, lambda x, y: x * y)
print(product_result)  # Output: 120
```

def custom_reduce(lst, func):
    """
    Reduces a list to a single value by applying a given function.

    Args:
        lst (list): The list to reduce.
        func (function): The function to apply to the list.

    Returns:
        The reduced value.
    """
    if not lst:
        raise ValueError("List is empty")

    result = lst[0]
    for element in lst[1:]:
        result = func(result, element)

    return result

# Example usage:
numbers = [1, 2, 3, 4, 5]
sum_result = custom_reduce(numbers, lambda x, y: x + y)
print(sum_result)  # Output: 15

product_result = custom_reduce(numbers, lambda x, y: x * y)
print(product_result)  # Output: 120
