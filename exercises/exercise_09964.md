## Exercise
Write a program that performs basic set operations such as union, intersection, and difference.

## Reference Solution
```python
class SetOperations:
    def __init__(self, set1, set2):
        """
        Initialize the SetOperations class with two sets.

        Args:
            set1 (set): The first set.
            set2 (set): The second set.
        """
        self.set1 = set1
        self.set2 = set2

    def union(self):
        """
        Calculate the union of the two sets.

        Returns:
            set: The union of the two sets.
        """
        return self.set1.union(self.set2)

    def intersection(self):
        """
        Calculate the intersection of the two sets.

        Returns:
            set: The intersection of the two sets.
        """
        return self.set1.intersection(self.set2)

    def difference(self):
        """
        Calculate the difference of the two sets (set1 - set2).

        Returns:
            set: The difference of the two sets.
        """
        return self.set1.difference(self.set2)

    def symmetric_difference(self):
        """
        Calculate the symmetric difference of the two sets.

        Returns:
            set: The symmetric difference of the two sets.
        """
        return self.set1.symmetric_difference(self.set2)


def main():
    # Define two sets
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}

    # Create a SetOperations object
    set_operations = SetOperations(set1, set2)

    # Perform set operations
    print("Set 1:", set1)
    print("Set 2:", set2)
    print("Union:", set_operations.union())
    print("Intersection:", set_operations.intersection())
    print("Difference (set1 - set2):", set_operations.difference())
    print("Symmetric Difference:", set_operations.symmetric_difference())


if __name__ == "__main__":
    main()
```
