## Exercise
Write a program to handle exceptions and errors using try-except blocks.

## Reference Solution
```python
import logging

# Set up logging
logging.basicConfig(level=logging.ERROR)

def divide_numbers(num1, num2):
    """
    Divide two numbers and handle ZeroDivisionError.
    """
    try:
        result = num1 / num2
        return result
    except ZeroDivisionError:
        logging.error("Cannot divide by zero!")
        return None

def custom_exception_example():
    """
    Raise and handle a custom exception.
    """
    class CustomException(Exception):
        pass

    try:
        raise CustomException("This is a custom exception")
    except CustomException as e:
        logging.error(f"Custom exception: {e}")

def error_logging_example():
    """
    Log errors using the logging module.
    """
    try:
        # Simulate an error
        x = 1 / 0
    except Exception as e:
        logging.error(f"Error occurred: {e}")

def main():
    # Test divide_numbers function
    print("Divide numbers example:")
    result = divide_numbers(10, 2)
    print(f"Result: {result}")
    result = divide_numbers(10, 0)
    print(f"Result: {result}")

    # Test custom exception example
    print("\nCustom exception example:")
    custom_exception_example()

    # Test error logging example
    print("\nError logging example:")
    error_logging_example()

if __name__ == "__main__":
    main()
```
