## Exercise
Write a program that finds the first duplicate in a list of numbers using a for loop.

## Reference Solution
```python
def find_first_duplicate(numbers):
    """
    Finds the first duplicate in a list of numbers.

    Args:
    numbers (list): A list of numbers.

    Returns:
    The first duplicate number in the list, or None if no duplicates are found.
    """
    seen = set()
    for num in numbers:
        if num in seen:
            return num
        seen.add(num)
    return None

# Example usage
def main():
    numbers = [2, 1, 3, 5, 3, 2]
    duplicate = find_first_duplicate(numbers)
    if duplicate is not None:
        print(f"The first duplicate in the list is: {duplicate}")
    else:
        print("No duplicates found in the list.")

if __name__ == "__main__":
    main()
```
