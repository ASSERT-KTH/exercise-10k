## Exercise
Write a program that takes user-provided input and parses it into a usable format

## Reference Solution
```python
import re

def parse_input(input_str):
    """
    Parse user input into a usable format.

    Args:
        input_str (str): User-provided input string.

    Returns:
        dict: Parsed input data.
    """
    # Initialize an empty dictionary to store parsed data
    parsed_data = {}

    # Use regular expression to extract key-value pairs
    pairs = re.findall(r"(\w+)\s*=\s*(\w+)", input_str)

    # Iterate over extracted pairs and store in dictionary
    for key, value in pairs:
        parsed_data[key] = value

    return parsed_data

def main():
    # Get user input
    input_str = input("Enter your input (e.g., name=John age=30): ")

    # Parse user input
    try:
        parsed_data = parse_input(input_str)
        print("Parsed Input Data:")
        for key, value in parsed_data.items():
            print(f"{key}: {value}")
    except Exception as e:
        print(f"Error: {str(e)}")

if __name__ == "__main__":
    main()
```
