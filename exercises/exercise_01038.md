## Exercise
Create a program to find the minimum number of coins needed to make change for a given amount.

## Reference Solution
```python
def min_coins(coins, amount):
    """
    Find the minimum number of coins needed to make change for a given amount.

    Args:
        coins (list): A list of coin denominations.
        amount (int): The amount to make change for.

    Returns:
        int: The minimum number of coins needed. Returns -1 if it's not possible to make change.
    """
    # Create a table to store the minimum number of coins needed for each amount
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # Base case: 0 coins needed to make 0 amount

    # Fill the table using dynamic programming
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] = min(dp[i], dp[i - coin] + 1)

    # If it's not possible to make change, return -1
    if dp[amount] == float('inf'):
        return -1
    else:
        return dp[amount]

# Example usage
coins = [1, 2, 5]
amount = 11
print(min_coins(coins, amount))  # Output: 3 (11 = 5 + 5 + 1)
```
