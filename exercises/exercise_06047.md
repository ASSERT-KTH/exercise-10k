## Exercise
Create a program that takes in a password and checks its validity using conditional statements, considering factors such as length and character type.

## Reference Solution
```python
def check_password(password):
    # Define the conditions for a valid password
    min_length = 8
    max_length = 20
    has_uppercase = any(char.isupper() for char in password)
    has_lowercase = any(char.islower() for char in password)
    has_digit = any(char.isdigit() for char in password)
    has_special_char = any(not char.isalnum() for char in password)

    # Check the conditions
    if len(password) < min_length:
        print("Password should be at least", min_length, "characters long.")
    elif len(password) > max_length:
        print("Password should be no more than", max_length, "characters long.")
    elif not has_uppercase:
        print("Password should have at least one uppercase letter.")
    elif not has_lowercase:
        print("Password should have at least one lowercase letter.")
    elif not has_digit:
        print("Password should have at least one digit.")
    elif not has_special_char:
        print("Password should have at least one special character.")
    else:
        print("Password is valid.")

def main():
    password = input("Enter your password: ")
    check_password(password)

if __name__ == "__main__":
    main()
```
