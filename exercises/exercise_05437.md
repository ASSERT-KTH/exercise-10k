## Exercise
Create a program that performs scientific calculations. If the user chooses to calculate the sine of an angle, use the formula sin(x). If the user chooses to calculate the cosine of an angle, use the formula cos(x).

## Reference Solution
```python
import math

def calculate_sine(angle):
    """
    Calculate the sine of an angle.
    
    Args:
        angle (float): Angle in radians.
    
    Returns:
        float: The sine of the angle.
    """
    return math.sin(angle)

def calculate_cosine(angle):
    """
    Calculate the cosine of an angle.
    
    Args:
        angle (float): Angle in radians.
    
    Returns:
        float: The cosine of the angle.
    """
    return math.cos(angle)

def main():
    print("Scientific Calculator")
    print("1. Calculate Sine")
    print("2. Calculate Cosine")
    print("3. Quit")
    
    while True:
        choice = input("Choose an option: ")
        
        if choice == "1":
            angle = float(input("Enter an angle in degrees: "))
            # Convert angle from degrees to radians
            angle_in_radians = math.radians(angle)
            sine = calculate_sine(angle_in_radians)
            print(f"The sine of {angle} degrees is {sine:.4f}")
        elif choice == "2":
            angle = float(input("Enter an angle in degrees: "))
            # Convert angle from degrees to radians
            angle_in_radians = math.radians(angle)
            cosine = calculate_cosine(angle_in_radians)
            print(f"The cosine of {angle} degrees is {cosine:.4f}")
        elif choice == "3":
            print("Goodbye!")
            break
        else:
            print("Invalid option. Please choose again.")

if __name__ == "__main__":
    main()
```
