## Exercise
Find the maximum contiguous subarray of an array of integers

## Reference Solution
```python
def max_contiguous_subarray(arr):
    """
    Find the maximum contiguous subarray of an array of integers.

    Args:
        arr (list): A list of integers.

    Returns:
        tuple: A tuple containing the maximum contiguous subarray and its sum.
    """
    max_sum = float('-inf')
    max_subarray = []

    current_sum = 0
    current_subarray = []

    for num in arr:
        current_sum += num
        current_subarray.append(num)

        if current_sum > max_sum:
            max_sum = current_sum
            max_subarray = current_subarray[:]

        if current_sum < 0:
            current_sum = 0
            current_subarray = []

    return max_subarray, max_sum

# Example usage:
arr = [-2, -3, 4, -1, -2, 1, 5, -3]
max_subarray, max_sum = max_contiguous_subarray(arr)
print("Maximum Contiguous Subarray:", max_subarray)
print("Maximum Sum:", max_sum)
```
