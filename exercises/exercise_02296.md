## Exercise
Write a program that checks whether a given password meets certain criteria (e.g., length, contains uppercase and lowercase letters, contains digits).

## Reference Solution
```python
import re

def validate_password(password: str) -> bool:
    """
    Validate a password based on certain criteria.

    Args:
    password (str): The password to be validated.

    Returns:
    bool: True if the password is valid, False otherwise.
    """
    # Define the criteria
    criteria = {
        "length": len(password) >= 8,
        "uppercase": any(char.isupper() for char in password),
        "lowercase": any(char.islower() for char in password),
        "digits": any(char.isdigit() for char in password)
    }

    # Check if all criteria are met
    return all(criteria.values())

def validate_password_regex(password: str) -> bool:
    """
    Validate a password based on certain criteria using regular expressions.

    Args:
    password (str): The password to be validated.

    Returns:
    bool: True if the password is valid, False otherwise.
    """
    # Define the regular expression pattern
    pattern = r"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).{8,}$"

    # Check if the password matches the pattern
    return bool(re.match(pattern, password))

def main():
    # Test the functions
    passwords = ["Short1", "LongEnough1A", "longenough1", "LongEnough1A"]

    print("Using validate_password function:")
    for password in passwords:
        print(f"Password: {password}, Valid: {validate_password(password)}")

    print("\nUsing validate_password_regex function:")
    for password in passwords:
        print(f"Password: {password}, Valid: {validate_password_regex(password)}")

if __name__ == "__main__":
    main()
```
