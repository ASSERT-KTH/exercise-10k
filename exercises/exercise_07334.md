## Exercise
Write a program that takes input from the user with SSL encryption and prints out the results.

## Reference Solution
```python
import socket
import ssl

def secure_input(host='localhost', port=12345):
    """
    Establishes a secure SSL connection and receives input from the user.
    
    Args:
        host (str): The hostname or IP address to connect to.
        port (int): The port number to connect to.
    
    Returns:
        str: The user's input.
    """
    # Create an SSL context
    context = ssl.create_default_context(ssl.Purpose.SERVER_AUTH)
    context.load_verify_locations('server.crt')  # Load the server's certificate
    
    # Create a socket object
    client_socket = socket.socket(socket.AF_INET)
    client_socket.settimeout(10)  # Set a timeout of 10 seconds
    
    # Wrap the socket with an SSL context
    ssl_socket = context.wrap_socket(client_socket, server_hostname=host)
    
    # Connect to the server
    ssl_socket.connect((host, port))
    
    # Receive input from the user
    input_data = ssl_socket.recv(1024).decode('utf-8')
    
    # Close the SSL socket
    ssl_socket.close()
    
    return input_data

def main():
    host = 'localhost'
    port = 12345
    
    # Create an SSL server
    context = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
    context.load_cert_chain('server.crt', 'server.key')  # Load the server's certificate and private key
    
    server_socket = socket.socket(socket.AF_INET)
    server_socket.bind((host, port))
    server_socket.listen(1)
    
    print(f"Server listening on {host}:{port}...")
    
    client_socket, address = server_socket.accept()
    ssl_socket = context.wrap_socket(client_socket, server_side=True)
    
    print("Secure connection established.")
    
    # Send a prompt to the client
    ssl_socket.send(b"Enter your input: ")
    
    # Receive input from the client
    input_data = ssl_socket.recv(1024).decode('utf-8')
    
    print(f"Received input: {input_data}")
    
    # Close the SSL socket
    ssl_socket.close()

if __name__ == "__main__":
    main()
```
