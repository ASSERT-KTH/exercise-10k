## Exercise
Write a program that simulates a traffic light. The program should print 'Red', 'Yellow', or 'Green' based on the current state of the light, and allow the user to change the state.

## Reference Solution
```python
class TrafficLight:
    def __init__(self):
        self.states = ['Red', 'Yellow', 'Green']
        self.current_state = self.states[0]

    def change_state(self, state):
        if state in self.states:
            self.current_state = state
        else:
            print("Invalid state. Please enter 'Red', 'Yellow', or 'Green'.")

    def print_state(self):
        print(self.current_state)


def main():
    light = TrafficLight()
    while True:
        print("\nOptions:")
        print("1. Change state")
        print("2. Print current state")
        print("3. Exit")
        choice = input("Enter your choice: ")
        
        if choice == '1':
            print("\nAvailable states:")
            for i, state in enumerate(light.states):
                print(f"{i+1}. {state}")
            state_choice = input("Enter the number of your desired state: ")
            try:
                state_choice = int(state_choice)
                if 1 <= state_choice <= 3:
                    light.change_state(light.states[state_choice - 1])
                else:
                    print("Invalid choice. Please enter a number between 1 and 3.")
            except ValueError:
                print("Invalid choice. Please enter a number.")
        elif choice == '2':
            light.print_state()
        elif choice == '3':
            break
        else:
            print("Invalid choice. Please enter a number between 1 and 3.")


if __name__ == "__main__":
    main()
```
