## Exercise
Write a program that extracts a substring from a given string based on a start and end index.

## Reference Solution
```python
def extract_substring(input_string, start_index, end_index):
    """
    This function extracts a substring from a given string based on a start and end index.

    Args:
        input_string (str): The string from which the substring will be extracted.
        start_index (int): The starting index of the substring.
        end_index (int): The ending index of the substring.

    Returns:
        str: The extracted substring.

    Raises:
        ValueError: If the start or end index is out of range.
    """
    if start_index < 0 or end_index > len(input_string) or start_index > end_index:
        raise ValueError("Invalid start or end index")

    return input_string[start_index:end_index]

def main():
    input_string = "Hello, World!"
    start_index = 7
    end_index = 12

    try:
        substring = extract_substring(input_string, start_index, end_index)
        print(f"The substring from index {start_index} to {end_index} is: {substring}")
    except ValueError as e:
        print(e)

if __name__ == "__main__":
    main()
```
